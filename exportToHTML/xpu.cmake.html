1    # ---[ xpu
2    
3    # Poor man's include guard
4    if(TARGET torch::xpurt) 
5      return() 
6    endif() 
7    
8    set(XPU_HOST_CXX_FLAGS) 
9    
10   # Find SYCL library.
11   find_package(SYCLToolkit REQUIRED) 
12   if(NOT SYCL_FOUND) 
13     set(PYTORCH_FOUND_XPU FALSE) 
14     return() 
15   endif() 
16   set(PYTORCH_FOUND_XPU TRUE) 
17   
18   # SYCL library interface
19   add_library(torch::sycl INTERFACE IMPORTED) 
20   
21   set_property( 
22       TARGET torch::sycl PROPERTY INTERFACE_INCLUDE_DIRECTORIES
23       ${SYCL_INCLUDE_DIR}) 
24   set_property( 
25       TARGET torch::sycl PROPERTY INTERFACE_LINK_LIBRARIES 
26       ${SYCL_LIBRARY}) 
27   
28   # xpurt
29   add_library(torch::xpurt INTERFACE IMPORTED) 
30   set_property( 
31       TARGET torch::xpurt PROPERTY INTERFACE_LINK_LIBRARIES 
32       torch::sycl) 
33   
34   # setting xpu arch flags
35   torch_xpu_get_arch_list(XPU_ARCH_FLAGS) 
36   # propagate to torch-xpu-ops
37   set(TORCH_XPU_ARCH_LIST ${XPU_ARCH_FLAGS}) 
38   
39   string(APPEND XPU_HOST_CXX_FLAGS " -DSYCL_COMPILER_VERSION=${SYCL_COMPILER_VERSION}") 
40   
41   if(DEFINED ENV{XPU_ENABLE_KINETO}) 
42     set(XPU_ENABLE_KINETO TRUE) 
43   else() 
44     set(XPU_ENABLE_KINETO FALSE) 
45   endif() 
46   
47   if(WIN32) 
48     if(${SYCL_COMPILER_VERSION} GREATER_EQUAL 20250101) 
49       set(XPU_ENABLE_KINETO TRUE) 
50     endif() 
51   else() 
52     set(XPU_ENABLE_KINETO TRUE) 
53   endif()
