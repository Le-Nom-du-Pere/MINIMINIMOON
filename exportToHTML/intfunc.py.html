22   def num_digits(n, base=10):
23       """Return the number of digits needed to express n in give base. 
24    
25       Examples 
26       ======== 
27    
28       >>> from sympy.core.intfunc import num_digits 
29       >>> num_digits(10) 
30       2 
31       >>> num_digits(10, 2)  # 1010 -> 4 digits 
32       4 
33       >>> num_digits(-100, 16)  # -64 -> 2 digits 
34       2 
35    
36    
37       Parameters 
38       ========== 
39    
40       n: integer 
41           The number whose digits are counted. 
42    
43       b: integer 
44           The base in which digits are computed. 
45    
46       See Also 
47       ======== 
48       sympy.ntheory.digits.digits, sympy.ntheory.digits.count_digits 
49       """
50       if base < 0:
51           raise ValueError('base must be int greater than 1')
52       if not n:
53           return 1
54       e, t = integer_log(abs(n), base)
55       return 1 + e

58   def integer_log(n, b):
59       r""" 
60       Returns ``(e, bool)`` where e is the largest nonnegative integer 
61       such that :math:`|n| \geq |b^e|` and ``bool`` is True if $n = b^e$. 
62    
63       Examples 
64       ======== 
65    
66       >>> from sympy import integer_log 
67       >>> integer_log(125, 5) 
68       (3, True) 
69       >>> integer_log(17, 9) 
70       (1, False) 
71    
72       If the base is positive and the number negative the 
73       return value will always be the same except for 2: 
74    
75       >>> integer_log(-4, 2) 
76       (2, False) 
77       >>> integer_log(-16, 4) 
78       (0, False) 
79    
80       When the base is negative, the returned value 
81       will only be True if the parity of the exponent is 
82       correct for the sign of the base: 
83    
84       >>> integer_log(4, -2) 
85       (2, True) 
86       >>> integer_log(8, -2) 
87       (3, False) 
88       >>> integer_log(-8, -2) 
89       (3, True) 
90       >>> integer_log(-4, -2) 
91       (2, False) 
92    
93       See Also 
94       ======== 
95       integer_nthroot 
96       sympy.ntheory.primetest.is_square 
97       sympy.ntheory.factor_.multiplicity 
98       sympy.ntheory.factor_.perfect_power 
99       """
100      n = as_int(n)
101      b = as_int(b)
102  
103      if b < 0:
104          e, t = integer_log(abs(n), -b)
105          # (-2)**3 == -8
106          # (-2)**2 = 4
107          t = t and e % 2 == (n < 0)
108          return e, t
109      if b <= 1:
110          raise ValueError('base must be 2 or more')
111      if n < 0:
112          if b != 2:
113              return 0, False
114          e, t = integer_log(-n, b)
115          return e, False
116      if n == 0:
117          raise ValueError('n cannot be 0')
118  
119      if n < b:
120          return 0, n == 1
121      if b == 2:
122          e = n.bit_length() - 1
123          return e, trailing(n) == e
124      t = trailing(b)
125      if 2**t == b:
126          e = int(n.bit_length() - 1)//t
127          n_ = 1 << (t*e)
128          return e, n_ == n
129  
130      d = math.floor(math.log10(n) / math.log10(b))
131      n_ = b ** d
132      while n_ <= n:  # this will iterate 0, 1 or 2 times
133          d += 1
134          n_ *= b
135      return d - (n_ > n), (n_ == n or n_//b == n)

138  def trailing(n):
139      """Count the number of trailing zero digits in the binary 
140      representation of n, i.e. determine the largest power of 2 
141      that divides n. 
142   
143      Examples 
144      ======== 
145   
146      >>> from sympy import trailing 
147      >>> trailing(128) 
148      7 
149      >>> trailing(63) 
150      0 
151   
152      See Also 
153      ======== 
154      sympy.ntheory.factor_.multiplicity 
155   
156      """
157      if not n:
158          return 0
159      return bit_scan1(int(n))

163  def igcd(*args):
164      """Computes nonnegative integer greatest common divisor. 
165   
166      Explanation 
167      =========== 
168   
169      The algorithm is based on the well known Euclid's algorithm [1]_. To 
170      improve speed, ``igcd()`` has its own caching mechanism. 
171      If you do not need the cache mechanism, using ``sympy.external.gmpy.gcd``. 
172   
173      Examples 
174      ======== 
175   
176      >>> from sympy import igcd 
177      >>> igcd(2, 4) 
178      2 
179      >>> igcd(5, 10, 15) 
180      5 
181   
182      References 
183      ========== 
184   
185      .. [1] https://en.wikipedia.org/wiki/Euclidean_algorithm 
186   
187      """
188      if len(args) < 2:
189          raise TypeError("igcd() takes at least 2 arguments (%s given)" % len(args))
190      return int(number_gcd(*map(as_int, args)))

196  def igcd_lehmer(a, b):
197      r"""Computes greatest common divisor of two integers. 
198   
199      Explanation 
200      =========== 
201   
202      Euclid's algorithm for the computation of the greatest 
203      common divisor ``gcd(a, b)``  of two (positive) integers 
204      $a$ and $b$ is based on the division identity 
205      $$ a = q \times b + r$$, 
206      where the quotient  $q$  and the remainder  $r$  are integers 
207      and  $0 \le r < b$. Then each common divisor of  $a$  and  $b$ 
208      divides  $r$, and it follows that  ``gcd(a, b) == gcd(b, r)``. 
209      The algorithm works by constructing the sequence 
210      r0, r1, r2, ..., where  r0 = a, r1 = b,  and each  rn 
211      is the remainder from the division of the two preceding 
212      elements. 
213   
214      In Python, ``q = a // b``  and  ``r = a % b``  are obtained by the 
215      floor division and the remainder operations, respectively. 
216      These are the most expensive arithmetic operations, especially 
217      for large  a  and  b. 
218   
219      Lehmer's algorithm [1]_ is based on the observation that the quotients 
220      ``qn = r(n-1) // rn``  are in general small integers even 
221      when  a  and  b  are very large. Hence the quotients can be 
222      usually determined from a relatively small number of most 
223      significant bits. 
224   
225      The efficiency of the algorithm is further enhanced by not 
226      computing each long remainder in Euclid's sequence. The remainders 
227      are linear combinations of  a  and  b  with integer coefficients 
228      derived from the quotients. The coefficients can be computed 
229      as far as the quotients can be determined from the chosen 
230      most significant parts of  a  and  b. Only then a new pair of 
231      consecutive remainders is computed and the algorithm starts 
232      anew with this pair. 
233   
234      References 
235      ========== 
236   
237      .. [1] https://en.wikipedia.org/wiki/Lehmer%27s_GCD_algorithm 
238   
239      """
240      a, b = abs(as_int(a)), abs(as_int(b))
241      if a < b:
242          a, b = b, a
243  
244      # The algorithm works by using one or two digit division
245      # whenever possible. The outer loop will replace the
246      # pair (a, b) with a pair of shorter consecutive elements
247      # of the Euclidean gcd sequence until a and b
248      # fit into two Python (long) int digits.
249      nbits = 2 * sys.int_info.bits_per_digit
250  
251      while a.bit_length() > nbits and b != 0:
252          # Quotients are mostly small integers that can
253          # be determined from most significant bits.
254          n = a.bit_length() - nbits
255          x, y = int(a >> n), int(b >> n)  # most significant bits
256  
257          # Elements of the Euclidean gcd sequence are linear
258          # combinations of a and b with integer coefficients.
259          # Compute the coefficients of consecutive pairs
260          #     a' = A*a + B*b, b' = C*a + D*b
261          # using small integer arithmetic as far as possible.
262          A, B, C, D = 1, 0, 0, 1  # initial values
263  
264          while True:
265              # The coefficients alternate in sign while looping.
266              # The inner loop combines two steps to keep track
267              # of the signs.
268  
269              # At this point we have
270              #   A > 0, B <= 0, C <= 0, D > 0,
271              #   x' = x + B <= x < x" = x + A,
272              #   y' = y + C <= y < y" = y + D,
273              # and
274              #   x'*N <= a' < x"*N, y'*N <= b' < y"*N,
275              # where N = 2**n.
276  
277              # Now, if y' > 0, and x"//y' and x'//y" agree,
278              # then their common value is equal to  q = a'//b'.
279              # In addition,
280              #   x'%y" = x' - q*y" < x" - q*y' = x"%y',
281              # and
282              #   (x'%y")*N < a'%b' < (x"%y')*N.
283  
284              # On the other hand, we also have  x//y == q,
285              # and therefore
286              #   x'%y" = x + B - q*(y + D) = x%y + B',
287              #   x"%y' = x + A - q*(y + C) = x%y + A',
288              # where
289              #    B' = B - q*D < 0, A' = A - q*C > 0.
290  
291              if y + C <= 0:
292                  break
293              q = (x + A) // (y + C)
294  
295              # Now  x'//y" <= q, and equality holds if
296              #   x' - q*y" = (x - q*y) + (B - q*D) >= 0.
297              # This is a minor optimization to avoid division.
298              x_qy, B_qD = x - q * y, B - q * D
299              if x_qy + B_qD < 0:
300                  break
301  
302              # Next step in the Euclidean sequence.
303              x, y = y, x_qy
304              A, B, C, D = C, D, A - q * C, B_qD
305  
306              # At this point the signs of the coefficients
307              # change and their roles are interchanged.
308              #   A <= 0, B > 0, C > 0, D < 0,
309              #   x' = x + A <= x < x" = x + B,
310              #   y' = y + D < y < y" = y + C.
311  
312              if y + D <= 0:
313                  break
314              q = (x + B) // (y + D)
315              x_qy, A_qC = x - q * y, A - q * C
316              if x_qy + A_qC < 0:
317                  break
318  
319              x, y = y, x_qy
320              A, B, C, D = C, D, A_qC, B - q * D
321              # Now the conditions on top of the loop
322              # are again satisfied.
323              #   A > 0, B < 0, C < 0, D > 0.
324  
325          if B == 0:
326              # This can only happen when y == 0 in the beginning
327              # and the inner loop does nothing.
328              # Long division is forced.
329              a, b = b, a % b
330              continue
331  
332          # Compute new long arguments using the coefficients.
333          a, b = A * a + B * b, C * a + D * b
334  
335      # Small divisors. Finish with the standard algorithm.
336      while b:
337          a, b = b, a % b
338  
339      return a

362  def igcdex(a, b):
363      """Returns x, y, g such that g = x*a + y*b = gcd(a, b). 
364   
365      Examples 
366      ======== 
367   
368      >>> from sympy.core.intfunc import igcdex 
369      >>> igcdex(2, 3) 
370      (-1, 1, 1) 
371      >>> igcdex(10, 12) 
372      (-1, 1, 2) 
373   
374      >>> x, y, g = igcdex(100, 2004) 
375      >>> x, y, g 
376      (-20, 1, 4) 
377      >>> x*100 + y*2004 
378      4 
379   
380      """
381      g, x, y = gcdext(int(a), int(b))
382      return x, y, g

385  def mod_inverse(a, m):
386      r""" 
387      Return the number $c$ such that, $a \times c = 1 \pmod{m}$ 
388      where $c$ has the same sign as $m$. If no such value exists, 
389      a ValueError is raised. 
390   
391      Examples 
392      ======== 
393   
394      >>> from sympy import mod_inverse, S 
395   
396      Suppose we wish to find multiplicative inverse $x$ of 
397      3 modulo 11. This is the same as finding $x$ such 
398      that $3x = 1 \pmod{11}$. One value of x that satisfies 
399      this congruence is 4. Because $3 \times 4 = 12$ and $12 = 1 \pmod{11}$. 
400      This is the value returned by ``mod_inverse``: 
401   
402      >>> mod_inverse(3, 11) 
403      4 
404      >>> mod_inverse(-3, 11) 
405      7 
406   
407      When there is a common factor between the numerators of 
408      `a` and `m` the inverse does not exist: 
409   
410      >>> mod_inverse(2, 4) 
411      Traceback (most recent call last): 
412      ... 
413      ValueError: inverse of 2 mod 4 does not exist 
414   
415      >>> mod_inverse(S(2)/7, S(5)/2) 
416      7/2 
417   
418      References 
419      ========== 
420   
421      .. [1] https://en.wikipedia.org/wiki/Modular_multiplicative_inverse 
422      .. [2] https://en.wikipedia.org/wiki/Extended_Euclidean_algorithm 
423      """
424      c = None
425      try:
426          a, m = as_int(a), as_int(m)
427          if m != 1 and m != -1:
428              x, _, g = igcdex(a, m)
429              if g == 1:
430                  c = x % m
431      except ValueError:
432          a, m = sympify(a), sympify(m)
433          if not (a.is_number and m.is_number):
434              raise TypeError(
435                  filldedent(
436                      """ 
437                  Expected numbers for arguments; symbolic `mod_inverse` 
438                  is not implemented 
439                  but symbolic expressions can be handled with the 
440                  similar function, 
441                  sympy.polys.polytools.invert"""
442                  )
443              )
444          big = m > 1
445          if big not in (S.true, S.false):
446              raise ValueError("m > 1 did not evaluate; try to simplify %s" % m)
447          elif big:
448              c = 1 / a
449      if c is None:
450          raise ValueError("inverse of %s (mod %s) does not exist" % (a, m))
451      return c

454  def isqrt(n):
455      r""" Return the largest integer less than or equal to `\sqrt{n}`. 
456   
457      Parameters 
458      ========== 
459   
460      n : non-negative integer 
461   
462      Returns 
463      ======= 
464   
465      int : `\left\lfloor\sqrt{n}\right\rfloor` 
466   
467      Raises 
468      ====== 
469   
470      ValueError 
471          If n is negative. 
472      TypeError 
473          If n is of a type that cannot be compared to ``int``. 
474          Therefore, a TypeError is raised for ``str``, but not for ``float``. 
475   
476      Examples 
477      ======== 
478   
479      >>> from sympy.core.intfunc import isqrt 
480      >>> isqrt(0) 
481      0 
482      >>> isqrt(9) 
483      3 
484      >>> isqrt(10) 
485      3 
486      >>> isqrt("30") 
487      Traceback (most recent call last): 
488          ... 
489      TypeError: '<' not supported between instances of 'str' and 'int' 
490      >>> from sympy.core.numbers import Rational 
491      >>> isqrt(Rational(-1, 2)) 
492      Traceback (most recent call last): 
493          ... 
494      ValueError: n must be nonnegative 
495   
496      """
497      if n < 0:
498          raise ValueError("n must be nonnegative")
499      return int(sqrt(int(n)))

502  def integer_nthroot(y, n):
503      """ 
504      Return a tuple containing x = floor(y**(1/n)) 
505      and a boolean indicating whether the result is exact (that is, 
506      whether x**n == y). 
507   
508      Examples 
509      ======== 
510   
511      >>> from sympy import integer_nthroot 
512      >>> integer_nthroot(16, 2) 
513      (4, True) 
514      >>> integer_nthroot(26, 2) 
515      (5, False) 
516   
517      To simply determine if a number is a perfect square, the is_square 
518      function should be used: 
519   
520      >>> from sympy.ntheory.primetest import is_square 
521      >>> is_square(26) 
522      False 
523   
524      See Also 
525      ======== 
526      sympy.ntheory.primetest.is_square 
527      integer_log 
528      """
529      x, b = iroot(as_int(y), as_int(n))
530      return int(x), b
