6    def format_bytecode(table):
7        # given a nested tuple, convert it to nested list
8        def listify(content):
9            if not isinstance(content, tuple):
10               return content
11           return [listify(i) for i in content]
12   
13       formatted_table = {}
14       for entry in table:
15           identifier = entry[0]
16           content = entry[1]
17           content = listify(content)
18           formatted_table[identifier] = content
19       return formatted_table

22   def generate_upgraders_bytecode() -> list:
23       yaml_content = []
24       upgraders_graph_map = _generate_upgraders_graph()
25       for upgrader_name, upgrader_graph in upgraders_graph_map.items():
26           bytecode_table = _compile_graph_to_code_table(upgrader_name, upgrader_graph)
27           entry = {upgrader_name: format_bytecode(bytecode_table)}
28           yaml_content.append(entry)
