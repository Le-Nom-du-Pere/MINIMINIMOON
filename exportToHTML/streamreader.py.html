12   @functional_datapipe("read_from_stream")
13   class StreamReaderIterDataPipe(IterDataPipe[tuple[str, bytes]]):
14       r""" 
15       Given IO streams and their label names, yield bytes with label name as tuple. 
16    
17       (functional name: ``read_from_stream``). 
18    
19       Args: 
20           datapipe: Iterable DataPipe provides label/URL and byte stream 
21           chunk: Number of bytes to be read from stream per iteration. 
22               If ``None``, all bytes will be read until the EOF. 
23    
24       Example: 
25           >>> # xdoctest: +SKIP 
26           >>> from torchdata.datapipes.iter import IterableWrapper, StreamReader 
27           >>> from io import StringIO 
28           >>> dp = IterableWrapper([("alphabet", StringIO("abcde"))]) 
29           >>> list(StreamReader(dp, chunk=1)) 
30           [('alphabet', 'a'), ('alphabet', 'b'), ('alphabet', 'c'), ('alphabet', 'd'), ('alphabet', 'e')] 
31       """
32   
33       def __init__(
34           self, datapipe: IterDataPipe[tuple[str, IOBase]], chunk: Optional[int] = None
35       ):
36           self.datapipe = datapipe
37           self.chunk = chunk
38   
39       def __iter__(self) -> Iterator[tuple[str, bytes]]:
40           for furl, stream in self.datapipe:
41               while True:
42                   d = stream.read(self.chunk)
43                   if not d:
44                       stream.close()
45                       break
46                   yield (furl, d)
