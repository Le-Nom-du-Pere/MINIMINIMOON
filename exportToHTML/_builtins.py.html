128  # ops in torch.functional are bound to torch
129  # in these cases, we want to resolve the function to their python implementation
130  # instead looking up a builtin "aten::" schema
131

133  def _gen_torch_functional_registered_ops():
134      # eventually ops should encompass all of torch/functional.py, (torch.functional.__all__)
135      # but we are currently only able to compile some of the functions. additionally,
136      # some functions directly map to their aten:: implementations.
137      # TODO: add support for more ops
138      ops = [
139          "stft",
140          "istft",
141          "lu",
142          "cdist",
143          "norm",
144          "unique",
145          "unique_consecutive",
146          "tensordot",
147      ]
148      return {getattr(torch.functional, name) for name in ops}
149  
150  
151  _functional_registered_ops = _gen_torch_functional_registered_ops()
152

154  def _is_special_functional_bound_op(fn):
155      return fn in _functional_registered_ops

158  # lazily built to ensure the correct initialization order
159  def _get_builtin_table():
160      global _builtin_table
161      if _builtin_table is not None:
162          return _builtin_table
163      _builtin_table = {}
164  
165      def register_all(mod):
166          for name in dir(mod):
167              v = getattr(mod, name)
168              if (
169                  callable(v)
170                  and not _is_special_functional_bound_op(v)
171                  and v is not torch.no_grad
172                  and v is not torch.autocast
173              ):
174                  # Fixup inconsistency in segment_reduce
175                  if name == "_segment_reduce":
176                      name = name[1:]
177                  _builtin_ops.append((v, "aten::" + name))
178  
179      for mod in _modules_containing_builtins:
180          register_all(mod)
181  
182      _builtin_ops.append((math.gcd, "aten::gcd"))
183      _builtin_ops.append((math.isfinite, "aten::isfinite"))
184      _builtin_ops.append((math.remainder, "aten::mathremainder"))  # type: ignore[attr-defined]
185  
186      import torch.distributed.autograd as dist_autograd
187  
188      if dist_autograd.is_available():
189          _builtin_ops.append((dist_autograd.get_gradients, "aten::get_gradients"))
190          _builtin_ops.append((dist_autograd.backward, "aten::dist_backward"))
191  
192      # populate the _builtin_table from _builtin_ops
193      for builtin, aten_op in _builtin_ops:
194          _builtin_table[id(builtin)] = aten_op
195  
196      return _builtin_table
197  
198  
199  def _register_builtin(fn, op):
200      _get_builtin_table()[id(fn)] = op
201  
202  
203  def _find_builtin(fn):
204      return _get_builtin_table().get(id(fn))
