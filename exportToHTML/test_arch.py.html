1    from sympy.physics.continuum_mechanics.arch import Arch
2    from sympy import Symbol, simplify
3    
4    x = Symbol('x')
5    t = Symbol('t')
6    
7    def test_arch_init():
8        a = Arch((0,0),(10,0),crown_x=5,crown_y=5)
9        assert a.get_loads == {'distributed': {}, 'concentrated': {}}
10       assert a.reaction_force == {Symbol('R_A_x'):0, Symbol('R_A_y'):0, Symbol('R_B_x'):0, Symbol('R_B_y'):0}
11       assert a.supports == {'left':'hinge', 'right':'hinge'}
12       assert a.left_support == (0,0)
13       assert a.right_support == (10,0)
14       assert a.get_shape_eqn == 5 - ((x-5)**2)/5
15   
16       a = Arch((0,0),(10,1),crown_x=6)
17       a.change_support_type(left_support='roller')
18       a.add_member(0.5)
19       assert a.supports == {'left':'roller', 'right':'hinge'}
20       assert simplify(a.get_shape_eqn) == simplify(9/5 - (x - 6)**2/20)

22   def test_arch_support():
23       a = Arch((0,0),(40,0),crown_x=20,crown_y=12)
24       a.apply_load(-1,'C',8,150,angle=270)
25       a.apply_load(0,'D',start=20,end=40,mag=-4)
26       a.solve()
27       assert abs(a.reaction_force[Symbol("R_A_x")] - 83.33333333333333) < 10e-12
28       assert abs(a.reaction_force[Symbol("R_B_y")] - 90.00000000000000) < 10e-12
29       assert abs(a.reaction_force[Symbol("R_B_x")] + 83.33333333333333) < 10e-12
30       assert abs(a.reaction_force[Symbol("R_A_y")] - 140.00000000000000) < 10e-12
31   
32   def test_arch_member():
33       a = Arch((0,0),(40,0),crown_x=20,crown_y=15)
34       a.change_support_type(right_support='roller')
35       a.add_member(0)
36       a.apply_load(-1,'D',start=12,mag=3,angle=270)
37       a.apply_load(-1,'E',start=6,mag=4,angle=270)
38       a.apply_load(-1,'C',start=30,mag=5,angle=270)
39       a.solve()
40       assert a.reaction_force[Symbol("R_A_x")] == 0
41       assert abs(a.reaction_force[Symbol("R_A_y")] - 6.750000000000000) < 10e-12
42       assert a.reaction_force[Symbol("R_B_x")] == 0
43       assert abs(a.reaction_force[Symbol("R_B_y")] - 5.250000000000000) < 10e-12

45   def test_symbol_magnitude():
46       a = Arch((0,0),(16,0),crown_x=8,crown_y=5)
47       a.apply_load(0,'C',start=3,end=5,mag=t)
48       a.solve()
49       assert a.reaction_force[Symbol("R_A_x")] == -(4*t)/5
50       assert a.reaction_force[Symbol("R_A_y")] == -(3*t)/2
51       assert a.reaction_force[Symbol("R_B_x")] == (4*t)/5
52       assert a.reaction_force[Symbol("R_B_y")] == -t/2
53       assert a.bending_moment_at(4) == -5*t/2
