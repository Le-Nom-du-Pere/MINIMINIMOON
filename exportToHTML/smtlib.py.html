98       def _is_legal_name(self, s: str):
99           if not s: return False
100          if s[0].isnumeric(): return False
101          return all(_.isalnum() or _ == '_' for _ in s)

103      def _s_expr(self, op: str, args: typing.Union[list, tuple]) -> str:
104          args_str = ' '.join(
105              a if isinstance(a, str)
106              else self._print(a)
107              for a in args
108          )
109          return f'({op} {args_str})'

111      def _print_Function(self, e):
112          if e in self._known_functions:
113              op = self._known_functions[e]
114          elif type(e) in self._known_functions:
115              op = self._known_functions[type(e)]
116          elif type(type(e)) == UndefinedFunction:
117              op = e.name
118          elif isinstance(e, AppliedBinaryRelation) and e.function in self._known_functions:
119              op = self._known_functions[e.function]
120              return self._s_expr(op, e.arguments)
121          else:
122              op = self._known_functions[e]  # throw KeyError
123  
124          return self._s_expr(op, e.args)

135      def _print_Unequality(self, e: Unequality):
136          if type(e) in self._known_functions:
137              return self._print_Relational(e)  # default
138          else:
139              eq_op = self._known_functions[Equality]
140              not_op = self._known_functions[Not]
141              return self._s_expr(not_op, [self._s_expr(eq_op, e.args)])

143      def _print_Piecewise(self, e: Piecewise):
144          def _print_Piecewise_recursive(args: typing.Union[list, tuple]):
145              e, c = args[0]
146              if len(args) == 1:
147                  assert (c is True) or isinstance(c, BooleanTrue)
148                  return self._print(e)
149              else:
150                  ite = self._known_functions[ITE]
151                  return self._s_expr(ite, [
152                      c, e, _print_Piecewise_recursive(args[1:])
153                  ])
154  
155          return _print_Piecewise_recursive(e.args)

157      def _print_Interval(self, e: Interval):
158          if e.start.is_infinite and e.end.is_infinite:
159              return ''
160          elif e.start.is_infinite != e.end.is_infinite:
161              raise ValueError(f'One-sided intervals (`{e}`) are not supported in SMT.')
162          else:
163              return f'[{e.start}, {e.end}]'

165      def _print_AppliedPredicate(self, e: AppliedPredicate):
166          if e.function == Q.positive:
167              rel = Q.gt(e.arguments[0],0)
168          elif e.function == Q.negative:
169              rel = Q.lt(e.arguments[0], 0)
170          elif e.function == Q.zero:
171              rel = Q.eq(e.arguments[0], 0)
172          elif e.function == Q.nonpositive:
173              rel = Q.le(e.arguments[0], 0)
174          elif e.function == Q.nonnegative:
175              rel = Q.ge(e.arguments[0], 0)
176          elif e.function == Q.nonzero:
177              rel = Q.ne(e.arguments[0], 0)
178          else:
179              raise ValueError(f"Predicate (`{e}`) is not handled.")
180  
181          return self._print_AppliedBinaryRelation(rel)
182  
183      def _print_AppliedBinaryRelation(self, e: AppliedPredicate):
184          if e.function == Q.ne:
185              return self._print_Unequality(Unequality(*e.arguments))
186          else:
187              return self._print_Function(e)

208      def _print_Float(self, x: Float):
209          dps = prec_to_dps(x._prec)
210          str_real = mlib_to_str(x._mpf_, dps, strip_zeros=True, min_fixed=None, max_fixed=None)
211  
212          if 'e' in str_real:
213              (mant, exp) = str_real.split('e')
214  
215              if exp[0] == '+':
216                  exp = exp[1:]
217  
218              mul = self._known_functions[Mul]
219              pow = self._known_functions[Pow]
220  
221              return r"(%s %s (%s 10 %s))" % (mul, mant, pow, exp)
222          elif str_real in ["+inf", "-inf"]:
223              raise ValueError("Infinite values are not supported in SMT.")
224          else:
225              return str_real

267  def smtlib_code(
268      expr,
269      auto_assert=True, auto_declare=True,
270      precision=None,
271      symbol_table=None,
272      known_types=None, known_constants=None, known_functions=None,
273      prefix_expressions=None, suffix_expressions=None,
274      log_warn=None
275  ):
276      r"""Converts ``expr`` to a string of smtlib code. 
277   
278      Parameters 
279      ========== 
280   
281      expr : Expr | List[Expr] 
282          A SymPy expression or system to be converted. 
283      auto_assert : bool, optional 
284          If false, do not modify expr and produce only the S-Expression equivalent of expr. 
285          If true, assume expr is a system and assert each boolean element. 
286      auto_declare : bool, optional 
287          If false, do not produce declarations for the symbols used in expr. 
288          If true, prepend all necessary declarations for variables used in expr based on symbol_table. 
289      precision : integer, optional 
290          The ``evalf(..)`` precision for numbers such as pi. 
291      symbol_table : dict, optional 
292          A dictionary where keys are ``Symbol`` or ``Function`` instances and values are their Python type i.e. ``bool``, ``int``, ``float``, or ``Callable[...]``. 
293          If incomplete, an attempt will be made to infer types from ``expr``. 
294      known_types: dict, optional 
295          A dictionary where keys are ``bool``, ``int``, ``float`` etc. and values are their corresponding SMT type names. 
296          If not given, a partial listing compatible with several solvers will be used. 
297      known_functions : dict, optional 
298          A dictionary where keys are ``Function``, ``Relational``, ``BooleanFunction``, or ``Expr`` instances and values are their SMT string representations. 
299          If not given, a partial listing optimized for dReal solver (but compatible with others) will be used. 
300      known_constants: dict, optional 
301          A dictionary where keys are ``NumberSymbol`` instances and values are their SMT variable names. 
302          When using this feature, extra caution must be taken to avoid naming collisions between user symbols and listed constants. 
303          If not given, constants will be expanded inline i.e. ``3.14159`` instead of ``MY_SMT_VARIABLE_FOR_PI``. 
304      prefix_expressions: list, optional 
305          A list of lists of ``str`` and/or expressions to convert into SMTLib and prefix to the output. 
306      suffix_expressions: list, optional 
307          A list of lists of ``str`` and/or expressions to convert into SMTLib and postfix to the output. 
308      log_warn: lambda function, optional 
309          A function to record all warnings during potentially risky operations. 
310          Soundness is a core value in SMT solving, so it is good to log all assumptions made. 
311   
312      Examples 
313      ======== 
314      >>> from sympy import smtlib_code, symbols, sin, Eq 
315      >>> x = symbols('x') 
316      >>> smtlib_code(sin(x).series(x).removeO(), log_warn=print) 
317      Could not infer type of `x`. Defaulting to float. 
318      Non-Boolean expression `x**5/120 - x**3/6 + x` will not be asserted. Converting to SMTLib verbatim. 
319      '(declare-const x Real)\n(+ x (* (/ -1 6) (pow x 3)) (* (/ 1 120) (pow x 5)))' 
320   
321      >>> from sympy import Rational 
322      >>> x, y, tau = symbols("x, y, tau") 
323      >>> smtlib_code((2*tau)**Rational(7, 2), log_warn=print) 
324      Could not infer type of `tau`. Defaulting to float. 
325      Non-Boolean expression `8*sqrt(2)*tau**(7/2)` will not be asserted. Converting to SMTLib verbatim. 
326      '(declare-const tau Real)\n(* 8 (pow 2 (/ 1 2)) (pow tau (/ 7 2)))' 
327   
328      ``Piecewise`` expressions are implemented with ``ite`` expressions by default. 
329      Note that if the ``Piecewise`` lacks a default term, represented by 
330      ``(expr, True)`` then an error will be thrown.  This is to prevent 
331      generating an expression that may not evaluate to anything. 
332   
333      >>> from sympy import Piecewise 
334      >>> pw = Piecewise((x + 1, x > 0), (x, True)) 
335      >>> smtlib_code(Eq(pw, 3), symbol_table={x: float}, log_warn=print) 
336      '(declare-const x Real)\n(assert (= (ite (> x 0) (+ 1 x) x) 3))' 
337   
338      Custom printing can be defined for certain types by passing a dictionary of 
339      PythonType : "SMT Name" to the ``known_types``, ``known_constants``, and ``known_functions`` kwargs. 
340   
341      >>> from typing import Callable 
342      >>> from sympy import Function, Add 
343      >>> f = Function('f') 
344      >>> g = Function('g') 
345      >>> smt_builtin_funcs = {  # functions our SMT solver will understand 
346      ...   f: "existing_smtlib_fcn", 
347      ...   Add: "sum", 
348      ... } 
349      >>> user_def_funcs = {  # functions defined by the user must have their types specified explicitly 
350      ...   g: Callable[[int], float], 
351      ... } 
352      >>> smtlib_code(f(x) + g(x), symbol_table=user_def_funcs, known_functions=smt_builtin_funcs, log_warn=print) 
353      Non-Boolean expression `f(x) + g(x)` will not be asserted. Converting to SMTLib verbatim. 
354      '(declare-const x Int)\n(declare-fun g (Int) Real)\n(sum (existing_smtlib_fcn x) (g x))' 
355      """
356      log_warn = log_warn or (lambda _: None)
357  
358      if not isinstance(expr, list): expr = [expr]
359      expr = [
360          sympy.sympify(_, strict=True, evaluate=False, convert_xor=False)
361          for _ in expr
362      ]
363  
364      if not symbol_table: symbol_table = {}
365      symbol_table = _auto_infer_smtlib_types(
366          *expr, symbol_table=symbol_table
367      )
368      # See [FALLBACK RULES]
369      # Need SMTLibPrinter to populate known_functions and known_constants first.
370  
371      settings = {}
372      if precision: settings['precision'] = precision
373      del precision
374  
375      if known_types: settings['known_types'] = known_types
376      del known_types
377  
378      if known_functions: settings['known_functions'] = known_functions
379      del known_functions
380  
381      if known_constants: settings['known_constants'] = known_constants
382      del known_constants
383  
384      if not prefix_expressions: prefix_expressions = []
385      if not suffix_expressions: suffix_expressions = []
386  
387      p = SMTLibPrinter(settings, symbol_table)
388      del symbol_table
389  
390      # [FALLBACK RULES]
391      for e in expr:
392          for sym in e.atoms(Symbol, Function):
393              if (
394                  sym.is_Symbol and
395                  sym not in p._known_constants and
396                  sym not in p.symbol_table
397              ):
398                  log_warn(f"Could not infer type of `{sym}`. Defaulting to float.")
399                  p.symbol_table[sym] = float
400              if (
401                  sym.is_Function and
402                  type(sym) not in p._known_functions and
403                  type(sym) not in p.symbol_table and
404                  not sym.is_Piecewise
405              ): raise TypeError(
406                  f"Unknown type of undefined function `{sym}`. "
407                  f"Must be mapped to ``str`` in known_functions or mapped to ``Callable[..]`` in symbol_table."
408              )
409  
410      declarations = []
411      if auto_declare:
412          constants = {sym.name: sym for e in expr for sym in e.free_symbols
413                       if sym not in p._known_constants}
414          functions = {fnc.name: fnc for e in expr for fnc in e.atoms(Function)
415                       if type(fnc) not in p._known_functions and not fnc.is_Piecewise}
416          declarations = \
417              [
418                  _auto_declare_smtlib(sym, p, log_warn)
419                  for sym in constants.values()
420              ] + [
421                  _auto_declare_smtlib(fnc, p, log_warn)
422                  for fnc in functions.values()
423              ]
424          declarations = [decl for decl in declarations if decl]
425  
426      if auto_assert:
427          expr = [_auto_assert_smtlib(e, p, log_warn) for e in expr]
428  
429      # return SMTLibPrinter().doprint(expr)
430      return '\n'.join([
431          # ';; PREFIX EXPRESSIONS',
432          *[
433              e if isinstance(e, str) else p.doprint(e)
434              for e in prefix_expressions
435          ],
436  
437          # ';; DECLARATIONS',
438          *sorted(e for e in declarations),
439  
440          # ';; EXPRESSIONS',
441          *[
442              e if isinstance(e, str) else p.doprint(e)
443              for e in expr
444          ],
445  
446          # ';; SUFFIX EXPRESSIONS',
447          *[
448              e if isinstance(e, str) else p.doprint(e)
449              for e in suffix_expressions
450          ],
451      ])
452

454  def _auto_declare_smtlib(sym: typing.Union[Symbol, Function], p: SMTLibPrinter, log_warn: typing.Callable[[str], None]):
455      if sym.is_Symbol:
456          type_signature = p.symbol_table[sym]
457          assert isinstance(type_signature, type)
458          type_signature = p._known_types[type_signature]
459          return p._s_expr('declare-const', [sym, type_signature])
460  
461      elif sym.is_Function:
462          type_signature = p.symbol_table[type(sym)]
463          assert callable(type_signature)
464          type_signature = [p._known_types[_] for _ in type_signature.__args__]
465          assert len(type_signature) > 0
466          params_signature = f"({' '.join(type_signature[:-1])})"
467          return_signature = type_signature[-1]
468          return p._s_expr('declare-fun', [type(sym), params_signature, return_signature])
469  
470      else:
471          log_warn(f"Non-Symbol/Function `{sym}` will not be declared.")
472          return None

475  def _auto_assert_smtlib(e: Expr, p: SMTLibPrinter, log_warn: typing.Callable[[str], None]):
476      if isinstance(e, Boolean) or (
477          e in p.symbol_table and p.symbol_table[e] == bool
478      ) or (
479          e.is_Function and
480          type(e) in p.symbol_table and
481          p.symbol_table[type(e)].__args__[-1] == bool
482      ):
483          return p._s_expr('assert', [e])
484      else:
485          log_warn(f"Non-Boolean expression `{e}` will not be asserted. Converting to SMTLib verbatim.")
486          return e

489  def _auto_infer_smtlib_types(
490      *exprs: Basic,
491      symbol_table: typing.Optional[dict] = None
492  ) -> dict:
493      # [TYPE INFERENCE RULES]
494      # X is alone in an expr => X is bool
495      # X in BooleanFunction.args => X is bool
496      # X matches to a bool param of a symbol_table function => X is bool
497      # X matches to an int param of a symbol_table function => X is int
498      # X.is_integer => X is int
499      # X == Y, where X is T => Y is T
500  
501      # [FALLBACK RULES]
502      # see _auto_declare_smtlib(..)
503      # X is not bool and X is not int and X is Symbol => X is float
504      # else (e.g. X is Function) => error. must be specified explicitly.
505  
506      _symbols = dict(symbol_table) if symbol_table else {}
507  
508      def safe_update(syms: set, inf):
509          for s in syms:
510              assert s.is_Symbol
511              if (old_type := _symbols.setdefault(s, inf)) != inf:
512                  raise TypeError(f"Could not infer type of `{s}`. Apparently both `{old_type}` and `{inf}`?")
513  
514      # EXPLICIT TYPES
515      safe_update({
516          e
517          for e in exprs
518          if e.is_Symbol
519      }, bool)
520  
521      safe_update({
522          symbol
523          for e in exprs
524          for boolfunc in e.atoms(BooleanFunction)
525          for symbol in boolfunc.args
526          if symbol.is_Symbol
527      }, bool)
528  
529      safe_update({
530          symbol
531          for e in exprs
532          for boolfunc in e.atoms(Function)
533          if type(boolfunc) in _symbols
534          for symbol, param in zip(boolfunc.args, _symbols[type(boolfunc)].__args__)
535          if symbol.is_Symbol and param == bool
536      }, bool)
537  
538      safe_update({
539          symbol
540          for e in exprs
541          for intfunc in e.atoms(Function)
542          if type(intfunc) in _symbols
543          for symbol, param in zip(intfunc.args, _symbols[type(intfunc)].__args__)
544          if symbol.is_Symbol and param == int
545      }, int)
546  
547      safe_update({
548          symbol
549          for e in exprs
550          for symbol in e.atoms(Symbol)
551          if symbol.is_integer
552      }, int)
553  
554      safe_update({
555          symbol
556          for e in exprs
557          for symbol in e.atoms(Symbol)
558          if symbol.is_real and not symbol.is_integer
559      }, float)
560  
561      # EQUALITY RELATION RULE
562      rels_eq = [rel for expr in exprs for rel in expr.atoms(Equality)]
563      rels = [
564                 (rel.lhs, rel.rhs) for rel in rels_eq if rel.lhs.is_Symbol
565             ] + [
566                 (rel.rhs, rel.lhs) for rel in rels_eq if rel.rhs.is_Symbol
567             ]
568      for infer, reltd in rels:
569          inference = (
570              _symbols[infer] if infer in _symbols else
571              _symbols[reltd] if reltd in _symbols else
572  
573              _symbols[type(reltd)].__args__[-1]
574              if reltd.is_Function and type(reltd) in _symbols else
575  
576              bool if reltd.is_Boolean else
577              int if reltd.is_integer or reltd.is_Integer else
578              float if reltd.is_real else
579              None
580          )
581          if inference: safe_update({infer}, inference)
582  
583      return _symbols
