43   def _to_tuple(a, b):
44       return tuple(a), tuple(b)

46   def _trim_tuple(a, b):
47       a, b = _to_tuple(a, b)
48       return tuple(a[0: 2] + a[len(a)//2 : len(a)//2 + 1] + a[-2:]), \
49           tuple(b[0: 2] + b[len(b)//2 : len(b)//2 + 1] + b[-2:])

51   def y_coordinate_equality(plot_data_func, evalf_func, system):
52       """Checks whether the y-coordinate value of the plotted 
53       data point is equal to the value of the function at a 
54       particular x."""
55       x, y = plot_data_func(system)
56       x, y = _trim_tuple(x, y)
57       y_exp = tuple(evalf_func(system, x_i) for x_i in x)
58       return all(Abs(y_exp_i - y_i) < 1e-8 for y_exp_i, y_i in zip(y_exp, y))

136      def bode_phase_evalf(system, point):
137          expr = system.to_expr()
138          _w = Dummy("w", real=True)
139          w_expr = expr.subs({system.var: I*_w})
140          return arg(w_expr).subs({_w: point}).evalf()
141  
142      def bode_mag_evalf(system, point):
143          expr = system.to_expr()
144          _w = Dummy("w", real=True)
145          w_expr = expr.subs({system.var: I*_w})
146          return 20*log(Abs(w_expr), 10).subs({_w: point}).evalf()
147  
148      def test_bode_data(sys):
149          return y_coordinate_equality(bode_magnitude_numerical_data, bode_mag_evalf, sys) \
150              and y_coordinate_equality(bode_phase_numerical_data, bode_phase_evalf, sys)

160      return all(isclose(*_, rel_tol=1e-1, abs_tol=1e-6
161          ) for _ in zip(a, b))

320  def test_nichols_expr():
321      m1, p1, w1 = nichols_plot_expr(tf1)
322      m2, p2, w2 = nichols_plot_expr(tf2)
323      m3, p3, w3 = nichols_plot_expr(tf3)
324      m4, p4, w4 = nichols_plot_expr(tf4)
325      assert m1 == 20*log(1/sqrt(w1**4 - 3.75*w1**2 + 4))/log(10)
326      assert p1 == 180*arg(1/(-w1**2 + 0.5*w1*I + 2))/pi
327      assert m2 == 20*log(Abs(w2)/sqrt(36*w2**4 - 3*w2**2 + 1))/log(10)
328      assert p2 == 180*arg(w2*I/(-6*w2**2 + 3*w2*I + 1))/pi
329      assert m3 == 20*log(Abs(w3)/sqrt(w3**6 + 1))/log(10)
330      assert p3 == 180*arg(-w3*I/(w3**3*I + 1))/pi
331      assert m4 == 20*log(10/(w4**2*Abs(w4)))/log(10)
332      assert p4 == 180*arg(I/w4**3)/pi
333
