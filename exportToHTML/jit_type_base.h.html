170    template <typename T>
171    class SingletonOrSharedTypePtr {
172     public:
173      using element_type = typename std::shared_ptr<T>::element_type;
174  
175      SingletonOrSharedTypePtr() = default;
176  
177      /* implicit */ SingletonOrSharedTypePtr(std::shared_ptr<T> x)
178          : repr_(std::move(x)) {}
179  
180      template <typename U, std::enable_if_t<std::is_convertible_v<U*, T*>, bool> = true>
181      /* implicit */ SingletonOrSharedTypePtr(std::shared_ptr<U> x)
182          : repr_(std::move(x)) {}
183  
184      /* implicit */ SingletonOrSharedTypePtr(std::nullptr_t)
185          : repr_(nullptr) {}
186  
187      /* implicit */ SingletonOrSharedTypePtr(SingletonTypePtr<T> p)
188          : repr_(p) {}
189  
190      template <typename U, std::enable_if_t<std::is_convertible_v<U*, T*>, bool> = true>
191      /* implicit */ SingletonOrSharedTypePtr(SingletonTypePtr<U> p)
192          : repr_(SingletonTypePtr<T>(p.get())) {}
193  
194  
195      // We need to support construction from T* for pybind. The problem
196      // is that it's not clear if we are supposed to be taking shared
197      // ownership or not.
198      //
199      // Case 1: if T is known statically to derive from SharedType, we should use
200      // shared_from_this() and take shared_ownership.
201      //
202      // Case 2: if T is exactly Type, we need to do a dynamic_cast to
203      // check if it's a SharedType and do the right thing.
204      //
205      // Case 3: Otherwise, T is not a SharedType. (debug-check this
206      // assumption!) Use a singleton pointer.
207  
208      template <typename U = T, std::enable_if_t<std::is_base_of_v<SharedType, U>, bool> = true>
209      /* implicit */ SingletonOrSharedTypePtr(T* p) : SingletonOrSharedTypePtr(static_cast<typename detail::as_shared_type<U>::type>(p)->shared_from_this()) {}
210  
211      template <typename U = T, std::enable_if_t<std::is_same_v<Type, U>, bool> = true>
212      /* implicit */ SingletonOrSharedTypePtr(T* p) {
213        if (auto* shared_p = dynamic_cast<typename detail::as_shared_type<U>::type>(p)) {
214          repr_ = Repr(shared_p->shared_from_this());
215        } else {
216          repr_ = Repr(p);
217        }
218      }
219  
220      template <typename U = T, std::enable_if_t<!std::is_same_v<Type, U> && !std::is_base_of_v<SharedType, U>, bool> = true>
221      /* implicit */ SingletonOrSharedTypePtr(T* p)
222          : repr_(p) {
223        TORCH_INTERNAL_ASSERT_DEBUG_ONLY(dynamic_cast<typename detail::as_shared_type<U>::type>(p) == nullptr);
224      }
225  
226      SingletonOrSharedTypePtr(const SingletonOrSharedTypePtr&) = default;
227      SingletonOrSharedTypePtr(SingletonOrSharedTypePtr&&) noexcept = default;
228      SingletonOrSharedTypePtr& operator=(const SingletonOrSharedTypePtr&) = default;
229      SingletonOrSharedTypePtr& operator=(SingletonOrSharedTypePtr&&) noexcept = default;
230      ~SingletonOrSharedTypePtr() = default;
231  
232      T* get() const {
233        return repr_.isSharedAndNonNull() ? repr_.shared_.repr_.get() : static_cast<T*>(repr_.rawRepr().first);
234      }
235  
236      operator bool() const {
237        return repr_.isNonNull();
238      }
239  
240      bool operator==(std::nullptr_t) const {
241        return !repr_.isNonNull();
242      }
243  
244      bool operator!=(std::nullptr_t) const {
245        return repr_.isNonNull();
246      }
247  
248      template <typename U = T, std::enable_if_t<!std::is_same_v<std::remove_const_t<U>, void>, bool> = true>
249      U& operator*() const {
250        return *get();
251      }
252  
253      T* operator->() const {
254        return get();
255      }
256  
257    private:
258      // NOTE: SharedPtrWrapper exists to work around a baffling bug in
259      // nvcc; see comment in destroy() below.
260      struct SharedPtrWrapper {
261        SharedPtrWrapper(std::shared_ptr<T> &&x)
262            : repr_(std::move(x)) {}
263        std::shared_ptr<T> repr_;
264      };
265      union Repr {
266        Repr() : Repr(nullptr) {}
267  
268        explicit Repr(std::shared_ptr<T> x)
269            : shared_(std::move(x)) {}
270  
271        explicit Repr(std::nullptr_t)
272            : singletonRepr_(nullptr) {}
273  
274        explicit Repr(SingletonTypePtr<T> p)
275            : singletonRepr_(p.get()) {}
276  
277        ~Repr() {
278          destroy();
279        }
280  
281        // NOTE: the only non-UB way to access our null state is through
282        // rawRepr(), because our copy operation doesn't preserve which
283        // union member is active for null pointers.
284        Repr(const Repr& rhs) {
285          if (rhs.isSharedAndNonNull()) {
286            new (&shared_) SharedPtrWrapper(rhs.shared_);
287          } else {
288            singletonRepr_.singleton_ = static_cast<T*>(rhs.rawRepr().first);
289            TORCH_INTERNAL_ASSERT_DEBUG_ONLY(rhs.singletonRepr_.unused_ == nullptr);
290            singletonRepr_.unused_ = nullptr;
291          }
292        }
293  
294        Repr(Repr&& rhs) noexcept {
295          if (rhs.isSharedAndNonNull()) {
296            new (&shared_) SharedPtrWrapper(std::move(rhs.shared_));
297          } else {
298            singletonRepr_.singleton_ = static_cast<T*>(rhs.rawRepr().first);
299            TORCH_INTERNAL_ASSERT_DEBUG_ONLY(rhs.singletonRepr_.unused_ == nullptr);
300            singletonRepr_.unused_ = nullptr;
301          }
302        }
303  
304        Repr& operator=(const Repr& rhs) {
305          if (&rhs == this) {
306            return *this;
307          }
308          if (rhs.isSharedAndNonNull()) {
309            if (isSharedAndNonNull()) {
310              shared_ = rhs.shared_;
311            } else {
312              new (&shared_) SharedPtrWrapper(rhs.shared_);
313            }
314          } else {
315            if (isSharedAndNonNull()) {
316              destroy();
317            }
318            singletonRepr_.singleton_ = static_cast<T*>(rhs.rawRepr().first);
319            TORCH_INTERNAL_ASSERT_DEBUG_ONLY(rhs.rawRepr().nullIfSingleton_ == nullptr);
320            singletonRepr_.unused_ = nullptr;
321          }
322          return *this;
323        }
324  
325        Repr& operator=(Repr&& rhs) noexcept {
326          if (&rhs == this) {
327            return *this;
328          }
329          if (rhs.isSharedAndNonNull()) {
330            if (isSharedAndNonNull()) {
331              shared_ = std::move(rhs.shared_);
332            } else {
333              new (&shared_) SharedPtrWrapper(std::move(rhs.shared_));
334            }
335          } else {
336            if (isSharedAndNonNull()) {
337              destroy();
338            }
339            singletonRepr_.singleton_ = static_cast<T*>(rhs.rawRepr().first);
340            TORCH_INTERNAL_ASSERT_DEBUG_ONLY(rhs.rawRepr().nullIfSingleton_ == nullptr);
341            singletonRepr_.unused_ = nullptr;
342          }
343          return *this;
344        }
345  
346        SharedPtrWrapper shared_;
347  
348        struct SingletonRepr {
349          explicit SingletonRepr(T* s) : singleton_(s) {}
350          T* singleton_;
351          void* unused_ = nullptr;
352        } singletonRepr_;
353        struct RawRepr {
354          void* first;
355          void* nullIfSingleton_;
356        };
357  
358        // It is UB to read the singleton part of Repr if it was
359        // constructed as a shared_ptr and vice versa, but memcpying out
360        // the representation is always OK, so here's an accessor to obey
361        // the letter of the law.
362        RawRepr rawRepr() const {
363          RawRepr repr{};
364          memcpy(&repr, reinterpret_cast<const char *>(this), sizeof(RawRepr));
365          return repr;
366        }
367  
368        bool isNonNull() const {
369          auto repr = rawRepr();
370          TORCH_INTERNAL_ASSERT_DEBUG_ONLY(repr.nullIfSingleton_ == nullptr || repr.first != nullptr);
371          return repr.first != nullptr;
372        }
373  
374        bool isSharedAndNonNull() const {
375          return rawRepr().nullIfSingleton_ != nullptr;
376        }
377  
378       private:
379        void destroy() {
380          if (isSharedAndNonNull()) {
381            // Without SharedPtrWrapper, this line would read
382            // `shared_.~shared_ptr()` and nvcc would complain with
383            // "error: expected primary-expression before '>' token"
384            // referring to the "t" in "shared_ptr". SharedPtrWrapper
385            // exists to work around this compiler bug.
386            shared_.~SharedPtrWrapper();
387          }
388        }
389      } repr_;
390    };

396    // subtyping relation. By default, we return true for the case
397    // when the type is exactly equal or if this <: T where rhs = Optional[T]
398  
399    // if this returns false and the why_not stream is non-null, it contains
400    // additional details that describe why this is not a subtype of 'rhs'.
401    // This additional information should only contain details that are not
402    // obvious from the annotation_str() that describes the type. For instance it
403    // is clear that `int <: str` is false but not clear why `Foo <: InterfaceBar`
404    // might be false.
405    virtual bool isSubtypeOfExt(const Type& rhs, std::ostream* why_not) const;
406    virtual bool is_module() const;
407    bool isSubtypeOf(const Type& rhs) const {
408      return isSubtypeOfExt(rhs, nullptr);

451    // How this type will appear as if it were a type annotation in Python
452    // which is sometimes different than how it appears in declarations (e.g.
453    // int[] vs List[int])
454    //
455    // Takes a custom printer that users can pass in to customize the output of
456    // this method.
457    std::string annotation_str(const TypePrinter& printer) const {
458      if (printer) {
459        // the printer can return std::nullopt to fall through to the default impl
460        if (auto renamed = printer(*this)) {
461          return *renamed;
462        }
463      }
464      return annotation_str_impl(printer);
465    }

467      // Overload instead of define a default value for `printer` to help
468      // debuggers out.
469      return annotation_str(nullptr);

502      }

508        TORCH_INTERNAL_ASSERT_DEBUG_ONLY(this == T::get().get());
509        return typename detail::CastReturnType<T>::type(static_cast<T*>(this));
510      }
511      return nullptr;

659  // Base class for Types that are guaranteed to be owned by std::shared_ptr.
660  struct TORCH_API SharedType : public Type, public std::enable_shared_from_this<SharedType> {
661    using Type::Type;
662  };
663  
664  inline TypePtr Type::withContained(std::vector<TypePtr> contained_types) {
665    auto current_contained = containedTypes();
666    // Types with no contained_types don't need this call. Check before calling!
667    //
668    // (We can't support this efficiently because types without
669    // contained types may be singletons, in which case
670    // shared_from_this will crash; we would have to provide a virtual
671    // typeptr_from_this or isSingleton.)
672    TORCH_INTERNAL_ASSERT(!current_contained.empty() && current_contained.size() == contained_types.size());
673    if (current_contained.equals(contained_types)) {
674      return std::static_pointer_cast<Type>(static_cast<SharedType *>(this)->shared_from_this());
675    }
676    return createWithContained(std::move(contained_types));
677  }

680  TORCH_API inline bool operator==(const Type& lhs, const Type& rhs) {
681    if (C10_UNLIKELY(!rhs.symmetric())) {
682      return rhs.equals(lhs);
683    }
684    return lhs.equals(rhs);
685  }

698          "If you add a new kind of NamedType, ",
