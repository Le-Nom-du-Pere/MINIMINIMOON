18   struct type_index final : IdWrapper<type_index, uint64_t> {
19     constexpr explicit type_index(uint64_t checksum) : IdWrapper(checksum) {}
20   
21     // Allow usage in std::map / std::set
22     // TODO Disallow this and rather use std::unordered_map/set everywhere
23     friend constexpr bool operator<(type_index lhs, type_index rhs) noexcept {
24       return lhs.underlyingId() < rhs.underlyingId();
25     }
26   
27     friend std::ostream& operator<<(std::ostream& stream, type_index typeId) {
28       return stream << typeId.underlyingId();
29     }
30   };

34   template <typename T>
35   inline constexpr c10::c10_string_view fully_qualified_type_name_impl() {
36   #if defined(_MSC_VER) && !defined(__clang__)
37     constexpr std::string_view fun_sig = __FUNCSIG__;
38   #if defined(__NVCC__)
39     constexpr std::string_view prefix =
40         "c10::basic_string_view<char> c10::util::detail::fully_qualified_type_name_impl<";
41     constexpr std::string_view suffix = ">()";
42   #else
43     constexpr std::string_view prefix =
44         "class c10::basic_string_view<char> __cdecl c10::util::detail::fully_qualified_type_name_impl<";
45     constexpr std::string_view suffix = ">(void)";
46   #endif
47   #elif defined(__clang__)
48     constexpr std::string_view fun_sig = __PRETTY_FUNCTION__;
49     constexpr std::string_view prefix =
50         "c10::c10_string_view c10::util::detail::fully_qualified_type_name_impl() [T = ";
51     constexpr std::string_view suffix = "]";
52   #elif defined(__GNUC__)
53     constexpr std::string_view fun_sig = __PRETTY_FUNCTION__;
54     constexpr std::string_view prefix =
55         "constexpr c10::c10_string_view c10::util::detail::fully_qualified_type_name_impl() [with T = ";
56     constexpr std::string_view suffix =
57         "; c10::c10_string_view = c10::basic_string_view<char>]";
58   #endif
59   #if !defined(__CUDA_ARCH__) && !defined(__CUDA_ARCH_LIST__)
60     static_assert(c10::starts_with(
61         static_cast<std::string_view>(fun_sig),
62         static_cast<std::string_view>(prefix)));
63     static_assert(c10::ends_with(
64         static_cast<std::string_view>(fun_sig),
65         static_cast<std::string_view>(suffix)));
66   #endif
67     return fun_sig.substr(
68         prefix.size(), fun_sig.size() - prefix.size() - suffix.size());
69   }

71   #if !defined(__CUDA_ARCH__) && !defined(__CUDA_ARCH_LIST__)
72   template <typename T>
73   inline constexpr uint64_t type_index_impl() {
74   // Idea: __PRETTY_FUNCTION__ (or __FUNCSIG__ on msvc) contains a qualified name
75   // of this function, including its template parameter, i.e. including the
76   // type we want an id for. We use this name and run crc64 on it to get a type
77   // id.
78   #if defined(_MSC_VER) && !defined(__clang__)
79     return crc64(__FUNCSIG__, sizeof(__FUNCSIG__)).checksum();
80   #elif defined(__clang__)
81     return crc64(__PRETTY_FUNCTION__, sizeof(__PRETTY_FUNCTION__)).checksum();
82   #elif defined(__GNUC__)
83     return crc64(__PRETTY_FUNCTION__, sizeof(__PRETTY_FUNCTION__)).checksum();
84   #endif
85   }
86   #endif

90   template <typename T>
91   inline constexpr type_index get_type_index() {
92   #if !defined(__CUDA_ARCH__) && !defined(__CUDA_ARCH_LIST__)
93     // To enforce that this is really computed at compile time, we pass the
94     // type index through std::integral_constant.
95     return type_index{std::integral_constant<
96         uint64_t,
97         detail::type_index_impl<std::decay_t<T>>()>::value};
98   #else
99     // There's nothing in theory preventing us from running this on device code
100    // except for nvcc throwing a compiler error if we enable it.
101    return (abort(), type_index(0));
102  #endif
103  }

105  #if !defined(TORCH_PEDANTIC)
106  // Use precomputed hashsum for std::string
107  // Needed to workaround ambiguity in class name resolution
108  // into __PRETTY_FUNCTION__ when abovementioned class is defined in inlined
109  // namespace. In multi-ABI C++ library, `std::string` is an alias to
110  // `std::__cxx11::basic_string<char>` which depending on compiler flags can be
111  // resolved to `basic_string<char>` either in `std` namespace or in
112  // `std::__cxx11` one (`__cxx11` is an inline namespace)
113  template <>
114  inline constexpr type_index get_type_index<std::string>() {
115    // hashsum for std::basic_string<char>
116    return type_index{4193213214807308375ULL};
117  }
118  #endif

120  template <typename T>
121  inline constexpr std::string_view get_fully_qualified_type_name() noexcept {
122    return static_cast<std::string_view>(
123        detail::fully_qualified_type_name_impl<T>());
124  }
