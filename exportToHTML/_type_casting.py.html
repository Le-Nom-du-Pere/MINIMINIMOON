6    def unpack_float4x2_as_uint8(tensor: torch.Tensor) -> np.ndarray:
7        """Convert a float4x2 tensor to unpacked uint8 np array."""
8        assert tensor.dtype == torch.float4_e2m1fn_x2
9        data = tensor.view(torch.uint8).numpy(force=True).flatten()
10       result_size = tensor.numel() * 2
11       result = np.empty([result_size], dtype=np.uint8)
12       array_low = data & np.uint8(0x0F)
13       array_high = data & np.uint8(0xF0)
14       array_high >>= np.uint8(4)
15       result[0::2] = array_low
16       result[1::2] = array_high
17       result.resize(get_float4_shape(tensor), refcheck=False)
18       return result

21   def get_float4_shape(tensor: torch.Tensor) -> tuple[int, ...]:
22       """Get the shape of an unpacked float4 tensor. 
23    
24       The float4_e2m1fn_x2 type is a shell type described in 
25       https://github.com/pytorch/pytorch/issues/146414. 
26    
27       the shell dtype is takes up 1 byte per element and semantically represents 
28       two fp4 values packed into 1 byte. Semantically it represents (*tensor.shape[:-1], tensor.shape[-1]*2) 
29       fp4 elements. 
30       """
31       assert tensor.dtype == torch.float4_e2m1fn_x2
32       return (*tensor.shape[:-1], tensor.shape[-1] * 2)
