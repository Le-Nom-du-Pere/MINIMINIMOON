154  class FrontendError(Exception):
155      def __init__(self, source_range, msg):
156          self.source_range = source_range
157          self.msg = msg
158  
159          # This has to be instantiated here so the ErrorReport is accurate to the
160          # call stack when the FrontendError was raised
161          self.error_report = torch._C.ErrorReport(self.source_range)
162  
163      def __str__(self):
164          return self.msg + self.error_report.what().lstrip()

171  class UnsupportedNodeError(NotSupportedError):
172      def __init__(self, ctx, offending_node, reason=""):
173          # If we don't have a specific token, we default to length of 1
174          node_type = type(offending_node)
175          range_len = len(node_start_tokens.get(node_type, " "))
176          source_range = ctx.make_range(
177              offending_node.lineno,
178              offending_node.col_offset,
179              offending_node.col_offset + range_len,
180          )
181          feature_name = pretty_node_names.get(node_type, node_type.__name__)
182          msg = f"{feature_name} {reason + ' ' if reason else ''}aren't supported"
183          super().__init__(source_range, msg)

200  def get_class_properties(cls, self_name):
201      """ 
202      Get a list of Property objects representing the properties of a class. 
203   
204      Args: 
205          cls:  The class to get properties of. 
206          self_name: The name of the class that the properties should belong to. 
207      Returns: 
208          A list of Property objects corresponding to the properties of cls. Property 
209          here refers to the subclass of TreeView. 
210      """
211      props = inspect.getmembers(cls, predicate=lambda m: isinstance(m, property))
212      # Any property that should not compiled must be in this list on the Module.
213      unused_properties = getattr(cls, "__jit_unused_properties__", [])
214  
215      # Create Property TreeView objects from inspected property objects.
216      properties = []
217      for prop in props:
218          if prop[0] not in unused_properties and not should_drop(prop[1].fget):
219              getter = get_jit_def(
220                  prop[1].fget, f"__{prop[0]}_getter", self_name=self_name
221              )
222              setter = (
223                  get_jit_def(prop[1].fset, f"__{prop[0]}_setter", self_name=self_name)
224                  if prop[1].fset
225                  else None
226              )
227              properties.append(
228                  Property(getter.range(), Ident(getter.range(), prop[0]), getter, setter)
229              )
230  
231      return properties

234  def get_class_assigns(ctx, cls_ast):
235      assigns = []
236  
237      def maybe_build_assign(builder, entry):
238          nonlocal assigns
239          try:
240              assigns.append(builder(ctx, entry))
241          except NotSupportedError:
242              pass
243  
244      for entry in cls_ast.body:
245          if isinstance(entry, ast.Assign):
246              maybe_build_assign(StmtBuilder.build_Assign, entry)
247          elif isinstance(entry, ast.AnnAssign):
248              maybe_build_assign(StmtBuilder.build_AnnAssign, entry)
249      return assigns

252  def get_jit_class_def(cls, self_name):
253      """Get definitions for each method within the current class independently. 
254   
255      Args: 
256          cls: The class to get definition of. 
257          self_name: The name of the class that the properties should belong to. 
258   
259      Returns: 
260          torch._C._jit_tree_views.ClassDef: A representation of the class, 
261              the methods in the class and their definition as a tree. 
262      """
263      # TODO: proper overriding analysis when implementing class inheritance
264      methods = inspect.getmembers(
265          cls,
266          predicate=lambda m: (inspect.ismethod(m) or inspect.isfunction(m))
267          and not is_static_fn(cls, m.__name__)
268          and m.__name__ in cls.__dict__
269          and not _is_drop_fn(m),
270      )
271  
272      def is_classmethod(fn):
273          return inspect.ismethod(fn) and getattr(fn, "__self__", None) == cls
274  
275      # Get and parse the source code for this class
276      sourcelines, file_lineno, filename = get_source_lines_and_file(
277          cls, torch._C.ErrorReport.call_stack()
278      )
279      source = "".join(sourcelines)
280  
281      dedent_src = dedent(source)
282      py_ast = ast.parse(dedent_src)
283  
284      class_ast = py_ast.body[0]
285      assert isinstance(class_ast, ast.ClassDef)
286  
287      # Special case for dataclasses. In general we need access to the source code for
288      # an object in order to JIT compile it. But the dataclasses module dynamically synthesizes
289      # magic methods for classes, and we can't get the source code for these methods. As a
290      # workaround, we synthesize TorchScript-friendly implementations ourselves.
291      if dataclasses.is_dataclass(cls):
292          # Detect whether the user manually implemented any of the magic methods. If they did,
293          # we don't want to synthesize/override them.
294          overrides = {
295              method.name
296              for method in class_ast.body
297              if isinstance(method, ast.FunctionDef)
298              and method.name in DATACLASS_MAGIC_METHODS
299          }
300          for i, (name, _) in enumerate(methods):
301              # Is this a magic method we can synthesize?
302              synthesizer_fn = DATACLASS_MAGIC_METHODS.get(name)
303              if synthesizer_fn and name not in overrides:
304                  parsed_def = synthesizer_fn(cls)
305                  methods[i] = name, parsed_def
306                  func = getattr(cls, name)
307                  _jit_internal.loader.cache(func, parsed_def.source)
308  
309      method_defs = [
310          get_jit_def(obj, name, self_name=self_name, is_classmethod=is_classmethod(obj))
311          for (name, obj) in methods
312      ]
313      properties = get_class_properties(cls, self_name)
314  
315      leading_whitespace_len = len(source.split("\n", 1)[0]) - len(
316          dedent_src.split("\n", 1)[0]
317      )
318      ctx = make_source_context(
319          source, filename, file_lineno, leading_whitespace_len, False
320      )
321      assigns = get_class_assigns(ctx, class_ast)
322  
323      return build_class_def(ctx, class_ast, method_defs, properties, self_name, assigns)

326  def get_jit_def(fn, def_name, self_name=None, is_classmethod=False):
327      """ 
328      Build a JIT AST (TreeView) from the given function. 
329   
330      Args: 
331          fn: A function object to compile or a pre-parsed ParsedDef object 
332          def_name: The name to give to the resulting AST object. This is not 
333              always the same as `fn.__name__`, for example: 
334                  def _forward(self): 
335                      ... 
336                  forward = _forward 
337              In this case, the `__name__` attribute of the function object is "_forward", 
338              but we want the result AST to have the name "forward". 
339          self_name: If this function is a method, what the type name of `self` is. 
340      """
341      parsed_def = parse_def(fn) if not isinstance(fn, _ParsedDef) else fn
342      type_line = torch.jit.annotations.get_type_line(parsed_def.source)
343      fn_def = parsed_def.ast.body[0]
344  
345      if is_classmethod:
346          arg_name = fn_def.args.args[0].arg  # type:ignore[union-attr]
347          # Insert a statement that assigns the first argument to the class
348          assign_stmt = ast.parse(f"{arg_name} = {self_name}").body[0]
349          fn_def.body.insert(0, assign_stmt)  # type:ignore[union-attr]
350  
351      # Swap out the function signature and body if it is unused
352      if should_drop(fn):
353          unused_fn_def = ast.parse(
354              'def unused_fn(self: Any):\n\traise RuntimeError("Cannot call @unused methods")'
355          )
356          if len(unused_fn_def.body) != 1 or not isinstance(
357              unused_fn_def.body[0], ast.FunctionDef
358          ):
359              raise RuntimeError(
360                  f"Expected a single top-level function: {parsed_def.filename}:{parsed_def.file_lineno}"
361              )
362          unused_def = unused_fn_def.body[0]
363          fn_def.body = unused_def.body  # type:ignore[union-attr]
364          # kwarg/vararg not supported by `build_def`
365          fn_def.args.kwarg = fn_def.args.vararg = None  # type:ignore[union-attr]
366          for arg in fn_def.args.args + fn_def.args.kwonlyargs:  # type:ignore[union-attr]
367              # Replace potentially unsupported type annotations by "Any"
368              arg.annotation = unused_def.args.args[0].annotation
369          if _is_drop_fn(fn):
370              # Dropping potentially unsupported return type annotation for jit._drop
371              fn_def.returns = None  # type:ignore[union-attr]
372              fn_def.type_comment = None  # type:ignore[union-attr]
373  
374      # If MonkeyType is installed, get all the consolidated type traces
375      # for the arguments from type_trace_db
376      type_trace_db = torch.jit._script._get_type_trace_db()
377      pdt_arg_types = None
378      if monkeytype_trace and not isinstance(fn, _ParsedDef):  # type: ignore[truthy-function]
379          qualname = get_qualified_name(fn)
380          pdt_arg_types = type_trace_db.get_args_types(qualname)
381  
382      return build_def(
383          parsed_def.ctx,
384          fn_def,
385          type_line,
386          def_name,
387          self_name=self_name,
388          pdt_arg_types=pdt_arg_types,
389      )

393  def is_torch_jit_ignore_context_manager(stmt):
394      # checks if the statement is torch.jit.ignore context manager
395      if isinstance(stmt.items[0].context_expr, ast.Call):
396          # extract torch part
397          function = stmt.items[0].context_expr.func
398          if isinstance(function, ast.Attribute):
399              attr_name = function.attr
400              attr_value = function.value
401              if attr_name == "_IgnoreContextManager" and isinstance(
402                  attr_value, ast.Attribute
403              ):
404                  # there should be at most two nested attributes (e.g torch.jit._IgnoreContextManager)
405                  if attr_value.attr == "jit" and isinstance(attr_value.value, ast.Name):
406                      if attr_value.value.id == "torch":
407                          return True
408      return False

428  def build_def(ctx, py_def, type_line, def_name, self_name=None, pdt_arg_types=None):
429      body = py_def.body
430      r = ctx.make_range(py_def.lineno, py_def.col_offset, py_def.col_offset + len("def"))
431  
432      param_list = build_param_list(ctx, py_def.args, self_name, pdt_arg_types)
433      return_type = None
434      if getattr(py_def, "returns", None) is not None:
435          return_type = build_expr(ctx, py_def.returns)
436  
437      decl = Decl(r, param_list, return_type)
438      is_method = self_name is not None
439      if type_line is not None:
440          type_comment_decl = torch._C.parse_type_comment(type_line)
441          decl = torch._C.merge_type_from_type_comment(decl, type_comment_decl, is_method)
442  
443      return Def(Ident(r, def_name), decl, build_stmts(ctx, body))

452  def build_param_list(ctx, py_args, self_name, pdt_arg_types=None):
453      if py_args.kwarg is not None:
454          expr = py_args.kwarg
455          ctx_range = ctx.make_range(
456              expr.lineno, expr.col_offset - 1, expr.col_offset + len(expr.arg)
457          )
458          raise NotSupportedError(ctx_range, _vararg_kwarg_err)
459      if py_args.vararg is not None:
460          expr = py_args.vararg
461          ctx_range = ctx.make_range(
462              expr.lineno, expr.col_offset - 1, expr.col_offset + len(expr.arg)
463          )
464          raise NotSupportedError(ctx_range, _vararg_kwarg_err)
465      if len(py_args.kw_defaults) > 0:
466          # kw_defaults is a list of the values for the kwargs (which default to None),
467          # so they don't actually have line numbers.
468          for arg in py_args.kw_defaults:
469              if arg is not None:
470                  ctx_range = build_expr(ctx, arg).range()
471                  raise NotSupportedError(ctx_range, _vararg_kwarg_err)
472  
473      # List of Tuple of args and type as inferred by profile directed typing
474      arg_and_types = [
475          (
476              arg,
477              pdt_arg_types[arg.arg]
478              if pdt_arg_types and bool(pdt_arg_types[arg.arg])
479              else None,
480          )
481          for arg in py_args.args
482      ]
483      arg_and_types_kwonlyargs = [
484          (
485              arg,
486              pdt_arg_types[arg.arg]
487              if pdt_arg_types and bool(pdt_arg_types[arg.arg])
488              else None,
489          )
490          for arg in py_args.kwonlyargs
491      ]
492  
493      result = [
494          build_param(ctx, arg, self_name, kwarg_only=False, pdt_arg_type=arg_type)
495          for arg, arg_type in arg_and_types
496      ]
497      result += [
498          build_param(ctx, arg, self_name, kwarg_only=True, pdt_arg_type=arg_type)
499          for arg, arg_type in arg_and_types_kwonlyargs
500      ]
501      return result

504  def build_param(ctx, py_arg, self_name, kwarg_only, pdt_arg_type=None):
505      # NB: In Python3 py_arg is a pair of (str arg, expr? annotation)
506      name = py_arg.arg
507      r = ctx.make_range(py_arg.lineno, py_arg.col_offset, py_arg.col_offset + len(name))
508      if getattr(py_arg, "annotation", None) is not None:
509          annotation_expr = build_expr(ctx, py_arg.annotation)
510      elif pdt_arg_type:
511          annotation_expr = Var(Ident(r, pdt_arg_type))
512      elif self_name is not None and name == "self":
513          annotation_expr = Var(Ident(r, self_name))
514      else:
515          annotation_expr = EmptyTypeAnnotation(r)
516      return Param(annotation_expr, Ident(r, name), kwarg_only)

519  def build_ignore_context_manager(ctx, stmt):
520      InputType = namedtuple("InputType", ["name", "ann"])
521      OutputType = namedtuple("OutputType", ["name", "ann"])
522  
523      def process_ins_outs(args):
524          # parse the context manager to figure out inputs and outputs
525          # with their annotated types
526          # TODO: add input, output validator
527          inputs = []
528          outputs = []
529          for arg in args:
530              var_name = arg.arg
531              var_ann = arg.value.value
532              var_decl_type, var_ann = var_ann.split(":")
533              if var_decl_type == "inp":
534                  inputs.append(InputType(var_name, var_ann))
535              if var_decl_type == "out":
536                  outputs.append(OutputType(var_name, var_ann))
537          return inputs, outputs
538  
539      def create_unique_name_ext(ctx, stmt):
540          # extension will be based on the full path filename plus
541          # the line number of original context manager
542          fn = re.sub(r"[^a-zA-Z0-9_]", "_", ctx.filename)
543          return f"{fn}_{stmt.lineno}"
544  
545      def build_return_ann_stmt(outputs):
546          return_type_ann = ""
547          return_statement_str = "return "
548          if len(outputs) == 0:
549              return_type_ann += " -> None"
550          if len(outputs) == 1:
551              return_type_ann = " -> " + outputs[0].ann
552              return_statement_str += outputs[0].name
553          if len(outputs) > 1:
554              return_type_ann = " -> tuple"
555              return_type_ann += "[" + ", ".join([var.ann for var in outputs]) + "]"
556              return_statement_str += ", ".join([var.name for var in outputs])
557          return return_type_ann, return_statement_str
558  
559      def build_args(args):
560          return ", ".join([arg.name for arg in args])
561  
562      inputs, outputs = process_ins_outs(stmt.items[0].context_expr.keywords)
563  
564      # build the replacement function str with given inputs and outputs
565      ignore_function_name = "func_ignore_" + create_unique_name_ext(ctx, stmt)
566      ignore_function_str = "\ndef " + ignore_function_name
567      ignore_function_str += (
568          "(" + ", ".join([var.name + " :" + var.ann for var in inputs]) + ")"
569      )
570  
571      return_ann, return_stmt = build_return_ann_stmt(outputs)
572      ignore_function_str += return_ann + ": pass"
573  
574      # first create the functionDef object from just declaration
575      ignore_function = ast.parse(ignore_function_str).body[0]
576  
577      # dump the body of context manager to dummy function
578      ignore_function.body = stmt.body  # type: ignore[attr-defined]
579  
580      # insert return statement to the function
581      return_stmt = ast.parse(return_stmt).body[0]
582      ignore_function.body.append(return_stmt)  # type: ignore[attr-defined]
583  
584      ignore_func_str = f"""\ 
585  # Backward compat: These used to be imported into the outer global scope so some
586  # code may still expect them.
587  from typing import List, Dict, Tuple
588  
589  @torch.jit.ignore
590  {astunparse.unparse(ignore_function)}
591  """
592      g = copy.copy(globals())
593      exec(ignore_func_str, g)  # noqa: P204
594      # registers the custom function in the global context
595      globals()[ignore_function_name] = g[ignore_function_name]
596  
597      # build the statements as:
598      # <out_1>, <out_2>, ... = torch.jit.frontend.<func>(<in_1>, <in_2>)
599      assign_str_lhs = build_args(outputs)
600      # this function will be registered in torch.jit.frontend module by default
601      assign_str_rhs = (
602          f"torch.jit.frontend.{ignore_function_name}(" + build_args(inputs) + ")"
603      )
604  
605      if len(outputs) > 0:
606          assign_str = assign_str_lhs + " = " + assign_str_rhs
607      else:
608          assign_str = assign_str_rhs
609      assign_ast = ast.parse(assign_str).body[0]
610      return assign_ast

613  def get_default_args(fn):
614      """ 
615      Get a dictionary of default arguments for a function. 
616   
617      Args: 
618          fn: Callable - The function to inspect for default arguments. 
619      Returns: 
620          (Dict[str, Any]): mapping argument names to their default values if 
621          :attr:`fn` is not None, else empty dictionary. 
622      """
623      if fn is None:
624          return {}
625  
626      signature = inspect.signature(fn)
627  
628      return {
629          k: v.default
630          for k, v in signature.parameters.items()
631          if v.default is not inspect.Parameter.empty
632      }

635  def get_default_args_for_class(cls):
636      """ 
637      Get default arguments for all methods in a class (except for static methods). 
638   
639      Args: 
640          cls: type - The class type to inspect for default arguments. 
641      Returns: 
642          A Dict[str, Dict[str, Any]] which maps each method name to a Dict[str, Any] 
643          that maps each argument name to its default value. 
644      """
645      # Get methods (except static methods because those are compiled separately as
646      # if they were independent script functions).
647      methods = inspect.getmembers(
648          cls,
649          predicate=lambda m: (inspect.ismethod(m) or inspect.isfunction(m))
650          and not is_static_fn(cls, m.__name__)
651          and m.__name__ in cls.__dict__,
652      )
653  
654      # Get method defaults. Property defaults do not need to be considered
655      # because setters cannot be invoked without a value.
656      defaults = {
657          method_name: get_default_args(method_impl)
658          for method_name, method_impl in methods
659      }
660  
661      return defaults

855  class ExprBuilder(Builder):
856      binop_map = {
857          ast.Add: "+",
858          ast.Sub: "-",
859          ast.Mult: "*",
860          ast.Div: "/",
861          ast.Pow: "**",
862          ast.Mod: "%",
863          ast.FloorDiv: "//",
864          ast.BitAnd: "&",
865          ast.BitXor: "^",
866          ast.BitOr: "|",
867          ast.LShift: "<<",
868          ast.RShift: ">>",
869      }
870  
871      binop_map[ast.MatMult] = "@"
872  
873      unop_map = {
874          ast.Not: "not",
875          ast.USub: "-",
876          ast.Invert: "~",
877      }
878  
879      boolop_map = {
880          ast.And: "and",
881          ast.Or: "or",
882      }

884      cmpop_map = {
885          ast.Eq: "==",
886          ast.NotEq: "!=",
887          ast.LtE: "<=",
888          ast.Lt: "<",
889          ast.GtE: ">=",
890          ast.Gt: ">",
891          ast.Is: "is",
892          ast.IsNot: "is not",
893          ast.In: "in",
894          ast.NotIn: "not in",

897      @staticmethod
898      def build_Attribute(ctx, expr):
899          base = build_expr(ctx, expr.value)
900          # expr.attr is just a string, so it's not annotated in any way, so we have
901          # to build the range manually
902          source = ctx.source.encode("utf-8")
903  
904          def get_char(index):
905              return chr(source[index])
906  
907          start_pos = base.range().end + 1
908          while get_char(start_pos) in string.whitespace:  # Skip whitespace
909              start_pos += 1
910          end_pos = start_pos + len(expr.attr)
911          name_range = ctx.make_raw_range(start_pos, end_pos)
912          return Select(base, Ident(name_range, expr.attr))

914      @staticmethod
915      def build_Call(ctx, expr):
916          func = build_expr(ctx, expr.func)
917          args = [build_expr(ctx, py_arg) for py_arg in expr.args]
918          if hasattr(expr, "starargs") and expr.starargs:
919              stararg_expr = build_expr(ctx, expr.starargs)
920              args += [Starred(stararg_expr.range(), stararg_expr)]
921          kwargs = []
922          for kw in expr.keywords:
923              kw_expr = build_expr(ctx, kw.value)
924              # XXX: we could do a better job at figuring out the range for the name here
925              if not kw.arg:
926                  raise NotSupportedError(
927                      kw_expr.range(), "keyword-arg expansion is not supported"
928                  )
929              kwargs.append(Attribute(Ident(kw_expr.range(), kw.arg), kw_expr))
930          return Apply(func, args, kwargs)

942          if expr.id.startswith(_reserved_prefix):
943              raise NotSupportedError(
944                  r,
945                  "names of variables used in JIT-ed functions "
946                  "can't start with " + _reserved_prefix,
947              )
948          if expr.id == "True":
949              return TrueLiteral(r)
950          elif expr.id == "False":
951              return FalseLiteral(r)
952          elif expr.id == "None":

961              expr.lineno, expr.col_offset, expr.col_offset + len(str(expr.value))
962          )

964              return TrueLiteral(r)
965          elif expr.value is False:
966              return FalseLiteral(r)
967          elif expr.value is None:
968              return NoneLiteral(r)

971          else:
972              raise ValueError("Name constant value unsupported: " + str(expr.value))

974      @staticmethod
975      def build_BinOp(ctx, expr):
976          lhs = build_expr(ctx, expr.left)
977          rhs = build_expr(ctx, expr.right)
978          op = type(expr.op)
979  
980          if op == ast.Div and not ctx.uses_true_division:
981              err_range = ctx.make_raw_range(lhs.range().end, rhs.range().start)
982              raise FrontendError(
983                  err_range,
984                  "Division of ints in TorchScript uses Python 3 true "
985                  "division semantics. Please put `from __future__ "
986                  "import division` at the top of your file",
987              )
988          op_token = ExprBuilder.binop_map.get(op)
989          if op_token is None:
990              err_range = ctx.make_raw_range(lhs.range().end, rhs.range().start)
991              raise NotSupportedError(
992                  err_range, "unsupported binary operator: " + op.__name__
993              )
994          return BinOp(op_token, lhs, rhs)

996      @staticmethod
997      def build_UnaryOp(ctx, expr):
998          sub_expr = build_expr(ctx, expr.operand)
999          op = type(expr.op)
1000         op_token = ExprBuilder.unop_map.get(op)
1001         if op_token is None:
1002             raise NotSupportedError(
1003                 expr.range(), "unsupported unary operator: " + op.__name__
1004             )
1005         r = ctx.make_range(
1006             expr.lineno, expr.col_offset, expr.col_offset + len(op_token)
1007         )
1008         return UnaryOp(r, op_token, sub_expr)

1010     @staticmethod
1011     def build_BoolOp(ctx, expr):
1012         if len(expr.values) < 2:
1013             raise AssertionError(
1014                 "expected at least 2 values in BoolOp, but got " + str(len(expr.values))
1015             )
1016         sub_exprs = [build_expr(ctx, sub_expr) for sub_expr in expr.values]
1017         op = type(expr.op)
1018         op_token = ExprBuilder.boolop_map.get(op)
1019         if op_token is None:
1020             err_range = ctx.make_raw_range(
1021                 sub_exprs[0].range().end, sub_exprs[1].range().start
1022             )
1023             raise NotSupportedError(
1024                 err_range, "unsupported boolean operator: " + op.__name__
1025             )
1026         lhs = sub_exprs[0]
1027         for rhs in sub_exprs[1:]:
1028             lhs = BinOp(op_token, lhs, rhs)
1029         return lhs

1039     @staticmethod
1040     def build_Compare(ctx, expr):
1041         operands = [build_expr(ctx, e) for e in [expr.left] + list(expr.comparators)]
1042         result = None
1043         for lhs, op_, rhs in zip(operands, expr.ops, operands[1:]):
1044             op = type(op_)
1045             op_token = ExprBuilder.cmpop_map.get(op)
1046             r = ctx.make_raw_range(lhs.range().end, rhs.range().start)
1047             if op_token is None:
1048                 raise NotSupportedError(
1049                     r, "unsupported comparison operator: " + op.__name__
1050                 )
1051 
1052             if op == ast.NotIn:
1053                 # NB: `not in` is just `not( in )`, so we don't introduce new tree view
1054                 # but just make it a nested call in our tree view structure
1055                 in_expr = BinOp("in", lhs, rhs)
1056                 cmp_expr = UnaryOp(r, "not", in_expr)
1057             else:
1058                 cmp_expr = BinOp(op_token, lhs, rhs)
1059 
1060             if result is None:
1061                 result = cmp_expr
1062             else:
1063                 result = BinOp("and", result, cmp_expr)
1064         return result

1066     @staticmethod
1067     def build_Subscript(ctx, expr):
1068         def build_SliceExpr(ctx, base, slice_expr):
1069             lower = (
1070                 build_expr(ctx, slice_expr.lower)
1071                 if slice_expr.lower is not None
1072                 else None
1073             )
1074             upper = (
1075                 build_expr(ctx, slice_expr.upper)
1076                 if slice_expr.upper is not None
1077                 else None
1078             )
1079             step = (
1080                 build_expr(ctx, slice_expr.step)
1081                 if slice_expr.step is not None
1082                 else None
1083             )
1084             return SliceExpr(base.range(), lower, upper, step)
1085 
1086         def build_Index(ctx, base, index_expr):
1087             if isinstance(index_expr.value, ast.Tuple):
1088                 raise NotSupportedError(
1089                     base.range(),
1090                     "slicing multiple dimensions with tuples not supported yet",
1091                 )
1092             return build_expr(ctx, index_expr.value)
1093 
1094         def build_ExtSlice(ctx, base, extslice):
1095             sub_exprs = []
1096             for expr in extslice.dims:
1097                 sub_type = type(expr)
1098                 if sub_type is ast.Index:
1099                     sub_exprs.append(build_Index(ctx, base, expr))
1100                 elif sub_type is ast.Slice:
1101                     sub_exprs.append(build_SliceExpr(ctx, base, expr))
1102                 elif sub_type is ast.Constant and expr.value is Ellipsis:
1103                     sub_exprs.append(Dots(base.range()))
1104                 else:
1105                     raise NotSupportedError(
1106                         base.range(),
1107                         f"slicing multiple dimensions with {sub_type} not supported",
1108                     )
1109             return sub_exprs
1110 
1111         base = build_expr(ctx, expr.value)
1112         sub_type = type(expr.slice)
1113         if sub_type is ast.Index:
1114             if isinstance(expr.slice.value, ast.Tuple):
1115                 # N-dimensional indexing using Tuple: x[(i, j, k)] is equivalent to x[i, j, k]
1116                 # XXX: Indexing using a list is **different**! It triggers advanced indexing.
1117                 indices = [
1118                     build_expr(ctx, index_expr) for index_expr in expr.slice.value.elts
1119                 ]
1120                 if not indices:
1121                     # `col_offset` is an int, but `end_col_offset` is
1122                     # `Optional[int]`. The magic number is here to make
1123                     # sure we can parse `()` on any machine
1124                     r = ctx.make_range(
1125                         expr.lineno,
1126                         expr.slice.value.col_offset,
1127                         expr.slice.value.col_offset + 2,
1128                     )
1129                     tup = TupleLiteral(r, [])
1130                     indices.append(tup)
1131                 return Subscript(base, indices)
1132             else:
1133                 return Subscript(base, [build_expr(ctx, expr.slice.value)])
1134         elif sub_type is ast.Slice:
1135             return Subscript(base, [build_SliceExpr(ctx, base, expr.slice)])
1136         elif sub_type is ast.ExtSlice:
1137             return Subscript(base, build_ExtSlice(ctx, base, expr.slice))
1138         else:  # In Python3.9 array indicies are not wrapped in ast.Index
1139             if sub_type is ast.Tuple:
1140                 # N-dimensional indexing using Tuple: x[(i, j, k)] is equivalent to x[i, j, k]
1141                 indices = []
1142                 for index_expr in expr.slice.elts:
1143                     if isinstance(index_expr, ast.Slice):
1144                         indices.append(build_SliceExpr(ctx, base, index_expr))
1145                     else:
1146                         indices.append(build_expr(ctx, index_expr))
1147                 # Special-case logic for `typing.Tuple[()]`
1148                 if not indices:
1149                     # See note above r.e. magic number
1150                     r = ctx.make_range(
1151                         expr.lineno, expr.slice.col_offset, expr.slice.col_offset + 2
1152                     )
1153                     tup = TupleLiteral(r, [])
1154                     indices.append(tup)
1155                 return Subscript(base, indices)
1156             return Subscript(base, [build_expr(ctx, expr.slice)])
