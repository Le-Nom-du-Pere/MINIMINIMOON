42   class Sum(AddWithLimits, ExprWithIntLimits):
43       r""" 
44       Represents unevaluated summation. 
45    
46       Explanation 
47       =========== 
48    
49       ``Sum`` represents a finite or infinite series, with the first argument 
50       being the general form of terms in the series, and the second argument 
51       being ``(dummy_variable, start, end)``, with ``dummy_variable`` taking 
52       all integer values from ``start`` through ``end``. In accordance with 
53       long-standing mathematical convention, the end term is included in the 
54       summation. 
55    
56       Finite sums 
57       =========== 
58    
59       For finite sums (and sums with symbolic limits assumed to be finite) we 
60       follow the summation convention described by Karr [1], especially 
61       definition 3 of section 1.4. The sum: 
62    
63       .. math:: 
64    
65           \sum_{m \leq i < n} f(i) 
66    
67       has *the obvious meaning* for `m < n`, namely: 
68    
69       .. math:: 
70    
71           \sum_{m \leq i < n} f(i) = f(m) + f(m+1) + \ldots + f(n-2) + f(n-1) 
72    
73       with the upper limit value `f(n)` excluded. The sum over an empty set is 
74       zero if and only if `m = n`: 
75    
76       .. math:: 
77    
78           \sum_{m \leq i < n} f(i) = 0  \quad \mathrm{for} \quad  m = n 
79    
80       Finally, for all other sums over empty sets we assume the following 
81       definition: 
82    
83       .. math:: 
84    
85           \sum_{m \leq i < n} f(i) = - \sum_{n \leq i < m} f(i)  \quad \mathrm{for} \quad  m > n 
86    
87       It is important to note that Karr defines all sums with the upper 
88       limit being exclusive. This is in contrast to the usual mathematical notation, 
89       but does not affect the summation convention. Indeed we have: 
90    
91       .. math:: 
92    
93           \sum_{m \leq i < n} f(i) = \sum_{i = m}^{n - 1} f(i) 
94    
95       where the difference in notation is intentional to emphasize the meaning, 
96       with limits typeset on the top being inclusive. 
97    
98       Examples 
99       ======== 
100   
101      >>> from sympy.abc import i, k, m, n, x 
102      >>> from sympy import Sum, factorial, oo, IndexedBase, Function 
103      >>> Sum(k, (k, 1, m)) 
104      Sum(k, (k, 1, m)) 
105      >>> Sum(k, (k, 1, m)).doit() 
106      m**2/2 + m/2 
107      >>> Sum(k**2, (k, 1, m)) 
108      Sum(k**2, (k, 1, m)) 
109      >>> Sum(k**2, (k, 1, m)).doit() 
110      m**3/3 + m**2/2 + m/6 
111      >>> Sum(x**k, (k, 0, oo)) 
112      Sum(x**k, (k, 0, oo)) 
113      >>> Sum(x**k, (k, 0, oo)).doit() 
114      Piecewise((1/(1 - x), Abs(x) < 1), (Sum(x**k, (k, 0, oo)), True)) 
115      >>> Sum(x**k/factorial(k), (k, 0, oo)).doit() 
116      exp(x) 
117   
118      Here are examples to do summation with symbolic indices.  You 
119      can use either Function of IndexedBase classes: 
120   
121      >>> f = Function('f') 
122      >>> Sum(f(n), (n, 0, 3)).doit() 
123      f(0) + f(1) + f(2) + f(3) 
124      >>> Sum(f(n), (n, 0, oo)).doit() 
125      Sum(f(n), (n, 0, oo)) 
126      >>> f = IndexedBase('f') 
127      >>> Sum(f[n]**2, (n, 0, 3)).doit() 
128      f[0]**2 + f[1]**2 + f[2]**2 + f[3]**2 
129   
130      An example showing that the symbolic result of a summation is still 
131      valid for seemingly nonsensical values of the limits. Then the Karr 
132      convention allows us to give a perfectly valid interpretation to 
133      those sums by interchanging the limits according to the above rules: 
134   
135      >>> S = Sum(i, (i, 1, n)).doit() 
136      >>> S 
137      n**2/2 + n/2 
138      >>> S.subs(n, -4) 
139      6 
140      >>> Sum(i, (i, 1, -4)).doit() 
141      6 
142      >>> Sum(-i, (i, -3, 0)).doit() 
143      6 
144   
145      An explicit example of the Karr summation convention: 
146   
147      >>> S1 = Sum(i**2, (i, m, m+n-1)).doit() 
148      >>> S1 
149      m**2*n + m*n**2 - m*n + n**3/3 - n**2/2 + n/6 
150      >>> S2 = Sum(i**2, (i, m+n, m-1)).doit() 
151      >>> S2 
152      -m**2*n - m*n**2 + m*n - n**3/3 + n**2/2 - n/6 
153      >>> S1 + S2 
154      0 
155      >>> S3 = Sum(i, (i, m, m-1)).doit() 
156      >>> S3 
157      0

213      def doit(self, **hints):
214          if hints.get('deep', True):
215              f = self.function.doit(**hints)
216          else:
217              f = self.function
218  
219          # first make sure any definite limits have summation
220          # variables with matching assumptions
221          reps = {}
222          for xab in self.limits:
223              d = _dummy_with_inherited_properties_concrete(xab)
224              if d:
225                  reps[xab[0]] = d
226          if reps:
227              undo = {v: k for k, v in reps.items()}
228              did = self.xreplace(reps).doit(**hints)
229              if isinstance(did, tuple):  # when separate=True
230                  did = tuple([i.xreplace(undo) for i in did])
231              elif did is not None:
232                  did = did.xreplace(undo)
233              else:
234                  did = self
235              return did
236  
237  
238          if self.function.is_Matrix:
239              expanded = self.expand()
240              if self != expanded:
241                  return expanded.doit()
242              return _eval_matrix_sum(self)
243  
244          for n, limit in enumerate(self.limits):
245              i, a, b = limit
246              dif = b - a
247              if dif == -1:
248                  # Any summation over an empty set is zero
249                  return S.Zero
250              if dif.is_integer and dif.is_negative:
251                  a, b = b + 1, a - 1
252                  f = -f
253  
254              newf = eval_sum(f, (i, a, b))
255              if newf is None:
256                  if f == self.function:
257                      zeta_function = self.eval_zeta_function(f, (i, a, b))
258                      if zeta_function is not None:
259                          return zeta_function
260                      return self
261                  else:
262                      return self.func(f, *self.limits[n:])
263              f = newf
264  
265          if hints.get('deep', True):
266              # eval_sum could return partially unevaluated
267              # result with Piecewise.  In this case we won't
268              # doit() recursively.
269              if not isinstance(f, Piecewise):
270                  return f.doit(**hints)
271  
272          return f

274      def eval_zeta_function(self, f, limits):
275          """ 
276          Check whether the function matches with the zeta function. 
277   
278          If it matches, then return a `Piecewise` expression because 
279          zeta function does not converge unless `s > 1` and `q > 0` 
280          """
281          i, a, b = limits
282          if a.is_comparable and b.is_comparable and a > b:
283              return self.eval_zeta_function(f, (i, b + S.One, a - S.One))
284          if b is not S.Infinity:
285              return
286          w, y, z = Wild('w', exclude=[i]), Wild('y', exclude=[i]), Wild('z', exclude=[i])
287          if result := f.match((w * i + y) ** (-z)):
288              coeff = 1 / result[w] ** result[z]
289              s = result[z]
290              q = result[y] / result[w] + a
291              return Piecewise((coeff * zeta(s, q),
292                                And(Not(Contains(-q, S.Naturals0)), re(s) > S.One)),
293                               (self, True))

295      def _eval_derivative(self, x):
296          """ 
297          Differentiate wrt x as long as x is not in the free symbols of any of 
298          the upper or lower limits. 
299   
300          Explanation 
301          =========== 
302   
303          Sum(a*b*x, (x, 1, a)) can be differentiated wrt x or b but not `a` 
304          since the value of the sum is discontinuous in `a`. In a case 
305          involving a limit variable, the unevaluated derivative is returned. 
306          """
307  
308          # diff already confirmed that x is in the free symbols of self, but we
309          # don't want to differentiate wrt any free symbol in the upper or lower
310          # limits
311          # XXX remove this test for free_symbols when the default _eval_derivative is in
312          if isinstance(x, Symbol) and x not in self.free_symbols:
313              return S.Zero
314  
315          # get limits and the function
316          f, limits = self.function, list(self.limits)
317  
318          limit = limits.pop(-1)
319  
320          if limits:  # f is the argument to a Sum
321              f = self.func(f, *limits)
322  
323          _, a, b = limit
324          if x in a.free_symbols or x in b.free_symbols:
325              return None
326          df = Derivative(f, x, evaluate=True)
327          rv = self.func(df, limit)
328          return rv

331          k, _, upper = self.args[-1]
332          new_upper = upper.subs(n, n + step)
333  
334          if len(self.args) == 2:

336          else:
337              f = self.func(*self.args[:-1])

341      def _eval_simplify(self, **kwargs):
342  
343          function = self.function
344  
345          if kwargs.get('deep', True):
346              function = function.simplify(**kwargs)
347  
348          # split the function into adds
349          terms = Add.make_args(expand(function))
350          s_t = [] # Sum Terms
351          o_t = [] # Other Terms
352  
353          for term in terms:
354              if term.has(Sum):
355                  # if there is an embedded sum here
356                  # it is of the form x * (Sum(whatever))
357                  # hence we make a Mul out of it, and simplify all interior sum terms
358                  subterms = Mul.make_args(expand(term))
359                  out_terms = []
360                  for subterm in subterms:
361                      # go through each term
362                      if isinstance(subterm, Sum):
363                          # if it's a sum, simplify it
364                          out_terms.append(subterm._eval_simplify(**kwargs))
365                      else:
366                          # otherwise, add it as is
367                          out_terms.append(subterm)
368  
369                  # turn it back into a Mul
370                  s_t.append(Mul(*out_terms))
371              else:
372                  o_t.append(term)
373  
374          # next try to combine any interior sums for further simplification
375          from sympy.simplify.simplify import factor_sum, sum_combine
376          result = Add(sum_combine(s_t), *o_t)
377  
378          return factor_sum(result, limits=self.limits)

380      def is_convergent(self):
381          r""" 
382          Checks for the convergence of a Sum. 
383   
384          Explanation 
385          =========== 
386   
387          We divide the study of convergence of infinite sums and products in 
388          two parts. 
389   
390          First Part: 
391          One part is the question whether all the terms are well defined, i.e., 
392          they are finite in a sum and also non-zero in a product. Zero 
393          is the analogy of (minus) infinity in products as 
394          :math:`e^{-\infty} = 0`. 
395   
396          Second Part: 
397          The second part is the question of convergence after infinities, 
398          and zeros in products, have been omitted assuming that their number 
399          is finite. This means that we only consider the tail of the sum or 
400          product, starting from some point after which all terms are well 
401          defined. 
402   
403          For example, in a sum of the form: 
404   
405          .. math:: 
406   
407              \sum_{1 \leq i < \infty} \frac{1}{n^2 + an + b} 
408   
409          where a and b are numbers. The routine will return true, even if there 
410          are infinities in the term sequence (at most two). An analogous 
411          product would be: 
412   
413          .. math:: 
414   
415              \prod_{1 \leq i < \infty} e^{\frac{1}{n^2 + an + b}} 
416   
417          This is how convergence is interpreted. It is concerned with what 
418          happens at the limit. Finding the bad terms is another independent 
419          matter. 
420   
421          Note: It is responsibility of user to see that the sum or product 
422          is well defined. 
423   
424          There are various tests employed to check the convergence like 
425          divergence test, root test, integral test, alternating series test, 
426          comparison tests, Dirichlet tests. It returns true if Sum is convergent 
427          and false if divergent and NotImplementedError if it cannot be checked. 
428   
429          References 
430          ========== 
431   
432          .. [1] https://en.wikipedia.org/wiki/Convergence_tests 
433   
434          Examples 
435          ======== 
436   
437          >>> from sympy import factorial, S, Sum, Symbol, oo 
438          >>> n = Symbol('n', integer=True) 
439          >>> Sum(n/(n - 1), (n, 4, 7)).is_convergent() 
440          True 
441          >>> Sum(n/(2*n + 1), (n, 1, oo)).is_convergent() 
442          False 
443          >>> Sum(factorial(n)/5**n, (n, 1, oo)).is_convergent() 
444          False 
445          >>> Sum(1/n**(S(6)/5), (n, 1, oo)).is_convergent() 
446          True 
447   
448          See Also 
449          ======== 
450   
451          Sum.is_absolutely_convergent 
452          sympy.concrete.products.Product.is_convergent 
453          """
454          p, q, r = symbols('p q r', cls=Wild)
455  
456          sym = self.limits[0][0]
457          lower_limit = self.limits[0][1]
458          upper_limit = self.limits[0][2]
459          sequence_term = self.function.simplify()
460  
461          if len(sequence_term.free_symbols) > 1:
462              raise NotImplementedError("convergence checking for more than one symbol "
463                                        "containing series is not handled")
464  
465          if lower_limit.is_finite and upper_limit.is_finite:
466              return S.true
467  
468          # transform sym -> -sym and swap the upper_limit = S.Infinity
469          # and lower_limit = - upper_limit
470          if lower_limit is S.NegativeInfinity:
471              if upper_limit is S.Infinity:
472                  return Sum(sequence_term, (sym, 0, S.Infinity)).is_convergent() and \
473                          Sum(sequence_term, (sym, S.NegativeInfinity, 0)).is_convergent()
474              from sympy.simplify.simplify import simplify
475              sequence_term = simplify(sequence_term.xreplace({sym: -sym}))
476              lower_limit = -upper_limit
477              upper_limit = S.Infinity
478  
479          sym_ = Dummy(sym.name, integer=True, positive=True)
480          sequence_term = sequence_term.xreplace({sym: sym_})
481          sym = sym_
482  
483          interval = Interval(lower_limit, upper_limit)
484  
485          # Piecewise function handle
486          if sequence_term.is_Piecewise:
487              for func, cond in sequence_term.args:
488                  # see if it represents something going to oo
489                  if cond == True or cond.as_set().sup is S.Infinity:
490                      s = Sum(func, (sym, lower_limit, upper_limit))
491                      return s.is_convergent()
492              return S.true
493  
494          ###  -------- Divergence test ----------- ###
495          try:
496              lim_val = limit_seq(sequence_term, sym)
497              if lim_val is not None and lim_val.is_zero is False:
498                  return S.false
499          except NotImplementedError:
500              pass
501  
502          try:
503              lim_val_abs = limit_seq(abs(sequence_term), sym)
504              if lim_val_abs is not None and lim_val_abs.is_zero is False:
505                  return S.false
506          except NotImplementedError:
507              pass
508  
509          order = O(sequence_term, (sym, S.Infinity))
510  
511          ### --------- p-series test (1/n**p) ---------- ###
512          p_series_test = order.expr.match(sym**p)
513          if p_series_test is not None:
514              if p_series_test[p] < -1:
515                  return S.true
516              if p_series_test[p] >= -1:
517                  return S.false
518  
519          ### ------------- comparison test ------------- ###
520          # 1/(n**p*log(n)**q*log(log(n))**r) comparison
521          n_log_test = (order.expr.match(1/(sym**p*log(1/sym)**q*log(-log(1/sym))**r)) or
522                        order.expr.match(1/(sym**p*(-log(1/sym))**q*log(-log(1/sym))**r)))
523          if n_log_test is not None:
524              if (n_log_test[p] > 1 or
525                  (n_log_test[p] == 1 and n_log_test[q] > 1) or
526                  (n_log_test[p] == n_log_test[q] == 1 and n_log_test[r] > 1)):
527                      return S.true
528              return S.false
529  
530          ### ------------- Limit comparison test -----------###
531          # (1/n) comparison
532          try:
533              lim_comp = limit_seq(sym*sequence_term, sym)
534              if lim_comp is not None and lim_comp.is_number and lim_comp > 0:
535                  return S.false
536          except NotImplementedError:
537              pass
538  
539          ### ----------- ratio test ---------------- ###
540          next_sequence_term = sequence_term.xreplace({sym: sym + 1})
541          from sympy.simplify.combsimp import combsimp
542          from sympy.simplify.powsimp import powsimp
543          ratio = combsimp(powsimp(next_sequence_term/sequence_term))
544          try:
545              lim_ratio = limit_seq(ratio, sym)
546              if lim_ratio is not None and lim_ratio.is_number and lim_ratio is not S.NaN:
547                  if abs(lim_ratio) > 1:
548                      return S.false
549                  if abs(lim_ratio) < 1:
550                      return S.true
551          except NotImplementedError:
552              lim_ratio = None
553  
554          ### ---------- Raabe's test -------------- ###
555          if lim_ratio == 1:  # ratio test inconclusive
556              test_val = sym*(sequence_term/
557                           sequence_term.subs(sym, sym + 1) - 1)
558              test_val = test_val.gammasimp()
559              try:
560                  lim_val = limit_seq(test_val, sym)
561                  if lim_val is not None and lim_val.is_number:
562                      if lim_val > 1:
563                          return S.true
564                      if lim_val < 1:
565                          return S.false
566              except NotImplementedError:
567                  pass
568  
569          ### ----------- root test ---------------- ###
570          # lim = Limit(abs(sequence_term)**(1/sym), sym, S.Infinity)
571          try:
572              lim_evaluated = limit_seq(abs(sequence_term)**(1/sym), sym)
573              if lim_evaluated is not None and lim_evaluated.is_number:
574                  if lim_evaluated < 1:
575                      return S.true
576                  if lim_evaluated > 1:
577                      return S.false
578          except NotImplementedError:
579              pass
580  
581          ### ------------- alternating series test ----------- ###
582          dict_val = sequence_term.match(S.NegativeOne**(sym + p)*q)
583          if not dict_val[p].has(sym) and is_decreasing(dict_val[q], interval):
584              return S.true
585  
586          ### ------------- integral test -------------- ###
587          check_interval = None
588          from sympy.solvers.solveset import solveset
589          maxima = solveset(sequence_term.diff(sym), sym, interval)
590          if not maxima:
591              check_interval = interval
592          elif isinstance(maxima, FiniteSet) and maxima.sup.is_number:
593              check_interval = Interval(maxima.sup, interval.sup)
594          if (check_interval is not None and
595              (is_decreasing(sequence_term, check_interval) or
596              is_decreasing(-sequence_term, check_interval))):
597                  integral_val = Integral(
598                      sequence_term, (sym, lower_limit, upper_limit))
599                  try:
600                      integral_val_evaluated = integral_val.doit()
601                      if integral_val_evaluated.is_number:
602                          return S(integral_val_evaluated.is_finite)
603                  except NotImplementedError:
604                      pass
605  
606          ### ----- Dirichlet and bounded times convergent tests ----- ###
607          # TODO
608          #
609          # Dirichlet_test
610          # https://en.wikipedia.org/wiki/Dirichlet%27s_test
611          #
612          # Bounded times convergent test
613          # It is based on comparison theorems for series.
614          # In particular, if the general term of a series can
615          # be written as a product of two terms a_n and b_n
616          # and if a_n is bounded and if Sum(b_n) is absolutely
617          # convergent, then the original series Sum(a_n * b_n)
618          # is absolutely convergent and so convergent.
619          #
620          # The following code can grows like 2**n where n is the
621          # number of args in order.expr
622          # Possibly combined with the potentially slow checks
623          # inside the loop, could make this test extremely slow
624          # for larger summation expressions.
625  
626          if order.expr.is_Mul:
627              args = order.expr.args
628              argset = set(args)
629  
630              ### -------------- Dirichlet tests -------------- ###
631              m = Dummy('m', integer=True)
632              def _dirichlet_test(g_n):
633                  try:
634                      ing_val = limit_seq(Sum(g_n, (sym, interval.inf, m)).doit(), m)
635                      if ing_val is not None and ing_val.is_finite:
636                          return S.true
637                  except NotImplementedError:
638                      pass

641              def _bounded_convergent_test(g1_n, g2_n):
642                  try:
643                      lim_val = limit_seq(g1_n, sym)
644                      if lim_val is not None and (lim_val.is_finite or (
645                          isinstance(lim_val, AccumulationBounds)
646                          and (lim_val.max - lim_val.min).is_finite)):
647                              if Sum(g2_n, (sym, lower_limit, upper_limit)).is_absolutely_convergent():

650                      pass

701      def euler_maclaurin(self, m=0, n=0, eps=0, eval_integral=True):
702          """ 
703          Return an Euler-Maclaurin approximation of self, where m is the 
704          number of leading terms to sum directly and n is the number of 
705          terms in the tail. 
706   
707          With m = n = 0, this is simply the corresponding integral 
708          plus a first-order endpoint correction. 
709   
710          Returns (s, e) where s is the Euler-Maclaurin approximation 
711          and e is the estimated error (taken to be the magnitude of 
712          the first omitted term in the tail): 
713   
714              >>> from sympy.abc import k, a, b 
715              >>> from sympy import Sum 
716              >>> Sum(1/k, (k, 2, 5)).doit().evalf() 
717              1.28333333333333 
718              >>> s, e = Sum(1/k, (k, 2, 5)).euler_maclaurin() 
719              >>> s 
720              -log(2) + 7/20 + log(5) 
721              >>> from sympy import sstr 
722              >>> print(sstr((s.evalf(), e.evalf()), full_prec=True)) 
723              (1.26629073187415, 0.0175000000000000) 
724   
725          The endpoints may be symbolic: 
726   
727              >>> s, e = Sum(1/k, (k, a, b)).euler_maclaurin() 
728              >>> s 
729              -log(a) + log(b) + 1/(2*b) + 1/(2*a) 
730              >>> e 
731              Abs(1/(12*b**2) - 1/(12*a**2)) 
732   
733          If the function is a polynomial of degree at most 2n+1, the 
734          Euler-Maclaurin formula becomes exact (and e = 0 is returned): 
735   
736              >>> Sum(k, (k, 2, b)).euler_maclaurin() 
737              (b**2/2 + b/2 - 1, 0) 
738              >>> Sum(k, (k, 2, b)).doit() 
739              b**2/2 + b/2 - 1 
740   
741          With a nonzero eps specified, the summation is ended 
742          as soon as the remainder term is less than the epsilon. 
743          """
744          m = int(m)
745          n = int(n)
746          f = self.function
747          if len(self.limits) != 1:
748              raise ValueError("More than 1 limit")
749          i, a, b = self.limits[0]
750          if (a > b) == True:
751              if a - b == 1:
752                  return S.Zero, S.Zero
753              a, b = b + 1, a - 1
754              f = -f
755          s = S.Zero
756          if m:
757              if b.is_Integer and a.is_Integer:
758                  m = min(m, b - a + 1)
759              if not eps or f.is_polynomial(i):
760                  s = Add(*[f.subs(i, a + k) for k in range(m)])
761              else:
762                  term = f.subs(i, a)
763                  if term:
764                      test = abs(term.evalf(3)) < eps
765                      if test == True:
766                          return s, abs(term)
767                      elif not (test == False):
768                          # a symbolic Relational class, can't go further
769                          return term, S.Zero
770                  s = term
771                  for k in range(1, m):
772                      term = f.subs(i, a + k)
773                      if abs(term.evalf(3)) < eps and term != 0:
774                          return s, abs(term)
775                      s += term
776              if b - a + 1 == m:
777                  return s, S.Zero
778              a += m
779          x = Dummy('x')
780          I = Integral(f.subs(i, x), (x, a, b))
781          if eval_integral:
782              I = I.doit()
783          s += I
784  
785          def fpoint(expr):
786              if b is S.Infinity:
787                  return expr.subs(i, a), 0
788              return expr.subs(i, a), expr.subs(i, b)
789          fa, fb = fpoint(f)
790          iterm = (fa + fb)/2
791          g = f.diff(i)
792          for k in range(1, n + 2):
793              ga, gb = fpoint(g)
794              term = bernoulli(2*k)/factorial(2*k)*(gb - ga)
795              if k > n:
796                  break
797              if eps and term:
798                  term_evalf = term.evalf(3)
799                  if term_evalf is S.NaN:
800                      return S.NaN, S.NaN
801                  if abs(term_evalf) < eps:
802                      break
803              s += term
804              g = g.diff(i, 2, simplify=False)
805          return s + iterm, abs(term)

808      def reverse_order(self, *indices):
809          """ 
810          Reverse the order of a limit in a Sum. 
811   
812          Explanation 
813          =========== 
814   
815          ``reverse_order(self, *indices)`` reverses some limits in the expression 
816          ``self`` which can be either a ``Sum`` or a ``Product``. The selectors in 
817          the argument ``indices`` specify some indices whose limits get reversed. 
818          These selectors are either variable names or numerical indices counted 
819          starting from the inner-most limit tuple. 
820   
821          Examples 
822          ======== 
823   
824          >>> from sympy import Sum 
825          >>> from sympy.abc import x, y, a, b, c, d 
826   
827          >>> Sum(x, (x, 0, 3)).reverse_order(x) 
828          Sum(-x, (x, 4, -1)) 
829          >>> Sum(x*y, (x, 1, 5), (y, 0, 6)).reverse_order(x, y) 
830          Sum(x*y, (x, 6, 0), (y, 7, -1)) 
831          >>> Sum(x, (x, a, b)).reverse_order(x) 
832          Sum(-x, (x, b + 1, a - 1)) 
833          >>> Sum(x, (x, a, b)).reverse_order(0) 
834          Sum(-x, (x, b + 1, a - 1)) 
835   
836          While one should prefer variable names when specifying which limits 
837          to reverse, the index counting notation comes in handy in case there 
838          are several symbols with the same name. 
839   
840          >>> S = Sum(x**2, (x, a, b), (x, c, d)) 
841          >>> S 
842          Sum(x**2, (x, a, b), (x, c, d)) 
843          >>> S0 = S.reverse_order(0) 
844          >>> S0 
845          Sum(-x**2, (x, b + 1, a - 1), (x, c, d)) 
846          >>> S1 = S0.reverse_order(1) 
847          >>> S1 
848          Sum(x**2, (x, b + 1, a - 1), (x, d + 1, c - 1)) 
849   
850          Of course we can mix both notations: 
851   
852          >>> Sum(x*y, (x, a, b), (y, 2, 5)).reverse_order(x, 1) 
853          Sum(x*y, (x, b + 1, a - 1), (y, 6, 1)) 
854          >>> Sum(x*y, (x, a, b), (y, 2, 5)).reverse_order(y, x) 
855          Sum(x*y, (x, b + 1, a - 1), (y, 6, 1)) 
856   
857          See Also 
858          ======== 
859   
860          sympy.concrete.expr_with_intlimits.ExprWithIntLimits.index, reorder_limit, 
861          sympy.concrete.expr_with_intlimits.ExprWithIntLimits.reorder 
862   
863          References 
864          ========== 
865   
866          .. [1] Michael Karr, "Summation in Finite Terms", Journal of the ACM, 
867                 Volume 28 Issue 2, April 1981, Pages 305-350 
868                 https://dl.acm.org/doi/10.1145/322248.322255 
869          """
870          l_indices = list(indices)
871  
872          for i, indx in enumerate(l_indices):
873              if not isinstance(indx, int):
874                  l_indices[i] = self.index(indx)
875  
876          e = 1
877          limits = []
878          for i, limit in enumerate(self.limits):
879              l = limit
880              if i in l_indices:
881                  e = -e
882                  l = (limit[0], limit[2] + 1, limit[1] - 1)
883              limits.append(l)
884  
885          return Sum(e * self.function, *limits)

971      ''' 
972      Tries to perform the summation using the telescopic property. 
973   
974      Return None if not possible. 
975      '''
976      (i, a, b) = limits
977      if L.is_Add or R.is_Add:
978          return None
979  
980      # We want to solve(L.subs(i, i + m) + R, m)
981      # First we try a simple match since this does things that
982      # solve doesn't do, e.g. solve(cos(k+m)-cos(k), m) gives
983      # a more complicated solution than m == 0.
984  
985      k = Wild("k")
986      sol = (-R).match(L.subs(i, i + k))
987      s = None
988      if sol and k in sol:
989          s = sol[k]
990          if not (s.is_Integer and L.subs(i, i + s) + R == 0):
991              # invalid match or match didn't work
992              s = None

994      # But there are things that match doesn't do that solve
995      # can do, e.g. determine that 1/(x + m) = 1/(1 - x) when m = 1

999          try:
1000             from sympy.solvers.solvers import solve
1001             sol = solve(L.subs(i, i + m) + R, m) or []
1002         except NotImplementedError:
1003             return None

1005                (L.subs(i, i + si) + R).expand().is_zero]
1006         if len(sol) != 1:
1007             return None
1008         s = sol[0]

1328 def eval_sum_hyper(f, i_a_b):
1329     i, a, b = i_a_b
1330 
1331     if f.is_hypergeometric(i) is False:
1332         return
1333 
1334     if (b - a).is_Integer:
1335         # We are never going to do better than doing the sum in the obvious way
1336         return None
1337 
1338     old_sum = Sum(f, (i, a, b))
1339 
1340     if b != S.Infinity:
1341         if a is S.NegativeInfinity:
1342             res = _eval_sum_hyper(f.subs(i, -i), i, -b)
1343             if res is not None:
1344                 return Piecewise(res, (old_sum, True))
1345         else:
1346             n_illegal = lambda x: sum(x.count(_) for _ in _illegal)
1347             had = n_illegal(f)
1348             # check that no extra illegals are introduced
1349             res1 = _eval_sum_hyper(f, i, a)
1350             if res1 is None or n_illegal(res1) > had:
1351                 return
1352             res2 = _eval_sum_hyper(f, i, b + 1)
1353             if res2 is None or n_illegal(res2) > had:
1354                 return
1355             (res1, cond1), (res2, cond2) = res1, res2
1356             cond = And(cond1, cond2)
1357             if cond == False:
1358                 return None
1359             return Piecewise((res1 - res2, cond), (old_sum, True))
1360 
1361     if a is S.NegativeInfinity:
1362         res1 = _eval_sum_hyper(f.subs(i, -i), i, 1)
1363         res2 = _eval_sum_hyper(f, i, 0)
1364         if res1 is None or res2 is None:
1365             return None
1366         res1, cond1 = res1
1367         res2, cond2 = res2
1368         cond = And(cond1, cond2)
1369         if cond == False or cond.as_set() == S.EmptySet:
1370             return None
1371         return Piecewise((res1 + res2, cond), (old_sum, True))
1372 
1373     # Now b == oo, a != -oo
1374     res = _eval_sum_hyper(f, i, a)
1375     if res is not None:
1376         r, c = res
1377         if c == False:
1378             if r.is_number:
1379                 f = f.subs(i, Dummy('i', integer=True, positive=True) + a)
1380                 if f.is_positive or f.is_zero:
1381                     return S.Infinity
1382                 elif f.is_negative:
1383                     return S.NegativeInfinity
1384             return None
1385         return Piecewise(res, (old_sum, True))

1388 def eval_sum_residue(f, i_a_b):
1389     r"""Compute the infinite summation with residues 
1390  
1391     Notes 
1392     ===== 
1393  
1394     If $f(n), g(n)$ are polynomials with $\deg(g(n)) - \deg(f(n)) \ge 2$, 
1395     some infinite summations can be computed by the following residue 
1396     evaluations. 
1397  
1398     .. math:: 
1399         \sum_{n=-\infty, g(n) \ne 0}^{\infty} \frac{f(n)}{g(n)} = 
1400         -\pi \sum_{\alpha|g(\alpha)=0} 
1401         \text{Res}(\cot(\pi x) \frac{f(x)}{g(x)}, \alpha) 
1402  
1403     .. math:: 
1404         \sum_{n=-\infty, g(n) \ne 0}^{\infty} (-1)^n \frac{f(n)}{g(n)} = 
1405         -\pi \sum_{\alpha|g(\alpha)=0} 
1406         \text{Res}(\csc(\pi x) \frac{f(x)}{g(x)}, \alpha) 
1407  
1408     Examples 
1409     ======== 
1410  
1411     >>> from sympy import Sum, oo, Symbol 
1412     >>> x = Symbol('x') 
1413  
1414     Doubly infinite series of rational functions. 
1415  
1416     >>> Sum(1 / (x**2 + 1), (x, -oo, oo)).doit() 
1417     pi/tanh(pi) 
1418  
1419     Doubly infinite alternating series of rational functions. 
1420  
1421     >>> Sum((-1)**x / (x**2 + 1), (x, -oo, oo)).doit() 
1422     pi/sinh(pi) 
1423  
1424     Infinite series of even rational functions. 
1425  
1426     >>> Sum(1 / (x**2 + 1), (x, 0, oo)).doit() 
1427     1/2 + pi/(2*tanh(pi)) 
1428  
1429     Infinite series of alternating even rational functions. 
1430  
1431     >>> Sum((-1)**x / (x**2 + 1), (x, 0, oo)).doit() 
1432     pi/(2*sinh(pi)) + 1/2 
1433  
1434     This also have heuristics to transform arbitrarily shifted summand or 
1435     arbitrarily shifted summation range to the canonical problem the 
1436     formula can handle. 
1437  
1438     >>> Sum(1 / (x**2 + 2*x + 2), (x, -1, oo)).doit() 
1439     1/2 + pi/(2*tanh(pi)) 
1440     >>> Sum(1 / (x**2 + 4*x + 5), (x, -2, oo)).doit() 
1441     1/2 + pi/(2*tanh(pi)) 
1442     >>> Sum(1 / (x**2 + 1), (x, 1, oo)).doit() 
1443     -1/2 + pi/(2*tanh(pi)) 
1444     >>> Sum(1 / (x**2 + 1), (x, 2, oo)).doit() 
1445     -1 + pi/(2*tanh(pi)) 
1446  
1447     References 
1448     ========== 
1449  
1450     .. [#] http://www.supermath.info/InfiniteSeriesandtheResidueTheorem.pdf 
1451  
1452     .. [#] Asmar N.H., Grafakos L. (2018) Residue Theory. 
1453            In: Complex Analysis with Applications. 
1454            Undergraduate Texts in Mathematics. Springer, Cham. 
1455            https://doi.org/10.1007/978-3-319-94063-2_5 
1456     """
1457     i, a, b = i_a_b
1458 
1459     # If lower limit > upper limit: Karr Summation Convention
1460     if a.is_comparable and b.is_comparable and a > b:
1461         return eval_sum_residue(f, (i, b + S.One, a - S.One))

1465         numer_even = all(i % 2 == 0 for (i,) in numer.monoms())
1466         denom_even = all(i % 2 == 0 for (i,) in denom.monoms())
1467         numer_odd = all(i % 2 == 1 for (i,) in numer.monoms())
1468         denom_odd = all(i % 2 == 1 for (i,) in denom.monoms())
