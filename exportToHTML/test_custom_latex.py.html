18   %override DIV_SYMBOL: DIV

26   def init_custom_parser(modification, transformer=None):
27       latex_grammar = Path(grammar_file).read_text(encoding="utf-8")
28       latex_grammar += modification
29   
30       with tempfile.NamedTemporaryFile() as f:
31           f.write(bytes(latex_grammar, encoding="utf8"))
32           f.flush()
33   
34           parser = LarkLaTeXParser(grammar_file=f.name, transformer=transformer)
35   
36       return parser
37   
38   def test_custom1():
39       # Removes the parser's ability to understand \cdot and \div.
40   
41       parser = init_custom_parser(modification1)
42   
43       with raises(lark.exceptions.UnexpectedCharacters):
44           parser.doparse(r"a \cdot b")
45           parser.doparse(r"x \div y")
46   
47   class CustomTransformer(TransformToSymPyExpr):
48       def number(self, tokens):
49           if "," in tokens[0]:
50               # The Float constructor expects a dot as the decimal separator
51               return sympy.core.numbers.Float(tokens[0].replace(",", "."))
52           else:
53               return sympy.core.numbers.Integer(tokens[0])
54   
55   def test_custom2():
56       # Makes the parser parse commas as the decimal separator instead of dots
57   
58       parser = init_custom_parser(modification2, CustomTransformer)
59   
60       with raises(lark.exceptions.UnexpectedCharacters):
61           # Asserting that the default parser cannot parse numbers which have commas as
62           # the decimal separator
63           parse_latex_lark("100,1")
64           parse_latex_lark("0,009")
65   
66       parser.doparse("100,1")
67       parser.doparse("0,009")
68       parser.doparse("2,71828")
69       parser.doparse("3,14159")
