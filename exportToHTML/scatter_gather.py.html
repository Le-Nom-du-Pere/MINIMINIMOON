22   def _is_namedtuple(obj: Any) -> bool:
23       # Check if type was created from collections.namedtuple or a typing.NamedTuple.
24       return (
25           isinstance(obj, tuple) and hasattr(obj, "_asdict") and hasattr(obj, "_fields")
26       )

49   def scatter(inputs, target_gpus, dim=0):
50       r"""Slice tensors into approximately equal chunks and distributes them across given GPUs. 
51    
52       Duplicates references to objects that are not tensors. 
53       """
54   
55       def scatter_map(obj):
56           if isinstance(obj, torch.Tensor):
57               return Scatter.apply(target_gpus, None, dim, obj)
58           if _is_namedtuple(obj):
59               return [type(obj)(*args) for args in zip(*map(scatter_map, obj))]
60           if isinstance(obj, tuple) and len(obj) > 0:
61               return list(zip(*map(scatter_map, obj)))
62           if isinstance(obj, list) and len(obj) > 0:
63               return [list(i) for i in zip(*map(scatter_map, obj))]
64           if isinstance(obj, dict) and len(obj) > 0:
65               return [type(obj)(i) for i in zip(*map(scatter_map, obj.items()))]
66           return [obj for _ in target_gpus]
67   
68       # After scatter_map is called, a scatter_map cell will exist. This cell
69       # has a reference to the actual function scatter_map, which has references
70       # to a closure that has a reference to the scatter_map cell (because the
71       # fn is recursive). To avoid this reference cycle, we set the function to
72       # None, clearing the cell
73       try:
74           res = scatter_map(inputs)
75       finally:
76           scatter_map = None  # type: ignore[assignment]
77       return res

80   def scatter_kwargs(
81       inputs: tuple[Any, ...],
82       kwargs: Optional[dict[str, Any]],
83       target_gpus: Sequence[Union[int, torch.device]],
84       dim: int = 0,
85   ) -> tuple[tuple[Any, ...], tuple[dict[str, Any], ...]]:
86       r"""Scatter with support for kwargs dictionary."""
87       scattered_inputs = scatter(inputs, target_gpus, dim) if inputs else []
88       scattered_kwargs = scatter(kwargs, target_gpus, dim) if kwargs else []
89       if len(scattered_inputs) < len(scattered_kwargs):
90           scattered_inputs.extend(
91               () for _ in range(len(scattered_kwargs) - len(scattered_inputs))
92           )
93       elif len(scattered_kwargs) < len(inputs):
94           scattered_kwargs.extend(
95               {} for _ in range(len(scattered_inputs) - len(scattered_kwargs))
96           )
97       return tuple(scattered_inputs), tuple(scattered_kwargs)

100  def gather(outputs: Any, target_device: Union[int, torch.device], dim: int = 0) -> Any:
101      r"""Gather tensors from different GPUs on a specified device. 
102   
103      This function is useful for gathering the results of a distributed computation. 
104      It takes a sequence of objects, one for each GPU, and returns a single object 
105      on the specified device. 
106   
107      Args: 
108          outputs (Any): A sequence of objects (potentially tensors) to gather. 
109          target_device (Union[int, torch.device]): The device to gather the tensors to. 
110              Use 'cpu' for CPU to avoid a deprecation warning. 
111          dim (int, optional): The dimension along which to gather. Default: 0. 
112   
113      Returns: 
114          Any: A gathered object (potentially tensor) on the specified device. 
115      """
116  
117      def gather_map(outputs):
118          out = outputs[0]
119          if isinstance(out, torch.Tensor):
120              return Gather.apply(target_device, dim, *outputs)
121          if out is None:
122              return None
123          if isinstance(out, dict):
124              if not all(len(out) == len(d) for d in outputs):
125                  raise ValueError("All dicts must have the same number of keys")
126              return type(out)((k, gather_map([d[k] for d in outputs])) for k in out)
127          if _is_namedtuple(out):
128              return type(out)._make(map(gather_map, zip(*outputs)))
129          return type(out)(map(gather_map, zip(*outputs)))
130  
131      # Recursive function calls like this create reference cycles.
132      # Setting the function to None clears the refcycle.
133      try:
134          res = gather_map(outputs)
135      finally:
136          gather_map = None  # type: ignore[assignment]
137      return res
