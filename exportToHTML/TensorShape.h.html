14   // Check to see if the shape of tensors is compatible
15   // for being concatenated along a given dimension.
16   inline void check_cat_shape_except_dim(
17       const Tensor& first,
18       const Tensor& second,
19       int64_t dimension,
20       int64_t index) {
21     int64_t first_dims = first.dim();
22     int64_t second_dims = second.dim();
23     TORCH_CHECK(
24         first_dims == second_dims,
25         "Tensors must have same number of dimensions: got ",
26         first_dims,
27         " and ",
28         second_dims);
29     for (const auto dim : c10::irange(first_dims)) {
30       if (dim == dimension) {
31         continue;
32       }
33       int64_t first_dim_size = first.sizes()[dim];
34       int64_t second_dim_size = second.sizes()[dim];
35       TORCH_CHECK(
36           first_dim_size == second_dim_size,
37           "Sizes of tensors must match except in dimension ",
38           dimension,
39           ". Expected size ",
40           static_cast<long long>(first_dim_size),
41           " but got size ",
42           static_cast<long long>(second_dim_size),
43           " for tensor number ",
44           index,
45           " in the list.");
46     }
47   }

49   inline void check_cat_no_zero_dim(const MaterializedITensorListRef& tensors) {
50     [[maybe_unused]] int64_t i = 0;
51     for (const Tensor& t : tensors) {
52       TORCH_CHECK(
53           t.dim() > 0,
54           "zero-dimensional tensor (at position ",
55           i,
56           ") cannot be concatenated");
57       i++;
58     }
59   }

61   inline int64_t get_num_splits(
62       const Tensor& self,
63       int64_t split_size,
64       int64_t dim) {
65     TORCH_CHECK(self.dim() != 0, "split expects at least a 1-dimensional tensor");
66     TORCH_CHECK(
67         split_size >= 0,
68         "split expects split_size be non-negative, but got split_size=",
69         split_size);
70     int64_t dim_size = self.size(dim);
71     TORCH_CHECK(
72         split_size > 0 || dim_size == 0,
73         "split_size can only be 0 if dimension size is 0, "
74         "but got dimension size of ",
75         dim_size);
76     // if split_size is 0 and dimension size is 0, there is 1 split.
77     int64_t num_splits = 1;
78     if (split_size != 0) {
79       // ensuring num_splits is at least 1 makes consistent the case where
80       // split_size > dim_size (returns a single split).  We might want to error
81       // here, but keep it for BC.
82       num_splits = std::max<int64_t>((dim_size + split_size - 1) / split_size, 1);
83     }
84     return num_splits;
85   }

87   inline bool have_same_ndims(TensorList tensors) {
88     auto ndim = tensors[0].dim();
89     for (const auto tensor_idx : c10::irange(tensors.size())) {
90       if (tensors[tensor_idx].dim() != ndim) {
91         return false;
92       }
93     }
94     return true;
95   }

97   inline void leading_dimension_matches(TensorList tensors, int64_t dim) {
98     auto tensor_zero_size = tensors[0].sizes();
99     std::vector<c10::SymInt> leading_dim_sizes(
100        tensor_zero_size.begin(), tensor_zero_size.begin() + dim);
101    for (const auto i : c10::irange(tensors.size())) {
102      at::Tensor tensor = tensors[i];
103      for (const auto j : c10::irange(dim)) {
104        TORCH_CHECK(
105            tensor.size(j) == leading_dim_sizes[j],
106            "_chunk_cat expects same sizes of 0,...,dim-1 dimensions for all tensors");
107      }
108    }
109  }

111  inline int64_t preprocess_chunk_cat_inputs(
112      TensorList tensors,
113      int64_t dim,
114      int64_t num_chunks) {
115    TORCH_CHECK(num_chunks >= 1, "_chunk_cat expects positive num_chunks");
116    TORCH_CHECK(
117        !tensors.empty(), "_chunk_cat expects a non-empty input tensor list");
118    auto expected_dtype = tensors[0].dtype();
119    auto expected_device = tensors[0].device();
120    for (const auto i : c10::irange(tensors.size())) {
121      TORCH_CHECK(tensors[i].numel() > 0, "_chunk_cat expects non-empty tensor");
122      TORCH_CHECK(
123          tensors[i].dtype() == expected_dtype,
124          "_chunk_cat expects all input tensors with the same dtype");
125      TORCH_CHECK(
126          tensors[i].device() == expected_device,
127          "_chunk_cat expects all inputs tensors on the same device");
128    }
129    if (have_same_ndims(tensors)) {
130      dim = maybe_wrap_dim(dim, tensors[0].dim());
131    } else {
132      TORCH_CHECK(
133          dim >= 0,
134          "_chunk_cat expects non-negative dim when input tensors have different ndims")
135      for (const auto i : c10::irange(tensors.size())) {
136        TORCH_CHECK(
137            dim < tensors[i].ndimension(),
138            "_chunk_cat expects dim < ndim for all input tensors");
139      }
140    }
141    leading_dimension_matches(tensors, dim);
142    return dim;
143  }
