55   def test_holder_formula():
56       # semiprime
57       assert _holder_formula({3, 5}) == 1
58       assert _holder_formula({5, 11}) == 2
59       # n in A003277 is always 1
60       for n in A003277:
61           assert _holder_formula(set(factorint(n).keys())) == 1
62       # otherwise
63       assert _holder_formula({2, 3, 5, 7}) == 12
