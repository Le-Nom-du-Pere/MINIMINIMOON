18   // RAII guard for "cudaStreamCaptureMode", a thread-local value
19   // that controls the error-checking strictness of a capture.
20   struct C10_CUDA_API CUDAStreamCaptureModeGuard {
21     CUDAStreamCaptureModeGuard(cudaStreamCaptureMode desired)
22         : strictness_(desired) {
23       C10_CUDA_CHECK(cudaThreadExchangeStreamCaptureMode(&strictness_));
24     }
25     CUDAStreamCaptureModeGuard(const CUDAStreamCaptureModeGuard&) = delete;
26     CUDAStreamCaptureModeGuard(CUDAStreamCaptureModeGuard&&) = delete;
27     CUDAStreamCaptureModeGuard& operator=(const CUDAStreamCaptureModeGuard&) =
28         delete;
29     CUDAStreamCaptureModeGuard& operator=(CUDAStreamCaptureModeGuard&&) = delete;
30     ~CUDAStreamCaptureModeGuard() {
31       C10_CUDA_CHECK_WARN(cudaThreadExchangeStreamCaptureMode(&strictness_));
32     }
33   
34    private:
35     cudaStreamCaptureMode strictness_;
36   };
37   
38   // Protects against enum cudaStreamCaptureStatus implementation changes.
39   // Some compilers seem not to like static_assert without the messages.
40   static_assert(
41       int(cudaStreamCaptureStatus::cudaStreamCaptureStatusNone) == 0,
42       "unexpected int(cudaStreamCaptureStatusNone) value");
43   static_assert(
44       int(cudaStreamCaptureStatus::cudaStreamCaptureStatusActive) == 1,
45       "unexpected int(cudaStreamCaptureStatusActive) value");
46   static_assert(
47       int(cudaStreamCaptureStatus::cudaStreamCaptureStatusInvalidated) == 2,
48       "unexpected int(cudaStreamCaptureStatusInvalidated) value");
49   
50   enum class CaptureStatus : int {
51     None = int(cudaStreamCaptureStatus::cudaStreamCaptureStatusNone),
52     Active = int(cudaStreamCaptureStatus::cudaStreamCaptureStatusActive),
53     Invalidated = int(cudaStreamCaptureStatus::cudaStreamCaptureStatusInvalidated)
54   };
55   
56   inline std::ostream& operator<<(std::ostream& os, CaptureStatus status) {
57     switch (status) {
58       case CaptureStatus::None:
59         os << "cudaStreamCaptureStatusNone";
60         break;
61       case CaptureStatus::Active:
62         os << "cudaStreamCaptureStatusActive";
63         break;
64       case CaptureStatus::Invalidated:
65         os << "cudaStreamCaptureStatusInvalidated";
66         break;
67       default:
68         TORCH_INTERNAL_ASSERT(
69             false, "Unknown CUDA graph CaptureStatus", int(status));
70     }
71     return os;
72   }
73   
74   // Use this version where you're sure a CUDA context exists already.
75   inline CaptureStatus currentStreamCaptureStatusMayInitCtx() {
76     cudaStreamCaptureStatus is_capturing{cudaStreamCaptureStatusNone};
77     C10_CUDA_CHECK(
78         cudaStreamIsCapturing(c10::cuda::getCurrentCUDAStream(), &is_capturing));
79     return CaptureStatus(is_capturing);
80   }
