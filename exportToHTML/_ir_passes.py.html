22   def rename_inputs(model: ir.Model, new_names: Sequence[str]) -> None:
23       # TODO: Ensure the names do not have duplicates
24       for input, new_name in zip(model.graph.inputs, new_names):
25           input.metadata_props["pkg.torch.onnx.original_node_name"] = str(input.name)

35   def _all_values(model: ir.Model):
36       """Yield all values in a model."""
37       # Yield all values in the model
38       yield from model.graph.inputs
39       yield from model.graph.initializers.values()
40       for node in ir.traversal.RecursiveGraphIterator(model.graph):
41           yield from node.outputs
42       # Yield all values in functions
43       for function in model.functions.values():
44           yield from function.inputs
45           for node in ir.traversal.RecursiveGraphIterator(function):
46               yield from node.outputs

49   def _replace_names(shape_expr: str, rename_mapping: dict[str, str]) -> str:
50       """Replace all known names in a shape expression with new names."""
51       for old_name, new_name in rename_mapping.items():
52           shape_expr = re.sub(
53               rf"(?<!\w){re.escape(old_name)}(?!\w)", new_name, shape_expr
54           )
55       return shape_expr

58   def rename_axis(model: ir.Model, rename_mapping: dict[str, str]) -> None:
59       """Rename dynamic axes in a model according to the specified dynamic_axes names."""
60   
61       # NOTE: Mapping needs to be srted by length because the shape expression
62       # could have multiple ways to be expressed, for example,
63       # {"s1": sequence_length, "s11": "past_sequence_length", "s1 + s11": "masked_sequence_length"}
64       # We prefer the replacement starts from the longest match.
65       sorted_rename_mapping = dict(
66           sorted(rename_mapping.items(), key=lambda item: len(item[0]), reverse=True)
67       )
68       for value in _all_values(model):
69           if value.shape is None:
70               continue
71           new_shape = []
72           changed = False
73           for dim in value.shape:
74               if not isinstance(dim, ir.SymbolicDim):
75                   new_shape.append(dim)
76                   continue
77               dim_name = dim.value
78               if dim_name in sorted_rename_mapping:
79                   new_shape.append(sorted_rename_mapping[dim_name])
80                   changed = True
81               elif dim_name is not None:
82                   # For example: "2*s1", "s1+1", "s1-1", "s1*s2", "s1/s2"
83                   new_name = _replace_names(dim_name, sorted_rename_mapping)
84                   new_shape.append(new_name)
85                   if new_name != dim_name:
86                       changed = True
87               else:
88                   new_shape.append(None)
89           if changed:
90               value.shape = ir.Shape(new_shape)

93   def add_torchlib_common_imports(
94       model: ir.Model, opset_version: int = _constants.TORCHLIB_OPSET
95   ) -> None:
96       """Hack to add torchlib common imports to the model."""
97   
98       try:
99           # TODO(justinchuby): Remove this hack and improved onnxscript
100          from onnxscript.function_libs.torch_lib.ops import common as common_ops
101  
102          model.opset_imports["pkg.onnxscript.torch_lib.common"] = 1
103          rank_func = ir.serde.deserialize_function(common_ops.Rank.to_function_proto())
104          rank_func.opset_imports[""] = opset_version
105          is_scalar_func = ir.serde.deserialize_function(
106              common_ops.IsScalar.to_function_proto()
107          )
108          is_scalar_func.opset_imports[""] = opset_version
109          model.functions[rank_func.identifier()] = rank_func
110          model.functions[is_scalar_func.identifier()] = is_scalar_func
111      except Exception:
112          logger.exception("Failed to add torchlib common imports to the model.")

115  def _maybe_set_opset_version(
116      opset_imports: dict[str, int], domain: str, version: int | None
117  ) -> None:
118      """Set the opset version for the domain."""
119      if domain in opset_imports and opset_imports[domain] != 1:
120          # Already set
121          return
122      if domain == _ONNX_DOMAIN:
123          opset_imports[domain] = _constants.TORCHLIB_OPSET
124          return
125      if version is None:
126          # We don't know the opset version, so set it to 1
127          # This is valid for the custom function domains like "pkg.torch.__subgraph__"
128          opset_imports[domain] = 1
129          return
130      # Set the known opset version for the domain
131      opset_imports[domain] = version

134  def add_opset_imports(model: ir.Model) -> None:
135      """Collect all opsets used and add opset imports to the model and functions."""
136      for node in ir.traversal.RecursiveGraphIterator(model.graph):
137          domain = node.domain
138          _maybe_set_opset_version(model.opset_imports, domain, node.version)
139  
140      for function in model.functions.values():
141          for node in ir.traversal.RecursiveGraphIterator(function):
142              domain = node.domain
143              _maybe_set_opset_version(function.opset_imports, domain, node.version)
144          for domain, version in function.opset_imports.items():
145              # Add all opsets used in the function to the model, because ONNX Runtime
146              # does not handle adding the opset imports to the model after inlining during inference.
147              # This should happen after all opsets are collected for the function from its nodes.
148              _maybe_set_opset_version(model.opset_imports, domain, version)
