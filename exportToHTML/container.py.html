35   def _addindent(s_, numSpaces):
36       s = s_.split("\n")
37       # don't do anything for single-line stuff
38       if len(s) == 1:
39           return s_
40       first = s.pop(0)
41       s = [(numSpaces * " ") + line for line in s]
42       s = "\n".join(s)
43       s = first + "\n" + s
44       return s

59   class Sequential(Module):
60       r"""A sequential container. 
61    
62       Modules will be added to it in the order they are passed in the 
63       constructor. Alternatively, an ``OrderedDict`` of modules can be 
64       passed in. The ``forward()`` method of ``Sequential`` accepts any 
65       input and forwards it to the first module it contains. It then 
66       "chains" outputs to inputs sequentially for each subsequent module, 
67       finally returning the output of the last module. 
68    
69       The value a ``Sequential`` provides over manually calling a sequence 
70       of modules is that it allows treating the whole container as a 
71       single module, such that performing a transformation on the 
72       ``Sequential`` applies to each of the modules it stores (which are 
73       each a registered submodule of the ``Sequential``). 
74    
75       What's the difference between a ``Sequential`` and a 
76       :class:`torch.nn.ModuleList`? A ``ModuleList`` is exactly what it 
77       sounds like--a list for storing ``Module`` s! On the other hand, 
78       the layers in a ``Sequential`` are connected in a cascading way. 
79    
80       Example:: 
81    
82           # Using Sequential to create a small model. When `model` is run, 
83           # input will first be passed to `Conv2d(1,20,5)`. The output of 
84           # `Conv2d(1,20,5)` will be used as the input to the first 
85           # `ReLU`; the output of the first `ReLU` will become the input 
86           # for `Conv2d(20,64,5)`. Finally, the output of 
87           # `Conv2d(20,64,5)` will be used as input to the second `ReLU` 
88           model = nn.Sequential( 
89               nn.Conv2d(1, 20, 5), nn.ReLU(), nn.Conv2d(20, 64, 5), nn.ReLU() 
90           ) 
91    
92           # Using Sequential with OrderedDict. This is functionally the 
93           # same as the above code 
94           model = nn.Sequential( 
95               OrderedDict( 
96                   [ 
97                       ("conv1", nn.Conv2d(1, 20, 5)), 
98                       ("relu1", nn.ReLU()), 
99                       ("conv2", nn.Conv2d(20, 64, 5)), 
100                      ("relu2", nn.ReLU()), 
101                  ] 
102              ) 
103          ) 
104      """
105  
106      _modules: dict[str, Module]  # type: ignore[assignment]
107  
108      @overload
109      def __init__(self, *args: Module) -> None: ...
110  
111      @overload
112      def __init__(self, arg: OrderedDict[str, Module]) -> None: ...
113  
114      def __init__(self, *args):
115          super().__init__()
116          if len(args) == 1 and isinstance(args[0], OrderedDict):
117              for key, module in args[0].items():
118                  self.add_module(key, module)
119          else:
120              for idx, module in enumerate(args):
121                  self.add_module(str(idx), module)
122  
123      def _get_item_by_idx(self, iterator: Iterable[_V], idx: int) -> _V:
124          """Get the idx-th item of the iterator."""
125          size = len(self)
126          idx = operator.index(idx)
127          if not -size <= idx < size:
128              raise IndexError(f"index {idx} is out of range")
129          idx %= size
130          return next(islice(iterator, idx, None))
131  
132      @_copy_to_script_wrapper
133      def __getitem__(self, idx: Union[slice, int]) -> Union[Sequential, Module]:
134          if isinstance(idx, slice):
135              return self.__class__(OrderedDict(list(self._modules.items())[idx]))
136          else:
137              return self._get_item_by_idx(self._modules.values(), idx)
138  
139      def __setitem__(self, idx: int, module: Module) -> None:
140          key: str = self._get_item_by_idx(self._modules.keys(), idx)
141          return setattr(self, key, module)
142  
143      def __delitem__(self, idx: Union[slice, int]) -> None:
144          if isinstance(idx, slice):
145              for key in list(self._modules.keys())[idx]:
146                  delattr(self, key)
147          else:
148              key = self._get_item_by_idx(self._modules.keys(), idx)
149              delattr(self, key)
150          # To preserve numbering
151          str_indices = [str(i) for i in range(len(self._modules))]
152          self._modules = OrderedDict(list(zip(str_indices, self._modules.values())))
153  
154      @_copy_to_script_wrapper
155      def __len__(self) -> int:
156          return len(self._modules)
157  
158      def __add__(self, other) -> Sequential:
159          if isinstance(other, Sequential):
160              ret = Sequential()
161              for layer in self:
162                  ret.append(layer)
163              for layer in other:
164                  ret.append(layer)
165              return ret
166          else:
167              raise ValueError(
168                  "add operator supports only objects "
169                  f"of Sequential class, but {str(type(other))} is given."
170              )
171  
172      def pop(self, key: Union[int, slice]) -> Module:
173          v = self[key]
174          del self[key]
175          return v
176  
177      def __iadd__(self, other) -> Self:
178          if isinstance(other, Sequential):
179              offset = len(self)
180              for i, module in enumerate(other):
181                  self.add_module(str(i + offset), module)
182              return self
183          else:
184              raise ValueError(
185                  "add operator supports only objects "
186                  f"of Sequential class, but {str(type(other))} is given."
187              )
188  
189      def __mul__(self, other: int) -> Sequential:
190          if not isinstance(other, int):
191              raise TypeError(
192                  f"unsupported operand type(s) for *: {type(self)} and {type(other)}"
193              )
194          elif other <= 0:
195              raise ValueError(
196                  f"Non-positive multiplication factor {other} for {type(self)}"
197              )
198          else:
199              combined = Sequential()
200              offset = 0
201              for _ in range(other):
202                  for module in self:
203                      combined.add_module(str(offset), module)
204                      offset += 1
205              return combined
206  
207      def __rmul__(self, other: int) -> Sequential:
208          return self.__mul__(other)
209  
210      def __imul__(self, other: int) -> Self:
211          if not isinstance(other, int):
212              raise TypeError(
213                  f"unsupported operand type(s) for *: {type(self)} and {type(other)}"
214              )
215          elif other <= 0:
216              raise ValueError(
217                  f"Non-positive multiplication factor {other} for {type(self)}"
218              )
219          else:
220              len_original = len(self)
221              offset = len(self)
222              for _ in range(other - 1):
223                  for i in range(len_original):
224                      self.add_module(str(i + offset), self._modules[str(i)])
225                  offset += len_original
226              return self
227  
228      @_copy_to_script_wrapper
229      def __dir__(self) -> list[str]:
230          keys = super().__dir__()
231          keys = [key for key in keys if not key.isdigit()]
232          return keys
233  
234      @_copy_to_script_wrapper
235      def __iter__(self) -> Iterator[Module]:
236          return iter(self._modules.values())

238      # NB: We can't really type check this function as the type of input
239      # may change dynamically (as is tested in
240      # TestScript.test_sequential_intermediary_types).  Cannot annotate
241      # with Any as TorchScript expects a more precise type
242      def forward(self, input):
243          for module in self:
244              input = module(input)
245          return input
246  
247      def append(self, module: Module) -> Self:
248          r"""Append a given module to the end. 
249   
250          Args: 
251              module (nn.Module): module to append 
252   
253          Example:: 
254   
255              >>> import torch.nn as nn 
256              >>> n = nn.Sequential(nn.Linear(1, 2), nn.Linear(2, 3)) 
257              >>> n.append(nn.Linear(3, 4)) 
258              Sequential( 
259                  (0): Linear(in_features=1, out_features=2, bias=True) 
260                  (1): Linear(in_features=2, out_features=3, bias=True) 
261                  (2): Linear(in_features=3, out_features=4, bias=True) 
262              ) 
263   
264          """
265          self.add_module(str(len(self)), module)
266          return self
267  
268      def insert(self, index: int, module: Module) -> Self:
269          """ 
270          Inserts a module into the Sequential container at the specified index. 
271   
272          Args: 
273              index (int): The index to insert the module. 
274              module (Module): The module to be inserted. 
275   
276          Example:: 
277   
278              >>> import torch.nn as nn 
279              >>> n = nn.Sequential(nn.Linear(1, 2), nn.Linear(2, 3)) 
280              >>> n.insert(0, nn.Linear(3, 4)) 
281              Sequential( 
282                  (0): Linear(in_features=3, out_features=4, bias=True) 
283                  (1): Linear(in_features=1, out_features=2, bias=True) 
284                  (2): Linear(in_features=2, out_features=3, bias=True) 
285              ) 
286   
287          """
288          if not isinstance(module, Module):
289              raise AssertionError(f"module should be of type: {Module}")
290          n = len(self._modules)
291          if not (-n <= index <= n):
292              raise IndexError(f"Index out of range: {index}")
293          if index < 0:
294              index += n
295          for i in range(n, index, -1):
296              self._modules[str(i)] = self._modules[str(i - 1)]
297          self._modules[str(index)] = module
298          return self
299  
300      def extend(self, sequential: Iterable[Module]) -> Self:
301          """ 
302          Extends the current Sequential container with layers from another Sequential container. 
303   
304          Args: 
305              sequential (Sequential): A Sequential container whose layers will be added to the current container. 
306   
307          Example:: 
308   
309              >>> import torch.nn as nn 
310              >>> n = nn.Sequential(nn.Linear(1, 2), nn.Linear(2, 3)) 
311              >>> other = nn.Sequential(nn.Linear(3, 4), nn.Linear(4, 5)) 
312              >>> n.extend(other) # or `n + other` 
313              Sequential( 
314                  (0): Linear(in_features=1, out_features=2, bias=True) 
315                  (1): Linear(in_features=2, out_features=3, bias=True) 
316                  (2): Linear(in_features=3, out_features=4, bias=True) 
317                  (3): Linear(in_features=4, out_features=5, bias=True) 
318              ) 
319   
320          """
321          for layer in sequential:
322              self.append(layer)
323          return self

326  class ModuleList(Module):
327      r"""Holds submodules in a list. 
328   
329      :class:`~torch.nn.ModuleList` can be indexed like a regular Python list, but 
330      modules it contains are properly registered, and will be visible by all 
331      :class:`~torch.nn.Module` methods. 
332   
333      Args: 
334          modules (iterable, optional): an iterable of modules to add 
335   
336      Example:: 
337   
338          class MyModule(nn.Module): 
339              def __init__(self) -> None: 
340                  super().__init__() 
341                  self.linears = nn.ModuleList([nn.Linear(10, 10) for i in range(10)]) 
342   
343              def forward(self, x): 
344                  # ModuleList can act as an iterable, or be indexed using ints 
345                  for i, l in enumerate(self.linears): 
346                      x = self.linears[i // 2](x) + l(x) 
347                  return x 
348      """
349  
350      _modules: dict[str, Module]  # type: ignore[assignment]
351  
352      def __init__(self, modules: Optional[Iterable[Module]] = None) -> None:
353          super().__init__()
354          if modules is not None:
355              self += modules
356  
357      def _get_abs_string_index(self, idx):
358          """Get the absolute index for the list of modules."""
359          idx = operator.index(idx)
360          if not (-len(self) <= idx < len(self)):
361              raise IndexError(f"index {idx} is out of range")
362          if idx < 0:
363              idx += len(self)
364          return str(idx)
365  
366      @overload
367      def __getitem__(self, idx: slice) -> ModuleList: ...
368  
369      @overload
370      def __getitem__(self, idx: int) -> Module: ...
371  
372      @_copy_to_script_wrapper
373      def __getitem__(self, idx: Union[int, slice]) -> Union[Module, ModuleList]:
374          if isinstance(idx, slice):
375              return self.__class__(list(self._modules.values())[idx])
376          else:
377              return self._modules[self._get_abs_string_index(idx)]
378  
379      def __setitem__(self, idx: int, module: Module) -> None:
380          idx = self._get_abs_string_index(idx)
381          return setattr(self, str(idx), module)
382  
383      def __delitem__(self, idx: Union[int, slice]) -> None:
384          if isinstance(idx, slice):
385              for k in range(len(self._modules))[idx]:
386                  delattr(self, str(k))
387          else:
388              delattr(self, self._get_abs_string_index(idx))
389          # To preserve numbering, self._modules is being reconstructed with modules after deletion
390          str_indices = [str(i) for i in range(len(self._modules))]
391          self._modules = OrderedDict(list(zip(str_indices, self._modules.values())))
392  
393      @_copy_to_script_wrapper
394      def __len__(self) -> int:
395          return len(self._modules)
396  
397      @_copy_to_script_wrapper
398      def __iter__(self) -> Iterator[Module]:
399          return iter(self._modules.values())
400  
401      def __iadd__(self, modules: Iterable[Module]) -> Self:
402          return self.extend(modules)
403  
404      def __add__(self, other: Iterable[Module]) -> ModuleList:
405          combined = ModuleList()
406          for i, module in enumerate(chain(self, other)):
407              combined.add_module(str(i), module)
408          return combined
409  
410      def __repr__(self) -> str:
411          """Return a custom repr for ModuleList that compresses repeated module representations."""
412          list_of_reprs = [repr(item) for item in self]
413          if len(list_of_reprs) == 0:
414              return self._get_name() + "()"
415  
416          start_end_indices = [[0, 0]]
417          repeated_blocks = [list_of_reprs[0]]
418          for i, r in enumerate(list_of_reprs[1:], 1):
419              if r == repeated_blocks[-1]:
420                  start_end_indices[-1][1] += 1
421                  continue
422  
423              start_end_indices.append([i, i])
424              repeated_blocks.append(r)
425  
426          lines = []
427          main_str = self._get_name() + "("
428          for (start_id, end_id), b in zip(start_end_indices, repeated_blocks):
429              local_repr = f"({start_id}): {b}"  # default repr
430  
431              if start_id != end_id:
432                  n = end_id - start_id + 1
433                  local_repr = f"({start_id}-{end_id}): {n} x {b}"
434  
435              local_repr = _addindent(local_repr, 2)
436              lines.append(local_repr)
437  
438          main_str += "\n  " + "\n  ".join(lines) + "\n"
439          main_str += ")"
440          return main_str
441  
442      @_copy_to_script_wrapper
443      def __dir__(self) -> list[str]:
444          keys = super().__dir__()
445          keys = [key for key in keys if not key.isdigit()]
446          return keys
447  
448      def insert(self, index: int, module: Module) -> None:
449          r"""Insert a given module before a given index in the list. 
450   
451          Args: 
452              index (int): index to insert. 
453              module (nn.Module): module to insert 
454          """
455          for i in range(len(self._modules), index, -1):
456              self._modules[str(i)] = self._modules[str(i - 1)]
457          self._modules[str(index)] = module
458  
459      def append(self, module: Module) -> Self:
460          r"""Append a given module to the end of the list. 
461   
462          Args: 
463              module (nn.Module): module to append 
464          """
465          self.add_module(str(len(self)), module)
466          return self
467  
468      def pop(self, key: Union[int, slice]) -> Module:
469          v = self[key]
470          del self[key]
471          return v
472  
473      def extend(self, modules: Iterable[Module]) -> Self:
474          r"""Append modules from a Python iterable to the end of the list. 
475   
476          Args: 
477              modules (iterable): iterable of modules to append 
478          """
479          if not isinstance(modules, container_abcs.Iterable):
480              raise TypeError(
481                  "ModuleList.extend should be called with an "
482                  "iterable, but got " + type(modules).__name__
483              )
484          offset = len(self)
485          for i, module in enumerate(modules):
486              self.add_module(str(offset + i), module)
487          return self

492  class ModuleDict(Module):
493      r"""Holds submodules in a dictionary. 
494   
495      :class:`~torch.nn.ModuleDict` can be indexed like a regular Python dictionary, 
496      but modules it contains are properly registered, and will be visible by all 
497      :class:`~torch.nn.Module` methods. 
498   
499      :class:`~torch.nn.ModuleDict` is an **ordered** dictionary that respects 
500   
501      * the order of insertion, and 
502   
503      * in :meth:`~torch.nn.ModuleDict.update`, the order of the merged 
504        ``OrderedDict``, ``dict`` (started from Python 3.6) or another 
505        :class:`~torch.nn.ModuleDict` (the argument to 
506        :meth:`~torch.nn.ModuleDict.update`). 
507   
508      Note that :meth:`~torch.nn.ModuleDict.update` with other unordered mapping 
509      types (e.g., Python's plain ``dict`` before Python version 3.6) does not 
510      preserve the order of the merged mapping. 
511   
512      Args: 
513          modules (iterable, optional): a mapping (dictionary) of (string: module) 
514              or an iterable of key-value pairs of type (string, module) 
515   
516      Example:: 
517   
518          class MyModule(nn.Module): 
519              def __init__(self) -> None: 
520                  super().__init__() 
521                  self.choices = nn.ModuleDict( 
522                      {"conv": nn.Conv2d(10, 10, 3), "pool": nn.MaxPool2d(3)} 
523                  ) 
524                  self.activations = nn.ModuleDict( 
525                      [["lrelu", nn.LeakyReLU()], ["prelu", nn.PReLU()]] 
526                  ) 
527   
528              def forward(self, x, choice, act): 
529                  x = self.choices[choice](x) 
530                  x = self.activations[act](x) 
531                  return x 
532      """
533  
534      _modules: dict[str, Module]  # type: ignore[assignment]
535  
536      def __init__(self, modules: Optional[Mapping[str, Module]] = None) -> None:
537          super().__init__()
538          if modules is not None:
539              self.update(modules)
540  
541      @_copy_to_script_wrapper
542      def __getitem__(self, key: str) -> Module:
543          return self._modules[key]
544  
545      def __setitem__(self, key: str, module: Module) -> None:
546          self.add_module(key, module)
547  
548      def __delitem__(self, key: str) -> None:
549          del self._modules[key]
550  
551      @_copy_to_script_wrapper
552      def __len__(self) -> int:
553          return len(self._modules)
554  
555      @_copy_to_script_wrapper
556      def __iter__(self) -> Iterator[str]:
557          return iter(self._modules)
558  
559      @_copy_to_script_wrapper
560      def __contains__(self, key: str) -> bool:
561          return key in self._modules
562  
563      def clear(self) -> None:
564          """Remove all items from the ModuleDict."""
565          self._modules.clear()
566  
567      def pop(self, key: str) -> Module:
568          r"""Remove key from the ModuleDict and return its module. 
569   
570          Args: 
571              key (str): key to pop from the ModuleDict 
572          """
573          v = self[key]
574          del self[key]
575          return v
576  
577      @_copy_to_script_wrapper
578      def keys(self) -> container_abcs.KeysView[str]:
579          r"""Return an iterable of the ModuleDict keys."""
580          return self._modules.keys()
581  
582      @_copy_to_script_wrapper
583      def items(self) -> container_abcs.ItemsView[str, Module]:
584          r"""Return an iterable of the ModuleDict key/value pairs."""
585          return self._modules.items()
586  
587      @_copy_to_script_wrapper
588      def values(self) -> container_abcs.ValuesView[Module]:
589          r"""Return an iterable of the ModuleDict values."""
590          return self._modules.values()
591  
592      def update(self, modules: Mapping[str, Module]) -> None:
593          r"""Update the :class:`~torch.nn.ModuleDict` with key-value pairs from a mapping, overwriting existing keys. 
594   
595          .. note:: 
596              If :attr:`modules` is an ``OrderedDict``, a :class:`~torch.nn.ModuleDict`, or 
597              an iterable of key-value pairs, the order of new elements in it is preserved. 
598   
599          Args: 
600              modules (iterable): a mapping (dictionary) from string to :class:`~torch.nn.Module`, 
601                  or an iterable of key-value pairs of type (string, :class:`~torch.nn.Module`) 
602          """
603          if not isinstance(modules, container_abcs.Iterable):
604              raise TypeError(
605                  "ModuleDict.update should be called with an "
606                  "iterable of key/value pairs, but got " + type(modules).__name__
607              )
608  
609          if isinstance(modules, (OrderedDict, ModuleDict, container_abcs.Mapping)):
610              for key, module in modules.items():
611                  self[key] = module
612          else:
613              # modules here can be a list with two items
614              for j, m in enumerate(modules):
615                  if not isinstance(m, container_abcs.Iterable):
616                      raise TypeError(
617                          "ModuleDict update sequence element "
618                          "#" + str(j) + " should be Iterable; is" + type(m).__name__
619                      )
620                  if not len(m) == 2:
621                      raise ValueError(
622                          "ModuleDict update sequence element "
623                          "#" + str(j) + " has length " + str(len(m)) + "; 2 is required"
624                      )
625                  # modules can be Mapping (what it's typed at), or a list: [(name1, module1), (name2, module2)]
626                  # that's too cumbersome to type correctly with overloads, so we add an ignore here
627                  self[m[0]] = m[1]  # type: ignore[assignment]

632  class ParameterList(Module):
633      r"""Holds parameters in a list. 
634   
635      :class:`~torch.nn.ParameterList` can be used like a regular Python 
636      list, but Tensors that are :class:`~torch.nn.Parameter` are properly registered, 
637      and will be visible by all :class:`~torch.nn.Module` methods. 
638   
639      Note that the constructor, assigning an element of the list, the 
640      :meth:`~torch.nn.ParameterList.append` method and the :meth:`~torch.nn.ParameterList.extend` 
641      method will convert any :class:`~torch.Tensor` into :class:`~torch.nn.Parameter`. 
642   
643      Args: 
644          parameters (iterable, optional): an iterable of elements to add to the list. 
645   
646      Example:: 
647   
648          class MyModule(nn.Module): 
649              def __init__(self) -> None: 
650                  super().__init__() 
651                  self.params = nn.ParameterList( 
652                      [nn.Parameter(torch.randn(10, 10)) for i in range(10)] 
653                  ) 
654   
655              def forward(self, x): 
656                  # ParameterList can act as an iterable, or be indexed using ints 
657                  for i, p in enumerate(self.params): 
658                      x = self.params[i // 2].mm(x) + p.mm(x) 
659                  return x 
660      """
661  
662      def __init__(self, values: Optional[Iterable[Any]] = None) -> None:
663          super().__init__()
664          self._size = 0
665          if values is not None:
666              self += values
667  
668      def _get_abs_string_index(self, idx):
669          """Get the absolute index for the list of modules."""
670          idx = operator.index(idx)
671          if not (-len(self) <= idx < len(self)):
672              raise IndexError(f"index {idx} is out of range")
673          if idx < 0:
674              idx += len(self)
675          return str(idx)
676  
677      @overload
678      def __getitem__(self, idx: int) -> Any: ...
679  
680      @overload
681      def __getitem__(self: T, idx: slice) -> T: ...
682  
683      def __getitem__(self, idx):
684          if isinstance(idx, slice):
685              start, stop, step = idx.indices(len(self))
686              out = self.__class__()
687              for i in range(start, stop, step):
688                  out.append(self[i])
689              return out
690          else:
691              idx = self._get_abs_string_index(idx)
692              return getattr(self, str(idx))
693  
694      def __setitem__(self, idx: int, param: Any) -> None:
695          # Note that all other function that add an entry to the list part of
696          # the ParameterList end up here. So this is the only place where we need
697          # to wrap things into Parameter if needed.
698          # Objects added via setattr() are not in the list part and thus won't
699          # call into this function.
700          idx = self._get_abs_string_index(idx)
701          if isinstance(param, torch.Tensor) and not isinstance(param, Parameter):
702              param = Parameter(param)
703          return setattr(self, str(idx), param)

705      def __len__(self) -> int:
706          return self._size
707  
708      def __iter__(self) -> Iterator[Any]:
709          return iter(self[i] for i in range(len(self)))

711      def __iadd__(self, parameters: Iterable[Any]) -> Self:
712          return self.extend(parameters)

715          keys = super().__dir__()
716          keys = [key for key in keys if not key.isdigit()]
717          return keys

719      def append(self, value: Any) -> Self:
720          """Append a given value at the end of the list. 
721   
722          Args: 
723              value (Any): value to append 
724          """
725          new_idx = len(self)
726          self._size += 1
727          self[new_idx] = value
728          return self

730      def extend(self, values: Iterable[Any]) -> Self:
731          """Append values from a Python iterable to the end of the list. 
732   
733          Args: 
734              values (iterable): iterable of values to append 
735          """
736          # Tensor is an iterable but we never want to unpack it here
737          if not isinstance(values, container_abcs.Iterable) or isinstance(
738              values, torch.Tensor
739          ):
740              raise TypeError(
741                  "ParameterList.extend should be called with an "
742                  "iterable, but got " + type(values).__name__
743              )
744          for value in values:
745              self.append(value)
746          return self

751              if isinstance(p, torch.Tensor):
752                  size_str = "x".join(str(size) for size in p.size())
753                  if p.device.type in ["cuda", torch._C._get_privateuse1_backend_name()]:
754                      device_str = f" ({p.device})"
755                  else:
756                      device_str = ""
757                  parastr = "{} containing: [{} of size {}{}]".format(
758                      "Parameter" if isinstance(p, Parameter) else "Tensor",
759                      p.dtype,
760                      size_str,
761                      device_str,
762                  )
763                  child_lines.append("  (" + str(k) + "): " + parastr)
764              else:
765                  child_lines.append(
766                      "  (" + str(k) + "): Object of type: " + type(p).__name__

776  class ParameterDict(Module):
777      r"""Holds parameters in a dictionary. 
778   
779      ParameterDict can be indexed like a regular Python dictionary, but Parameters it 
780      contains are properly registered, and will be visible by all Module methods. 
781      Other objects are treated as would be done by a regular Python dictionary 
782   
783      :class:`~torch.nn.ParameterDict` is an **ordered** dictionary. 
784      :meth:`~torch.nn.ParameterDict.update` with other unordered mapping 
785      types (e.g., Python's plain ``dict``) does not preserve the order of the 
786      merged mapping. On the other hand, ``OrderedDict`` or another :class:`~torch.nn.ParameterDict` 
787      will preserve their ordering. 
788   
789      Note that the constructor, assigning an element of the dictionary and the 
790      :meth:`~torch.nn.ParameterDict.update` method will convert any :class:`~torch.Tensor` into 
791      :class:`~torch.nn.Parameter`. 
792   
793      Args: 
794          values (iterable, optional): a mapping (dictionary) of 
795              (string : Any) or an iterable of key-value pairs 
796              of type (string, Any) 
797   
798      Example:: 
799   
800          class MyModule(nn.Module): 
801              def __init__(self) -> None: 
802                  super().__init__() 
803                  self.params = nn.ParameterDict( 
804                      { 
805                          "left": nn.Parameter(torch.randn(5, 10)), 
806                          "right": nn.Parameter(torch.randn(5, 10)), 
807                      } 
808                  ) 
809   
810              def forward(self, x, choice): 
811                  x = self.params[choice].mm(x) 
812                  return x 
813      """
814  
815      def __init__(self, parameters: Any = None) -> None:
816          super().__init__()
817          self._keys: dict[str, None] = {}
818          if parameters is not None:
819              self.update(parameters)
820  
821      def _key_to_attr(self, key: str) -> str:
822          if not isinstance(key, str):
823              raise TypeError(
824                  "Index given to ParameterDict cannot be used as a key as it is "
825                  f"not a string (type is '{type(key).__name__}'). Open an issue on "
826                  "github if you need non-string keys."
827              )
828          else:
829              # Use the key as-is so that `.named_parameters()` returns the right thing
830              return key
831  
832      def __getitem__(self, key: str) -> Any:
833          attr = self._key_to_attr(key)
834          return getattr(self, attr)
835  
836      def __setitem__(self, key: str, value: Any) -> None:
837          # Note that all other function that add an entry to the dictionary part of
838          # the ParameterDict end up here. So this is the only place where we need
839          # to wrap things into Parameter if needed.
840          # Objects added via setattr() are not in the dictionary part and thus won't
841          # call into this function.
842          self._keys[key] = None
843          attr = self._key_to_attr(key)
844          if isinstance(value, torch.Tensor) and not isinstance(value, Parameter):
845              value = Parameter(value)
846          setattr(self, attr, value)
847  
848      def __delitem__(self, key: str) -> None:
849          del self._keys[key]
850          attr = self._key_to_attr(key)
851          delattr(self, attr)
852  
853      def __len__(self) -> int:
854          return len(self._keys)
855  
856      def __iter__(self) -> Iterator[str]:
857          return iter(self._keys)
858  
859      def __reversed__(self) -> Iterator[str]:
860          return reversed(self._keys)
861  
862      def copy(self) -> ParameterDict:
863          """Return a copy of this :class:`~torch.nn.ParameterDict` instance."""
864          # We have to use an OrderedDict because the ParameterDict constructor
865          # behaves differently on plain dict vs OrderedDict
866          return ParameterDict(OrderedDict((k, self[k]) for k in self._keys))
867  
868      def __contains__(self, key: str) -> bool:
869          return key in self._keys
870  
871      def setdefault(self, key: str, default: Optional[Any] = None) -> Any:
872          """Set the default for a key in the Parameterdict. 
873   
874          If key is in the ParameterDict, return its value. 
875          If not, insert `key` with a parameter `default` and return `default`. 
876          `default` defaults to `None`. 
877   
878          Args: 
879              key (str): key to set default for 
880              default (Any): the parameter set to the key 
881          """
882          if key not in self:
883              self[key] = default
884          return self[key]
885  
886      def clear(self) -> None:
887          """Remove all items from the ParameterDict."""
888          for k in self._keys.copy():
889              del self[k]
890  
891      def pop(self, key: str) -> Any:
892          r"""Remove key from the ParameterDict and return its parameter. 
893   
894          Args: 
895              key (str): key to pop from the ParameterDict 
896          """
897          v = self[key]
898          del self[key]
899          return v
900  
901      def popitem(self) -> tuple[str, Any]:
902          """Remove and return the last inserted `(key, parameter)` pair from the ParameterDict."""
903          k, _ = self._keys.popitem()
904          # We need the key in the _keys to be able to access/del
905          self._keys[k] = None
906          val = self[k]
907          del self[k]
908          return k, val
909  
910      def get(self, key: str, default: Optional[Any] = None) -> Any:
911          r"""Return the parameter associated with key if present. Otherwise return default if provided, None if not. 
912   
913          Args: 
914              key (str): key to get from the ParameterDict 
915              default (Parameter, optional): value to return if key not present 
916          """
917          return self[key] if key in self else default
918  
919      def fromkeys(
920          self, keys: Iterable[str], default: Optional[Any] = None
921      ) -> ParameterDict:
922          r"""Return a new ParameterDict with the keys provided. 
923   
924          Args: 
925              keys (iterable, string): keys to make the new ParameterDict from 
926              default (Parameter, optional): value to set for all keys 
927          """
928          return ParameterDict((k, default) for k in keys)
929  
930      def keys(self) -> container_abcs.KeysView[str]:
931          r"""Return an iterable of the ParameterDict keys."""
932          return self._keys.keys()
933  
934      def items(self) -> Iterable[tuple[str, Any]]:
935          r"""Return an iterable of the ParameterDict key/value pairs."""
936          return ((k, self[k]) for k in self._keys)
937  
938      def values(self) -> Iterable[Any]:
939          r"""Return an iterable of the ParameterDict values."""
940          return (self[k] for k in self._keys)

942      def update(self, parameters: Union[Mapping[str, Any], ParameterDict]) -> None:
943          r"""Update the :class:`~torch.nn.ParameterDict` with key-value pairs from ``parameters``, overwriting existing keys. 
944   
945          .. note:: 
946              If :attr:`parameters` is an ``OrderedDict``, a :class:`~torch.nn.ParameterDict`, or 
947              an iterable of key-value pairs, the order of new elements in it is preserved. 
948   
949          Args: 
950              parameters (iterable): a mapping (dictionary) from string to 
951                  :class:`~torch.nn.Parameter`, or an iterable of 
952                  key-value pairs of type (string, :class:`~torch.nn.Parameter`) 
953          """
954          if not isinstance(parameters, container_abcs.Iterable):
955              raise TypeError(
956                  "ParametersDict.update should be called with an "
957                  "iterable of key/value pairs, but got " + type(parameters).__name__
958              )
959  
960          if isinstance(parameters, (OrderedDict, ParameterDict)):
961              for key, parameter in parameters.items():
962                  self[key] = parameter
963          elif isinstance(parameters, container_abcs.Mapping):
964              for key, parameter in sorted(parameters.items()):
965                  self[key] = parameter
966          else:
967              for j, p in enumerate(parameters):
968                  if not isinstance(p, container_abcs.Iterable):
969                      raise TypeError(
970                          "ParameterDict update sequence element "
971                          "#" + str(j) + " should be Iterable; is" + type(p).__name__
972                      )
973                  if not len(p) == 2:
974                      raise ValueError(
975                          "ParameterDict update sequence element "
976                          "#" + str(j) + " has length " + str(len(p)) + "; 2 is required"
977                      )
978                  # parameters as length-2 list too cumbersome to type, see ModuleDict.update comment
979                  self[p[0]] = p[1]  # type: ignore[assignment]
980  
981      def extra_repr(self) -> str:
982          child_lines = []
983          for k, p in self.items():
984              if isinstance(p, torch.Tensor):
985                  size_str = "x".join(str(size) for size in p.size())
986                  if p.device.type in ["cuda", torch._C._get_privateuse1_backend_name()]:
987                      device_str = f" ({p.device})"
988                  else:
989                      device_str = ""
990                  parastr = "{} containing: [{} of size {}{}]".format(
991                      "Parameter" if isinstance(p, Parameter) else "Tensor",
992                      torch.typename(p),
993                      size_str,
994                      device_str,
995                  )
996                  child_lines.append("  (" + str(k) + "): " + parastr)
997              else:
998                  child_lines.append(
999                      "  (" + str(k) + "): Object of type: " + type(p).__name__
1000                 )
1001         tmpstr = "\n".join(child_lines)
1002         return tmpstr
