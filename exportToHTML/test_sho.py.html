1    from sympy.core import symbols, Rational, Function, diff

6    def test_sho_R_nl():
7        omega, r = symbols('omega r')
8        l = symbols('l', integer=True)
9        u = Function('u')
10   
11       # check that it obeys the Schrodinger equation
12       for n in range(5):
13           schreq = ( -diff(u(r), r, 2)/2 + ((l*(l + 1))/(2*r**2)
14                       + omega**2*r**2/2 - E_nl(n, l, omega))*u(r) )
15           result = schreq.subs(u(r), r*R_nl(n, l, omega/2, r))
16           assert simplify(result.doit()) == 0
