8    /** 
9     * The `VecMask` class provides a convenient interface for working with 
10    * vectorized masks in SIMD operations. It encapsulates a `Vectorized<T, N>` 
11    * mask that can be directly usable in masked vectorized operations. It provides 
12    * various methods for manipulating and accessing the mask elements: 
13    * 1. `from` and `to`: Conversion between a vector of boolean values and a 
14    * vectorized mask. 
15    * 2. `cast`: Casts the mask to a different base type. 
16    * 3. `all_zero`: Checks if all mask elements are zero. 
17    * 4. `is_masked`: Checks if a specific element is masked. 
18    * 5. `loadu`: Loads data from memory using the mask. 
19    * 6. `all_masked`: Checks if all mask elements are masked. 
20    * 
21    * Some helper template classes are provided to simplify the specialization of 
22    * the `VecMask` for the specific CPU arch: 
23    * 1. `VecMaskLoad`: Loads data from memory using the mask. 
24    * 2. `VecMaskTo`: Converts the mask to boolean. 
25    * 3. `VecMaskCast`: Casts the mask to a different base type. 
26    * 
27    */
28   template <typename T, int N>
29   class VecMask;
30   
31   template <
32       typename data_t,
33       int data_n,
34       typename mask_t,
35       int mask_n,
36       typename Enabled = void>
37   struct VecMaskLoad {
38     static inline VectorizedN<data_t, data_n> apply(
39         const data_t* ptr,
40         const VecMask<mask_t, mask_n>& vec_mask) {
41       constexpr typename VecMask<mask_t, mask_n>::size_type size =
42           VecMask<mask_t, mask_n>::size();
43       static_assert(VectorizedN<data_t, data_n>::size() >= size);
44       __at_align__ data_t data[size];
45       __at_align__ mask_t mask[size];
46       auto mask_ = VectorizedN<mask_t, mask_n>(vec_mask);
47       mask_.store(mask);
48       for (int i = 0; i < size; i++) {
49         data[i] = mask[i] ? ptr[i] : static_cast<data_t>(0);
50       }
51       return VectorizedN<data_t, data_n>::loadu(data, size);
52     }
53   };
54   
55   template <
56       typename dst_t,
57       int dst_n,
58       typename src_t,
59       int src_n,
60       typename Enabled = void>
61   struct VecMaskTo {
62     static inline VecMask<dst_t, dst_n> apply(
63         const VecMask<src_t, src_n>& vec_mask) {
64       auto zeros = VectorizedN<dst_t, dst_n>(static_cast<dst_t>(0));
65       auto ones = VectorizedN<dst_t, dst_n>(static_cast<dst_t>(1));
66       return VectorizedN<dst_t, dst_n>::blendv(
67           zeros, ones, vec_mask.template cast<dst_t, dst_n>());
68     }
69   };
70   
71   template <
72       typename dst_t,
73       int dst_n,
74       typename src_t,
75       int src_n,
76       typename Enabled = void>
77   struct VecMaskCast {
78     static inline VecMask<dst_t, dst_n> apply(
79         const VecMask<src_t, src_n>& vec_mask) {
80       return VecMask<dst_t, dst_n>::from(VectorizedN<src_t, src_n>(vec_mask));
81     }
82   };
83   
84   template <typename T, int N>
85   struct VecMaskCast<T, N, T, N> {
86     static inline VecMask<T, N> apply(const VecMask<T, N>& vec_mask) {
87       return vec_mask;
88     }
89   };
90   
91   template <typename T, int N>
92   struct VecMaskCheck {
93     static inline bool all_zero(const VectorizedN<T, N>& vec_mask) {
94       __at_align__ T mask[VectorizedN<T, N>::size()];
95       vec_mask.store(mask);
96       return std::all_of(mask, mask + VectorizedN<T, N>::size(), [](T m) {
97         return m == static_cast<T>(0);
98       });
99     }
100  
101    static inline bool all_masked(const VectorizedN<T, N>& vec_mask) {
102      __at_align__ T mask[VectorizedN<T, N>::size()];
103      vec_mask.store(mask);
104      return std::all_of(mask, mask + VectorizedN<T, N>::size(), [](T m) {
105        return m != static_cast<T>(0);
106      });
107    }
108  
109    static inline bool is_masked(const VectorizedN<T, N>& vec_mask, int i) {
110      __at_align__ T mask[VectorizedN<T, N>::size()];
111      vec_mask.store(mask);
112      return mask[i] != static_cast<T>(0);
113    }
114  };
115  
116  template <typename T, int N>
117  class VecMask {
118   public:
119    using size_type = int;
120    static constexpr size_type size() {
121      return VectorizedN<T, N>::size();
122    }
123  
124   private:
125    VectorizedN<T, N> mask_;
126  
127   public:
128    VecMask() : mask_(static_cast<T>(0)) {}
129    VecMask(const VectorizedN<T, N>& mask) : mask_(mask) {}
130  
131    template <int L = N, typename std::enable_if_t<L == 1, int> = 0>
132    VecMask(const Vectorized<T>& mask) : mask_(mask) {}
133  
134    template <typename U, int L>
135    static VecMask<T, N> from(const VectorizedN<U, L>& b_vec) {
136      __at_align__ U b_buf[size()];
137      if constexpr (size() >= VectorizedN<U, L>::size()) {
138        b_vec.store(b_buf);
139        for (int i = VectorizedN<U, L>::size(); i < size(); i++) {
140          b_buf[i] = static_cast<U>(0);
141        }
142      } else {
143        b_vec.store(b_buf, size());
144      }
145      return from(b_buf);
146    }
147  
148    template <typename U>
149    static VecMask<T, N> from(U b) {
150      using int_t = int_same_size_t<T>;
151      T mask = b ? c10::bit_cast<T>((int_t)(~(int_t)0)) : (T)0;
152      return VectorizedN<T, N>(mask);
153    }
154  
155    template <typename U>
156    static VecMask<T, N> from(U* b) {
157      using int_t = int_same_size_t<T>;
158      __at_align__ T mask[size()];
159  #ifndef __msvc_cl__
160  #pragma unroll
161  #endif
162      for (int i = 0; i < size(); i++) {
163        *(int_t*)(mask + i) = b[i] ? ~(int_t)0 : (int_t)0;
164      }
165      return VectorizedN<T, N>(VectorizedN<T, N>::loadu(mask));
166    }
167  
168    static VecMask<T, N> blendv(
169        const VecMask<T, N>& c,
170        const VecMask<T, N>& b,
171        const VecMask<T, N>& a) {
172      VectorizedN<T, N> result = VectorizedN<T, N>::blendv(
173          VectorizedN<T, N>(c), VectorizedN<T, N>(b), VectorizedN<T, N>(a));
174      return result;
175    }
176  
177    static VecMask<T, N> set(
178        const VecMask<T, N>& a,
179        const VecMask<T, N>& b,
180        int64_t count = size()) {
181      VectorizedN<T, N> result = VectorizedN<T, N>::set(
182          VectorizedN<T, N>(a), VectorizedN<T, N>(b), count);
183      return result;
184    }
185  
186    void store(bool* b, int count = size()) {
187      constexpr int L =
188          (VectorizedN<T, N>::size() + Vectorized<bool>::size() - 1) /
189          Vectorized<bool>::size();
190      auto res = this->to<bool, L>();
191      res.store(b, count);
192      return;
193    }
194  
195    template <typename U, int L, std::enable_if_t<L >= 2, int> = 0>
196    inline VectorizedN<U, L> to() const {
197      return VecMaskTo<U, L, T, N>::apply(*this);
198    }
199  
200    template <typename U, int L, std::enable_if_t<L == 1, int> = 0>
201    inline Vectorized<U> to() const {
202      return VecMaskTo<U, L, T, N>::apply(*this);
203    }
204  
205    template <typename U, int L>
206    inline VecMask<U, L> cast() const {
207      return VecMaskCast<U, L, T, N>::apply(*this);
208    }
209  
210    inline bool all_zero() const {
211      return VecMaskCheck<T, N>::all_zero(mask_);
212    }
213  
214    inline bool all_masked() const {
215      return VecMaskCheck<T, N>::all_masked(mask_);
216    }
217  
218    inline bool is_masked(int i) const {
219      return VecMaskCheck<T, N>::is_masked(mask_, i);
220    }
221  
222    inline operator VectorizedN<T, N>() const {
223      return mask_;
224    }
225  
226    template <int L = N, typename std::enable_if_t<L == 1, int> = 0>
227    inline operator Vectorized<T>() const {
228      return mask_[0];
229    }
230  
231    inline Vectorized<T> operator[](int i) const {
232      return mask_[i];
233    }
234  
235    template <
236        typename U,
237        int L,
238        std::enable_if_t<L >= 2 && VectorizedN<U, L>::size() >= size(), int> = 0>
239    VectorizedN<U, L> loadu(const U* ptr) const {
240      return VecMaskLoad<U, L, T, N>::apply(ptr, *this);
241    }
242  
243    template <
244        typename U,
245        int L,
246        std::enable_if_t<L == 1 && Vectorized<U>::size() >= size(), int> = 0>
247    Vectorized<U> loadu(const U* ptr) const {
248      return VecMaskLoad<U, L, T, N>::apply(ptr, *this);
249    }
250  };
