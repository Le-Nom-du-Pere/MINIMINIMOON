7    def _create_rolling_disc():
8        # Define symbols and coordinates
9        t = dynamicsymbols._t
10       q1, q2, q3, q4, q5, u1, u2, u3, u4, u5 = dynamicsymbols('q1:6 u1:6')
11       g, r, m = symbols('g r m')
12       # Define bodies and frames
13       ground = RigidBody('ground')
14       disc = RigidBody('disk', mass=m)
15       disc.inertia = (m * r ** 2 / 4 * inertia(disc.frame, 1, 2, 1),
16                       disc.masscenter)
17       ground.masscenter.set_vel(ground.frame, 0)
18       disc.masscenter.set_vel(disc.frame, 0)
19       int_frame = ReferenceFrame('int_frame')
20       # Orient frames
21       int_frame.orient_body_fixed(ground.frame, (q1, q2, 0), 'zxy')
22       disc.frame.orient_axis(int_frame, int_frame.y, q3)
23       g_w_d = disc.frame.ang_vel_in(ground.frame)
24       disc.frame.set_ang_vel(ground.frame,
25                              u1 * disc.x + u2 * disc.y + u3 * disc.z)
26       # Define points
27       cp = ground.masscenter.locatenew('contact_point',
28                                        q4 * ground.x + q5 * ground.y)
29       cp.set_vel(ground.frame, u4 * ground.x + u5 * ground.y)
30       disc.masscenter.set_pos(cp, r * int_frame.z)
31       disc.masscenter.set_vel(ground.frame, cross(
32           disc.frame.ang_vel_in(ground.frame), disc.masscenter.pos_from(cp)))
33       # Define kinematic differential equations
34       kdes = [g_w_d.dot(disc.x) - u1, g_w_d.dot(disc.y) - u2,
35               g_w_d.dot(disc.z) - u3, q4.diff(t) - u4, q5.diff(t) - u5]
36       # Define nonholonomic constraints
37       v0 = cp.vel(ground.frame) + cross(
38           disc.frame.ang_vel_in(int_frame), cp.pos_from(disc.masscenter))
39       fnh = [v0.dot(ground.x), v0.dot(ground.y)]
40       # Define loads
41       loads = [(disc.masscenter, -disc.mass * g * ground.z)]
42       bodies = [disc]
43       return {
44           'frame': ground.frame,
45           'q_ind': [q1, q2, q3, q4, q5],
46           'u_ind': [u1, u2, u3],
47           'u_dep': [u4, u5],
48           'kdes': kdes,
49           'fnh': fnh,
50           'bodies': bodies,
51           'loads': loads
52       }

55   def _verify_rolling_disc_numerically(kane, all_zero=False):
56       q, u, p = dynamicsymbols('q1:6'), dynamicsymbols('u1:6'), symbols('g r m')
57       eval_sys = lambdify((q, u, p), (kane.mass_matrix_full, kane.forcing_full),
58                           cse=True)
59       solve_sys = lambda q, u, p: Matrix.LUsolve(
60           *(Matrix(mat) for mat in eval_sys(q, u, p)))
61       solve_u_dep = lambdify((q, u[:3], p), kane._Ars * Matrix(u[:3]), cse=True)
62       eps = 1e-10
63       p_vals = (9.81, 0.26, 3.43)
64       # First numeric test
65       q_vals = (0.3, 0.1, 1.97, -0.35, 2.27)
66       u_vals = [-0.2, 1.3, 0.15]
67       u_vals.extend(solve_u_dep(q_vals, u_vals, p_vals)[:2, 0])
68       expected = Matrix([
69           0.126603940595934, 0.215942571601660, 1.28736069604936,
70           0.319764288376543, 0.0989146857254898, -0.925848952664489,
71           -0.0181350656532944, 2.91695398184589, -0.00992793421754526,
72           0.0412861634829171])
73       assert all(abs(x) < eps for x in
74                  (solve_sys(q_vals, u_vals, p_vals) - expected))
75       # Second numeric test
76       q_vals = (3.97, -0.28, 8.2, -0.35, 2.27)
77       u_vals = [-0.25, -2.2, 0.62]
78       u_vals.extend(solve_u_dep(q_vals, u_vals, p_vals)[:2, 0])
79       expected = Matrix([
80           0.0259159090798597, 0.668041660387416, -2.19283799213811,
81           0.385441810852219, 0.420109283790573, 1.45030568179066,
82           -0.0110924422400793, -8.35617840186040, -0.154098542632173,
83           -0.146102664410010])
84       assert all(abs(x) < eps for x in
85                  (solve_sys(q_vals, u_vals, p_vals) - expected))
86       if all_zero:
87           q_vals = (0, 0, 0, 0, 0)
88           u_vals = (0, 0, 0, 0, 0)
89           assert solve_sys(q_vals, u_vals, p_vals) == zeros(10, 1)

92   def test_kane_rolling_disc_lu():
93       props = _create_rolling_disc()
94       kane = KanesMethod(props['frame'], props['q_ind'], props['u_ind'],
95                          props['kdes'], u_dependent=props['u_dep'],
96                          velocity_constraints=props['fnh'],
97                          bodies=props['bodies'], forcelist=props['loads'],
98                          explicit_kinematics=False, constraint_solver='LU')
99       kane.kanes_equations()
100      _verify_rolling_disc_numerically(kane)

103  def test_kane_rolling_disc_kdes_callable():
104      props = _create_rolling_disc()
105      kane = KanesMethod(
106          props['frame'], props['q_ind'], props['u_ind'], props['kdes'],
107          u_dependent=props['u_dep'], velocity_constraints=props['fnh'],
108          bodies=props['bodies'], forcelist=props['loads'],
109          explicit_kinematics=False,
110          kd_eqs_solver=lambda A, b: simplify(A.LUsolve(b)))
111      q, u, p = dynamicsymbols('q1:6'), dynamicsymbols('u1:6'), symbols('g r m')
112      qd = dynamicsymbols('q1:6', 1)
113      eval_kdes = lambdify((q, qd, u, p), tuple(kane.kindiffdict().items()))
114      eps = 1e-10
115      # Test with only zeros. If 'LU' would be used this would result in nan.
116      p_vals = (9.81, 0.25, 3.5)
117      zero_vals = (0, 0, 0, 0, 0)
118      assert all(abs(qdi - fui) < eps for qdi, fui in
119                 eval_kdes(zero_vals, zero_vals, zero_vals, p_vals))
120      # Test with some arbitrary values
121      q_vals = tuple(map(float, (pi / 6, pi / 3, pi / 2, 0.42, 0.62)))
122      qd_vals = tuple(map(float, (4, 1 / 3, 4 - 2 * sqrt(3),
123                                  0.25 * (2 * sqrt(3) - 3),
124                                  0.25 * (2 - sqrt(3)))))
125      u_vals = tuple(map(float, (-2, 4, 1 / 3, 0.25 * (-3 + 2 * sqrt(3)),
126                                 0.25 * (-sqrt(3) + 2))))
127      assert all(abs(qdi - fui) < eps for qdi, fui in
128                 eval_kdes(q_vals, qd_vals, u_vals, p_vals))
