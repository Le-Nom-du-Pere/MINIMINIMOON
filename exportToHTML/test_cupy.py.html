14   def test_cupy_print():
15       prntr = CuPyPrinter()
16       assert prntr.doprint(logaddexp(a, b)) == 'cupy.logaddexp(a, b)'
17       assert prntr.doprint(sqrt(x)) == 'cupy.sqrt(x)'
18       assert prntr.doprint(log(x)) == 'cupy.log(x)'
19       assert prntr.doprint("acos(x)") == 'cupy.arccos(x)'
20       assert prntr.doprint("exp(x)") == 'cupy.exp(x)'
21       assert prntr.doprint("Abs(x)") == 'abs(x)'
22   
23   def test_not_cupy_print():
24       prntr = CuPyPrinter()

28   def test_cupy_sum():
29       if not cp:
30           skip("CuPy not installed")
31   
32       s = Sum(x ** i, (i, a, b))
33       f = lambdify((a, b, x), s, 'cupy')
34   
35       a_, b_ = 0, 10
36       x_ = cp.linspace(-1, +1, 10)
37       assert cp.allclose(f(a_, b_, x_), sum(x_ ** i_ for i_ in range(a_, b_ + 1)))
38   
39       s = Sum(i * x, (i, a, b))
40       f = lambdify((a, b, x), s, 'numpy')
41   
42       a_, b_ = 0, 10
43       x_ = cp.linspace(-1, +1, 10)
44       assert cp.allclose(f(a_, b_, x_), sum(i_ * x_ for i_ in range(a_, b_ + 1)))
