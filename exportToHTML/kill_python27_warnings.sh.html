kill_python27_warnings.sh 
1    #!/bin/bash
2    
3    # Eliminar configuraciones de inspección Python 2.7
4    find .idea -name "*.xml" -exec sed -i '' '/<inspection_tool class="Python.*Compatibility"/,/<\/inspection_tool>/d' {} \; 2>/dev/null
5    find .idea -name "*.xml" -exec sed -i '' '/Python.*2\.7.*compatibility/d' {} \; 2>/dev/null
6    find .idea -name "*.xml" -exec sed -i '' '/languageLevel.*2\.7/d' {} \; 2>/dev/null
7    
8    # Forzar Python 3.11 en misc.xml
9    if [ -f ".idea/misc.xml" ]; then
10     sed -i '' 's/languageLevel="[^"]*"/languageLevel="JDK_11"/g' .idea/misc.xml
11   fi
12   
13   # Eliminar perfiles de inspección completos si existen
14   rm -rf .idea/inspectionProfiles/ 2>/dev/null
15   
16   echo "Eliminadas todas las inspecciones Python 2.7"
17
