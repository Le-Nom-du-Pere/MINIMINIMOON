13   namespace detail {

15   // Does not delete anything
16   C10_API void deleteNothing(void*);
17   
18   // A detail::UniqueVoidPtr is an owning smart pointer like unique_ptr, but
19   // with three major differences:
20   //
21   //    1) It is specialized to void
22   //
23   //    2) It is specialized for a function pointer deleter
24   //       void(void* ctx); i.e., the deleter doesn't take a
25   //       reference to the data, just to a context pointer
26   //       (erased as void*).  In fact, internally, this pointer
27   //       is implemented as having an owning reference to
28   //       context, and a non-owning reference to data; this is why
29   //       you release_context(), not release() (the conventional
30   //       API for release() wouldn't give you enough information
31   //       to properly dispose of the object later.)
32   //
33   //    3) The deleter is guaranteed to be called when the unique
34   //       pointer is destructed and the context is non-null; this is different
35   //       from std::unique_ptr where the deleter is not called if the
36   //       data pointer is null.
37   //
38   // Some of the methods have slightly different types than std::unique_ptr
39   // to reflect this.
40   //
41   class UniqueVoidPtr {
42    private:
43     // Lifetime tied to ctx_
44     void* data_;
45     std::unique_ptr<void, DeleterFnPtr> ctx_;
46   
47    public:
48     UniqueVoidPtr() : data_(nullptr), ctx_(nullptr, &deleteNothing) {}
49     explicit UniqueVoidPtr(void* data)
50         : data_(data), ctx_(nullptr, &deleteNothing) {}
51     UniqueVoidPtr(void* data, void* ctx, DeleterFnPtr ctx_deleter)
52         : data_(data), ctx_(ctx, ctx_deleter ? ctx_deleter : &deleteNothing) {}
53     void* operator->() const {
54       return data_;
55     }
56     void clear() {
57       ctx_ = nullptr;
58       data_ = nullptr;
59     }
60     void* get() const {
61       return data_;
62     }
63   
64     bool /* success */ unsafe_reset_data_and_ctx(void* new_data_and_ctx) {
65       if (C10_UNLIKELY(ctx_.get_deleter() != &deleteNothing)) {
66         return false;
67       }
68       // seems quicker than calling the no-op deleter when we reset
69       // NOLINTNEXTLINE(bugprone-unused-return-value)
70       ctx_.release();
71       ctx_.reset(new_data_and_ctx);
72       data_ = new_data_and_ctx;
73       return true;
74     }
75   
76     void* get_context() const {
77       return ctx_.get();
78     }
79     void* release_context() {
80       return ctx_.release();
81     }
82     std::unique_ptr<void, DeleterFnPtr>&& move_context() {
83       return std::move(ctx_);
84     }
85     [[nodiscard]] bool compare_exchange_deleter(
86         DeleterFnPtr expected_deleter,
87         DeleterFnPtr new_deleter) {
88       if (get_deleter() != expected_deleter)
89         return false;
90       ctx_ = std::unique_ptr<void, DeleterFnPtr>(ctx_.release(), new_deleter);
91       return true;
92     }
93   
94     template <typename T>
95     T* cast_context(DeleterFnPtr expected_deleter) const {
96       if (get_deleter() != expected_deleter)
97         return nullptr;
98       return static_cast<T*>(get_context());
99     }
100    operator bool() const {
101      return data_ || ctx_;
102    }
103    DeleterFnPtr get_deleter() const {
104      return ctx_.get_deleter();
105    }
106  };

108  // Note [How UniqueVoidPtr is implemented]
109  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
110  // UniqueVoidPtr solves a common problem for allocators of tensor data, which
111  // is that the data pointer (e.g., float*) which you are interested in, is not
112  // the same as the context pointer (e.g., DLManagedTensor) which you need
113  // to actually deallocate the data.  Under a conventional deleter design, you
114  // have to store extra context in the deleter itself so that you can actually
115  // delete the right thing.  Implementing this with standard C++ is somewhat
116  // error-prone: if you use a std::unique_ptr to manage tensors, the deleter will
117  // not be called if the data pointer is nullptr, which can cause a leak if the
118  // context pointer is non-null (and the deleter is responsible for freeing both
119  // the data pointer and the context pointer).
120  //
121  // So, in our reimplementation of unique_ptr, which just store the context
122  // directly in the unique pointer, and attach the deleter to the context
123  // pointer itself.  In simple cases, the context pointer is just the pointer
124  // itself.
125  
126  inline bool operator==(const UniqueVoidPtr& sp, std::nullptr_t) noexcept {
127    return !sp;
128  }
129  inline bool operator==(std::nullptr_t, const UniqueVoidPtr& sp) noexcept {
130    return !sp;
131  }
132  inline bool operator!=(const UniqueVoidPtr& sp, std::nullptr_t) noexcept {
133    return sp;
134  }
135  inline bool operator!=(std::nullptr_t, const UniqueVoidPtr& sp) noexcept {
136    return sp;
137  }
