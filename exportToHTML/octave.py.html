123      def _get_loop_opening_ending(self, indices):
124          open_lines = []
125          close_lines = []
126          for i in indices:
127              # Octave arrays start at 1 and end at dimension
128              var, start, stop = map(self._print,
129                      [i.label, i.lower + 1, i.upper + 1])
130              open_lines.append("for %s = %s:%s" % (var, start, stop))
131              close_lines.append("end")
132          return open_lines, close_lines

135      def _print_Mul(self, expr):
136          # print complex numbers nicely in Octave
137          if (expr.is_number and expr.is_imaginary and
138                  (S.ImaginaryUnit*expr).is_Integer):
139              return "%si" % self._print(-S.ImaginaryUnit*expr)
140  
141          # cribbed from str.py
142          prec = precedence(expr)
143  
144          c, e = expr.as_coeff_Mul()
145          if c < 0:
146              expr = _keep_coeff(-c, e)
147              sign = "-"
148          else:
149              sign = ""
150  
151          a = []  # items in the numerator
152          b = []  # items that are in the denominator (if any)
153  
154          pow_paren = []  # Will collect all pow with more than one base element and exp = -1
155  
156          if self.order not in ('old', 'none'):
157              args = expr.as_ordered_factors()
158          else:
159              # use make_args in case expr was something like -x -> x
160              args = Mul.make_args(expr)
161  
162          # Gather args for numerator/denominator
163          for item in args:
164              if (item.is_commutative and item.is_Pow and item.exp.is_Rational
165                      and item.exp.is_negative):
166                  if item.exp != -1:
167                      b.append(Pow(item.base, -item.exp, evaluate=False))
168                  else:
169                      if len(item.args[0].args) != 1 and isinstance(item.base, Mul):   # To avoid situations like #14160
170                          pow_paren.append(item)
171                      b.append(Pow(item.base, -item.exp))
172              elif item.is_Rational and item is not S.Infinity:
173                  if item.p != 1:
174                      a.append(Rational(item.p))
175                  if item.q != 1:
176                      b.append(Rational(item.q))
177              else:
178                  a.append(item)
179  
180          a = a or [S.One]
181  
182          a_str = [self.parenthesize(x, prec) for x in a]
183          b_str = [self.parenthesize(x, prec) for x in b]
184  
185          # To parenthesize Pow with exp = -1 and having more than one Symbol
186          for item in pow_paren:
187              if item.base in b:
188                  b_str[b.index(item.base)] = "(%s)" % b_str[b.index(item.base)]
189  
190          # from here it differs from str.py to deal with "*" and ".*"
191          def multjoin(a, a_str):
192              # here we probably are assuming the constants will come first
193              r = a_str[0]
194              for i in range(1, len(a)):
195                  mulsym = '*' if a[i-1].is_number else '.*'
196                  r = r + mulsym + a_str[i]
197              return r
198  
199          if not b:
200              return sign + multjoin(a, a_str)
201          elif len(b) == 1:
202              divsym = '/' if b[0].is_number else './'
203              return sign + multjoin(a, a_str) + divsym + b_str[0]
204          else:
205              divsym = '/' if all(bi.is_number for bi in b) else './'
206              return (sign + multjoin(a, a_str) +
207                      divsym + "(%s)" % multjoin(b, b_str))

215      def _print_Pow(self, expr):
216          powsymbol = '^' if all(x.is_number for x in expr.args) else '.^'
217  
218          PREC = precedence(expr)
219  
220          if equal_valued(expr.exp, 0.5):
221              return "sqrt(%s)" % self._print(expr.base)
222  
223          if expr.is_commutative:
224              if equal_valued(expr.exp, -0.5):
225                  sym = '/' if expr.base.is_number else './'
226                  return "1" + sym + "sqrt(%s)" % self._print(expr.base)
227              if equal_valued(expr.exp, -1):
228                  sym = '/' if expr.base.is_number else './'
229                  return "1" + sym + "%s" % self.parenthesize(expr.base, PREC)
230  
231          return '%s%s%s' % (self.parenthesize(expr.base, PREC), powsymbol,
232                             self.parenthesize(expr.exp, PREC))

263      def _print_Assignment(self, expr):
264          from sympy.codegen.ast import Assignment
265          from sympy.functions.elementary.piecewise import Piecewise
266          from sympy.tensor.indexed import IndexedBase
267          # Copied from codeprinter, but remove special MatrixSymbol treatment
268          lhs = expr.lhs
269          rhs = expr.rhs
270          # We special case assignments that take multiple lines
271          if not self._settings["inline"] and isinstance(expr.rhs, Piecewise):
272              # Here we modify Piecewise so each expression is now
273              # an Assignment, and then continue on the print.
274              expressions = []
275              conditions = []
276              for (e, c) in rhs.args:
277                  expressions.append(Assignment(lhs, e))
278                  conditions.append(c)
279              temp = Piecewise(*zip(expressions, conditions))
280              return self._print(temp)
281          if self._settings["contract"] and (lhs.has(IndexedBase) or
282                  rhs.has(IndexedBase)):
283              # Here we check if there is looping to be done, and if so
284              # print the required loops.
285              return self._doprint_loops(rhs, lhs)
286          else:
287              lhs_code = self._print(lhs)
288              rhs_code = self._print(rhs)
289              return self._get_statement("%s = %s" % (lhs_code, rhs_code))

327      def _print_MatrixBase(self, A):
328          # Handle zero dimensions:
329          if (A.rows, A.cols) == (0, 0):
330              return '[]'
331          elif S.Zero in A.shape:
332              return 'zeros(%s, %s)' % (A.rows, A.cols)
333          elif (A.rows, A.cols) == (1, 1):
334              # Octave does not distinguish between scalars and 1x1 matrices
335              return self._print(A[0, 0])
336          return "[%s]" % "; ".join(" ".join([self._print(a) for a in A[r, :]])
337                                    for r in range(A.rows))

356      def _print_MatrixSlice(self, expr):
357          def strslice(x, lim):
358              l = x[0] + 1
359              h = x[1]
360              step = x[2]
361              lstr = self._print(l)
362              hstr = 'end' if h == lim else self._print(h)
363              if step == 1:
364                  if l == 1 and h == lim:
365                      return ':'
366                  if l == h:
367                      return lstr
368                  else:
369                      return lstr + ':' + hstr
370              else:
371                  return ':'.join((lstr, self._print(step), hstr))
372          return (self._print(expr.parent) + '(' +
373                  strslice(expr.rowslice, expr.parent.shape[0]) + ', ' +
374                  strslice(expr.colslice, expr.parent.shape[1]) + ')')

490      def _print_Piecewise(self, expr):
491          if expr.args[-1].cond != True:
492              # We need the last conditional to be a True, otherwise the resulting
493              # function may not return a result.
494              raise ValueError("All Piecewise expressions must contain an "
495                               "(expr, True) statement to be used as a default "
496                               "condition. Without one, the generated "
497                               "expression may not evaluate to anything under "
498                               "some condition.")
499          lines = []
500          if self._settings["inline"]:
501              # Express each (cond, expr) pair in a nested Horner form:
502              #   (condition) .* (expr) + (not cond) .* (<others>)
503              # Expressions that result in multiple statements won't work here.
504              ecpairs = ["({0}).*({1}) + (~({0})).*(".format
505                         (self._print(c), self._print(e))
506                         for e, c in expr.args[:-1]]
507              elast = "%s" % self._print(expr.args[-1].expr)
508              pw = " ...\n".join(ecpairs) + elast + ")"*len(ecpairs)
509              # Note: current need these outer brackets for 2*pw.  Would be
510              # nicer to teach parenthesize() to do this for us when needed!
511              return "(" + pw + ")"
512          else:
513              for i, (e, c) in enumerate(expr.args):
514                  if i == 0:
515                      lines.append("if (%s)" % self._print(c))
516                  elif i == len(expr.args) - 1 and c == True:
517                      lines.append("else")
518                  else:
519                      lines.append("elseif (%s)" % self._print(c))
520                  code0 = self._print(e)
521                  lines.append(code0)
522                  if i == len(expr.args) - 1:
523                      lines.append("end")
524              return "\n".join(lines)

535      def indent_code(self, code):
536          """Accepts a string of code or a list of code lines"""
537  
538          # code mostly copied from ccode
539          if isinstance(code, str):
540              code_lines = self.indent_code(code.splitlines(True))
541              return ''.join(code_lines)
542  
543          tab = "  "
544          inc_regex = ('^function ', '^if ', '^elseif ', '^else$', '^for ')
545          dec_regex = ('^end$', '^elseif ', '^else$')
546  
547          # pre-strip left-space from the code
548          code = [ line.lstrip(' \t') for line in code ]
549  
550          increase = [ int(any(search(re, line) for re in inc_regex))
551                       for line in code ]
552          decrease = [ int(any(search(re, line) for re in dec_regex))
553                       for line in code ]
554  
555          pretty = []
556          level = 0
557          for n, line in enumerate(code):
558              if line in ('', '\n'):
559                  pretty.append(line)
560                  continue
561              level -= decrease[n]
562              pretty.append("%s%s" % (tab*level, line))
563              level += increase[n]
564          return pretty

567  def octave_code(expr, assign_to=None, **settings):
568      r"""Converts `expr` to a string of Octave (or Matlab) code. 
569   
570      The string uses a subset of the Octave language for Matlab compatibility. 
571   
572      Parameters 
573      ========== 
574   
575      expr : Expr 
576          A SymPy expression to be converted. 
577      assign_to : optional 
578          When given, the argument is used as the name of the variable to which 
579          the expression is assigned.  Can be a string, ``Symbol``, 
580          ``MatrixSymbol``, or ``Indexed`` type.  This can be helpful for 
581          expressions that generate multi-line statements. 
582      precision : integer, optional 
583          The precision for numbers such as pi  [default=16]. 
584      user_functions : dict, optional 
585          A dictionary where keys are ``FunctionClass`` instances and values are 
586          their string representations.  Alternatively, the dictionary value can 
587          be a list of tuples i.e. [(argument_test, cfunction_string)].  See 
588          below for examples. 
589      human : bool, optional 
590          If True, the result is a single string that may contain some constant 
591          declarations for the number symbols.  If False, the same information is 
592          returned in a tuple of (symbols_to_declare, not_supported_functions, 
593          code_text).  [default=True]. 
594      contract: bool, optional 
595          If True, ``Indexed`` instances are assumed to obey tensor contraction 
596          rules and the corresponding nested loops over indices are generated. 
597          Setting contract=False will not generate loops, instead the user is 
598          responsible to provide values for the indices in the code. 
599          [default=True]. 
600      inline: bool, optional 
601          If True, we try to create single-statement code instead of multiple 
602          statements.  [default=True]. 
603   
604      Examples 
605      ======== 
606   
607      >>> from sympy import octave_code, symbols, sin, pi 
608      >>> x = symbols('x') 
609      >>> octave_code(sin(x).series(x).removeO()) 
610      'x.^5/120 - x.^3/6 + x' 
611   
612      >>> from sympy import Rational, ceiling 
613      >>> x, y, tau = symbols("x, y, tau") 
614      >>> octave_code((2*tau)**Rational(7, 2)) 
615      '8*sqrt(2)*tau.^(7/2)' 
616   
617      Note that element-wise (Hadamard) operations are used by default between 
618      symbols.  This is because its very common in Octave to write "vectorized" 
619      code.  It is harmless if the values are scalars. 
620   
621      >>> octave_code(sin(pi*x*y), assign_to="s") 
622      's = sin(pi*x.*y);' 
623   
624      If you need a matrix product "*" or matrix power "^", you can specify the 
625      symbol as a ``MatrixSymbol``. 
626   
627      >>> from sympy import Symbol, MatrixSymbol 
628      >>> n = Symbol('n', integer=True, positive=True) 
629      >>> A = MatrixSymbol('A', n, n) 
630      >>> octave_code(3*pi*A**3) 
631      '(3*pi)*A^3' 
632   
633      This class uses several rules to decide which symbol to use a product. 
634      Pure numbers use "*", Symbols use ".*" and MatrixSymbols use "*". 
635      A HadamardProduct can be used to specify componentwise multiplication ".*" 
636      of two MatrixSymbols.  There is currently there is no easy way to specify 
637      scalar symbols, so sometimes the code might have some minor cosmetic 
638      issues.  For example, suppose x and y are scalars and A is a Matrix, then 
639      while a human programmer might write "(x^2*y)*A^3", we generate: 
640   
641      >>> octave_code(x**2*y*A**3) 
642      '(x.^2.*y)*A^3' 
643   
644      Matrices are supported using Octave inline notation.  When using 
645      ``assign_to`` with matrices, the name can be specified either as a string 
646      or as a ``MatrixSymbol``.  The dimensions must align in the latter case. 
647   
648      >>> from sympy import Matrix, MatrixSymbol 
649      >>> mat = Matrix([[x**2, sin(x), ceiling(x)]]) 
650      >>> octave_code(mat, assign_to='A') 
651      'A = [x.^2 sin(x) ceil(x)];' 
652   
653      ``Piecewise`` expressions are implemented with logical masking by default. 
654      Alternatively, you can pass "inline=False" to use if-else conditionals. 
655      Note that if the ``Piecewise`` lacks a default term, represented by 
656      ``(expr, True)`` then an error will be thrown.  This is to prevent 
657      generating an expression that may not evaluate to anything. 
658   
659      >>> from sympy import Piecewise 
660      >>> pw = Piecewise((x + 1, x > 0), (x, True)) 
661      >>> octave_code(pw, assign_to=tau) 
662      'tau = ((x > 0).*(x + 1) + (~(x > 0)).*(x));' 
663   
664      Note that any expression that can be generated normally can also exist 
665      inside a Matrix: 
666   
667      >>> mat = Matrix([[x**2, pw, sin(x)]]) 
668      >>> octave_code(mat, assign_to='A') 
669      'A = [x.^2 ((x > 0).*(x + 1) + (~(x > 0)).*(x)) sin(x)];' 
670   
671      Custom printing can be defined for certain types by passing a dictionary of 
672      "type" : "function" to the ``user_functions`` kwarg.  Alternatively, the 
673      dictionary value can be a list of tuples i.e., [(argument_test, 
674      cfunction_string)].  This can be used to call a custom Octave function. 
675   
676      >>> from sympy import Function 
677      >>> f = Function('f') 
678      >>> g = Function('g') 
679      >>> custom_functions = { 
680      ...   "f": "existing_octave_fcn", 
681      ...   "g": [(lambda x: x.is_Matrix, "my_mat_fcn"), 
682      ...         (lambda x: not x.is_Matrix, "my_fcn")] 
683      ... } 
684      >>> mat = Matrix([[1, x]]) 
685      >>> octave_code(f(x) + g(x) + g(mat), user_functions=custom_functions) 
686      'existing_octave_fcn(x) + my_fcn(x) + my_mat_fcn([1 x])' 
687   
688      Support for loops is provided through ``Indexed`` types. With 
689      ``contract=True`` these expressions will be turned into loops, whereas 
690      ``contract=False`` will just print the assignment expression that should be 
691      looped over: 
692   
693      >>> from sympy import Eq, IndexedBase, Idx 
694      >>> len_y = 5 
695      >>> y = IndexedBase('y', shape=(len_y,)) 
696      >>> t = IndexedBase('t', shape=(len_y,)) 
697      >>> Dy = IndexedBase('Dy', shape=(len_y-1,)) 
698      >>> i = Idx('i', len_y-1) 
699      >>> e = Eq(Dy[i], (y[i+1]-y[i])/(t[i+1]-t[i])) 
700      >>> octave_code(e.rhs, assign_to=e.lhs, contract=False) 
701      'Dy(i) = (y(i + 1) - y(i))./(t(i + 1) - t(i));' 
702      """
703      return OctaveCodePrinter(settings).doprint(expr, assign_to)

707      """Prints the Octave (or Matlab) representation of the given expression. 
708   
709      See `octave_code` for the meaning of the optional arguments. 
710      """
711      print(octave_code(expr, **settings))
