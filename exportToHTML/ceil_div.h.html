1    #pragma once
2    #include <c10/macros/Macros.h>
3    #include <type_traits>

5    namespace at {
6    
7    /** 
8       Computes ceil(a / b) 
9    */
10   template <typename T, typename = std::enable_if_t<std::is_integral_v<T>>>
11   C10_ALWAYS_INLINE C10_HOST_DEVICE T ceil_div(T a, T b) {
12     return (a + b - 1) / b;
13   }
14   
15   /** 
16      Computes ceil(a / b) * b; i.e., rounds up `a` to the next highest 
17      multiple of b 
18   */
19   template <typename T>
20   C10_ALWAYS_INLINE C10_HOST_DEVICE T round_up(T a, T b) {
21     return ceil_div(a, b) * b;
22   }
