28   // Note on naming: it is unconventional.
29   // grad_in does not mean that it is a gradient wrt to input,
30   // grad_in/grad_out is just an input/output of unfold_backward kernel.
31   
32   [[maybe_unused]] static TensorIterator _make_unfold_backward_iter_over_grad_out(
33       Tensor& grad_out,
34       const Tensor& grad_in,
35       int64_t dim,
36       int64_t size,
37       int64_t step) {
38     dim = maybe_wrap_dim(dim, grad_out.dim());
39     // last dim stores the folds
40   
41     auto grad_out_dim_size = ensure_nonempty_size(grad_out, dim);
42     auto grad_in_dim_size = ensure_nonempty_size(grad_in, dim);
43     // dictates the number of elements to iterate over
44     // in dimension `dim`
45     auto iter_dim_size = std::min(
46       grad_out_dim_size,
47       (grad_in_dim_size - 1) * step + size
48     );
49   
50     /* prepare grad_out for TensorIterator { */
51     auto grad_out_strides = ensure_nonempty_vec(grad_out.strides().vec());
52     auto grad_out_sizes = ensure_nonempty_vec(grad_out.sizes().vec());
53     grad_out_sizes[dim] = iter_dim_size;
54     auto grad_out_restrided = grad_out.as_strided(
55       grad_out_sizes, grad_out_strides
56     );
57     /* } */
58   
59     /* prepare grad_in for TensorIterator { */
60     auto grad_in_strides = ensure_nonempty_vec(grad_in.strides().vec());
61     auto grad_in_sizes = ensure_nonempty_vec(grad_in.sizes().vec());
62   
63     // set strides for dim to 0
64     // and size to 1 because
65     // this dimension is indexed inside the kernel
66     grad_in_strides[dim] = 0;
67     grad_in_sizes[dim] = 1;
68   
69     grad_in_strides.pop_back();
70     grad_in_sizes.pop_back();
71   
72     auto grad_in_restrided = grad_in.squeeze(-1).as_strided(
73       grad_in_sizes, grad_in_strides
74     );
75     /* } */
76   
77     // During the TensorIterator iteration we have to know
78     // i_dim in grad_out[i_1,...,i_dim,...i_n],
79     // idx_dim stores this information
80     /* prepare idx_dim for TensorIterator { */
81     auto idx_dim = at::arange(
82       0, iter_dim_size, grad_in.options().dtype(at::kLong)
83     );
84   
85     auto grad_out_dim = ensure_nonempty_dim(grad_out.dim());
86   
87     auto idx_dim_strides = std::vector<int64_t>(grad_out_dim, 0);
88     auto idx_dim_sizes = std::vector<int64_t>(grad_out_dim, 1);
89   
90     idx_dim_strides[dim] = 1;
91     idx_dim_sizes[dim] = iter_dim_size;
92   
93     // idx_dim size will broadcast over determined by grad_out sizes in TensorIterator
94     auto idx_dim_restrided = idx_dim.as_strided(idx_dim_sizes, idx_dim_strides);
95     /* } */
96   
97     auto iter = TensorIteratorConfig()
98       .set_check_mem_overlap(false)
99       .check_all_same_dtype(false)
100      .resize_outputs(false)
101      .add_owned_output(grad_out_restrided)
102      .add_owned_const_input(grad_in_restrided)
103      .add_owned_const_input(idx_dim_restrided)
104      .build();
105  
106    return iter;
107  }
