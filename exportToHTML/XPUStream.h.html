9    /* 
10    * Note [Stream Management] 
11    * 
12    * An XPUStream is an abstraction of an actual SYCL queue in which SYCL kernel 
13    * can execute. Currently, there are several pools per device to manage SYCL 
14    * queue, and a device's pool is lazily created. 
15    * 
16    * There are two pools per device. The first pool contains "normal priority" 
17    * queues. The second pool is the "high priority" queues. There are 32 queues in 
18    * per pool per device, and when a queue is requested one of these queues is 
19    * returned round-robin. That is, the first queue requested is at index 0, the 
20    * second at index 1... to index 31, then index 0 again. 
21    * 
22    * This means that if 33 queues are requested, the first and last queues 
23    * requested are actually the same queue (under the covers) and kernels enqueued 
24    * on them cannot run concurrently. 
25    * 
26    * It is safe to enqueue a kernel on the same queue from two different 
27    * threads as the SYCL specification described. 
28    */
29   
30   static constexpr int max_compile_time_stream_priorities = 3;
31   
32   /* 
33    * This serves as a wrapper around c10::Stream and acts as a representation for 
34    * a SYCL queue, which allows asynchronous execution of XPU tasks. 
35    */
36   class C10_XPU_API XPUStream {
37    public:
38     enum Unchecked { UNCHECKED };
39   
40     /// Construct a XPUStream from a Stream. This construction is checked, and
41     /// will raise an error if the Stream is not, in fact, a XPU stream.
42     explicit XPUStream(Stream stream) : stream_(stream) {
43       TORCH_CHECK(stream_.device_type() == DeviceType::XPU);
44     }
45   
46     /// Construct a XPUStream from a Stream with no error checking.
47     explicit XPUStream(Unchecked, Stream stream) : stream_(stream) {}
48   
49     bool operator==(const XPUStream& other) const noexcept {
50       return unwrap() == other.unwrap();
51     }
52   
53     bool operator!=(const XPUStream& other) const noexcept {
54       return unwrap() != other.unwrap();
55     }
56   
57     /// Implicit conversion to sycl::queue&.
58     operator sycl::queue&() const {
59       return queue();
60     }
61   
62     /// Implicit conversion to sycl::queue*.
63     operator sycl::queue*() const {
64       return &queue();
65     }
66   
67     /// Implicit conversion to Stream (a.k.a., forget that the stream is a
68     /// XPU stream).
69     operator Stream() const {
70       return unwrap();
71     }
72   
73     /// Get the XPU device type that this stream is associated with.
74     DeviceType device_type() const {
75       return DeviceType::XPU;
76     }
77   
78     /// Get the XPU device index that this stream is associated with.
79     DeviceIndex device_index() const {
80       return stream_.device_index();
81     }
82   
83     /// Get the full Device that this stream is associated with. The Device is
84     /// guaranteed to be a XPU device.
85     Device device() const {
86       return Device(DeviceType::XPU, device_index());
87     }
88   
89     /// Return the stream ID corresponding to this particular stream. StreamId is
90     /// a int64_t representation generated by its type and index.
91     StreamId id() const {
92       return stream_.id();
93     }
94   
95     /// Return true if all enqueued tasks in this stream have been completed,
96     /// otherwise return false.
97     bool query() const {
98       return queue().ext_oneapi_empty();
99     }
100  
101    /// Performs a blocking wait for the completion of all enqueued tasks in this
102    /// stream.
103    void synchronize() const {
104      queue().wait_and_throw();
105      const c10::impl::PyInterpreter* interp = c10::impl::GPUTrace::get_trace();
106      if (C10_UNLIKELY(interp)) {
107        (*interp)->trace_gpu_stream_synchronization(
108            c10::kXPU, reinterpret_cast<uintptr_t>(&queue()));
109      }
110    }
111  
112    /// Return the priority that this stream is associated with. Lower numbers
113    /// represent higher priority.
114    int priority() const;
115  
116    /// Explicit conversion to sycl::queue&.
117    sycl::queue& queue() const;
118  
119    /// Explicit conversion to Stream.
120    Stream unwrap() const {
121      return stream_;
122    }
123  
124    /// Reversibly pack a XPUStream into a struct representation. The XPUStream
125    /// can be unpacked using unpack3().
126    struct c10::StreamData3 pack3() const {
127      return stream_.pack3();
128    }
129  
130    /// Unpack a XPUStream from the 3 fields generated by pack3().
131    static XPUStream unpack3(
132        StreamId stream_id,
133        DeviceIndex device_index,
134        DeviceType device_type) {
135      return XPUStream(Stream::unpack3(stream_id, device_index, device_type));
136    }
137  
138    /// Return the range of priority **supported by PyTorch**.
139    static std::tuple<int, int> priority_range() {
140      // See Note [XPU Stream priorities]
141      return std::make_tuple(1, -max_compile_time_stream_priorities + 2);
142    }
143  
144   private:
145    Stream stream_;
146  };

150   * 
151   * You can request a stream from the highest priority pool by setting 
152   * isHighPriority to true for a specific device. 
153   */
154  C10_XPU_API XPUStream
155  getStreamFromPool(const bool isHighPriority = false, DeviceIndex device = -1);
156  
157  /** 
158   * Get a stream from the pool in a round-robin fashion. 
159   * 
160   * You can request a stream by setting a priority value for a specific device. 
161   * The priority number lower, the priority higher. 
162   */
163  C10_XPU_API XPUStream
164  getStreamFromPool(const int priority, DeviceIndex device = -1);
165  
166  /** 
167   * Get an XPUStream from an external SYCL queue. 
168   * 
169   * This function allows interoperability with other libraries by enabling 
170   * the use of an external SYCL queue that was not created by PyTorch. This 
171   * can be useful for data exchange or other operations where integration 
172   * with non-PyTorch queues is required. 
173   * 
174   * NOTE: It is the user's responsibility to ensure that the referenced SYCL 
175   * queue remains alive while the corresponding XPUStream, or any c10::Stream 
176   * derived from it, is in use. The different SYCL queue pointers will result in 
177   * distinct XPUStream instances, even if the SYCL queues they dereference are 
178   * equivalent. 
179   */
180  C10_XPU_API XPUStream
181  getStreamFromExternal(sycl::queue* ext_queue, DeviceIndex device_index);
