48   def _get_dynamo_graph(function: Callable, inputs) -> torch.fx.Graph:
49       gm, _ = torchdynamo.export(function, aten_graph=True)(*inputs)
50       gm.graph.eliminate_dead_code()
51       return gm.graph

54   def _get_linear_patterns(input_size: list[int]):
55       in_channels = input_size[-1]
56       out_channels = 8  # hard coding but this should not matter
57       weight = torch.ones((out_channels, in_channels))
58       bias = torch.ones((out_channels,))
59       act = torch.ones(input_size)
60   
61       def linear_op(act, weight, bias=None):
62           return F.linear(act, weight, bias)
63   
64       pattern_w_bias = _get_dynamo_graph(linear_op, (act, weight, bias))
65       pattern_wo_bias = _get_dynamo_graph(linear_op, (act, weight))
66       return [pattern_w_bias, pattern_wo_bias]

69   def _supported_symmetric_quantized_operators() -> dict[str, list[OperatorPatternType]]:
70       supported_operators: dict[str, list[OperatorPatternType]] = {
71           # Both conv and linear should be able to handle relu + hardtanh fusion since
72           # those are clamp ops
73           "conv2d": [
74               [torch.nn.Conv2d, torch.nn.ReLU],
75               [torch.nn.Conv2d, F.relu],
76               [F.conv2d, torch.nn.ReLU],
77               [F.conv2d, F.relu],
78           ],
79           "linear": [[torch.nn.Linear], [F.linear]],
80           "add": [[torch.add]],
81           "adaptive_avg_pool2d": [
82               [torch.nn.AdaptiveAvgPool2d],
83               [F.adaptive_avg_pool2d],
84           ],
85       }
86       return copy.deepcopy(supported_operators)

89   def _get_supported_symmetric_config_and_operators() -> list[OperatorConfig]:
90       supported_config_and_operators: list[OperatorConfig] = []
91       for quantization_config in [
92           get_symmetric_quantization_config(),
93           get_symmetric_quantization_config(is_qat=True),
94           get_symmetric_quantization_config(is_per_channel=True),
95           get_symmetric_quantization_config(is_per_channel=True, is_qat=True),
96       ]:
97           ops = _supported_symmetric_quantized_operators()
98           supported_config_and_operators.extend(
99               OperatorConfig(quantization_config, pattern_list)
100              for pattern_list in ops.values()
101          )
102      return copy.deepcopy(supported_config_and_operators)

105  @functools.lru_cache
106  def get_symmetric_quantization_config(
107      is_per_channel: bool = False,
108      is_qat: bool = False,
109      is_dynamic: bool = False,
110      act_qmin: int = -128,
111      act_qmax: int = 127,
112      weight_qmin: int = -127,
113      weight_qmax: int = 127,
114  ):
115      extra_args: dict[str, Any] = {"eps": 2**-12}
116      if is_qat:
117          if is_dynamic:
118              act_observer_or_fake_quant_ctr = FakeQuantize
119              dynamic_quant_observer = MovingAverageMinMaxObserver.with_args(
120                  averaging_constant=1
121              )
122              extra_args["observer"] = dynamic_quant_observer
123          else:
124              act_observer_or_fake_quant_ctr = FusedMovingAvgObsFakeQuantize  # type: ignore[assignment]
125      else:
126          if is_dynamic:
127              act_observer_or_fake_quant_ctr = PlaceholderObserver  # type: ignore[assignment]
128          else:
129              act_observer_or_fake_quant_ctr = HistogramObserver  # type: ignore[assignment]
130  
131      act_quantization_spec = QuantizationSpec(
132          dtype=torch.int8,
133          quant_min=act_qmin,
134          quant_max=act_qmax,
135          qscheme=torch.per_tensor_affine,
136          is_dynamic=is_dynamic,
137          observer_or_fake_quant_ctr=act_observer_or_fake_quant_ctr.with_args(
138              **extra_args,
139          ),
140      )
141      weight_qscheme = (
142          torch.per_channel_symmetric if is_per_channel else torch.per_tensor_symmetric
143      )
144      weight_observer_or_fake_quant_ctr: _ObserverOrFakeQuantizeConstructor = (
145          MinMaxObserver
146      )
147      if is_qat:
148          # TODO: qat + per channel?
149          weight_observer_or_fake_quant_ctr = FusedMovingAvgObsFakeQuantize
150      elif is_per_channel:
151          weight_observer_or_fake_quant_ctr = PerChannelMinMaxObserver
152  
153      extra_args: dict[str, Any] = {"eps": 2**-12}
154      if is_qat:
155          if weight_qscheme == torch.per_tensor_symmetric:
156              extra_args["observer"] = MovingAverageMinMaxObserver
157          else:
158              extra_args["observer"] = MovingAveragePerChannelMinMaxObserver  # type: ignore[dict-item]
159      weight_quantization_spec = QuantizationSpec(
160          dtype=torch.int8,
161          quant_min=weight_qmin,
162          quant_max=weight_qmax,
163          qscheme=weight_qscheme,
164          ch_axis=0,
165          is_dynamic=False,
166          observer_or_fake_quant_ctr=weight_observer_or_fake_quant_ctr.with_args(
167              **extra_args
168          ),
169      )
170  
171      bias_quantization_spec = None
172      if is_dynamic:
173          quantization_config = QuantizationConfig(
174              act_quantization_spec,
175              None,
176              weight_quantization_spec,
177              bias_quantization_spec,
178              is_qat,
179          )
180      else:
181          quantization_config = QuantizationConfig(
182              act_quantization_spec,
183              act_quantization_spec,
184              weight_quantization_spec,
185              bias_quantization_spec,
186              is_qat,
187          )
188      return quantization_config

195  def _get_module_type_filter(tp: Callable):
196      """Get the module_type_filter function for a given module type, the filter accepts 
197      a node and checks if the node comes from a module that has certain module type 
198   
199      For example: 
200          node: linear_op = call_function[...](...)  # comes from a module with type Block -> Sub -> Linear 
201   
202   
203      >> module_type_filter = _get_module_type_filter(Sub)  # submodule with type `Sub`, under the `Block` submodule 
204      >> print(module_type_filter(node)) 
205      True  # the node is from the submodule `Sub` (same for `Block` and `Linear` as well) 
206      """
207  
208      tp_str = tp.__module__ + "." + tp.__qualname__
209  
210      def module_type_filter(n: Node) -> bool:
211          # example: {
212          #     'L__self___sub': ("L['self'].sub", <class '....Sub'>),
213          #     'L__self___sub_linear': ("L['self'].sub.linear", <class 'torch.nn.modules.linear.Linear'>)
214          # }
215          nn_module_stack = n.meta.get("nn_module_stack", {})
216          types = []
217          for _, t in nn_module_stack.values():
218              # export() returns str, but older APIs (e.g. capture_pre_autograd_graph)
219              # return type. Handle both cases.
220              if isinstance(t, type):
221                  t = t.__module__ + "." + t.__qualname__
222              types.append(t)
223          return tp_str in types
224  
225      return module_type_filter

228  def _get_not_module_type_or_name_filter(
229      tp_list: list[Callable], module_name_list: list[str]
230  ) -> Callable[[Node], bool]:
231      module_type_filters = [_get_module_type_filter(tp) for tp in tp_list]
232      module_name_list_filters = [_get_module_name_filter(m) for m in module_name_list]
233  
234      def not_module_type_or_name_filter(n: Node) -> bool:
235          return not any(f(n) for f in module_type_filters + module_name_list_filters)
236  
237      return not_module_type_or_name_filter

292      @classmethod
293      def get_supported_quantization_configs(cls) -> list[QuantizationConfig]:
294          op_configs: set[QuantizationConfig] = {
295              spec for spec, _ in cls.supported_config_and_operators

297          return list(op_configs)

299      @classmethod
300      def get_supported_operator_for_quantization_config(
301          cls, quantization_config: Optional[QuantizationConfig]
302      ) -> list[OperatorPatternType]:
303          if quantization_config is None:
304              all_ops = []
305              for _, ops in cls.supported_config_and_operators:
306                  all_ops.extend(ops)
307              return all_ops
308  
309          for config, ops in cls.supported_config_and_operators:
310              # note: this assumes each entry in cls.supported_spec_and_operators
311              # corresponds to one spec, e.g. we don't have
312              # [(spec1, op_list1), (spec1, op_list2), (spec2, op_list3)]
313              # where the first and second entry have the same spec but did not
314              # merge the op list
315              if config == quantization_config:
316                  return ops
317          return []

360      def annotate(self, model: torch.fx.GraphModule) -> torch.fx.GraphModule:
361          """just handling global spec for now"""
362          # hacked for handling dynamic linear quant. will fix later.
363          if self.global_config and self.global_config.input_activation.is_dynamic:  # type: ignore[union-attr]
364              model = self._annotate_for_dynamic_quantization_config(model)
365          else:
366              model = self._annotate_for_static_quantization_config(model)
367          propagate_annotation(model)
368          return model

370      def _annotate_all_static_patterns(
371          self,
372          model: torch.fx.GraphModule,
373          quantization_config: Optional[QuantizationConfig],
374          filter_fn: Optional[Callable[[Node], bool]] = None,
375      ) -> torch.fx.GraphModule:
376          # TODO: implement the support for None to be canceling out previous annotations
377          if quantization_config is None:
378              return model
379  
380          if quantization_config.is_qat:
381              for op in self.STATIC_QAT_ONLY_OPS:
382                  OP_TO_ANNOTATOR[op](model, quantization_config, filter_fn)
383          for op in self.STATIC_OPS:
384              OP_TO_ANNOTATOR[op](model, quantization_config, filter_fn)
385          return model

387      def _annotate_all_dynamic_patterns(
388          self,
389          model: torch.fx.GraphModule,
390          quantization_config: Optional[QuantizationConfig],
391          filter_fn: Optional[Callable[[Node], bool]] = None,
392      ) -> torch.fx.GraphModule:
393          # TODO: implement the support for None to be canceling out previous annotations
394          if quantization_config is None:
395              return model
396  
397          for op in self.DYNAMIC_OPS:
398              OP_TO_ANNOTATOR[op](model, quantization_config, filter_fn)
399          return model

401      def _annotate_for_static_quantization_config(
402          self, model: torch.fx.GraphModule
403      ) -> torch.fx.GraphModule:
404          module_name_list = list(self.module_name_config.keys())
405          for module_name, config in self.module_name_config.items():
406              self._annotate_all_static_patterns(
407                  model, config, _get_module_name_filter(module_name)
408              )
409  
410          tp_list = list(self.module_type_config.keys())
411          for module_type, config in self.module_type_config.items():
412              self._annotate_all_static_patterns(
413                  model, config, _get_module_type_filter(module_type)
414              )
415  
416          self._annotate_all_static_patterns(
417              model,
418              self.global_config,
419              _get_not_module_type_or_name_filter(tp_list, module_name_list),
420          )
421          return model

428              self._annotate_all_dynamic_patterns(

430              )

432          tp_list = list(self.module_type_config.keys())
433          for module_type, config in self.module_type_config.items():
434              self._annotate_all_dynamic_patterns(
435                  model, config, _get_module_type_filter(module_type)

438          self._annotate_all_dynamic_patterns(

440              self.global_config,
