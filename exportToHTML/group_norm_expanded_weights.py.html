9    from .expanded_weights_impl import ExpandedWeight, implements_per_sample_grads
10   from .expanded_weights_utils import (
11       forward_helper,
12       set_grad_sample_if_exists,
13       standard_kwargs,
14       unpack_expanded_weight_or_tensor,
15   )

18   @implements_per_sample_grads(F.group_norm)
19   class GroupNormPerSampleGrad(torch.autograd.Function):
20       @staticmethod
21       def forward(ctx, kwarg_names, _, *expanded_args_and_kwargs):
22           expanded_args, expanded_kwargs = standard_kwargs(
23               kwarg_names, expanded_args_and_kwargs
24           )
25           input, num_groups = expanded_args
26           N = input.shape[0]
27           C = input.shape[1]
28           HxW = reduce(operator.mul, input.shape[2:], 1)
29           weight, bias, eps = (
30               expanded_kwargs["weight"],
31               expanded_kwargs["bias"],
32               expanded_kwargs["eps"],
33           )
34           output, mean, rstd = forward_helper(
35               torch.native_group_norm,
36               (input, weight, bias, N, C, HxW, num_groups, eps),
37               {},
38           )
39           ctx.input, ctx.num_groups = input, num_groups
40           ctx.weight, ctx.eps = weight, eps
41           ctx.mean, ctx.rstd = mean, rstd
42           if isinstance(bias, ExpandedWeight):
43               ctx.bias = bias
44           if input.requires_grad and isinstance(weight, ExpandedWeight):
45               ctx.weight = weight
46           return output
47   
48       @staticmethod
49       def backward(ctx, grad_output):
50           input, num_groups = ctx.input, ctx.num_groups
51           weight, bias, eps = ctx.weight, ctx.bias, ctx.eps
52           mean, rstd = ctx.mean, ctx.rstd
53   
54           results: list[Optional[torch.Tensor]] = []
55           results.append(None)  # for kwarg names
56           results.append(None)  # for op reference
57   
58           if input.requires_grad:
59               weight_c = unpack_expanded_weight_or_tensor(
60                   weight, lambda t: t.contiguous()
61               )
62               input_c = input.contiguous()
63               grad_output_c = (
64                   grad_output.contiguous() if grad_output is not None else None
65               )
66               N = input.shape[0]
67               C = input.shape[1]
68               HxW = 1
69               for s in input.shape[2:]:
70                   HxW *= s
71               bw_fn = torch.ops.aten.native_group_norm_backward
72               results.append(
73                   bw_fn(
74                       grad_output_c,
75                       input_c,
76                       mean,
77                       rstd,
78                       weight_c,
79                       N,
80                       C,
81                       HxW,
82                       num_groups,
83                       (True, False, False),
84                   )[0]
85               )
86           else:
87               results.append(None)
88   
89           # weight and bias don't compute batched gradients; no other arguments are differentiable
90           results = results + [None] * 4
91   
92           # set grad_sample field for weight and bias with per sample gradients
93           if hasattr(ctx, "weight"):
94               set_grad_sample_if_exists(
95                   weight,
96                   lambda _: torch.einsum(
97                       "ni...->ni", F.group_norm(input, num_groups, eps=eps) * grad_output
98                   ),
99               )
100          if hasattr(ctx, "bias"):
101              set_grad_sample_if_exists(
102                  bias, lambda _: torch.einsum("ni...->ni", grad_output)
103              )
104          return tuple(results)
