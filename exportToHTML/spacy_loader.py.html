17   class SpacyModelLoader:
18       """ 
19       Robust spaCy model loader with automatic download, retry logic, and degraded mode fallback. 
20       """
21   
22       def __init__(self, max_retries: int = 2, retry_delay: float = 1.0):
23           self.max_retries = max_retries
24           self.retry_delay = retry_delay
25           self.degraded_mode = False
26           self.loaded_models = {}
27           return None
28   
29       def load_model(
30           self, model_name: str, disable: Optional[list] = None
31       ) -> Optional[Language]:
32           """ 
33           Load a spaCy model with automatic download and retry logic. 
34    
35           Args: 
36               model_name: Name of the spaCy model to load 
37               disable: List of pipeline components to disable 
38    
39           Returns: 
40               Loaded spaCy model or None if loading fails completely 
41           """
42           if model_name in self.loaded_models:
43               return self.loaded_models[model_name]
44   
45           # First attempt to load the model
46           model = self._try_load_model(model_name, disable)
47           if model is not None:
48               self.loaded_models[model_name] = model
49               return model
50   
51           # Model not found, attempt automatic download
52           logger.warning(
53               f"spaCy model '{model_name}' not found. Attempting automatic download..."
54           )
55   
56           if self._download_model_with_retry(model_name):
57               # Try loading again after successful download
58               model = self._try_load_model(model_name, disable)
59               if model is not None:
60                   self.loaded_models[model_name] = model
61                   logger.info(
62                       f"Successfully loaded spaCy model '{model_name}' after download"
63                   )
64                   return model
65   
66           # All attempts failed, enter degraded mode
67           logger.error(
68               f"Failed to load spaCy model '{model_name}'. Operating in degraded mode."
69           )
70           self.degraded_mode = True
71           return None
72   
73       def _try_load_model(
74           self, model_name: str, disable: Optional[list] = None
75       ) -> Optional[Language]:
76           """ 
77           Attempt to load a spaCy model without downloading. 
78    
79           Args: 
80               model_name: Name of the spaCy model to load 
81               disable: List of pipeline components to disable 
82    
83           Returns: 
84               Loaded spaCy model or None if loading fails 
85           """
86           try:
87               return spacy.load(model_name, disable=disable or [])
88           except Exception as e:  # Catch all exceptions to prevent SystemExit
89               log_debug_with_text(
90                   logger, f"Failed to load model '{model_name}': {e}")
91               return None
92   
93       def _download_model_with_retry(self, model_name: str) -> bool:
94           """ 
95           Download a spaCy model with retry logic. 
96    
97           Args: 
98               model_name: Name of the spaCy model to download 
99    
100          Returns: 
101              True if download successful, False otherwise 
102          """
103          for attempt in range(self.max_retries + 1):
104              try:
105                  logger.info(
106                      f"Downloading spaCy model '{model_name}' (attempt {attempt + 1}/{self.max_retries + 1})"
107                  )
108                  spacy.cli.download(model_name)
109                  logger.info(
110                      f"Successfully downloaded spaCy model '{model_name}'")
111                  return True
112  
113              except Exception as e:
114                  logger.warning(
115                      f"Download attempt {attempt + 1} failed for model '{model_name}': {e}"
116                  )
117  
118                  if attempt < self.max_retries:
119                      logger.info(f"Retrying in {self.retry_delay} seconds...")
120                      time.sleep(self.retry_delay)
121                  else:
122                      logger.error(
123                          f"All download attempts failed for model '{model_name}'. "
124                          f"Possible causes: offline environment, insufficient permissions, "
125                          f"or model name not found."
126                      )
127  
128          return False
129  
130      def is_degraded_mode(self) -> bool:
131          """Check if the loader is operating in degraded mode."""
132          return self.degraded_mode
133  
134      def get_loaded_models(self) -> dict:
135          """Get dictionary of successfully loaded models."""
136          return self.loaded_models.copy()

139  class SafeSpacyProcessor:
140      """ 
141      Example processor that gracefully handles missing spaCy models. 
142      """
143  
144      def __init__(self, preferred_model: str = "en_core_web_sm"):
145          self.loader = SpacyModelLoader()
146          self.model = self.loader.load_model(preferred_model)
147          self.preferred_model = preferred_model
148          return None
149      def process_text(self, text: str) -> dict:
150          """ 
151          Process text with available spaCy functionality or fallback methods. 
152   
153          Args: 
154              text: Input text to process 
155   
156          Returns: 
157              Dictionary with processing results 
158          """
159          if self.model is not None:
160              # Full functionality available
161              doc = self.model(text)
162              return {
163                  "tokens": [token.text for token in doc],
164                  "lemmas": [token.lemma_ for token in doc],
165                  "pos_tags": [token.pos_ for token in doc],
166                  "entities": [(ent.text, ent.label_) for ent in doc.ents],
167                  "sentences": [sent.text for sent in doc.sents],
168                  "processing_mode": "full",
169              }
170          else:
171              # Degraded mode - basic text processing
172              log_warning_with_text(
173                  logger,
174                  f"Processing text in degraded mode (no spaCy model available)",
175                  text,
176              )
177              return {
178                  "tokens": text.split(),  # Basic whitespace tokenization
179                  "lemmas": [],  # Not available in degraded mode
180                  "pos_tags": [],  # Not available in degraded mode
181                  "entities": [],  # Not available in degraded mode
182                  "sentences": text.split("."),  # Basic sentence splitting
183                  "processing_mode": "degraded",
184              }
185  
186      def is_fully_functional(self) -> bool:
187          """Check if processor has full spaCy functionality."""
188          return self.model is not None and not self.loader.is_degraded_mode()

191  def setup_logging():
192      """ 
193      Setup logging with RotatingFileHandler and configurable log directory. 
194      Falls back to current working directory if LOG_DIR is not set or not writable. 
195      """
196      # Get log directory from environment variable or fallback to current directory
197      log_dir = os.getenv("LOG_DIR", os.getcwd())
198  
199      # Ensure log directory exists and is writable
200      try:
201          os.makedirs(log_dir, exist_ok=True)
202          # Test if directory is writable
203          test_file = os.path.join(log_dir, ".write_test")
204          with open(test_file, "w") as f:
205              f.write("test")
206          os.remove(test_file)
207      except (PermissionError, OSError, IOError) as e:
208          # Fallback to current working directory
209          warning_msg = f"Cannot write to LOG_DIR '{log_dir}': {e}. Falling back to current directory."
210          log_dir = os.getcwd()
211          # Log warning to console since file logging isn't set up yet
212          print(f"WARNING: {warning_msg}")
213  
214          # Try current directory as final fallback
215          try:
216              test_file = os.path.join(log_dir, ".write_test")
217              with open(test_file, "w") as f:
218                  f.write("test")
219              os.remove(test_file)
220          except (PermissionError, OSError, IOError) as fallback_e:
221              print(f"WARNING: Cannot write to fallback directory '{log_dir}': {fallback_e}")
222              return None  # Skip file logging setup if no writable directory available
223  
224      log_file = os.path.join(log_dir, "spacy_loader.log")
225      # Configure root logger
226      root_logger = logging.getLogger()
227      root_logger.setLevel(logging.INFO)
228  
229      # Remove existing handlers to avoid duplicates
230      for handler in root_logger.handlers[:]:
231          root_logger.removeHandler(handler)
232  
233      try:
234          # Setup RotatingFileHandler with appropriate parameters
235          file_handler = RotatingFileHandler(
236              filename=log_file,
237              maxBytes=10 * 1024 * 1024,  # 10MB per file
238              backupCount=5,  # Keep 5 backup files
239              encoding="utf-8",
240          )
241          file_formatter = logging.Formatter(
242              "%(asctime)s - %(name)s - %(levelname)s - %(message)s"
243          )
244          file_handler.setFormatter(file_formatter)
245          root_logger.addHandler(file_handler)
246  
247          # Also add console handler for immediate feedback
248          console_handler = logging.StreamHandler()
249          console_formatter = logging.Formatter("%(levelname)s - %(message)s")
250          console_handler.setFormatter(console_formatter)
251          root_logger.addHandler(console_handler)
252  
253          logger.info(
254              f"Logging configured successfully. Log files in: {log_dir}")
255  
256      except (PermissionError, OSError, IOError) as e:
257          print(
258              f"WARNING: Failed to setup file logging: {e}. Using console logging only."
259          )
260          # Fallback to console-only logging
261          console_handler = logging.StreamHandler()
262          console_formatter = logging.Formatter("%(levelname)s - %(message)s")
263          console_handler.setFormatter(console_formatter)
264          root_logger.addHandler(console_handler)
265      
266      return None
