10   set(SYCL_ROOT "") 
11   if(DEFINED ENV{SYCL_ROOT}) 
12     set(SYCL_ROOT $ENV{SYCL_ROOT}) 
13   elseif(DEFINED ENV{CMPLR_ROOT}) 
14     set(SYCL_ROOT $ENV{CMPLR_ROOT}) 
15   else() 
16     # Use the default path to ensure proper linking with torch::xpurt when the user is working with libtorch.
17     if(CMAKE_SYSTEM_NAME MATCHES "Linux") 
18       set(SYCL_ROOT "/opt/intel/oneapi/compiler/latest") 
19     elseif(CMAKE_SYSTEM_NAME MATCHES "Windows") 
20       set(SYCL_ROOT "C:/Program Files (x86)/Intel/oneAPI/compiler/latest") 
21     endif() 
22     if(NOT EXISTS ${SYCL_ROOT}) 
23       set(SYCL_ROOT "") 
24     endif()

27   string(COMPARE EQUAL "${SYCL_ROOT}" "" nosyclfound) 
28   if(nosyclfound) 
29     set(SYCL_FOUND False) 
30     set(SYCL_REASON_FAILURE "SYCL library not set!!") 
31     set(SYCL_NOT_FOUND_MESSAGE "${SYCL_REASON_FAILURE}")

44   function(parse_sycl_compiler_version version_number) 
45     # Execute the SYCL compiler with the --version flag to match the version string.
46     execute_process(COMMAND ${SYCL_COMPILER} --version OUTPUT_VARIABLE SYCL_VERSION_STRING) 
47     string(REGEX REPLACE "Intel\\(R\\) (.*) Compiler ([0-9]+\\.[0-9]+\\.[0-9]+) (.*)" "\\2"
48                  SYCL_VERSION_STRING_MATCH ${SYCL_VERSION_STRING}) 
49     string(REPLACE "." ";" SYCL_VERSION_LIST ${SYCL_VERSION_STRING_MATCH}) 
50     # Split the version number list into major, minor, and patch components.
51     list(GET SYCL_VERSION_LIST 0 VERSION_MAJOR) 
52     list(GET SYCL_VERSION_LIST 1 VERSION_MINOR) 
53     list(GET SYCL_VERSION_LIST 2 VERSION_PATCH) 
54     # Calculate the version number in the format XXXXYYZZ, using the formula (major * 10000 + minor * 100 + patch).
55     math(EXPR VERSION_NUMBER_MATCH "${VERSION_MAJOR} * 10000 + ${VERSION_MINOR} * 100 + ${VERSION_PATCH}") 
56     set(${version_number} "${VERSION_NUMBER_MATCH}" PARENT_SCOPE) 
57   endfunction()

59   if(SYCL_COMPILER) 
60     parse_sycl_compiler_version(SYCL_COMPILER_VERSION) 
61   endif() 
62   
63   if(NOT SYCL_COMPILER_VERSION) 
64     set(SYCL_FOUND False) 
65     set(SYCL_REASON_FAILURE "Cannot parse sycl compiler version to get SYCL_COMPILER_VERSION!") 
66     set(SYCL_NOT_FOUND_MESSAGE "${SYCL_REASON_FAILURE}") 
67     return() 
68   endif()

78   # Find include/sycl path from include path.
79   find_file( 
80     SYCL_INCLUDE_SYCL_DIR 
81     NAMES sycl 
82     HINTS ${SYCL_ROOT}/include/ 
83     NO_DEFAULT_PATH

86   # Due to the unrecognized compilation option `-fsycl` in other compiler.
87   list(APPEND SYCL_INCLUDE_DIR ${SYCL_INCLUDE_SYCL_DIR})

89   # Find library directory from binary.
90   find_file( 
91     SYCL_LIBRARY_DIR 
92     NAMES lib lib64 
93     HINTS ${SYCL_ROOT}
94     NO_DEFAULT_PATH 
95     )

97   # Define the old version of SYCL toolkit that is compatible with the current version of PyTorch.
98   set(PYTORCH_2_5_SYCL_TOOLKIT_VERSION 20249999) 
99   
100  # By default, we use libsycl.so on Linux and sycl.lib on Windows as the SYCL library name.
101  if (SYCL_COMPILER_VERSION VERSION_LESS_EQUAL PYTORCH_2_5_SYCL_TOOLKIT_VERSION) 
102    # Don't use if(WIN32) here since this requires cmake>=3.25 and file is installed
103    # and used by other projects.
104    # See: https://cmake.org/cmake/help/v3.25/variable/LINUX.html
105    if(CMAKE_SYSTEM_NAME MATCHES "Windows") 
106      # On Windows, the SYCL library is named sycl7.lib until PYTORCH_2_5_SYCL_TOOLKIT_VERSION.
107      # sycl.lib is supported in the later version.
108      set(sycl_lib_suffix "7") 
109    endif() 
110  endif() 
111  
112  # Find SYCL library fullname.
113  find_library( 
114    SYCL_LIBRARY 
115    NAMES "sycl${sycl_lib_suffix}"
116    HINTS ${SYCL_LIBRARY_DIR}
117    NO_DEFAULT_PATH 
118  ) 
119  
120  # Find OpenCL library fullname, which is a dependency of oneDNN.
121  find_library( 
122    OCL_LIBRARY 
123    NAMES OpenCL 
124    HINTS ${SYCL_LIBRARY_DIR}
125    NO_DEFAULT_PATH 
126  ) 
127  
128  if((NOT SYCL_LIBRARY) OR (NOT OCL_LIBRARY)) 
129    set(SYCL_FOUND False) 
130    set(SYCL_REASON_FAILURE "SYCL library is incomplete!!") 
131    set(SYCL_NOT_FOUND_MESSAGE "${SYCL_REASON_FAILURE}") 
132    return() 
133  endif()
