18   @compatibility(is_backward_compatible=True)
19   class Partition:
20       def __init__(self, name: str):
21           self.name: str = name
22           self.submod_name = f"submod_{name}"
23           self.node_names: list[str] = []
24           self.inputs: dict[str, None] = {}
25           self.outputs: dict[str, None] = {}
26           self.dependencies: dict[str, None] = {}
27           self.dependents: dict[str, None] = {}
28           self.graph: torch.fx.graph.Graph = torch.fx.graph.Graph()
29           self.environment: dict[Node, Node] = {}
30           self.targets: dict[str, Any] = {}
31   
32       def __repr__(self) -> str:
33           return (
34               f"name: {self.name},\n"
35               f" nodes: {self.node_names},\n"
36               f" inputs: {self.inputs},\n"
37               f" outputs: {self.outputs},\n"
38               f" partitions depended on: {self.dependencies},\n"
39               f" partition dependents: {self.dependents}"
40           )

43   def _get_attr_from_qualname(mod: torch.nn.Module, qualname: str) -> Any:
44       attr_val = mod
45       for atom in qualname.split("."):  # type: ignore[union-attr]
46           if not hasattr(attr_val, atom):
47               raise AttributeError(f"Node target {qualname} not found!")
48           attr_val = getattr(attr_val, atom)
49       return attr_val

52   # Creates subgraphs out of main graph
53   @compatibility(is_backward_compatible=True)
54   def split_module(
55       m: GraphModule,
56       root_m: torch.nn.Module,
57       split_callback: Callable[[Node], int],
58       qualname_map: Optional[dict[str, str]] = None,
59       keep_original_order: Optional[bool] = False,
60       keep_original_node_name: Optional[bool] = False,
61       keep_original_input_name: bool = True,
62   ):
63       """ 
64       Creates subgraphs out of main graph 
65    
66       Args: 
67           m (GraphModule): Graph module to split 
68           root_m (torch.nn.Module): root nn module. Not currently used. Included 
69               because the root nn module is usually transformed via 
70               torch.fx._symbolic_trace.symbolic_trace (see example below) 
71           split_callback (Callable[[Node], int]): Callable function 
72               that maps a given Node instance to a numeric partition identifier. 
73               split_module will use this function as the policy for which operations 
74               appear in which partitions in the output Module. 
75           qualname_map: Optional[Dict[str, str]]: optional output parameter that returns a 
76               mapping from new target names in the module after split to old target 
77               names in the original module. 
78           keep_original_order: Optional[bool]: keep the original order of the GraphModule 
79               or use the Topological order of the new constructed GraphModule 
80           keep_original_node_name: Optional[bool]: If the partitioned graphs should 
81               have the same node names as the original graph. 
82           keep_original_input_name: bool: If the partitioned graphs should 
83               have the same input names as the original graph. 
84    
85       Returns: 
86           GraphModule: the module after split. 
87    
88       Example: 
89    
90           This is a sample setup: 
91    
92               import torch 
93               from torch.fx.symbolic_trace import symbolic_trace 
94               from torch.fx.graph_module import GraphModule 
95               from torch.fx.node import Node 
96               from torch.fx.passes.split_module import split_module 
97    
98               class MyModule(torch.nn.Module): 
99                   def __init__(self) -> None: 
100                      super().__init__() 
101                      self.param = torch.nn.Parameter(torch.rand(3, 4)) 
102                      self.linear = torch.nn.Linear(4, 5) 
103   
104                  def forward(self, x, y): 
105                      z = self.linear(x + self.param).clamp(min=0.0, max=1.0) 
106                      w = self.linear(y).clamp(min=0.0, max=1.0) 
107                      return z + w 
108   
109              # symbolically trace model 
110              my_module = MyModule() 
111              my_module_traced = symbolic_trace(my_module) 
112   
113              # random mod partitioning 
114              partition_counter = 0 
115              NPARTITIONS = 3 
116   
117              def mod_partition(node: Node): 
118                  global partition_counter 
119                  partition = partition_counter % NPARTITIONS 
120                  partition_counter = (partition_counter + 1) % NPARTITIONS 
121                  return partition 
122   
123              # split module in module with submodules 
124              module_with_submodules = split_module( 
125                  my_module_traced, my_module, mod_partition 
126              ) 
127   
128          Output looks like this. Original graph is broken into partitions 
129   
130              > print(module_with_submodules) 
131              GraphModule( 
132                  (submod_0): GraphModule( 
133                      (linear): Linear(in_features=4, out_features=5, bias=True) 
134                  ) 
135                  (submod_1): GraphModule( 
136                      (linear): Linear(in_features=4, out_features=5, bias=True) 
137                  ) 
138                  (submod_2): GraphModule() 
139              ) 
140   
141              def forward(self, x, y): 
142                  param = self.param 
143                  submod_0 = self.submod_0(x, param, y);  x = param = y = None 
144                  getitem = submod_0[0] 
145                  getitem_1 = submod_0[1];  submod_0 = None 
146                  submod_1 = self.submod_1(getitem, getitem_1);  getitem = getitem_1 = None 
147                  getitem_2 = submod_1[0] 
148                  getitem_3 = submod_1[1];  submod_1 = None 
149                  submod_2 = self.submod_2(getitem_2, getitem_3);  getitem_2 = getitem_3 = None 
150                  return submod_2 
151   
152          Output of split module is the same as output of input traced module. 
153          This is an example within a test setting: 
154   
155              > orig_out = my_module_traced(x, y) 
156              > submodules_out = module_with_submodules(x, y) 
157              > self.assertEqual(orig_out, submodules_out) 
158              True 
159      """

166      def construct_graph(
167          node: Node,
168          base_mod_env: dict[str, Node],
169          base_mod_attrs: dict[str, torch.fx.graph_module.GraphModule],
170      ):
171          if node.op == "placeholder":
172              default_value = (
173                  node.args[0] if len(node.args) > 0 else inspect.Signature.empty
174              )
175              if keep_original_node_name:
176                  args = (
177                      () if default_value is inspect.Signature.empty else (default_value,)
178                  )
179                  base_mod_env[node.name] = base_mod_graph.create_node(
180                      "placeholder",
181                      node.name,
182                      args=args,  # type: ignore[arg-type]
183                      type_expr=node.type,
184                  )
185              else:
186                  base_mod_env[node.name] = base_mod_graph.placeholder(
187                      node.target,  # type: ignore[arg-type]
188                      type_expr=node.type,
189                      default_value=default_value,
190                  )
191              base_mod_env[node.name].meta = node.meta.copy()
192          elif node.op == "get_attr":
193              base_mod_env[node.name] = base_mod_graph.get_attr(node.target)  # type: ignore[arg-type]
194              base_mod_env[node.name].meta = node.meta.copy()
195              assert isinstance(node.target, str)
196              attr_val = _get_attr_from_qualname(m, node.target)
197              base_mod_attrs[node.target] = attr_val  # type: ignore[index]
198          return base_mod_env, base_mod_attrs

202      partitions: dict[str, Partition] = {}
203      orig_nodes: dict[str, Node] = {}
204      symbol_to_node: dict[sympy.Symbol, Node] = {}
205  
206      def record_cross_partition_use(def_node: Node, use_node: Optional[Node]):
207          from torch.fx.experimental.symbolic_shapes import free_symbols
208  
209          defined = getattr(def_node, "_fx_partition", None)
210          used = getattr(use_node, "_fx_partition", None)
211  
212          log.debug(
213              "record_cross_partition_use %s (%s) %s (%s)",
214              def_node.name,
215              defined,
216              use_node.name if use_node is not None else "-",
217              used,
218          )
219  
220          if defined != used:
221              if defined is not None:
222                  def_partition = partitions[defined]
223                  def_partition.outputs.setdefault(def_node.name)
224                  if used is not None:
225                      def_partition.dependents.setdefault(used)
226  
227              if used is not None:
228                  use_partition = partitions[used]
229                  use_partition.inputs.setdefault(def_node.name)
230                  # We have made def_node an input to the use_partition.  If
231                  # this input has symbolic symbols in its size, those also must
232                  # be made as inputs to the partition
233                  if (def_val := def_node.meta.get("example_value")) is not None:
234                      for s in sorted(free_symbols(def_val), key=str):
235                          s_node = symbol_to_node[s]
236                          use_partition.inputs.setdefault(s_node.name)
237                          if symbol_to_node[s].op != "placeholder":
238                              # If the node that defines the symbol is not a
239                              # placeholder, we must make it an output of the
240                              # partition.  Note that this may be in a different
241                              # partition than defined!  Although, this doesn't
242                              # really make a difference for correctness, since
243                              # defined is guaranteed to have the symbol in
244                              # scope and can return it; you just get less
245                              # optimal codegen in this case.
246                              s_defined = getattr(s_node, "_fx_partition", None)
247                              if s_defined is not None:
248                                  s_def_partition = partitions[s_defined]
249                                  s_def_partition.outputs.setdefault(s_node.name)
250                                  s_def_partition.dependents.setdefault(used)
251                  if defined is not None:
252                      use_partition.dependencies.setdefault(defined)

254      def instantiate_node_partition_mapping(node):
255          partition_name = str(split_callback(node))
256          log.debug(
257              "instantiate_node_partition_mapping %s (%s)", node.name, partition_name
258          )
259  
260          # add node to partitions
261          partition = partitions.get(partition_name)
262          if partition is None:
263              partitions[partition_name] = partition = Partition(partition_name)
264  
265          partition.node_names.append(node.name)
266          node._fx_partition = partition_name
267  
268      # Global State Nodes are nodes which by their global state effects,
269      # "taint" all downstream nodes while they are active.
270      GLOBAL_STATE_NODES = [
271          torch.amp._enter_autocast,
272          torch.amp._exit_autocast,
273          torch._C._set_grad_enabled,
274      ]
275  
276      # For grad regions:
277      # ------------------------
278      # 1. first region: we do nothing
279      # 2. subsequent regions: we insert the set_grad at the beginning
280      grad_regions: OrderedDict[Node, set[int]] = OrderedDict()
281  
282      # For autocast regions:
283      # ------------------------
284      # 1. first region: we will only insert the _exit at the end
285      # 2. intermediate regions: we will insert both the
286      #    _enter at the beginning and _exit at the end
287      # 3. last region: we will only insert _enter at the beginning
288      # We will do so in the order in which the autocasts were instantiated.
289      autocast_regions: OrderedDict[Node, set[int]] = OrderedDict()
290      autocast_exits: dict[Node, Optional[Node]] = {}
291  
292      active_grad = None
293      active_autocasts = set()
294  
295      for node in m.graph.nodes:
296          # This will prefer placeholder bindings, because those come first.
297          # This is a little dangerous though: it is possible that an unbacked
298          # symbol is used without any binding site for it, in which case we
299          # will get a KeyError not able to find it.  I'd like to fix this by
300          # having passes.runtime_assert establish some invariants that I can
301          # rely on later, but this needs some extra work.  Quick fix first.
302          # See https://github.com/pytorch/pytorch/issues/130534
303          if (
304              (val := node.meta.get("example_value")) is not None
305              and isinstance(val, (torch.SymInt, torch.SymFloat))
306              and isinstance(s0 := val.node.expr, sympy.Symbol)
307              and s0 not in symbol_to_node
308          ):
309              symbol_to_node[val.node.expr] = node
310  
311          if node.op in ["placeholder", "get_attr", "output"]:
312              continue
313  
314          instantiate_node_partition_mapping(node)
315  
316          if node.op == "call_function" and node.target in GLOBAL_STATE_NODES:
317              if node.target == torch._C._set_grad_enabled:
318                  assert len(node.args) == 1
319                  assert isinstance(node.args[0], bool)
320                  active_grad = node
321                  grad_regions[active_grad] = set({split_callback(node)})
322              elif node.target == torch.amp._enter_autocast:
323                  # Should all be python constants
324                  assert all(not isinstance(arg, Node) for arg in node.args)
325                  active_autocasts.add(node)
326                  autocast_regions[node] = set({split_callback(node)})
327                  autocast_exits[node] = None
328              elif node.target == torch.amp._exit_autocast:
329                  assert len(node.args) == 1
330                  autocast_regions[node.args[0]].add(split_callback(node))
331                  active_autocasts.remove(node.args[0])
332                  autocast_exits[node.args[0]] = node
333  
334          if active_grad is not None:
335              grad_regions[active_grad].add(split_callback(node))
336  
337          for a in active_autocasts:
338              autocast_regions[a].add(split_callback(node))
339  
340      assert all(v is not None for v in autocast_exits.values()), "autocast must exit"
341  
342      autocast_regions = {k: sorted(v) for k, v in autocast_regions.items()}
343      grad_regions = {k: sorted(v) for k, v in grad_regions.items()}
344  
345      if _LOGGER.isEnabledFor(logging.DEBUG):
346          _LOGGER.debug("autocast_regions: %s", autocast_regions)
347          _LOGGER.debug("grad_regions: %s", grad_regions)
348  
349      assert_monotonically_increasing = bool(autocast_regions) or bool(grad_regions)
350  
351      # split nodes into partitions
352      highest_partition = -1
353      for node in m.graph.nodes:
354          orig_nodes[node.name] = node
355  
356          # TODO currently placeholders/parameters aren't put into random partitions,
357          # rather they're added to the graphs where they are used down below
358          if node.op in ["placeholder", "get_attr"]:
359              continue
360          if node.op == "output":
361              torch.fx.graph.map_arg(
362                  node.args[0], lambda n: record_cross_partition_use(n, None)
363              )
364              continue
365  
366          if assert_monotonically_increasing:
367              pid = split_callback(node)
368              assert highest_partition <= pid, (
369                  "autocast or set_grad_enabled require monotonically increasing partitions:"
370                  f"highest: {highest_partition}, this node's: {pid}"
371              )
372              highest_partition = pid
373  
374          # do not capture cross-partition dependencies for global state nodes as they will be
375          # self-contained - their setup and unwind will be isolated to each partition submodule.
376          if node.target not in GLOBAL_STATE_NODES:
377              torch.fx.graph.map_arg(
378                  node.args, lambda def_node: record_cross_partition_use(def_node, node)
379              )
380              torch.fx.graph.map_arg(
381                  node.kwargs, lambda def_node: record_cross_partition_use(def_node, node)
382              )  # noqa: B950
383  
384      original_partition_order = list(partitions.keys())
385      # find partitions with no dependencies
386      root_partitions: list[str] = []
387      for partition_name, partition in partitions.items():
388          if not len(partition.dependencies):
389              root_partitions.append(partition_name)
390  
391      # check partitions for circular dependencies and create topological partition ordering
392      sorted_partitions: list[str] = []
393      while root_partitions:
394          root_partition = root_partitions.pop()
395          sorted_partitions.append(root_partition)
396          for dependent in partitions[root_partition].dependents:
397              partitions[dependent].dependencies.pop(root_partition)
398              if not partitions[dependent].dependencies:
399                  root_partitions.append(dependent)
400      if len(sorted_partitions) != len(partitions):
401          raise RuntimeError("cycle exists between partitions!")
402  
403      # Enter prelude
404      for regions_mapping in [autocast_regions, grad_regions]:
405          for node, regions in regions_mapping.items():
406              assert len(regions) > 0
407              partitions[str(regions[0])].environment[node] = node
408              for r in regions[1:]:
409                  partition = partitions[str(r)]
410                  new_node = partition.graph.create_node(
411                      op=node.op,
412                      target=node.target,
413                      args=tuple(arg for arg in node.args),
414                      kwargs={},
415                      type_expr=node.type,
416                  )
417                  new_node.meta = (
418                      node.meta.copy()
419                  )  # is it really a good idea to copy this?
420                  partition.environment[node] = new_node
421  
422      # add placeholders to partition inputs
423      for partition_name in sorted_partitions:
424          partition = partitions[partition_name]
425          new_inputs: dict[str, None] = {}
426  
427          counter = 0
428  
429          for inp in partition.inputs:
430              orig_node = orig_nodes[inp]
431              # We don't pass in get_attr nodes as inputs to the partition, but
432              # instead set them as targets and use getattr within the module
433  
434              def add_placeholder():
435                  if keep_original_input_name:
436                      name = inp
437                  else:
438                      nonlocal counter
439                      name = f"arg_{counter}"
440                      counter += 1
441                  placeholder = partition.graph.placeholder(
442                      name,
443                      type_expr=orig_nodes[inp].type,
444                  )
445                  new_inputs[inp] = None
446                  return placeholder
447  
448              if orig_node.op == "get_attr":
449                  assert isinstance(orig_node.target, str)
450  
451                  orig_attr = _get_attr_from_qualname(m, orig_node.target)
452                  if isinstance(orig_attr, torch.nn.Module):
453                      placeholder = partition.graph.get_attr(orig_node.target)
454                      partition.targets[orig_node.target] = orig_attr
455                  else:
456                      placeholder = add_placeholder()
457              else:
458                  placeholder = add_placeholder()
459              placeholder.meta = orig_nodes[inp].meta.copy()
460              partition.environment[orig_nodes[inp]] = placeholder
461          partition.inputs = new_inputs
462  
463      # Transform nodes and collect targets for partition's submodule
464      for node in m.graph.nodes:
465          if hasattr(node, "_fx_partition"):
466              partition = partitions[node._fx_partition]
467  
468              # swap out old graph nodes in kw/args with references to new nodes in this submodule
469              environment = partition.environment
470              gathered_args = torch.fx.graph.map_arg(node.args, lambda n: environment[n])
471              gathered_kwargs = torch.fx.graph.map_arg(
472                  node.kwargs, lambda n: environment[n]
473              )
474  
475              if node.op not in ["call_module", "get_attr"]:
476                  target = node.target
477              else:
478                  target_attr = _get_attr_from_qualname(m, node.target)
479                  target = node.target.replace(".", "_")
480                  partition.targets[target] = target_attr
481                  # Fill in the passed-in mapping from new qualname to old qualname
482                  if qualname_map is not None:
483                      # When creating the split module later, the submodules will have
484                      # path prefix matching the corresponding partition's submod_name
485                      qualname = f"{partition.submod_name}.{target}"
486                      qualname_map[qualname] = node.target
487  
488              assert isinstance(gathered_args, tuple)
489              assert isinstance(gathered_kwargs, dict)
490              name = node.name if keep_original_node_name else None
491              new_node = partition.graph.create_node(
492                  op=node.op,
493                  target=target,
494                  args=gathered_args,
495                  kwargs=gathered_kwargs,
496                  type_expr=node.type,
497                  name=name,
498              )
499              new_node.meta = node.meta.copy()
500              partition.environment[node] = new_node
501  
502      # Exit epilogue
503      for regions_mapping in [autocast_regions]:
504          for node in reversed(regions_mapping):
505              regions = regions_mapping[node]
506              assert len(regions) > 0
507              for r in regions[:-1]:
508                  partition = partitions[str(r)]
509                  exit_node = autocast_exits[node]
510                  assert exit_node is not None, "Missing exit node"
511                  new_node = partition.graph.create_node(
512                      op=exit_node.op,
513                      target=exit_node.target,
514                      args=(partition.environment[node],),
515                      kwargs={},
516                      type_expr=exit_node.type,
517                  )
518                  new_node.meta = (
519                      exit_node.meta.copy()
520                  )  # is it really a good idea to copy this?
521  
522      # original module environment dict mapping node names to nodes
523      orig_mod_env: dict[str, Node] = {}
524      # Set up values to construct base module
525      base_mod_env: dict[str, Node] = {}
526      base_mod_graph: torch.fx.graph.Graph = torch.fx.graph.Graph()
527      base_mod_attrs: dict[str, torch.fx.graph_module.GraphModule] = {}
528      if not keep_original_order:
529          for node in m.graph.nodes:
530              base_mod_env, base_mod_attrs = construct_graph(
531                  node, base_mod_env, base_mod_attrs
532              )
533  
534      else:
535          # Go through the graph to construct the mapping dict
536          for node in m.graph.nodes:
537              orig_mod_env[node.name] = node
538  
539      # Do some things iterating over the partitions in topological order again:
540      # 1) Finish off submodule Graphs by setting corresponding outputs
541      # 2) Construct GraphModules for each submodule
542      # 3) Construct the base graph by emitting calls to those submodules in
543      #    topological order or original order specified by keep_original_order
544  
545      construct_order_partitions = (
546          sorted_partitions if not keep_original_order else original_partition_order
547      )
548  
549      already_constructed_attr_nodes = set()
550  
551      # We actually need to insert the placeholder nodes in the original order
552      # otherwise graph signature will be wrong.
553      original_order = [node for node in m.graph.nodes if node.op == "placeholder"]
554  
555      for partition_name in construct_order_partitions:
556          partition = partitions[partition_name]
557  
558          # Set correct output values
559          output_vals = tuple(
560              partition.environment[orig_nodes[name]] for name in partition.outputs
561          )
562  
563          # skip output node generation if there are no output values
564          num_output_vals = len(output_vals)
565          if num_output_vals == 1:
566              partition.graph.output(output_vals[0])
567          elif num_output_vals > 1:
568              partition.graph.output(output_vals)
569          else:
570              # Invariant - Graph should always have an output node.
571              partition.graph.output(())
572  
573          if keep_original_order:
574              # first get the attr nodes required by this partition
575              orig_mod_attr_nodes: list[Node] = [
576                  orig_mod_env[key]
577                  for key in partition.inputs
578                  if key not in original_order
579              ]
580  
581              for node in original_order:
582                  if node in already_constructed_attr_nodes:
583                      continue  # already added this attr to the base graph
584                  base_mod_env, _based_mod_attrs = construct_graph(
585                      node, base_mod_env, base_mod_attrs
586                  )
587                  already_constructed_attr_nodes.add(node)
588  
589              # Construct GraphModule for this partition
590              for node in orig_mod_attr_nodes:  # type: ignore[attr-defined]
591                  if node in already_constructed_attr_nodes:
592                      continue
593                  base_mod_env, base_mod_attrs = construct_graph(
594                      node, base_mod_env, base_mod_attrs
595                  )
596                  already_constructed_attr_nodes.add(node)
597  
598          base_mod_attrs[partition.submod_name] = torch.fx.graph_module.GraphModule(
599              partition.targets, partition.graph
600          )  # noqa: B950
601  
602          # Emit call in base graph to this submodule
603          output_val = base_mod_graph.call_module(
604              partition.submod_name,
605              tuple(base_mod_env[name] for name in partition.inputs),
606          )
607  
608          num_outputs = len(partition.outputs)
609          if num_outputs > 1:
610              # Unpack multiple return values from submodule
611              output_val_proxy = torch.fx.proxy.Proxy(output_val)
612              for i, output_name in enumerate(partition.outputs):
613                  base_mod_env[output_name] = output_val_proxy[i].node  # type: ignore[index]
614          elif num_outputs == 1:
615              base_mod_env[next(iter(partition.outputs))] = output_val
616  
617      # When keep_original_order=True and if the graph doesn't have any
618      # `call_function` node then `base_mod_graph`, `base_mod_env` and `base_mod_attrs`
619      # are never populated.
620      # For this case, we call `construct_graph` here which takes care of updating them.
621      if keep_original_order and not base_mod_env:
622          for node in m.graph.nodes:
623              base_mod_env, base_mod_attrs = construct_graph(
624                  node, base_mod_env, base_mod_attrs
625              )
626  
627      # Add output node to `base_mod_graph` (i.e. the split graph) which will be returned.
628      for node in m.graph.nodes:
629          if node.op == "output":
630              base_mod_graph.output(
631                  torch.fx.graph.map_arg(node.args[0], lambda n: base_mod_env[n.name])
632              )  # noqa: B950
633  
634      ret = torch.fx.graph_module.GraphModule(base_mod_attrs, base_mod_graph)
635      log.debug(
636          "%s",
637          lazy_format_graph_code("post split_module", ret, colored=True),
638      )
639      return ret
