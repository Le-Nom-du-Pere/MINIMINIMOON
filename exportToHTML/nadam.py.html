32   class NAdam(Optimizer):  # noqa: D101
33       def __init__(
34           self,
35           params: ParamsT,
36           lr: Union[float, Tensor] = 2e-3,
37           betas: tuple[float, float] = (0.9, 0.999),
38           eps: float = 1e-8,
39           weight_decay: float = 0,
40           momentum_decay: float = 4e-3,
41           decoupled_weight_decay: bool = False,
42           *,
43           foreach: Optional[bool] = None,
44           maximize: bool = False,
45           capturable: bool = False,
46           differentiable: bool = False,
47       ):  # noqa: D107
48           if isinstance(lr, Tensor) and lr.numel() != 1:
49               raise ValueError("Tensor lr must be 1-element")
50           if not 0.0 <= lr:
51               raise ValueError(f"Invalid learning rate: {lr}")
52           if not 0.0 <= eps:
53               raise ValueError(f"Invalid epsilon value: {eps}")
54           if not 0.0 <= betas[0] < 1.0:
55               raise ValueError(f"Invalid beta parameter at index 0: {betas[0]}")
56           if not 0.0 <= betas[1] < 1.0:
57               raise ValueError(f"Invalid beta parameter at index 1: {betas[1]}")
58           if not 0.0 <= weight_decay:
59               raise ValueError(f"Invalid weight_decay value: {weight_decay}")
60           if not 0.0 <= momentum_decay:
61               raise ValueError(f"Invalid momentum_decay value: {momentum_decay}")
62           defaults = dict(
63               lr=lr,
64               betas=betas,
65               eps=eps,
66               weight_decay=weight_decay,
67               momentum_decay=momentum_decay,
68               decoupled_weight_decay=decoupled_weight_decay,
69               maximize=maximize,
70               foreach=foreach,
71               capturable=capturable,
72               differentiable=differentiable,
73           )
74           super().__init__(params, defaults)
75   
76       def __setstate__(self, state):  # noqa: D105
77           super().__setstate__(state)
78           for group in self.param_groups:
79               group.setdefault("maximize", False)
80               group.setdefault("foreach", None)
81               group.setdefault("capturable", False)
82               group.setdefault("differentiable", False)
83               group.setdefault("decoupled_weight_decay", False)
84               for p in group["params"]:
85                   p_state = self.state.get(p, [])
86                   if len(p_state) != 0:
87                       if not torch.is_tensor(p_state["step"]):
88                           step_val = float(p_state["step"])
89                           p_state["step"] = (
90                               torch.tensor(
91                                   step_val, dtype=_get_scalar_dtype(), device=p.device
92                               )
93                               if group["capturable"]
94                               else torch.tensor(step_val, dtype=_get_scalar_dtype())
95                           )
96                       if not torch.is_tensor(p_state["mu_product"]):
97                           mu_prod_val = p_state["mu_product"]
98                           p_state["mu_product"] = (
99                               torch.tensor(
100                                  mu_prod_val, dtype=_get_scalar_dtype(), device=p.device
101                              )
102                              if group["capturable"]
103                              else torch.tensor(mu_prod_val, dtype=_get_scalar_dtype())
104                          )
105  
106      def _init_group(
107          self,
108          group,
109          params_with_grad,
110          grads,
111          exp_avgs,
112          exp_avg_sqs,
113          mu_products,
114          state_steps,
115      ):
116          has_complex = False
117          for p in group["params"]:
118              if p.grad is not None:
119                  has_complex |= torch.is_complex(p)
120                  params_with_grad.append(p)
121                  if p.grad.is_sparse:
122                      raise RuntimeError("NAdam does not support sparse gradients")
123                  grads.append(p.grad)
124  
125                  state = self.state[p]
126                  # Lazy state initialization
127                  if len(state) == 0:
128                      # note(crcrpar): [special device hosting for step]
129                      # Deliberately host `step` and `mu_product` on CPU if capturable is False.
130                      # This is because kernel launches are costly on CUDA and XLA.
131                      state["step"] = (
132                          torch.zeros((), dtype=_get_scalar_dtype(), device=p.device)
133                          if group["capturable"]
134                          else torch.tensor(0.0, dtype=_get_scalar_dtype())
135                      )
136                      state["mu_product"] = (
137                          torch.ones((), dtype=_get_scalar_dtype(), device=p.device)
138                          if group["capturable"]
139                          else torch.tensor(1.0, dtype=_get_scalar_dtype())
140                      )
141                      # Exponential moving average of gradient values
142                      state["exp_avg"] = torch.zeros_like(
143                          p, memory_format=torch.preserve_format
144                      )
145                      # Exponential moving average of squared gradient values
146                      state["exp_avg_sq"] = torch.zeros_like(
147                          p, memory_format=torch.preserve_format
148                      )
149  
150                  exp_avgs.append(state["exp_avg"])
151                  exp_avg_sqs.append(state["exp_avg_sq"])
152                  mu_products.append(state["mu_product"])
153                  state_steps.append(state["step"])
154          return has_complex

156      @_use_grad_for_differentiable
157      def step(self, closure=None):
158          """Perform a single optimization step. 
159   
160          Args: 
161              closure (Callable, optional): A closure that reevaluates the model 
162                  and returns the loss. 
163          """
164          self._cuda_graph_capture_health_check()
165  
166          loss = None
167          if closure is not None:
168              with torch.enable_grad():
169                  loss = closure()
170  
171          for group in self.param_groups:
172              params_with_grad: list[Tensor] = []
173              grads: list[Tensor] = []
174              exp_avgs: list[Tensor] = []
175              exp_avg_sqs: list[Tensor] = []
176              mu_products: list[Tensor] = []
177              state_steps: list[Tensor] = []
178              beta1, beta2 = cast(tuple[float, float], group["betas"])
179  
180              has_complex = self._init_group(
181                  group,
182                  params_with_grad,
183                  grads,
184                  exp_avgs,
185                  exp_avg_sqs,
186                  mu_products,
187                  state_steps,
188              )
189  
190              nadam(
191                  params_with_grad,
192                  grads,
193                  exp_avgs,
194                  exp_avg_sqs,
195                  mu_products,
196                  state_steps,
197                  beta1=beta1,
198                  beta2=beta2,
199                  lr=group["lr"],
200                  weight_decay=group["weight_decay"],
201                  momentum_decay=group["momentum_decay"],
202                  eps=group["eps"],
203                  maximize=group["maximize"],
204                  decoupled_weight_decay=group["decoupled_weight_decay"],
205                  foreach=group["foreach"],
206                  capturable=group["capturable"],
207                  differentiable=group["differentiable"],
208                  has_complex=has_complex,
209              )
210  
211          return loss

215      r"""Implements NAdam algorithm. 
216   
217      .. math:: 
218         \begin{aligned} 
219              &\rule{110mm}{0.4pt}                                                                 \\ 
220              &\textbf{input}      : \gamma_t \text{ (lr)}, \: \beta_1,\beta_2 \text{ (betas)}, 
221                  \: \theta_0 \text{ (params)}, \: f(\theta) \text{ (objective)}                   \\ 
222              &\hspace{13mm} \: \lambda \text{ (weight decay)}, \:\psi \text{ (momentum decay)}    \\ 
223              &\hspace{13mm} \: \textit{decoupled\_weight\_decay}, \:\textit{maximize}             \\ 
224              &\textbf{initialize} :  m_0 \leftarrow 0 \text{ ( first moment)}, 
225                  v_0 \leftarrow 0 \text{ ( second moment)}                                 \\[-1.ex] 
226              &\rule{110mm}{0.4pt}                                                                 \\ 
227              &\textbf{for} \: t=1 \: \textbf{to} \: \ldots \: \textbf{do}                         \\ 
228              &\hspace{5mm}\textbf{if} \: \textit{maximize}:                                       \\ 
229              &\hspace{10mm}g_t           \leftarrow   -\nabla_{\theta} f_t (\theta_{t-1})         \\ 
230              &\hspace{5mm}\textbf{else}                                                           \\ 
231              &\hspace{10mm}g_t           \leftarrow   \nabla_{\theta} f_t (\theta_{t-1})          \\ 
232              &\hspace{5mm} \theta_t \leftarrow \theta_{t-1}                                       \\ 
233              &\hspace{5mm} \textbf{if} \: \lambda \neq 0                                          \\ 
234              &\hspace{10mm}\textbf{if} \: \textit{decoupled\_weight\_decay}                       \\ 
235              &\hspace{15mm} \theta_t \leftarrow \theta_{t-1} - \gamma \lambda \theta_{t-1}                    \\ 
236              &\hspace{10mm}\textbf{else}                                                          \\ 
237              &\hspace{15mm} g_t \leftarrow g_t + \lambda \theta_{t-1}                             \\ 
238              &\hspace{5mm} \mu_t \leftarrow \beta_1 \big(1 - \frac{1}{2}  0.96^{t \psi} \big)     \\ 
239              &\hspace{5mm} \mu_{t+1} \leftarrow \beta_1 \big(1 - \frac{1}{2} 0.96^{(t+1)\psi}\big)\\ 
240              &\hspace{5mm}m_t           \leftarrow   \beta_1 m_{t-1} + (1 - \beta_1) g_t          \\ 
241              &\hspace{5mm}v_t           \leftarrow   \beta_2 v_{t-1} + (1-\beta_2) g^2_t          \\ 
242              &\hspace{5mm}\widehat{m_t} \leftarrow \mu_{t+1} m_t/(1-\prod_{i=1}^{t+1}\mu_i)\\[-1.ex] 
243              & \hspace{11mm} + (1-\mu_t) g_t /(1-\prod_{i=1}^{t} \mu_{i})                         \\ 
244              &\hspace{5mm}\widehat{v_t} \leftarrow   v_t/\big(1-\beta_2^t \big)                   \\ 
245              &\hspace{5mm}\theta_t \leftarrow \theta_t - \gamma \widehat{m_t}/ 
246                  \big(\sqrt{\widehat{v_t}} + \epsilon \big)                                       \\ 
247              &\rule{110mm}{0.4pt}                                                          \\[-1.ex] 
248              &\bf{return} \:  \theta_t                                                     \\[-1.ex] 
249              &\rule{110mm}{0.4pt}                                                          \\[-1.ex] 
250         \end{aligned} 
251   
252      For further details regarding the algorithm we refer to `Incorporating Nesterov Momentum into Adam`_. 
253      """
254      + rf"""
255      Args:
256          {_params_doc}
257          lr (float, Tensor, optional): learning rate (default: 2e-3)
258          betas (Tuple[float, float], optional): coefficients used for computing
259              running averages of gradient and its square (default: (0.9, 0.999))
260          eps (float, optional): term added to the denominator to improve
261              numerical stability (default: 1e-8)
262          weight_decay (float, optional): weight decay (L2 penalty) (default: 0)
263          momentum_decay (float, optional): momentum momentum_decay (default: 4e-3)
264          decoupled_weight_decay (bool, optional): whether to decouple the weight
265              decay as in AdamW to obtain NAdamW. If True, the algorithm does not

267          {_foreach_doc}

281  def _single_tensor_nadam(
282      params: list[Tensor],
283      grads: list[Tensor],
284      exp_avgs: list[Tensor],
285      exp_avg_sqs: list[Tensor],
286      mu_products: list[Tensor],
287      state_steps: list[Tensor],
288      *,
289      beta1: float,
290      beta2: float,
291      lr: float,
292      weight_decay: float,
293      momentum_decay: float,
294      eps: float,
295      decoupled_weight_decay: bool,
296      maximize: bool,
297      capturable: bool,
298      differentiable: bool,
299      has_complex: bool,
300  ):
301      if not torch.jit.is_scripting():
302          lr = _to_scalar(lr)
303  
304      for i, param in enumerate(params):
305          grad = grads[i] if not maximize else -grads[i]
306          exp_avg = exp_avgs[i]
307          exp_avg_sq = exp_avg_sqs[i]
308          mu_product = mu_products[i]
309          step_t = state_steps[i]
310  
311          if torch.is_complex(param):
312              param = torch.view_as_real(param)
313              grad = torch.view_as_real(grad)
314              exp_avg = torch.view_as_real(exp_avg)
315              exp_avg_sq = torch.view_as_real(exp_avg_sq)
316  
317          # If compiling, the compiler will handle cudagraph checks, see note [torch.compile x capturable]
318          if not torch.compiler.is_compiling() and capturable:
319              capturable_supported_devices = _get_capturable_supported_devices()
320              assert (
321                  param.device.type == mu_product.device.type == step_t.device.type
322                  and param.device.type in capturable_supported_devices
323              ), (
324                  f"If capturable=True, params, mu_products and state_steps must be "
325                  f"on supported devices: {capturable_supported_devices}."
326              )
327  
328          # update step
329          step_t += 1
330  
331          if capturable:
332              step = step_t
333          else:
334              step = _get_value(step_t)
335  
336          bias_correction2 = 1 - beta2**step
337  
338          if weight_decay != 0:
339              if decoupled_weight_decay:
340                  # Perform stepweight decay
341                  param.mul_(1 - lr * weight_decay)
342              else:
343                  grad = grad.add(param, alpha=weight_decay)
344  
345          # calculate the momentum cache \mu^{t} and \mu^{t+1}
346          mu = beta1 * (1.0 - 0.5 * (0.96 ** (step * momentum_decay)))
347          mu_next = beta1 * (1.0 - 0.5 * (0.96 ** ((step + 1) * momentum_decay)))
348  
349          # update mu_product
350          mu_product *= mu
351  
352          # decay the first and second moment running average coefficient
353          exp_avg.lerp_(grad, 1 - beta1)
354          exp_avg_sq.mul_(beta2).addcmul_(grad, grad, value=1 - beta2)
355          denom = exp_avg_sq.div(bias_correction2).sqrt()
356  
357          if differentiable or capturable:
358              denom = denom.add(eps)
359              # Make autograd track the operations
360              # by updating the grad and exp_avg directly and not using the
361              # scalar "value" argument of addcdiv.
362              mu_product_next = mu_product * mu_next
363              grad = grad * (-lr * (1.0 - mu) / (1.0 - mu_product))
364              exp_avg = exp_avg * (-lr * mu_next / (1.0 - mu_product_next))
365              param.addcdiv_(grad, denom)
366              param.addcdiv_(exp_avg, denom)
367          else:
368              mu_product_next = _get_value(mu_product) * mu_next
369              denom.add_(eps)
370              param.addcdiv_(
371                  grad, denom, value=(-lr * (1.0 - mu) / (1.0 - _get_value(mu_product)))
372              )
373              param.addcdiv_(
374                  exp_avg, denom, value=(-lr * mu_next) / (1.0 - mu_product_next)
375              )

378  def _multi_tensor_nadam(
379      params: list[Tensor],
380      grads: list[Tensor],
381      exp_avgs: list[Tensor],
382      exp_avg_sqs: list[Tensor],
383      mu_products: list[Tensor],
384      state_steps: list[Tensor],
385      *,
386      beta1: float,
387      beta2: float,
388      lr: float,
389      weight_decay: float,
390      momentum_decay: float,
391      eps: float,
392      decoupled_weight_decay: bool,
393      maximize: bool,
394      capturable: bool,
395      differentiable: bool,
396      has_complex: bool,
397  ):
398      if len(params) == 0:
399          return
400  
401      assert not differentiable, "_foreach ops don't support autograd"
402  
403      # If compiling, the compiler will handle cudagraph checks, see note [torch.compile x capturable]
404      if not torch.compiler.is_compiling() and capturable:
405          capturable_supported_devices = _get_capturable_supported_devices(
406              supports_xla=False
407          )
408          assert all(
409              p.device.type == mp.device.type == step.device.type
410              and p.device.type in capturable_supported_devices
411              for p, mp, step in zip(params, mu_products, state_steps)
412          ), (
413              "If capturable=True, "
414              "params, mu_products, and state_steps must be on supported devices: "
415              f"{capturable_supported_devices}."
416          )
417  
418      lr = _to_scalar(lr)
419  
420      grouped_tensors = Optimizer._group_tensors_by_device_and_dtype(
421          [params, grads, exp_avgs, exp_avg_sqs, mu_products, state_steps]  # type: ignore[list-item]
422      )
423      for (
424          grouped_params_,
425          grouped_grads_,
426          grouped_exp_avgs_,
427          grouped_exp_avg_sqs_,
428          grouped_mu_products_,
429          grouped_state_steps_,
430      ), _ in grouped_tensors.values():
431          grouped_params = cast(list[Tensor], grouped_params_)
432          grouped_grads = cast(list[Tensor], grouped_grads_)
433          grouped_exp_avgs = cast(list[Tensor], grouped_exp_avgs_)
434          grouped_exp_avg_sqs = cast(list[Tensor], grouped_exp_avg_sqs_)
435          grouped_mu_products = cast(list[Tensor], grouped_mu_products_)
436          grouped_state_steps = cast(list[Tensor], grouped_state_steps_)
437  
438          # handle complex
439          if has_complex:
440              _view_as_real(
441                  grouped_params, grouped_grads, grouped_exp_avgs, grouped_exp_avg_sqs
442              )
443  
444          if maximize:
445              grouped_grads = torch._foreach_neg(grouped_grads)  # type: ignore[assignment]
446  
447          # Update steps
448          # If steps are on CPU, foreach will fall back to the slow path, which is a for-loop calling t.add(1) over
449          # and over. 1 will then be wrapped into a Tensor over and over again, which is slower than if we just
450          # wrapped it once now. The alpha is required to assure we go to the right overload.
451          if not torch.compiler.is_compiling() and grouped_state_steps[0].is_cpu:
452              torch._foreach_add_(
453                  grouped_state_steps, torch.tensor(1.0, device="cpu"), alpha=1.0
454              )
455          else:
456              torch._foreach_add_(grouped_state_steps, 1)
457  
458          if weight_decay != 0:
459              if decoupled_weight_decay:
460                  # Perform stepweight decay
461                  torch._foreach_mul_(grouped_params, 1 - lr * weight_decay)
462              else:
463                  # Re-use the intermediate memory (grouped_grads) already allocated for maximize
464                  if maximize:
465                      torch._foreach_add_(
466                          grouped_grads, grouped_params, alpha=weight_decay
467                      )
468                  else:
469                      grouped_grads = torch._foreach_add(  # type: ignore[assignment]
470                          grouped_grads, grouped_params, alpha=weight_decay
471                      )
472  
473          # Decay the first and second moment running average coefficient
474          torch._foreach_lerp_(grouped_exp_avgs, grouped_grads, 1 - beta1)
475  
476          torch._foreach_mul_(grouped_exp_avg_sqs, beta2)
477          torch._foreach_addcmul_(
478              grouped_exp_avg_sqs, grouped_grads, grouped_grads, 1 - beta2
479          )
480  
481          exp_avg_sq_sqrt = torch._foreach_sqrt(grouped_exp_avg_sqs)
482  
483          bias_correction_sqrt: Union[tuple[Tensor, ...], list[Tensor]]
484          mus: Union[tuple[Tensor, ...], list[Tensor]]
485          mu_nexts: Union[tuple[Tensor, ...], list[Tensor]]
486          if capturable:
487              # mus will be beta1 * (1 - 0.5 * 0.96 ** (step * momentum_decay))
488              exponent = torch._foreach_mul(grouped_state_steps, momentum_decay)
489              mus = torch._foreach_pow(0.96, exponent)
490              torch._foreach_mul_(mus, -0.5)
491              torch._foreach_add_(mus, 1.0)
492              torch._foreach_mul_(mus, beta1)
493  
494              # mu_nexts will be beta1 * (1 - 0.5 * 0.96 ** ((step + 1) * momentum_decay))
495              torch._foreach_add_(exponent, momentum_decay)
496              mu_nexts = torch._foreach_pow(0.96, exponent)
497              torch._foreach_mul_(mu_nexts, -0.5)
498              torch._foreach_add_(mu_nexts, 1.0)
499              torch._foreach_mul_(mu_nexts, beta1)
500  
501              # save peak memory as we don't need exponent anymore
502              del exponent
503  
504              bias_correction_sqrt = torch._foreach_pow(beta2, grouped_state_steps)
505              # foreach_sub doesn't allow a scalar as the first arg
506              torch._foreach_sub_(bias_correction_sqrt, 1.0)
507              torch._foreach_neg_(bias_correction_sqrt)
508              torch._foreach_sqrt_(bias_correction_sqrt)
509          else:
510              bias_correction_sqrt = [
511                  (1 - beta2 ** _get_value(step)) ** 0.5 for step in grouped_state_steps
512              ]
513              mus = [
514                  beta1 * (1.0 - 0.5 * (0.96 ** (_get_value(step) * momentum_decay)))
515                  for step in grouped_state_steps
516              ]
517              mu_nexts = [
518                  beta1
519                  * (1.0 - 0.5 * (0.96 ** ((_get_value(step) + 1) * momentum_decay)))
520                  for step in grouped_state_steps
521              ]
522  
523          # update mu_products
524          torch._foreach_mul_(grouped_mu_products, mus)
525  
526          torch._foreach_div_(exp_avg_sq_sqrt, bias_correction_sqrt)
527          torch._foreach_add_(exp_avg_sq_sqrt, eps)
528  
529          # explicitly delete bias_correction refs to save memory
530          del bias_correction_sqrt
531  
532          if capturable:
533              # Build up the step_size multiplier for grad, reusing mus' memory
534              torch._foreach_sub_(mus, 1.0)
535              torch._foreach_mul_(mus, lr)
536              # foreach_sub doesn't allow a scalar as the first arg
537              denom = torch._foreach_sub(grouped_mu_products, 1.0)
538              torch._foreach_neg_(denom)
539              torch._foreach_div_(mus, denom)
540              # - lr * (1 - mu) / (1 - mu_product)
541              step_size_grads = mus
542              # explicitly delete denom to save memory
543              del denom
544  
545              # Build up the step_size multiplier for exp_avg, reusing mu_nexts' memory
546              denom = torch._foreach_mul(grouped_mu_products, mu_nexts)
547              torch._foreach_mul_(mu_nexts, lr)
548              # foreach_sub doesn't allow a scalar as the first arg, but it's okay because
549              # we need a negative here anyway
550              torch._foreach_sub_(denom, 1.0)
551              torch._foreach_div_(mu_nexts, denom)
552              # - lr * mu_next / (1 - mu_product * mu_next)
553              step_size_expavg = mu_nexts
554              # explicitly delete denom to save memory
555              del denom
556  
557              # we cannot inplace into step_size_grads cuz it is a list of ScalarTensors
558              # and mul'ing with grouped_grads will result in a list of bigger Tensors
559              numerator = torch._foreach_mul(step_size_grads, grouped_grads)
560              torch._foreach_addcmul_(numerator, step_size_expavg, grouped_exp_avgs)
561  
562              # finally, update params
563              torch._foreach_addcdiv_(grouped_params, numerator, exp_avg_sq_sqrt)
564          else:
565              step_size_grads = _stack_if_compiling(
566                  [
567                      (_get_value(lr) * (1.0 - mu) / (1.0 - _get_value(mu_product))) * -1
568                      for mu_product, mu in zip(grouped_mu_products, mus)
569                  ]
570              )
571              step_size_expavg = _stack_if_compiling(
572                  [
573                      (
574                          _get_value(lr)
575                          * mu_next
576                          / (1.0 - _get_value(mu_product) * mu_next)
577                      )
578                      * -1
579                      for mu_product, mu_next in zip(grouped_mu_products, mu_nexts)
580                  ]
581              )
582  
583              torch._foreach_addcdiv_(
584                  grouped_params,
585                  grouped_grads,
586                  exp_avg_sq_sqrt,
587                  step_size_grads,  # type: ignore[arg-type]
588              )
589              torch._foreach_addcdiv_(
590                  grouped_params,
591                  grouped_exp_avgs,
592                  exp_avg_sq_sqrt,
593                  step_size_expavg,  # type: ignore[arg-type]
594              )
