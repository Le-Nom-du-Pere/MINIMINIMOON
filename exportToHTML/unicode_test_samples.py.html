1    #!/usr/bin/env python3
2    """ 
3    Test script to demonstrate Unicode normalization effects on text samples. 
4    """
5    
6    import unicodedata
7    import sys
8    sys.path.append('.')
9    from feasibility_scorer import FeasibilityScorer
10   
11   def test_unicode_normalization():
12       """Test Unicode normalization on various text samples."""
13       scorer = FeasibilityScorer()
14       
15       # Test samples with various Unicode characters
16       test_samples = [
17           {
18               'text': 'Incrementar la línea base de 65％ a una "meta" de 85％',  # Full-width percent, smart quotes
19               'description': 'Full-width percent signs and smart quotes'
20           },
21           {
22               'text': 'Alcanzar objetivo de 1‚500 millones en año ２０２５',        # Different comma, full-width numbers  
23               'description': 'Different comma character and full-width numbers'
24           },
25           {
26               'text': 'baseline "50%" target "80%" by 2024',                    # Curly quotes
27               'description': 'Curly quotation marks'
28           },
29           {
30               'text': 'Horizonte 2020—2025 para meta',                         # Em dash
31               'description': 'Em dash instead of hyphen'
32           },
33           {
34               'text': 'Coeficiente línea base',                                # fi ligature (if present)
35               'description': 'Potential ligature characters'
36           }
37       ]
38       
39       print('UNICODE NORMALIZATION COMPARISON')
40       print('=' * 60)
41       
42       total_improvements = 0
43       
44       for i, sample in enumerate(test_samples, 1):
45           text = sample['text']
46           description = sample['description']
47           
48           print(f'\n{i}. Test: {description}')
49           print(f'   Original:   {repr(text)}')
50           
51           # Normalize the text
52           normalized = unicodedata.normalize('NFKC', text)
53           print(f'   Normalized: {repr(normalized)}')
54           print(f'   Changed: {text != normalized}')
55           
56           # Test regex matching before and after normalization
57           print(f'\n   Pattern Matching Comparison:')
58           
59           # Score original text (scorer will normalize it internally)
60           result_original = scorer.calculate_feasibility_score(text)
61           result_normalized = scorer.calculate_feasibility_score(normalized)
62           
63           components_original = len(result_original.components_detected)
64           components_normalized = len(result_normalized.components_detected)
65           
66           print(f'   Components detected (original):   {components_original}')
67           print(f'   Components detected (normalized): {components_normalized}')
68           print(f'   Score (original):   {result_original.feasibility_score:.3f}')
69           print(f'   Score (normalized): {result_normalized.feasibility_score:.3f}')
70           
71           if components_normalized >= components_original:
72               total_improvements += 1
73               print(f'   ✓ Normalization maintained or improved detection')
74           else:
75               print(f'   ✗ Normalization reduced detection')
76           
77           # Show detailed component matches
78           if result_original.detailed_matches:
79               print(f'   Detailed matches:')
80               for match in result_original.detailed_matches:
81                   print(f'     - {match.component_type.value}: "{match.matched_text}"')
82       
83       print(f'\n{"=" * 60}')
84       print(f'SUMMARY: {total_improvements}/{len(test_samples)} cases maintained/improved detection')
85       improvement_rate = total_improvements / len(test_samples) * 100
86       print(f'Improvement rate: {improvement_rate:.1f}%')
87   
88   if __name__ == '__main__':
89       test_unicode_normalization()
