1    """Utility to lazily import modules."""
2    
3    from __future__ import annotations
4    
5    import importlib
6    from typing import Any, TYPE_CHECKING
7    
8    
9    class _LazyModule:
10       """Lazily import a module."""
11   
12       def __init__(self, module_name: str) -> None:
13           self._name = module_name
14           self._module: Any = None
15   
16       def __repr__(self) -> str:
17           return f"<lazy module '{self._name}'>"
18   
19       def __getattr__(self, attr: str) -> object:
20           if self._module is None:
21               self._module = importlib.import_module(".", self._name)
22           return getattr(self._module, attr)
23
