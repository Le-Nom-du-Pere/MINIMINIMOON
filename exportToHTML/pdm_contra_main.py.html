147  class ContradictionDetector:
148      """ 
149      Hybrid contradiction detector for PDM analysis. 
150      Combines pattern matching, NLI, and policy validation. 
151      """
152  
153      def __init__(
154          self,
155          competence_matrix_path: Optional[Path] = None,
156          language: str = "es",
157          mode_light: bool = False,
158      ):
159          """ 
160          Initialize the contradiction detector. 
161   
162          Args: 
163              competence_matrix_path: Path to JSON file with competence matrix 
164              language: Language code (default: "es" for Spanish) 
165              mode_light: Use lightweight models if True 
166          """
167          self.language = language
168          self.mode_light = mode_light
169  
170          # Initialize components
171          self.pattern_matcher = PatternMatcher(language=language)
172          self.nli_detector = SpanishNLIDetector(light_mode=mode_light)
173          self.competence_validator = CompetenceValidator(
174              matrix_path=competence_matrix_path
175          )
176          self.risk_scorer = RiskScorer()
177          self.explainer = ExplanationTracer()
178  
179          # Load sentence encoder for semantic similarity
180          model_name = (
181              "sentence-transformers/all-MiniLM-L6-v2"
182              if mode_light
183              else "sentence-transformers/multilingual-e5-large"
184          )
185          self.encoder = SentenceTransformer(model_name)
186  
187          logger.info(
188              f"Initialized ContradictionDetector in {'light' if mode_light else 'full'} mode"
189          )

191      def detect_contradictions(
192          self,
193          text: str,
194          sectors: Optional[List[str]] = None,
195          pdm_structure: Optional[Dict[str, Any]] = None,
196      ) -> ContradictionAnalysis:
197          """ 
198          Detect contradictions in PDM text. 
199   
200          Args: 
201              text: Input PDM text 
202              sectors: List of sectors to analyze 
203              pdm_structure: Optional structured PDM data 
204   
205          Returns: 
206              ContradictionAnalysis with all findings 
207          """
208          # Normalize text
209          normalized_text = self._normalize_text(text)
210  
211          # Extract document segments
212          segments = self._segment_document(normalized_text, pdm_structure)
213  
214          contradictions = []
215          competence_issues = []
216          agenda_gaps = []
217  
218          # Analyze each segment
219          for seg_type, seg_text, seg_meta in segments:
220              # Pattern-based detection
221              pattern_matches = self.pattern_matcher.find_adversatives(seg_text)
222  
223              # NLI-based detection for semantic contradictions
224              if pdm_structure and seg_type in ["objetivos", "metas", "indicadores"]:
225                  nli_results = self._check_nli_contradictions(
226                      seg_text, pdm_structure, seg_type
227                  )
228                  pattern_matches.extend(nli_results)
229  
230              # Competence validation
231              if sectors and seg_type in ["programas", "acciones"]:
232                  comp_results = self.competence_validator.validate_segment(
233                      seg_text, sectors, seg_meta.get("nivel", "municipal")
234                  )
235                  competence_issues.extend(comp_results)
236  
237              # Agenda-setting alignment check
238              if pdm_structure:
239                  agenda_issues = self._check_agenda_alignment(
240                      seg_text, seg_type, pdm_structure
241                  )
242                  agenda_gaps.extend(agenda_issues)
243  
244              contradictions.extend(pattern_matches)
245  
246          # Calculate risk scores with conformal prediction
247          risk_analysis = self.risk_scorer.calculate_risk(
248              contradictions, competence_issues, agenda_gaps
249          )
250  
251          # Generate explanations
252          explanations = self.explainer.generate_explanations(
253              contradictions, competence_issues, agenda_gaps
254          )
255  
256          return ContradictionAnalysis(
257              contradictions=contradictions,
258              competence_mismatches=competence_issues,
259              agenda_gaps=agenda_gaps,
260              total_contradictions=len(contradictions),
261              total_competence_issues=len(competence_issues),
262              total_agenda_gaps=len(agenda_gaps),
263              risk_score=risk_analysis["overall_risk"],
264              risk_level=risk_analysis["risk_level"],
265              confidence_intervals=risk_analysis["confidence_intervals"],
266              explanations=explanations,
267              calibration_info={
268                  "method": "conformal_prediction",
269                  "alpha": 0.1,
270                  "coverage": risk_analysis.get("empirical_coverage", 0.9),
271              },
272          )

274      @staticmethod
275      def _normalize_text(text: str) -> str:
276          """Normalize text using NFKC and clean whitespace."""
277          if not text:
278              return ""
279          normalized = unicodedata.normalize("NFKC", text)
280          normalized = re.sub(r"\s+", " ", normalized).strip()
281          return normalized

283      @staticmethod
284      def _segment_document(
285          text: str, structure: Optional[Dict[str, Any]]
286      ) -> List[Tuple[str, str, Dict]]:
287          """ 
288          Segment PDM document into analyzable chunks. 
289   
290          Returns: 
291              List of (segment_type, text, metadata) tuples 
292          """
293          segments = []
294  
295          # Common PDM section patterns
296          section_patterns = {
297              "diagnostico": r"(?i)(diagnóstico|análisis\s+situacional|contexto)",
298              "objetivos": r"(?i)(objetivos?\s+estratégicos?|objetivos?\s+generales?)",
299              "metas": r"(?i)(metas?\s+de?\s+resultado|metas?\s+de?\s+producto)",
300              "indicadores": r"(?i)(indicadores?\s+de?\s+gestión|indicadores?\s+de?\s+impacto)",
301              "programas": r"(?i)(programas?\s+y?\s+proyectos?|líneas?\s+estratégicas?)",
302              "presupuesto": r"(?i)(presupuesto|plan\s+plurianual|recursos)",
303          }
304  
305          # Try to extract sections
306          current_pos = 0
307          for sec_type, pattern in section_patterns.items():
308              matches = list(re.finditer(pattern, text))
309              for match in matches:
310                  # Extract text until next section or 2000 chars
311                  start = match.end()
312                  end = min(start + 2000, len(text))
313  
314                  # Find next section boundary
315                  next_sections = []
316                  for other_pattern in section_patterns.values():
317                      next_match = re.search(other_pattern, text[start:end])
318                      if next_match:
319                          next_sections.append(start + next_match.start())
320  
321                  if next_sections:
322                      end = min(next_sections)
323  
324                  segment_text = text[start:end].strip()
325                  if segment_text:
326                      segments.append(
327                          (
328                              sec_type,
329                              segment_text,
330                              {"start": start, "end": end, "header": match.group()},
331                          )
332                      )
333  
334          # If no sections found, treat as single segment
335          if not segments:
336              segments.append(("general", text, {"start": 0, "end": len(text)}))
337  
338          return segments

340      def _check_nli_contradictions(
341          self, text: str, structure: Dict[str, Any], segment_type: str
342      ) -> List[ContradictionMatch]:
343          """ 
344          Check for semantic contradictions using NLI. 
345   
346          Args: 
347              text: Segment text 
348              structure: PDM structure data 
349              segment_type: Type of segment being analyzed 
350   
351          Returns: 
352              List of contradiction matches found via NLI 
353          """
354          nli_matches = []
355  
356          # Extract comparison pairs based on segment type
357          pairs = []
358          if segment_type == "objetivos" and "metas" in structure:
359              for objetivo in self._extract_items(text, "objetivo"):
360                  for meta in structure.get("metas", []):
361                      pairs.append((objetivo, meta, "objetivo-meta"))
362  
363          elif segment_type == "metas" and "indicadores" in structure:
364              for meta in self._extract_items(text, "meta"):
365                  for indicador in structure.get("indicadores", []):
366                      pairs.append((meta, indicador, "meta-indicador"))
367  
368          # Run NLI on pairs
369          for premise, hypothesis, pair_type in pairs:
370              result = self.nli_detector.check_contradiction(premise, hypothesis)
371  
372              if result["label"] == "contradiction" and result["score"] > 0.7:
373                  match = ContradictionMatch(
374                      type="semantic_contradiction",
375                      pair_type=pair_type,
376                      premise=premise[:200],
377                      hypothesis=hypothesis[:200],
378                      nli_score=result["score"],
379                      nli_label=result["label"],
380                      confidence=result["score"],
381                      risk_level=self._score_to_risk(result["score"]),
382                      explanation=f"Contradicción semántica detectada entre {pair_type}",
383                  )
384                  nli_matches.append(match)
385  
386          return nli_matches

388      def _check_agenda_alignment(
389          self, text: str, segment_type: str, structure: Dict[str, Any]
390      ) -> List[Dict[str, Any]]:
391          """ 
392          Check alignment in the agenda-setting chain. 
393   
394          Returns: 
395              List of agenda alignment issues 
396          """
397          issues = []
398  
399          # Check for missing links in chain
400          chain = [
401              "diagnostico",
402              "objetivos",
403              "estrategias",
404              "metas",
405              "indicadores",
406              "presupuesto",
407          ]
408  
409          if segment_type in chain:
410              idx = chain.index(segment_type)
411  
412              # Check backward alignment
413              if idx > 0:
414                  prev_type = chain[idx - 1]
415                  if prev_type in structure and not self._has_alignment(
416                      text, structure[prev_type]
417                  ):
418                      issues.append(
419                          {
420                              "type": "missing_backward_alignment",
421                              "from": segment_type,
422                              "to": prev_type,
423                              "severity": "medium",
424                              "explanation": f"Falta alineación entre {segment_type} y {prev_type}",
425                          }
426                      )
427  
428              # Check forward alignment
429              if idx < len(chain) - 1:
430                  next_type = chain[idx + 1]
431                  if next_type not in structure or not structure[next_type]:
432                      issues.append(
433                          {
434                              "type": "missing_forward_element",
435                              "from": segment_type,
436                              "expected": next_type,
437                              "severity": "high",
438                              "explanation": f"No se encontró {next_type} para {segment_type}",
439                          }
440                      )
441  
442          return issues

444      @staticmethod
445      def _extract_items(text: str, item_type: str) -> List[str]:
446          """Extract items of a specific type from text."""
447          items = []
448  
449          patterns = {
450              "objetivo": r"objetivo\s*\d*[:.]?\s*([^.]+\.)",
451              "meta": r"meta\s*\d*[:.]?\s*([^.]+\.)",
452              "indicador": r"indicador\s*[:.]?\s*([^.]+\.)",
453          }
454  
455          if item_type in patterns:
456              matches = re.finditer(patterns[item_type], text, re.IGNORECASE)
457              items = [match.group(1).strip() for match in matches]
458  
459          return items

461      def _has_alignment(self, text1: str, text2_items: List[str]) -> bool:
462          """Check if text1 has semantic alignment with items in text2."""
463          if not text2_items:
464              return False
465  
466          # Encode texts
467          emb1 = self.encoder.encode(text1[:500], convert_to_tensor=True)
468          emb2 = self.encoder.encode(text2_items[:5], convert_to_tensor=True)
469  
470          # Calculate similarities
471          similarities = util.pytorch_cos_sim(emb1, emb2)
472          max_sim = torch.max(similarities).item()
473  
474          return max_sim > 0.6  # Threshold for considering alignment
475  
476      @staticmethod
477      def _score_to_risk(score: float) -> RiskLevel:
478          """Convert confidence score to risk level."""
479          if score >= 0.8:
480              return RiskLevel.HIGH
481          elif score >= 0.6:
482              return RiskLevel.MEDIUM_HIGH
483          elif score >= 0.4:
484              return RiskLevel.MEDIUM
485          else:
486              return RiskLevel.LOW
