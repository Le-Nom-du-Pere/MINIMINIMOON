15   def itermonomials(variables, max_degrees, min_degrees=None):
16       r""" 
17       ``max_degrees`` and ``min_degrees`` are either both integers or both lists. 
18       Unless otherwise specified, ``min_degrees`` is either ``0`` or 
19       ``[0, ..., 0]``. 
20    
21       A generator of all monomials ``monom`` is returned, such that 
22       either 
23       ``min_degree <= total_degree(monom) <= max_degree``, 
24       or 
25       ``min_degrees[i] <= degree_list(monom)[i] <= max_degrees[i]``, 
26       for all ``i``. 
27    
28       Case I. ``max_degrees`` and ``min_degrees`` are both integers 
29       ============================================================= 
30    
31       Given a set of variables $V$ and a min_degree $N$ and a max_degree $M$ 
32       generate a set of monomials of degree less than or equal to $N$ and greater 
33       than or equal to $M$. The total number of monomials in commutative 
34       variables is huge and is given by the following formula if $M = 0$: 
35    
36           .. math:: 
37               \frac{(\#V + N)!}{\#V! N!} 
38    
39       For example if we would like to generate a dense polynomial of 
40       a total degree $N = 50$ and $M = 0$, which is the worst case, in 5 
41       variables, assuming that exponents and all of coefficients are 32-bit long 
42       and stored in an array we would need almost 80 GiB of memory! Fortunately 
43       most polynomials, that we will encounter, are sparse. 
44    
45       Consider monomials in commutative variables $x$ and $y$ 
46       and non-commutative variables $a$ and $b$:: 
47    
48           >>> from sympy import symbols 
49           >>> from sympy.polys.monomials import itermonomials 
50           >>> from sympy.polys.orderings import monomial_key 
51           >>> from sympy.abc import x, y 
52    
53           >>> sorted(itermonomials([x, y], 2), key=monomial_key('grlex', [y, x])) 
54           [1, x, y, x**2, x*y, y**2] 
55    
56           >>> sorted(itermonomials([x, y], 3), key=monomial_key('grlex', [y, x])) 
57           [1, x, y, x**2, x*y, y**2, x**3, x**2*y, x*y**2, y**3] 
58    
59           >>> a, b = symbols('a, b', commutative=False) 
60           >>> set(itermonomials([a, b, x], 2)) 
61           {1, a, a**2, b, b**2, x, x**2, a*b, b*a, x*a, x*b} 
62    
63           >>> sorted(itermonomials([x, y], 2, 1), key=monomial_key('grlex', [y, x])) 
64           [x, y, x**2, x*y, y**2] 
65    
66       Case II. ``max_degrees`` and ``min_degrees`` are both lists 
67       =========================================================== 
68    
69       If ``max_degrees = [d_1, ..., d_n]`` and 
70       ``min_degrees = [e_1, ..., e_n]``, the number of monomials generated 
71       is: 
72    
73       .. math:: 
74           (d_1 - e_1 + 1) (d_2 - e_2 + 1) \cdots (d_n - e_n + 1) 
75    
76       Let us generate all monomials ``monom`` in variables $x$ and $y$ 
77       such that ``[1, 2][i] <= degree_list(monom)[i] <= [2, 4][i]``, 
78       ``i = 0, 1`` :: 
79    
80           >>> from sympy import symbols 
81           >>> from sympy.polys.monomials import itermonomials 
82           >>> from sympy.polys.orderings import monomial_key 
83           >>> from sympy.abc import x, y 
84    
85           >>> sorted(itermonomials([x, y], [2, 4], [1, 2]), reverse=True, key=monomial_key('lex', [x, y])) 
86           [x**2*y**4, x**2*y**3, x**2*y**2, x*y**4, x*y**3, x*y**2] 
87       """
88       if is_sequence(max_degrees):
89           n = len(variables)
90           if len(max_degrees) != n:
91               raise ValueError('Argument sizes do not match')
92           if min_degrees is None:
93               min_degrees = [0]*n
94           elif not is_sequence(min_degrees):
95               raise ValueError('min_degrees is not a list')
96           else:
97               if len(min_degrees) != n:
98                   raise ValueError('Argument sizes do not match')
99               if any(i < 0 for i in min_degrees):
100                  raise ValueError("min_degrees cannot contain negative numbers")
101          if any(min_degrees[i] > max_degrees[i] for i in range(n)):
102              raise ValueError('min_degrees[i] must be <= max_degrees[i] for all i')
103          power_lists = []
104          for var, min_d, max_d in zip(variables, min_degrees, max_degrees):
105              power_lists.append([var**i for i in range(min_d, max_d + 1)])
106          for powers in product(*power_lists):
107              yield Mul(*powers)
108      else:
109          max_degree = max_degrees
110          if max_degree < 0:
111              raise ValueError("max_degrees cannot be negative")
112          if min_degrees is None:
113              min_degree = 0
114          else:
115              if min_degrees < 0:
116                  raise ValueError("min_degrees cannot be negative")
117              min_degree = min_degrees
118          if min_degree > max_degree:
119              return
120          if not variables or max_degree == 0:
121              yield S.One
122              return
123          # Force to list in case of passed tuple or other incompatible collection
124          variables = list(variables) + [S.One]
125          if all(variable.is_commutative for variable in variables):
126              it = combinations_with_replacement(variables, max_degree)
127          else:
128              it = product(variables, repeat=max_degree)
129          monomials_set = set()
130          d = max_degree - min_degree
131          for item in it:
132              count = 0
133              for variable in item:
134                  if variable == 1:
135                      count += 1
136                      if d < count:
137                          break
138              else:
139                  monomials_set.add(Mul(*item))
140          yield from monomials_set

142  def monomial_count(V, N):
143      r""" 
144      Computes the number of monomials. 
145   
146      The number of monomials is given by the following formula: 
147   
148      .. math:: 
149   
150          \frac{(\#V + N)!}{\#V! N!} 
151   
152      where `N` is a total degree and `V` is a set of variables. 
153   
154      Examples 
155      ======== 
156   
157      >>> from sympy.polys.monomials import itermonomials, monomial_count 
158      >>> from sympy.polys.orderings import monomial_key 
159      >>> from sympy.abc import x, y 
160   
161      >>> monomial_count(2, 2) 
162      6 
163   
164      >>> M = list(itermonomials([x, y], 2)) 
165   
166      >>> sorted(M, key=monomial_key('grlex', [y, x])) 
167      [1, x, y, x**2, x*y, y**2] 
168      >>> len(M) 
169      6 
170   
171      """
172      from sympy.functions.combinatorial.factorials import factorial
173      return factorial(V + N) / factorial(V) / factorial(N)

175  def monomial_mul(A, B):
176      """ 
177      Multiplication of tuples representing monomials. 
178   
179      Examples 
180      ======== 
181   
182      Lets multiply `x**3*y**4*z` with `x*y**2`:: 
183   
184          >>> from sympy.polys.monomials import monomial_mul 
185   
186          >>> monomial_mul((3, 4, 1), (1, 2, 0)) 
187          (4, 6, 1) 
188   
189      which gives `x**4*y**5*z`. 
190   
191      """
192      return tuple([ a + b for a, b in zip(A, B) ])

194  def monomial_div(A, B):
195      """ 
196      Division of tuples representing monomials. 
197   
198      Examples 
199      ======== 
200   
201      Lets divide `x**3*y**4*z` by `x*y**2`:: 
202   
203          >>> from sympy.polys.monomials import monomial_div 
204   
205          >>> monomial_div((3, 4, 1), (1, 2, 0)) 
206          (2, 2, 1) 
207   
208      which gives `x**2*y**2*z`. However:: 
209   
210          >>> monomial_div((3, 4, 1), (1, 2, 2)) is None 
211          True 
212   
213      `x*y**2*z**2` does not divide `x**3*y**4*z`. 
214   
215      """
216      C = monomial_ldiv(A, B)
217  
218      if all(c >= 0 for c in C):
219          return tuple(C)
220      else:
221          return None

223  def monomial_ldiv(A, B):
224      """ 
225      Division of tuples representing monomials. 
226   
227      Examples 
228      ======== 
229   
230      Lets divide `x**3*y**4*z` by `x*y**2`:: 
231   
232          >>> from sympy.polys.monomials import monomial_ldiv

240          (2, 2, -1)

242      which gives `x**2*y**2*z**-1`.

252      """ 
253      Greatest common divisor of tuples representing monomials. 
254   
255      Examples 
256      ======== 
257   
258      Lets compute GCD of `x*y**4*z` and `x**3*y**2`:: 
259   
260          >>> from sympy.polys.monomials import monomial_gcd

263          (1, 2, 0) 
264   
265      which gives `x*y**2`. 
266   
267      """
268      return tuple([ min(a, b) for a, b in zip(A, B) ])

272      Least common multiple of tuples representing monomials.

289  def monomial_divides(A, B):
290      """ 
291      Does there exist a monomial X such that XA == B? 
292   
293      Examples 
294      ======== 
295   
296      >>> from sympy.polys.monomials import monomial_divides 
297      >>> monomial_divides((1, 2), (3, 4)) 
298      True 
299      >>> monomial_divides((1, 2), (0, 2)) 
300      False 
301      """
302      return all(a <= b for a, b in zip(A, B))

304  def monomial_max(*monoms):
305      """ 
306      Returns maximal degree for each variable in a set of monomials. 
307   
308      Examples 
309      ======== 
310   
311      Consider monomials `x**3*y**4*z**5`, `y**5*z` and `x**6*y**3*z**9`. 
312      We wish to find out what is the maximal degree for each of `x`, `y` 
313      and `z` variables:: 
314   
315          >>> from sympy.polys.monomials import monomial_max 
316   
317          >>> monomial_max((3,4,5), (0,5,1), (6,3,9)) 
318          (6, 5, 9) 
319   
320      """
321      M = list(monoms[0])
322  
323      for N in monoms[1:]:
324          for i, n in enumerate(N):
325              M[i] = max(M[i], n)
326  
327      return tuple(M)

331      Returns minimal degree for each variable in a set of monomials. 
332   
333      Examples 
334      ======== 
335   
336      Consider monomials `x**3*y**4*z**5`, `y**5*z` and `x**6*y**3*z**9`. 
337      We wish to find out what is the minimal degree for each of `x`, `y`

341   
342          >>> monomial_min((3,4,5), (0,5,1), (6,3,9)) 
343          (0, 3, 1) 
344   
345      """
346      M = list(monoms[0])

348      for N in monoms[1:]:
349          for i, n in enumerate(N):

369  def term_div(a, b, domain):
370      """Division of two terms in over a ring/field. """
371      a_lm, a_lc = a
372      b_lm, b_lc = b
373  
374      monom = monomial_div(a_lm, b_lm)
375  
376      if domain.is_Field:
377          if monom is not None:
378              return monom, domain.quo(a_lc, b_lc)
379          else:
380              return None
381      else:
382          if not (monom is None or a_lc % b_lc):
383              return monom, domain.quo(a_lc, b_lc)
384          else:
385              return None

387  class MonomialOps:
388      """Code generator of fast monomial arithmetic functions. """
389  
390      @cacheit
391      def __new__(cls, ngens):
392          obj = super().__new__(cls)
393          obj.ngens = ngens
394          return obj
395  
396      def __getnewargs__(self):
397          return (self.ngens,)
398  
399      def _build(self, code, name):
400          ns = {}
401          exec(code, ns)
402          return ns[name]
403  
404      def _vars(self, name):
405          return [ "%s%s" % (name, i) for i in range(self.ngens) ]
406  
407      @cacheit
408      def mul(self):
409          name = "monomial_mul"
410          template = dedent("""\ 
411          def %(name)s(A, B): 
412              (%(A)s,) = A 
413              (%(B)s,) = B 
414              return (%(AB)s,) 
415          """)
416          A = self._vars("a")
417          B = self._vars("b")
418          AB = [ "%s + %s" % (a, b) for a, b in zip(A, B) ]
419          code = template % {"name": name, "A": ", ".join(A), "B": ", ".join(B), "AB": ", ".join(AB)}
420          return self._build(code, name)
421  
422      @cacheit
423      def pow(self):
424          name = "monomial_pow"
425          template = dedent("""\ 
426          def %(name)s(A, k): 
427              (%(A)s,) = A 
428              return (%(Ak)s,) 
429          """)
430          A = self._vars("a")
431          Ak = [ "%s*k" % a for a in A ]
432          code = template % {"name": name, "A": ", ".join(A), "Ak": ", ".join(Ak)}
433          return self._build(code, name)
434  
435      @cacheit
436      def mulpow(self):
437          name = "monomial_mulpow"
438          template = dedent("""\ 
439          def %(name)s(A, B, k): 
440              (%(A)s,) = A 
441              (%(B)s,) = B 
442              return (%(ABk)s,) 
443          """)
444          A = self._vars("a")
445          B = self._vars("b")
446          ABk = [ "%s + %s*k" % (a, b) for a, b in zip(A, B) ]
447          code = template % {"name": name, "A": ", ".join(A), "B": ", ".join(B), "ABk": ", ".join(ABk)}
448          return self._build(code, name)
449  
450      @cacheit
451      def ldiv(self):
452          name = "monomial_ldiv"
453          template = dedent("""\ 
454          def %(name)s(A, B): 
455              (%(A)s,) = A 
456              (%(B)s,) = B 
457              return (%(AB)s,) 
458          """)
459          A = self._vars("a")
460          B = self._vars("b")
461          AB = [ "%s - %s" % (a, b) for a, b in zip(A, B) ]
462          code = template % {"name": name, "A": ", ".join(A), "B": ", ".join(B), "AB": ", ".join(AB)}
463          return self._build(code, name)
464  
465      @cacheit
466      def div(self):
467          name = "monomial_div"
468          template = dedent("""\ 
469          def %(name)s(A, B): 
470              (%(A)s,) = A 
471              (%(B)s,) = B 
472              %(RAB)s 
473              return (%(R)s,) 
474          """)
475          A = self._vars("a")
476          B = self._vars("b")
477          RAB = [ "r%(i)s = a%(i)s - b%(i)s\n    if r%(i)s < 0: return None" % {"i": i} for i in range(self.ngens) ]
478          R = self._vars("r")
479          code = template % {"name": name, "A": ", ".join(A), "B": ", ".join(B), "RAB": "\n    ".join(RAB), "R": ", ".join(R)}
480          return self._build(code, name)
481  
482      @cacheit
483      def lcm(self):
484          name = "monomial_lcm"
485          template = dedent("""\ 
486          def %(name)s(A, B): 
487              (%(A)s,) = A 
488              (%(B)s,) = B 
489              return (%(AB)s,) 
490          """)
491          A = self._vars("a")
492          B = self._vars("b")
493          AB = [ "%s if %s >= %s else %s" % (a, a, b, b) for a, b in zip(A, B) ]
494          code = template % {"name": name, "A": ", ".join(A), "B": ", ".join(B), "AB": ", ".join(AB)}
495          return self._build(code, name)
496  
497      @cacheit
498      def gcd(self):
499          name = "monomial_gcd"
500          template = dedent("""\ 
501          def %(name)s(A, B): 
502              (%(A)s,) = A 
503              (%(B)s,) = B 
504              return (%(AB)s,) 
505          """)
506          A = self._vars("a")
507          B = self._vars("b")
508          AB = [ "%s if %s <= %s else %s" % (a, a, b, b) for a, b in zip(A, B) ]
509          code = template % {"name": name, "A": ", ".join(A), "B": ", ".join(B), "AB": ", ".join(AB)}
510          return self._build(code, name)

512  @public
513  class Monomial(PicklableWithSlots):
514      """Class representing a monomial, i.e. a product of powers. """
515  
516      __slots__ = ('exponents', 'gens')
517  
518      def __init__(self, monom, gens=None):
519          if not iterable(monom):
520              rep, gens = dict_from_expr(sympify(monom), gens=gens)
521              if len(rep) == 1 and list(rep.values())[0] == 1:
522                  monom = list(rep.keys())[0]
523              else:
524                  raise ValueError("Expected a monomial got {}".format(monom))
525  
526          self.exponents = tuple(map(int, monom))
527          self.gens = gens
528  
529      def rebuild(self, exponents, gens=None):
530          return self.__class__(exponents, gens or self.gens)
531  
532      def __len__(self):
533          return len(self.exponents)
534  
535      def __iter__(self):
536          return iter(self.exponents)
537  
538      def __getitem__(self, item):
539          return self.exponents[item]
540  
541      def __hash__(self):
542          return hash((self.__class__.__name__, self.exponents, self.gens))
543  
544      def __str__(self):
545          if self.gens:
546              return "*".join([ "%s**%s" % (gen, exp) for gen, exp in zip(self.gens, self.exponents) ])
547          else:
548              return "%s(%s)" % (self.__class__.__name__, self.exponents)
549  
550      def as_expr(self, *gens):
551          """Convert a monomial instance to a SymPy expression. """
552          gens = gens or self.gens
553  
554          if not gens:
555              raise ValueError(
556                  "Cannot convert %s to an expression without generators" % self)
557  
558          return Mul(*[ gen**exp for gen, exp in zip(gens, self.exponents) ])
559  
560      def __eq__(self, other):
561          if isinstance(other, Monomial):
562              exponents = other.exponents
563          elif isinstance(other, (tuple, Tuple)):
564              exponents = other
565          else:
566              return False
567  
568          return self.exponents == exponents
569  
570      def __ne__(self, other):
571          return not self == other
572  
573      def __mul__(self, other):
574          if isinstance(other, Monomial):
575              exponents = other.exponents
576          elif isinstance(other, (tuple, Tuple)):
577              exponents = other
578          else:
579              raise NotImplementedError
580  
581          return self.rebuild(monomial_mul(self.exponents, exponents))
582  
583      def __truediv__(self, other):
584          if isinstance(other, Monomial):
585              exponents = other.exponents
586          elif isinstance(other, (tuple, Tuple)):
587              exponents = other
588          else:
589              raise NotImplementedError
590  
591          result = monomial_div(self.exponents, exponents)
592  
593          if result is not None:
594              return self.rebuild(result)
595          else:
596              raise ExactQuotientFailed(self, Monomial(other))
597  
598      __floordiv__ = __truediv__
599  
600      def __pow__(self, other):
601          n = int(other)
602          if n < 0:
603              raise ValueError("a non-negative integer expected, got %s" % other)
604          return self.rebuild(monomial_pow(self.exponents, n))
605  
606      def gcd(self, other):
607          """Greatest common divisor of monomials. """
608          if isinstance(other, Monomial):
609              exponents = other.exponents
610          elif isinstance(other, (tuple, Tuple)):
611              exponents = other
612          else:
613              raise TypeError(
614                  "an instance of Monomial class expected, got %s" % other)
615  
616          return self.rebuild(monomial_gcd(self.exponents, exponents))
617  
618      def lcm(self, other):
619          """Least common multiple of monomials. """
620          if isinstance(other, Monomial):
621              exponents = other.exponents
622          elif isinstance(other, (tuple, Tuple)):
623              exponents = other
624          else:
625              raise TypeError(
626                  "an instance of Monomial class expected, got %s" % other)
627  
628          return self.rebuild(monomial_lcm(self.exponents, exponents))
