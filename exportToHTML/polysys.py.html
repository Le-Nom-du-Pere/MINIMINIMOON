39   def solve_poly_system(seq, *gens, strict=False, **args):
40       """ 
41       Return a list of solutions for the system of polynomial equations 
42       or else None. 
43    
44       Parameters 
45       ========== 
46    
47       seq: a list/tuple/set 
48           Listing all the equations that are needed to be solved 
49       gens: generators 
50           generators of the equations in seq for which we want the 
51           solutions 
52       strict: a boolean (default is False) 
53           if strict is True, NotImplementedError will be raised if 
54           the solution is known to be incomplete (which can occur if 
55           not all solutions are expressible in radicals) 
56       args: Keyword arguments 
57           Special options for solving the equations. 
58    
59    
60       Returns 
61       ======= 
62    
63       List[Tuple] 
64           a list of tuples with elements being solutions for the 
65           symbols in the order they were passed as gens 
66       None 
67           None is returned when the computed basis contains only the ground. 
68    
69       Examples 
70       ======== 
71    
72       >>> from sympy import solve_poly_system 
73       >>> from sympy.abc import x, y 
74    
75       >>> solve_poly_system([x*y - 2*y, 2*y**2 - x**2], x, y) 
76       [(0, 0), (2, -sqrt(2)), (2, sqrt(2))] 
77    
78       >>> solve_poly_system([x**5 - x + y**3, y**2 - 1], x, y, strict=True) 
79       Traceback (most recent call last): 
80       ... 
81       UnsolvableFactorError 
82    
83       """
84       try:
85           polys, opt = parallel_poly_from_expr(seq, *gens, **args)
86       except PolificationFailed as exc:
87           raise ComputationFailed('solve_poly_system', len(seq), exc)
88   
89       if len(polys) == len(opt.gens) == 2:
90           f, g = polys
91   
92           if all(i <= 2 for i in f.degree_list() + g.degree_list()):
93               try:
94                   return solve_biquadratic(f, g, opt)
95               except SolveFailed:
96                   pass
97   
98       return solve_generic(polys, opt, strict=strict)

101  def solve_biquadratic(f, g, opt):
102      """Solve a system of two bivariate quadratic polynomial equations. 
103   
104      Parameters 
105      ========== 
106   
107      f: a single Expr or Poly 
108          First equation 
109      g: a single Expr or Poly 
110          Second Equation 
111      opt: an Options object 
112          For specifying keyword arguments and generators 
113   
114      Returns 
115      ======= 
116   
117      List[Tuple] 
118          a list of tuples with elements being solutions for the 
119          symbols in the order they were passed as gens 
120      None 
121          None is returned when the computed basis contains only the ground. 
122   
123      Examples 
124      ======== 
125   
126      >>> from sympy import Options, Poly 
127      >>> from sympy.abc import x, y 
128      >>> from sympy.solvers.polysys import solve_biquadratic 
129      >>> NewOption = Options((x, y), {'domain': 'ZZ'}) 
130   
131      >>> a = Poly(y**2 - 4 + x, y, x, domain='ZZ') 
132      >>> b = Poly(y*2 + 3*x - 7, y, x, domain='ZZ') 
133      >>> solve_biquadratic(a, b, NewOption) 
134      [(1/3, 3), (41/27, 11/9)] 
135   
136      >>> a = Poly(y + x**2 - 3, y, x, domain='ZZ') 
137      >>> b = Poly(-y + x - 4, y, x, domain='ZZ') 
138      >>> solve_biquadratic(a, b, NewOption) 
139      [(7/2 - sqrt(29)/2, -sqrt(29)/2 - 1/2), (sqrt(29)/2 + 7/2, -1/2 + \ 
140        sqrt(29)/2)] 
141      """
142      G = groebner([f, g])
143  
144      if len(G) == 1 and G[0].is_ground:
145          return None
146  
147      if len(G) != 2:
148          raise SolveFailed
149  
150      x, y = opt.gens
151      p, q = G
152      if not p.gcd(q).is_ground:
153          # not 0-dimensional
154          raise SolveFailed
155  
156      p = Poly(p, x, expand=False)
157      p_roots = [rcollect(expr, y) for expr in roots(p).keys()]
158  
159      q = q.ltrim(-1)
160      q_roots = list(roots(q).keys())
161  
162      solutions = [(p_root.subs(y, q_root), q_root) for q_root, p_root in
163                   itertools.product(q_roots, p_roots)]
164  
165      return sorted(solutions, key=default_sort_key)

168  def solve_generic(polys, opt, strict=False):
169      """ 
170      Solve a generic system of polynomial equations. 
171   
172      Returns all possible solutions over C[x_1, x_2, ..., x_m] of a 
173      set F = { f_1, f_2, ..., f_n } of polynomial equations, using 
174      Groebner basis approach. For now only zero-dimensional systems 
175      are supported, which means F can have at most a finite number 
176      of solutions. If the basis contains only the ground, None is 
177      returned. 
178   
179      The algorithm works by the fact that, supposing G is the basis 
180      of F with respect to an elimination order (here lexicographic 
181      order is used), G and F generate the same ideal, they have the 
182      same set of solutions. By the elimination property, if G is a 
183      reduced, zero-dimensional Groebner basis, then there exists an 
184      univariate polynomial in G (in its last variable). This can be 
185      solved by computing its roots. Substituting all computed roots 
186      for the last (eliminated) variable in other elements of G, new 
187      polynomial system is generated. Applying the above procedure 
188      recursively, a finite number of solutions can be found. 
189   
190      The ability of finding all solutions by this procedure depends 
191      on the root finding algorithms. If no solutions were found, it 
192      means only that roots() failed, but the system is solvable. To 
193      overcome this difficulty use numerical algorithms instead. 
194   
195      Parameters 
196      ========== 
197   
198      polys: a list/tuple/set 
199          Listing all the polynomial equations that are needed to be solved 
200      opt: an Options object 
201          For specifying keyword arguments and generators 
202      strict: a boolean 
203          If strict is True, NotImplementedError will be raised if the solution 
204          is known to be incomplete 
205   
206      Returns 
207      ======= 
208   
209      List[Tuple] 
210          a list of tuples with elements being solutions for the 
211          symbols in the order they were passed as gens 
212      None 
213          None is returned when the computed basis contains only the ground. 
214   
215      References 
216      ========== 
217   
218      .. [Buchberger01] B. Buchberger, Groebner Bases: A Short 
219      Introduction for Systems Theorists, In: R. Moreno-Diaz, 
220      B. Buchberger, J.L. Freire, Proceedings of EUROCAST'01, 
221      February, 2001 
222   
223      .. [Cox97] D. Cox, J. Little, D. O'Shea, Ideals, Varieties 
224      and Algorithms, Springer, Second Edition, 1997, pp. 112 
225   
226      Raises 
227      ======== 
228   
229      NotImplementedError 
230          If the system is not zero-dimensional (does not have a finite 
231          number of solutions) 
232   
233      UnsolvableFactorError 
234          If ``strict`` is True and not all solution components are 
235          expressible in radicals 
236   
237      Examples 
238      ======== 
239   
240      >>> from sympy import Poly, Options 
241      >>> from sympy.solvers.polysys import solve_generic 
242      >>> from sympy.abc import x, y 
243      >>> NewOption = Options((x, y), {'domain': 'ZZ'}) 
244   
245      >>> a = Poly(x - y + 5, x, y, domain='ZZ') 
246      >>> b = Poly(x + y - 3, x, y, domain='ZZ') 
247      >>> solve_generic([a, b], NewOption) 
248      [(-1, 4)] 
249   
250      >>> a = Poly(x - 2*y + 5, x, y, domain='ZZ') 
251      >>> b = Poly(2*x - y - 3, x, y, domain='ZZ') 
252      >>> solve_generic([a, b], NewOption) 
253      [(11/3, 13/3)] 
254   
255      >>> a = Poly(x**2 + y, x, y, domain='ZZ') 
256      >>> b = Poly(x + y*4, x, y, domain='ZZ') 
257      >>> solve_generic([a, b], NewOption) 
258      [(0, 0), (1/4, -1/16)] 
259   
260      >>> a = Poly(x**5 - x + y**3, x, y, domain='ZZ') 
261      >>> b = Poly(y**2 - 1, x, y, domain='ZZ') 
262      >>> solve_generic([a, b], NewOption, strict=True) 
263      Traceback (most recent call last): 
264      ... 
265      UnsolvableFactorError 
266   
267      """
268      def _is_univariate(f):
269          """Returns True if 'f' is univariate in its last variable. """
270          for monom in f.monoms():
271              if any(monom[:-1]):
272                  return False
273  
274          return True
275  
276      def _subs_root(f, gen, zero):
277          """Replace generator with a root so that the result is nice. """
278          p = f.as_expr({gen: zero})
279  
280          if f.degree(gen) >= 2:
281              p = p.expand(deep=False)
282  
283          return p
284  
285      def _solve_reduced_system(system, gens, entry=False):
286          """Recursively solves reduced polynomial systems. """
287          if len(system) == len(gens) == 1:
288              # the below line will produce UnsolvableFactorError if
289              # strict=True and the solution from `roots` is incomplete
290              zeros = list(roots(system[0], gens[-1], strict=strict).keys())
291              return [(zero,) for zero in zeros]
292  
293          basis = groebner(system, gens, polys=True)
294  
295          if len(basis) == 1 and basis[0].is_ground:
296              if not entry:
297                  return []
298              else:
299                  return None
300  
301          univariate = list(filter(_is_univariate, basis))
302  
303          if len(basis) < len(gens):
304              raise NotImplementedError(filldedent(''' 
305                  only zero-dimensional systems supported 
306                  (finite number of solutions) 
307                  '''))
308  
309          if len(univariate) == 1:
310              f = univariate.pop()
311          else:
312              raise NotImplementedError(filldedent(''' 
313                  only zero-dimensional systems supported 
314                  (finite number of solutions) 
315                  '''))
316  
317          gens = f.gens
318          gen = gens[-1]
319  
320          # the below line will produce UnsolvableFactorError if
321          # strict=True and the solution from `roots` is incomplete
322          zeros = list(roots(f.ltrim(gen), strict=strict).keys())
323  
324          if not zeros:
325              return []
326  
327          if len(basis) == 1:
328              return [(zero,) for zero in zeros]
329  
330          solutions = []
331  
332          for zero in zeros:
333              new_system = []
334              new_gens = gens[:-1]
335  
336              for b in basis[:-1]:
337                  eq = _subs_root(b, gen, zero)
338  
339                  if eq is not S.Zero:
340                      new_system.append(eq)
341  
342              for solution in _solve_reduced_system(new_system, new_gens):
343                  solutions.append(solution + (zero,))
344  
345          if solutions and len(solutions[0]) != len(gens):
346              raise NotImplementedError(filldedent(''' 
347                  only zero-dimensional systems supported 
348                  (finite number of solutions) 
349                  '''))
350          return solutions
351  
352      try:
353          result = _solve_reduced_system(polys, opt.gens, entry=True)
354      except CoercionFailed:
355          raise NotImplementedError
356  
357      if result is not None:
358          return sorted(result, key=default_sort_key)

361  def solve_triangulated(polys, *gens, **args):
362      """ 
363      Solve a polynomial system using Gianni-Kalkbrenner algorithm. 
364   
365      The algorithm proceeds by computing one Groebner basis in the ground 
366      domain and then by iteratively computing polynomial factorizations in 
367      appropriately constructed algebraic extensions of the ground domain. 
368   
369      Parameters 
370      ========== 
371   
372      polys: a list/tuple/set 
373          Listing all the equations that are needed to be solved 
374      gens: generators 
375          generators of the equations in polys for which we want the 
376          solutions 
377      args: Keyword arguments 
378          Special options for solving the equations 
379   
380      Returns 
381      ======= 
382   
383      List[Tuple] 
384          A List of tuples. Solutions for symbols that satisfy the 
385          equations listed in polys 
386   
387      Examples 
388      ======== 
389   
390      >>> from sympy import solve_triangulated 
391      >>> from sympy.abc import x, y, z 
392   
393      >>> F = [x**2 + y + z - 1, x + y**2 + z - 1, x + y + z**2 - 1] 
394   
395      >>> solve_triangulated(F, x, y, z) 
396      [(0, 0, 1), (0, 1, 0), (1, 0, 0)] 
397   
398      Using extension for algebraic solutions. 
399   
400      >>> solve_triangulated(F, x, y, z, extension=True) #doctest: +NORMALIZE_WHITESPACE 
401      [(0, 0, 1), (0, 1, 0), (1, 0, 0), 
402       (CRootOf(x**2 + 2*x - 1, 0), CRootOf(x**2 + 2*x - 1, 0), CRootOf(x**2 + 2*x - 1, 0)), 
403       (CRootOf(x**2 + 2*x - 1, 1), CRootOf(x**2 + 2*x - 1, 1), CRootOf(x**2 + 2*x - 1, 1))] 
404   
405      References 
406      ========== 
407   
408      1. Patrizia Gianni, Teo Mora, Algebraic Solution of System of 
409      Polynomial Equations using Groebner Bases, AAECC-5 on Applied Algebra, 
410      Algebraic Algorithms and Error-Correcting Codes, LNCS 356 247--257, 1989 
411   
412      """
413      opt = build_options(gens, args)
414  
415      G = groebner(polys, gens, polys=True)
416      G = list(reversed(G))
417  
418      extension = opt.get('extension', False)
419      if extension:
420          def _solve_univariate(f):
421              return [r for r, _ in f.all_roots(multiple=False, radicals=False)]
422      else:
423          domain = opt.get('domain')
424  
425          if domain is not None:
426              for i, g in enumerate(G):
427                  G[i] = g.set_domain(domain)
428  
429          def _solve_univariate(f):
430              return list(f.ground_roots().keys())
431  
432      f, G = G[0].ltrim(-1), G[1:]
433      dom = f.get_domain()
434  
435      zeros = _solve_univariate(f)
436  
437      if extension:
438          solutions = {((zero,), dom.algebraic_field(zero)) for zero in zeros}
439      else:
440          solutions = {((zero,), dom) for zero in zeros}
441  
442      var_seq = reversed(gens[:-1])
443      vars_seq = postfixes(gens[1:])
444  
445      for var, vars in zip(var_seq, vars_seq):
446          _solutions = set()
447  
448          for values, dom in solutions:
449              H, mapping = [], list(zip(vars, values))
450  
451              for g in G:
452                  _vars = (var,) + vars
453  
454                  if g.has_only_gens(*_vars) and g.degree(var) != 0:
455                      if extension:
456                          g = g.set_domain(g.domain.unify(dom))
457                      h = g.ltrim(var).eval(dict(mapping))
458  
459                      if g.degree(var) == h.degree():
460                          H.append(h)
461  
462              p = min(H, key=lambda h: h.degree())
463              zeros = _solve_univariate(p)
464  
465              for zero in zeros:
466                  if not (zero in dom):
467                      dom_zero = dom.algebraic_field(zero)
468                  else:
469                      dom_zero = dom
470  
471                  _solutions.add(((zero,) + values, dom_zero))
472  
473          solutions = _solutions
474      return sorted((s for s, _ in solutions), key=default_sort_key)

477  def factor_system(eqs: Sequence[Expr | complex], gens: Sequence[Expr] = (), **kwargs: Any) -> list[list[Expr]]:
478      """ 
479      Factorizes a system of polynomial equations into 
480      irreducible subsystems. 
481   
482      Parameters 
483      ========== 
484   
485      eqs : list 
486          List of expressions to be factored. 
487          Each expression is assumed to be equal to zero. 
488   
489      gens : list, optional 
490          Generator(s) of the polynomial ring. 
491          If not provided, all free symbols will be used. 
492   
493      **kwargs : dict, optional 
494          Same optional arguments taken by ``factor`` 
495   
496      Returns 
497      ======= 
498   
499      list[list[Expr]] 
500          A list of lists of expressions, where each sublist represents 
501          an irreducible subsystem. When solved, each subsystem gives 
502          one component of the solution. Only generic solutions are 
503          returned (cases not requiring parameters to be zero). 
504   
505      Examples 
506      ======== 
507   
508      >>> from sympy.solvers.polysys import factor_system, factor_system_cond 
509      >>> from sympy.abc import x, y, a, b, c 
510   
511      A simple system with multiple solutions: 
512   
513      >>> factor_system([x**2 - 1, y - 1]) 
514      [[x + 1, y - 1], [x - 1, y - 1]] 
515   
516      A system with no solution: 
517   
518      >>> factor_system([x, 1]) 
519      [] 
520   
521      A system where any value of the symbol(s) is a solution: 
522   
523      >>> factor_system([x - x, (x + 1)**2 - (x**2 + 2*x + 1)]) 
524      [[]] 
525   
526      A system with no generic solution: 
527   
528      >>> factor_system([a*x*(x-1), b*y, c], [x, y]) 
529      [] 
530   
531      If c is added to the unknowns then the system has a generic solution: 
532   
533      >>> factor_system([a*x*(x-1), b*y, c], [x, y, c]) 
534      [[x - 1, y, c], [x, y, c]] 
535   
536      Alternatively :func:`factor_system_cond` can be used to get degenerate 
537      cases as well: 
538   
539      >>> factor_system_cond([a*x*(x-1), b*y, c], [x, y]) 
540      [[x - 1, y, c], [x, y, c], [x - 1, b, c], [x, b, c], [y, a, c], [a, b, c]] 
541   
542      Each of the above cases is only satisfiable in the degenerate case `c = 0`. 
543   
544      The solution set of the original system represented 
545      by eqs is the union of the solution sets of the 
546      factorized systems. 
547   
548      An empty list [] means no generic solution exists. 
549      A list containing an empty list [[]] means any value of 
550      the symbol(s) is a solution. 
551   
552      See Also 
553      ======== 
554   
555      factor_system_cond : Returns both generic and degenerate solutions 
556      factor_system_bool : Returns a Boolean combination representing all solutions 
557      sympy.polys.polytools.factor : Factors a polynomial into irreducible factors 
558                                     over the rational numbers 
559      """
560  
561      systems = _factor_system_poly_from_expr(eqs, gens, **kwargs)
562      systems_generic = [sys for sys in systems if not _is_degenerate(sys)]
563      systems_expr = [[p.as_expr() for p in system] for system in systems_generic]
564      return systems_expr
565

567  def _is_degenerate(system: list[Poly]) -> bool:
568      """Helper function to check if a system is degenerate"""
569      return any(p.is_ground for p in system)

572  def factor_system_bool(eqs: Sequence[Expr | complex], gens: Sequence[Expr] = (), **kwargs: Any) -> Boolean:
573      """ 
574      Factorizes a system of polynomial equations into irreducible DNF. 
575   
576      The system of expressions(eqs) is taken and a Boolean combination 
577      of equations is returned that represents the same solution set. 
578      The result is in disjunctive normal form (OR of ANDs). 
579   
580      Parameters 
581      ========== 
582   
583      eqs : list 
584         List of expressions to be factored. 
585         Each expression is assumed to be equal to zero. 
586   
587      gens : list, optional 
588         Generator(s) of the polynomial ring. 
589         If not provided, all free symbols will be used. 
590   
591      **kwargs : dict, optional 
592         Optional keyword arguments 
593   
594   
595      Returns 
596      ======= 
597   
598      Boolean: 
599         A Boolean combination of equations. The result is typically in 
600         the form of a conjunction (AND) of a disjunctive normal form 
601         with additional conditions. 
602   
603      Examples 
604      ======== 
605   
606      >>> from sympy.solvers.polysys import factor_system_bool 
607      >>> from sympy.abc import x, y, a, b, c 
608      >>> factor_system_bool([x**2 - 1]) 
609      Eq(x - 1, 0) | Eq(x + 1, 0) 
610   
611      >>> factor_system_bool([x**2 - 1, y - 1]) 
612      (Eq(x - 1, 0) & Eq(y - 1, 0)) | (Eq(x + 1, 0) & Eq(y - 1, 0)) 
613   
614      >>> eqs = [a * (x - 1), b] 
615      >>> factor_system_bool([a*(x - 1), b]) 
616      (Eq(a, 0) & Eq(b, 0)) | (Eq(b, 0) & Eq(x - 1, 0)) 
617   
618      >>> factor_system_bool([a*x**2 - a, b*(x + 1), c], [x]) 
619      (Eq(c, 0) & Eq(x + 1, 0)) | (Eq(a, 0) & Eq(b, 0) & Eq(c, 0)) | (Eq(b, 0) & Eq(c, 0) & Eq(x - 1, 0)) 
620   
621      >>> factor_system_bool([x**2 + 2*x + 1 - (x + 1)**2]) 
622      True 
623   
624      The result is logically equivalent to the system of equations 
625      i.e. eqs. The function returns ``True`` when all values of 
626      the symbol(s) is a solution and ``False`` when the system 
627      cannot be solved. 
628   
629      See Also 
630      ======== 
631   
632      factor_system : Returns factors and solvability condition separately 
633      factor_system_cond : Returns both factors and conditions 
634   
635      """
636  
637      systems = factor_system_cond(eqs, gens, **kwargs)
638      return Or(*[And(*[Eq(eq, 0) for eq in sys]) for sys in systems])

641  def factor_system_cond(eqs: Sequence[Expr | complex], gens: Sequence[Expr] = (), **kwargs: Any) -> list[list[Expr]]:
642      """ 
643      Factorizes a polynomial system into irreducible components and returns 
644      both generic and degenerate solutions. 
645   
646      Parameters 
647      ========== 
648   
649      eqs : list 
650          List of expressions to be factored. 
651          Each expression is assumed to be equal to zero. 
652   
653      gens : list, optional 
654          Generator(s) of the polynomial ring. 
655          If not provided, all free symbols will be used. 
656   
657      **kwargs : dict, optional 
658          Optional keyword arguments. 
659   
660      Returns 
661      ======= 
662   
663      list[list[Expr]] 
664          A list of lists of expressions, where each sublist represents 
665          an irreducible subsystem. Includes both generic solutions and 
666          degenerate cases requiring equality conditions on parameters. 
667   
668      Examples 
669      ======== 
670   
671      >>> from sympy.solvers.polysys import factor_system_cond 
672      >>> from sympy.abc import x, y, a, b, c 
673   
674      >>> factor_system_cond([x**2 - 4, a*y, b], [x, y]) 
675      [[x + 2, y, b], [x - 2, y, b], [x + 2, a, b], [x - 2, a, b]] 
676   
677      >>> factor_system_cond([a*x*(x-1), b*y, c], [x, y]) 
678      [[x - 1, y, c], [x, y, c], [x - 1, b, c], [x, b, c], [y, a, c], [a, b, c]] 
679   
680      An empty list [] means no solution exists. 
681      A list containing an empty list [[]] means any value of 
682      the symbol(s) is a solution. 
683   
684      See Also 
685      ======== 
686   
687      factor_system : Returns only generic solutions 
688      factor_system_bool : Returns a Boolean combination representing all solutions 
689      sympy.polys.polytools.factor : Factors a polynomial into irreducible factors 
690                                     over the rational numbers 
691      """
692      systems_poly = _factor_system_poly_from_expr(eqs, gens, **kwargs)
693      systems = [[p.as_expr() for p in system] for system in systems_poly]
694      return systems

697  def _factor_system_poly_from_expr(
698          eqs: Sequence[Expr | complex], gens: Sequence[Expr], **kwargs: Any
699  ) -> list[list[Poly]]:
700      """ 
701      Convert expressions to polynomials and factor the system. 
702   
703      Takes a sequence of expressions, converts them to 
704      polynomials, and factors the resulting system. Handles both regular 
705      polynomial systems and purely numerical cases. 
706      """
707      try:
708          polys, opts = parallel_poly_from_expr(eqs, *gens, **kwargs)
709          only_numbers = False
710      except (GeneratorsNeeded, PolificationFailed):
711          _u = Dummy('u')
712          polys, opts = parallel_poly_from_expr(eqs, [_u], **kwargs)
713          assert opts['domain'].is_Numerical
714          only_numbers = True
715  
716      if only_numbers:
717          return [[]] if all(p == 0 for p in polys) else []
718  
719      return factor_system_poly(polys)

722  def factor_system_poly(polys: list[Poly]) -> list[list[Poly]]:
723      """ 
724      Factors a system of polynomial equations into irreducible subsystems 
725   
726      Core implementation that works directly with Poly instances. 
727   
728      Parameters 
729      ========== 
730   
731      polys : list[Poly] 
732          A list of Poly instances to be factored. 
733   
734      Returns 
735      ======= 
736   
737      list[list[Poly]] 
738          A list of lists of polynomials, where each sublist represents 
739          an irreducible component of the solution. Includes both 
740          generic and degenerate cases. 
741   
742      Examples 
743      ======== 
744   
745      >>> from sympy import symbols, Poly, ZZ 
746      >>> from sympy.solvers.polysys import factor_system_poly 
747      >>> a, b, c, x = symbols('a b c x') 
748      >>> p1 = Poly((a - 1)*(x - 2), x, domain=ZZ[a,b,c]) 
749      >>> p2 = Poly((b - 3)*(x - 2), x, domain=ZZ[a,b,c]) 
750      >>> p3 = Poly(c, x, domain=ZZ[a,b,c]) 
751   
752      The equation to be solved for x is ``x - 2 = 0`` provided either 
753      of the two conditions on the parameters ``a`` and ``b`` is nonzero 
754      and the constant parameter ``c`` should be zero. 
755   
756      >>> sys1, sys2 = factor_system_poly([p1, p2, p3]) 
757      >>> sys1 
758      [Poly(x - 2, x, domain='ZZ[a,b,c]'), 
759       Poly(c, x, domain='ZZ[a,b,c]')] 
760      >>> sys2 
761      [Poly(a - 1, x, domain='ZZ[a,b,c]'), 
762       Poly(b - 3, x, domain='ZZ[a,b,c]'), 
763       Poly(c, x, domain='ZZ[a,b,c]')] 
764   
765       An empty list [] when returned means no solution exists. 
766       Whereas a list containing an empty list [[]] means any value is a solution. 
767   
768      See Also 
769      ======== 
770   
771      factor_system : Returns only generic solutions 
772      factor_system_bool : Returns a Boolean combination representing the solutions 
773      factor_system_cond : Returns both generic and degenerate solutions 
774      sympy.polys.polytools.factor : Factors a polynomial into irreducible factors 
775                                     over the rational numbers 
776      """
777      if not all(isinstance(poly, Poly) for poly in polys):
778          raise TypeError("polys should be a list of Poly instances")
779      if not polys:
780          return [[]]
781  
782      base_domain = polys[0].domain
783      base_gens = polys[0].gens
784      if not all(poly.domain == base_domain and poly.gens == base_gens for poly in polys[1:]):
785          raise DomainError("All polynomials must have the same domain and generators")
786  
787      factor_sets = []
788      for poly in polys:
789          constant, factors_mult = poly.factor_list()
790  
791          if constant.is_zero is True:
792              continue
793          elif constant.is_zero is False:
794              if not factors_mult:
795                  return []
796              factor_sets.append([f for f, _ in factors_mult])
797          else:
798              constant = sqf_part(factor_terms(constant).as_coeff_Mul()[1])
799              constp = Poly(constant, base_gens, domain=base_domain)
800              factors = [f for f, _ in factors_mult]
801              factors.append(constp)
802              factor_sets.append(factors)
803  
804      if not factor_sets:
805          return [[]]
806  
807      result = _factor_sets(factor_sets)
808      return _sort_systems(result)

811  def _factor_sets_slow(eqs: list[list]) -> set[frozenset]:
812      """ 
813      Helper to find the minimal set of factorised subsystems that is 
814      equivalent to the original system. 
815   
816      The result is in DNF. 
817      """
818      if not eqs:
819          return {frozenset()}
820      systems_set = {frozenset(sys) for sys in cartes(*eqs)}
821      return {s1 for s1 in systems_set if not any(s1 > s2 for s2 in systems_set)}
822

824  def _factor_sets(eqs: list[list]) -> set[frozenset]:
825      """ 
826      Helper that builds factor combinations. 
827      """
828      if not eqs:
829          return {frozenset()}
830  
831      current_set = min(eqs, key=len)
832      other_sets = [s for s in eqs if s is not current_set]
833  
834      stack = [(factor, [s for s in other_sets if factor not in s], {factor})
835               for factor in current_set]
836  
837      result = set()
838  
839      while stack:
840          factor, remaining_sets, current_solution = stack.pop()
841  
842          if not remaining_sets:
843              result.add(frozenset(current_solution))
844              continue
845  
846          next_set = min(remaining_sets, key=len)
847          next_remaining = [s for s in remaining_sets if s is not next_set]
848  
849          for next_factor in next_set:
850              valid_remaining = [s for s in next_remaining if next_factor not in s]
851              new_solution = current_solution | {next_factor}
852              stack.append((next_factor, valid_remaining, new_solution))
853  
854      return {s1 for s1 in result if not any(s1 > s2 for s2 in result)}

857  def _sort_systems(systems: Iterable[Iterable[Poly]]) -> list[list[Poly]]:
858      """Sorts a list of lists of polynomials"""
859      systems_list = [sorted(s, key=_poly_sort_key, reverse=True) for s in systems]
860      return sorted(systems_list, key=_sys_sort_key, reverse=True)

863  def _poly_sort_key(poly):
864      """Sort key for polynomials"""
865      if poly.domain.is_FF:
866          poly = poly.set_domain(ZZ)
867      return poly.degree_list(), poly.rep.to_list()

870  def _sys_sort_key(sys):
871      """Sort key for lists of polynomials"""
872      return list(zip(*map(_poly_sort_key, sys)))
