348  static hipblasOperation_t _hipblasOpFromChar(char op) {
349    switch (op) {
350      case 'n':
351      case 'N':
352        return HIPBLAS_OP_N;
353      case 't':
354      case 'T':
355        return HIPBLAS_OP_T;
356      case 'c':
357      case 'C':
358        return HIPBLAS_OP_C;
359    }
360    TORCH_CHECK(false,
361        "_hipblasOpFromChar input should be 't', 'n' or 'c' but got `", op, "`");
362  }
363  
364  static char _charFromhipblasOp(hipblasOperation_t op) {
365    switch (op) {
366      case HIPBLAS_OP_N:
367        return 'N';
368      case HIPBLAS_OP_T:
369        return 'T';
370      case HIPBLAS_OP_C:
371        return 'C';
372    }
373    TORCH_CHECK(false,
374        "_charFromhipblasOp input should be HIPBLAS_OP_N/T/C but got `", op, "`");
375  }

377  static hipblasOperation_t MapLayoutToHipBlasLt(BlasOp layout) {
378    if (layout == BlasOp::N) {
379      return HIPBLAS_OP_N;
380    }
381    return HIPBLAS_OP_T;
382  }

384  static size_t GetHipblasltWorkspaceSize() {
385    static const auto env = c10::utils::get_env("HIPBLASLT_WORKSPACE_SIZE");
386    // 256MB is max workspace size allowed for hipblaslt
387    // hipblaslt-bench uses 32MB
388    // recommendation from hipblaslt author was 76MB
389    // TunableOp hipBLASLt workspace size is aligned with
390    // PyTorch's default in CUDABlas.cpp (_parseChosenWorkspaceSize)
391    size_t workspace_size = 76*1024;
392    if (env) {
393      try {
394        workspace_size = std::stoi(env.value());
395      } catch(std::invalid_argument const& e) {
396        TORCH_WARN("invalid HIPBLASLT_WORKSPACE_SIZE,",
397                   " using default workspace size of ", workspace_size, " KiB.");
398      } catch(std::out_of_range const& e) {
399        TORCH_WARN("HIPBLASLT_WORKSPACE_SIZE out of range,",
400                   " using default workspace size of ", workspace_size, " KiB.");
401      }
402    }
403    return workspace_size * 1024;
404  }

429  class HipBlasLtMatmulDescriptor : public HipBlasLtDescriptor<
430                                       hipblasLtMatmulDescOpaque_t,
431                                       &hipblasLtMatmulDescDestroy> {
432   public:
433    HipBlasLtMatmulDescriptor(
434        hipblasComputeType_t compute_type,
435        hipDataType scale_type) {
436      hipblasLtMatmulDesc_t raw_descriptor = nullptr;
437      TORCH_HIPBLASLT_CHECK(
438          hipblasLtMatmulDescCreate(&raw_descriptor, compute_type, scale_type));
439      descriptor_.reset(raw_descriptor);
440    }

443      TORCH_HIPBLASLT_CHECK(::hipblasLtMatmulDescSetAttribute(descriptor(), attr, &value, sizeof(T)));
444    }
445  };

447  template <typename AT, typename BT, typename CT, BlasOp ALayout, BlasOp BLayout, typename ParamsT>
448  class HipblasltGemmOp : public Callable<ParamsT> {
449    public:
450      HipblasltGemmOp(hipblasLtMatmulAlgo_t algo) : algo_{algo} {}
451  
452      TuningStatus Call(const ParamsT* params) override {
453        hipblasOperation_t transa_outer = MapLayoutToHipBlasLt(ALayout);
454        hipblasOperation_t transb_outer = MapLayoutToHipBlasLt(BLayout);
455        auto a_datatype = HipDataTypeFor<AT>();
456        auto b_datatype = HipDataTypeFor<BT>();
457        auto in_out_datatype = HipDataTypeFor<CT>();
458        auto opa = _hipblasOpFromChar(params->transa);
459        auto opb = _hipblasOpFromChar(params->transb);
460  
461        TORCH_CHECK(transa_outer == opa && transb_outer == opb, "trans mismatch, shouldn't happen");
462  
463        float alpha = GetAlphaFromParams<CT>(params);
464        float beta = GetBetaFromParams<CT>(params);
465  
466        hipblasLtMatrixLayout_t mat_a, mat_b, mat_c;
467        if (opa == HIPBLAS_OP_N) {
468          TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutCreate(&mat_a, a_datatype, params->m, params->k, params->lda));
469        }
470        else {
471          TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutCreate(&mat_a, a_datatype, params->k, params->m, params->lda));
472        }
473        if (opb == HIPBLAS_OP_N) {
474          TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutCreate(&mat_b, b_datatype, params->k, params->n, params->ldb));
475        }
476        else {
477          TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutCreate(&mat_b, b_datatype, params->n, params->k, params->ldb));
478        }
479        TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutCreate(&mat_c, in_out_datatype, params->m, params->n, params->ldc));
480  
481        // specific to batched gemmm
482        int batch = GetBatchFromParams<CT>(params);
483        if (batch > 1) {
484          int64_t stride_a = GetStrideAFromParams<CT>(params);
485          int64_t stride_b = GetStrideBFromParams<CT>(params);
486          int64_t stride_c = GetStrideCFromParams<CT>(params);
487          TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutSetAttribute(
488              mat_a, HIPBLASLT_MATRIX_LAYOUT_BATCH_COUNT, &batch, sizeof(batch)));
489          TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutSetAttribute(
490              mat_a, HIPBLASLT_MATRIX_LAYOUT_STRIDED_BATCH_OFFSET, &stride_a, sizeof(stride_a)));
491          TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutSetAttribute(
492              mat_b, HIPBLASLT_MATRIX_LAYOUT_BATCH_COUNT, &batch, sizeof(batch)));
493          TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutSetAttribute(
494              mat_b, HIPBLASLT_MATRIX_LAYOUT_STRIDED_BATCH_OFFSET, &stride_b, sizeof(stride_b)));
495          TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutSetAttribute(
496              mat_c, HIPBLASLT_MATRIX_LAYOUT_BATCH_COUNT, &batch, sizeof(batch)));
497          TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutSetAttribute(
498              mat_c, HIPBLASLT_MATRIX_LAYOUT_STRIDED_BATCH_OFFSET, &stride_c, sizeof(stride_c)));
499        }
500  
501        hipblasComputeType_t computeType = HIPBLAS_COMPUTE_32F;
502        if (at::globalContext().allowTF32CuBLAS()) {
503          computeType = HIPBLAS_COMPUTE_32F_FAST_TF32;
504        }
505        HipBlasLtMatmulDescriptor matmul(computeType, HIP_R_32F);
506        matmul.setAttribute(HIPBLASLT_MATMUL_DESC_TRANSA, opa);
507        matmul.setAttribute(HIPBLASLT_MATMUL_DESC_TRANSB, opb);
508  
509        // specific to scaled gemm
510        const void* mat1_scale_ptr = GetAScalePointerFromParams<CT>(params);
511        const void* mat2_scale_ptr = GetBScalePointerFromParams<CT>(params);
512        const void* result_scale_ptr = GetDScalePointerFromParams<CT>(params);
513        if (mat1_scale_ptr && mat2_scale_ptr) {
514  #ifdef HIPBLASLT_VEC_EXT
515          if (GetUseRowwiseFromParams<CT>(params)) {
516            matmul.setAttribute(HIPBLASLT_MATMUL_DESC_A_SCALE_POINTER_VEC_EXT, mat1_scale_ptr);
517            matmul.setAttribute(HIPBLASLT_MATMUL_DESC_B_SCALE_POINTER_VEC_EXT, mat2_scale_ptr);
518          }
519          else
520  #endif
521          {
522            matmul.setAttribute(HIPBLASLT_MATMUL_DESC_A_SCALE_POINTER, mat1_scale_ptr);
523            matmul.setAttribute(HIPBLASLT_MATMUL_DESC_B_SCALE_POINTER, mat2_scale_ptr);
524          }
525  #ifdef HIPBLASLT_OUTER_VEC
526          if (GetUseRowwiseFromParams<CT>(params)) {
527            matmul.setAttribute(HIPBLASLT_MATMUL_DESC_A_SCALE_MODE, HIPBLASLT_MATMUL_MATRIX_SCALE_OUTER_VEC_32F);
528            matmul.setAttribute(HIPBLASLT_MATMUL_DESC_B_SCALE_MODE, HIPBLASLT_MATMUL_MATRIX_SCALE_OUTER_VEC_32F);
529          }
530  #endif
531        }
532        if (result_scale_ptr) {
533          matmul.setAttribute(HIPBLASLT_MATMUL_DESC_D_SCALE_POINTER, result_scale_ptr);
534        }
535  
536        const void* bias_ptr = GetBiasPointerFromParams<CT>(params);
537        auto bias_datatype = GetBiasTypeFromParams<CT>(params);
538        if (bias_ptr) {
539          matmul.setAttribute(HIPBLASLT_MATMUL_DESC_BIAS_POINTER, bias_ptr);
540          matmul.setAttribute(HIPBLASLT_MATMUL_DESC_BIAS_DATA_TYPE, bias_datatype);
541          auto activation = GetActivationFromParams<CT>(params);
542          if (activation == at::cuda::blas::GEMMAndBiasActivationEpilogue::RELU) {
543            matmul.setAttribute(HIPBLASLT_MATMUL_DESC_EPILOGUE, HIPBLASLT_EPILOGUE_RELU_BIAS);
544          }
545          else if (activation == at::cuda::blas::GEMMAndBiasActivationEpilogue::GELU) {
546            matmul.setAttribute(HIPBLASLT_MATMUL_DESC_EPILOGUE, HIPBLASLT_EPILOGUE_GELU_BIAS);
547          }
548          else {
549            matmul.setAttribute(HIPBLASLT_MATMUL_DESC_EPILOGUE, HIPBLASLT_EPILOGUE_BIAS);
550          }
551        }
552  
553        size_t workspace_size = GetHipblasltWorkspaceSize();
554  
555        auto op_handle = at::cuda::getCurrentCUDABlasLtHandle();
556  
557        size_t ret_workspace_size = 0;
558        auto status = hipblaslt_ext::matmulIsAlgoSupported(op_handle,
559            matmul.descriptor(),
560            &alpha,
561            mat_a,
562            mat_b,
563            &beta,
564            mat_c,
565            mat_c,
566            algo_,
567            ret_workspace_size);
568  
569        if (status == HIPBLAS_STATUS_SUCCESS) {
570          if (ret_workspace_size >= workspace_size) {
571            return FAIL;
572          }
573        }
574        else {
575          return FAIL;
576        }
577  
578        void* workspace_buffer = nullptr;
579        if (workspace_size > 0) {
580          workspace_buffer = c10::cuda::CUDACachingAllocator::raw_alloc(workspace_size);
581        }
582  
583        TORCH_HIPBLASLT_CHECK(hipblasLtMatmul(op_handle,
584              matmul.descriptor(),
585              &alpha,
586              params->a,
587              mat_a,
588              params->b,
589              mat_b,
590              &beta,
591              params->c,
592              mat_c,
593              params->c,
594              mat_c,
595              &algo_,
596              workspace_buffer,
597              workspace_size,
598              at::cuda::getCurrentCUDAStream()));
599  
600        //TORCH_HIPBLASLT_CHECK(hipblasLtMatmulDescDestroy(matmul));
601        TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutDestroy(mat_a));
602        TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutDestroy(mat_b));
603        TORCH_HIPBLASLT_CHECK(hipblasLtMatrixLayoutDestroy(mat_c));
604        if (workspace_size > 0) {
605          c10::cuda::CUDACachingAllocator::raw_delete(workspace_buffer);
606        }
607        return OK;
608      }
609  
610    private:
611      hipblasLtMatmulAlgo_t algo_;
612  };

614  template <typename AT, typename BT, typename CT, BlasOp ALayout, BlasOp BLayout, typename ParamsT>
615  auto GetHipBlasLtTypeStringAndOps() {
616    hipblasOperation_t transa_outer = MapLayoutToHipBlasLt(ALayout);
617    hipblasOperation_t transb_outer = MapLayoutToHipBlasLt(BLayout);
618    auto a_datatype = HipDataTypeFor<AT>();
619    auto b_datatype = HipDataTypeFor<BT>();
620    auto in_out_datatype = HipDataTypeFor<CT>();
621    std::vector<hipblasLtMatmulHeuristicResult_t> heuristic_result;
622  #if ROCM_VERSION == 60400
623    // hipblaslt TT fp32 regression on ROCm 6.4, cannot use
624    if ((a_datatype == HIP_R_32F || b_datatype == HIP_R_32F || in_out_datatype == HIP_R_32F)
625            && (transa_outer == HIPBLAS_OP_T && transb_outer == HIPBLAS_OP_T)) {
626      std::vector<std::pair<std::string, std::unique_ptr<Callable<ParamsT>>>> ignore;
627      return ignore;
628    }
629  #endif
630  
631    hipblasComputeType_t computeType = HIPBLAS_COMPUTE_32F;
632    if (at::globalContext().allowTF32CuBLAS()) {
633      computeType = HIPBLAS_COMPUTE_32F_FAST_TF32;
634    }
635  
636    hipblasLtHandle_t handle;
637    TORCH_HIPBLASLT_CHECK(hipblasLtCreate(&handle));
638    TORCH_HIPBLASLT_CHECK(hipblaslt_ext::getAllAlgos(handle,
639          hipblaslt_ext::GemmType::HIPBLASLT_GEMM,
640          transa_outer,
641          transb_outer,
642          a_datatype,
643          b_datatype,
644          in_out_datatype,
645          in_out_datatype,
646          computeType,
647          heuristic_result));
648    TORCH_HIPBLASLT_CHECK(hipblasLtDestroy(handle));
649  
650    int returned_algo_count = heuristic_result.size();
651    std::vector<std::pair<std::string, std::unique_ptr<Callable<ParamsT>>>> ret;
652    for (int i = 0; i < returned_algo_count; i++) {
653      auto algo = heuristic_result[i].algo;
654      int algo_index = hipblaslt_ext::getIndexFromAlgo(algo);
655      auto callable = std::make_unique<HipblasltGemmOp<AT, BT, CT, ALayout, BLayout, ParamsT>>(algo);
656      std::string type_string = fmt::sprintf("Gemm_Hipblaslt_%d", algo_index);
657      ret.emplace_back(type_string, std::move(callable));
658    }
659  
660    return ret;
661  }
