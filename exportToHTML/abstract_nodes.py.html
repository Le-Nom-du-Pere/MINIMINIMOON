1    """This module provides containers for python objects that are valid

6    from sympy.core.containers import Tuple
7    
8    
9    class List(Tuple):
10       """Represents a (frozen) (Python) list (for code printing purposes)."""
11       def __eq__(self, other):
12           if isinstance(other, list):
13               return self == List(*other)
14           else:
15               return self.args == other
16   
17       def __hash__(self):
18           return super().__hash__()
