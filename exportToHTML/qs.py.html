9    class SievePolynomial:
10       def __init__(self, a, b, N):
11           """This class denotes the sieve polynomial. 
12           Provide methods to compute `(a*x + b)**2 - N` and 
13           `a*x + b` when given `x`. 
14    
15           Parameters 
16           ========== 
17    
18           a : parameter of the sieve polynomial 
19           b : parameter of the sieve polynomial 
20           N : number to be factored 
21    
22           """
23           self.a = a
24           self.b = b
25           self.a2 = a**2
26           self.ab = 2*a*b
27           self.b2 = b**2 - N
28   
29       def eval_u(self, x):
30           return self.a*x + self.b
31   
32       def eval_v(self, x):
33           return (self.a2*x + self.ab)*x + self.b2

36   class FactorBaseElem:
37       """This class stores an element of the `factor_base`. 
38       """
39       def __init__(self, prime, tmem_p, log_p):
40           """ 
41           Initialization of factor_base_elem. 
42    
43           Parameters 
44           ========== 
45    
46           prime : prime number of the factor_base 
47           tmem_p : Integer square root of x**2 = n mod prime 
48           log_p : Compute Natural Logarithm of the prime 
49           """
50           self.prime = prime
51           self.tmem_p = tmem_p
52           self.log_p = log_p
53           # `soln1` and `soln2` are solutions to
54           # the equation `(a*x + b)**2 - N = 0 (mod p)`.
55           self.soln1 = None
56           self.soln2 = None
57           self.b_ainv = None

60   def _generate_factor_base(prime_bound, n):
61       """Generate `factor_base` for Quadratic Sieve. The `factor_base` 
62       consists of all the points whose ``legendre_symbol(n, p) == 1`` 
63       and ``p < num_primes``. Along with the prime `factor_base` also stores 
64       natural logarithm of prime and the residue n modulo p. 
65       It also returns the of primes numbers in the `factor_base` which are 
66       close to 1000 and 5000. 
67    
68       Parameters 
69       ========== 
70    
71       prime_bound : upper prime bound of the factor_base 
72       n : integer to be factored 
73       """
74       from sympy.ntheory.generate import sieve
75       factor_base = []
76       idx_1000, idx_5000 = None, None
77       for prime in sieve.primerange(1, prime_bound):
78           if pow(n, (prime - 1) // 2, prime) == 1:
79               if prime > 1000 and idx_1000 is None:
80                   idx_1000 = len(factor_base) - 1
81               if prime > 5000 and idx_5000 is None:
82                   idx_5000 = len(factor_base) - 1
83               residue = _sqrt_mod_prime_power(n, prime, 1)[0]
84               log_p = round(log(prime)*2**10)
85               factor_base.append(FactorBaseElem(prime, residue, log_p))
86       return idx_1000, idx_5000, factor_base

89   def _generate_polynomial(N, M, factor_base, idx_1000, idx_5000, randint):
90       """ Generate sieve polynomials indefinitely. 
91       Information such as `soln1` in the `factor_base` associated with 
92       the polynomial is modified in place. 
93    
94       Parameters 
95       ========== 
96    
97       N : Number to be factored 
98       M : sieve interval 
99       factor_base : factor_base primes 
100      idx_1000 : index of prime number in the factor_base near 1000 
101      idx_5000 : index of prime number in the factor_base near to 5000 
102      randint : A callable that takes two integers (a, b) and returns a random integer 
103                n such that a <= n <= b, similar to `random.randint`. 
104      """
105      approx_val = log(2*N)/2 - log(M)
106      start = idx_1000 or 0
107      end = idx_5000 or (len(factor_base) - 1)
108      while True:
109          # Choose `a` that is close to `sqrt(2*N) / M`
110          best_a, best_q, best_ratio = None, None, None
111          for _ in range(50):
112              a = 1
113              q = []
114              while log(a) < approx_val:
115                  rand_p = 0
116                  while(rand_p == 0 or rand_p in q):
117                      rand_p = randint(start, end)
118                  p = factor_base[rand_p].prime
119                  a *= p
120                  q.append(rand_p)
121              ratio = exp(log(a) - approx_val)
122              if best_ratio is None or abs(ratio - 1) < abs(best_ratio - 1):
123                  best_q = q
124                  best_a = a
125                  best_ratio = ratio
126  
127          # Set `b` using the Chinese remainder theorem
128          a = best_a
129          q = best_q
130          B = []
131          for val in q:
132              q_l = factor_base[val].prime
133              gamma = factor_base[val].tmem_p * invert(a // q_l, q_l) % q_l
134              if 2*gamma > q_l:
135                  gamma = q_l - gamma
136              B.append(a//q_l*gamma)
137          b = sum(B)
138          g = SievePolynomial(a, b, N)
139          for fb in factor_base:
140              if a % fb.prime == 0:
141                  fb.soln1 = None
142                  continue
143              a_inv = invert(a, fb.prime)
144              fb.b_ainv = [2*b_elem*a_inv % fb.prime for b_elem in B]
145              fb.soln1 = (a_inv*(fb.tmem_p - b)) % fb.prime
146              fb.soln2 = (a_inv*(-fb.tmem_p - b)) % fb.prime
147          yield g
148  
149          # Update `b` with Gray code
150          for i in range(1, 2**(len(B)-1)):
151              v = bit_scan1(i)
152              neg_pow = 2*((i >> (v + 1)) % 2) - 1
153              b = g.b + 2*neg_pow*B[v]
154              a = g.a
155              g = SievePolynomial(a, b, N)
156              for fb in factor_base:
157                  if fb.soln1 is None:
158                      continue
159                  fb.soln1 = (fb.soln1 - neg_pow*fb.b_ainv[v]) % fb.prime
160                  fb.soln2 = (fb.soln2 - neg_pow*fb.b_ainv[v]) % fb.prime
161              yield g

164  def _gen_sieve_array(M, factor_base):
165      """Sieve Stage of the Quadratic Sieve. For every prime in the factor_base 
166      that does not divide the coefficient `a` we add log_p over the sieve_array 
167      such that ``-M <= soln1 + i*p <=  M`` and ``-M <= soln2 + i*p <=  M`` where `i` 
168      is an integer. When p = 2 then log_p is only added using 
169      ``-M <= soln1 + i*p <=  M``. 
170   
171      Parameters 
172      ========== 
173   
174      M : sieve interval 
175      factor_base : factor_base primes 
176      """
177      sieve_array = [0]*(2*M + 1)
178      for factor in factor_base:
179          if factor.soln1 is None: #The prime does not divides a
180              continue
181          for idx in range((M + factor.soln1) % factor.prime, 2*M, factor.prime):
182              sieve_array[idx] += factor.log_p
183          if factor.prime == 2:
184              continue
185          #if prime is 2 then sieve only with soln_1_p
186          for idx in range((M + factor.soln2) % factor.prime, 2*M, factor.prime):
187              sieve_array[idx] += factor.log_p
188      return sieve_array

191  def _check_smoothness(num, factor_base):
192      r""" Check if `num` is smooth with respect to the given `factor_base` 
193      and compute its factorization vector. 
194   
195      Parameters 
196      ========== 
197   
198      num : integer whose smootheness is to be checked 
199      factor_base : factor_base primes 
200      """
201      if num < 0:
202          num *= -1
203          vec = 1
204      else:
205          vec = 0
206      for i, fb in enumerate(factor_base, 1):
207          if num % fb.prime:
208              continue
209          e = 1
210          num //= fb.prime
211          while num % fb.prime == 0:
212              e += 1
213              num //= fb.prime
214          if e % 2:
215              vec += 1 << i
216      return vec, num

219  def _trial_division_stage(N, M, factor_base, sieve_array, sieve_poly, partial_relations, ERROR_TERM):
220      """Trial division stage. Here we trial divide the values generetated 
221      by sieve_poly in the sieve interval and if it is a smooth number then 
222      it is stored in `smooth_relations`. Moreover, if we find two partial relations 
223      with same large prime then they are combined to form a smooth relation. 
224      First we iterate over sieve array and look for values which are greater 
225      than accumulated_val, as these values have a high chance of being smooth 
226      number. Then using these values we find smooth relations. 
227      In general, let ``t**2 = u*p modN`` and ``r**2 = v*p modN`` be two partial relations 
228      with the same large prime p. Then they can be combined ``(t*r/p)**2 = u*v modN`` 
229      to form a smooth relation. 
230   
231      Parameters 
232      ========== 
233   
234      N : Number to be factored 
235      M : sieve interval 
236      factor_base : factor_base primes 
237      sieve_array : stores log_p values 
238      sieve_poly : polynomial from which we find smooth relations 
239      partial_relations : stores partial relations with one large prime 
240      ERROR_TERM : error term for accumulated_val 
241      """
242      accumulated_val = (log(M) + log(N)/2 - ERROR_TERM) * 2**10
243      smooth_relations = []
244      proper_factor = set()
245      partial_relation_upper_bound = 128*factor_base[-1].prime
246      for x, val in enumerate(sieve_array, -M):
247          if val < accumulated_val:
248              continue
249          v = sieve_poly.eval_v(x)
250          vec, num = _check_smoothness(v, factor_base)
251          if num == 1:
252              smooth_relations.append((sieve_poly.eval_u(x), v, vec))
253          elif num < partial_relation_upper_bound and isprime(num):
254              if N % num == 0:
255                  proper_factor.add(num)
256                  continue
257              u = sieve_poly.eval_u(x)
258              if num in partial_relations:
259                  u_prev, v_prev, vec_prev = partial_relations.pop(num)
260                  u = u*u_prev*invert(num, N) % N
261                  v = v*v_prev // num**2
262                  vec ^= vec_prev
263                  smooth_relations.append((u, v, vec))
264              else:
265                  partial_relations[num] = (u, v, vec)
266      return smooth_relations, proper_factor

269  def _find_factor(N, smooth_relations, col):
270      """ Finds proper factor of N using fast gaussian reduction for modulo 2 matrix. 
271   
272      Parameters 
273      ========== 
274   
275      N : Number to be factored 
276      smooth_relations : Smooth relations vectors matrix 
277      col : Number of columns in the matrix 
278   
279      Reference 
280      ========== 
281   
282      .. [1] A fast algorithm for gaussian elimination over GF(2) and 
283      its implementation on the GAPP. Cetin K.Koc, Sarath N.Arachchige 
284      """
285      matrix = [s_relation[2] for s_relation in smooth_relations]
286      row = len(matrix)
287      mark = [False] * row
288      for pos in range(col):
289          m = 1 << pos
290          for i in range(row):
291              if p := matrix[i] & m:
292                  add_col = p ^ matrix[i]
293                  matrix[i] = m
294                  mark[i] = True
295                  for j in range(i + 1, row):
296                      if matrix[j] & m:
297                          matrix[j] ^= add_col
298                  break
299  
300      for m, mat, rel in zip(mark, matrix, smooth_relations):
301          if m:
302              continue
303          u, v = rel[0], rel[1]
304          for m1, mat1, rel1 in zip(mark, matrix, smooth_relations):
305              if m1 and mat & mat1:
306                  u *= rel1[0]
307                  v *= rel1[1]
308          # assert is_square(v)
309          v = isqrt(v)
310          if 1 < (g := gcd(u - v, N)) < N:
311              yield g

314  def qs(N, prime_bound, M, ERROR_TERM=25, seed=1234):
315      """Performs factorization using Self-Initializing Quadratic Sieve. 
316      In SIQS, let N be a number to be factored, and this N should not be a 
317      perfect power. If we find two integers such that ``X**2 = Y**2 modN`` and 
318      ``X != +-Y modN``, then `gcd(X + Y, N)` will reveal a proper factor of N. 
319      In order to find these integers X and Y we try to find relations of form 
320      t**2 = u modN where u is a product of small primes. If we have enough of 
321      these relations then we can form ``(t1*t2...ti)**2 = u1*u2...ui modN`` such that 
322      the right hand side is a square, thus we found a relation of ``X**2 = Y**2 modN``. 
323   
324      Here, several optimizations are done like using multiple polynomials for 
325      sieving, fast changing between polynomials and using partial relations. 
326      The use of partial relations can speeds up the factoring by 2 times. 
327   
328      Parameters 
329      ========== 
330   
331      N : Number to be Factored 
332      prime_bound : upper bound for primes in the factor base 
333      M : Sieve Interval 
334      ERROR_TERM : Error term for checking smoothness 
335      seed : seed of random number generator 
336   
337      Returns 
338      ======= 
339   
340      set(int) : A set of factors of N without considering multiplicity. 
341                 Returns ``{N}`` if factorization fails. 
342   
343      Examples 
344      ======== 
345   
346      >>> from sympy.ntheory import qs 
347      >>> qs(25645121643901801, 2000, 10000) 
348      {5394769, 4753701529} 
349      >>> qs(9804659461513846513, 2000, 10000) 
350      {4641991, 2112166839943} 
351   
352      See Also 
353      ======== 
354   
355      qs_factor 
356   
357      References 
358      ========== 
359   
360      .. [1] https://pdfs.semanticscholar.org/5c52/8a975c1405bd35c65993abf5a4edb667c1db.pdf 
361      .. [2] https://www.rieselprime.de/ziki/Self-initializing_quadratic_sieve 
362      """
363      return set(qs_factor(N, prime_bound, M, ERROR_TERM, seed))

366  def qs_factor(N, prime_bound, M, ERROR_TERM=25, seed=1234):
367      """ Performs factorization using Self-Initializing Quadratic Sieve. 
368   
369      Parameters 
370      ========== 
371   
372      N : Number to be Factored 
373      prime_bound : upper bound for primes in the factor base 
374      M : Sieve Interval 
375      ERROR_TERM : Error term for checking smoothness 
376      seed : seed of random number generator 
377   
378      Returns 
379      ======= 
380   
381      dict[int, int] : Factors of N. 
382                       Returns ``{N: 1}`` if factorization fails. 
383                       Note that the key is not always a prime number. 
384   
385      Examples 
386      ======== 
387   
388      >>> from sympy.ntheory import qs_factor 
389      >>> qs_factor(1009 * 100003, 2000, 10000) 
390      {1009: 1, 100003: 1} 
391   
392      See Also 
393      ======== 
394   
395      qs 
396   
397      """
398      if N < 2:
399          raise ValueError("N should be greater than 1")
400      factors = {}
401      smooth_relations = []
402      partial_relations = {}
403      # Eliminate the possibility of even numbers,
404      # prime numbers, and perfect powers.
405      if N % 2 == 0:
406          e = 1
407          N //= 2
408          while N % 2 == 0:
409              N //= 2
410              e += 1
411          factors[2] = e
412      if isprime(N):
413          factors[N] = 1
414          return factors
415      if result := _perfect_power(N, 3):
416          n, e = result
417          factors[n] = e
418          return factors
419      N_copy = N
420      randint = _randint(seed)
421      idx_1000, idx_5000, factor_base = _generate_factor_base(prime_bound, N)
422      threshold = len(factor_base) * 105//100
423      for g in _generate_polynomial(N, M, factor_base, idx_1000, idx_5000, randint):
424          sieve_array = _gen_sieve_array(M, factor_base)
425          s_rel, p_f = _trial_division_stage(N, M, factor_base, sieve_array, g, partial_relations, ERROR_TERM)
426          smooth_relations += s_rel
427          for p in p_f:
428              if N_copy % p:
429                  continue
430              e = 1
431              N_copy //= p
432              while N_copy % p == 0:
433                  N_copy //= p
434                  e += 1
435              factors[p] = e
436          if threshold <= len(smooth_relations):
437              break
438  
439      for factor in _find_factor(N, smooth_relations, len(factor_base) + 1):
440          if N_copy % factor == 0:
441              e = 1
442              N_copy //= factor
443              while N_copy % factor == 0:
444                  N_copy //= factor
445                  e += 1
446              factors[factor] = e
447              if N_copy == 1 or isprime(N_copy):
448                  break
449      if N_copy != 1:
450          factors[N_copy] = 1
451      return factors
