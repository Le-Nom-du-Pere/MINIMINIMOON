34   def _get_example_value(node: fx.Node) -> Optional[str]:
35       """ 
36       Get the example value key for a node, since dynamo uses "example_value" 
37       while non-strict export uses "val. 
38       """
39       if "example_value" in node.meta:
40           return node.meta["example_value"]
41       elif "val" in node.meta:
42           return node.meta["val"]
43       else:
44           return None
45   
46   
47   def _get_sym_val(node: fx.Node) -> Optional["sympy.Expr"]:
48       val = _get_example_value(node)
49       if isinstance(val, py_sym_types):
50           return val.node.expr
51       return None

54   @compatibility(is_backward_compatible=True)
55   def insert_deferred_runtime_asserts(
56       gm: GraphModule,
57       shape_env: ShapeEnv,
58       name: str,
59       export: bool = False,
60   ) -> None:
61       """ 
62       During tracing, we may have discovered that some data-dependent values 
63       had runtime assert on them; e.g., torch.empty(x.item()) induces a runtime 
64       that x.item() >= 0.  This asserts can happen unpredictably during fake 
65       tensor propagation, so we cannot conveniently insert them into the FX graph 
66       when they occur.  Instead, we accumulate them in the ShapeEnv, and in this 
67       pass insert them into the graph as proper tests. 
68    
69       This pass also deduplicates size-related computation, CSE-ing ops that produce 
70       symbolic values and/or are involved in runtime asserts. Additionally, shape calls 
71       (size/stride/storage_offset) are turned into compute on input sizes if possible, 
72       allowing intermediate tensors to be freed earlier. For example, here dynamo will 
73       DCE the cat and repeat calls: 
74    
75           z = torch.cat([x, x], dim=0)  # 2*s0 
76           w = z.repeat(y.shape[0])  # 2*s0*s1 
77           _w = w.shape[0] 
78           # something with _w, but not w ... 
79    
80           # turns into -> 
81           _w0 = 2 * s0 
82           _w = _w0 * s1 
83    
84           # where s0, s1 are either SymInt graph inputs, or the result of added size calls 
85    
86       Redundant torch._check or torch.ops.aten._assert_scalar.default calls that assert 
87       the same expression, and redundant constrain_range calls are also deduplicated. 
88       Additionally, because single-symbol bound checks (e.g. u0 >= 0, u0 <= 5) accumulate 
89       information in the ShapeEnv, the ShapeEnv contains min/max bounds for each symbol, 
90       and we delete all previous calls, adding bound checks at the end of this pass. 
91       """
92   
93       # Import sympy locally
94       import sympy
95   
96       from torch._export.passes._node_metadata_hook import _set_node_metadata_hook
97       from torch.fx.experimental.symbolic_shapes import (
98           _get_placeholder_expr,
99           _has_uninterpretable_sympy_function,
100          CallMethodKey,
101          cast_symbool_to_symint_guardless,
102          ConvertIntKey,
103          DivideByKey,
104          free_symbols,
105          InnerTensorKey,
106          resolve_unbacked_bindings,
107      )
108      from torch.utils._sympy.numbers import int_oo
109      from torch.utils._sympy.reference import (
110          OptimizedPythonReferenceAnalysis,
111          PythonReferenceAnalysis,
112      )
113      from torch.utils._sympy.value_ranges import ValueRanges

115      # TODO: Request simplification on runtime asserts before emitting them
116      ras_by_symbol = shape_env.deferred_runtime_asserts.copy()
117      graph = gm.graph
118      tracer = fx.proxy.GraphAppendingTracer(graph)
119      graph_code_log.debug(
120          "%s",
121          lazy_format_graph_code(
122              f"pre insert_deferred_runtime_asserts {name}", gm, colored=True
123          ),
124      )

126      # We are going to mutate the dict
127      expr_to_proxy: dict[sympy.Expr, fx.Proxy] = {}
128      placeholders = set()
129      first_non_placeholder = None
130      for node in graph.nodes:
131          if node.op != "placeholder":
132              first_non_placeholder = node
133              break
134          else:
135              placeholders.add(node)

137      def _is_intermediate_tensor_sym_call(node: fx.Node) -> bool:
138          """ 
139          If a size/stride/storage offset call on an intermediate tensor, 
140          we can try to compute the value from input shapes instead. 
141          """
142          return (
143              (val := _get_sym_val(node)) is not None
144              and not isinstance(val, sympy.Number)
145              # this holds back from reifying anything in torch.utils._sympy.functions.py that's unsupported
146              and not _has_uninterpretable_sympy_function(val)
147              and any(
148                  isinstance(arg, fx.Node)
149                  and isinstance(_get_example_value(arg), (torch.Tensor, torch.Size))
150                  and arg.op != "placeholder"
151                  for arg in node.args
152              )
153          )
154  
155      # Figure out what key to use, val or example_value
156      val_key = "val"
157      for node in graph.nodes:
158          if "example_value" in node.meta:
159              val_key = "example_value"
160              break
161          elif "val" in node.meta:
162              break

164      def _node_metadata_hook(
165          node: torch.fx.Node,
166          stack_trace: Optional[str] = None,
167          nn_module_stack: Optional[dict[str, Any]] = None,
168      ) -> None:
169          fake_args = pytree.tree_map(
170              lambda arg: (
171                  _get_example_value(arg) if isinstance(arg, torch.fx.Node) else arg
172              ),
173              node.args,
174          )
175          try:
176              target = node.target
177              if node.op == "call_method":
178                  assert isinstance(node.target, str)
179                  target = getattr(fake_args[0], node.target)
180                  fake_args = fake_args[1:]
181              node.meta[val_key] = target(*fake_args)  # type: ignore[operator]
182          except NotImplementedError:
183              # This can happen when attempting to reify a symbol with an unsupported call_function node,
184              # e.g. with NestedTensors + sym_size.int via match_symbol().
185              # This seems to be fine, as the node gets CSE'd and deleted later in favor of a SymInt graph input.
186              pass
187          if stack_trace is not None:
188              node.meta["stack_trace"] = stack_trace
189          if nn_module_stack is not None:
190              node.meta["nn_module_stack"] = nn_module_stack
191  
192      # Track asserts/checks we've added
193      added_asserts: set[sympy.Expr] = set()
194      constrained_unbacked_symbols: set[sympy.Symbol] = set()
195  
196      Analysis = PythonReferenceAnalysis if export else OptimizedPythonReferenceAnalysis
197  
198      def _sympy_interp(expr_to_proxy, expr):
199          # sympy_interp() with hash consing
200          from sympy import Integer, Number, Symbol
201          from sympy.logic.boolalg import BooleanAtom
202  
203          from torch.utils._sympy.interp import _run_sympy_handler, sympy_interp
204  
205          # hash cons
206          if expr in expr_to_proxy:
207              return expr_to_proxy[expr]
208          # base cases, don't cache
209          if isinstance(expr, (Integer, Number, Symbol, BooleanAtom)):
210              return sympy_interp(Analysis, expr_to_proxy, expr)
211  
212          # hash cons on arguments, run expr handler
213          expr_to_proxy[expr] = _run_sympy_handler(
214              Analysis,
215              [_sympy_interp(expr_to_proxy, arg) for arg in expr.args],
216              expr,
217          )
218          return expr_to_proxy[expr]

220      def _is_bound_expr_for_symbol(expr: "sympy.Expr") -> bool:
221          # This is probably unnecessary, but since torch._check() calls for single-symbol bounds
222          # like u0 >= 0, 10 >= u0 accumulate range info in the ShapeEnv, we designate these calls as redundant
223          # and instead add 2 runtime asserts at the end of this pass, if the min/max bounds are non-trivial.
224          if len(expr.args) != 2 or expr.func not in (sympy.LessThan, sympy.GreaterThan):
225              return False
226          lhs, rhs = expr.args
227          return (isinstance(lhs, sympy.Symbol) and isinstance(rhs, sympy.Number)) or (
228              isinstance(rhs, sympy.Symbol) and isinstance(lhs, sympy.Number)
229          )

231      def add_runtime_asserts(ras):
232          for ra in ras:
233              if (
234                  # redundant
235                  ra.expr in added_asserts
236                  # if we've already added a constrain_range call for this symbol,
237                  # then single-symbol bound asserts like u0 >= 0, u0 <= 5 are redundant.
238                  or (
239                      len(ra.expr.free_symbols) == 1
240                      and next(iter(ra.expr.free_symbols)) in constrained_unbacked_symbols
241                      and _is_bound_expr_for_symbol(ra.expr)
242                  )
243                  # don't try to reify sympy functions we can't turn into FX nodes
244                  or _has_uninterpretable_sympy_function(ra.expr)
245              ):
246                  continue
247  
248              log.debug("inserting runtime assert %s", ra.expr)
249              # Need to process ALL free symbols, not just unbacked ones
250              fvs = free_symbols(ra.expr)
251              missing = fvs - expr_to_proxy.keys()
252              if missing:
253                  i1 = min(missing, key=str)
254                  # TODO: Remove relaxing assert on unbacked_symint https://github.com/pytorch/pytorch/issues/119689
255                  # assert shape_env.is_unbacked_symint(i1), i1
256                  ras_by_symbol.setdefault(i1, []).append(ra)
257              else:
258                  # Convert the sympy expression into a sequence of FX
259                  # nodes
260                  with _set_node_metadata_hook(gm, _node_metadata_hook):
261                      res = _sympy_interp(expr_to_proxy, ra.expr).node
262  
263                      graph.call_function(
264                          torch.ops.aten._assert_scalar.default,
265                          # TODO: use ra.msg here, but it's pretty
266                          # useless right now
267                          (
268                              res,
269                              f"Runtime assertion failed for expression {ra.expr} on node '{res}'",
270                          ),
271                      )
272                  added_asserts.add(ra.expr)
273  
274      nodes = list(graph.nodes)
275      for i, node in enumerate(nodes[:-1]):
276          # Placeholders can match symbols, but when we destructure them
277          # with size we have to make sure we insert the nodes after all
278          # the placeholders
279          with graph.inserting_before(
280              nodes[i + 1] if node not in placeholders else first_non_placeholder
281          ):
282              # Unfortunately, this logic still must remain because manual
283              # make_fx calls may not explicitly bind all symbolic ints as
284              # arguments to the function, so we must infer it from the other
285              # arguments
286              if (
287                  node in placeholders
288                  and (example_value := _get_example_value(node)) is not None
289              ):
290  
291                  def match_symbol(symint, cb):
292                      if (
293                          isinstance(symint, torch.SymInt)
294                          and isinstance(symint.node, SymNode)
295                          and isinstance(
296                              s := _get_placeholder_expr(symint.node), sympy.Symbol
297                          )
298                          and s not in expr_to_proxy
299                      ):
300                          with _set_node_metadata_hook(gm, _node_metadata_hook):
301                              expr_to_proxy[s] = fx.Proxy(cb(), tracer=tracer)
302                          log.debug("expr_to_proxy[%s] = %s", s, expr_to_proxy[s])
303  
304                  match_symbol(example_value, lambda: node)
305                  if isinstance(t := example_value, torch.Tensor):
306                      for i, s in enumerate(t.size()):
307                          match_symbol(
308                              s,
309                              lambda: graph.call_function(
310                                  torch.ops.aten.sym_size.int, (node, i)
311                              ),
312                          )
313                      if not is_sparse_any(t):
314                          for i, s in enumerate(t.stride()):
315                              match_symbol(
316                                  s,
317                                  lambda: graph.call_function(
318                                      torch.ops.aten.sym_stride.int, (node, i)
319                                  ),
320                              )
321                          match_symbol(
322                              t.storage_offset(),
323                              lambda: graph.call_function(
324                                  torch.ops.aten.sym_storage_offset.default, (node,)
325                              ),
326                          )
327  
328              # Handle asserts that aren't associated with any symbol.  This
329              # doesn't really have to be in the loop as it will only run once,
330              # it just needs to happen right after the placeholders.
331              # insert this after placeholders & added sym nodes, and before non-placeholders.
332              if node == first_non_placeholder:
333                  add_runtime_asserts(ras_by_symbol.pop(None, []))  # type: ignore[call-overload]
334  
335              # deduplicate asserts already present in graph, and remove trivial asserts
336              if node.target in (
337                  torch._check,
338                  torch.ops.aten._assert_scalar.default,
339              ):
340                  if (
341                      node.args[0] == True  # noqa: E712
342                      or (assert_expr := _get_sym_val(node.args[0])) in expr_to_proxy
343                      and assert_expr in added_asserts
344                  ):
345                      arg = node.args[0]
346                      gm.graph.erase_node(node)
347                      if isinstance(arg, fx.Node) and not arg.users:
348                          gm.graph.erase_node(arg)
349                  else:
350                      added_asserts.add(assert_expr)  # type: ignore[arg-type]
351  
352              # hash cons, replace function calls that return torch.SymInts with direct references to
353              # FX nodes built up to reify the sympy expression.
354              if (
355                  node.op != "placeholder"
356                  and (sym_expr := _get_sym_val(node)) is not None
357              ):
358                  # this guards against deleting calls like item() that produce new untracked symbols
359                  def has_new_untracked_symbols():
360                      for symbol in sym_expr.free_symbols:
361                          if symbol not in expr_to_proxy:
362                              return True
363                      return False
364  
365                  # this guards against deleting calls that produce unbacked bindings we haven't yet seen.
366                  # in this case looking at sym_expr.free_symbols might not be enough, if the example value has a hint
367                  # (is backed), but produces an unbacked symbol. In this case keep the node alive.
368                  resolved_unbacked_bindings = resolve_unbacked_bindings(
369                      shape_env, node.meta.get("unbacked_bindings", {})
370                  )
371  
372                  assert resolved_unbacked_bindings is not None
373  
374                  def has_new_unbacked_bindings():
375                      for key in resolved_unbacked_bindings.keys():
376                          if key not in expr_to_proxy:
377                              return True
378                      return False
379  
380                  # maybe re-reify expression, replace current node
381                  if (
382                      sym_expr in expr_to_proxy
383                      or (  # example value is redundant
384                          _is_intermediate_tensor_sym_call(node)
385                          # shape call on intermediate tensor, turn into computation on input shapes
386                          and not has_new_untracked_symbols()
387                      )
388                  ) and not has_new_unbacked_bindings():
389                      if _is_intermediate_tensor_sym_call(
390                          node
391                      ):  # reify from input shapes
392                          with _set_node_metadata_hook(
393                              gm,
394                              functools.partial(
395                                  _node_metadata_hook,
396                                  stack_trace=node.meta.get("stack_trace"),
397                                  nn_module_stack=node.meta.get("nn_module_stack"),
398                              ),
399                          ):
400                              expr_to_proxy[sym_expr] = _sympy_interp(
401                                  expr_to_proxy, sym_expr
402                              )  # type: ignore[arg-type]
403                          # won't try DCE-ing tensor compute here
404                      hash_node = expr_to_proxy[sym_expr].node  # type: ignore[arg-type]
405                      node.replace_all_uses_with(hash_node)
406                      gm.graph.erase_node(node)
407                      log.debug(
408                          "CSE node %s -> %s for expr %s", node, hash_node, sym_expr
409                      )
410  
411                  # store node in hash cons, don't delete/replace
412                  elif sym_expr not in expr_to_proxy and not isinstance(
413                      sym_expr, (sympy.Number, sympy.logic.boolalg.BooleanAtom)
414                  ):  # don't hash cons primitives
415                      expr_to_proxy[sym_expr] = fx.Proxy(node, tracer=tracer)  # type: ignore[arg-type]
416  
417              # We add sym_constrain_range calls for symbols later in any case if they're size-like or range-constrained,
418              # so calls before that are redundant.
419              if node.target in (
420                  torch.ops.aten.sym_constrain_range.default,
421                  torch.ops.aten.sym_constrain_range_for_size.default,
422              ):
423                  gm.graph.erase_node(node)
424  
425              defs = []
426  
427              # AOTAutograd will create new symbols as the unbacked_bindings keys, which PropagateSymInts will set as
428              # equivalent, but the refinement calls we perform in this pass may struggle with associating the two.
429              # More concretely, when re-exporting/tracing, constraining only the new symbol may not communicate enough
430              # information about the old symbol when we re-export, raising errors on data-dependent guards.
431              # Call resolve_unbacked_bindings() to get the original symbol if present, otherwise we take it as is.
432              if unbacked_bindings := resolve_unbacked_bindings(
433                  shape_env, node.meta.get("unbacked_bindings")
434              ):
435                  for s, keypath in unbacked_bindings.items():
436                      defs.append(s)
437  
438                      # TODO: some CSE when generating these nodes can probably
439                      # help reduce graph size and improve compile time
440                      def go(node, keypath):
441                          if keypath == ():
442                              return node
443                          if (
444                              len(keypath) >= 2
445                              and isinstance(keypath[0], CallMethodKey)
446                              and isinstance(keypath[1], pytree.SequenceKey)
447                          ):
448                              if keypath[0].name == "size":
449                                  return go(
450                                      graph.call_function(
451                                          torch.ops.aten.sym_size.int,
452                                          (node, keypath[1].idx),
453                                      ),
454                                      keypath[2:],
455                                  )
456                              if keypath[0].name == "stride":
457                                  return go(
458                                      graph.call_function(
459                                          torch.ops.aten.sym_stride.int,
460                                          (node, keypath[1].idx),
461                                      ),
462                                      keypath[2:],
463                                  )
464                              return go(
465                                  graph.call_method(
466                                      keypath[0].name, (node, keypath[1].idx)
467                                  ),
468                                  keypath[2:],
469                              )
470                          elif isinstance(keypath[0], CallMethodKey):
471                              return go(
472                                  graph.call_method(keypath[0].name, (node,)), keypath[1:]
473                              )
474                          elif isinstance(keypath[0], pytree.SequenceKey):
475                              return go(
476                                  graph.call_function(
477                                      operator.getitem, (node, keypath[0].idx)
478                                  ),
479                                  keypath[1:],
480                              )
481                          elif isinstance(keypath[0], ConvertIntKey):
482                              return go(
483                                  graph.call_function(
484                                      cast_symbool_to_symint_guardless, (node,)
485                                  ),
486                                  keypath[1:],
487                              )
488                          elif isinstance(keypath[0], DivideByKey):
489                              # TODO: need to assert divisibility
490                              return go(
491                                  graph.call_function(
492                                      operator.floordiv, (node, keypath[0].divisor)
493                                  ),
494                                  keypath[1:],
495                              )
496                          elif isinstance(keypath[0], InnerTensorKey):
497                              return go(
498                                  graph.call_function(
499                                      getattr, (node, keypath[0].inner_name)
500                                  ),
501                                  keypath[1:],
502                              )
503                          else:
504                              raise AssertionError(f"unrecognized keypath {keypath}")
505  
506                      if s not in expr_to_proxy:
507                          with _set_node_metadata_hook(gm, _node_metadata_hook):
508                              expr_to_proxy[s] = fx.Proxy(
509                                  go(node, keypath), tracer=tracer
510                              )
511                          log.debug("expr_to_proxy[%s] = %s", s, expr_to_proxy[s])
512  
513              for i0 in defs:
514                  ras = ras_by_symbol.pop(i0, [])
515                  # Before we perform any asserts, first apply range
516                  # refinement.  This is important, because if we are going
517                  # to retrace the graph (and we typically are if we send
518                  # the graph to AOTAutograd), we need to make sure we apply
519                  # range refinement (ala _check_is_size) first, BEFORE we
520                  # run any of the asserts.  Otherwise, we may decide to
521                  # perform substitutions based on the asserts which we then
522                  # can't back out, because value ranges can only be applied
523                  # to asserts.)
524                  #
525                  # A perhaps better long term plan is to avoid this order
526                  # dependence by making it possible to refine ranges on
527                  # arbitrary expressions, not just symbols.  But it is not
528                  # so easy to make use of this information, see
529                  # https://twitter.com/ezyang/status/1745801370299482492
530                  # We actually made an attempt at this in
531                  # https://github.com/pytorch/pytorch/pull/119043
532                  # which didn't work.
533                  #
534                  # Another ideas for how to do this:
535                  # - Have bound_sympy be the source of truth of the ranges of any expression
536                  # - Cache intermediate results for every subexpression of bound_sympy
537                  # - This cache should be possible to edit to refine ranges
538                  #
539                  # One issue with this proposal is that if
540                  # we have a bound on 2x, we are not going to be able to
541                  # apply it for 4x.  Similarly, we may have bounds for an
542                  # equivalent expression that we are not applying because
543                  # it's not a perfect match (e.g. x < y vs y > x)".
544                  #
545                  # The first issue we already have it and it's impossible
546                  # to solve in general, so any implementation on a best
547                  # effort basis should do.
548                  #
549                  # The second issue is a preexisting one. It can be mitigated
550                  # with a normalization algorithm. In general, it may also
551                  # be on a best effort basis, but since our grammar is not
552                  # terribly difficult, chances are we could even fully
553                  # normalize SymPy expressions... who knows.
554                  if i0 in constrained_unbacked_symbols:
555                      continue  # constrain symbol just once
556  
557                  if i0 in shape_env.size_like:
558                      if export:
559                          graph.call_function(
560                              torch.ops.aten.sym_constrain_range_for_size.default,
561                              (expr_to_proxy[i0].node,),
562                          )
563                      else:
564                          graph.call_function(
565                              torch._check_is_size, (expr_to_proxy[i0].node,)
566                          )
567  
568                  vr = shape_env.var_to_range[i0]
569                  if vr.is_int and vr.upper == sys.maxsize - 1:
570                      # treat upper bound == sys.maxsize - 1 for int symbols as +oo
571                      # to avoid redundant runtime assert
572                      vr = ValueRanges(vr.lower, int_oo)
573                  if not shape_env._default_unspecified_value_range().issubset(vr):
574                      # The runtime range is constrained, so add a runtime
575                      # assert and also explicitly refine the range
576                      # (refinement should not be necessary once runtime
577                      # asserts cause refinement, but that's NYI)
578                      def convert(s):
579                          if s in (int_oo, -int_oo):
580                              return None
581                          try:
582                              return int(s)
583                          except TypeError:
584                              return None
585  
586                      if (
587                          expr_to_proxy[i0].node.target
588                          != cast_symbool_to_symint_guardless
589                      ):
590                          # TODO(pianpwk): calling sym_constrain_range_for_size or adding bound asserts
591                          # raises AOTAutograd errors on cast_symbool_to_symint_guardless
592  
593                          with _set_node_metadata_hook(
594                              gm,
595                              functools.partial(
596                                  _node_metadata_hook,
597                                  stack_trace=node.meta.get("stack_trace"),
598                                  nn_module_stack=node.meta.get("nn_module_stack"),
599                              ),
600                          ):
601                              if (min_val := convert(vr.lower)) is not None:
602                                  ge = _sympy_interp(expr_to_proxy, i0 >= min_val).node
603                                  graph.call_function(
604                                      torch.ops.aten._assert_scalar.default,
605                                      (
606                                          ge,
607                                          f"Runtime assertion failed for expression {i0 >= min_val} on node '{ge}'",
608                                      ),
609                                  )
