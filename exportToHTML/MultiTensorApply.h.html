119  std::pair<id<MTLComputePipelineState>, id<MTLFunction>> getFusedAdamCPLState(const std::string& fname);
120  template <int depth, uint32_t kThreadGroupSize, typename encoder_func_t, typename... ArgTypes>
121  static void multi_tensor_apply_for_fused_optimizer(const std::string& kernel_name,
122                                                     std::vector<std::vector<at::Tensor>>& tensor_lists,
123                                                     at::TensorList state_steps,
124                                                     encoder_func_t encode,
125                                                     ArgTypes... args) {
126    const auto num_tensors = tensor_lists[0].size();
127  
128    if (num_tensors == 0) {
129      return;
130    }
131  
132    TORCH_CHECK(tensor_lists.size() == depth, "Number of tensor lists has to match the depth");
133    for (const auto& d : c10::irange(depth)) {
134      const auto scalar_type = tensor_lists[d][0].scalar_type();
135      TORCH_CHECK(scalar_type == kFloat || scalar_type == kHalf || scalar_type == kBFloat16,
136                  "Only float, bfloat and half are supported");
137    }
138  
139    id<MTLDevice> device = MPSDevice::getInstance()->device();
140    MPSStream* mpsStream = getCurrentMPSStream();
141  
142    // Remove comment for debugging
143    /* 
144    mpsStream->addCompletedHandler(^(id<MTLCommandBuffer> cb) { 
145      [cb.logs enumerateObjectsUsingBlock:^(NSString* log, NSUInteger idx, BOOL* stop) { 
146        NSLog(@"MPSStream: %@", log); 
147        } 
148      ]; 
149    }); 
150    */
151  
152    dispatch_sync_with_rethrow(mpsStream->queue(), ^() {
153      @autoreleasepool {
154        id<MTLComputeCommandEncoder> computeEncoder = mpsStream->commandEncoder();
155        auto [fusedOptimizerPSO, fusedOptimizerFunc] = getFusedAdamCPLState(kernel_name);
156  
157        // this function call is a no-op if MPS Profiler is not enabled
158        getMPSProfiler().beginProfileKernel(fusedOptimizerPSO, kernel_name, {tensor_lists[0]});
159  
160        [computeEncoder setComputePipelineState:fusedOptimizerPSO];
161  
162        // BufferIndex is the index in the kernel function
163        auto tensorArgumentEncoder = [[fusedOptimizerFunc newArgumentEncoderWithBufferIndex:0] autorelease];
164        id<MTLBuffer> tensorArgumentBuffer = [[device newBufferWithLength:tensorArgumentEncoder.encodedLength
165                                                                  options:0] autorelease];
166        [tensorArgumentEncoder setArgumentBuffer:tensorArgumentBuffer offset:0];
167  
168        int64_t tensor_loc = 0;
169        int64_t threadgroup_loc = 0;
170        MetadataArguments metadata_arguments;
171  
172        for (const auto tensor_index : c10::irange(num_tensors)) {
173          // short-circuit to avoid adding empty tensors to tensorListMeta
174          if (tensor_lists[0][tensor_index].numel() == 0) {
175            continue;
176          }
177  
178          for (const auto& d : c10::irange(depth)) {
179            mtl_setBuffer(tensorArgumentEncoder, tensor_lists[d][tensor_index], d * kmaxTensors + tensor_loc);
180            [computeEncoder useResource:getMTLBufferStorage(tensor_lists[d][tensor_index])
181                                  usage:MTLResourceUsageRead | MTLResourceUsageWrite];
182          }
183          if (!state_steps.empty()) {
184            mtl_setBuffer(tensorArgumentEncoder, state_steps[tensor_index], depth * kmaxTensors + tensor_loc);
185            [computeEncoder useResource:getMTLBufferStorage(state_steps[tensor_index]) usage:MTLResourceUsageRead];
186          }
187          metadata_arguments.numels[tensor_loc] = tensor_lists[0][tensor_index].numel();
188  
189          tensor_loc++;
190  
191          const auto numel = tensor_lists[0][tensor_index].numel();
192          const auto chunks = numel / kChunkSize + (numel % kChunkSize != 0);
193          TORCH_CHECK(chunks > -1);
194  
195          for (const auto& chunk : c10::irange(chunks)) {
196            metadata_arguments.threadgroup_to_tensor[threadgroup_loc] = tensor_loc - 1;
197            metadata_arguments.threadgroup_to_chunk[threadgroup_loc] = chunk;
198  
199            threadgroup_loc++;
200  
201            const auto tensor_full = tensor_loc == kmaxTensors && chunk == chunks - 1;
202            // Reach the maximum threadgroups per dispatch
203            const auto blocks_full = threadgroup_loc == kmaxThreadGroups;
204  
205            if (tensor_full || blocks_full) {
206              encode(computeEncoder, tensorArgumentBuffer, metadata_arguments, args...);
207              MTLSize gridSize = MTLSizeMake(threadgroup_loc, 1, 1);
208              uint32_t maxThreadsPerGroup = [fusedOptimizerPSO maxTotalThreadsPerThreadgroup];
209              MTLSize threadGroupSize = MTLSizeMake(std::min(maxThreadsPerGroup, kThreadGroupSize), 1, 1);
210              [computeEncoder dispatchThreadgroups:gridSize threadsPerThreadgroup:threadGroupSize];
211  
212              // Reset
213              threadgroup_loc = 0;
214              if (chunk == chunks - 1) {
215                // last chunk
216                tensor_loc = 0;
217                tensorArgumentBuffer = [[device newBufferWithLength:tensorArgumentEncoder.encodedLength
218                                                            options:0] autorelease];
219                [tensorArgumentEncoder setArgumentBuffer:tensorArgumentBuffer offset:0];
220              } else {
221                // reuse the current tensor since the current one isn't done.
222                metadata_arguments.numels[0] = metadata_arguments.numels[tensor_loc - 1];
223  
224                tensorArgumentBuffer = [[device newBufferWithLength:tensorArgumentEncoder.encodedLength
225                                                            options:0] autorelease];
226                [tensorArgumentEncoder setArgumentBuffer:tensorArgumentBuffer offset:0];
227  
228                for (const auto& d : c10::irange(depth)) {
229                  mtl_setBuffer(tensorArgumentEncoder, tensor_lists[d][tensor_index], d * kmaxTensors);
230                  [computeEncoder useResource:getMTLBufferStorage(tensor_lists[d][tensor_index])
231                                        usage:MTLResourceUsageWrite | MTLResourceUsageRead];
232                }
233                if (!state_steps.empty()) {
234                  mtl_setBuffer(tensorArgumentEncoder, state_steps[tensor_index], depth * kmaxTensors);
235                  [computeEncoder useResource:getMTLBufferStorage(state_steps[tensor_index]) usage:MTLResourceUsageRead];
236                }
237                tensor_loc = 1;
238              }
239            }
240          }
241        }
242  
243        if (threadgroup_loc != 0) {
244          encode(computeEncoder, tensorArgumentBuffer, metadata_arguments, args...);
245          MTLSize gridSize = MTLSizeMake(threadgroup_loc, 1, 1);
246          uint32_t maxThreadsPerGroup = [fusedOptimizerPSO maxTotalThreadsPerThreadgroup];
247          MTLSize threadGroupSize = MTLSizeMake(std::min(maxThreadsPerGroup, kThreadGroupSize), 1, 1);
248          [computeEncoder dispatchThreadgroups:gridSize threadsPerThreadgroup:threadGroupSize];
249        }
250  
251        getMPSProfiler().endProfileKernel(fusedOptimizerPSO);
252      }
253    });
254  }

256  std::pair<id<MTLComputePipelineState>, id<MTLFunction>> getAmpCPLState(const std::string& fname);
257  template <int depth, typename... ArgTypes>
258  void multi_tensor_apply(const std::string& kernel_name,
259                          std::vector<std::vector<at::Tensor>>& tensor_lists,
260                          ArgTypes... args) {
261    const auto num_tensors = tensor_lists[0].size();
262    if (num_tensors == 0) {
263      return;
264    }
265  
266    TORCH_CHECK(tensor_lists.size() == depth, "Number of tensor lists must match depth.");
267  
268    id<MTLDevice> device = MPSDevice::getInstance()->device();
269    MPSStream* mpsStream = getCurrentMPSStream();
270  
271    dispatch_sync_with_rethrow(mpsStream->queue(), ^() {
272      @autoreleasepool {
273        id<MTLComputeCommandEncoder> computeEncoder = mpsStream->commandEncoder();
274        auto [pipeline, function] = getAmpCPLState(kernel_name);
275        [computeEncoder setComputePipelineState:pipeline];
276  
277        id<MTLArgumentEncoder> argumentEncoder = [function newArgumentEncoderWithBufferIndex:0];
278        auto tensorArgumentBuffer = [[device newBufferWithLength:argumentEncoder.encodedLength options:0] autorelease];
279        [argumentEncoder setArgumentBuffer:tensorArgumentBuffer offset:0];
280  
281        int tensor_loc = 0;
282        int threadgroup_loc = 0;
283        MetadataArguments metadata_arguments;
284        std::memset(&metadata_arguments, 0, sizeof(metadata_arguments));
285  
286        for (size_t t = 0; t < num_tensors; t++) {
287          if (tensor_lists[0][t].numel() == 0)
288            continue;
289  
290          // bind each tensor in this list to the correct slots across depths
291          for (int d = 0; d < depth; d++) {
292            mtl_setBuffer(argumentEncoder, tensor_lists[d][t], d * kmaxTensors + tensor_loc);
293            [computeEncoder useResource:getMTLBufferStorage(tensor_lists[d][t])
294                                  usage:(MTLResourceUsageRead | MTLResourceUsageWrite)];
295          }
296  
297          // save number of elements for this tensor
298          metadata_arguments.numels[tensor_loc] = tensor_lists[0][t].numel();
299          int currentTensorIndex = tensor_loc;
300          tensor_loc++;
301  
302          const auto numel = tensor_lists[0][t].numel();
303          const auto chunks = numel / kChunkSize + ((numel % kChunkSize) ? 1 : 0);
304  
305          // process tensor in chunks based on max chunk size
306          for (uint chunk = 0; chunk < chunks; chunk++) {
307            metadata_arguments.threadgroup_to_tensor[threadgroup_loc] = currentTensorIndex;
308            metadata_arguments.threadgroup_to_chunk[threadgroup_loc] = chunk;
309            threadgroup_loc++;
310  
311            // dispatch when we've filled the threadgroup array or finished the chunks
312            const bool dispatch_now = (threadgroup_loc == kmaxThreadGroups) || (chunk == chunks - 1);
313            if (dispatch_now) {
314              // check for a partial dispatch (i.e. more chunks remain for the current tensor)
315              bool partial = (chunk != chunks - 1);
316              uint carried_numels = 0;
317              if (partial) {
318                carried_numels = metadata_arguments.numels[currentTensorIndex];
319              }
320  
321              mtl_setArgs(computeEncoder, tensorArgumentBuffer, metadata_arguments, args...);
322              MTLSize gridSize = MTLSizeMake(threadgroup_loc, 1, 1);
323              uint32_t maxThreads = [pipeline maxTotalThreadsPerThreadgroup];
324              MTLSize threadGroupSize = MTLSizeMake(std::min(maxThreads, (uint32_t)64), 1, 1);
325              [computeEncoder dispatchThreadgroups:gridSize threadsPerThreadgroup:threadGroupSize];
326  
327              // prepare for the next batch: reset threadgroup count and create a new buffer
328              threadgroup_loc = 0;
329              tensorArgumentBuffer = [[device newBufferWithLength:argumentEncoder.encodedLength options:0] autorelease];
330              [argumentEncoder setArgumentBuffer:tensorArgumentBuffer offset:0];
331  
332              if (partial) {
333                // for a partial dispatch, rebind the partially processed tensor to slot 0
334                // so that its metadata is in the correct location
335                for (int d = 0; d < depth; d++) {
336                  mtl_setBuffer(argumentEncoder, tensor_lists[d][t], d * kmaxTensors + 0);
337                  [computeEncoder useResource:getMTLBufferStorage(tensor_lists[d][t])
338                                        usage:(MTLResourceUsageRead | MTLResourceUsageWrite)];
339                }
340                metadata_arguments.numels[0] = carried_numels;
341                // the currently processed tensor now lives at index 0
342                currentTensorIndex = 0;
343                tensor_loc = 1;
344              } else {
345                tensor_loc = 0;
346              }
347            }
348          }
349        }
350  
351        if (threadgroup_loc != 0) {
352          mtl_setArgs(computeEncoder, tensorArgumentBuffer, metadata_arguments, args...);
353          MTLSize gridSize = MTLSizeMake(threadgroup_loc, 1, 1);
354          uint32_t maxThreads = [pipeline maxTotalThreadsPerThreadgroup];
355          MTLSize threadGroupSize = MTLSizeMake(std::min(maxThreads, static_cast<uint32_t>(64)), 1, 1);
356          [computeEncoder dispatchThreadgroups:gridSize threadsPerThreadgroup:threadGroupSize];
357        }
358      }
359    });
