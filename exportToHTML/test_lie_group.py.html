23   def test_heuristic1():
24       a, b, c, a4, a3, a2, a1, a0 = symbols("a b c a4 a3 a2 a1 a0")
25       df = f(x).diff(x)
26       eq = Eq(df, x**2*f(x))
27       eq1 = f(x).diff(x) + a*f(x) - c*exp(b*x)
28       eq2 = f(x).diff(x) + 2*x*f(x) - x*exp(-x**2)
29       eq3 = (1 + 2*x)*df + 2 - 4*exp(-f(x))
30       eq4 = f(x).diff(x) - (a4*x**4 + a3*x**3 + a2*x**2 + a1*x + a0)**Rational(-1, 2)
31       eq5 = x**2*df - f(x) + x**2*exp(x - (1/x))
32       eqlist = [eq, eq1, eq2, eq3, eq4, eq5]
33   
34       i = infinitesimals(eq, hint='abaco1_simple')
35       assert i == [{eta(x, f(x)): exp(x**3/3), xi(x, f(x)): 0},
36           {eta(x, f(x)): f(x), xi(x, f(x)): 0},
37           {eta(x, f(x)): 0, xi(x, f(x)): x**(-2)}]
38       i1 = infinitesimals(eq1, hint='abaco1_simple')
39       assert i1 == [{eta(x, f(x)): exp(-a*x), xi(x, f(x)): 0}]
40       i2 = infinitesimals(eq2, hint='abaco1_simple')
41       assert i2 == [{eta(x, f(x)): exp(-x**2), xi(x, f(x)): 0}]
42       i3 = infinitesimals(eq3, hint='abaco1_simple')
43       assert i3 == [{eta(x, f(x)): 0, xi(x, f(x)): 2*x + 1},
44           {eta(x, f(x)): 0, xi(x, f(x)): 1/(exp(f(x)) - 2)}]
45       i4 = infinitesimals(eq4, hint='abaco1_simple')
46       assert i4 == [{eta(x, f(x)): 1, xi(x, f(x)): 0},
47           {eta(x, f(x)): 0,
48           xi(x, f(x)): sqrt(a0 + a1*x + a2*x**2 + a3*x**3 + a4*x**4)}]
49       i5 = infinitesimals(eq5, hint='abaco1_simple')
50       assert i5 == [{xi(x, f(x)): 0, eta(x, f(x)): exp(-1/x)}]
51   
52       ilist = [i, i1, i2, i3, i4, i5]
53       for eq, i in (zip(eqlist, ilist)):
54           check = checkinfsol(eq, i)
55           assert check[0]
56   
57       # This ODE can be solved by the Lie Group method, when there are
58       # better assumptions
59       eq6 = df - (f(x)/x)*(x*log(x**2/f(x)) + 2)
60       i = infinitesimals(eq6, hint='abaco1_product')
61       assert i == [{eta(x, f(x)): f(x)*exp(-x), xi(x, f(x)): 0}]
62       assert checkinfsol(eq6, i)[0]
63   
64       eq7 = x*(f(x).diff(x)) + 1 - f(x)**2
65       i = infinitesimals(eq7, hint='chi')
66       assert checkinfsol(eq7, i)[0]

105  def test_heuristic_abaco2_unique_unknown():
106  
107      a, b = symbols("a b")
108      F = Function('F')
109      eq = f(x).diff(x) - x**(a - 1)*(f(x)**(1 - b))*F(x**a/a + f(x)**b/b)

140  def test_user_infinitesimals():
141      x = Symbol("x") # assuming x is real generates an error
142      eq = x*(f(x).diff(x)) + 1 - f(x)**2
143      sol = Eq(f(x), (C1 + x**2)/(C1 - x**2))
144      infinitesimals = {'xi':sqrt(f(x) - 1)/sqrt(f(x) + 1), 'eta':0}
145      assert dsolve(eq, hint='lie_group', **infinitesimals) == sol
146      assert checkodesol(eq, sol) == (True, 0)
