32   @pytest.mark.slow
33   def test_mixed_threads_processes(x):
34       # Test that the fft threadpool is safe to use before & after fork
35   
36       expect = fft.fft(x, workers=2)
37   
38       with multiprocessing.Pool(2) as p:
39           res = p.map(_mt_fft, [x for _ in range(4)])
40   
41       for r in res:
42           assert_allclose(r, expect)
43   
44       fft.fft(x, workers=2)
