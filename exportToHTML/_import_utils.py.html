9    def _check_module_exists(name: str) -> bool:
10       r"""Returns if a top-level module with :attr:`name` exists *without** 
11       importing it. This is generally safer than try-catch block around a 
12       `import X`. It avoids third party libraries breaking assumptions of some of 
13       our tests, e.g., setting multiprocessing start method when imported 
14       (see librosa/#747, torchvision/#544). 
15       """
16       try:
17           spec = importlib.util.find_spec(name)
18           return spec is not None
19       except ImportError:
20           return False

23   @functools.lru_cache
24   def dill_available() -> bool:
25       return (
26           _check_module_exists("dill")
27           # dill fails to import under torchdeploy
28           and not torch._running_with_deploy()
29       )
30   
31   
32   @functools.lru_cache
33   def import_dill() -> Optional[ModuleType]:
34       if not dill_available():
35           return None
36   
37       import dill
38   
39       # XXX: By default, dill writes the Pickler dispatch table to inject its
40       # own logic there. This globally affects the behavior of the standard library
41       # pickler for any user who transitively depends on this module!
42       # Undo this extension to avoid altering the behavior of the pickler globally.
43       dill.extend(use_dill=False)
44       return dill
