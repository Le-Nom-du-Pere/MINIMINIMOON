3    from pip._internal.utils.misc import splitext
4    
5    WHEEL_EXTENSION = ".whl"
6    BZ2_EXTENSIONS: tuple[str, ...] = (".tar.bz2", ".tbz")
7    XZ_EXTENSIONS: tuple[str, ...] = (
8        ".tar.xz",
9        ".txz",
10       ".tlz",
11       ".tar.lz",
12       ".tar.lzma",
13   )
14   ZIP_EXTENSIONS: tuple[str, ...] = (".zip", WHEEL_EXTENSION)
15   TAR_EXTENSIONS: tuple[str, ...] = (".tar.gz", ".tgz", ".tar")
16   ARCHIVE_EXTENSIONS = ZIP_EXTENSIONS + BZ2_EXTENSIONS + TAR_EXTENSIONS + XZ_EXTENSIONS
17   
18   
19   def is_archive_file(name: str) -> bool:
20       """Return True if `name` is a considered as an archive file."""
21       ext = splitext(name)[1].lower()
22       if ext in ARCHIVE_EXTENSIONS:
23           return True
24       return False
