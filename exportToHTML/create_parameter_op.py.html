9    # See [Note: Metadata mutation in proxy tracing] for why sacrificial parameter mutates
10   # metadata during proxy tracing and we should remove the sacrificial parameter logic.
11   doc = """ 
12   This is used when dynamo traces torch.nn.Parameter, which normally would not trace properly 
13   with AOTAutograd.  We instead create a placeholder torch.nn.Parameter before the graph, which 
14   becomes a graph arg and has no storage backing it.  At the point in the graph where the parameter 
15   actually should be created we mutate this sacrificial placeholder into it.  This allows gradients 
16   to flow into the parameter as if it were an input to the graph (which is the only thing we are 
17   allowed to compute gradients on). 
18   """.strip()
19

21   class TracableCreateParameter(torch.autograd.Function):
22       @staticmethod
23       def forward(ctx: Any, tensor: Any, placeholder: Any) -> torch.nn.Parameter:
24           assert not tensor.requires_grad
25           return placeholder.set_(tensor)
26   
27       @staticmethod
28       def backward(ctx: Any, *grad_outputs: torch.Tensor) -> tuple[None, torch.Tensor]:
29           grad = grad_outputs[0]
30           return None, grad  # grad flows to placeholder
31   
32   
33   def tracable_create_parameter(
34       tensor: torch.Tensor, placeholder: torch.nn.Parameter
35   ) -> torch.nn.Parameter:
36       with torch.set_grad_enabled(placeholder.requires_grad):
37           out = TracableCreateParameter.apply(tensor, placeholder)
38       return out

41   def new_parameter_placeholder(
42       size: tuple[int, ...], dtype: torch.dtype, device: torch.device, requires_grad: bool
43   ) -> torch.nn.Parameter:
44       """Create a placeholder to be passed to the above functions"""
45       result = torch.nn.Parameter(
46           torch.empty(size, dtype=dtype, device=device), requires_grad=requires_grad
47       )
48       # TODO(jansel): alloc followed by free is inefficient, need a way to allocate an unbacked tensor.
49       # Allocating a zero tensor would causes assert failures in autograd.
50       result.untyped_storage().resize_(0)
51       return result

57   @contextmanager
58   def do_not_convert_to_tracable_parameter() -> Generator[bool, None, None]:
59       old_flag = getattr(_TLS, "convert_tracable_parameter", True)
60       _TLS.convert_tracable_parameter = False
61       try:
62           yield False
63       finally:
64           _TLS.convert_tracable_parameter = old_flag

66   
67   def can_convert_to_tracable_parameter() -> bool:
68       return getattr(_TLS, "convert_tracable_parameter", True)
