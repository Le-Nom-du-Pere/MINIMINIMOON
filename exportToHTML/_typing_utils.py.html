_typing_utils.py 
1    """Miscellaneous utilities to aid with typing."""
2    
3    from typing import Optional, TypeVar
4    
5    
6    # Helper to turn Optional[T] into T when we know None either isn't
7    # possible or should trigger an exception.
8    T = TypeVar("T")
9    
10   
11   def not_none(obj: Optional[T]) -> T:
12       if obj is None:
13           raise TypeError("Invariant encountered: value was None when it should not be")
14       return obj
