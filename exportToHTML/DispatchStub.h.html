10   // Implements instruction set specific function dispatch.
11   //
12   // Kernels that may make use of specialized instruction sets (e.g. AVX2) are
13   // compiled multiple times with different compiler flags (e.g. -mavx2). A
14   // DispatchStub contains a table of function pointers for a kernel. At runtime,
15   // the fastest available kernel is chosen based on the features reported by
16   // cpuinfo.
17   //
18   // Example:
19   //
20   // In native/MyKernel.h:
21   //   using fn_type = void(*)(const Tensor& x);
22   //   DECLARE_DISPATCH(fn_type, stub)
23   //
24   // In native/MyKernel.cpp
25   //   DEFINE_DISPATCH(stub);
26   //
27   // In native/cpu/MyKernel.cpp:
28   //   namespace {
29   //     // use anonymous namespace so that different cpu versions won't conflict
30   //     void kernel(const Tensor& x) { ... }
31   //   }
32   //   REGISTER_DISPATCH(stub, &kernel);
33   //
34   // To call:
35   //   stub(kCPU, tensor);
36   //
37   // TODO: CPU instruction set selection should be folded into whatever
38   // the main dispatch mechanism is.
39   //
40   // Supported device types for registration:
41   //   - CPU: Central Processing Unit
42   //   - CUDA: NVIDIA GPUs
43   //   - HIP: AMD GPUs
44   //   - MPS: Apple Silicon GPUs (Metal Performance Shaders)
45   //   - MTIA: Meta Training and Inference Devices
46   //   - XPU: Intel GPUs
47   //   - HPU: Reserved for HPU (Intel Gaudi) device types
48   //   - PrivateUse1: Reserved for private/custom device types
49   //
50   // If you want to update the list of supported devices, add a new dispatch_ptr
51   // member in DispatchStubImpl.h and update the get_call_ptr switch.
52   // As well you will need to update the inlined list in 'is_device_supported`
53   //
54   //
55   // ignore warnings about DispatchStub::DEFAULT, AVX, AVX2 defined elsewhere

216  template <typename rT, typename T, typename... Args>
217  struct DispatchStub<rT (*)(Args...), T> {
218    using FnPtr = rT (*) (Args...);
219  
220    DispatchStub() = default;
221    DispatchStub(const DispatchStub&) = delete;
222    DispatchStub& operator=(const DispatchStub&) = delete;
223  
224  private:
225    FnPtr get_call_ptr(const c10::DeviceType device_type) {
226      return reinterpret_cast<FnPtr>(
227        impl.get_call_ptr(device_type
228        , reinterpret_cast<void*>(DEFAULT)
229  #ifdef HAVE_AVX512_CPU_DEFINITION
230        , reinterpret_cast<void*>(AVX512)
231  #endif
232  #ifdef HAVE_AVX2_CPU_DEFINITION
233        , reinterpret_cast<void*>(AVX2)
234  #endif
235  #ifdef HAVE_VSX_CPU_DEFINITION
236        , reinterpret_cast<void*>(VSX)
237  #endif
238  #ifdef HAVE_ZVECTOR_CPU_DEFINITION
239        , reinterpret_cast<void*>(ZVECTOR)
240  #endif
241  #ifdef HAVE_SVE256_CPU_DEFINITION
242        , reinterpret_cast<void*>(SVE256)
243  #endif
244        )
245      );
246    }
247  
248  public:
249    template <typename... ArgTypes>
250    rT operator()(c10::DeviceType device_type, ArgTypes&&... args) {
251      FnPtr call_ptr = get_call_ptr(device_type);
252      return (*call_ptr)(std::forward<ArgTypes>(args)...);
253    }
254  
255    void set_cuda_dispatch_ptr(FnPtr fn_ptr) {
256      impl.cuda_dispatch_ptr = reinterpret_cast<void*>(fn_ptr);
257    }
258  
259    #if defined(USE_XPU)
260    void set_xpu_dispatch_ptr(FnPtr fn_ptr){
261      impl.xpu_dispatch_ptr = reinterpret_cast<void*>(fn_ptr);
262    }
263    #endif
264  
265    void set_hpu_dispatch_ptr(FnPtr fn_ptr) {
266      impl.hpu_dispatch_ptr = reinterpret_cast<void*>(fn_ptr);
267    }
268  
269    void set_hip_dispatch_ptr(FnPtr fn_ptr) {
270      impl.hip_dispatch_ptr = reinterpret_cast<void*>(fn_ptr);
271    }
272  
273    void set_mps_dispatch_ptr(FnPtr fn_ptr) {
274      impl.mps_dispatch_ptr = reinterpret_cast<void*>(fn_ptr);
275    }
276  
277      void set_mtia_dispatch_ptr(FnPtr fn_ptr) {

281    void set_privateuse1_dispatch_ptr(FnPtr fn_ptr) {

285    // Returns true if the dispatcher has a kernel registered for this device
286    // type.
287    bool is_device_supported(const c10::DeviceType device_type) {
288      auto result = impl.try_get_call_ptr(device_type
289        , reinterpret_cast<void*>(DEFAULT)
290  #ifdef HAVE_AVX512_CPU_DEFINITION
291        , reinterpret_cast<void*>(AVX512)
292  #endif
293  #ifdef HAVE_AVX2_CPU_DEFINITION
294        , reinterpret_cast<void*>(AVX2)
295  #endif
296  #ifdef HAVE_VSX_CPU_DEFINITION
297        , reinterpret_cast<void*>(VSX)
298  #endif
299  #ifdef HAVE_ZVECTOR_CPU_DEFINITION
300        , reinterpret_cast<void*>(ZVECTOR)
301  #endif
302  #ifdef HAVE_SVE256_CPU_DEFINITION
303        , reinterpret_cast<void*>(SVE256)
304  #endif
305        );
306      if (std::holds_alternative<ErrorType>(result)){
307        return false;
308      }
309      return true;
310    }

384  // Compiler will complain if you put things like std::tuple<Tensor, Tensor> in
385  // the `fn` argument of DECLARE_DISPATCH. Some possible workarounds, e.g.,
386  // adding parentheses and using helper struct to get rid of the parentheses, do
387  // not work with MSVC. So do a `using`-declaration if you need to pass in such
388  // `fn`, e.g., grid_sampler_2d_backward_cpu_kernel in GridSampleKernel.h.
389  #define DECLARE_DISPATCH(fn, name)                                                         \
390    struct name##_DECLARE_DISPATCH_type : DispatchStub<fn, name##_DECLARE_DISPATCH_type> {   \
391      name##_DECLARE_DISPATCH_type() = default;                                              \
392      name##_DECLARE_DISPATCH_type(const name##_DECLARE_DISPATCH_type&) = delete;            \
393      name##_DECLARE_DISPATCH_type& operator=(const name##_DECLARE_DISPATCH_type&) = delete; \
394      name##_DECLARE_DISPATCH_type(name##_DECLARE_DISPATCH_type&&) = delete;                 \
395      name##_DECLARE_DISPATCH_type& operator=(name##_DECLARE_DISPATCH_type&&) = delete;      \
396      ~name##_DECLARE_DISPATCH_type() = default;                                             \
397    };                                                                                       \
398    extern TORCH_API struct name##_DECLARE_DISPATCH_type name;
399  
400  #define DEFINE_DISPATCH(name) struct name##_DECLARE_DISPATCH_type name
401  
402  #define REGISTER_ARCH_DISPATCH(name, arch, fn) \
403    template <> name##_DECLARE_DISPATCH_type::FnPtr TORCH_API DispatchStub<name##_DECLARE_DISPATCH_type::FnPtr, struct name##_DECLARE_DISPATCH_type>::arch = fn;
