14   def execute(
15       gm: GraphModule,
16       *args: Unpack[Ts],
17       executor: str = "aten",
18       executor_parameters: Optional[dict] = None,
19   ) -> Any:
20       """ 
21       Prototype ATen executor. 
22    
23       Just executes the context's graph. 
24       """
25   
26       if executor == "aten":
27           return gm.forward(*args)
28   
29       msg = f"Received unexpected value for 'executor': {executor}. Allowed values are: aten."
30       raise ValueError(msg)

33   def make_traced(fn: Callable[P, T]) -> Callable[P, T]:
34       """ 
35       Returns a function that, when called, will 
36       trace its torch operations to prims and then 
37       execute those prims on the requested trace executor 
38       (possibly lowering them to that trace executor first). 
39    
40       Only supports the torch operations defined in _torch_to_reference_map 
41       in context.py and operations with positional args. All args must 
42       be tensors. 
43       In the near future all these restrictions will be lifted. 
44    
45       Example usage: 
46    
47       def foo(a, b): 
48         return torch.add(a, b) 
49    
50       traced_foo = make_traced(foo) 
51    
52       a = torch.randn((1, 2, 3, 4, 5), device='cuda') 
53       b = torch.randn((1, 2, 3, 4, 5), device='cuda') 
54       result = traced_foo(a, b, executor='aten') 
55       """
56   
57       def _traced(*args: P.args, **kwargs: P.kwargs) -> T:
58           executor = str(kwargs.pop("executor", "aten"))
59   
60           # TODO: caching
61           wrapped, all_args = wrapper_and_args_for_make_fx(fn, args, kwargs)
62   
63           with TorchRefsMode():
64               gm = make_fx(wrapped)(all_args)
65           return execute(gm, all_args, executor=executor)
66   
67       return _traced  # type: ignore[return-value]
