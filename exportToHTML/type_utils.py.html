20   # Enable both TorchScriptTensor and torch.Tensor to be tested
21   # for dtype in OpSchemaWrapper.
22   @runtime_checkable
23   class TensorLike(Protocol):
24       @property
25       def dtype(self) -> torch.dtype | None: ...

28   def is_torch_complex_dtype(tensor_dtype: torch.dtype) -> bool:
29       # NOTE: This is needed as TorchScriptTensor is nor supported by torch.is_complex()
30       return tensor_dtype in _COMPLEX_TO_FLOAT
31   
32   
33   def from_complex_to_float(dtype: torch.dtype) -> torch.dtype:
34       return _COMPLEX_TO_FLOAT[dtype]
35   
36   
37   def from_sym_value_to_torch_dtype(sym_value: SYM_VALUE_TYPE) -> torch.dtype:
38       return _SYM_TYPE_TO_TORCH_DTYPE[type(sym_value)]
39   
40   
41   def is_optional_onnx_dtype_str(onnx_type_str: str) -> bool:
42       return onnx_type_str in _OPTIONAL_ONNX_DTYPE_STR
43   
44   
45   def from_torch_dtype_to_onnx_dtype_str(dtype: torch.dtype | type) -> set[str]:
46       return _TORCH_DTYPE_TO_COMPATIBLE_ONNX_TYPE_STRINGS[dtype]
47   
48   
49   def from_python_type_to_onnx_attribute_type(
50       dtype: type, is_sequence: bool = False
51   ) -> onnx.defs.OpSchema.AttrType | None:
52       import onnx.defs  # type: ignore[import]
53   
54       _PYTHON_TYPE_TO_ONNX_ATTRIBUTE_TYPE = {
55           float: onnx.defs.OpSchema.AttrType.FLOAT,
56           int: onnx.defs.OpSchema.AttrType.INT,
57           str: onnx.defs.OpSchema.AttrType.STRING,
58           bool: onnx.defs.OpSchema.AttrType.INT,
59       }
60   
61       _SEQUENCE_TYPE_TO_ONNX_ATTRIBUTE_TYPE = {
62           float: onnx.defs.OpSchema.AttrType.FLOATS,
63           int: onnx.defs.OpSchema.AttrType.INTS,
64           str: onnx.defs.OpSchema.AttrType.STRINGS,
65           bool: onnx.defs.OpSchema.AttrType.INTS,
66       }
67   
68       if is_sequence:
69           return _SEQUENCE_TYPE_TO_ONNX_ATTRIBUTE_TYPE.get(dtype)
70       return _PYTHON_TYPE_TO_ONNX_ATTRIBUTE_TYPE.get(dtype)
71   
72   
73   def is_torch_symbolic_type(value: Any) -> bool:
74       return isinstance(value, (torch.SymBool, torch.SymInt, torch.SymFloat))
75   
76   
77   def from_torch_dtype_to_abbr(dtype: torch.dtype | None) -> str:
78       if dtype is None:
79           return ""
80       return _TORCH_DTYPE_TO_ABBREVIATION.get(dtype, "")
81   
82   
83   def from_scalar_type_to_torch_dtype(scalar_type: type) -> torch.dtype | None:
84       return _SCALAR_TYPE_TO_TORCH_DTYPE.get(scalar_type)
85

88   # It's used in dispatcher to find the best match overload for the input dtypes
89   _TORCH_DTYPE_TO_COMPATIBLE_ONNX_TYPE_STRINGS: dict[torch.dtype | type, set[str]] = {

92       torch.float64: {"tensor(double)"},
93       torch.float32: {"tensor(float)"},
94       torch.float16: {"tensor(float16)"},
95       torch.float8_e4m3fn: {"tensor(float8_e4m3fn)"},
96       torch.float8_e4m3fnuz: {"tensor(float8_e4m3fnuz)"},
97       torch.float8_e5m2: {"tensor(float8_e5m2)"},
98       torch.float8_e5m2fnuz: {"tensor(float8_e5m2fnuz)"},
99       torch.int16: {"tensor(int16)"},
100      torch.int32: {"tensor(int32)"},
101      torch.int64: {"tensor(int64)"},
102      torch.int8: {"tensor(int8)"},
103      torch.uint8: {"tensor(uint8)"},
104      str: {"tensor(string)"},
105      int: {"tensor(int16)", "tensor(int32)", "tensor(int64)"},
106      float: {"tensor(float16)", "tensor(float)", "tensor(double)"},
107      bool: {"tensor(int32)", "tensor(int64)", "tensor(bool)"},
108      complex: {"tensor(float)", "tensor(double)"},
109      torch.complex32: {"tensor(float16)"},
110      torch.complex64: {"tensor(float)"},
111      torch.complex128: {"tensor(double)"},

114  _OPTIONAL_ONNX_DTYPE_STR: set[str] = {
115      f"optional({value})"
116      for value_set in _TORCH_DTYPE_TO_COMPATIBLE_ONNX_TYPE_STRINGS.values()
117      for value in value_set
118  }

120  _PYTHON_TYPE_TO_TORCH_DTYPE = {
121      bool: torch.bool,
122      int: torch.int64,

124      complex: torch.complex64,

127  _COMPLEX_TO_FLOAT: dict[torch.dtype, torch.dtype] = {
128      torch.complex32: torch.float16,
129      torch.complex64: torch.float32,
130      torch.complex128: torch.float64,  # NOTE: ORT doesn't support torch.float64

133  _SYM_TYPE_TO_TORCH_DTYPE = {
134      torch.SymInt: torch.int64,

139  _SCALAR_TYPE_TO_TORCH_DTYPE: dict[type, torch.dtype] = {
