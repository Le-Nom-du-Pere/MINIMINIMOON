4    def pin_memory(data_ptr: int, size: int) -> None:
5        cudart = torch.cuda.cudart()
6        succ = int(
7            cudart.cudaHostRegister(
8                data_ptr,
9                size,
10               1,  # lines up with 'cudaHostRegisterPortable'
11           )
12       )
13   
14       if succ != 0:
15           raise RuntimeError(
16               f"Registering memory failed with cudaError: {succ}."
17               " It's possible that this is an asynchronous error raised from a previous cuda operation."
18               " Consider launching with CUDA_LAUNCH_BLOCKING=1 to debug."
19           )

22   def unpin_memory(data_ptr: int) -> None:
23       succ = int(torch.cuda.cudart().cudaHostUnregister(data_ptr))
