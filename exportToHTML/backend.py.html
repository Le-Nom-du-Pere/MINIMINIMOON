16       def sympify(a, *, strict=False):
17           """ 
18           Notes 
19           ===== 
20    
21           SymEngine's ``sympify`` does not accept keyword arguments and is 
22           therefore not compatible with SymPy's ``sympify`` with ``strict=True`` 
23           (which ensures that only the types for which an explicit conversion has 
24           been defined are converted). This wrapper adds an additional parameter 
25           ``strict`` (with default ``False``) that will raise a ``SympifyError`` 
26           if ``strict=True`` and the argument passed to the parameter ``a`` is a 
27           string. 
28    
29           See Also 
30           ======== 
31    
32           sympify: Converts an arbitrary expression to a type that can be used 
33               inside SymPy. 
34    
35           """
36           # The parameter ``a`` is used for this function to keep compatibility
37           # with the SymEngine docstring.
38           if strict and isinstance(a, str):
39               raise SympifyError(a)
40           return sympify_symengine(a)

74   #
75   # XXX: Handling of immutable and mutable matrices in SymEngine is inconsistent
76   # with SymPy's matrix classes in at least SymEngine version 0.7.0. Until that
77   # is fixed the function below is needed for consistent behaviour when
78   # attempting to simplify a matrix.
79   #
80   # Expected behaviour of a SymPy mutable/immutable matrix .simplify() method:
81   #
82   #   Matrix.simplify() : works in place, returns None
83   #   ImmutableMatrix.simplify() : returns a simplified copy
84   #
85   # In SymEngine both mutable and immutable matrices simplify in place and return
86   # None. This is inconsistent with the matrix being "immutable" and also the
87   # returned None leads to problems in the mechanics module.
88   #
89   # The simplify function should not be used because simplify(M) sympifies the
90   # matrix M and the SymEngine matrices all sympify to SymPy matrices. If we want
91   # to work with SymEngine matrices then we need to use their .simplify() method
92   # but that method does not work correctly with immutable matrices.
93   #
94   # The _simplify_matrix function can be removed when the SymEngine bug is fixed.
95   # Since this should be a temporary problem we do not make this function part of
96   # the public API.
97   #
98   #   SymEngine issue: https://github.com/symengine/symengine.py/issues/363
99   #
100  
101  def _simplify_matrix(M):
102      """Return a simplified copy of the matrix M"""
103      if not isinstance(M, (Matrix, ImmutableMatrix)):
104          raise TypeError("The matrix M must be an instance of Matrix or ImmutableMatrix")
105      Mnew = M.as_mutable() # makes a copy if mutable
106      Mnew.simplify()
107      if isinstance(M, ImmutableMatrix):
108          Mnew = Mnew.as_immutable()
109      return Mnew
