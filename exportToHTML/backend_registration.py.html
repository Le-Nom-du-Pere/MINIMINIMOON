18   def rename_privateuse1_backend(backend_name: str) -> None:
19       r""" 
20       Rename the privateuse1 backend device to make it more convenient to use as a device name within PyTorch APIs. 
21    
22       The steps are: 
23    
24       (1) (In C++) implement kernels for various torch operations, and register them 
25           to the PrivateUse1 dispatch key. 
26       (2) (In python) call torch.utils.rename_privateuse1_backend("foo") 
27    
28       You can now use "foo" as an ordinary device string in python. 
29    
30       Note: this API can only be called once per process. Attempting to change 
31       the external backend after it's already been set will result in an error. 
32    
33       Note(AMP): If you want to support AMP on your device, you can register a custom backend module. 
34       The backend must register a custom backend module with ``torch._register_device_module("foo", BackendModule)``. 
35       BackendModule needs to have the following API's: 
36    
37       (1) ``get_amp_supported_dtype() -> List[torch.dtype]`` 
38           get the supported dtypes on your "foo" device in AMP, maybe the "foo" device supports one more dtype. 
39    
40       Note(random): If you want to support to set seed for your device, BackendModule needs to have the following API's: 
41    
42       (1) ``_is_in_bad_fork() -> bool`` 
43           Return ``True`` if now it is in bad_fork, else return ``False``. 
44    
45       (2) ``manual_seed_all(seed int) -> None`` 
46           Sets the seed for generating random numbers for your devices. 
47    
48       (3) ``device_count() -> int`` 
49           Returns the number of "foo"s available. 
50    
51       (4) ``get_rng_state(device: Union[int, str, torch.device] = 'foo') -> Tensor`` 
52           Returns a list of ByteTensor representing the random number states of all devices. 
53    
54       (5) ``set_rng_state(new_state: Tensor, device: Union[int, str, torch.device] = 'foo') -> None`` 
55           Sets the random number generator state of the specified "foo" device. 
56    
57       And there are some common funcs: 
58    
59       (1) ``is_available() -> bool`` 
60           Returns a bool indicating if "foo" is currently available. 
61    
62       (2) ``current_device() -> int`` 
63           Returns the index of a currently selected device. 
64    
65       For more details, see https://pytorch.org/tutorials/advanced/extend_dispatcher.html#get-a-dispatch-key-for-your-backend 
66       For an existing example, see https://github.com/bdhirsh/pytorch_open_registration_example 
67    
68       Example:: 
69    
70           >>> # xdoctest: +SKIP("failing") 
71           >>> torch.utils.rename_privateuse1_backend("foo") 
72           # This will work, assuming that you've implemented the right C++ kernels 
73           # to implement torch.ones. 
74           >>> a = torch.ones(2, device="foo") 
75    
76       """
77       _rename_privateuse1_backend(backend_name)
78       global _privateuse1_backend_name
79       _privateuse1_backend_name = backend_name

82   def _check_register_once(module, attr):
83       if hasattr(module, attr):
84           raise RuntimeError(
85               f"The custom device module of {module} has already been registered with {attr}"
86           )

89   def _normalization_device(
90       custom_backend_name: str, device: Optional[Union[int, str, torch.device]] = None
91   ) -> int:
92       def _get_current_device_index():
93           _get_device_index = "current_device"
94           if hasattr(torch, custom_backend_name) and hasattr(
95               getattr(torch, custom_backend_name), _get_device_index
96           ):
97               return getattr(getattr(torch, custom_backend_name), _get_device_index)()
98           else:
99               # The default device index is 0.
100              return 0
101  
102      if device is None:
103          return _get_current_device_index()
104      # if isinstance(device, str), this means that the parameter passed in is in the string format "foo:0"
105      # convert str object to torch.device object, and then process it uniformly
106      elif isinstance(device, str):
107          device = torch.device(device)
108  
109      # variable devcie can only be torch.device type or int type
110      if isinstance(device, torch.device):
111          if device.type != custom_backend_name:
112              raise RuntimeError(f"Invalid device, must be {custom_backend_name} device")
113          elif device.index is None:
114              device_idx = _get_current_device_index()
115          else:
116              device_idx = device.index
117      # if isinstance(device, int), we can take the index number directly
118      else:
119          device_idx = device
120      return device_idx

123  def _generate_tensor_methods_for_privateuse1_backend(custom_backend_name: str) -> None:
124      @property  # type: ignore[misc]
125      def wrap_tensor_backend(self: torch.Tensor) -> bool:
126          if has_torch_function_unary(self):
127              # TODO mypy doesn't support @property, see: https://github.com/python/mypy/issues/6185
128              return handle_torch_function(wrap_tensor_backend.__get__, (self,), self)  # type: ignore[attr-defined]
129          return self.device.type == custom_backend_name
130  
131      _check_register_once(torch.Tensor, f"is_{custom_backend_name}")
132      wrap_tensor_backend.fget.__name__ = f"is_{custom_backend_name}"  # type: ignore[attr-defined]
133      setattr(torch.Tensor, f"is_{custom_backend_name}", wrap_tensor_backend)
134  
135      def wrap_tensor_to(
136          self: torch.Tensor,
137          device: Optional[Union[int, torch.device]] = None,
138          non_blocking=False,
139          **kwargs,
140      ) -> torch.Tensor:
141          r"""Perform Tensor device conversion. Call the to operator implementation. 
142   
143          .. note:: 
144              If the ``self`` Tensor already 
145              has the correct :class:`torch.device`, then ``self`` is returned. 
146              Otherwise, the returned tensor is a copy of ``self`` with the desired :class:`torch.device`. 
147   
148          Args: 
149              device (int, optional): if specified, all parameters will be copied to that device 
150              non_blocking (bool): If ``True`` and the source is in pinned memory, 
151                  the copy will be asynchronous with respect to the host. Otherwise, 
152                  the argument has no effect. 
153              **kwargs (dict): For compatibility, may contain the key ``memory_format`` argument. 
154          """
155          if has_torch_function_unary(self):
156              return handle_torch_function(
157                  wrap_tensor_to,
158                  (self,),
159                  self,
160                  device=device,
161                  non_blocking=False,
162                  **kwargs,
163              )
164          device_idx = _normalization_device(custom_backend_name, device)
165          return self.to(
166              device=torch.device(f"{custom_backend_name}:{device_idx}"),
167              non_blocking=non_blocking,
168              **kwargs,
169          )
170  
171      _check_register_once(torch.Tensor, custom_backend_name)
172      wrap_tensor_to.__name__ = custom_backend_name
173      setattr(torch.Tensor, custom_backend_name, wrap_tensor_to)

176  def _generate_module_methods_for_privateuse1_backend(custom_backend_name: str) -> None:
177      # Generate Module attributes and methods depends on Tensor methods,
178      # so we need to check whether Tensor methods is already registered.
179      if not hasattr(torch.Tensor, custom_backend_name):
180          raise RuntimeError(
181              f"Can not automatically generate {custom_backend_name}() method for torch.nn.Module."
182              f"Because torch.Tensor doesn't has the method {custom_backend_name}()."
183              f"For this error, you can try setting for_tensor=True."
184          )

186      def wrap_module_to(
187          self: torch.nn.modules.module.T,
188          device: Optional[Union[int, torch.device]] = None,
189      ) -> torch.nn.modules.module.T:
190          r"""Move all model parameters and buffers to the custom device. 
191   
192          This also makes associated parameters and buffers different objects. So 
193          it should be called before constructing optimizer if the module will 
194          live on device while being optimized. 
195   
196          .. note:: 
197              This method modifies the module in-place. 
198   
199          Args: 
200              device (int, optional): if specified, all parameters will be copied to that device 
201          """
202          return self._apply(lambda t: getattr(t, custom_backend_name)(device))
203  
204      _check_register_once(torch.nn.Module, custom_backend_name)
205      setattr(torch.nn.Module, custom_backend_name, wrap_module_to)

235      def wrap_module_to(
236          self: torch.nn.utils.rnn.PackedSequence, *args, **kwargs
237      ) -> torch.nn.utils.rnn.PackedSequence:
238          r"""Move all model parameters and buffers to the custom device. 
239   
240          This also makes associated parameters and buffers different objects. So 
241          it should be called before constructing optimizer if the module will 
242          live on device while being optimized. 
243   
244          .. note:: 
245              This method modifies the module in-place. 
246   
247          Args: 
248              device (int, optional): if specified, all parameters will be copied to that device 
249          """
250          ex = torch.tensor((), dtype=self.data.dtype, device=self.data.device).to(
251              *args, **kwargs
252          )
253          if ex.device.type == custom_backend_name:
254              return self.to(*args, **kwargs)
255          kwargs.update({"device": custom_backend_name})
256          return self.to(*args, **kwargs)
257  
258      _check_register_once(torch.nn.utils.rnn.PackedSequence, custom_backend_name)
259      setattr(torch.nn.utils.rnn.PackedSequence, custom_backend_name, wrap_module_to)

262  def _generate_storage_methods_for_privateuse1_backend(
263      custom_backend_name: str, unsupported_dtype: Optional[list[torch.dtype]] = None
264  ) -> None:
265      # Attribute is registered in the _StorageBase class
266      # and UntypedStorage obtains through inheritance.
267      @property  # type: ignore[misc]
268      def wrap_storage_backend(self: torch.storage._StorageBase) -> bool:
269          r"""Return the internal :class:`torch.UntypedStorage`."""
270          return self.device.type == custom_backend_name
271  
272      _check_register_once(torch.storage._StorageBase, f"is_{custom_backend_name}")
273      setattr(
274          torch.storage._StorageBase, f"is_{custom_backend_name}", wrap_storage_backend
275      )

277      def wrap_storage_to(self, device=None, non_blocking=False):
278          r"""Return a copy of this object in custom device memory. 
279   
280          If this object is already in device memory and on the correct device, then 
281          no copy is performed and the original object is returned. 
282   
283          Args: 
284              device (int): The destination device id. Defaults to the current device. 
285              non_blocking (bool): If ``True`` and the source is in pinned memory, 
286              the copy will be asynchronous with respect to the host. Otherwise, 
287              the argument has no effect. 
288          """
289          # There should be a judgment related to storage device and a judgment related to storage type,
290          # but it depends on the extended function, so this part is temporarily omitted in the automatic generation.
291          device_idx = _normalization_device(custom_backend_name, device)
292  
293          if getattr(self, f"is_{custom_backend_name}"):
294              # storage has already on expected device.
295              if self.get_device() == device_idx:
296                  return self
297          # For sparse storage, custom need to extend the implementation by themselves.
298          if self.is_sparse:
299              raise RuntimeError(
300                  f"Can not support a sparse storage move to {custom_backend_name} backend"
301              )
302          # create untyped_storage and copy data
303          untyped_storage = torch.UntypedStorage(
304              self.size(), device=torch.device(f"{custom_backend_name}:{device_idx}")
305          )
306          untyped_storage.copy_(self, non_blocking)
307          return untyped_storage
308  
309      _check_register_once(torch.storage._StorageBase, custom_backend_name)
310      setattr(torch.storage._StorageBase, custom_backend_name, wrap_storage_to)
311  
312      # Register the corresponding attribute for the TypedStorage class.
313      # When the TypedStorage class is removed, the registration is also removed.
314  
315      @property  # type: ignore[misc]
316      def wrap_typed_storage_backend(self: torch.storage.TypedStorage) -> bool:
317          torch.storage._warn_typed_storage_removal()
318          return self._untyped_storage.device.type == custom_backend_name
319  
320      _check_register_once(torch.TypedStorage, f"is_{custom_backend_name}")
321      setattr(
322          torch.storage.TypedStorage,
323          f"is_{custom_backend_name}",
324          wrap_typed_storage_backend,
325      )

327      def wrap_typed_storage_to(
328          self: torch.storage.TypedStorage, device=None, non_blocking=False, **kwargs
329      ) -> torch.storage.TypedStorage:
330          torch.storage._warn_typed_storage_removal()
331          if unsupported_dtype and self.dtype in unsupported_dtype:
332              raise RuntimeError(
333                  f"Cannot create {custom_backend_name} storage "
334                  f"as {self.dtype} dtype is not supported by this backend"
335              )
336          custom_backend_storage: torch.UntypedStorage = getattr(
337              self._untyped_storage, custom_backend_name
338          )(device, non_blocking, **kwargs)
339          return self._new_wrapped_storage(custom_backend_storage)
340  
341      _check_register_once(torch.TypedStorage, custom_backend_name)
342      setattr(torch.TypedStorage, custom_backend_name, wrap_typed_storage_to)

345  def generate_methods_for_privateuse1_backend(
346      for_tensor: bool = True,
347      for_module: bool = True,
348      for_packed_sequence: bool = True,
349      for_storage: bool = False,
350      unsupported_dtype: Optional[list[torch.dtype]] = None,
351  ) -> None:
352      r""" 
353      Automatically generate attributes and methods for the custom backend after rename privateuse1 backend. 
354   
355      In the default scenario, storage-related methods will not be generated automatically. 
356   
357      When you implement kernels for various torch operations, and register them to the PrivateUse1 dispatch key. 
358      And call the function torch.rename_privateuse1_backend("foo") to rename your backend name. 
359      At this point, you can easily register specific methods and attributes by calling this function. 
360      Just like torch.Tensor.foo(), torch.Tensor.is_foo, torch.Storage.foo(), torch.Storage.is_foo. 
361   
362      Note: We recommend you use generic functions (check devices are equal or to(device=)). 
363      We provide these methods for convenience only and they will be "monkey patched" onto the objects 
364      and so will not be properly typed. For Storage methods generate, if you need to support sparse data storage, 
365      you need to extend the implementation yourself. 
366   
367      Args: 
368          for_tensor (bool): whether register related methods for torch.Tensor class. 
369          for_module (bool): whether register related methods for torch.nn.Module class. 
370          for_storage (bool): whether register related methods for torch.Storage class. 
371          unsupported_dtype (List[torch.dtype]): takes effect only when the storage method needs to be generated, 
372              indicating that the storage does not support the torch.dtype type. 
373   
374      Example:: 
375   
376          >>> # xdoctest: +SKIP("failing") 
377          >>> torch.utils.rename_privateuse1_backend("foo") 
378          >>> torch.utils.generate_methods_for_privateuse1_backend() 
379          # Then automatically generate backend-related attributes and methods. 
380          >>> a = torch.tensor(2).foo() 
381          >>> a.is_foo 
382          >>> hasattr(torch.nn.Module, 'foo') 
383      """
384      custom_backend_name = _get_privateuse1_backend_name()
385  
386      if for_tensor:
387          _generate_tensor_methods_for_privateuse1_backend(custom_backend_name)
388  
389      if for_module:
390          _generate_module_methods_for_privateuse1_backend(custom_backend_name)
391  
392      if for_storage:
393          _generate_storage_methods_for_privateuse1_backend(
394              custom_backend_name, unsupported_dtype
395          )
396  
397      if for_packed_sequence:
398          _generate_packed_sequence_methods_for_privateuse1_backend(custom_backend_name)

401  def _get_custom_mod_func(func_name: str):
402      r""" 
403      Return the func named `func_name` defined in custom device module. If not defined, 
404      return `None`. And the func is registered with `torch.utils.rename_privateuse1_backend('foo')` 
405      and `torch._register_device_module('foo', BackendModule)`. 
406      If the custom device module or the func is not defined, it will give warning or error message. 
407      Args: 
408          func_name (str): return the callable func named func_name defined in custom device module. 
409      Example:: 
410          class DummyfooModule: 
411              @staticmethod 
412              def is_available(): 
413                  return True 
414              @staticmethod 
415              def func_name(*args, **kwargs): 
416                  .... 
417          torch.utils.rename_privateuse1_backend("foo") 
418          torch._register_device_module("foo", DummyfooModule) 
419          foo_is_available_func = torch.utils.backend_registration._get_custom_mod_func("is_available") 
420          if foo_is_available_func: 
421              foo_is_available = foo_is_available_func() 
422          func_ = torch.utils.backend_registration._get_custom_mod_func("func_name") 
423          if func_: 
424              result = func_(*args, **kwargs) 
425      Attention: This function is not meant to be used directly by users, which is why 
426      it is marked as private. It is a convenience function for backend implementers to 
427      more easily call the hooks into their backend extensions. 
428      """
429      assert isinstance(
430          func_name, str
431      ), f"func_name must be `str`, but got `{type(func_name)}`."
432      backend_name = _get_privateuse1_backend_name()
433      custom_device_mod = getattr(torch, backend_name, None)  # type: ignore[arg-type]
434      function = getattr(custom_device_mod, func_name, None)  # type: ignore[arg-type]
435      if custom_device_mod is None or function is None:
436          message = f"Try to call torch.{backend_name}.{func_name}. The backend must register a custom backend "
437          message += f"module with `torch._register_device_module('{backend_name}', BackendModule)`. And "
438          message += f"BackendModule needs to have the following API's:\n `{func_name}(*args, **kwargs)`. \n"
439          raise RuntimeError(message)
440      return function
