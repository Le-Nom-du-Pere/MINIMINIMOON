74     T buf[Vectorized<T>::size()];
75     vec.store(buf);
76     stream << "vec[";
77     for (int i = 0; i != Vectorized<T>::size(); i++) {
78       if (i != 0) {
79         stream << ", ";

158  template <>
159  Vectorized<int64_t> inline convert_to_int_of_same_size<double>(
160      const Vectorized<double>& src) {
161    auto x = _mm256_add_pd(src, _mm256_set1_pd(0x0018000000000000));
162    return _mm256_sub_epi64(
163        _mm256_castpd_si256(x),
164        _mm256_castpd_si256(_mm256_set1_pd(0x0018000000000000)));
165  }

173  // From: https://stackoverflow.com/a/41148578
174  template <>
175  Vectorized<double> inline convert_to_fp_of_same_size<double>(
176      const Vectorized<int64_t>& src) {
177    __m256i magic_i_lo = _mm256_set1_epi64x(0x4330000000000000); /* 2^52 */
178    __m256i magic_i_hi32 =
179        _mm256_set1_epi64x(0x4530000080000000); /* 2^84 + 2^63 */
180    __m256i magic_i_all =
181        _mm256_set1_epi64x(0x4530000080100000); /* 2^84 + 2^63 + 2^52 */
182    __m256d magic_d_all = _mm256_castsi256_pd(magic_i_all);
183  
184    __m256i v_lo = _mm256_blend_epi32(
185        magic_i_lo, src, 0b01010101); /* v_low = low32 + 2^52 */
186    __m256i v_hi = _mm256_srli_epi64(src, 32);
187    v_hi = _mm256_xor_si256(
188        v_hi, magic_i_hi32); /* v_hi = high32*2^32 + 2^84 + 2^63 */
189    /* int64 = low32 + high32*2^32 = v_hi + v_lo - 2^52 - 2^63 - 2^84 */
190    __m256d v_hi_dbl = _mm256_sub_pd(_mm256_castsi256_pd(v_hi), magic_d_all);
191    __m256d result = _mm256_add_pd(v_hi_dbl, _mm256_castsi256_pd(v_lo));
192    return result;
193  }

203  template <>
204  std::pair<Vectorized<double>, Vectorized<double>> inline interleave2<double>(
205      const Vectorized<double>& a,
206      const Vectorized<double>& b) {
207    // inputs:
208    //   a = {a0, a1, a2, a3}
209    //   b = {b0, b1, b2, b3}
210  
211    // swap lanes:
212    //   a_swapped = {a0, a1, b0, b1}
213    //   b_swapped = {a2, a3, b2, b3}
214    auto a_swapped =
215        _mm256_permute2f128_pd(a, b, 0b0100000); // 0, 2.   4 bits apart
216    auto b_swapped =
217        _mm256_permute2f128_pd(a, b, 0b0110001); // 1, 3.   4 bits apart
218  
219    // group cols crossing lanes:
220    //   return {a0, b0, a1, b1}
221    //          {a2, b2, a3, b3}
222    return std::make_pair(
223        _mm256_permute4x64_pd(a_swapped, 0b11011000), // 0, 2, 1, 3
224        _mm256_permute4x64_pd(b_swapped, 0b11011000)); // 0, 2, 1, 3
225  }

227  template <>
228  std::pair<Vectorized<float>, Vectorized<float>> inline interleave2<float>(
229      const Vectorized<float>& a,
230      const Vectorized<float>& b) {
231    // inputs:
232    //   a = {a0, a1, a2, a3, a4, a5, a6, a7}
233    //   b = {b0, b1, b2, b3, b4, b5, b6, b7}
234  
235    // swap lanes:
236    //   a_swapped = {a0, a1, a2, a3, b0, b1, b2, b3}
237    //   b_swapped = {a4, a5, a6, a7, b4, b5, b6, b7}
238    // TODO: can we support caching this?
239    auto a_swapped =
240        _mm256_permute2f128_ps(a, b, 0b0100000); // 0, 2.   4 bits apart
241    auto b_swapped =
242        _mm256_permute2f128_ps(a, b, 0b0110001); // 1, 3.   4 bits apart
243  
244    // group cols crossing lanes:
245    //   return {a0, b0, a1, b1, a2, b2, a3, b3}
246    //          {a4, b4, a5, b5, a6, b6, a7, b7}
247    const __m256i group_ctrl = _mm256_setr_epi32(0, 4, 1, 5, 2, 6, 3, 7);
248    return std::make_pair(
249        _mm256_permutevar8x32_ps(a_swapped, group_ctrl),
250        _mm256_permutevar8x32_ps(b_swapped, group_ctrl));
251  }

255  template <>
256  std::pair<Vectorized<double>, Vectorized<double>> inline deinterleave2<double>(
257      const Vectorized<double>& a,
258      const Vectorized<double>& b) {
259    // inputs:
260    //   a = {a0, b0, a1, b1}
261    //   b = {a2, b2, a3, b3}
262  
263    // group cols crossing lanes:
264    //   a_grouped = {a0, a1, b0, b1}
265    //   b_grouped = {a2, a3, b2, b3}
266    auto a_grouped = _mm256_permute4x64_pd(a, 0b11011000); // 0, 2, 1, 3
267    auto b_grouped = _mm256_permute4x64_pd(b, 0b11011000); // 0, 2, 1, 3
268  
269    // swap lanes:
270    //   return {a0, a1, a2, a3}
271    //          {b0, b1, b2, b3}
272    return std::make_pair(
273        _mm256_permute2f128_pd(
274            a_grouped, b_grouped, 0b0100000), // 0, 2.   4 bits apart
275        _mm256_permute2f128_pd(
276            a_grouped, b_grouped, 0b0110001)); // 1, 3.   4 bits apart
277  }

279  template <>
280  std::pair<Vectorized<float>, Vectorized<float>> inline deinterleave2<float>(
281      const Vectorized<float>& a,
282      const Vectorized<float>& b) {
283    // inputs:
284    //   a = {a0, b0, a1, b1, a2, b2, a3, b3}
285    //   b = {a4, b4, a5, b5, a6, b6, a7, b7}
286  
287    // group cols crossing lanes:
288    //   a_grouped = {a0, a1, a2, a3, b0, b1, b2, b3}
289    //   b_grouped = {a4, a5, a6, a7, b4, b5, b6, b7}
290    // TODO: can we support caching this?
291    const __m256i group_ctrl = _mm256_setr_epi32(0, 2, 4, 6, 1, 3, 5, 7);
292    auto a_grouped = _mm256_permutevar8x32_ps(a, group_ctrl);
293    auto b_grouped = _mm256_permutevar8x32_ps(b, group_ctrl);
294  
295    // swap lanes:
296    //   return {a0, a1, a2, a3, a4, a5, a6, a7}
297    //          {b0, b1, b2, b3, b4, b5, b6, b7}
298    return std::make_pair(
299        _mm256_permute2f128_ps(
300            a_grouped, b_grouped, 0b0100000), // 0, 2.   4 bits apart
301        _mm256_permute2f128_ps(
302            a_grouped, b_grouped, 0b0110001)); // 1, 3.   4 bits apart
303  }

416  inline Vectorized<bool> operator&&(
417      const Vectorized<bool>& self,
418      const Vectorized<bool>& other) {
419    const __m256i* self_ = reinterpret_cast<const __m256i*>(self.as_bytes());
420    const __m256i* other_ = reinterpret_cast<const __m256i*>(other.as_bytes());
421    __m256i out = _mm256_and_si256(*self_, *other_);
422    Vectorized<bool> ret;
423    std::memcpy(ret, &out, ret.size() * sizeof(bool));
424    return ret;
425  }
