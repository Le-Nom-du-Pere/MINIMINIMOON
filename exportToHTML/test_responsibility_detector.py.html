10   class TestResponsibilityDetector:
11       @pytest.fixture

16       @staticmethod
17       def test_government_entity_detection(detector):
18           """Test detection of government entities with high confidence."""
19           text = "La Alcaldía Municipal coordinará con la Secretaría de Salud."
20           result = detector.calculate_responsibility_score(text)
21   
22           assert result["factibility_score"] > 0.7
23           assert result["has_government_entities"] is True
24   
25           # Check for government entities
26           gov_entities = [e for e in result["entities"] if e.is_government]
27           assert len(gov_entities) >= 1
28   
29           # Government entities should have high confidence
30           for entity in gov_entities:
31               assert entity.confidence >= 0.8
32

137      @staticmethod
138      def test_pattern_coverage(detector):
139          """Test that key government patterns are covered."""
140          test_cases = [
141              ("alcaldía", True),
142              ("secretaría de educación", True),
143              ("programa social", True),
144              ("ministerio del interior", True),
145              ("alcalde municipal", True),
146              ("director general", True),
147              ("coordinador técnico", True),
148          ]
149  
150          for text, should_detect in test_cases:
151              entities = detector.detect_entities(text)
152              if should_detect:
153                  assert len(
154                      entities) > 0, f"Failed to detect entities in: {text}"
155                  assert any(e.confidence > 0.5 for e in entities), (
156                      f"Low confidence for: {text}"
157                  )
