45   # Check if we can safely replicate the module.
46   # there are two types of module:
47   # 1. python modules
48   # 2. ScriptModule
49   #
50   # currently a module cannot be replicated properly if the descendants of
51   # any ScriptModule contains python module (type 1 above)
52   def _replicatable_module(module: Module, memo: Optional[set[Module]] = None) -> bool:
53       # module.modules() contains module itself as the first element
54       def descendant_modules(module: Module) -> Iterator[Module]:
55           gen = module.modules()
56           next(gen)
57           return gen
58   
59       if not _is_jit_enabled():
60           return True
61       if memo is None:
62           memo = set()
63   
64       # memoize visited modules
65       memo.add(module)
66       if _is_script_module(module):
67           memo.update(descendant_modules(module))
68           return all(
69               _is_script_module(descendant) for descendant in descendant_modules(module)
70           )
71   
72       for child in module.children():
73           # since any unreplicatable module will cause the check to return
74           # False early, visited modules here can be safely ignored.
75           if child in memo:
76               continue
77           if not _replicatable_module(child, memo):
78               return False
79   
80       return True

83   def _broadcast_coalesced_reshape(
84       tensors: Sequence[torch.Tensor],
85       devices: Sequence[Union[int, torch.device]],
86       detach: bool = False,
87   ) -> list[list[torch.Tensor]]:
88       from torch.nn.parallel._functions import Broadcast
89   
90       if detach:
91           return comm.broadcast_coalesced(tensors, devices)
92       else:
93           # Use the autograd function to broadcast if not detach
94           if len(tensors) > 0:
95               tensor_copies = Broadcast.apply(devices, *tensors)
96               return [
97                   tensor_copies[i : i + len(tensors)]
98                   for i in range(0, len(tensor_copies), len(tensors))
99               ]
100          else:
101              return []

107  def replicate(
108      network: T,
109      devices: Sequence[Union[int, torch.device]],
110      detach: bool = False,
111  ) -> list[T]:
112      if not _replicatable_module(network):
113          raise RuntimeError(
114              "Cannot replicate network where python modules are "
115              "childrens of ScriptModule"
116          )
117  
118      if not devices:
119          return []
120  
121      devices = [_get_device_index(x, True) for x in devices]
122      num_replicas = len(devices)
123  
124      params = list(network.parameters())
125      param_indices = {param: idx for idx, param in enumerate(params)}
126      param_copies = _broadcast_coalesced_reshape(params, devices, detach)
127  
128      buffers = list(network.buffers())
129      buffers_rg: list[torch.Tensor] = []
130      buffers_not_rg: list[torch.Tensor] = []
131      for buf in buffers:
132          if buf.requires_grad and not detach:
133              buffers_rg.append(buf)
134          else:
135              buffers_not_rg.append(buf)
136  
137      buffer_indices_rg = {buf: idx for idx, buf in enumerate(buffers_rg)}
138      buffer_indices_not_rg = {buf: idx for idx, buf in enumerate(buffers_not_rg)}
139  
140      buffer_copies_rg = _broadcast_coalesced_reshape(buffers_rg, devices, detach=detach)
141      buffer_copies_not_rg = _broadcast_coalesced_reshape(
142          buffers_not_rg, devices, detach=True
143      )
144  
145      modules = list(network.modules())
146      module_copies: list[list[Module]] = [[] for _ in devices]
147      module_indices: dict[Module, int] = {}
148  
149      for i, module in enumerate(modules):
150          module_indices[module] = i
151          for j in range(num_replicas):
152              replica = module._replicate_for_data_parallel()
153              # This is a temporary fix for DDP. DDP needs to access the
154              # replicated model parameters. It used to do so through
155              # `mode.parameters()`. The fix added in #33907 for DP stops the
156              # `parameters()` API from exposing the replicated parameters.
157              # Hence, we add a `_former_parameters` dict here to support DDP.
158              replica._former_parameters = OrderedDict()
159  
160              module_copies[j].append(replica)
161  
162      for i, module in enumerate(modules):
163          for key, child in module._modules.items():
164              if child is None:
165                  for j in range(num_replicas):
166                      replica = module_copies[j][i]
167                      replica._modules[key] = None
168              else:
169                  module_idx = module_indices[child]
170                  for j in range(num_replicas):
171                      replica = module_copies[j][i]
172                      setattr(replica, key, module_copies[j][module_idx])
173          for key, param in module._parameters.items():
174              if param is None:
175                  for j in range(num_replicas):
176                      replica = module_copies[j][i]
177                      replica._parameters[key] = None
178              else:
179                  param_idx = param_indices[param]
180                  for j in range(num_replicas):
181                      replica = module_copies[j][i]
182                      param_copy = param_copies[j][param_idx]
183                      # parameters in replicas are no longer leaves,
184                      # so setattr them as non-parameter attributes
185                      setattr(replica, key, param_copy)
186                      # expose the parameter for DDP
187                      replica._former_parameters[key] = param_copy  # type: ignore[operator, index]
188          for key, buf in module._buffers.items():  # type: ignore[assignment]
189              if buf is None:
190                  for j in range(num_replicas):
191                      replica = module_copies[j][i]
192                      replica._buffers[key] = None
193              else:
194                  if buf.requires_grad and not detach:
195                      buffer_copies = buffer_copies_rg
196                      buffer_idx = buffer_indices_rg[buf]
197                  else:
198                      buffer_copies = buffer_copies_not_rg
199                      buffer_idx = buffer_indices_not_rg[buf]
200                  for j in range(num_replicas):
201                      replica = module_copies[j][i]
202                      setattr(replica, key, buffer_copies[j][buffer_idx])
203  
204      return [cast(T, module_copies[j][0]) for j in range(num_replicas)]
