meijerint_doc.py 
1    """ This module cooks up a docstring when imported. Its only purpose is to 
2        be displayed in the sphinx documentation. """
3    
4    from __future__ import annotations
5    from typing import Any
6    
7    from sympy.integrals.meijerint import _create_lookup_table
8    from sympy.core.add import Add
9    from sympy.core.basic import Basic
10   from sympy.core.expr import Expr
11   from sympy.core.relational import Eq
12   from sympy.core.symbol import Symbol
13   from sympy.printing.latex import latex
14   
15   t: dict[tuple[type[Basic], ...], list[Any]] = {}
16   _create_lookup_table(t)
17   
18   
19   doc = ""
20   for about, category in t.items():
21       if about == ():
22           doc += 'Elementary functions:\n\n'
23       else:
24           doc += 'Functions involving ' + ', '.join('`%s`' % latex(
25               list(category[0][0].atoms(func))[0]) for func in about) + ':\n\n'
26       for formula, gs, cond, hint in category:
27           if not isinstance(gs, list):
28               g: Expr = Symbol('\\text{generated}')
29           else:
30               g = Add(*[fac*f for (fac, f) in gs])
31           obj = Eq(formula, g)
32           if cond is True:
33               cond = ""
34           else:
35               cond = ',\\text{ if } %s' % latex(cond)
36           doc += ".. math::\n  %s%s\n\n" % (latex(obj), cond)
37   
38   __doc__ = doc
