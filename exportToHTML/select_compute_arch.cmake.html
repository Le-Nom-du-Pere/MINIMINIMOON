1    # Synopsis:
2    #   CUDA_SELECT_NVCC_ARCH_FLAGS(out_variable [target_CUDA_architectures])
3    #   -- Selects GPU arch flags for nvcc based on target_CUDA_architectures
4    #      target_CUDA_architectures : Auto | Common | All | LIST(ARCH_AND_PTX ...)
5    #       - "Auto" detects local machine GPU compute arch at runtime.
6    #       - "Common" and "All" cover common and entire subsets of architectures
7    #      ARCH_AND_PTX : NAME | NUM.NUM | NUM.NUM(NUM.NUM) | NUM.NUM+PTX
8    #      NAME: Kepler Maxwell Kepler+Tegra Kepler+Tesla Maxwell+Tegra Pascal Volta Turing Ampere
9    #      NUM: Any number. Only those pairs are currently accepted by NVCC though:
10   #            3.5 3.7 5.0 5.2 5.3 6.0 6.2 7.0 7.2 7.5 8.0
11   #      Returns LIST of flags to be added to CUDA_NVCC_FLAGS in ${out_variable}
12   #      Additionally, sets ${out_variable}_readable to the resulting numeric list
13   #      Example:
14   #       CUDA_SELECT_NVCC_ARCH_FLAGS(ARCH_FLAGS 3.0 3.5+PTX 5.2(5.0) Maxwell)
15   #        LIST(APPEND CUDA_NVCC_FLAGS ${ARCH_FLAGS})
16   #
17   #      More info on CUDA architectures: https://en.wikipedia.org/wiki/CUDA
18   #

98   # A function for automatic detection of GPUs installed  (if autodetection is enabled)
99   # Usage:
100  #   CUDA_DETECT_INSTALLED_GPUS(OUT_VARIABLE)
101  #
102  function(CUDA_DETECT_INSTALLED_GPUS OUT_VARIABLE) 
103    if(NOT CUDA_GPU_DETECT_OUTPUT) 
104      if(CMAKE_CUDA_COMPILER_LOADED) # CUDA as a language
105        set(file "${PROJECT_BINARY_DIR}/detect_cuda_compute_capabilities.cu") 
106      else() 
107        set(file "${PROJECT_BINARY_DIR}/detect_cuda_compute_capabilities.cpp") 
108      endif() 
109  
110      file(WRITE ${file} ""
111        "#include <cuda_runtime.h>\n"
112        "#include <cstdio>\n"
113        "int main()\n"
114        "{\n"
115        "  int count = 0;\n"
116        "  if (cudaSuccess != cudaGetDeviceCount(&count)) return -1;\n"
117        "  if (count == 0) return -1;\n"
118        "  for (int device = 0; device < count; ++device)\n"
119        "  {\n"
120        "    cudaDeviceProp prop;\n"
121        "    if (cudaSuccess == cudaGetDeviceProperties(&prop, device))\n"
122        "      std::printf(\"%d.%d \", prop.major, prop.minor);\n"
123        "  }\n"
124        "  return 0;\n"
125        "}\n") 
126  
127      if(CMAKE_CUDA_COMPILER_LOADED) # CUDA as a language
128        try_run(run_result compile_result ${PROJECT_BINARY_DIR} ${file}
129                RUN_OUTPUT_VARIABLE compute_capabilities) 
130      else() 
131        try_run(run_result compile_result ${PROJECT_BINARY_DIR} ${file}
132                CMAKE_FLAGS "-DINCLUDE_DIRECTORIES=${CUDA_INCLUDE_DIRS}"
133                LINK_LIBRARIES ${CUDA_LIBRARIES}
134                RUN_OUTPUT_VARIABLE compute_capabilities) 
135      endif() 
136  
137      # Filter unrelated content out of the output.
138      string(REGEX MATCHALL "[0-9]+\\.[0-9]+" compute_capabilities "${compute_capabilities}") 
139  
140      if(run_result EQUAL 0) 
141        string(REPLACE "2.1" "2.1(2.0)" compute_capabilities "${compute_capabilities}") 
142        set(CUDA_GPU_DETECT_OUTPUT ${compute_capabilities}
143          CACHE INTERNAL "Returned GPU architectures from detect_gpus tool" FORCE) 
144      endif() 
145    endif() 
146  
147    if(NOT CUDA_GPU_DETECT_OUTPUT) 
148      message(STATUS "Automatic GPU detection failed. Building for common architectures.") 
149      set(${OUT_VARIABLE} ${CUDA_COMMON_GPU_ARCHITECTURES} PARENT_SCOPE) 
150    else() 
151      # Filter based on CUDA version supported archs
152      set(CUDA_GPU_DETECT_OUTPUT_FILTERED "") 
153      separate_arguments(CUDA_GPU_DETECT_OUTPUT) 
154      foreach(ITEM IN ITEMS ${CUDA_GPU_DETECT_OUTPUT}) 
155          if(CUDA_LIMIT_GPU_ARCHITECTURE AND (ITEM VERSION_GREATER CUDA_LIMIT_GPU_ARCHITECTURE OR
156                                              ITEM VERSION_EQUAL CUDA_LIMIT_GPU_ARCHITECTURE)) 
157          list(GET CUDA_COMMON_GPU_ARCHITECTURES -1 NEWITEM) 
158          string(APPEND CUDA_GPU_DETECT_OUTPUT_FILTERED " ${NEWITEM}") 
159        else() 
160          string(APPEND CUDA_GPU_DETECT_OUTPUT_FILTERED " ${ITEM}") 
161        endif() 
162      endforeach() 
163  
164      set(${OUT_VARIABLE} ${CUDA_GPU_DETECT_OUTPUT_FILTERED} PARENT_SCOPE) 
165    endif() 
166  endfunction()

170  # Function for selecting GPU arch flags for nvcc based on CUDA architectures from parameter list
171  # Usage:
172  #   SELECT_NVCC_ARCH_FLAGS(out_variable [list of CUDA compute archs])
173  function(CUDA_SELECT_NVCC_ARCH_FLAGS out_variable) 
174    set(CUDA_ARCH_LIST "${ARGN}") 
175  
176    if("X${CUDA_ARCH_LIST}" STREQUAL "X" ) 
177      set(CUDA_ARCH_LIST "Auto") 
178    endif() 
179  
180    set(cuda_arch_bin) 
181    set(cuda_arch_ptx) 
182  
183    if("${CUDA_ARCH_LIST}" STREQUAL "All") 
184      set(CUDA_ARCH_LIST ${CUDA_KNOWN_GPU_ARCHITECTURES}) 
185    elseif("${CUDA_ARCH_LIST}" STREQUAL "Common") 
186      set(CUDA_ARCH_LIST ${CUDA_COMMON_GPU_ARCHITECTURES}) 
187    elseif("${CUDA_ARCH_LIST}" STREQUAL "Auto") 
188      CUDA_DETECT_INSTALLED_GPUS(CUDA_ARCH_LIST) 
189      message(STATUS "Autodetected CUDA architecture(s): ${CUDA_ARCH_LIST}") 
190    endif() 
191  
192    # Now process the list and look for names
193    string(REGEX REPLACE "[ \t]+" ";" CUDA_ARCH_LIST "${CUDA_ARCH_LIST}") 
194    list(REMOVE_DUPLICATES CUDA_ARCH_LIST) 
195    foreach(arch_name ${CUDA_ARCH_LIST}) 
196      set(arch_bin) 
197      set(arch_ptx) 
198      set(add_ptx FALSE) 
199      # Check to see if we are compiling PTX
200      if(arch_name MATCHES "(.*)\\+PTX$") 
201        set(add_ptx TRUE) 
202        set(arch_name ${CMAKE_MATCH_1}) 
203      endif() 
204      if(arch_name MATCHES "^([0-9]+\\.[0-9]a?(\\([0-9]+\\.[0-9]\\))?)$") 
205        set(arch_bin ${CMAKE_MATCH_1}) 
206        set(arch_ptx ${arch_bin}) 
207      else() 
208        # Look for it in our list of known architectures
209        if(${arch_name} STREQUAL "Kepler+Tesla") 
210          set(arch_bin 3.7) 
211        elseif(${arch_name} STREQUAL "Kepler") 
212          set(arch_bin 3.5) 
213          set(arch_ptx 3.5) 
214        elseif(${arch_name} STREQUAL "Maxwell+Tegra") 
215          set(arch_bin 5.3) 
216        elseif(${arch_name} STREQUAL "Maxwell") 
217          set(arch_bin 5.0 5.2) 
218          set(arch_ptx 5.2) 
219        elseif(${arch_name} STREQUAL "Pascal") 
220          set(arch_bin 6.0 6.1) 
221          set(arch_ptx 6.1) 
222       elseif(${arch_name} STREQUAL "Volta+Tegra") 
223          set(arch_bin 7.2) 
224        elseif(${arch_name} STREQUAL "Volta") 
225          set(arch_bin 7.0 7.0) 
226          set(arch_ptx 7.0) 
227        elseif(${arch_name} STREQUAL "Turing") 
228          set(arch_bin 7.5) 
229          set(arch_ptx 7.5) 
230        elseif(${arch_name} STREQUAL "Ampere+Tegra") 
231          set(arch_bin 8.7) 
232        elseif(${arch_name} STREQUAL "Ampere") 
233          set(arch_bin 8.0 8.6) 
234          set(arch_ptx 8.0 8.6) 
235        elseif(${arch_name} STREQUAL "Ada") 
236          set(arch_bin 8.9) 
237          set(arch_ptx 8.9) 
238        elseif(${arch_name} STREQUAL "Hopper") 
239          set(arch_bin 9.0) 
240          set(arch_ptx 9.0) 
241        elseif(${arch_name} STREQUAL "Blackwell+Tegra") 
242          set(arch_bin 10.1) 
243        elseif(${arch_name} STREQUAL "Blackwell") 
244          set(arch_bin 10.0 12.0) 
245          set(arch_ptx 10.0 12.0) 
246        else() 
247          message(SEND_ERROR "Found Unknown CUDA Architecture Name in CUDA_SELECT_NVCC_ARCH_FLAGS: ${arch_name} ") 
248        endif() 
249      endif() 
250      if(NOT arch_bin) 
251        message(SEND_ERROR "arch_bin wasn't set for some reason") 
252      endif() 
253      list(APPEND cuda_arch_bin ${arch_bin}) 
254      if(add_ptx) 
255        if (NOT arch_ptx) 
256          set(arch_ptx ${arch_bin}) 
257        endif() 
258        list(APPEND cuda_arch_ptx ${arch_ptx}) 
259      endif() 
260    endforeach() 
261  
262    # remove dots and convert to lists
263    string(REGEX REPLACE "\\." "" cuda_arch_bin "${cuda_arch_bin}") 
264    string(REGEX REPLACE "\\." "" cuda_arch_ptx "${cuda_arch_ptx}") 
265    string(REGEX MATCHALL "[0-9()]+a?" cuda_arch_bin "${cuda_arch_bin}") 
266    string(REGEX MATCHALL "[0-9]+a?"   cuda_arch_ptx "${cuda_arch_ptx}") 
267  
268    if(cuda_arch_bin) 
269      list(REMOVE_DUPLICATES cuda_arch_bin) 
270    endif() 
271    if(cuda_arch_ptx) 
272      list(REMOVE_DUPLICATES cuda_arch_ptx) 
273    endif() 
274  
275    set(nvcc_flags "") 
276    set(nvcc_archs_readable "") 
277  
278    # Tell NVCC to add binaries for the specified GPUs
279    foreach(arch ${cuda_arch_bin}) 
280      if(arch MATCHES "([0-9]+)\\(([0-9]+)\\)") 
281        # User explicitly specified ARCH for the concrete CODE
282        list(APPEND nvcc_flags -gencode arch=compute_${CMAKE_MATCH_2},code=sm_${CMAKE_MATCH_1}) 
283        list(APPEND nvcc_archs_readable sm_${CMAKE_MATCH_1}) 
284      else() 
285        # User didn't explicitly specify ARCH for the concrete CODE, we assume ARCH=CODE
286        list(APPEND nvcc_flags -gencode arch=compute_${arch},code=sm_${arch}) 
287        list(APPEND nvcc_archs_readable sm_${arch}) 
288      endif() 
289    endforeach() 
290  
291    # Tell NVCC to add PTX intermediate code for the specified architectures
292    foreach(arch ${cuda_arch_ptx}) 
293      list(APPEND nvcc_flags -gencode arch=compute_${arch},code=compute_${arch}) 
294      list(APPEND nvcc_archs_readable compute_${arch}) 
295    endforeach() 
296  
297    string(REPLACE ";" " " nvcc_archs_readable "${nvcc_archs_readable}") 
298    set(${out_variable}          ${nvcc_flags}          PARENT_SCOPE) 
299    set(${out_variable}_readable ${nvcc_archs_readable} PARENT_SCOPE) 
300  endfunction()
