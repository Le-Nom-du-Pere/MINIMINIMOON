62   def expand_to_tensor_dim(t, n):
63       """ 
64       Expand a type to the desired tensor dimension if possible 
65       Raise an error otherwise. 
66       - t is the given type 
67       - n is a number of dimensions to expand to 
68       """
69       if t == Dyn:
70           dims = [Dyn] * n
71           return TensorType(tuple(dims))
72       elif isinstance(t, TensorType):
73           if len(t.__args__) != n:
74               raise TypeError(
75                   f"Cannot extend tensor. Tensor {t} has rank {len(t.__args__)}. It should have rank {n}"
76               )
77           return t
78       else:
79           raise TypeError(f"Cannot match the type {t}")

82   def broadcast_types(t1, t2):
83       """ 
84       Applies broadcasting to both given types such that they 
85       become consistent with eachother and returns two new 
86       resulting types 
87       """
88   
89       # if either type is Dyn, do nothing since the types are already consistent
90       if t1 == Dyn or t2 == Dyn or isinstance(t1, Var) or isinstance(t2, Var):
91           return t1, t2
92   
93       if isinstance(t1, TensorType) and isinstance(t2, TensorType):
94           s1 = len(t1.__args__)
95           s2 = len(t2.__args__)
96   
97           new_t1 = list(t1.__args__)
98           new_t2 = list(t2.__args__)
99   
100          # We make the types the same length which is the first requirement
101          # for consistency
102          if s1 > s2:
103              for i in range(s1 - s2):
104                  new_t2.insert(0, 1)
105  
106          elif s2 > s1:
107              for i in range(s2 - s1):
108                  new_t1.insert(0, 1)
109  
110          # we replace occurrences of "1" with each tensor with
111          # the corresponding type from the other tensor
112          for i, (x, y) in enumerate(zip(new_t1, new_t2)):
113              if x == 1:
114                  new_t1[i] = y
115              elif y == 1:
116                  new_t2[i] = x
117  
118          # at this point our tensors should be consistent
119          # and we can apply the element-wise operation and find the right dimension
120          # for the output of the operation
121          (t1, t2) = TensorType(tuple(new_t1)), TensorType(tuple(new_t2))
122          return (t1, t2)
123      else:
124          raise TypeError(f"Cannot broadcast types {t1} and {t2}")

164  @register_inference_rule(operator.add)
165  def add_inference_rule(n: Node):
166      """ 
167      Apply the addition inference rule. This includes: 
168      - scalar addition 
169      - broadcasting semantics 
170   
171      Note that we always return the least precise type between 
172      the operands (after applying broadcasting) to be the final type of the operation 
173   
174      Note that we do not modify the operand types themselves after applying broadcasting 
175      to them. We only use them to calculate the final type 
176      """
177      assert isinstance(n.args[0], Node)
178      assert isinstance(n.args[1], Node)
179      t1 = n.args[0].type
180      t2 = n.args[1].type
181  
182      # handle scalar addition
183      if t1 == int and isinstance(t2, TensorType):
184          n.type = t2
185          return n.type
186  
187      # handle scalar addition
188      elif t2 == int and isinstance(t1, TensorType):
189          n.type = t1
190          return n.type
191  
192      # we bring the new types to the point where
193      # we can check for consistency
194      # any inconsistency would not have been caused
195      # by broadcasting at this point
196      (new_t1, new_t2) = broadcast_types(t1, t2)
197  
198      if new_t1 != t1 or new_t2 != t2:
199          n.meta["broadcast"] = True
200          n.meta[str(n.args[0])] = new_t1
201          n.meta[str(n.args[1])] = new_t2
202  
203      else:
204          n.meta["broadcast"] = False
205  
206      new_t1 = t1 if not n.meta["broadcast"] else new_t1
207      new_t2 = t2 if not n.meta["broadcast"] else new_t2
208  
209      # we check for consistency between the new types
210      if is_consistent(new_t1, new_t2):
211          # we return the less precise type because
212          # broadcasting may have happened
213          # for operands with shape [1,2,Dyn] and [1,2,1]
214          # we have to assign the node [1,2,Dyn]
215          if is_more_precise(new_t1, new_t2):
216              n.type = new_t2
217          else:
218              n.type = new_t1
219          return n.type
220      else:
221          raise TypeError(
222              f"Cannot add arguments {n.args[0]} ({n.args[0].type}) and {n.args[1]} ({n.args[1].type}) in node {n}."
223              f" Types should match "
224          )

246  @register_inference_rule(torch.transpose)
247  def transpose_inference_rule(n: Node):
248      """ 
249      We check that dimensions for the transpose operations 
250      are within range of the tensor type of the node 
251      """
252      if n.target == torch.transpose:
253          assert isinstance(n.args[0], Node)
254          t = n.args[0].type
255  
256          assert isinstance(n.args[1], int)
257          assert isinstance(n.args[2], int)
258          dim1, dim2 = n.args[1], n.args[2]
259  
260          if t == Dyn:
261              n.type = Dyn
262              return n.type
263  
264          elif isinstance(t, TensorType):
265              if 0 <= dim1 < len(t.__args__) and 0 <= dim2 < len(t.__args__):
266                  new_type = list(t.__args__)
267                  new_type[dim1], new_type[dim2] = new_type[dim2], new_type[dim1]
268                  final = TensorType(new_type)
269                  n.type = get_greatest_upper_bound(n.type, final)
270                  return n.type
271              else:
272                  raise TypeError(
273                      f"Cannot transpose {dim1} and {dim2} in type {t} for node {n}"
274                  )
275          else:
276              raise TypeError(
277                  f"Cannot transpose {dim1} and {dim2} in type {t} for node {n}"
278              )

281  @register_inference_rule(torch.reshape)
282  def reshape_inference_rule(n: Node):
283      """ 
284      Without dynamism, the rule checks that the 
285      product of the elements of the argument tensor 
286      type is equal to the product of the elements 
287      of the required shape. We gradualize this rule 
288      by adding a case to handle fully dynamic input 
289      as well as input where some of the tensor dimensions 
290      are unknown. In this case we check for divisibility 
291      """
292      assert isinstance(n.args[0], Node)
293      t1 = n.args[0].type
294  
295      assert isinstance(n.args[1], list)
296      t2 = n.args[1]
297      t2_type = TensorType([Dyn if elem == -1 else elem for elem in t2])
298  
299      # if we do not know the original tensor dimension,
300      # we return the required dimension
301      if t1 == Dyn:
302          n.type = t2_type
303          return t2_type
304  
305      # if any of the dimensions are unknown,
306      # we check for divisibility
307      elif isinstance(t1, TensorType):
308          assert isinstance(t1, TensorType)
309          a = [e if e != Dyn else 1 for e in t1.__args__]
310          p1 = reduce(operator.mul, a)
311          p2 = reduce(operator.mul, t2)
312          if p1 % p2 == 0 or p2 % p1 == 0:
313              n.type = t2_type
314              return t2_type
315          else:
316              raise TypeError(f"Cannot reshape in node {n} from {t1} to {t2_type}")
317      else:
318          raise TypeError(f"Cannot reshape in node {n} from {t1} to {t2_type}")

321  @register_inference_rule(BatchNorm2d)
322  def bn2d_inference_rule(n: Node, module_instance):
323      """ 
324      Given a BatchNorm2D instance and a node check the following conditions: 
325      - the input type can be expanded to a size 4 tensor: t =  (x_1, x_2, x_3, x_4) 
326      - the current node type can be expanded to a size 4 tensor: t' =  (x_1', x_2', x_3', x_4') 
327      - t is consistent with t' 
328      - x_2 is consistent with the module's num_features 
329      - x_2' is consistent with the module's num_features 
330      output type: the more precise type of t and t' 
331      """
332      assert isinstance(n.args[0], Node)
333      n.args[0].type = expand_to_tensor_dim(n.args[0].type, 4)
334      arg_type = n.args[0].type
335      n.type = expand_to_tensor_dim(n.type, 4)
336  
337      # we check the conditions on the incoming argument
338      # and any existing annotation
339      # we also check for consistency between both annotations
340      if (
341          is_consistent(arg_type.__args__[1], module_instance.num_features)
342          and is_consistent(n.type.__args__[1], module_instance.num_features)
343          and is_consistent(arg_type, n.type)
344      ):
345          # we choose the more precise type
346          # to be the node type
347          # so if an incoming argument has more type information
348          # we set this node's type to be the argument type
349          n.type = get_greatest_upper_bound(arg_type, n.type)
350          return n.type
351      else:
352          raise TypeError(
353              f"Cannot apply {module_instance} with input type {arg_type} and existing type {n.type} on {n}"
354          )

357  def calculate_out_dimension(d_in, module_instance, index):
358      """ 
359      For calculating h_in and w_out according to the conv2D documentation 
360      """
361      padding = (
362          (module_instance.padding, module_instance.padding)
363          if isinstance(module_instance.padding, int)
364          else module_instance.padding
365      )
366      kernel_size = (
367          (module_instance.kernel_size, module_instance.kernel_size)
368          if isinstance(module_instance.kernel_size, int)
369          else module_instance.kernel_size
370      )
371      stride = (
372          (module_instance.stride, module_instance.stride)
373          if isinstance(module_instance.stride, int)
374          else module_instance.stride
375      )
376      dilation = (
377          (module_instance.dilation, module_instance.dilation)
378          if isinstance(module_instance.dilation, int)
379          else module_instance.dilation
380      )
381  
382      DIMENSION_TYPES = (int, sympy.Symbol)
383  
384      if d_in == Dyn:
385          return Dyn
386  
387      elif isinstance(d_in, DIMENSION_TYPES):
388          n = d_in + 2 * padding[index] - dilation[index] * (kernel_size[index] - 1) - 1
389  
390          return (n // stride[0]) + 1
391  
392      else:
393          raise TypeError(
394              f"{d_in} in {module_instance} must be a number or Dyn. Received {type(d_in)}"
395          )

398  def get_greatest_upper_bound(type1, type2):
399      """ 
400      Get the most precise type that's consistent with the given types 
401      """
402      if type1 == Dyn:
403          return type2
404      elif type2 == Dyn:
405          return type1
406      elif isinstance(type1, TensorType) and isinstance(type2, TensorType):
407          if not is_consistent(type1, type2):
408              raise TypeError(f"Inconsistent types {type1}, {type2}")
409          gub = [
410              t1 if is_more_precise(t1, t2) else t2
411              for (t1, t2) in zip(type1.__args__, type2.__args__)
412          ]
413          return TensorType(tuple(gub))

416  @register_inference_rule(Conv2d)
417  def conv2d_inference_rule(n: Node, module_instance):
418      """ 
419      Given a Conv2D instance and a node check the following conditions: 
420      - the input type can be expanded to a size 4 tensor: t =  (x_1, x_2, H, W) 
421      - the current node type can be expanded to a size 4 tensor: t' =  (x_1', x_2', x_3', x_4') 
422      - x_2 is consistent with the module's in_channels 
423      - let o = (x_1, out_channels, H_out, W_out) 
424      then the output is the greatest upper bound of o and the existing node type t'. 
425      """
426      assert isinstance(n.args[0], Node)
427      n.args[0].type = expand_to_tensor_dim(n.args[0].type, 4)
428      arg_type = n.args[0].type
429      curr_node_type = expand_to_tensor_dim(n.type, 4)
430  
431      if is_consistent(arg_type.__args__[1], module_instance.in_channels):
432          w_in = arg_type.__args__[3]
433          h_in = arg_type.__args__[2]
434          h_out = calculate_out_dimension(h_in, module_instance, 0)
435          w_out = calculate_out_dimension(w_in, module_instance, 1)
436          new_type = TensorType(
437              (arg_type.__args__[0], module_instance.out_channels, h_out, w_out)
438          )
439          gub = get_greatest_upper_bound(new_type, curr_node_type)
440          n.type = gub
441          return n.type
442      else:
443          raise TypeError(
444              f"Cannot apply {module_instance} with input type {arg_type} and existing type {n.type} on {n}"
445          )

463  def maxpool2d_check(typ, module_instance):
464      """ 
465      Applies the maxpool2d shape information to the input 
466      this affects the last two dimensions 
467      """
468      new_type_list = list(typ.__args__)
469      if len(new_type_list) == 4 or len(new_type_list) == 3:
470          w_in = new_type_list[-1]
471          h_in = new_type_list[-2]
472  
473          h_out = calculate_out_dimension(h_in, module_instance, 0)
474          w_out = calculate_out_dimension(w_in, module_instance, 1)
475  
476          new_type_list[-1] = w_out
477          new_type_list[-2] = h_out
478          return TensorType(tuple(new_type_list))
479  
480      else:
481          raise TypeError(f"Wrong size {typ} for {module_instance}")

484  @register_inference_rule(torch.nn.MaxPool2d)
485  def maxpool2d_inference_rule(n: Node, module_instance):
486      """ 
487      Given a MaxPool2D instance and a node check the following conditions: 
488      - Input size matches size 3 or 4 
489      - Current node type is consistent with the output type we will calculate 
490      - Input size matches output size and the last two dimensions of the output 
491        are w_out and h_out. The remaining dimensions are the same as the input 
492      - Our final result is the greatest upper bound of the output we calculate 
493        and the current node type. 
494      """
495      assert isinstance(n.args[0], Node)
496  
497      if n.args[0].type == Dyn and isinstance(n.type, TensorType):
498          n.args[0].type = expand_to_tensor_dim(n.args[0].type, len(n.type.__args__))
499      if isinstance(n.args[0].type, TensorType):
500          output = maxpool2d_check(n.args[0].type, module_instance)
501          n.type = get_greatest_upper_bound(output, n.type)
502      return n.type

505  def linear_check(tensor_type, module_instance):
506      """ 
507      Checks that an input tensor type satisfies the conditions for linear operation 
508      and returns the output type based on in and out features given by module_instance 
509      """
510      if len(tensor_type.__args__) >= 2:
511          if is_consistent(module_instance.in_features, tensor_type.__args__[-1]):
512              new_type_args = list(tensor_type.__args__)
513              new_type_args[-1] = module_instance.out_features
514              return TensorType(tuple(new_type_args))
515          else:
516              raise TypeError(
517                  f"Inconsistent {module_instance.in_features} and {tensor_type.__args__[-1]} in {module_instance}"
518              )
519      else:
520          raise TypeError(f"Type {tensor_type} must have rank 2 or more.")

538  def adaptiveavgpool2d_check(tensor_type, module_instance):
539      output_size = module_instance.output_size
540      if isinstance(output_size, int):
541          output_size = [output_size, output_size]
542      elif isinstance(output_size, tuple):
543          output_size = list(output_size)
544          if output_size[0] is None:
545              output_size[0] = output_size[1]
546          if output_size[1] is None:
547              output_size[1] = output_size[0]
548  
549      new_type_list = list(tensor_type.__args__)
550  
551      if len(tensor_type.__args__) == 4 or len(tensor_type.__args__) == 3:
552          new_type_list[-1] = output_size[1]
553          new_type_list[-2] = output_size[0]
554  
555          return TensorType(tuple(new_type_list))
556  
557      else:
558          raise TypeError(f"Tensor ranks must be 3 or 4. Got {tensor_type}")

576  def flatten_check(tensor_type, start_dim, end_dim):
577      l = len(tensor_type.__args__)
578  
579      start_dim = l if start_dim == -1 else abs(start_dim)
580      end_dim = l + end_dim + 1 if end_dim < 0 else end_dim + 1
581  
582      if 0 <= start_dim <= (l - 1) and 0 <= end_dim <= l and start_dim < end_dim:
583          my_args = list(tensor_type.__args__)
584          lhs = my_args[0:start_dim]
585          rhs = my_args[end_dim:]
586          mid = my_args[start_dim:end_dim]
587          if Dyn in mid:
588              mid = [Dyn]
589          else:
590              mid = [reduce(operator.mul, my_args[start_dim:end_dim])]
591          new_type_list = lhs + mid + rhs
592          return TensorType(tuple(new_type_list))
593      else:
594          raise TypeError(
595              f"Incompatible dimensions {start_dim}, {end_dim - 1} in type {tensor_type}"
596          )

599  @register_inference_rule(torch.flatten)
600  def flatten_inference_rule(n: Node):
601      """ 
602      Applies the flatten shape information to the input then gets the 
603      greatest upper bound of the resulting type and the existing type 
604      """
605      assert isinstance(n.args[0], Node)
606  
607      # set the default start and end dims
608      start_dim = 1
609      end_dim = -1
610  
611      if len(n.args) > 1:
612          assert isinstance(n.args[1], int)
613          start_dim = n.args[1]
614  
615      if len(n.args) > 2:
616          assert isinstance(n.args[2], int)
617          end_dim = n.args[2]
618  
619      if n.args[0].type == Dyn and isinstance(n.type, TensorType):
620          n.args[0].type = expand_to_tensor_dim(n.args[0].type, len(n.type.__args__))
621  
622      if isinstance(n.args[0].type, TensorType):
623          output_type = flatten_check(n.args[0].type, start_dim, end_dim)
624          n.type = get_greatest_upper_bound(output_type, n.type)
625  
626      return n.type

629  class GraphTypeChecker:
630      def __init__(self, env, traced):
631          self.env = env
632          self.traced = traced
633  
634      def type_check(self):
635          """ 
636          A gradual type checker for graphs 
637          Effect: every node's field type will be 
638          populated with a type after type-checking is done 
639          """
640          graph = self.traced.graph
641  
642          # type check every node with gradual type rules
643          # if any node does not type check return false
644          for n in graph.nodes:
645              self.type_check_node(n)
646          return True
647  
648      def type_check_node(self, n: Node):
649          """ 
650          Type check a given fx node. 
651          Current operations: 
652          - Reshape 
653          - Transpose 
654          - Add 
655          - Relu 
656          - conv2d 
657          - batchnorm2d 
658          - flatten 
659          - maxpool2d 
660          - adaptiveavgpool2d 
661          - linear 
662          """
663          if n.type is None:
664              n.type = Dyn
665  
666          if n.op == "placeholder":
667              return n.type
668  
669          elif n.op == "get_attr":
670              t = get_parameter(self.traced, n.target)  # type: ignore[arg-type]
671              if isinstance(t.data, torch.Tensor):
672                  n.type = TensorType(t.data.shape)
673              return n.type
674  
675          elif n.op == "call_function":
676              if n.target == getattr:
677                  assert getattr in _INFERENCE_RULES
678                  return _INFERENCE_RULES[n.target](n, self.traced)
679  
680              elif n.target in _INFERENCE_RULES:
681                  return _INFERENCE_RULES[n.target](n)
682              else:
683                  raise RuntimeError(
684                      f"No inference rule registered for target {n.target}!"
685                  )
686  
687          elif n.op == "call_module":
688              module_instance = self.traced.get_submodule(n.target)
689              if type(module_instance) in _INFERENCE_RULES:
690                  return _INFERENCE_RULES[type(module_instance)](n, module_instance)
691              else:
692                  raise RuntimeError(
693                      f"No inference rule registered for class {type(module_instance)}!"
694                  )
695  
696          elif n.op == "output":
697  
698              def get_node_type(a):
699                  return a.type
700  
701              n.type = torch.fx.node.map_arg(n.args[0], get_node_type)
702              return n.type
703  
704          else:
705              raise NotImplementedError(f"Method {n.op} not yet implemented")

769  @register_refinement_rule(torch.add)
770  @register_refinement_rule(operator.add)
771  def element_wise_eq(n: Node):
772      """ 
773      For element-wise operations and handles broadcasting. 
774      Note that after applying broadcasting to the arguments 
775      we are able to determine if certain dimensions have not been broadcast 
776      if they are symbolicallu equal. 
777   
778      in this case, we can establish equality between those dimensions and the 
779      corresponding output dimensions. 
780   
781      Note that it takes two iterations for this result. One iteration to establish 
782      equality between certain dimensions of the operands (requiring the whole solver 
783      including unification) and another iteration to establish equality between the operands 
784      and the resulting type, requiring another round of constraint generation and unificaiton. 
785      """
786      res = []
787      if isinstance(n.args[0], Node) and isinstance(n.args[1], Node):
788          arg_type1 = n.args[0].type
789          arg_type2 = n.args[1].type
790          if (
791              isinstance(arg_type1, TensorType)
792              and isinstance(arg_type2, TensorType)
793              and isinstance(n.type, TensorType)
794          ):
795              args1, args2 = broadcast_types(arg_type1, arg_type2)
796              # by this point, we know that args1 and args2 are the same size.
797              a1 = args1.__args__
798              a2 = args2.__args__
799              a3 = n.type.__args__
800  
801              # we would be here in the second iteration where we establish equality
802              # between operand type dimensions and the resulting type dimensions
803              r = []
804              for x, y, z in zip(a1, a2, a3):
805                  if x == y:
806                      r.append(Equality(x, z))
807              res = r
808      return res

811  @register_refinement_rule(torch.flatten)
812  def flatten_refinement_rule(n: Node):
813      """ 
814      Generates equality constraints between the dimensions of the input and output 
815      that will not be involved in the flatten operation 
816      """
817      assert isinstance(n.args[0], Node)
818  
819      eq_const = []
820  
821      start_dim = 1
822      end_dim = -1
823  
824      if len(n.args) > 1:
825          assert isinstance(n.args[1], int)
826          start_dim = n.args[1]
827  
828      if len(n.args) > 2:
829          assert isinstance(n.args[2], int)
830          end_dim = n.args[2]
831  
832      if isinstance(n.type, TensorType) and isinstance(n.args[0].type, TensorType):
833          l = len(n.type.__args__)
834          arg_type = n.args[0].type
835          start_dim = l if start_dim == -1 else start_dim
836          end_dim = l + end_dim + 1 if end_dim < 0 else end_dim + 1
837  
838          for t1, t2 in zip(n.type.__args__[0:start_dim], arg_type.__args__[0:start_dim]):
839              eq_const.append(Equality(t1, t2))
840  
841          for t1, t2 in zip(n.type.__args__[end_dim:], arg_type.__args__[end_dim:]):
842              eq_const.append(Equality(t1, t2))
843      return eq_const

864  class Refine:
865      """ 
866      Symbolic shape inference. 
867      Generates constraints over type variables. 
868      Currently all constraints are equality constraints. 
869      """
870  
871      def __init__(self, traced):
872          self.constraints = []
873          self.traced = traced
874          self.symbol_iter = itertools.count(start=0, step=1)
875  
876      def refine(self):
877          """ 
878          Generates constraints for 
879          every node in the graph based on 
880          the operation. 
881          """
882          graph = self.traced.graph
883          for n in graph.nodes:
884              self.refine_node(n)
885          return True
886  
887      def symbolic_relations(self):
888          """ 
889          Infers algebraic relations 
890          """
891          graph = self.traced.graph
892          for n in graph.nodes:
893              self.infer_symbolic_relations(n)
894          return True
895  
896      def replace_dyn_with_fresh_var(self, typ):
897          """ 
898          Replace all unknown types with fresh type variables. 
899          """
900          if typ == Dyn:
901              new_symbol = Var(next(self.symbol_iter))
902              return new_symbol
903          elif isinstance(typ, TensorType):
904              new_args = [self.replace_dyn_with_fresh_var(a) for a in typ.__args__]
905              return TensorType(tuple(new_args))
906          elif isinstance(typ, list):
907              return [self.replace_dyn_with_fresh_var(t) for t in typ]
908          elif isinstance(typ, tuple):
909              return (self.replace_dyn_with_fresh_var(t) for t in typ)
910          else:
911              return typ
912  
913      def convert_to_sympy_symbols(self, typ):
914          """ 
915          Replace all unknown types with fresh type variables. 
916          """
917          if isinstance(typ, Var):
918              return sympy.symbols(str(typ))
919          elif isinstance(typ, TensorType):
920              new_args = [self.convert_to_sympy_symbols(a) for a in typ.__args__]
921              return TensorType(tuple(new_args))
922          elif isinstance(typ, list):
923              return [self.convert_to_sympy_symbols(t) for t in typ]
924          elif isinstance(typ, tuple):
925              return (self.convert_to_sympy_symbols(t) for t in typ)
926          else:
927              return typ
928  
929      def refine_node(self, n: Node):
930          """ 
931          Returns a list of equality constraints for 
932          call_module and call_function nodes. 
933          Models the relation between input and output dimensions 
934          using constraints in case they are both tensors. 
935          All operations used in resnet50 are defined. 
936          """
937          if n.type is None:
938              n.type = Dyn
939  
940          n.type = self.replace_dyn_with_fresh_var(n.type)
941  
942          if n.op == "call_function":
943              if n.target in _REFINEMENT_RULES:
944                  self.constraints += _REFINEMENT_RULES[n.target](n)
945              else:
946                  pass
947  
948          if n.op == "call_module":
949              module_instance = self.traced.get_submodule(n.target)
950              if type(module_instance) in _REFINEMENT_RULES:
951                  self.constraints += _REFINEMENT_RULES[type(module_instance)](n)
952              else:
953                  pass
954  
955          if n.op == "output":
956  
957              def get_node_type(a):
958                  return a.type
959  
960              n.type = torch.fx.node.map_arg(n.args[0], get_node_type)
961              return n.type
962  
963          else:
964              pass
965  
966      def infer_symbolic_relations(self, n: Node):
967          n.type = self.convert_to_sympy_symbols(n.type)
968          if n.op == "call_function":
969              if n.target in _RULES:
970                  return _RULES[n.target](n)
971              else:
972                  pass
973  
974          if n.op == "call_module":
975              module_instance = self.traced.get_submodule(n.target)
976              if type(module_instance) in _RULES:
977                  return _RULES[type(module_instance)](n, module_instance)
978              else:
979                  pass
980  
981          if n.op == "output":
982  
983              def get_node_type(a):
984                  return a.type
985  
986              n.type = torch.fx.node.map_arg(n.args[0], get_node_type)
987              return n.type
988  
989          else:
990              pass

993  def get_parameter(traced, target: str):
994      """ 
995      Returns the parameter given by ``target`` if it exists, 
996      otherwise throws an error. 
997   
998      See the docstring for ``get_submodule`` for a more detailed 
999      explanation of this method's functionality as well as how to 
1000     correctly specify ``target``. 
1001  
1002     Args: 
1003         target: The fully-qualified string name of the Parameter 
1004             to look for. (See ``get_submodule`` for how to specify a 
1005             fully-qualified string.) 
1006  
1007     Returns: 
1008         torch.nn.Parameter: The Parameter referenced by ``target`` 
1009  
1010     Raises: 
1011         AttributeError: If the target string references an invalid 
1012             path or resolves to something that is not an 
1013             ``nn.Parameter`` 
1014     """
1015     module_path, _, param_name = target.rpartition(".")
1016 
1017     mod: torch.nn.Module = traced.get_submodule(module_path)
1018 
1019     if not hasattr(mod, param_name):
1020         raise AttributeError(mod._get_name() + " has no attribute `" + param_name + "`")
1021 
1022     param: torch.nn.Parameter = getattr(mod, param_name)
1023 
1024     return param
