24   class Rationals(Set, metaclass=Singleton):
25       """ 
26       Represents the rational numbers. This set is also available as 
27       the singleton ``S.Rationals``. 
28    
29       Examples 
30       ======== 
31    
32       >>> from sympy import S 
33       >>> S.Half in S.Rationals 
34       True 
35       >>> iterable = iter(S.Rationals) 
36       >>> [next(iterable) for i in range(12)] 
37       [0, 1, -1, 1/2, 2, -1/2, -2, 1/3, 3, -1/3, -3, 2/3] 
38       """
39   
40       is_iterable = True
41       _inf = S.NegativeInfinity
42       _sup = S.Infinity
43       is_empty = False
44       is_finite_set = False
45   
46       def _contains(self, other):
47           if not isinstance(other, Expr):
48               return S.false
49           return tfn[other.is_rational]
50   
51       def __iter__(self):
52           yield S.Zero
53           yield S.One
54           yield S.NegativeOne
55           d = 2
56           while True:
57               for n in range(d):
58                   if igcd(n, d) == 1:
59                       yield Rational(n, d)
60                       yield Rational(d, n)
61                       yield Rational(-n, d)
62                       yield Rational(-d, n)
63               d += 1
64   
65       @property
66       def _boundary(self):
67           return S.Reals
68   
69       def _kind(self):
70           return SetKind(NumberKind)

73   class Naturals(Set, metaclass=Singleton):
74       """ 
75       Represents the natural numbers (or counting numbers) which are all 
76       positive integers starting from 1. This set is also available as 
77       the singleton ``S.Naturals``. 
78    
79       Examples 
80       ======== 
81    
82       >>> from sympy import S, Interval, pprint 
83       >>> 5 in S.Naturals 
84       True 
85       >>> iterable = iter(S.Naturals) 
86       >>> next(iterable) 
87       1 
88       >>> next(iterable) 
89       2 
90       >>> next(iterable) 
91       3 
92       >>> pprint(S.Naturals.intersect(Interval(0, 10))) 
93       {1, 2, ..., 10} 
94    
95       See Also 
96       ======== 
97    
98       Naturals0 : non-negative integers (i.e. includes 0, too) 
99       Integers : also includes negative integers 
100      """
101  
102      is_iterable = True
103      _inf: Integer = S.One
104      _sup = S.Infinity
105      is_empty = False
106      is_finite_set = False
107  
108      def _contains(self, other):
109          if not isinstance(other, Expr):
110              return S.false
111          elif other.is_positive and other.is_integer:
112              return S.true
113          elif other.is_integer is False or other.is_positive is False:
114              return S.false
115  
116      def _eval_is_subset(self, other):
117          return Range(1, oo).is_subset(other)
118  
119      def _eval_is_superset(self, other):
120          return Range(1, oo).is_superset(other)
121  
122      def __iter__(self):
123          i = self._inf
124          while True:
125              yield i
126              i = i + 1
127  
128      @property
129      def _boundary(self):
130          return self
131  
132      def as_relational(self, x):
133          return And(Eq(floor(x), x), x >= self.inf, x < oo)
134  
135      def _kind(self):
136          return SetKind(NumberKind)

291  class ImageSet(Set):
292      """ 
293      Image of a set under a mathematical function. The transformation 
294      must be given as a Lambda function which has as many arguments 
295      as the elements of the set upon which it operates, e.g. 1 argument 
296      when acting on the set of integers or 2 arguments when acting on 
297      a complex region. 
298   
299      This function is not normally called directly, but is called 
300      from ``imageset``. 
301   
302   
303      Examples 
304      ======== 
305   
306      >>> from sympy import Symbol, S, pi, Dummy, Lambda 
307      >>> from sympy import FiniteSet, ImageSet, Interval 
308   
309      >>> x = Symbol('x') 
310      >>> N = S.Naturals 
311      >>> squares = ImageSet(Lambda(x, x**2), N) # {x**2 for x in N} 
312      >>> 4 in squares 
313      True 
314      >>> 5 in squares 
315      False 
316   
317      >>> FiniteSet(0, 1, 2, 3, 4, 5, 6, 7, 9, 10).intersect(squares) 
318      {1, 4, 9} 
319   
320      >>> square_iterable = iter(squares) 
321      >>> for i in range(4): 
322      ...     next(square_iterable) 
323      1 
324      4 
325      9 
326      16 
327   
328      If you want to get value for `x` = 2, 1/2 etc. (Please check whether the 
329      `x` value is in ``base_set`` or not before passing it as args) 
330   
331      >>> squares.lamda(2) 
332      4 
333      >>> squares.lamda(S(1)/2) 
334      1/4 
335   
336      >>> n = Dummy('n') 
337      >>> solutions = ImageSet(Lambda(n, n*pi), S.Integers) # solutions of sin(x) = 0 
338      >>> dom = Interval(-1, 1) 
339      >>> dom.intersect(solutions) 
340      {0} 
341   
342      See Also 
343      ======== 
344   
345      sympy.sets.sets.imageset 
346      """
347      def __new__(cls, flambda, *sets):
348          if not isinstance(flambda, Lambda):
349              raise ValueError('First argument must be a Lambda')
350  
351          signature = flambda.signature
352  
353          if len(signature) != len(sets):
354              raise ValueError('Incompatible signature')
355  
356          sets = [_sympify(s) for s in sets]
357  
358          if not all(isinstance(s, Set) for s in sets):
359              raise TypeError("Set arguments to ImageSet should of type Set")
360  
361          if not all(cls._check_sig(sg, st) for sg, st in zip(signature, sets)):
362              raise ValueError("Signature %s does not match sets %s" % (signature, sets))
363  
364          if flambda is S.IdentityFunction and len(sets) == 1:
365              return sets[0]
366  
367          if not set(flambda.variables) & flambda.expr.free_symbols:
368              is_empty = fuzzy_or(s.is_empty for s in sets)
369              if is_empty == True:
370                  return S.EmptySet
371              elif is_empty == False:
372                  return FiniteSet(flambda.expr)
373  
374          return Basic.__new__(cls, flambda, *sets)
375  
376      lamda = property(lambda self: self.args[0])
377      base_sets = property(lambda self: self.args[1:])
378  
379      @property
380      def base_set(self):
381          # XXX: Maybe deprecate this? It is poorly defined in handling
382          # the multivariate case...
383          sets = self.base_sets
384          if len(sets) == 1:
385              return sets[0]
386          else:
387              return ProductSet(*sets).flatten()
388  
389      @property
390      def base_pset(self):
391          return ProductSet(*self.base_sets)
392  
393      @classmethod
394      def _check_sig(cls, sig_i, set_i):
395          if sig_i.is_symbol:
396              return True
397          elif isinstance(set_i, ProductSet):
398              sets = set_i.sets
399              if len(sig_i) != len(sets):
400                  return False
401              # Recurse through the signature for nested tuples:
402              return all(cls._check_sig(ts, ps) for ts, ps in zip(sig_i, sets))
403          else:
404              # XXX: Need a better way of checking whether a set is a set of
405              # Tuples or not. For example a FiniteSet can contain Tuples
406              # but so can an ImageSet or a ConditionSet. Others like
407              # Integers, Reals etc can not contain Tuples. We could just
408              # list the possibilities here... Current code for e.g.
409              # _contains probably only works for ProductSet.
410              return True # Give the benefit of the doubt
411  
412      def __iter__(self):
413          already_seen = set()
414          for i in self.base_pset:
415              val = self.lamda(*i)
416              if val in already_seen:
417                  continue
418              else:
419                  already_seen.add(val)
420                  yield val

422      def _is_multivariate(self):
423          return len(self.lamda.variables) > 1
424  
425      def _contains(self, other):
426          from sympy.solvers.solveset import _solveset_multi
427  
428          def get_symsetmap(signature, base_sets):
429              '''Attempt to get a map of symbols to base_sets'''
430              queue = list(zip(signature, base_sets))
431              symsetmap = {}
432              for sig, base_set in queue:
433                  if sig.is_symbol:
434                      symsetmap[sig] = base_set
435                  elif base_set.is_ProductSet:
436                      sets = base_set.sets
437                      if len(sig) != len(sets):
438                          raise ValueError("Incompatible signature")
439                      # Recurse
440                      queue.extend(zip(sig, sets))
441                  else:
442                      # If we get here then we have something like sig = (x, y) and
443                      # base_set = {(1, 2), (3, 4)}. For now we give up.
444                      return None
445  
446              return symsetmap
447  
448          def get_equations(expr, candidate):
449              '''Find the equations relating symbols in expr and candidate.'''
450              queue = [(expr, candidate)]
451              for e, c in queue:
452                  if not isinstance(e, Tuple):
453                      yield Eq(e, c)
454                  elif not isinstance(c, Tuple) or len(e) != len(c):
455                      yield False
456                      return
457                  else:
458                      queue.extend(zip(e, c))
459  
460          # Get the basic objects together:
461          other = _sympify(other)
462          expr = self.lamda.expr
463          sig = self.lamda.signature
464          variables = self.lamda.variables
465          base_sets = self.base_sets
466  
467          # Use dummy symbols for ImageSet parameters so they don't match
468          # anything in other
469          rep = {v: Dummy(v.name) for v in variables}
470          variables = [v.subs(rep) for v in variables]
471          sig = sig.subs(rep)
472          expr = expr.subs(rep)
473  
474          # Map the parts of other to those in the Lambda expr
475          equations = []
476          for eq in get_equations(expr, other):
477              # Unsatisfiable equation?
478              if eq is False:
479                  return S.false
480              equations.append(eq)
481  
482          # Map the symbols in the signature to the corresponding domains
483          symsetmap = get_symsetmap(sig, base_sets)
484          if symsetmap is None:
485              # Can't factor the base sets to a ProductSet
486              return None
487  
488          # Which of the variables in the Lambda signature need to be solved for?
489          symss = (eq.free_symbols for eq in equations)
490          variables = set(variables) & reduce(set.union, symss, set())
491  
492          # Use internal multivariate solveset
493          variables = tuple(variables)
494          base_sets = [symsetmap[v] for v in variables]
495          solnset = _solveset_multi(equations, variables, base_sets)
496          if solnset is None:
497              return None
498          return tfn[fuzzy_not(solnset.is_empty)]
499  
500      @property
501      def is_iterable(self):
502          return all(s.is_iterable for s in self.base_sets)
503  
504      def doit(self, **hints):
505          from sympy.sets.setexpr import SetExpr
506          f = self.lamda
507          sig = f.signature
508          if len(sig) == 1 and sig[0].is_symbol and isinstance(f.expr, Expr):
509              base_set = self.base_sets[0]
510              return SetExpr(base_set)._eval_func(f).set
511          if all(s.is_FiniteSet for s in self.base_sets):
512              return FiniteSet(*(f(*a) for a in product(*self.base_sets)))
513          return self
514  
515      def _kind(self):
516          return SetKind(self.lamda.expr.kind)

519  class Range(Set):
520      """ 
521      Represents a range of integers. Can be called as ``Range(stop)``, 
522      ``Range(start, stop)``, or ``Range(start, stop, step)``; when ``step`` is 
523      not given it defaults to 1. 
524   
525      ``Range(stop)`` is the same as ``Range(0, stop, 1)`` and the stop value 
526      (just as for Python ranges) is not included in the Range values. 
527   
528          >>> from sympy import Range 
529          >>> list(Range(3)) 
530          [0, 1, 2] 
531   
532      The step can also be negative: 
533   
534          >>> list(Range(10, 0, -2)) 
535          [10, 8, 6, 4, 2] 
536   
537      The stop value is made canonical so equivalent ranges always 
538      have the same args: 
539   
540          >>> Range(0, 10, 3) 
541          Range(0, 12, 3) 
542   
543      Infinite ranges are allowed. ``oo`` and ``-oo`` are never included in the 
544      set (``Range`` is always a subset of ``Integers``). If the starting point 
545      is infinite, then the final value is ``stop - step``. To iterate such a 
546      range, it needs to be reversed: 
547   
548          >>> from sympy import oo 
549          >>> r = Range(-oo, 1) 
550          >>> r[-1] 
551          0 
552          >>> next(iter(r)) 
553          Traceback (most recent call last): 
554          ... 
555          TypeError: Cannot iterate over Range with infinite start 
556          >>> next(iter(r.reversed)) 
557          0 
558   
559      Although ``Range`` is a :class:`Set` (and supports the normal set 
560      operations) it maintains the order of the elements and can 
561      be used in contexts where ``range`` would be used. 
562   
563          >>> from sympy import Interval 
564          >>> Range(0, 10, 2).intersect(Interval(3, 7)) 
565          Range(4, 8, 2) 
566          >>> list(_) 
567          [4, 6] 
568   
569      Although slicing of a Range will always return a Range -- possibly 
570      empty -- an empty set will be returned from any intersection that 
571      is empty: 
572   
573          >>> Range(3)[:0] 
574          Range(0, 0, 1) 
575          >>> Range(3).intersect(Interval(4, oo)) 
576          EmptySet 
577          >>> Range(3).intersect(Range(4, oo)) 
578          EmptySet 
579   
580      Range will accept symbolic arguments but has very limited support 
581      for doing anything other than displaying the Range: 
582   
583          >>> from sympy import Symbol, pprint 
584          >>> from sympy.abc import i, j, k 
585          >>> Range(i, j, k).start 
586          i 
587          >>> Range(i, j, k).inf 
588          Traceback (most recent call last): 
589          ... 
590          ValueError: invalid method for symbolic range 
591   
592      Better success will be had when using integer symbols: 
593   
594          >>> n = Symbol('n', integer=True) 
595          >>> r = Range(n, n + 20, 3) 
596          >>> r.inf 
597          n 
598          >>> pprint(r) 
599          {n, n + 3, ..., n + 18} 
600      """
601  
602      def __new__(cls, *args):
603          if len(args) == 1:
604              if isinstance(args[0], range):
605                  raise TypeError(
606                      'use sympify(%s) to convert range to Range' % args[0])
607  
608          # expand range
609          slc = slice(*args)
610  
611          if slc.step == 0:
612              raise ValueError("step cannot be 0")
613  
614          start, stop, step = slc.start or 0, slc.stop, slc.step or 1
615          try:
616              ok = []
617              for w in (start, stop, step):
618                  w = sympify(w)
619                  if w in [S.NegativeInfinity, S.Infinity] or (
620                          w.has(Symbol) and w.is_integer != False):
621                      ok.append(w)
622                  elif not w.is_Integer:
623                      if w.is_infinite:
624                          raise ValueError('infinite symbols not allowed')
625                      raise ValueError
626                  else:
627                      ok.append(w)
628          except ValueError:
629              raise ValueError(filldedent(''' 
630      Finite arguments to Range must be integers; `imageset` can define 
631      other cases, e.g. use `imageset(i, i/10, Range(3))` to give 
632      [0, 1/10, 1/5].'''))
633          start, stop, step = ok
634  
635          null = False
636          if any(i.has(Symbol) for i in (start, stop, step)):
637              dif = stop - start
638              n = dif/step
639              if n.is_Rational:
640                  if dif == 0:
641                      null = True
642                  else:  # (x, x + 5, 2) or (x, 3*x, x)
643                      n = floor(n)
644                      end = start + n*step
645                      if dif.is_Rational:  # (x, x + 5, 2)
646                          if (end - stop).is_negative:
647                              end += step
648                      else:  # (x, 3*x, x)
649                          if (end/stop - 1).is_negative:
650                              end += step
651              elif n.is_extended_negative:
652                  null = True
653              else:
654                  end = stop  # other methods like sup and reversed must fail
655          elif start.is_infinite:
656              span = step*(stop - start)
657              if span is S.NaN or span <= 0:
658                  null = True
659              elif step.is_Integer and stop.is_infinite and abs(step) != 1:
660                  raise ValueError(filldedent(''' 
661                      Step size must be %s in this case.''' % (1 if step > 0 else -1)))
662              else:
663                  end = stop
664          else:
665              oostep = step.is_infinite
666              if oostep:
667                  step = S.One if step > 0 else S.NegativeOne
668              n = ceiling((stop - start)/step)
669              if n <= 0:
670                  null = True
671              elif oostep:
672                  step = S.One  # make it canonical
673                  end = start + step
674              else:
675                  end = start + n*step
676          if null:
677              start = end = S.Zero
678              step = S.One
679          return Basic.__new__(cls, start, end, step)
680  
681      start = property(lambda self: self.args[0])
682      stop = property(lambda self: self.args[1])
683      step = property(lambda self: self.args[2])
684  
685      @property
686      def reversed(self):
687          """Return an equivalent Range in the opposite order. 
688   
689          Examples 
690          ======== 
691   
692          >>> from sympy import Range 
693          >>> Range(10).reversed 
694          Range(9, -1, -1) 
695          """
696          if self.has(Symbol):
697              n = (self.stop - self.start)/self.step
698              if not n.is_extended_positive or not all(
699                      i.is_integer or i.is_infinite for i in self.args):
700                  raise ValueError('invalid method for symbolic range')
701          if self.start == self.stop:
702              return self
703          return self.func(
704              self.stop - self.step, self.start - self.step, -self.step)
705  
706      def _kind(self):
707          return SetKind(NumberKind)
708  
709      def _contains(self, other):
710          if self.start == self.stop:
711              return S.false
712          if other.is_infinite:
713              return S.false
714          if not other.is_integer:
715              return tfn[other.is_integer]
716          if self.has(Symbol):
717              n = (self.stop - self.start)/self.step
718              if not n.is_extended_positive or not all(
719                      i.is_integer or i.is_infinite for i in self.args):
720                  return
721          else:
722              n = self.size
723          if self.start.is_finite:
724              ref = self.start
725          elif self.stop.is_finite:
726              ref = self.stop
727          else:  # both infinite; step is +/- 1 (enforced by __new__)
728              return S.true
729          if n == 1:
730              return Eq(other, self[0])
731          res = (ref - other) % self.step
732          if res == S.Zero:
733              if self.has(Symbol):
734                  d = Dummy('i')
735                  return self.as_relational(d).subs(d, other)
736              return And(other >= self.inf, other <= self.sup)
737          elif res.is_Integer:  # off sequence
738              return S.false
739          else:  # symbolic/unsimplified residue modulo step
740              return None
741  
742      def __iter__(self):
743          n = self.size  # validate
744          if not (n.has(S.Infinity) or n.has(S.NegativeInfinity) or n.is_Integer):
745              raise TypeError("Cannot iterate over symbolic Range")
746          if self.start in [S.NegativeInfinity, S.Infinity]:
747              raise TypeError("Cannot iterate over Range with infinite start")
748          elif self.start != self.stop:
749              i = self.start
750              if n.is_infinite:
751                  while True:
752                      yield i
753                      i += self.step
754              else:
755                  for _ in range(n):
756                      yield i
757                      i += self.step
758  
759      @property
760      def is_iterable(self):
761          # Check that size can be determined, used by __iter__
762          dif = self.stop - self.start
763          n = dif/self.step
764          if not (n.has(S.Infinity) or n.has(S.NegativeInfinity) or n.is_Integer):
765              return False
766          if self.start in [S.NegativeInfinity, S.Infinity]:
767              return False
768          if not (n.is_extended_nonnegative and all(i.is_integer for i in self.args)):
769              return False
770          return True
771  
772      def __len__(self):
773          rv = self.size
774          if rv is S.Infinity:
775              raise ValueError('Use .size to get the length of an infinite Range')
776          return int(rv)
777  
778      @property
779      def size(self):
780          if self.start == self.stop:
781              return S.Zero
782          dif = self.stop - self.start
783          n = dif/self.step
784          if n.is_infinite:
785              return S.Infinity
786          if  n.is_extended_nonnegative and all(i.is_integer for i in self.args):
787              return abs(floor(n))
788          raise ValueError('Invalid method for symbolic Range')
789  
790      @property
791      def is_finite_set(self):
792          if self.start.is_integer and self.stop.is_integer:
793              return True
794          return self.size.is_finite
795  
796      @property
797      def is_empty(self):
798          try:
799              return self.size.is_zero
800          except ValueError:
801              return None
802  
803      def __bool__(self):
804          # this only distinguishes between definite null range
805          # and non-null/unknown null; getting True doesn't mean
806          # that it actually is not null
807          b = is_eq(self.start, self.stop)
808          if b is None:
809              raise ValueError('cannot tell if Range is null or not')
810          return not bool(b)
811  
812      def __getitem__(self, i):
813          ooslice = "cannot slice from the end with an infinite value"
814          zerostep = "slice step cannot be zero"
815          infinite = "slicing not possible on range with infinite start"
816          # if we had to take every other element in the following
817          # oo, ..., 6, 4, 2, 0
818          # we might get oo, ..., 4, 0 or oo, ..., 6, 2
819          ambiguous = "cannot unambiguously re-stride from the end " + \
820              "with an infinite value"
821          if isinstance(i, slice):
822              if self.size.is_finite:  # validates, too
823                  if self.start == self.stop:
824                      return Range(0)
825                  start, stop, step = i.indices(self.size)
826                  n = ceiling((stop - start)/step)
827                  if n <= 0:
828                      return Range(0)
829                  canonical_stop = start + n*step
830                  end = canonical_stop - step
831                  ss = step*self.step
832                  return Range(self[start], self[end] + ss, ss)
833              else:  # infinite Range
834                  start = i.start
835                  stop = i.stop
836                  if i.step == 0:
837                      raise ValueError(zerostep)
838                  step = i.step or 1
839                  ss = step*self.step
840                  #---------------------
841                  # handle infinite Range
842                  #   i.e. Range(-oo, oo) or Range(oo, -oo, -1)
843                  # --------------------
844                  if self.start.is_infinite and self.stop.is_infinite:
845                      raise ValueError(infinite)
846                  #---------------------
847                  # handle infinite on right
848                  #   e.g. Range(0, oo) or Range(0, -oo, -1)
849                  # --------------------
850                  if self.stop.is_infinite:
851                      # start and stop are not interdependent --
852                      # they only depend on step --so we use the
853                      # equivalent reversed values
854                      return self.reversed[
855                          stop if stop is None else -stop + 1:
856                          start if start is None else -start:
857                          step].reversed
858                  #---------------------
859                  # handle infinite on the left
860                  #   e.g. Range(oo, 0, -1) or Range(-oo, 0)
861                  # --------------------
862                  # consider combinations of
863                  # start/stop {== None, < 0, == 0, > 0} and
864                  # step {< 0, > 0}
865                  if start is None:
866                      if stop is None:
867                          if step < 0:
868                              return Range(self[-1], self.start, ss)
869                          elif step > 1:
870                              raise ValueError(ambiguous)
871                          else:  # == 1
872                              return self
873                      elif stop < 0:
874                          if step < 0:
875                              return Range(self[-1], self[stop], ss)
876                          else:  # > 0
877                              return Range(self.start, self[stop], ss)
878                      elif stop == 0:
879                          if step > 0:
880                              return Range(0)
881                          else:  # < 0
882                              raise ValueError(ooslice)
883                      elif stop == 1:
884                          if step > 0:
885                              raise ValueError(ooslice)  # infinite singleton
886                          else:  # < 0
887                              raise ValueError(ooslice)
888                      else:  # > 1
889                          raise ValueError(ooslice)
890                  elif start < 0:
891                      if stop is None:
892                          if step < 0:
893                              return Range(self[start], self.start, ss)
894                          else:  # > 0
895                              return Range(self[start], self.stop, ss)
896                      elif stop < 0:
897                          return Range(self[start], self[stop], ss)
898                      elif stop == 0:
899                          if step < 0:
900                              raise ValueError(ooslice)
901                          else:  # > 0
902                              return Range(0)
903                      elif stop > 0:
904                          raise ValueError(ooslice)
905                  elif start == 0:
906                      if stop is None:
907                          if step < 0:
908                              raise ValueError(ooslice)  # infinite singleton
909                          elif step > 1:
910                              raise ValueError(ambiguous)
911                          else:  # == 1
912                              return self
913                      elif stop < 0:
914                          if step > 1:
915                              raise ValueError(ambiguous)
916                          elif step == 1:
917                              return Range(self.start, self[stop], ss)
918                          else:  # < 0
919                              return Range(0)
920                      else:  # >= 0
921                          raise ValueError(ooslice)
922                  elif start > 0:
923                      raise ValueError(ooslice)
924          else:
925              if self.start == self.stop:
926                  raise IndexError('Range index out of range')
927              if not (all(i.is_integer or i.is_infinite
928                      for i in self.args) and ((self.stop - self.start)/
929                      self.step).is_extended_positive):
930                  raise ValueError('Invalid method for symbolic Range')
931              if i == 0:
932                  if self.start.is_infinite:
933                      raise ValueError(ooslice)
934                  return self.start
935              if i == -1:
936                  if self.stop.is_infinite:
937                      raise ValueError(ooslice)
938                  return self.stop - self.step
939              n = self.size  # must be known for any other index
940              rv = (self.stop if i < 0 else self.start) + i*self.step
941              if rv.is_infinite:
942                  raise ValueError(ooslice)
943              val = (rv - self.start)/self.step
944              rel = fuzzy_or([val.is_infinite,
945                              fuzzy_and([val.is_nonnegative, (n-val).is_nonnegative])])
946              if rel:
947                  return rv
948              if rel is None:
949                  raise ValueError('Invalid method for symbolic Range')
950              raise IndexError("Range index out of range")
951  
952      @property
953      def _inf(self):
954          if not self:
955              return S.EmptySet.inf
956          if self.has(Symbol):
957              if all(i.is_integer or i.is_infinite for i in self.args):
958                  dif = self.stop - self.start
959                  if self.step.is_positive and dif.is_positive:
960                      return self.start
961                  elif self.step.is_negative and dif.is_negative:
962                      return self.stop - self.step
963              raise ValueError('invalid method for symbolic range')
964          if self.step > 0:
965              return self.start
966          else:
967              return self.stop - self.step
968  
969      @property
970      def _sup(self):
971          if not self:
972              return S.EmptySet.sup
973          if self.has(Symbol):
974              if all(i.is_integer or i.is_infinite for i in self.args):
975                  dif = self.stop - self.start
976                  if self.step.is_positive and dif.is_positive:
977                      return self.stop - self.step
978                  elif self.step.is_negative and dif.is_negative:
979                      return self.start
980              raise ValueError('invalid method for symbolic range')
981          if self.step > 0:
982              return self.stop - self.step
983          else:
984              return self.start
985  
986      @property
987      def _boundary(self):
988          return self

990      def as_relational(self, x):
991          """Rewrite a Range in terms of equalities and logic operators. """
992          if self.start.is_infinite:
993              assert not self.stop.is_infinite  # by instantiation
994              a = self.reversed.start
995          else:
996              a = self.start
997          step = self.step
998          in_seq = Eq(Mod(x - a, step), 0)
999          ints = And(Eq(Mod(a, 1), 0), Eq(Mod(step, 1), 0))
1000         n = (self.stop - self.start)/self.step
1001         if n == 0:
1002             return S.EmptySet.as_relational(x)
1003         if n == 1:
1004             return And(Eq(x, a), ints)
1005         try:
1006             a, b = self.inf, self.sup
1007         except ValueError:
1008             a = None
1009         if a is not None:
1010             range_cond = And(
1011                 x > a if a.is_infinite else x >= a,
1012                 x < b if b.is_infinite else x <= b)
1013         else:
1014             a, b = self.start, self.stop - self.step
1015             range_cond = Or(
1016                 And(self.step >= 1, x > a if a.is_infinite else x >= a,
1017                 x < b if b.is_infinite else x <= b),
1018                 And(self.step <= -1, x < a if a.is_infinite else x <= a,
1019                 x > b if b.is_infinite else x >= b))
1020         return And(in_seq, ints, range_cond)

1025 def normalize_theta_set(theta):
1026     r""" 
1027     Normalize a Real Set `theta` in the interval `[0, 2\pi)`. It returns 
1028     a normalized value of theta in the Set. For Interval, a maximum of 
1029     one cycle $[0, 2\pi]$, is returned i.e. for theta equal to $[0, 10\pi]$, 
1030     returned normalized value would be $[0, 2\pi)$. As of now intervals 
1031     with end points as non-multiples of ``pi`` is not supported. 
1032  
1033     Raises 
1034     ====== 
1035  
1036     NotImplementedError 
1037         The algorithms for Normalizing theta Set are not yet 
1038         implemented. 
1039     ValueError 
1040         The input is not valid, i.e. the input is not a real set. 
1041     RuntimeError 
1042         It is a bug, please report to the github issue tracker. 
1043  
1044     Examples 
1045     ======== 
1046  
1047     >>> from sympy.sets.fancysets import normalize_theta_set 
1048     >>> from sympy import Interval, FiniteSet, pi 
1049     >>> normalize_theta_set(Interval(9*pi/2, 5*pi)) 
1050     Interval(pi/2, pi) 
1051     >>> normalize_theta_set(Interval(-3*pi/2, pi/2)) 
1052     Interval.Ropen(0, 2*pi) 
1053     >>> normalize_theta_set(Interval(-pi/2, pi/2)) 
1054     Union(Interval(0, pi/2), Interval.Ropen(3*pi/2, 2*pi)) 
1055     >>> normalize_theta_set(Interval(-4*pi, 3*pi)) 
1056     Interval.Ropen(0, 2*pi) 
1057     >>> normalize_theta_set(Interval(-3*pi/2, -pi/2)) 
1058     Interval(pi/2, 3*pi/2) 
1059     >>> normalize_theta_set(FiniteSet(0, pi, 3*pi)) 
1060     {0, pi} 
1061  
1062     """
1063     from sympy.functions.elementary.trigonometric import _pi_coeff
1064 
1065     if theta.is_Interval:
1066         interval_len = theta.measure
1067         # one complete circle
1068         if interval_len >= 2*S.Pi:
1069             if interval_len == 2*S.Pi and theta.left_open and theta.right_open:
1070                 k = _pi_coeff(theta.start)
1071                 return Union(Interval(0, k*S.Pi, False, True),
1072                         Interval(k*S.Pi, 2*S.Pi, True, True))
1073             return Interval(0, 2*S.Pi, False, True)
1074 
1075         k_start, k_end = _pi_coeff(theta.start), _pi_coeff(theta.end)
1076 
1077         if k_start is None or k_end is None:
1078             raise NotImplementedError("Normalizing theta without pi as coefficient is "
1079                                     "not yet implemented")
1080         new_start = k_start*S.Pi
1081         new_end = k_end*S.Pi
1082 
1083         if new_start > new_end:
1084             return Union(Interval(S.Zero, new_end, False, theta.right_open),
1085                          Interval(new_start, 2*S.Pi, theta.left_open, True))
1086         else:
1087             return Interval(new_start, new_end, theta.left_open, theta.right_open)
1088 
1089     elif theta.is_FiniteSet:
1090         new_theta = []
1091         for element in theta:
1092             k = _pi_coeff(element)
1093             if k is None:
1094                 raise NotImplementedError('Normalizing theta without pi as '
1095                                           'coefficient, is not Implemented.')
1096             else:
1097                 new_theta.append(k*S.Pi)
1098         return FiniteSet(*new_theta)
1099 
1100     elif theta.is_Union:
1101         return Union(*[normalize_theta_set(interval) for interval in theta.args])
1102 
1103     elif theta.is_subset(S.Reals):
1104         raise NotImplementedError("Normalizing theta when, it is of type %s is not "
1105                                   "implemented" % type(theta))
1106     else:
1107         raise ValueError(" %s is not a real set" % (theta))

1110 class ComplexRegion(Set):
1111     r""" 
1112     Represents the Set of all Complex Numbers. It can represent a 
1113     region of Complex Plane in both the standard forms Polar and 
1114     Rectangular coordinates. 
1115  
1116     * Polar Form 
1117       Input is in the form of the ProductSet or Union of ProductSets 
1118       of the intervals of ``r`` and ``theta``, and use the flag ``polar=True``. 
1119  
1120       .. math:: Z = \{z \in \mathbb{C} \mid z = r\times (\cos(\theta) + I\sin(\theta)), r \in [\texttt{r}], \theta \in [\texttt{theta}]\} 
1121  
1122     * Rectangular Form 
1123       Input is in the form of the ProductSet or Union of ProductSets 
1124       of interval of x and y, the real and imaginary parts of the Complex numbers in a plane. 
1125       Default input type is in rectangular form. 
1126  
1127     .. math:: Z = \{z \in \mathbb{C} \mid z = x + Iy, x \in [\operatorname{re}(z)], y \in [\operatorname{im}(z)]\} 
1128  
1129     Examples 
1130     ======== 
1131  
1132     >>> from sympy import ComplexRegion, Interval, S, I, Union 
1133     >>> a = Interval(2, 3) 
1134     >>> b = Interval(4, 6) 
1135     >>> c1 = ComplexRegion(a*b)  # Rectangular Form 
1136     >>> c1 
1137     CartesianComplexRegion(ProductSet(Interval(2, 3), Interval(4, 6))) 
1138  
1139     * c1 represents the rectangular region in complex plane 
1140       surrounded by the coordinates (2, 4), (3, 4), (3, 6) and 
1141       (2, 6), of the four vertices. 
1142  
1143     >>> c = Interval(1, 8) 
1144     >>> c2 = ComplexRegion(Union(a*b, b*c)) 
1145     >>> c2 
1146     CartesianComplexRegion(Union(ProductSet(Interval(2, 3), Interval(4, 6)), ProductSet(Interval(4, 6), Interval(1, 8)))) 
1147  
1148     * c2 represents the Union of two rectangular regions in complex 
1149       plane. One of them surrounded by the coordinates of c1 and 
1150       other surrounded by the coordinates (4, 1), (6, 1), (6, 8) and 
1151       (4, 8). 
1152  
1153     >>> 2.5 + 4.5*I in c1 
1154     True 
1155     >>> 2.5 + 6.5*I in c1 
1156     False 
1157  
1158     >>> r = Interval(0, 1) 
1159     >>> theta = Interval(0, 2*S.Pi) 
1160     >>> c2 = ComplexRegion(r*theta, polar=True)  # Polar Form 
1161     >>> c2  # unit Disk 
1162     PolarComplexRegion(ProductSet(Interval(0, 1), Interval.Ropen(0, 2*pi))) 
1163  
1164     * c2 represents the region in complex plane inside the 
1165       Unit Disk centered at the origin. 
1166  
1167     >>> 0.5 + 0.5*I in c2 
1168     True 
1169     >>> 1 + 2*I in c2 
1170     False 
1171  
1172     >>> unit_disk = ComplexRegion(Interval(0, 1)*Interval(0, 2*S.Pi), polar=True) 
1173     >>> upper_half_unit_disk = ComplexRegion(Interval(0, 1)*Interval(0, S.Pi), polar=True) 
1174     >>> intersection = unit_disk.intersect(upper_half_unit_disk) 
1175     >>> intersection 
1176     PolarComplexRegion(ProductSet(Interval(0, 1), Interval(0, pi))) 
1177     >>> intersection == upper_half_unit_disk 
1178     True 
1179  
1180     See Also 
1181     ========

1184     PolarComplexRegion 
1185     Complexes

1348     def _contains(self, other):
1349         from sympy.functions import arg, Abs
1350 
1351         isTuple = isinstance(other, Tuple)
1352         if isTuple and len(other) != 2:
1353             raise ValueError('expecting Tuple of length 2')
1354 
1355         # If the other is not an Expression, and neither a Tuple
1356         if not isinstance(other, (Expr, Tuple)):
1357             return S.false
1358 
1359         # self in rectangular form
1360         if not self.polar:
1361             re, im = other if isTuple else other.as_real_imag()
1362             return tfn[fuzzy_or(fuzzy_and([
1363                 pset.args[0]._contains(re),
1364                 pset.args[1]._contains(im)])
1365                 for pset in self.psets)]
1366 
1367         # self in polar form
1368         elif self.polar:
1369             if other.is_zero:
1370                 # ignore undefined complex argument
1371                 return tfn[fuzzy_or(pset.args[0]._contains(S.Zero)
1372                     for pset in self.psets)]
1373             if isTuple:
1374                 r, theta = other
1375             else:
1376                 r, theta = Abs(other), arg(other)
1377             if theta.is_real and theta.is_number:
1378                 # angles in psets are normalized to [0, 2pi)
1379                 theta %= 2*S.Pi
1380                 return tfn[fuzzy_or(fuzzy_and([
1381                     pset.args[0]._contains(r),
1382                     pset.args[1]._contains(theta)])
1383                     for pset in self.psets)]

1386 class CartesianComplexRegion(ComplexRegion):
1387     r""" 
1388     Set representing a square region of the complex plane. 
1389  
1390     .. math:: Z = \{z \in \mathbb{C} \mid z = x + Iy, x \in [\operatorname{re}(z)], y \in [\operatorname{im}(z)]\} 
1391  
1392     Examples 
1393     ======== 
1394  
1395     >>> from sympy import ComplexRegion, I, Interval 
1396     >>> region = ComplexRegion(Interval(1, 3) * Interval(4, 6)) 
1397     >>> 2 + 5*I in region 
1398     True 
1399     >>> 5*I in region 
1400     False 
1401  
1402     See also 
1403     ======== 
1404  
1405     ComplexRegion 
1406     PolarComplexRegion 
1407     Complexes 
1408     """
1409 
1410     polar = False
1411     variables = symbols('x, y', cls=Dummy)
1412 
1413     def __new__(cls, sets):
1414 
1415         if sets == S.Reals*S.Reals:
1416             return S.Complexes
1417 
1418         if all(_a.is_FiniteSet for _a in sets.args) and (len(sets.args) == 2):
1419 
1420             # ** ProductSet of FiniteSets in the Complex Plane. **
1421             # For Cases like ComplexRegion({2, 4}*{3}), It
1422             # would return {2 + 3*I, 4 + 3*I}
1423 
1424             # FIXME: This should probably be handled with something like:
1425             # return ImageSet(Lambda((x, y), x+I*y), sets).rewrite(FiniteSet)
1426             complex_num = []
1427             for x in sets.args[0]:
1428                 for y in sets.args[1]:
1429                     complex_num.append(x + S.ImaginaryUnit*y)
1430             return FiniteSet(*complex_num)
1431         else:
1432             return Set.__new__(cls, sets)
1433 
1434     @property
1435     def expr(self):
1436         x, y = self.variables
1437         return x + S.ImaginaryUnit*y

1440 class PolarComplexRegion(ComplexRegion):
1441     r""" 
1442     Set representing a polar region of the complex plane. 
1443  
1444     .. math:: Z = \{z \in \mathbb{C} \mid z = r\times (\cos(\theta) + I\sin(\theta)), r \in [\texttt{r}], \theta \in [\texttt{theta}]\} 
1445  
1446     Examples 
1447     ======== 
1448  
1449     >>> from sympy import ComplexRegion, Interval, oo, pi, I 
1450     >>> rset = Interval(0, oo) 
1451     >>> thetaset = Interval(0, pi) 
1452     >>> upper_half_plane = ComplexRegion(rset * thetaset, polar=True) 
1453     >>> 1 + I in upper_half_plane 
1454     True 
1455     >>> 1 - I in upper_half_plane 
1456     False 
1457  
1458     See also 
1459     ======== 
1460  
1461     ComplexRegion 
1462     CartesianComplexRegion 
1463     Complexes 
1464  
1465     """
1466 
1467     polar = True
1468     variables = symbols('r, theta', cls=Dummy)
1469 
1470     def __new__(cls, sets):
1471 
1472         new_sets = []
1473         # sets is Union of ProductSets
1474         if not sets.is_ProductSet:
1475             for k in sets.args:
1476                 new_sets.append(k)
1477         # sets is ProductSets
1478         else:
1479             new_sets.append(sets)
1480         # Normalize input theta
1481         for k, v in enumerate(new_sets):
1482             new_sets[k] = ProductSet(v.args[0],
1483                                      normalize_theta_set(v.args[1]))
1484         sets = Union(*new_sets)
1485         return Set.__new__(cls, sets)
1486 
1487     @property
1488     def expr(self):
1489         r, theta = self.variables
1490         return r*(cos(theta) + S.ImaginaryUnit*sin(theta))
