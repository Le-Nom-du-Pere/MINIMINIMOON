38   def test_arrayexpr_convert_matrix_to_array():
39   
40       expr = M*N
41       result = ArrayContraction(ArrayTensorProduct(M, N), (1, 2))
42       assert convert_matrix_to_array(expr) == result
43   
44       expr = M*N*M
45       result = _array_contraction(ArrayTensorProduct(M, N, M), (1, 2), (3, 4))
46       assert convert_matrix_to_array(expr) == result

50

54       expr = 3*M*N
55       res = convert_matrix_to_array(expr)
56       rexpr = convert_array_to_matrix(res)
57       assert expr == rexpr

59       expr = 3*M + N*M.T*M + 4*k*N
60       res = convert_matrix_to_array(expr)
61       rexpr = convert_array_to_matrix(res)
62       assert expr == rexpr
63   
64       expr = Inverse(M)*N
65       rexpr = convert_array_to_matrix(convert_matrix_to_array(expr))
66       assert expr == rexpr
67   
68       expr = M**2
69       rexpr = convert_array_to_matrix(convert_matrix_to_array(expr))
70       assert expr == rexpr
71   
72       expr = M*(2*N + 3*M)
73       res = convert_matrix_to_array(expr)
74       rexpr = convert_array_to_matrix(res)
75       assert expr == rexpr
