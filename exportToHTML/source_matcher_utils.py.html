33   @compatibility(is_backward_compatible=False)
34   @dataclass
35   class SourcePartition:
36       # Nodes in a particular partition
37       nodes: list[Node]
38   
39       # The source these nodes decomposed from
40       source: Any
41   
42       # Nodes in the graph that are needed as inputs to the partition
43       # These do not include the params of the partition
44       input_nodes: list[Node] = field(default_factory=list)
45   
46       # Nodes in the partition that are being used by nodes outside of the
47       # partition
48       output_nodes: list[Node] = field(default_factory=list)
49   
50       # Parameters that are being used
51       params: list[Node] = field(default_factory=list)

54   @compatibility(is_backward_compatible=False)  # type: ignore[misc]
55   def get_source_partitions(
56       graph: Graph,
57       wanted_sources: list[Any],
58       filter_fn: Optional[Callable[[Node], bool]] = None,
59   ) -> dict[Any, list[SourcePartition]]:
60       """ 
61       Args: 
62           graph: The graph we want to partition 
63           wanted_sources: List of sources of nodes that were decomposed from this 
64               source. This can be a function (ex. torch.nn.functional.linear) or a 
65               leaf module type (ex. torch.nn.Linear). 
66    
67       Returns: 
68           Dictionary mapping sources that were given to a list of SourcePartitions 
69           that correspond to the list of nodes that were decomposed from the given 
70           source. 
71       """
72       modules: dict[type, dict[str, list[Node]]] = {}
73   
74       for node in graph.nodes:
75           # The metadata source_fn should contain a tuple of a unique name for the
76           # source, and the source function if the node is decomposed from a
77           # function, or the type of module if the node is decomposed from a leaf
78           # module
79   
80           # TODO: Bypass "torch_fn" when "source_fn_stack" because now "torch_fn" can
81           # be different from "source_fn_stack", for example for the add_ node
82           # decomposed from batch norm. We should remove the check on "source_fn_stack"
83           # after we fix "torch_fn". T199561090
84           if (source_fn_st := node.meta.get("source_fn_stack", None)) is None and (
85               torch_fn := node.meta.get("torch_fn", None)
86           ) is not None:
87               node_fqn, source_fn = torch_fn
88               source_fn_name = source_fn.split(".")[1]
89               if source_fn_name in wanted_sources:
90                   diff_modules = modules.setdefault(source_fn_name, {})
91                   partition = diff_modules.setdefault(node_fqn, [])
92                   partition.append(node)
93   
94           if (source_fn_st := node.meta.get("source_fn_stack", None)) is not None:
95               source_fn = source_fn_st[-1]
96               if source_fn[1] in wanted_sources:
97                   diff_modules = modules.setdefault(source_fn[1], {})
98                   partition = diff_modules.setdefault(source_fn[0], [])
99                   partition.append(node)
100  
101      def make_partition(nodes: list[Node], module_type: type) -> SourcePartition:
102          input_nodes = set()
103          output_nodes = set()
104          params = set()
105          for node in nodes:
106              for arg in node.args:
107                  if isinstance(arg, Node) and arg not in nodes and arg.op != "get_attr":
108                      input_nodes.add(arg)
109  
110              if node.op == "get_attr":
111                  params.add(node)
112                  # get_attr nodes won't be output nodes
113                  continue
114  
115              for user in node.users.keys():
116                  if user not in nodes:
117                      output_nodes.add(node)
118  
119          return SourcePartition(
120              nodes,
121              module_type,
122              list(input_nodes),
123              list(output_nodes),
124              list(params),  # type: ignore[arg-type]
125          )
126  
127      ret: dict[type[Any], list[SourcePartition]] = {}
128  
129      if filter_fn:
130          # for each partition, we apply filter_fn to filter out all partitions that doesn't satisfy the
131          # filter condition
132          filtered_modules = {}
133          for tp, name_to_partition in modules.items():
134              filtered_name_to_partition = {
135                  name: partition
136                  for name, partition in name_to_partition.items()
137                  if all(map(filter_fn, partition))
138              }
139              filtered_modules[tp] = filtered_name_to_partition
140          modules = filtered_modules
141  
142      for k, v in modules.items():
143          ret[k] = [make_partition(partition, k) for partition in v.values()]
144  
145      return ret

148  @compatibility(is_backward_compatible=False)  # type: ignore[misc]
149  def check_subgraphs_connected(
150      subgraph1: SourcePartition, subgraph2: SourcePartition
151  ) -> bool:
152      """ 
153      Given two subgraphs A and B (in the form of a list of nodes), checks if 
154      A has nodes connecting to at least one node in B -- aka there exists a node 
155      in B that uses a node in A (not the other way around). 
156      """
157  
158      for node in reversed(subgraph1.nodes):
159          for user in node.users.keys():
160              if user in subgraph2.nodes:
161                  return True
162      return False
