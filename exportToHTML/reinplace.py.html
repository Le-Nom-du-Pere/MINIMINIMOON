26   def _is_view_op(tgt):
27       if tgt is not None and isinstance(tgt, torch._ops.OpOverload):
28           schema = tgt._schema
29           if len(schema.arguments) > 0:
30               first_arg = schema.arguments[0]
31               # check if op is a view
32               return (
33                   first_arg.alias_info is not None and not first_arg.alias_info.is_write
34               )

37   def _get_view_type(tgt) -> _ViewType:
38       if tgt is not None and isinstance(tgt, torch._ops.OpOverload):
39           schema = tgt._schema
40           if len(schema.arguments) > 0:
41               first_arg = schema.arguments[0]
42               # check if op is a view
43               if first_arg.alias_info is not None and not first_arg.alias_info.is_write:
44                   # check if op is a multi-output view
45                   if "*" in first_arg.alias_info.after_set:
46                       return _ViewType.MultiOutputView
47                   else:
48                       return _ViewType.SingleOutputView
49       return _ViewType.NonView

52   # Stores a bunch of metadata related to functionalization each node.
53   # Relevant metadata:
54   # n.meta['fake_result']: FakeTensor (same type as the output of the node, but with FakeTenors instead of Tensors)
55   #   The fake tensor output from running the current node
56   # n.meta['view_of']: Node
57   #   If the current node n is a view of some base tensor, the 'view_of' field tells us which
58   #   view node was used to generate the current node (a view tensor).
59   #   This information actually makes `fake_result` redundant, but we can use `fake_result`
60   #   to sanity check that our aliasing information is correct.
61   @compatibility(is_backward_compatible=False)
62   class _FunctionalizationMetadataProp(torch.fx.Interpreter):
63       def run_node(self, node: Node):
64           self.node_counter += 1
65           result = super().run_node(node)
66           node.meta["fake_result"] = result
67           node.meta["node_idx"] = self.node_counter
68   
69           # (1) Update metadata with the list of nodes that are used by this node
70           # copy_() doesn't read from its first argument; it writes to it, overwriting previous data.
71           # We don't want to treat it as "being used as an input".
72           node_args = node.args
73           if node.target is torch.ops.aten.copy_.default:
74               node_args = node_args[1:]
75   
76           # (2) Update metadata to track aliasing information about view tensor nodes.
77           if node.op == "call_function":
78               view_type = _get_view_type(node.target)
79               if view_type == _ViewType.SingleOutputView:
80                   assert isinstance(node.args[0], Node)
81                   node.meta["view_of"] = node.args[0]
82               elif view_type == _ViewType.MultiOutputView:
83                   self.multi_output_view_nodes[node] = node.args[0]
84   
85               # Check if we returned a multi-output view,
86               # and we're now grabbing the individual views from the output.
87               #
88               # For multi-output views, we want to map each output view to the base,
89               # but this mapping involves two separate nodes in FX IR.
90               # e.g. "a, b = x_1.split(...)" becomes:
91               #    %split_tensor : [num_users=2] = call_function[target=torch.ops.aten.split.Tensor](args = (%x_1, 2), kwargs = {})
92               #    %getitem : [num_users=1] = call_function[target=operator.getitem](args = (%split_tensor, 0), kwargs = {})
93               #    %getitem_1 : [num_users=1] = call_function[target=operator.getitem](args = (%split_tensor, 1), kwargs = {})
94               # And we'd like to set:
95               #    getitem1.meta['view_of'] = x_1
96               elif node.target is _operator.getitem:
97                   list_arg = node.args[0]
98                   maybe_base_of_view = self.multi_output_view_nodes.get(list_arg, None)
99                   if maybe_base_of_view is not None:
100                      # Note: we could also track indexing info here for multi-output views.
101                      # I don't think this metadata is strictly needed for de-functionalization.
102                      assert isinstance(maybe_base_of_view, Node)
103                      node.meta["view_of"] = maybe_base_of_view
104  
105          if "view_of" in node.meta:
106              # We're linking the current node with its first argument as views.
107              # Assert here that this is actually the case, and their storages are the same.
108              assert isinstance(node.meta["fake_result"], FakeTensor)
109              assert isinstance(node.meta["view_of"].meta["fake_result"], FakeTensor)
110              view_storage = StorageWeakRef(node.meta["fake_result"]._typed_storage())
111              base_storage = StorageWeakRef(
112                  node.meta["view_of"].meta["fake_result"]._typed_storage()
113              )
114              assert view_storage == base_storage
115          return result
116  
117      def propagate(self, *args):
118          self.multi_output_view_nodes = {}
119          self.node_counter = -1
120  
121          with FakeTensorMode() as mode:
122              fake_args = [
123                  mode.from_tensor(a) if isinstance(a, torch.Tensor) else a for a in args
124              ]
125              return super().run(*fake_args)

128  def _schemas_match(functional_schema, inplace_schema):
129      names_match = (
130          inplace_schema.name.endswith("_")
131          and inplace_schema.name[:-1] == functional_schema.name
132      )
133      arg_types_match = len(functional_schema.arguments) == len(
134          inplace_schema.arguments
135      ) and all(
136          a1.type == a2.type
137          for a1, a2 in zip(functional_schema.arguments, inplace_schema.arguments)
138      )
139      # for the inplace op, its first argument should be mutable
140      assert (
141          inplace_schema.arguments[0].alias_info is not None
142          and inplace_schema.arguments[0].alias_info.is_write
143      )
144      # and its remaining arguments shouldn't be.
145      assert all(a.alias_info is None for a in inplace_schema.arguments[1:])
146      return names_match and arg_types_match

153  def _maybe_get_inplace_op(op):
154      # __module__ seems broken; it returns torch._ops.aten which doesn't exist
155      if not isinstance(op, torch._ops.OpOverload):
156          return None
157      # Some view ops have inplace variants (as_strided_, etc),
158      # but we do NOT want the reinplacing pass to directly add these into the program.
159      # (they'll require extra special handling, aren't aren't really useful for perf anyway)
160      if _is_view_op(op):
161          return None
162      op_namespace = op.__module__.split(".")[-1]
163      op_base_name = op.overloadpacket.__name__
164      maybe_namespace_module = getattr(torch.ops, op_namespace)
165      maybe_inplace_op = (
166          None
167          if maybe_namespace_module is None
168          else getattr(maybe_namespace_module, f"{op_base_name}_", None)
169      )
170      if maybe_inplace_op is None:
171          return None
172  
173      inplace_overloads = [
174          getattr(maybe_inplace_op, overload_name)
175          for overload_name in maybe_inplace_op.overloads()
176      ]
177      inplace_overloads_with_matching_schemas = [
178          f for f in inplace_overloads if _schemas_match(op._schema, f._schema)
179      ]
180      # Just because foo() and foo_() are both existing operators,
181      # They aren't guaranteed to have compatible schemas.
182      # For example, pow.Scalar(Scalar self, Tensor exponent) has no valid inplace variant,
183      # Even though several overloads of pow_ exist.
184      if len(inplace_overloads_with_matching_schemas) == 0:
185          return None
186      assert len(inplace_overloads_with_matching_schemas) == 1
187      inplace_op = inplace_overloads_with_matching_schemas[0]
188      return inplace_op

199  # This function, given a set of set of (aliased) tensor nodes,
200  # Returns any nodes in the graph that *use* any of the aliases, that occur *after* op_index
201  # in the node ordering.
202  def _get_all_later_node_usages(tensor_aliases: set[Node], op_index: int):
203      def _add_if_tensor(x, set_):
204          if isinstance(x, FakeTensor):
205              set_.add(StorageWeakRef(x._typed_storage()))
206  
207      nodes_used_after = set()
208      for t in tensor_aliases:
209          # get all nodes that use the current alias
210          usage_nodes = t.users
211          for n in usage_nodes:
212              # We only care about usages after the current node
213              if "node_idx" not in n.meta or n.meta["node_idx"] <= op_index:
214                  continue
215              # We also don't care about intermediate view ops.
216              # They only matter if their output is then used elsewhere
217              # (either in an out-of-place op, or as an output to the function).
218              if n in tensor_aliases:
219                  if (
220                      isinstance(n.target, torch._ops.OpOverload)
221                      or n.target == _operator.getitem
222                  ):
223                      continue
224              nodes_used_after.add(n)
225      return nodes_used_after

228  # Given an op that we're trying to re-inplace, "b = foo(a)",
229  # And given a {view}_scatter op that shows up later in the graph, "y = {view}_scatter(base, x, args...)"
230  # Then re-inplacing `foo()` would allow us to remove the `{view}_scatter` op entirely, IF:
231  # If there are any aliases in the alias_set(a) that satisfy:
232  # (1) The base of "alias", "alias_base", has the same size/stride/offset metadata as "base"
233  # (2) The output of running {view}(alias, args...) gives you the same size/stride/offset metadata
234  #     as "alias"
235  def _get_view_inverse_node_usages(
236      later_node_usages: set[Node], self_aliases: set[Node]
237  ) -> set[Node]:
238      def matching_view_metadata(a, b):
239          return (
240              a.size() == b.size()
241              and a.stride() == b.stride()
242              and a.storage_offset() == b.storage_offset()
243          )
244  
245      view_inverse_nodes = set()
246      # Go through them in node order, so we can see chains of view_scatter ops.
247      for n in sorted(later_node_usages, key=lambda x: x.meta["node_idx"]):
248          if n.target not in _VIEW_INVERSE_MAP:
249              continue
250          base = n.args[0]
251          mutated_view = n.args[1]
252          assert isinstance(base, Node)
253          assert isinstance(base.meta["fake_result"], FakeTensor)
254          assert isinstance(mutated_view, Node)
255          assert isinstance(mutated_view.meta["fake_result"], FakeTensor)
256          assert not isinstance(n.target, str)
257          # Check that this view_inverse op actually corresponds to taking doing the inverse
258          # of one of our existing self_alias nodes.
259          original_view = _VIEW_INVERSE_MAP[n.target]
260          for self_alias in self_aliases:
261              # We're looking for some alias of the self arg, "alias",
262              # that was created from some op `alias = foo(base, args...)`
263              # such that the current _scatter op "inverts" that foo call.
264              # We can check that by running the original op again, and checking that the strides match.
265              if "view_of" not in self_alias.meta:
266                  continue
267              self_alias_base = self_alias.meta["view_of"]
268              try:
269                  # The we're trying to re-use the args from the view_scatter call inside of the corresponding
270                  # view op, which might throw. This just indicates that view_scatter op isn't a valid inverse
271                  # of the current alias we're looking at.
272                  view_replay_metadata = original_view(
273                      self_alias_base.meta["fake_result"], *n.args[2:], **n.kwargs
274                  )
275                  expected_metadata = self_alias.meta["fake_result"]
276                  # If the alias and its base both have matching metadata, then this view_scatter op is valid to re-inplace.
277                  if matching_view_metadata(
278                      self_alias_base.meta["fake_result"], base.meta["fake_result"]
279                  ) and matching_view_metadata(view_replay_metadata, expected_metadata):
280                      view_inverse_nodes.add(n)
281              except Exception:
282                  continue
283  
284      return view_inverse_nodes

287  @compatibility(is_backward_compatible=True)
288  def reinplace(gm, *sample_args):
289      """ 
290      Given an fx.GraphModule, modifies it to perform "reinplacing", 
291      mutating the nodes of the graph. 
292      We look for out-of-place op call sites like `b = a.add(...)`, 
293      and convert them to be inplace (`b = a.add_(...)`), 
294      as long as the input to the current operator ("a") isn't re-used 
295      anywhere later in the graph. 
296   
297      This pass currently expects to operate on a **functional, ATen** graph. 
298      This can be obtained by running `make_fx(functionalize(f))`. 
299   
300      Sample inputs are needed to determine aliasing relationships of the inputs. 
301      In general, we can't reinplace node `b = a.add(...)` if "a" aliases any of the 
302      inputs to the program. 
303   
304      Given a node "b = foo(a, args...) the algorithm for re-inplacing is as follows: 
305   
306      (1) Perform some initial checks on the metadata of "a" and "args..." 
307          that can disqualify them from being reinplaced. 
308   
309        (1a) Check that the self argument we're attempting to reinplace 
310             has acceptable dtype/size metadata to reinplace with. 
311   
312             For example, if we have: 
313               a = torch.ones(1) 
314               b = torch.ones(10) 
315               out = torch.add(a, b) 
316             We can't turn that into 
317               a.add_(b) 
318             Because that would require resizing "a". 
319   
320             Similarly, we can't convert torch.ge(a, b) into a.ge_(b), 
321             because that would require changing a's dtype (from e.g. float32 to bool). 
322             Note that in this specific example, we could technically do better.. 
323   
324             If we see the pattern: 
325               a_1 = a.ge(b) 
326               a_2 = aten._to_copy(a_1, a.dtype) 
327             Then we this should be valid to completely re-inplace 
328             (this is exactly what functionalization will emit when it sees a.ge_(b)). 
329   
330             This optimization is only really important for user programs 
331             that directly use inplace comparison ops though. 
332   
333             We also cannot re-inplace on tensors that have overlapping memory, 
334             e.g. torch.ones(1).expand(4, 4).add_(1) 
335   
336        (1b) Check if "a" is an alias of any of the program inputs. 
337   
338            If it is, skip and move to the next node. 
339            Inplace'ing an op that would cause it to mutate a program is not sound, 
340            because that would be a side effect visible to the user. 
341   
342            NOTE: there's a future optimization that we should make: 
343            if "a" is a (alias of a)  program input, but later in the program 
344            there is a node that looks like "a.copy_(...)", 
345            Then re-inplacing is ok to do - we are temporarily re-using a's buffer, 
346            which will later be overwritten by the copy_() call. 
347   
348            This will be an important optimization to have for programs that mutate 
349            their inputs. It currently isn't implemented though. 
350   
351        (1c) Check if "a" and "args..." alias 
352   
353            For example, re-inplacing to create code like the below 
354            isn't guaranteed to be sound: 
355   
356              aten.mul_(a, a) 
357   
358      (2) Check that "a" and all of its outstanding aliases are not used anywhere 
359          later in the graph. If this is the case, then it's safe to re-inplace 
360          to "b = foo_(a)". 
361   
362          There are a few caveats to this, explained in more detail below: 
363          (a) If "a" is used later as an argument to a view op, that is okay. 
364              It's only a problem if "a" (or that view) is later passed 
365              into a normal operator, or if it is returned as the program output. 
366          (b) If "a" is a repeat argument in `foo()`, then don't reinplace. 
367              Most ATen kernels don't make any guarantees that this is sound, 
368              e.g. if you do aten.mul_(a, a). 
369              So we'll just ban re-inplacing in this case. 
370              It's only a problem if "a" (or that view) is later passed 
371          (c) If "a" is used as an input into a view "inverse" / "scatter" 
372              operator, it is potentially fine to re-inplace 
373              (and remove that scatter operator from the graph). 
374              See below for a more detailed example. 
375   
376          NOTE: there is an optimization in this step that is crucial 
377          to fully recovering performance from functionalization. 
378   
379          Given this program: 
380          def f(x): 
381              a = torch.ops.aten.add(x, x) 
382              b = torch.ops.aten.diagonal(a) 
383              torch.ops.aten.fill_(b, 0) 
384              return d 
385   
386          Functionalization will emit the following: 
387          def f(x): 
388              a = torch.ops.aten.add(x, x) 
389              b = torch.ops.aten.diagonal(a, 0, 1) 
390              b_updated = torch.ops.aten.fill(b, 0) 
391              a_updated = torch.ops.aten.diagonal_scatter(a, b_updated, 0, 1) 
392              return a_updated 
393   
394          Ordinarily, we would not be able to reinplace the fill, 
395          because "b" aliases with "a" which is used by the diagonal_scatter call. 
396   
397          "re-inplacing" is on the hook for figuring out that it is ok to 
398          completely, the expensive diagonal_scatter call, if we re-inplace the add(). 
399   
400          So, for every `alias in alias_set(a)`, instead of checking 
401          that "alias" is not used anywhere later in the graph, 
402          we check that 
403              EITHER: 
404            (a) alias is not used anywhere later in the graph 
405              OR: 
406            (b) alias is used exactly once later on in the graph, 
407                in the following op: 
408   
409                  out = foo_scatter(alias, x, args...) 
410   
411                where the following must hold: 
412                  (i) "foo_scatter" is the "inverse" operator for foo. 
413                      This only applies to "foo" ops that are view operators, 
414                      which view into a subset of the original tensor's memory. 
415                      In practice, there are ~4 operators where this applies: 
416                        diagonal -> diagonal_scatter 
417                        slice -> slice_scatter 
418                        select -> select_scatter 
419                        as_strided -> as_strided_scatter 
420                  (ii) "args..." are the same between the foo() and foo_scatter() calls. 
421   
422      (3) Perform the actual re-inplacing on foo! 
423   
424        (3b) is the common case, but special care is needed for {view}_scatter (3a) 
425   
426        (3a) {view}_scatter ops. 
427   
428          Consider this program: 
429            a = torch.zeros(2, 2) 
430            b = torch.ones(2) 
431            a[0] = b 
432   
433          Post functionalization, that will look like: 
434            a = torch.zeros(2) 
435            b = torch.ones(1) 
436            a_updated = torch.select_scatter(a, b, 0, 0) 
437   
438          In this case though, there is no "functional" op to re-inplace! 
439          Instead, we'd like to directly remove toe select_scatter call. 
440          We already know from (3) that this is valid, 
441          because "a" has no later usages in the graph. 
442   
443          We perform the re-inplacing on the {view}_scatter op like so 
444          Before: 
445            a_updated = torch.select_scatter(a, b, args...) 
446          After: 
447            a_slice = a.select(a, args...) 
448            a_slice.copy_(b) 
449   
450        (3b) Otherwise, replace the functional op with its inplace variant. 
451          Before: 
452            b = foo(a, args...) 
453          After: 
454            a.foo_(args...) 
455   
456      (4) Finally, after converting either: 
457            Before: 
458              b = foo(a) 
459            After: 
460              foo_(a) 
461          or 
462            Before: 
463              b = {slice}_scatter(a, mutated_slice, args...) 
464            After: 
465              slice = {slice}(a, args...) 
466              slice.copy_(mutated_slice) 
467   
468          We now need to find all later nodes that use "b" as an argument 
469          and update them to take in "a" instead. 
470   
471          Note that for the majority of inplace ops, this isn't actually necessary 
472          (because most inplace ops return "self" as their output). 
473          This isn't generally true for all mutable ops though, which is why 
474          we need to actually replace all of the arguments. 
475   
476          We also need to update our metadata of Dict[StorageWeakRef, Set[Node]], 
477          That maps a given tensor storage to the set of all nodes that take in that storage 
478          as an input. 
479          Specifically, re-inplacing `b = foo(a)` causes "a" and "b"'s sets to get fused 
480          together. 
481   
482      (5) Any "view_inverse/scatter" nodes that were identified as "it's ok to ignore them" 
483          during step (3) get manually deleted from the graph. 
484          Their outputs are no longer used, so technically standard DCE would be able 
485          to do this, but we can no longer run FX's DCE pass now that we have mutable 
486          ops in the graph. 
487      """
488      _FunctionalizationMetadataProp(gm).propagate(*sample_args)
489  
490      # Useful debug printing
491      # def _print(x):
492      # if isinstance(x, FakeTensor):
493      # print(f'fake_result: {StorageWeakRef(x._typed_storage()).cdata}')
494  
495      # for n in gm.graph.nodes:
496      # print(n.format_node())
497      # if hasattr(n, 'meta'):
498      # print(f'node_idx: {n.meta["node_idx"]}')
499      # if 'fake_result' in n.meta:
500      # tree_map(_print, n.meta['fake_result'])
501      # if 'view_of' in n.meta:
502      # print(f'view_of: {str(n.meta["view_of"])}')
503      # print()
504  
505      # We need to know which nodes correspond to inputs (or their aliases)
506      # so we know not to re-inplace them.
507      # NOTE: later, we'll need to add an optimization for fully recovering performance
508      # on programs that mutate inputs.
509      input_storages = {
510          StorageWeakRef(node.meta["fake_result"]._typed_storage())
511          for node in gm.graph.nodes
512          if (
513              node.op == "placeholder"
514              and isinstance(node.meta["fake_result"], torch.Tensor)
515          )
516      }
517  
518      # We also need to know for a given node, what are all of its aliasing nodes.
519      storage_to_nodes: dict[StorageWeakRef, set[Node]] = defaultdict(set)
520      for n in gm.graph.nodes:
521          if "fake_result" in n.meta:
522              # Tree-mapping because some ops can return lists of tensors.
523              def _add_to_map(x):
524                  if isinstance(x, FakeTensor):
525                      storage_to_nodes[StorageWeakRef(x._typed_storage())].add(n)
526  
527              pytree.tree_map_(_add_to_map, n.meta["fake_result"])
528  
529      # inplace-ify functional ops, subject to the constraints written below.
530      all_later_view_inverse_nodes_to_delete = set()
531      for node in gm.graph.nodes:
532          if node.op == "call_function":
533              # Today, the re-inplace pass on directly acts on:
534              # - functional ops with an inplace variant
535              # - {view}_scatter ops that can be potentially removed from the graph.
536              # Both of these ops take in tensor first args, so filtering on this condition
537              # makes the later code simpler.
538              # We should revisit this at some point though, particularly when we also want
539              # the reinplacer to be able to handle out= and mutable operators
540              # and tensorlist first args (like `_foreach_` ops).
541              if not isinstance(node.target, torch._ops.OpOverload):
542                  continue
543              if len(node.target._schema.arguments) < 1:
544                  continue
545              if type(node.target._schema.arguments[0].type) != torch.TensorType:
546                  continue
547  
548              # Step 1a: Check that the self argument we're attempting to reinplace
549              # has the same size/stride as the output.
550              # For example, we shouldn't try to reinplace torch.add(scalar_tensor, larger_tensor)
551              # As it would require resizing scalar_tensor.
552              # (We could potentially swizzle this into larger_tensor.add_(scalar_tensor),
553              # this is probably an optimization to revisit later).
554              self_arg = node.args[0]
555              self_flattened = pytree.tree_leaves(self_arg.meta["fake_result"])
556              node_flattened = pytree.tree_leaves(node.meta["fake_result"])
557              self_has_wrong_metadata = False
558              if len(self_flattened) == len(node_flattened):
559                  for self_meta, node_meta in zip(self_flattened, node_flattened):
560                      if self_meta.numel() != node_meta.numel():
561                          self_has_wrong_metadata = True
562                      if self_meta.dtype != node_meta.dtype:
563                          self_has_wrong_metadata = True
564                      # We also cannot re-inplace on tensors that have internal memory overlap.
565                      # e.g. torch.ones(1).expand(4, 4).add_(1)
566                      if torch._debug_has_internal_overlap(self_meta) == 1:
567                          self_has_wrong_metadata = True
568              # Here, we (optimistically) assume that a.resize(b) is valid to re-inplace,
569              # Since users should never really be calling the functional "torch.ops.aten.resize"
570              # op directly in their programs.
571              if self_has_wrong_metadata and node.target != torch.ops.aten.resize.default:
572                  continue
573  
574              # Step 1b: ensure that the op we're trying to re-inplace isn't a program input
575              self_arg_storage = StorageWeakRef(
576                  self_arg.meta["fake_result"]._typed_storage()
577              )
578              if self_arg_storage in input_storages:
579                  # TODO: later, add the optimization for handling `copy_()` calls in the graph.
580                  continue
581              if len([x for x in node.args if x is self_arg]) > 1:
582                  # Step 1c:
583                  # Calling stuff like aten.mul_(a, a) isn't guaranteed to be sound,
584                  # so we prevent re-inplacing in this case.
585                  continue
586  
587              self_arg_storage = StorageWeakRef(
588                  self_arg.meta["fake_result"]._typed_storage()
589              )
590              self_aliases = storage_to_nodes[self_arg_storage]
591  
592              # First, we find all later usages of any of the aliases of self_arg.
593              later_node_usages = _get_all_later_node_usages(
594                  self_aliases, node.meta["node_idx"]
595              )
596              # Then, we check if any of those later usages are actually view_scatter ops
597              # that are safe to fully remove.
598              later_view_inverse_node_usages = _get_view_inverse_node_usages(
599                  later_node_usages, self_aliases
600              )
601  
602              # Step 2: Check to see if the input to the op is re-used later in the graph.
603              # If not (same goes for its aliases), then this op is safe to re-in place.
604              # This is a slightly roundabout way to check that there are no later usages of the current self argument.
605              # (later_view_inverse_node_usages corresponds to "view_scatter" nodes that we are allowed to delete)
606              can_reinplace = len(later_node_usages - later_view_inverse_node_usages) == 0
607              if not can_reinplace:
608                  continue
609  
610              # Step 3a: Special handling for when we see *_scatter operators.
611              # When we see an operator like `b = torch.slice_scatter(a, ...)`,
612              # instead of trying to "inplace" it into a.slice_scatter_(..._),
613              # we would prefer to remove it from the graph entirely,
614              # and instead copy_() the slice directly into the larger tensor.
615              # See the description of the algorithm for a full example.
616              if (
617                  node.target in _VIEW_INVERSE_MAP
618                  and node not in all_later_view_inverse_nodes_to_delete
619              ):
620                  view_op = _VIEW_INVERSE_MAP[node.target]
621                  # Before:
622                  #   base_updated = torch.ops.aten.slice_scatter.default(base, mutated_slice, args...)
623                  # After:
624                  #   slice = torch.ops.aten.slice.default(base, args...)
625                  #   slice.copy_(mutated_slice)
626                  with gm.graph.inserting_before(node):
627                      mutated_slice_node = node.args[1]
628                      remaining_slice_args = node.args[2:]
629                      slice_node = gm.graph.create_node(
630                          "call_function",
631                          view_op,
632                          (self_arg,) + tuple(remaining_slice_args),
633                          node.kwargs,
634                      )
635                      gm.graph.create_node(
636                          "call_function",
637                          torch.ops.aten.copy_.default,
638                          (
639                              slice_node,
640                              mutated_slice_node,
641                          ),
642                          {},
643                      )
644                  # Add the slice_scatter node to our "nodes to delete" list.
645                  all_later_view_inverse_nodes_to_delete.add(node)
646  
647              else:
648                  # Step 3b: Check to see if this operator has an inplace variant.
649                  maybe_inplace_op = _maybe_get_inplace_op(node.target)
650                  if maybe_inplace_op is None:
651                      continue
652                  # And if so, replace it with its inplace variant.
653                  node.target = maybe_inplace_op
654  
655              # At this point, 'storage_to_nodes' will be stale.
656              # Now that we're inplacing `b = foo(a)`, we need to effectively
657              # union together the dict values for b and a's storage.
658              # Hmm... morally I think we also want to keep the `fake_result` metadata
659              # up to date here, but I'm not sure how easy it is to do.
660              # Maybe it's fine to wait until the end of the pass to update it.
661              curr_node_storage = StorageWeakRef(
662                  node.meta["fake_result"]._typed_storage()
663              )
664              storage_to_nodes[self_arg_storage].update(
665                  storage_to_nodes[curr_node_storage]
666              )
667              storage_to_nodes[curr_node_storage].update(
668                  storage_to_nodes[self_arg_storage]
669              )
670  
671              # Need to remember the view_scatter view nodes we found so we can remove them alter.
672              all_later_view_inverse_nodes_to_delete.update(
673                  later_view_inverse_node_usages
674              )
675  
676              # Step 4:
677              # Now that we've replaced b = a.foo() with a.foo_(),
678              # We need to replace any later usages of "b" with "a"
679              for old in itertools.chain([node], later_view_inverse_node_usages):
680                  new = old.args[0]
681                  nodes_to_update = [
682                      n for n in old.users if n.meta["node_idx"] > node.meta["node_idx"]
683                  ]
684                  for node_to_update in nodes_to_update:
685  
686                      def replace_arg(a):
687                          if a == old:
688                              return new
689                          return a
690  
691                      # First, replace usages of "b" with "a"
692                      node_to_update.args = tree_map_only(
693                          Node, replace_arg, node_to_update.args
694                      )
695                      node_to_update.kwargs = tree_map_only(
696                          Node, replace_arg, node_to_update.kwargs
697                      )
698  
699                      # Second, update our storage_to_nodes data structure.
700                      old_flattened_res = pytree.tree_leaves(old.meta["fake_result"])
701                      node_flattened_res = pytree.tree_leaves(
702                          node_to_update.meta["fake_result"]
703                      )
704  
705                      old_res_storage = {
706                          StorageWeakRef(x._typed_storage())
707                          for x in old_flattened_res
708                          if isinstance(x, FakeTensor)
709                      }
710                      node_res_storage = {
711                          StorageWeakRef(x._typed_storage())
712                          for x in node_flattened_res
713                          if isinstance(x, FakeTensor)
714                      }
715  
716                      # This will happen if we're updating a view op, e.g.
717                      # e.g. replacing
718                      #     x = view(old)
719                      #     x = view(new)
720                      # When that happens, we need to make sure to keep our
721                      # storage mapping up to date.
722                      #
723                      # We're checking for len(...) == 1 here because all view ops are guaranteed to return either a single tensor,
724                      # or multiple tensors that all share the same storage.
725                      # We can't just check equality because we might encounter FX nodes that return zero tensor outputs.
726                      if (
727                          len(old_res_storage) == 1
728                          and len(node_res_storage) == 1
729                          and old_res_storage == node_res_storage
730                      ):
731                          new_flattened_res = pytree.tree_leaves(new.meta["fake_result"])
732                          new_res_storage = {
733                              StorageWeakRef(x._typed_storage())
734                              for x in new_flattened_res
735                              if isinstance(x, FakeTensor)
736                          }
737                          assert len(new_res_storage) == 1
738                          (new_ref,) = new_res_storage
739                          (node_ref,) = node_res_storage
740                          # Technically, "old_ref" and all its aliases will remain
741                          # in our mapping.
742                          # That should be fine though, since we deleted "old"
743                          # from the graph at this point.
744                          storage_to_nodes[node_ref].update(storage_to_nodes[new_ref])
745                          storage_to_nodes[new_ref].update(storage_to_nodes[node_ref])
746  
747      # Step 4: delete any _scatter nodes that we de-functionalized
748      # Need to take care not to delete any of these nodes until after *all* modifications
749      # to the graph are finished.
750      for to_delete in all_later_view_inverse_nodes_to_delete:
751          gm.graph.erase_node(to_delete)
752  
753      gm.recompile()
754      return gm
