1    from sympy import symbols, sin, cos
2    from sympy.physics.mechanics import (dynamicsymbols, ReferenceFrame, Point,
3                KanesMethod)
4    from sympy.testing import pytest
5    from sympy.solvers.solveset import NonlinearError
6    
7    def test_linearity_of_motion_constraints():
8        # Test that an error is raised by KanesMethod if nonlinear velocity
9        # constraints are supplied.
10       # It is a simple pendulum.
11       t = dynamicsymbols._t
12       N, A = ReferenceFrame('N'), ReferenceFrame('A')
13       O, P = Point('O'), Point('P')
14       O.set_vel(N, 0)
15   
16       l = symbols('l')
17       q, x, y, u, ux, uy = dynamicsymbols('q x y u ux uy')
18   
19       A.orient_axis(N, q, N.z)
20       A.set_ang_vel(N, u * N.z)
21       P.set_pos(O, -l * A.y)
22       P.v2pt_theory(O, N, A)
23   
24       kd = [u - q.diff(t), ux - x.diff(t), uy - y.diff(t)]
25       config_constr = [x - l * sin(q), y - l * cos(q)]
26   
27       q_ind = [q]
28       q_dep = [x, y]
29       u_ind = [u]
30       u_dep = [ux, uy]
31   
32       # Make sure an error is raised if nonlinear velocity constraints are
33       # supplied.
34       speed_constr = [ux - l * q.diff(t) * cos(q), sin(uy) +
35           l * q.diff(t) * sin(q)]
36   
37       with pytest.raises(NonlinearError):
38           KanesMethod(N, q_ind=q_ind, q_dependent=q_dep, u_ind=u_ind,
39               u_dependent=u_dep, kd_eqs=kd,
40               configuration_constraints=config_constr,
41               velocity_constraints=speed_constr)
