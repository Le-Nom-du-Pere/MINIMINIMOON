235  struct MPSKernelCache {
236    typedef MPSCachedKernel* (^CreateCachedKernelBlock)();
237  
238    struct CacheEntry {
239      CacheEntry(const std::string& key, MPSCachedKernel* cachedKernel) : cachedKernel_(cachedKernel), key_(key) {}
240      MPSCachedKernel* cachedKernel_ = nullptr;
241      std::string key_;
242    };
243  
244   public:
245    static MPSKernelCache* getInstance() {
246      if (_instance_cache == nullptr) {
247        _instance_cache = new MPSKernelCache();
248      }
249      return _instance_cache;
250    }
251  
252    ~MPSKernelCache() {
253      dispatch_release(serialQueue_);
254      for (const auto& i : cache_) {
255        delete i.second.cachedKernel_;
256      }
257    }
258  
259    // Disallow the copy constructor and operator= functions
260    MPSKernelCache(const MPSKernelCache&) = delete;
261    void operator=(const MPSKernelCache&) = delete;
262  
263    MPSCachedKernel* CreateCachedKernel(const std::string& key, CreateCachedKernelBlock createCacheBlock) {
264      __block MPSCachedKernel* cachedKernel = nil;
265      MPSCacheKey hash = std::hash<std::string>{}(key);
266      dispatch_sync_with_rethrow(serialQueue_, ^() {
267        if (cache_.count(hash) != 0) {
268          auto& entry = cache_.at(hash);
269          TORCH_INTERNAL_ASSERT_DEBUG_ONLY(key == entry.key_, "Key collision in the MPS cached kernel!\n");
270          cachedKernel = entry.cachedKernel_;
271        } else {
272          cachedKernel = createCacheBlock();
273          CacheEntry entry(key, cachedKernel);
274          cache_.emplace(hash, entry);
275        }
276      });
277      return cachedKernel;
278    }
279    template <typename T>
280    inline T* CreateCachedKernelAs(const std::string& key, CreateCachedKernelBlock createCacheBlock) {
281      return static_cast<T*>(CreateCachedKernel(key, createCacheBlock));
282    }
283  
284    MPSCachedKernel* LookUp(const std::string& key) const {
285      __block MPSCachedKernel* cachedKernel = nil;
286  
287      MPSCacheKey hash = std::hash<std::string>{}(key);
288      dispatch_sync_with_rethrow(serialQueue_, ^() {
289        if (cache_.count(hash) != 0) {
290          auto& entry = cache_.at(hash);
291          TORCH_INTERNAL_ASSERT_DEBUG_ONLY(key == entry.key_, "Key collision in the MPS cached kernel!\n");
292          cachedKernel = entry.cachedKernel_;
293        }
294      });
295      return cachedKernel;
296    }
297  
298    template <typename T>
299    inline T* LookUpAs(const std::string& key) const {
300      return static_cast<T*>(LookUp(key));
301    }
302  
303   private:
304    MPSKernelCache() {
305      serialQueue_ = dispatch_queue_create("kernel cache queue", DISPATCH_QUEUE_SERIAL);
306    }
307  
308    static MPSKernelCache* _instance_cache;
309    std::unordered_map<MPSCacheKey, CacheEntry> cache_;
310    dispatch_queue_t serialQueue_ = nullptr;
311  };

314  template <typename T>
315  inline T* LookUpOrCreateCachedKernel(const std::string& key, std::function<MPSKernel*()> instantiate) {
316    auto cache_ = MPSKernelCache::getInstance();
317    if (auto rc = cache_->LookUpAs<T>(key)) {
318      return rc;

460  template <typename encoder_t,
461            typename = std::enable_if_t<std::is_same_v<id<MTLComputeCommandEncoder>, encoder_t> ||
462                                        std::is_same_v<id<MTLArgumentEncoder>, encoder_t>>>
463  static inline void mtl_setBuffer(encoder_t encoder, const TensorBase& t, unsigned idx) {
464    if (C10_UNLIKELY(t.device().type() == kCPU)) {
465      if constexpr (std::is_same_v<id<MTLComputeCommandEncoder>, encoder_t>) {
466        TORCH_CHECK(t.dim() == 0, "Passed CPU tensor to MPS op");
467        // MPS does not support doubles, silently downcast CPU scalar to float
468        if (C10_UNLIKELY(t.scalar_type() == kDouble)) {
469          auto val = static_cast<float>(*reinterpret_cast<const double*>(t.const_data_ptr()));
470          [encoder setBytes:&val length:sizeof(val) atIndex:idx];
471          return;
472        }
473        if (C10_UNLIKELY(t.scalar_type() == kComplexDouble)) {
474          auto val = static_cast<c10::complex<float>>(*reinterpret_cast<const c10::complex<double>*>(t.const_data_ptr()));
475          [encoder setBytes:&val length:sizeof(val) atIndex:idx];
476          return;
477        }
478        [encoder setBytes:t.storage().data() length:t.element_size() atIndex:idx];
479      } else {
480        TORCH_CHECK(false, "Passed CPU tensor to MPS op");
481      }
482      return;
483    }
484    [encoder setBuffer:getMTLBufferStorage(t) offset:t.storage_offset() * t.element_size() atIndex:idx];
485  }

487  // Implementation of setBytes for containers vs trivially copiable types must be separate
488  // Containers like `std::array` could have been uploaded directly, but `c10::ArrayRef`,
489  // while trivially copiable, includes padding  which if copied as Metal shader parameters
490  // might overwrite other values
491  template <
492      typename T,
493      typename = std::enable_if_t<std::is_integral_v<T> || std::is_same_v<T, float> ||
494                                  (std::is_class_v<T> && std::is_trivially_copyable_v<T> && !detail::has_size_type_v<T>)>>
495  static inline void mtl_setBytes(id<MTLComputeCommandEncoder> encoder, const T val, unsigned idx) {
496    [encoder setBytes:&val length:sizeof(T) atIndex:idx];
497  }

500  static inline void mtl_setBytes(id<MTLComputeCommandEncoder> encoder, const Container& values, unsigned idx) {
501    [encoder setBytes:values.data() length:sizeof(typename Container::value_type) * values.size() atIndex:idx];
502  }

508  static size_t iter_tensor_offset(TensorIteratorBase& iter, unsigned idx) {
509    // At the moment, MPS storage data is not the real GPU pointer, but rather a pointer to id<MTLBuffer> object
510    // But TensorIterator constructs data_ptr as if base was just a raw pointer
511    // Workaround this problem by computing an offset from the start of the tensor, which works for both
512    // tensor views and sliced 64-bit iterators
513    return reinterpret_cast<size_t>(iter.data_ptr(idx)) -
514        reinterpret_cast<size_t>(iter.tensor_base(idx).storage().data());
515  }
516  
517  static inline void bind_iter_tensors(id<MTLComputeCommandEncoder> encoder,
518                                       TensorIteratorBase& iter,
519                                       std::optional<size_t> ntensors = std::nullopt) {
520    for (auto idx : c10::irange(ntensors.value_or(iter.ntensors()))) {
521      auto& t = iter.tensor_base(idx);
522      // Handle CPU scalars
523      if (C10_UNLIKELY(t.device().type() == kCPU)) {
524        mtl_setBuffer(encoder, t, idx);
525        continue;
526      }
527      auto offs = iter_tensor_offset(iter, idx);
528      [encoder setBuffer:getMTLBufferStorage(t) offset:offs atIndex:idx];
529    }
530  }

577  static inline void mtl_dispatch1DJob(id<MTLComputeCommandEncoder> encoder,
578                                       id<MTLComputePipelineState> cplState,
579                                       NSUInteger length) {
580    static_assert(sizeof(NSUInteger) == sizeof(uint64_t));
581    const auto maxThreadsPerGroup = [cplState maxTotalThreadsPerThreadgroup];
582    auto size = MTLSizeMake(length, 1, 1);
583    auto threadGroupSize = MTLSizeMake(std::min(maxThreadsPerGroup, length), 1, 1);
584    [encoder dispatchThreads:size threadsPerThreadgroup:threadGroupSize];
585  }

628  inline bool supportedFloatingType(ScalarType dtype) {
629    return dtype == kFloat || dtype == kHalf || dtype == kBFloat16;
630  }

632  inline bool supportedFloatingType(const TensorBase& t) {
633    return supportedFloatingType(t.scalar_type());
634  }

636  inline bool supportedFloatingOrComplexType(ScalarType dtype) {
637    if (dtype == kComplexFloat || dtype == kComplexHalf) {
638      return supportsComplex();
639    }
640    return supportedFloatingType(dtype);
641  }
642  inline bool supportedFloatingOrComplexType(const TensorBase& t) {
643    return supportedFloatingOrComplexType(t.scalar_type());
644  }

648                     "MPS bfloat16 type is supported on MacOS 14.0 or newer.");
649  }

651  inline bool needsGather(const TensorBase& t) {
652    static const bool is_macOS_15_0_or_newer = is_macos_13_or_newer(MacOSVersion::MACOS_VER_15_0_PLUS);
653    return !is_macOS_15_0_or_newer && (!t.is_contiguous() || t.storage_offset());
