37   static inline void cvtbf16_fp32(const __m128i& a, __m256& o) {
38     o = _mm256_castsi256_ps(_mm256_slli_epi32(_mm256_cvtepu16_epi32(a), 16));
39   }

41   static inline void cvtbf16_fp32(const __m256i& a, __m256& o1, __m256& o2) {
42     __m128i lo = _mm256_extractf128_si256(a, 0);
43     __m128i hi = _mm256_extractf128_si256(a, 1);
44     cvtbf16_fp32(lo, o1);
45     cvtbf16_fp32(hi, o2);
46   }

48   static inline __m128i cvtfp32_bf16(const __m256& src) {
49     __m256i value = _mm256_castps_si256(src);
50     __m256i nan = _mm256_set1_epi32(0xffff);
51     __m256i mask = _mm256_castps_si256(_mm256_cmp_ps(src, src, _CMP_ORD_Q));
52     __m256i ones = _mm256_set1_epi32(0x1);
53     __m256i vec_bias = _mm256_set1_epi32(0x7fff);
54     // uint32_t lsb = (input >> 16) & 1;
55     auto t_value = _mm256_and_si256(_mm256_srli_epi32(value, 16), ones);
56     // uint32_t rounding_bias = 0x7fff + lsb;
57     t_value = _mm256_add_epi32(t_value, vec_bias);
58     // input += rounding_bias;
59     t_value = _mm256_add_epi32(t_value, value);
60     // input = input >> 16;
61     t_value = _mm256_srli_epi32(t_value, 16);
62     // Check NaN before converting back to bf16
63     t_value = _mm256_blendv_epi8(nan, t_value, mask);
64     t_value =
65         _mm256_packus_epi32(t_value, t_value); // t[4-7] t[4-7] t[0-4] t[0-4]
66     t_value = _mm256_permute4x64_epi64(t_value, 0xd8); // 11     01     10     00
67     return _mm256_castsi256_si128(t_value);
68   }

70   static inline __m256i cvtfp32_bf16(const __m256& a, const __m256& b) {
71     __m256i lo = _mm256_castps_si256(a);
72     __m256i hi = _mm256_castps_si256(b);
73     __m256i nan = _mm256_set1_epi32(0xffff);
74     __m256i mask_lo = _mm256_castps_si256(_mm256_cmp_ps(a, a, _CMP_ORD_Q));
75     __m256i mask_hi = _mm256_castps_si256(_mm256_cmp_ps(b, b, _CMP_ORD_Q));
76     __m256i ones = _mm256_set1_epi32(0x1);
77     __m256i vec_bias = _mm256_set1_epi32(0x7fff);
78     // uint32_t lsb = (input >> 16) & 1;
79     auto t_lo = _mm256_and_si256(_mm256_srli_epi32(lo, 16), ones);
80     auto t_hi = _mm256_and_si256(_mm256_srli_epi32(hi, 16), ones);
81     // uint32_t rounding_bias = 0x7fff + lsb;
82     t_lo = _mm256_add_epi32(t_lo, vec_bias);
83     t_hi = _mm256_add_epi32(t_hi, vec_bias);
84     // input += rounding_bias;
85     t_lo = _mm256_add_epi32(t_lo, lo);
86     t_hi = _mm256_add_epi32(t_hi, hi);
87     // input = input >> 16;
88     t_lo = _mm256_srli_epi32(t_lo, 16);
89     t_hi = _mm256_srli_epi32(t_hi, 16);
90     // Check NaN before converting back to bf16
91     t_lo = _mm256_blendv_epi8(nan, t_lo, mask_lo);
92     t_hi = _mm256_blendv_epi8(nan, t_hi, mask_hi);
93   
94     t_lo = _mm256_packus_epi32(
95         t_lo, t_hi); // t_hi[4-7] t_lo[4-7] t_hi[0-4] t_lo[0-4]
96     return _mm256_permute4x64_epi64(t_lo, 0xd8); // 11        01        10 00
97   }

99   static inline __m256i merge_compare_result(const __m256& a, const __m256& b) {
100    __m256i lo = _mm256_castps_si256(a);
101    __m256i hi = _mm256_castps_si256(b);
102    lo = _mm256_srli_epi32(lo, 16);
103    hi = _mm256_srli_epi32(hi, 16);
104    auto out = _mm256_packus_epi32(lo, hi);
105    return _mm256_permute4x64_epi64(out, 0xd8);
106  }

120  static inline __m128i cvtfp32_fp16(const __m256& src) {
121    return _mm256_cvtps_ph(src, (_MM_FROUND_TO_NEAREST_INT | _MM_FROUND_NO_EXC));
122  }

124  static inline __m256i cvtfp32_fp16(const __m256& a, const __m256& b) {
125    __m128i lo =
126        _mm256_cvtps_ph(a, (_MM_FROUND_TO_NEAREST_INT | _MM_FROUND_NO_EXC));
127    __m128i hi =
128        _mm256_cvtps_ph(b, (_MM_FROUND_TO_NEAREST_INT | _MM_FROUND_NO_EXC));
129    return _mm256_insertf128_si256(_mm256_castsi128_si256(lo), hi, 1);
130  }

244    int zero_mask() const {
245      // returns an integer mask where all zero elements are translated to 1-bit
246      // and others are translated to 0-bit

250    static Vectorized<T> loadu(const void* ptr, int16_t count = size()) {
251      if (count == size())
252        return _mm256_loadu_si256(reinterpret_cast<const __m256i*>(ptr));
253  
254      __at_align__ int16_t tmp_values[size()];
255  #ifndef __msvc_cl__
256  #pragma unroll
257  #endif
258      for (const auto i : c10::irange(count, size())) {
259        tmp_values[i] = 0;
260      }
261      std::memcpy(tmp_values, ptr, count * sizeof(int16_t));
262      return _mm256_loadu_si256(reinterpret_cast<const __m256i*>(tmp_values));
263    }
264    void store(void* ptr, int count = size()) const {
265      if (count == size()) {
266        _mm256_storeu_si256(reinterpret_cast<__m256i*>(ptr), values);
267      } else if (count > 0) {
268        __at_align__ int16_t tmp_values[size()];
269        _mm256_storeu_si256(reinterpret_cast<__m256i*>(tmp_values), values);
270        std::memcpy(ptr, tmp_values, count * sizeof(int16_t));
271      }
272    }
273    template <int64_t mask>
274    static Vectorized<T> blend(const Vectorized<T>& a, const Vectorized<T>& b) {
275      __at_align__ int16_t tmp_values[size()];
276      a.store(tmp_values);
277      if (mask & 0x01)
278        tmp_values[0] = _mm256_extract_epi16(b.values, 0);
279      if (mask & 0x02)
280        tmp_values[1] = _mm256_extract_epi16(b.values, 1);
281      if (mask & 0x04)
282        tmp_values[2] = _mm256_extract_epi16(b.values, 2);
283      if (mask & 0x08)
284        tmp_values[3] = _mm256_extract_epi16(b.values, 3);
285      if (mask & 0x10)
286        tmp_values[4] = _mm256_extract_epi16(b.values, 4);
287      if (mask & 0x20)
288        tmp_values[5] = _mm256_extract_epi16(b.values, 5);
289      if (mask & 0x40)
290        tmp_values[6] = _mm256_extract_epi16(b.values, 6);
291      if (mask & 0x80)
292        tmp_values[7] = _mm256_extract_epi16(b.values, 7);
293      if (mask & 0x100)
294        tmp_values[8] = _mm256_extract_epi16(b.values, 8);
295      if (mask & 0x200)
296        tmp_values[9] = _mm256_extract_epi16(b.values, 9);
297      if (mask & 0x400)
298        tmp_values[10] = _mm256_extract_epi16(b.values, 10);
299      if (mask & 0x800)
300        tmp_values[11] = _mm256_extract_epi16(b.values, 11);
301      if (mask & 0x1000)
302        tmp_values[12] = _mm256_extract_epi16(b.values, 12);
303      if (mask & 0x2000)
304        tmp_values[13] = _mm256_extract_epi16(b.values, 13);
305      if (mask & 0x4000)
306        tmp_values[14] = _mm256_extract_epi16(b.values, 14);
307      if (mask & 0x8000)
308        tmp_values[15] = _mm256_extract_epi16(b.values, 15);
309      return loadu(tmp_values);
310    }

392    Vectorized<T> isnan() const {
393      __m256 lo, hi;
394      cvt_to_fp32<T>(values, lo, hi);
395      lo = _mm256_cmp_ps(lo, _mm256_set1_ps(0.0f), _CMP_UNORD_Q);

399    Vectorized<T> abs() const {

402    Vectorized<T> angle() const {
403      __m256 lo, hi;
404      cvt_to_fp32<T>(values, lo, hi);
405      auto angle_lambda = [](__m256 values_2) {
406        const auto zero_vec = _mm256_set1_ps(0.f);
407        const auto nan_vec = _mm256_set1_ps(NAN);
408        const auto not_nan_mask = _mm256_cmp_ps(values_2, values_2, _CMP_EQ_OQ);
409        const auto nan_mask = _mm256_cmp_ps(not_nan_mask, zero_vec, _CMP_EQ_OQ);
410        const auto pi = _mm256_set1_ps(c10::pi<float>);
411  
412        const auto neg_mask = _mm256_cmp_ps(values_2, zero_vec, _CMP_LT_OQ);
413        auto angle = _mm256_blendv_ps(zero_vec, pi, neg_mask);
414        angle = _mm256_blendv_ps(angle, nan_vec, nan_mask);
415        return angle;
416      };
417      auto o1 = angle_lambda(lo);
418      auto o2 = angle_lambda(hi);
419      return cvt_from_fp32<T>(o1, o2);
420    }

454    Vectorized<T> copysign(const Vectorized<T>& sign) const {
455      // copy sign bit (0x8000) from sign and remaining bits from values
456      __m256i mask_value = _mm256_set1_epi32(~0x80008000);
457      __m256i mask_signbit = _mm256_set1_epi32(0x80008000);
458      return Vectorized<T>(_mm256_or_si256(
459          _mm256_and_si256(values, mask_value),
460          _mm256_and_si256(sign, mask_signbit)));
461    }

526    Vectorized<T> i0e() const {
527      __m256 lo, hi;

697    template <typename Op, typename VectorizedType>
698    Vectorized<T> inline binary_compare(const VectorizedType& b, Op op) const {
699      __m256 a_lo, a_hi;
700      __m256 b_lo, b_hi;
701      cvt_to_fp32<T>(values, a_lo, a_hi);
702      cvt_to_fp32<T>(b.values, b_lo, b_hi);
703      auto o1 = op(a_lo, b_lo);
704      auto o2 = op(a_hi, b_hi);
705      return cvt_from_fp32<T, /*is_compare_op*/ true>(o1, o2);
706    }

709    Vectorized<T> inline operator>(const Vectorized<T>& other) const {
710      return binary_compare(other, [](__m256 x, __m256 y) {
711        return _mm256_cmp_ps(x, y, _CMP_GT_OQ);
712      });
713    }

738    }

741  template <typename T, typename Op>
742  static inline Vectorized<T> binary_op_as_fp32(
743      const Vectorized<T>& a,
744      const Vectorized<T>& b,
745      Op op) {
746    __m256 a_lo, a_hi;
747    __m256 b_lo, b_hi;
748    cvt_to_fp32<T>(__m256i(a), a_lo, a_hi);
749    cvt_to_fp32<T>(__m256i(b), b_lo, b_hi);
750    auto o1 = op(a_lo, b_lo);
751    auto o2 = op(a_hi, b_hi);
752    return cvt_from_fp32<T>(o1, o2);
753  }
