194  def test_ccode_boolean():
195      assert ccode(True) == "true"
196      assert ccode(S.true) == "true"
197      assert ccode(False) == "false"
198      assert ccode(S.false) == "false"
199      assert ccode(x & y) == "x && y"
200      assert ccode(x | y) == "x || y"
201      assert ccode(~x) == "!x"
202      assert ccode(x & y & z) == "x && y && z"
203      assert ccode(x | y | z) == "x || y || z"
204      assert ccode((x & y) | z) == "z || x && y"
205      assert ccode((x | y) & z) == "z && (x || y)"
206      # Automatic rewrites
207      assert ccode(x ^ y) == '(x || y) && (!x || !y)'
208      assert ccode((x ^ y) ^ z) == '(x || y || z) && (x || !y || !z) && (y || !x || !z) && (z || !x || !y)'
209      assert ccode(Implies(x, y)) == 'y || !x'
210      assert ccode(Equivalent(x, z ^ y, Implies(z, x))) == '(x || (y || !z) && (z || !y)) && (z && !x || (y || z) && (!y || !z))'
