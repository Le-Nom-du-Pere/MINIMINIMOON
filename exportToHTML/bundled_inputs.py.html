15   class InflatableArg(NamedTuple):
16       """Helper type for bundled inputs. 
17    
18       'value' is the compressed/deflated input that is stored in the model. Value 
19       must be of the same type as the argument to the function that it is a deflated 
20       input for. 
21    
22       'fmt' is a formatable code string that is executed to inflate the compressed data into 
23       the appropriate input. It can use 'value' as an input to the format str. It must result 
24       in a value of the same type as 'value'. 
25    
26       'fmt_fn' is a formatable function code string that is executed to inflate the compressed 
27       data into the appropriate input. It must result in a value of the same type as 'value'. 
28       The function name should be the formatable part of the string. 
29    
30       Note: Only top level InflatableArgs can be inflated. i.e. you cannot place 
31       an inflatable arg inside of some other structure. You should instead create 
32       an inflatable arg such that the fmt code string returns the full structure 
33       of your input. 
34       """
35   
36       value: Any
37       fmt: str = "{}"
38       fmt_fn: str = ""

41   def bundle_inputs(
42           model: torch.jit.ScriptModule,
43           inputs: Union[Optional[Sequence[tuple[Any, ...]]], dict[Callable, Optional[Sequence[tuple[Any, ...]]]]],
44           info: Optional[Union[list[str], dict[Callable, list[str]]]] = None,
45           *,
46           _receive_inflate_expr: Optional[list[str]] = None,
47   ) -> torch.jit.ScriptModule:
48       """Create and return a copy of the specified model with inputs attached. 
49    
50       The original model is not mutated or changed in any way. 
51    
52       Models with bundled inputs can be invoked in a uniform manner by 
53       benchmarking and code coverage tools. 
54    
55       If inputs is passed in as a list then the inputs will be bundled for 'forward'. 
56       If inputs is instead passed in as a map then all the methods specified in the map 
57       will have their corresponding inputs bundled. Info should match watchever type is 
58       chosen for the inputs. 
59    
60       The returned model will support the following methods: 
61    
62           `get_all_bundled_inputs_for_<function_name>() -> List[Tuple[Any, ...]]` 
63               Returns a list of tuples suitable for passing to the model like 
64               `for inp in model.get_all_bundled_inputs_for_foo(): model.foo(*inp)` 
65    
66           `get_bundled_inputs_functions_and_info() -> Dict[str, Dict[str: List[str]]]` 
67               Returns a dictionary mapping function names to a metadata dictionary. 
68               This nested dictionary maps preset strings like: 
69                   'get_inputs_function_name' -> the name of a function attribute in this model that can be 
70                       run to get back a list of inputs corresponding to that function. 
71                   'info' -> the user provided extra information about the bundled inputs 
72    
73       If forward has bundled inputs then these following functions will also be defined on the returned module: 
74    
75           `get_all_bundled_inputs() -> List[Tuple[Any, ...]]` 
76               Returns a list of tuples suitable for passing to the model like 
77               `for inp in model.get_all_bundled_inputs(): model(*inp)` 
78    
79           `get_num_bundled_inputs() -> int` 
80               Equivalent to `len(model.get_all_bundled_inputs())`, 
81               but slightly easier to call from C++. 
82    
83       Inputs can be specified in one of two ways: 
84    
85         - The model can define `_generate_bundled_inputs_for_<function_name>`. 
86           If the user chooses this method inputs[<function>] should map to None 
87    
88         - The `inputs` argument to this function can be a dictionary mapping functions to a 
89           list of inputs, of the same form that will be returned by get_all_bundled_inputs_for_<function_name>. 
90           Alternatively if only bundling inputs for forward the map can be omitted and a singular list of inputs 
91           can be provided instead. 
92    
93           The type of the inputs is List[Tuple[Any, ...]]. The outer list corresponds with a 
94           list of inputs, the inner tuple is the list of args that together make up one input. 
95           For inputs of functions that take one arg, this will be a tuple of length one. The Any, ... 
96           is the actual data that makes up the args, e.g. a tensor. 
97    
98       Info is an optional parameter that maps functions to a list of strings providing extra information about that 
99       function's bundled inputs. Alternatively if only bundling inputs for forward the map can be omitted and 
100      a singular list of information can be provided instead. This could be descriptions, expected outputs, etc. 
101          - Ex: info={model.forward : ['man eating icecream', 'an airplane', 'a dog']} 
102   
103      This function will attempt to optimize arguments so that (e.g.) 
104      arguments like `torch.zeros(1000)` will be represented compactly. 
105      Only top-level arguments will be optimized. 
106      Tensors in lists or tuples will not. 
107      """
108      if not isinstance(model, torch.jit.ScriptModule):
109          raise Exception("Only ScriptModule is supported.")  # noqa: TRY002
110  
111      ignored_methods, ignored_attrs = _get_bundled_inputs_attributes_and_methods(model)
112      clone = torch._C._hack_do_not_use_clone_module_with_class(  # type: ignore[attr-defined]
113          model._c,
114          ignored_methods,
115          ignored_attrs,
116      )
117  
118      # The above cloning function returns a torch._C.scriptmodule and we need a torch.jit.scriptmodule.
119      # Fortunately theres a function in _recursive that does exactly that conversion.
120      cloned_module = wrap_cpp_module(clone)
121      if isinstance(inputs, dict):
122          assert isinstance(info, dict) or info is None
123          augment_many_model_functions_with_bundled_inputs(cloned_module, inputs, _receive_inflate_expr, info)
124      else:
125          assert isinstance(info, list) or info is None
126          augment_model_with_bundled_inputs(cloned_module, inputs, _receive_inflate_expr, info)
127      return cloned_module

129  def augment_model_with_bundled_inputs(
130          model: torch.jit.ScriptModule,
131          inputs: Optional[Sequence[tuple[Any, ...]]] = None,
132          _receive_inflate_expr: Optional[list[str]] = None,  # For debugging.
133          info: Optional[list[str]] = None,  # Optional argument to provide info about forward or its inputs
134          skip_size_check=False,
135  ) -> None:
136      """Add bundled sample inputs to a model for the forward function. 
137   
138      Models with bundled inputs can be invoked in a uniform manner by 
139      benchmarking and code coverage tools. 
140   
141      Augmented models will support the following methods: 
142   
143          `get_all_bundled_inputs() -> List[Tuple[Any, ...]]` 
144              Returns a list of tuples suitable for passing to the model like 
145              `for inp in model.get_all_bundled_inputs(): model(*inp)` 
146   
147          `get_num_bundled_inputs() -> int` 
148              Equivalent to `len(model.get_all_bundled_inputs())`, 
149              but slightly easier to call from C++. 
150   
151          `get_bundled_inputs_functions_and_info() -> Dict[str, Dict[str: List[str]]]` 
152              Returns a dictionary mapping function names to a metadata dictionary. 
153              This nested dictionary maps preset strings like: 
154                  'get_inputs_function_name' -> the name of a function attribute in this model that can be 
155                      run to get back a list of inputs corresponding to that function. 
156                  'info' -> the user provided extra information about the bundled inputs 
157   
158      Inputs can be specified in one of two ways: 
159   
160        - The model can define `_generate_bundled_inputs_for_forward`. 
161          If the user chooses this method inputs should be None 
162   
163        - `inputs` is a list of inputs of form List[Tuple[Any, ...]]. A list of tuples where the elements 
164          of each tuple are the args that make up one input. 
165      """
166      if not isinstance(model, torch.jit.ScriptModule):
167          raise Exception("Only ScriptModule is supported.")  # noqa: TRY002
168  
169      forward: Callable = model.forward
170  
171      # Sometimes forward won't have a name attached so just in case
172      if not hasattr(forward, "__name__"):
173          forward.__name__ = 'forward'
174      augment_many_model_functions_with_bundled_inputs(
175          model,
176          inputs={forward : inputs},
177          _receive_inflate_expr=_receive_inflate_expr,
178          info={forward : info} if info else None,
179          skip_size_check=skip_size_check,
180      )

183  def augment_many_model_functions_with_bundled_inputs(
184          model: torch.jit.ScriptModule,
185          inputs: dict[Callable, Optional[Sequence[tuple[Any, ...]]]],
186          _receive_inflate_expr: Optional[list[str]] = None,  # For debugging.
187          info: Optional[dict[Callable, list[str]]] = None,  # Optional argument to provide info about the function or its inputs
188          skip_size_check=False,
189  ) -> None:
190      """Add bundled sample inputs to a model for an arbitrary list of public functions. 
191   
192      Models with bundled inputs can be invoked in a uniform manner by 
193      benchmarking and code coverage tools. 
194   
195      Augmented models will support the following methods: 
196   
197          `get_all_bundled_inputs_for_<function_name>() -> List[Tuple[Any, ...]]` 
198              Returns a list of tuples suitable for passing to the model like 
199              `for inp in model.get_all_bundled_inputs_for_foo(): model.foo(*inp)` 
200   
201          `get_bundled_inputs_functions_and_info() -> Dict[str, Dict[str: List[str]]]` 
202              Returns a dictionary mapping function names to a metadata dictionary. 
203              This nested dictionary maps preset strings like: 
204                  'get_inputs_function_name' -> the name of a function attribute in this model that can be 
205                      run to get back a list of inputs corresponding to that function. 
206                  'info' -> the user provided extra information about the bundled inputs 
207   
208      If forward has bundled inputs then these following functions are also defined: 
209   
210          `get_all_bundled_inputs() -> List[Tuple[Any, ...]]` 
211              Returns a list of tuples suitable for passing to the model like 
212              `for inp in model.get_all_bundled_inputs(): model(*inp)` 
213   
214          `get_num_bundled_inputs() -> int` 
215              Equivalent to `len(model.get_all_bundled_inputs())`, 
216              but slightly easier to call from C++. 
217   
218      Inputs can be specified in one of two ways: 
219   
220        - The model can define `_generate_bundled_inputs_for_<function_name>`. 
221          If the user chooses this method inputs[<function>] should map to None 
222   
223        - The `inputs` argument to this function can be a dictionary mapping functions to a 
224          list of inputs, of the same form that will be returned by get_all_bundled_inputs_for_<function_name>. 
225          The type of the inputs is List[Tuple[Any, ...]]. The outer list corresponds with a 
226          list of inputs, the inner tuple is the list of args that together make up one input. 
227          For inputs of functions that take one arg, this will be a tuple of length one. The Any, ... 
228          is the actual data that makes up the args, e.g. a tensor. 
229   
230      Info is an optional parameter that maps functions to a list of strings providing extra information about that 
231      function's bundled inputs. This could be descriptions, expected outputs, etc. 
232          - Ex: info={model.forward : ['man eating icecream', 'an airplane', 'a dog']} 
233   
234      This function will attempt to optimize arguments so that (e.g.) 
235      arguments like `torch.zeros(1000)` will be represented compactly. 
236      Only top-level arguments will be optimized. 
237      Tensors in lists or tuples will not. 
238      """
239      if not isinstance(model, torch.jit.ScriptModule):
240          raise Exception("Only ScriptModule is supported.")  # noqa: TRY002
241  
242      if not inputs:
243          raise Exception("Please provide inputs for at least 1 function")  # noqa: TRY002
244  
245      if hasattr(model, "get_all_bundled_inputs") or hasattr(model, "get_bundled_inputs_functions_and_info"):
246          raise Exception(  # noqa: TRY002
247              "Models can only be augmented with bundled inputs once. "
248              "This Model seems to have already been augmented with "
249              "bundled inputs. Please start afresh with one that "
250              "doesn't have bundled inputs.",
251          )
252  
253      get_bundled_inputs_functions_and_info_template = ""
254  
255      for function, input_list in inputs.items():
256          if hasattr(function, "__name__"):
257              function_name = function.__name__
258          else:
259              if hasattr(function, "name"):
260                  function_name = function.name  # type: ignore[attr-defined]
261              else:
262                  raise Exception(  # noqa: TRY002
263                      'At least one of your functions has no attribute name please ensure all have one. m.foo.name = "foo"')
264  
265  
266          if input_list is not None and not isinstance(input_list, Sequence):
267              raise TypeError(f"Error inputs for function {function_name} is not a Sequence")
268  
269          function_arg_types = [arg.type for arg in function.schema.arguments[1:]]  # type: ignore[attr-defined]
270          deflated_inputs_type: ListType = ListType(TupleType(function_arg_types))
271          model._c._register_attribute(f"_bundled_inputs_deflated_{function_name}", deflated_inputs_type, [])
272  
273          if hasattr(model, "_generate_bundled_inputs_for_" + function_name):
274              if input_list is not None:
275                  raise Exception(  # noqa: TRY002
276                      f"inputs[{function_name}] is not None, but _generate_bundled_inputs_for_{function_name} is already defined"
277                  )
278              # Model author already defined _generate_bundled_inputs_for_<function_name>.
279          elif input_list is None or len(input_list) == 0:
280              raise Exception(  # noqa: TRY002
281                  f"inputs for {function_name} must be specified if "
282                  f"_generate_bundled_inputs_for_{function_name} is not already defined"
283              )
284          else:
285              # Iterate over the inputs and args in each input.
286              # Accumulate `deflated_inputs` as (possibly) compressed values
287              # and `parts` to be joined into the expression that unpacks them.
288              deflated_inputs = []
289              parts = []
290              for inp_idx, args in enumerate(input_list):
291                  if not isinstance(args, tuple) and not isinstance(args, list):  # type: ignore[arg-type]
292                      raise TypeError(
293                          f"Error bundled input for function {function_name} idx: {inp_idx} is not a Tuple or a List"
294                      )
295                  deflated_args = []
296                  parts.append("(")
297                  for arg_idx, arg in enumerate(args):
298                      inflate_helper_fn_name = _get_inflate_helper_fn_name(arg_idx, inp_idx, function_name)
299                      deflated, inflater, helper_definition = _inflate_expr(
300                          arg,
301                          f"deflated[{inp_idx}][{arg_idx}]",
302                          inflate_helper_fn_name,
303                          skip_size_check=skip_size_check,
304                      )
305                      deflated_args.append(deflated)
306                      parts.append(f"    {inflater},")
307                      if helper_definition:
308                          model.define(textwrap.dedent(helper_definition))
309                  deflated_inputs.append(tuple(deflated_args))
310                  parts.append("),")
311              parts.append("")
312              expr = "\n".join(parts)
313  
314              # Back-channel return this expr for debugging.
315              if _receive_inflate_expr is not None:
316                  _receive_inflate_expr.append(expr)
317              setattr(model, f"_bundled_inputs_deflated_{function_name}", deflated_inputs)
318              definition = textwrap.dedent(""" 
319                  def _generate_bundled_inputs_for_{name}(self): 
320                      deflated = self._bundled_inputs_deflated_{name} 
321                      return [ 
322                  {expr} 
323                      ] 
324                  """).format(expr=expr, name=function_name)
325              model.define(definition)
326  
327          # Define get_all_bundled_inputs_for_<function_name> that caches the generated inputs.
328          model.define(textwrap.dedent(""" 
329              def get_all_bundled_inputs_for_{name}(self): 
330                  all_inputs = self._generate_bundled_inputs_for_{name}() 
331                  assert all_inputs is not None 
332                  return all_inputs 
333              """).format(name=function_name))
334  
335          # Add to the high level helper methods
336          inputs_info = repr(info[function]) if info and function in info else '[]'
337          get_bundled_inputs_functions_and_info_template += f"""
338              temp_dict : Dict[str,List[str]] = {{}}
339              info: List[str] = {inputs_info}
340  
341              temp_dict['info'] = info
342              temp_dict['get_inputs_function_name'] = ['get_all_bundled_inputs_for_{function_name}']
343              all_inputs['{function_name}'] = temp_dict
344              """
345  
346          # To ensure backwards compatibility and a streamlined api for forward these wrappers are provided
347          if function_name == 'forward':
348              model.define(textwrap.dedent(""" 
349                  def get_all_bundled_inputs(self): 
350                      return self.get_all_bundled_inputs_for_forward() 
351                  """))
352              model.define(textwrap.dedent(""" 
353                  def get_num_bundled_inputs(self): 
354                      return len(self.get_all_bundled_inputs_for_forward()) 
355                  """))
356  
357      # Define some high level helper methods that act on all bundled inputs
358      model.define(textwrap.dedent(f"""
359          def get_bundled_inputs_functions_and_info(self):
360              all_inputs : Dict[str, Dict[str,List[str]]] = {{}}
361              {get_bundled_inputs_functions_and_info_template}
362              return all_inputs
363          """))

365  def _inflate_expr(
366      arg: T, ref: str, inflate_helper_fn_name: str, skip_size_check: bool = False
367  ) -> tuple[Union[T, torch.Tensor], str, Optional[str]]:
368      # Allow custom inflation expressions any object.
369      # For example, calling custom image-decoding ops.
370      # Or just use "{}" as the format string to ignore size limits.
371      if isinstance(arg, InflatableArg):
372          if arg.fmt_fn:
373              if arg.fmt not in ["{}", ""]:
374                  raise Exception(  # noqa: TRY002
375                      f"Bundled input argument at position '{ref}' has "
376                      f"both arg.fmt_fn => \n{arg.fmt_fn} "
377                      f"\n and arg.fmt  => {arg.fmt}. "
378                      "Please choose `arg.fmt` if the deflater is straightforward or "
379                      "`arg.fmt_fn` if you need a function."
380                  )
381  
382              helper_definition = arg.fmt_fn.format(inflate_helper_fn_name)
383              expr = f"self.{inflate_helper_fn_name}({ref})"
384  
385              return arg.value, expr, helper_definition
386          else:
387              return arg.value, arg.fmt.format(ref), None
388  
389      if isinstance(arg, torch.Tensor):
390          # Small-storage tensors can just be saved directly.
391          if arg._typed_storage().size() <= MAX_RAW_TENSOR_SIZE or skip_size_check:
392              return arg, ref, None
393          # Small contiguous tensors can be cloned to have small storage.
394          # TODO: Should we do this even for non-contiguous tensors?
395          if arg.is_contiguous() and arg.numel() <= MAX_RAW_TENSOR_SIZE:
396              return arg.clone(), ref, None
397          # Example inputs commonly come from torch.zeros, torch.ones, or torch.full.
398          # These can be represented compactly.
399          for fmt in [torch.contiguous_format, torch.channels_last]:
400              if arg.is_contiguous(memory_format=fmt) and (arg == arg.flatten()[0]).all().item():
401                  return (arg.flatten()[0].clone().expand(*arg.size()),
402                          f"{ref}.contiguous(memory_format={fmt})", None)
403          # Prevent big tensors from being bundled by default.
404          # TODO: Provide more useful diagnostics.
405          raise Exception(  # noqa: TRY002
406              f"Bundled input argument at position '{ref}' is "
407              f"a tensor with storage size {arg._typed_storage().size()}. "
408              f"You probably don't want to bundle this as an input. "
409          )
410      else:
411          return arg, ref, None

413  def _get_bundled_inputs_attributes_and_methods(script_module: torch.jit.ScriptModule) -> tuple[list[str], list[str]]:
414      methods: list[str] = []
415      attributes: list[str] = []
416  
417      # Has bundled inputs for forward
418      if hasattr(script_module, 'get_all_bundled_inputs'):
419          methods.append('get_all_bundled_inputs')
420          methods.append('get_num_bundled_inputs')
421          methods.append('run_on_bundled_input')
422  
423      if hasattr(script_module, 'get_bundled_inputs_functions_and_info'):
424          methods.append('get_bundled_inputs_functions_and_info')
425          all_info = script_module.get_bundled_inputs_functions_and_info()
426          for function_name in all_info:
427              methods.append("get_all_bundled_inputs_for_" + function_name)
428              methods.append("_generate_bundled_inputs_for_" + function_name)
429              attributes.append("_bundled_inputs_deflated_" + function_name)
430  
431              bundled_inputs_fn = getattr(
432                  script_module,
433                  f"get_all_bundled_inputs_for_{function_name}"
434              )
435              num_bundled_inputs: int = len(bundled_inputs_fn())
436  
437              # Check inflate helper functions for each function, argument and bundled input
438              func = getattr(script_module, function_name)
439              for arg_idx in range(len(func.schema.arguments) - 1):
440                  for input_idx in range(num_bundled_inputs):
441                      helper_fn_name = _get_inflate_helper_fn_name(
442                          arg_idx=arg_idx,
443                          input_idx=input_idx,
444                          function_name=function_name
445                      )
446                      # if the arg has an InflatableArg with fmt_fn, add the helper function name
447                      if hasattr(script_module, helper_fn_name):
448                          methods.append(helper_fn_name)
449  
450      return (methods, attributes)

453  def _get_inflate_helper_fn_name(
454      arg_idx: int,
455      input_idx: int,
456      function_name: str,
457  ) -> str:
458      return f"_inflate_helper_for_{function_name}_input_{input_idx}_arg_{arg_idx}"

462  def bundle_randn(*size, dtype=None):
463      """Generate a tensor that will be inflated with torch.randn."""
464      stub = torch.zeros(1, dtype=dtype).expand(*size)
465      return InflatableArg(value=stub, fmt="torch.randn_like({})")

468  def bundle_large_tensor(t):
469      """Wrap a tensor to allow bundling regardless of size."""
470      return InflatableArg(value=t, fmt="{}")
