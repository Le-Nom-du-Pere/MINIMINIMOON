30   // TODO: Move the MPSGuardImpl to inherit from NoOpDeviceGuardImpl
31   // https://github.com/pytorch/pytorch/issues/77170
32   struct TORCH_API MPSGuardImpl final
33       : public c10::impl::DeviceGuardImplInterface {
34     static constexpr c10::DeviceType static_type = c10::DeviceType::MPS;
35   
36     // constructor
37     MPSGuardImpl() {}
38     explicit MPSGuardImpl(c10::DeviceType t) {
39       TORCH_CHECK(
40           t == DeviceType::MPS,
41           "MPSGuardImpl initialized with non-MPS DeviceType: ",
42           t);
43     }
44   
45     // returns the type
46     c10::DeviceType type() const override {
47       return c10::DeviceType::MPS;
48     }
49   
50     Device exchangeDevice(Device d) const override {
51       return Device(c10::DeviceType::MPS, 0);
52     }
53   
54     Device getDevice() const override {
55       return Device(c10::DeviceType::MPS, 0);
56     }
57   
58     std::optional<Device> uncheckedGetDevice() const noexcept {
59       return Device(c10::DeviceType::MPS, 0);
60     }
61   
62     void setDevice(Device d) const override {
63       TORCH_CHECK(d.is_mps(), "Expected a MPS device, but got ", d);
64     }
65   
66     void uncheckedSetDevice(Device d) const noexcept override {
67       // TODO: Currently setting only device 0
68     }
69   
70     Stream getStream(Device d) const override {
71       return Stream(Stream::DEFAULT, Device(c10::DeviceType::MPS, 0));
72     }
73   
74     Stream getNewStream(Device, int priority = 0) const override {
75       (void)priority;
76       return Stream(Stream::DEFAULT, Device(c10::DeviceType::MPS, 0));
77     }
78   
79     Stream getDefaultStream(Device d) const override {
80       return Stream(Stream::DEFAULT, Device(c10::DeviceType::MPS, 0));
81     }
82   
83     // NB: These do NOT set the current device
84     Stream exchangeStream(Stream s) const override {
85       return Stream(Stream::DEFAULT, Device(c10::DeviceType::MPS, 0));
86     }
87     DeviceIndex deviceCount() const noexcept override {
88       if (at::hasMPS()) {
89         // TODO: extend it for multi-device case
90         return 1;
91       } else {
92         return 0;
93       }
94     }
95   
96     // Event-related functions
97     void createEvent(mpsEvent_t* event, const EventFlag flag) const;
98   
99     void destroyEvent(void* event, const DeviceIndex device_index)
100        const noexcept override;
101  
102    void record(
103        void** event,
104        const Stream& stream,
105        const DeviceIndex device_index,
106        const EventFlag flag) const override;
107  
108    void block(void* event, const Stream& stream) const override;
109  
110    bool queryEvent(void* event) const override;
111  
112    void synchronizeEvent(void* event) const override;
113  
114    double elapsedTime(void* event1, void* event2, const DeviceIndex device_index)
115        const override;
116  
117    void synchronizeDevice(const DeviceIndex device_index) const override;
118  };

120  /// A variant of OptionalDeviceGuard that is specialized for MPS.
121  struct OptionalMPSGuard {
122    explicit OptionalMPSGuard() : guard_() {}
123  
124    explicit OptionalMPSGuard(std::optional<Device> device_opt)
125        : guard_(device_opt) {}
126  
127    /// Set the current MPS device to the passed device index, if it is not
128    /// nullopt
129    explicit OptionalMPSGuard(std::optional<DeviceIndex> device_index_opt)
130        : guard_(device_index_opt) {}
131  
132    // Copy is not allowed
133    OptionalMPSGuard(const OptionalMPSGuard&) = delete;
134    OptionalMPSGuard& operator=(const OptionalMPSGuard&) = delete;
135    OptionalMPSGuard(OptionalMPSGuard&& other) = delete;
136    OptionalMPSGuard& operator=(OptionalMPSGuard&& other) = delete;
137  
138    /// Sets the MPS device to the given device, initializing the guard if it
139    /// is not already initialized.  Errors if the given device is not a MPS
140    /// device.
141    void set_device(Device device) {
142      guard_.set_device(device);
143    }
144  
145    /// Sets the MPS device to the given device, initializing the guard if it is
146    /// not already initialized.  Errors if the given device is not a MPS device.
147    void reset_device(Device device) {
148      guard_.reset_device(device);
149    }
150  
151    /// Sets the MPS device to the given device index, initializing the guard if
152    /// it is not already initialized.
153    void set_index(DeviceIndex device_index) {
154      guard_.set_index(device_index);
155    }
156  
157    /// Returns the device that was set immediately prior to initialization of the
158    /// guard, or nullopt if the guard is uninitialized.
159    std::optional<Device> original_device() const {
160      return guard_.original_device();
161    }
162  
163    /// Returns the most recent device that was set using this device guard,
164    /// either from construction, or via set_device, if the guard is initialized,
165    /// or nullopt if the guard is uninitialized.
166    std::optional<Device> current_device() const {
167      return guard_.current_device();
168    }
169  
170    /// Restore the original MPS device, resetting this guard to uninitialized
171    /// state.
172    void reset() {
173      guard_.reset();
174    }
175  
176   private:
177    c10::impl::InlineOptionalDeviceGuard<MPSGuardImpl> guard_;
178  };
