34   def _roots(poly, var):
35       """ like roots, but works on higher-order polynomials. """
36       r = roots(poly, var, multiple=True)
37       n = degree(poly)
38       if len(r) != n:
39           r = [rootof(poly, var, k) for k in range(n)]
40       return r

42   def gbt(tf, sample_per, alpha):
43       r""" 
44       Returns falling coefficients of H(z) from numerator and denominator. 
45    
46       Explanation 
47       =========== 
48    
49       Where H(z) is the corresponding discretized transfer function, 
50       discretized with the generalised bilinear transformation method. 
51       H(z) is obtained from the continuous transfer function H(s) 
52       by substituting $s(z) = \frac{z-1}{T(\alpha z + (1-\alpha))}$ into H(s), where T is the 
53       sample period. 
54       Coefficients are falling, i.e. $H(z) = \frac{az+b}{cz+d}$ is returned 
55       as [a, b], [c, d]. 
56    
57       Examples 
58       ======== 
59    
60       >>> from sympy.physics.control.lti import TransferFunction, gbt 
61       >>> from sympy.abc import s, L, R, T 
62    
63       >>> tf = TransferFunction(1, s*L + R, s) 
64       >>> numZ, denZ = gbt(tf, T, 0.5) 
65       >>> numZ 
66       [T/(2*(L + R*T/2)), T/(2*(L + R*T/2))] 
67       >>> denZ 
68       [1, (-L + R*T/2)/(L + R*T/2)] 
69    
70       >>> numZ, denZ = gbt(tf, T, 0) 
71       >>> numZ 
72       [T/L] 
73       >>> denZ 
74       [1, (-L + R*T)/L] 
75    
76       >>> numZ, denZ = gbt(tf, T, 1) 
77       >>> numZ 
78       [T/(L + R*T), 0] 
79       >>> denZ 
80       [1, -L/(L + R*T)] 
81    
82       >>> numZ, denZ = gbt(tf, T, 0.3) 
83       >>> numZ 
84       [3*T/(10*(L + 3*R*T/10)), 7*T/(10*(L + 3*R*T/10))] 
85       >>> denZ 
86       [1, (-L + 7*R*T/10)/(L + 3*R*T/10)] 
87    
88       References 
89       ========== 
90    
91       .. [1] https://www.polyu.edu.hk/ama/profile/gfzhang/Research/ZCC09_IJC.pdf 
92       """
93       if not tf.is_SISO:
94           raise NotImplementedError("Not implemented for MIMO systems.")
95   
96       T = sample_per  # and sample period T
97       s = tf.var
98       z =  s         # dummy discrete variable z
99   
100      np = tf.num.as_poly(s).all_coeffs()
101      dp = tf.den.as_poly(s).all_coeffs()
102      alpha = Rational(alpha).limit_denominator(1000)
103  
104      # The next line results from multiplying H(z) with z^N/z^N
105      N = max(len(np), len(dp)) - 1
106      num = Add(*[ T**(N-i) * c * (z-1)**i * (alpha * z + 1 - alpha)**(N-i) for c, i in zip(np[::-1], range(len(np))) ])
107      den = Add(*[ T**(N-i) * c * (z-1)**i * (alpha * z + 1 - alpha)**(N-i) for c, i in zip(dp[::-1], range(len(dp))) ])
108  
109      num_coefs = num.as_poly(z).all_coeffs()
110      den_coefs = den.as_poly(z).all_coeffs()
111  
112      para = den_coefs[0]
113      num_coefs = [coef/para for coef in num_coefs]
114      den_coefs = [coef/para for coef in den_coefs]
115  
116      return num_coefs, den_coefs

120      Returns falling coefficients of H(z) from numerator and denominator. 
121   
122      Explanation 
123      =========== 
124

143      >>> denZ 
144      [1, (-L + R*T/2)/(L + R*T/2)]

208  def phase_margin(system):
209      r""" 
210      Returns the phase margin of a continuous time system. 
211      Only applicable to Transfer Functions which can generate valid bode plots. 
212   
213      Raises 
214      ====== 
215   
216      NotImplementedError 
217          When time delay terms are present in the system. 
218   
219      ValueError 
220          When a SISO LTI system is not passed. 
221   
222          When more than one free symbol is present in the system. 
223          The only variable in the transfer function should be 
224          the variable of the Laplace transform. 
225   
226      Examples 
227      ======== 
228   
229      >>> from sympy.physics.control import TransferFunction, phase_margin 
230      >>> from sympy.abc import s 
231   
232      >>> tf = TransferFunction(1, s**3 + 2*s**2 + s, s) 
233      >>> phase_margin(tf) 
234      180*(-pi + atan((-1 + (-2*18**(1/3)/(9 + sqrt(93))**(1/3) + 12**(1/3)*(9 + sqrt(93))**(1/3))**2/36)/(-12**(1/3)*(9 + sqrt(93))**(1/3)/3 + 2*18**(1/3)/(3*(9 + sqrt(93))**(1/3)))))/pi + 180 
235      >>> phase_margin(tf).n() 
236      21.3863897518751 
237   
238      >>> tf1 = TransferFunction(s**3, s**2 + 5*s, s) 
239      >>> phase_margin(tf1) 
240      -180 + 180*(atan(sqrt(2)*(-51/10 - sqrt(101)/10)*sqrt(1 + sqrt(101))/(2*(sqrt(101)/2 + 51/2))) + pi)/pi 
241      >>> phase_margin(tf1).n() 
242      -25.1783920627277 
243   
244      >>> tf2 = TransferFunction(1, s + 1, s) 
245      >>> phase_margin(tf2) 
246      -180 
247   
248      See Also 
249      ======== 
250   
251      gain_margin 
252   
253      References 
254      ========== 
255   
256      .. [1] https://en.wikipedia.org/wiki/Phase_margin 
257   
258      """
259      from sympy.functions import arg
260  
261      if not isinstance(system, SISOLinearTimeInvariant):
262          raise ValueError("Margins are only applicable for SISO LTI systems.")
263  
264      _w = Dummy("w", real=True)
265      repl = I*_w
266      expr = system.to_expr()
267      len_free_symbols = len(expr.free_symbols)
268      if expr.has(exp):
269          raise NotImplementedError("Margins for systems with Time delay terms are not supported.")
270      elif len_free_symbols > 1:
271          raise ValueError("Extra degree of freedom found. Make sure"
272              " that there are no free symbols in the dynamical system other"
273              " than the variable of Laplace transform.")
274  
275      w_expr = expr.subs({system.var: repl})
276  
277      mag = 20*log(Abs(w_expr), 10)
278      mag_sol = list(solveset(mag, _w, Interval(0, oo, left_open=True)))
279  
280      if (len(mag_sol) == 0):
281          pm = S(-180)
282      else:
283          wcp = mag_sol[0]
284          pm = ((arg(w_expr)*S(180)/pi).subs({_w:wcp}) + S(180)) % 360
285  
286      if(pm >= 180):
287          pm = pm - 360
288  
289      return pm

327

329   
330      References

332   
333      https://en.wikipedia.org/wiki/Bode_plot 
334

348              " than the variable of Laplace transform.")

351

364  class LinearTimeInvariant(Basic, EvalfMixin):
365      """A common class for all the Linear Time-Invariant Dynamical Systems."""
366  
367      _clstype: Type
368  
369      # Users should not directly interact with this class.
370      def __new__(cls, *system, **kwargs):
371          if cls is LinearTimeInvariant:
372              raise NotImplementedError('The LTICommon class is not meant to be used directly.')
373          return super(LinearTimeInvariant, cls).__new__(cls, *system, **kwargs)
374  
375      @classmethod
376      def _check_args(cls, args):
377          if not args:
378              raise ValueError("At least 1 argument must be passed.")
379          if not all(isinstance(arg, cls._clstype) for arg in args):
380              raise TypeError(f"All arguments must be of type {cls._clstype}.")
381          var_set = {arg.var for arg in args}
382          if len(var_set) != 1:
383              raise ValueError(filldedent(f"""
384                  All transfer functions should use the same complex variable
385                  of the Laplace transform. {len(var_set)} different
386                  values found."""))

390          """Returns `True` if the passed LTI system is SISO else returns False."""

439  class TransferFunction(SISOLinearTimeInvariant):
440      r""" 
441      A class for representing LTI (Linear, time-invariant) systems that can be strictly described 
442      by ratio of polynomials in the Laplace transform complex variable. The arguments 
443      are ``num``, ``den``, and ``var``, where ``num`` and ``den`` are numerator and 
444      denominator polynomials of the ``TransferFunction`` respectively, and the third argument is 
445      a complex variable of the Laplace transform used by these polynomials of the transfer function. 
446      ``num`` and ``den`` can be either polynomials or numbers, whereas ``var`` 
447      has to be a :py:class:`~.Symbol`. 
448   
449      Explanation 
450      =========== 
451   
452      Generally, a dynamical system representing a physical model can be described in terms of Linear 
453      Ordinary Differential Equations like - 
454   
455              $b_{m}y^{\left(m\right)}+b_{m-1}y^{\left(m-1\right)}+\dots+b_{1}y^{\left(1\right)}+b_{0}y= 
456              a_{n}x^{\left(n\right)}+a_{n-1}x^{\left(n-1\right)}+\dots+a_{1}x^{\left(1\right)}+a_{0}x$ 
457   
458      Here, $x$ is the input signal and $y$ is the output signal and superscript on both is the order of derivative 
459      (not exponent). Derivative is taken with respect to the independent variable, $t$. Also, generally $m$ is greater 
460      than $n$. 
461   
462      It is not feasible to analyse the properties of such systems in their native form therefore, we use 
463      mathematical tools like Laplace transform to get a better perspective. Taking the Laplace transform 
464      of both the sides in the equation (at zero initial conditions), we get - 
465   
466              $\mathcal{L}[b_{m}y^{\left(m\right)}+b_{m-1}y^{\left(m-1\right)}+\dots+b_{1}y^{\left(1\right)}+b_{0}y]= 
467              \mathcal{L}[a_{n}x^{\left(n\right)}+a_{n-1}x^{\left(n-1\right)}+\dots+a_{1}x^{\left(1\right)}+a_{0}x]$ 
468   
469      Using the linearity property of Laplace transform and also considering zero initial conditions 
470      (i.e. $y(0^{-}) = 0$, $y'(0^{-}) = 0$ and so on), the equation 
471      above gets translated to - 
472   
473              $b_{m}\mathcal{L}[y^{\left(m\right)}]+\dots+b_{1}\mathcal{L}[y^{\left(1\right)}]+b_{0}\mathcal{L}[y]= 
474              a_{n}\mathcal{L}[x^{\left(n\right)}]+\dots+a_{1}\mathcal{L}[x^{\left(1\right)}]+a_{0}\mathcal{L}[x]$ 
475   
476      Now, applying Derivative property of Laplace transform, 
477   
478              $b_{m}s^{m}\mathcal{L}[y]+\dots+b_{1}s\mathcal{L}[y]+b_{0}\mathcal{L}[y]= 
479              a_{n}s^{n}\mathcal{L}[x]+\dots+a_{1}s\mathcal{L}[x]+a_{0}\mathcal{L}[x]$ 
480   
481      Here, the superscript on $s$ is **exponent**. Note that the zero initial conditions assumption, mentioned above, is very important 
482      and cannot be ignored otherwise the dynamical system cannot be considered time-independent and the simplified equation above 
483      cannot be reached. 
484   
485      Collecting $\mathcal{L}[y]$ and $\mathcal{L}[x]$ terms from both the sides and taking the ratio 
486      $\frac{ \mathcal{L}\left\{y\right\} }{ \mathcal{L}\left\{x\right\} }$, we get the typical rational form of transfer 
487      function. 
488   
489      The numerator of the transfer function is, therefore, the Laplace transform of the output signal 
490      (The signals are represented as functions of time) and similarly, the denominator 
491      of the transfer function is the Laplace transform of the input signal. It is also a convention 
492      to denote the input and output signal's Laplace transform with capital alphabets like shown below. 
493   
494              $H(s) = \frac{Y(s)}{X(s)} = \frac{ \mathcal{L}\left\{y(t)\right\} }{ \mathcal{L}\left\{x(t)\right\} }$ 
495   
496      $s$, also known as complex frequency, is a complex variable in the Laplace domain. It corresponds to the 
497      equivalent variable $t$, in the time domain. Transfer functions are sometimes also referred to as the Laplace 
498      transform of the system's impulse response. Transfer function, $H$, is represented as a rational 
499      function in $s$ like, 
500   
501              $H(s) =\ \frac{a_{n}s^{n}+a_{n-1}s^{n-1}+\dots+a_{1}s+a_{0}}{b_{m}s^{m}+b_{m-1}s^{m-1}+\dots+b_{1}s+b_{0}}$ 
502   
503      Parameters 
504      ========== 
505   
506      num : Expr, Number 
507          The numerator polynomial of the transfer function. 
508      den : Expr, Number 
509          The denominator polynomial of the transfer function. 
510      var : Symbol 
511          Complex variable of the Laplace transform used by the 
512          polynomials of the transfer function. 
513   
514      Raises 
515      ====== 
516   
517      TypeError 
518          When ``var`` is not a Symbol or when ``num`` or ``den`` is not a 
519          number or a polynomial. 
520      ValueError 
521          When ``den`` is zero. 
522   
523      Examples 
524      ======== 
525   
526      >>> from sympy.abc import s, p, a 
527      >>> from sympy.physics.control.lti import TransferFunction 
528      >>> tf1 = TransferFunction(s + a, s**2 + s + 1, s) 
529      >>> tf1 
530      TransferFunction(a + s, s**2 + s + 1, s) 
531      >>> tf1.num 
532      a + s 
533      >>> tf1.den 
534      s**2 + s + 1 
535      >>> tf1.var 
536      s 
537      >>> tf1.args 
538      (a + s, s**2 + s + 1, s) 
539   
540      Any complex variable can be used for ``var``. 
541   
542      >>> tf2 = TransferFunction(a*p**3 - a*p**2 + s*p, p + a**2, p) 
543      >>> tf2 
544      TransferFunction(a*p**3 - a*p**2 + p*s, a**2 + p, p) 
545      >>> tf3 = TransferFunction((p + 3)*(p - 1), (p - 1)*(p + 5), p) 
546      >>> tf3 
547      TransferFunction((p - 1)*(p + 3), (p - 1)*(p + 5), p) 
548   
549      To negate a transfer function the ``-`` operator can be prepended: 
550   
551      >>> tf4 = TransferFunction(-a + s, p**2 + s, p) 
552      >>> -tf4 
553      TransferFunction(a - s, p**2 + s, p) 
554      >>> tf5 = TransferFunction(s**4 - 2*s**3 + 5*s + 4, s + 4, s) 
555      >>> -tf5 
556      TransferFunction(-s**4 + 2*s**3 - 5*s - 4, s + 4, s) 
557   
558      You can use a float or an integer (or other constants) as numerator and denominator: 
559   
560      >>> tf6 = TransferFunction(1/2, 4, s) 
561      >>> tf6.num 
562      0.500000000000000 
563      >>> tf6.den 
564      4 
565      >>> tf6.var 
566      s 
567      >>> tf6.args 
568      (0.5, 4, s) 
569   
570      You can take the integer power of a transfer function using the ``**`` operator: 
571   
572      >>> tf7 = TransferFunction(s + a, s - a, s) 
573      >>> tf7**3 
574      TransferFunction((a + s)**3, (-a + s)**3, s) 
575      >>> tf7**0 
576      TransferFunction(1, 1, s) 
577      >>> tf8 = TransferFunction(p + 4, p - 3, p) 
578      >>> tf8**-1 
579      TransferFunction(p - 3, p + 4, p) 
580   
581      Addition, subtraction, and multiplication of transfer functions can form 
582      unevaluated ``Series`` or ``Parallel`` objects. 
583   
584      >>> tf9 = TransferFunction(s + 1, s**2 + s + 1, s) 
585      >>> tf10 = TransferFunction(s - p, s + 3, s) 
586      >>> tf11 = TransferFunction(4*s**2 + 2*s - 4, s - 1, s) 
587      >>> tf12 = TransferFunction(1 - s, s**2 + 4, s) 
588      >>> tf9 + tf10 
589      Parallel(TransferFunction(s + 1, s**2 + s + 1, s), TransferFunction(-p + s, s + 3, s)) 
590      >>> tf10 - tf11 
591      Parallel(TransferFunction(-p + s, s + 3, s), TransferFunction(-4*s**2 - 2*s + 4, s - 1, s)) 
592      >>> tf9 * tf10 
593      Series(TransferFunction(s + 1, s**2 + s + 1, s), TransferFunction(-p + s, s + 3, s)) 
594      >>> tf10 - (tf9 + tf12) 
595      Parallel(TransferFunction(-p + s, s + 3, s), TransferFunction(-s - 1, s**2 + s + 1, s), TransferFunction(s - 1, s**2 + 4, s)) 
596      >>> tf10 - (tf9 * tf12)

646

901      def _eval_rewrite_as_StateSpace(self, *args):
902          """ 
903          Returns the equivalent space model of the transfer function model. 
904          The state space model will be returned in the controllable canonical form. 
905   
906          Unlike the space state to transfer function model conversion, the transfer function 
907          to state space model conversion is not unique. There can be multiple state space 
908          representations of a given transfer function model. 
909   
910          Examples 
911          ======== 
912   
913          >>> from sympy.abc import s 
914          >>> from sympy.physics.control import TransferFunction, StateSpace 
915          >>> tf = TransferFunction(s**2 + 1, s**3 + 2*s + 10, s) 
916          >>> tf.rewrite(StateSpace) 
917          StateSpace(Matrix([ 
918          [  0,  1, 0], 
919          [  0,  0, 1], 
920          [-10, -2, 0]]), Matrix([ 
921          [0], 
922          [0], 
923          [1]]), Matrix([[1, 0, 1]]), Matrix([[0]])) 
924   
925          """
926          if not self.is_proper:
927              raise ValueError("Transfer Function must be proper.")
928  
929          num_poly = Poly(self.num, self.var)
930          den_poly = Poly(self.den, self.var)
931          n = den_poly.degree()
932  
933          num_coeffs = num_poly.all_coeffs()
934          den_coeffs = den_poly.all_coeffs()
935          diff = n - num_poly.degree()
936          num_coeffs = [0]*diff + num_coeffs
937  
938          a = den_coeffs[1:]
939          a_mat = Matrix([[(-1)*coefficient/den_coeffs[0] for coefficient in reversed(a)]])
940          vert = zeros(n-1, 1)
941          mat = eye(n-1)
942          A = vert.row_join(mat)
943          A = A.col_join(a_mat)
944  
945          B = zeros(n, 1)
946          B[n-1] = 1
947  
948          i = n
949          C = []
950          while(i > 0):
951              C.append(num_coeffs[i] - den_coeffs[i]*num_coeffs[0])
952              i -= 1
953          C = Matrix([C])
954  
955          D = Matrix([num_coeffs[0]])
956  
957          return StateSpace(A, B, C, D)

2626     r""" 
2627     A class for representing closed-loop feedback interconnection between two 
2628     SISO input/output systems. 
2629  
2630     The first argument, ``sys1``, is the feedforward part of the closed-loop 
2631     system or in simple words, the dynamical model representing the process 
2632     to be controlled. The second argument, ``sys2``, is the feedback system 
2633     and controls the fed back signal to ``sys1``. Both ``sys1`` and ``sys2`` 
2634     can either be ``Series``, ``StateSpace`` or ``TransferFunction`` objects. 
2635  
2636     Parameters 
2637     ========== 
2638  
2639     sys1 : Series, StateSpace, TransferFunction 
2640         The feedforward path system. 
2641     sys2 : Series, StateSpace, TransferFunction, optional 
2642         The feedback path system (often a feedback controller). 
2643         It is the model sitting on the feedback path. 
2644  
2645         If not specified explicitly, the sys2 is 
2646         assumed to be unit (1.0) transfer function. 
2647     sign : int, optional 
2648         The sign of feedback. Can either be ``1`` 
2649         (for positive feedback) or ``-1`` (for negative feedback). 
2650         Default value is `-1`. 
2651  
2652     Raises 
2653     ====== 
2654  
2655     ValueError 
2656         When ``sys1`` and ``sys2`` are not using the 
2657         same complex variable of the Laplace transform. 
2658  
2659         When a combination of ``sys1`` and ``sys2`` yields 
2660         zero denominator. 
2661  
2662     TypeError 
2663         When either ``sys1`` or ``sys2`` is not a ``Series``, ``StateSpace`` or 
2664         ``TransferFunction`` object. 
2665  
2666     Examples 
2667     ======== 
2668  
2669     >>> from sympy import Matrix 
2670     >>> from sympy.abc import s 
2671     >>> from sympy.physics.control.lti import StateSpace, TransferFunction, Feedback

2973             I_outputs = eye(self.num_outputs)

3449         [-------  ----------------] 
3450         [6*s - 1         2        ] 
3451         [             6*s  - s    ]{t} 
3452  
3453         """
3454         if self.is_StateSpace_object:
3455             sys1_ss = self.sys1.doit().rewrite(StateSpace)
3456             sys2_ss = self.sys2.doit().rewrite(StateSpace)
3457             A1, B1, C1, D1 = sys1_ss.A, sys1_ss.B, sys1_ss.C, sys1_ss.D
3458             A2, B2, C2, D2 = sys2_ss.A, sys2_ss.B, sys2_ss.C, sys2_ss.D
3459 
3460             # Create identity matrices
3461             I_inputs = eye(self.num_inputs)
3462             I_outputs = eye(self.num_outputs)
3463 
3464             # Compute F and its inverse
3465             F = I_inputs - self.sign * D2 * D1
3466             E = F.inv()
3467 
3468             # Compute intermediate matrices
3469             E_D2 = E * D2
3470             E_C2 = E * C2
3471             T1 = I_outputs + self.sign * D1 * E_D2
3472             T2 = I_inputs + self.sign * E_D2 * D1
3473             A = Matrix.vstack(
3474             Matrix.hstack(A1 + self.sign * B1 * E_D2 * C1, self.sign * B1 * E_C2),
3475             Matrix.hstack(B2 * T1 * C1, A2 + self.sign * B2 * D1 * E_C2)
3476             )
3477             B = Matrix.vstack(B1 * T2, B2 * D1 * T2)
3478             C = Matrix.hstack(T1 * C1, self.sign * D1 * E_C2)
3479             D = D1 * T2
3480             return StateSpace(A, B, C, D)
3481 
3482         _mat = self.sensitivity * self.sys1.doit()._expr_mat
3483 
3484         _resultant_tfm = _to_TFM(_mat, self.var)
3485 
3486         if cancel:
3487             _resultant_tfm = _resultant_tfm.simplify()
3488 
3489         if expand:
3490             _resultant_tfm = _resultant_tfm.expand()
3491 
3492         return _resultant_tfm
3493 
3494     def _eval_rewrite_as_TransferFunctionMatrix(self, sys1, sys2, sign, **kwargs):
3495         return self.doit()

3498         return MIMOFeedback(-self.sys1, -self.sys2, self.sign)

4187  
4188     Represents the standard state-space model with A, B, C, D as state-space matrices. 
4189     This makes the linear control system: 
4190  
4191         (1) x'(t) = A * x(t) + B * u(t);    x in R^n , u in R^k 
4192         (2) y(t)  = C * x(t) + D * u(t);    y in R^m 
4193  
4194     where u(t) is any input signal, y(t) the corresponding output, and x(t) the system's state. 
4195  
4196     Parameters 
4197     ========== 
4198  
4199     A : Matrix 
4200         The State matrix of the state space model. 
4201     B : Matrix 
4202         The Input-to-State matrix of the state space model.

4204         The State-to-Output matrix of the state space model. 
4205     D : Matrix

4209     ========

4212     >>> from sympy.physics.control import StateSpace

4215

4468     def dsolve(self, initial_conditions=None, input_vector=None, var=Symbol('t')):
4469         r""" 
4470         Returns `y(t)` or output of StateSpace given by the solution of equations: 
4471             x'(t) = A * x(t) + B * u(t) 
4472             y(t)  = C * x(t) + D * u(t) 
4473  
4474         Parameters 
4475         ============ 
4476  
4477         initial_conditions : Matrix 
4478             The initial conditions of `x` state vector. If not provided, it defaults to a zero vector. 
4479         input_vector : Matrix 
4480             The input vector for state space. If not provided, it defaults to a zero vector. 
4481         var : Symbol 
4482             The symbol representing time. If not provided, it defaults to `t`. 
4483  
4484         Examples 
4485         ========== 
4486  
4487         >>> from sympy import Matrix 
4488         >>> from sympy.physics.control import StateSpace 
4489         >>> A = Matrix([[-2, 0], [1, -1]]) 
4490         >>> B = Matrix([[1], [0]]) 
4491         >>> C = Matrix([[2, 1]]) 
4492         >>> ip = Matrix([5]) 
4493         >>> i = Matrix([0, 0]) 
4494         >>> ss = StateSpace(A, B, C) 
4495         >>> ss.dsolve(input_vector=ip, initial_conditions=i).simplify() 
4496         Matrix([[15/2 - 5*exp(-t) - 5*exp(-2*t)/2]]) 
4497  
4498         If no input is provided it defaults to solving the system with zero initial conditions and zero input. 
4499  
4500         >>> ss.dsolve() 
4501         Matrix([[0]]) 
4502  
4503         References 
4504         ========== 
4505         .. [1] https://web.mit.edu/2.14/www/Handouts/StateSpaceResponse.pdf 
4506         .. [2] https://docs.sympy.org/latest/modules/solvers/ode.html#sympy.solvers.ode.systems.linodesolve 
4507  
4508         """
4509 
4510         if not isinstance(var, Symbol):
4511             raise ValueError("Variable for representing time must be a Symbol.")
4512         if not initial_conditions:
4513             initial_conditions = zeros(self._A.shape[0], 1)
4514         elif initial_conditions.shape != (self._A.shape[0], 1):
4515             raise ShapeError("Initial condition vector should have the same number of "
4516                              "rows as the state matrix.")
4517         if not input_vector:
4518             input_vector = zeros(self._B.shape[1], 1)
4519         elif input_vector.shape != (self._B.shape[1], 1):
4520             raise ShapeError("Input vector should have the same number of "
4521                              "columns as the input matrix.")
4522         sol = linodesolve(A=self._A, t=var, b=self._B*input_vector, type='type2', doit=True)
4523         mat1 = Matrix(sol)
4524         mat2 = mat1.replace(var, 0)
4525         free1 = self._A.free_symbols | self._B.free_symbols | input_vector.free_symbols
4526         free2 = mat2.free_symbols
4527         # Get all the free symbols form the matrix
4528         dummy_symbols = list(free2-free1)
4529         # Convert the matrix to a Coefficient matrix
4530         r1, r2 = linear_eq_to_matrix(mat2, dummy_symbols)
4531         s = linsolve((r1, initial_conditions+r2))
4532         res_tuple = next(iter(s))
4533         for ind, v in enumerate(res_tuple):
4534             mat1 = mat1.replace(dummy_symbols[ind], v)
4535         res = self._C*mat1 + self._D*input_vector
4536         return res
