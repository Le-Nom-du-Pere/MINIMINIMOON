22   static const size_t kMaxSmallAlloc = MB(1); // largest "small" allocation is 1 MiB
23   static const size_t kMinLargeAlloc = MB(10); // allocations between 1 and 10 MiB may use kLargeHeap
24   static const size_t kRoundLarge = MB(2); // round up large allocations to 2 MiB
25   static const size_t kSmallHeap = MB(8); // "small" allocations are packed in 8 MiB heaps
26   static const size_t kLargeHeap = MB(32); // "large" allocations may be packed in 32 MiB heaps
27   static const size_t kXLargeHeapD =
28       MB(128); // "extra large" allocations on Discrete devices may be packed in 128 MiB heaps
29   static const size_t kXLargeHeapU =
30       MB(1024); // "extra large" allocations on Unified devices may be packed in 1 GiB heaps
31   static const size_t kMaxScalarAlloc = (sizeof(int64_t)); // largest "scalar" allocation
32   
33   // buffer pools could be customized with a combination of usage flags
34   enum UsageFlags : uint32_t {
35     PRIVATE = 0,
36     SMALL = (1 << 0), // small heaps have sizes of kSmallHeap, and large ones kLargeHeap
37     SHARED = (1 << 1), // shared pools allocated on devices with unified memory; otherwise, private between host/device
38     MANAGED = (1 << 2), // managed storage mode
39     HAZARD = (1 << 3), // enables Automatic Hazard Tracking for the resources allocated on the pool
40     SCALAR = (1 << 4), // used to import CPU scalar values to GPU and use them in MPS Stream
41   };
42   // debug verbosity flags
43   enum DebugVerbosity : uint32_t {
44     SILENT = 0,
45     PROFILING = (1 << 0), // print generic profiling data for total system memory usage
46     ALLOCATIONS = (1 << 1), // print buffer allocations
47     RECYCLES = (1 << 2), // print buffer recycling
48     RELEASES = (1 << 3), // print buffer releases
49     LARGE_ONLY = (1 << 4), // only log large buffer pool transactions
50   };

54   struct BufferBlock {
55     id<MTLBuffer> buffer;
56     void* cpu_ptr = nullptr; // stores the pointer to CPU mapping of a Shared MTLBuffer
57     size_t size; // size after alignment
58     size_t requested_size; // requested size (before alignment)
59     // buffer shape is used for retrieving base of views in cached graphs
60     std::vector<int64_t> shape;
61     bool in_use = false;
62     HeapBlock* heap;
63     id_t buf_id;
64     // counter to candidate least recently used buffers for garbage collection
65     uint32_t gc_count = 0;
66     uint32_t use_count = 0;
67     // counter to assign unique ids to buffer blocks
68     static uint64_t buffer_counter;
69     // Metal events used to sync GPU/CPU operations on the shared-storage buffers
70     MPSEventPtr event;
71   
72     BufferBlock(size_t Size, size_t RequestedSize = 0, const id<MTLBuffer> Buffer = nullptr, HeapBlock* Heap = nullptr)
73         : buffer(Buffer), size(Size), requested_size(RequestedSize), heap(Heap), buf_id(Buffer ? ++buffer_counter : 0) {}
74   
75     static bool Comparator(const BufferBlock* a, const BufferBlock* b) {
76       return (a->size != b->size) ? a->size < b->size : (uintptr_t)a->buffer < (uintptr_t)b->buffer;
77     }
78     static size_t alignUp(size_t Size, size_t Alignment) {
79       assert(((Alignment - 1) & Alignment) == 0);
80       return ((Size + Alignment - 1) & ~(Alignment - 1));
81     }
82     uint32_t retainCount() const {
83       return [buffer retainCount];
84     }
85   };
86   typedef bool (*BufferComparison)(const BufferBlock*, const BufferBlock*);

107  struct HeapBlock {
108    id<MTLHeap> heap;
109    struct {
110      size_t total, available;
111    } size;
112    BufferPool* pool;
113    unsigned int n_buffers = 0;
114    id_t heap_id;
115    // indicates if we split this heap to sub-allocate 'several' buffers (otherwise single buffer)
116    bool is_split;
117    // counter to assign unique ids to heap blocks
118    static uint64_t heap_counter;
119  
120    HeapBlock(size_t Size, const id<MTLHeap> Heap = nullptr, BufferPool* Pool = nullptr)
121        : heap(Heap),
122          size({.total = Size, .available = Size}),
123          pool(Pool),
124          heap_id(Heap ? ++heap_counter : 0),
125          is_split(true) {}
126  
127    static MTLResourceOptions getOptions(uint32_t usage) {
128      // TODO: check the caching performance of write-combined mode
129      MTLResourceOptions options = MTLResourceCPUCacheModeDefaultCache;
130  
131      if (usage & UsageFlags::MANAGED)
132        options |= MTLResourceStorageModeManaged;
133      else if (usage & UsageFlags::SHARED)
134        options |= MTLResourceStorageModeShared;
135      else
136        options |= MTLResourceStorageModePrivate;
137  
138      options |=
139          (usage & UsageFlags::HAZARD) ? MTLResourceHazardTrackingModeTracked : MTLResourceHazardTrackingModeUntracked;
140  
141      return options;
142    }
143  
144    static HeapBlock* createHeapBlock(AllocParams& params, id<MTLDevice> device, uint32_t usage) {
145      HeapBlock* heapBlock = nullptr;
146      bool is_split = true;
147      const size_t size = params.size();
148      MTLHeapDescriptor* d = [MTLHeapDescriptor new];
149      if (d) {
150        const size_t kXLargeHeap = params.has_unified_memory ? kXLargeHeapU : kXLargeHeapD;
151        if (size <= kMaxSmallAlloc) {
152          d.size = kSmallHeap;
153        } else if (size < kMinLargeAlloc) {
154          d.size = kLargeHeap;
155        } else if (size < kXLargeHeap / 2 && !params.has_memory_pressure) {
156          d.size = kXLargeHeap;
157        } else {
158          d.size = kRoundLarge * ((size + kRoundLarge - 1) / kRoundLarge);
159          is_split = false;
160        }
161        d.storageMode = (usage & UsageFlags::SHARED) ? MTLStorageModeShared : MTLStorageModePrivate;
162        d.cpuCacheMode = MTLCPUCacheModeDefaultCache;
163        // this automatically handles Metal buffer access synchronizations at the
164        // cost of slightly lower performance.
165        d.hazardTrackingMode =
166            (usage & UsageFlags::HAZARD) ? MTLHazardTrackingModeTracked : MTLHazardTrackingModeUntracked;
167        d.resourceOptions = getOptions(usage);
168        d.type = MTLHeapTypeAutomatic;
169        id<MTLHeap> heap = [device newHeapWithDescriptor:d];
170        if (heap) {
171          [heap setPurgeableState:MTLPurgeableStateNonVolatile];
172          const size_t heap_size = heapAvailableSize(heap);
173          heapBlock = new HeapBlock(heap_size, heap, params.pool);
174          if (heapBlock) {
175            heapBlock->is_split = is_split;
176          }
177        }
178        [d release];
179      }
180      return heapBlock;
181    }
182    static bool Comparator(const HeapBlock* a, const HeapBlock* b) {
183      return (a->size.available != b->size.available) ? a->size.available < b->size.available
184                                                      : (uintptr_t)a->heap < (uintptr_t)b->heap;
185    }
186    static NSUInteger heapAvailableSize(id<MTLHeap> heap, size_t Alignment = vm_page_size) {
187      return [heap maxAvailableSizeWithAlignment:Alignment];
188    }
189    NSUInteger Size() {
190      return [heap size];
191    }
192    id<MTLBuffer> newMTLBuffer(size_t length, uint32_t usage) {
193      id<MTLBuffer> buf = [heap newBufferWithLength:length options:getOptions(usage)];
194      if (buf) {
195        updateAvailableSize();
196        n_buffers++;
197      }
198      return buf;
199    }
200    // returns the retainCount before releasing the buffer
201    uint32_t releaseMTLBuffer(id<MTLBuffer>& buffer) {
202      const uint32_t retainCount = [buffer retainCount];
203      [buffer release];
204      buffer = nil;
205      updateAvailableSize();
206      n_buffers--;
207      return retainCount;
208    }
209    // returns the retainCount before releasing the heap
210    uint32_t releaseMTLHeap() {
211      const uint32_t retainCount = [heap retainCount];
212      TORCH_INTERNAL_ASSERT(!n_buffers); // assert if heap isn't empty
213      [heap setPurgeableState:MTLPurgeableStateEmpty];
214      [heap release];
215      heap = nil;
216      size.available = 0;
217      return retainCount;
218    }
219    uint32_t retainCount() const {
220      return [heap retainCount];
221    }
222    void updateAvailableSize() {
223      size.available = heapAvailableSize(heap);
224    }
225  };
226  typedef bool (*HeapComparison)(const HeapBlock*, const HeapBlock*);

228  struct BufferPool {
229    enum class Kind {
230      PRIVATE_SMALL,
231      PRIVATE_LARGE,
232      SHARED_SMALL,
233      SHARED_LARGE,
234      SCALAR,
235    };
236  
237    BufferPool(const id<MTLDevice> Device, uint32_t Usage)
238        : device(Device), usage(Usage), heaps(HeapBlock::Comparator), available_buffers(BufferBlock::Comparator) {}
239  
240    const id<MTLDevice> device;
241    // usage flags to customize the pool for various purposes (see UsageFlags enum)
242    const uint32_t usage;
243    // total number of buffers in the pool
244    uint32_t n_buffers = 0;
245    // total allocations size on this pool
246    size_t allocated_size = 0;
247    // total memory available in the pool
248    size_t available_size = 0;
249    // list of heaps ordered by their "available" (not total) memory size
250    std::set<HeapBlock*, HeapComparison> heaps;
251    // list of only "available" buffers in the pool (i.e., buffers not in-use)
252    std::set<BufferBlock*, BufferComparison> available_buffers;
253    // list of buffers that are in a state of "limbo" where they've already been freed
254    // from PyTorch-side, but were not returned to pool due to still being
255    // in-use by command buffers with retainCount > 1. In this state, the buffer is
256    // neither ready to be recycled, nor could be returned to pool as available.
257    // These buffers will be returned to pool once the command buffer's
258    // completionHandler callbacks are called.
259    std::unordered_set<BufferBlock*> buffers_pending_free;
260    // list of heaps pending size update
261    std::unordered_set<HeapBlock*> heaps_pending_update;
262  };

428    bool trigger_memory_callbacks(BufferBlock* buffer_block, IMpsAllocatorCallback::EventType event) const {
429      for (const auto& name : MPSAllocatorCallbacksRegistry()->Keys()) {
430        MPSAllocatorCallbacksRegistry()->Create(name)->executeMPSAllocatorCallback(
431            buffer_block ? buffer_block->buffer : nullptr, event);
432      }
433      return true;
434    }
