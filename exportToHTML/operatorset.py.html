1    """ A module for mapping operators to their corresponding eigenstates 
2    and vice versa 
3     
4    It contains a global dictionary with eigenstate-operator pairings. 
5    If a new state-operator pair is created, this dictionary should be 
6    updated as well. 
7     
8    It also contains functions operators_to_state and state_to_operators 
9    for mapping between the two. These can handle both classes and 
10   instances of operators and states. See the individual function 
11   descriptions for details. 
12    
13   TODO List:

15   """

20   from sympy.physics.quantum.state import StateBase, BraBase, Ket

29   #state_mapping stores the mappings between states and their associated
30   #operators or tuples of operators. This should be updated when new
31   #classes are written! Entries are of the form PxKet : PxOp or
32   #something like 3DKet : (ROp, ThetaOp, PhiOp)
33   
34   #frozenset is used so that the reverse mapping can be made
35   #(regular sets are not hashable because they are mutable
36   state_mapping = { JxKet: frozenset((J2Op, JxOp)),
37                     JyKet: frozenset((J2Op, JyOp)),
38                     JzKet: frozenset((J2Op, JzOp)),
39                     Ket: Operator,
40                     PositionKet3D: frozenset((XOp, YOp, ZOp)),
41                     PxKet: PxOp,
42                     XKet: XOp }
43   
44   op_mapping = {v: k for k, v in state_mapping.items()}

47   def operators_to_state(operators, **options):
48       """ Returns the eigenstate of the given operator or set of operators 
49    
50       A global function for mapping operator classes to their associated 
51       states. It takes either an Operator or a set of operators and 
52       returns the state associated with these. 
53    
54       This function can handle both instances of a given operator or 
55       just the class itself (i.e. both XOp() and XOp) 
56    
57       There are multiple use cases to consider: 
58    
59       1) A class or set of classes is passed: First, we try to 
60       instantiate default instances for these operators. If this fails, 
61       then the class is simply returned. If we succeed in instantiating 
62       default instances, then we try to call state._operators_to_state 
63       on the operator instances. If this fails, the class is returned. 
64       Otherwise, the instance returned by _operators_to_state is returned. 
65    
66       2) An instance or set of instances is passed: In this case, 
67       state._operators_to_state is called on the instances passed. If 
68       this fails, a state class is returned. If the method returns an 
69       instance, that instance is returned. 
70    
71       In both cases, if the operator class or set does not exist in the 
72       state_mapping dictionary, None is returned. 
73    
74       Parameters 
75       ========== 
76    
77       arg: Operator or set 
78            The class or instance of the operator or set of operators 
79            to be mapped to a state 
80    
81       Examples 
82       ======== 
83    
84       >>> from sympy.physics.quantum.cartesian import XOp, PxOp 
85       >>> from sympy.physics.quantum.operatorset import operators_to_state 
86       >>> from sympy.physics.quantum.operator import Operator 
87       >>> operators_to_state(XOp) 
88       |x> 
89       >>> operators_to_state(XOp()) 
90       |x> 
91       >>> operators_to_state(PxOp) 
92       |px> 
93       >>> operators_to_state(PxOp()) 
94       |px> 
95       >>> operators_to_state(Operator) 
96       |psi> 
97       >>> operators_to_state(Operator()) 
98       |psi> 
99       """
100  
101      if not (isinstance(operators, (Operator, set)) or issubclass(operators, Operator)):
102          raise NotImplementedError("Argument is not an Operator or a set!")
103  
104      if isinstance(operators, set):
105          for s in operators:
106              if not (isinstance(s, Operator)
107                     or issubclass(s, Operator)):
108                  raise NotImplementedError("Set is not all Operators!")
109  
110          ops = frozenset(operators)
111  
112          if ops in op_mapping:  # ops is a list of classes in this case
113              #Try to get an object from default instances of the
114              #operators...if this fails, return the class
115              try:
116                  op_instances = [op() for op in ops]
117                  ret = _get_state(op_mapping[ops], set(op_instances), **options)
118              except NotImplementedError:
119                  ret = op_mapping[ops]
120  
121              return ret
122          else:
123              tmp = [type(o) for o in ops]
124              classes = frozenset(tmp)
125  
126              if classes in op_mapping:
127                  ret = _get_state(op_mapping[classes], ops, **options)
128              else:
129                  ret = None
130  
131              return ret
132      else:
133          if operators in op_mapping:
134              try:
135                  op_instance = operators()
136                  ret = _get_state(op_mapping[operators], op_instance, **options)
137              except NotImplementedError:
138                  ret = op_mapping[operators]
139  
140              return ret
141          elif type(operators) in op_mapping:
142              return _get_state(op_mapping[type(operators)], operators, **options)
143          else:
144              return None

147  def state_to_operators(state, **options):
148      """ Returns the operator or set of operators corresponding to the 
149      given eigenstate 
150   
151      A global function for mapping state classes to their associated 
152      operators or sets of operators. It takes either a state class 
153      or instance. 
154   
155      This function can handle both instances of a given state or just 
156      the class itself (i.e. both XKet() and XKet) 
157   
158      There are multiple use cases to consider: 
159   
160      1) A state class is passed: In this case, we first try 
161      instantiating a default instance of the class. If this succeeds, 
162      then we try to call state._state_to_operators on that instance. 
163      If the creation of the default instance or if the calling of 
164      _state_to_operators fails, then either an operator class or set of 
165      operator classes is returned. Otherwise, the appropriate 
166      operator instances are returned. 
167   
168      2) A state instance is returned: Here, state._state_to_operators 
169      is called for the instance. If this fails, then a class or set of 
170      operator classes is returned. Otherwise, the instances are returned. 
171   
172      In either case, if the state's class does not exist in 
173      state_mapping, None is returned. 
174   
175      Parameters 
176      ========== 
177   
178      arg: StateBase class or instance (or subclasses) 
179           The class or instance of the state to be mapped to an 
180           operator or set of operators 
181   
182      Examples 
183      ======== 
184   
185      >>> from sympy.physics.quantum.cartesian import XKet, PxKet, XBra, PxBra 
186      >>> from sympy.physics.quantum.operatorset import state_to_operators 
187      >>> from sympy.physics.quantum.state import Ket, Bra 
188      >>> state_to_operators(XKet) 
189      X 
190      >>> state_to_operators(XKet()) 
191      X 
192      >>> state_to_operators(PxKet) 
193      Px 
194      >>> state_to_operators(PxKet()) 
195      Px 
196      >>> state_to_operators(PxBra) 
197      Px 
198      >>> state_to_operators(XBra) 
199      X 
200      >>> state_to_operators(Ket) 
201      O 
202      >>> state_to_operators(Bra) 
203      O 
204      """
205  
206      if not (isinstance(state, StateBase) or issubclass(state, StateBase)):
207          raise NotImplementedError("Argument is not a state!")
208  
209      if state in state_mapping:  # state is a class
210          state_inst = _make_default(state)
211          try:
212              ret = _get_ops(state_inst,
213                             _make_set(state_mapping[state]), **options)
214          except (NotImplementedError, TypeError):
215              ret = state_mapping[state]
216      elif type(state) in state_mapping:
217          ret = _get_ops(state,
218                         _make_set(state_mapping[type(state)]), **options)
219      elif isinstance(state, BraBase) and state.dual_class() in state_mapping:
220          ret = _get_ops(state,
221                         _make_set(state_mapping[state.dual_class()]))
222      elif issubclass(state, BraBase) and state.dual_class() in state_mapping:
223          state_inst = _make_default(state)
224          try:
225              ret = _get_ops(state_inst,
226                             _make_set(state_mapping[state.dual_class()]))
227          except (NotImplementedError, TypeError):
228              ret = state_mapping[state.dual_class()]
229      else:
230          ret = None
231  
232      return _make_set(ret)

235  def _make_default(expr):
236      # XXX: Catching TypeError like this is a bad way of distinguishing between
237      # classes and instances. The logic using this function should be rewritten
238      # somehow.
239      try:
240          ret = expr()
241      except TypeError:
242          ret = expr
243  
244      return ret

247  def _get_state(state_class, ops, **options):
248      # Try to get a state instance from the operator INSTANCES.
249      # If this fails, get the class
250      try:
251          ret = state_class._operators_to_state(ops, **options)
252      except NotImplementedError:
253          ret = _make_default(state_class)
254  
255      return ret
256  
257  
258  def _get_ops(state_inst, op_classes, **options):
259      # Try to get operator instances from the state INSTANCE.
260      # If this fails, just return the classes
261      try:
262          ret = state_inst._state_to_operators(op_classes, **options)
263      except NotImplementedError:
264          if isinstance(op_classes, (set, tuple, frozenset)):
265              ret = tuple(_make_default(x) for x in op_classes)
266          else:
267              ret = _make_default(op_classes)
268  
269      if isinstance(ret, set) and len(ret) == 1:
270          return ret[0]
271  
272      return ret

275  def _make_set(ops):
276      if isinstance(ops, (tuple, list, frozenset)):
277          return set(ops)
278      else:
279          return ops
