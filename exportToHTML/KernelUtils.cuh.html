14   __device__ inline __hip_bfloat162 preview_unsafeAtomicAdd(__hip_bfloat162* address, __hip_bfloat162 value) {
15   #if (defined(__gfx942__)) && \ 
16     __has_builtin(__builtin_amdgcn_flat_atomic_fadd_v2bf16)
17     typedef unsigned short __attribute__((ext_vector_type(2))) vec_short2;
18     static_assert(sizeof(vec_short2) == sizeof(__hip_bfloat162_raw));
19     union {
20       __hip_bfloat162_raw bf162_raw;
21       vec_short2 vs2;
22     } u{static_cast<__hip_bfloat162_raw>(value)};
23     u.vs2 = __builtin_amdgcn_flat_atomic_fadd_v2bf16((vec_short2*)address, u.vs2);
24     return static_cast<__hip_bfloat162>(u.bf162_raw);
25   #else
26     static_assert(sizeof(unsigned int) == sizeof(__hip_bfloat162_raw));
27     union u_hold {
28       __hip_bfloat162_raw h2r;
29       unsigned int u32;
30     };
31     u_hold old_val, new_val;
32     old_val.u32 = __hip_atomic_load((unsigned int*)address, __ATOMIC_RELAXED, __HIP_MEMORY_SCOPE_AGENT);
33     do {
34       new_val.h2r = __hadd2(old_val.h2r, value);
35     } while (!__hip_atomic_compare_exchange_strong(
36           (unsigned int*)address, &old_val.u32, new_val.u32,
37           __ATOMIC_RELAXED, __ATOMIC_RELAXED, __HIP_MEMORY_SCOPE_AGENT));
38     return old_val.h2r;
39   #endif
40   }

97   // fastSpecializedAtomicAdd (and fastAtomicAdd) are an optimization 
98   // that speed up half-precision atomics.  The situation with half 
99   // precision atomics is that we have a slow __half atomic, and 
100  // a fast vectored __half2 atomic (this can be worth up to a 6x 
101  // speedup, see https://github.com/pytorch/pytorch/pull/21879). 
102  // We can convert a __half atomic into a __half2 atomic by simply 
103  // pairing the __half with a zero entry on the left/right depending 
104  // on alignment... but only if this wouldn't cause an out of bounds 
105  // access!  Thus, you must specify tensor and numel so we can check 
106  // if you would be out-of-bounds and use a plain __half atomic if 
107  // you would be. 
108  template <
109      typename scalar_t,
110      typename index_t,
111      typename std::enable_if_t<std::is_same_v<c10::Half, scalar_t>>* =
112          nullptr>
113  __device__ __forceinline__ void fastSpecializedAtomicAdd(
114      scalar_t* tensor,
115      index_t index,
116      const index_t numel,
117      scalar_t value) {
118  #if (                      \ 
119      (defined(USE_ROCM) && ROCM_VERSION < 60201) || \ 
120      (defined(__CUDA_ARCH__) && (__CUDA_ARCH__ < 700)))
121    gpuAtomicAddNoReturn(
122        reinterpret_cast<at::Half*>(tensor) + index,
123        static_cast<at::Half>(value));
124  #else
125    // Accounts for the chance tensor falls on an odd 16 bit alignment (ie, not 32 bit aligned) 
126    __half* target_addr = reinterpret_cast<__half*>(tensor + index);
127    bool low_byte = (reinterpret_cast<std::uintptr_t>(target_addr) % sizeof(__half2) == 0);
128  
129    if (low_byte && index < (numel - 1)) {
130      __half2 value2;
131      value2.x = static_cast<__half>(value);
132      value2.y = __int2half_rz(0);
133      ATOMICADD(reinterpret_cast<__half2*>(target_addr), value2);
134  
135    } else if (!low_byte && index > 0) {
136      __half2 value2;
137      value2.x = __int2half_rz(0);
138      value2.y = static_cast<__half>(value);
139      ATOMICADD(reinterpret_cast<__half2*>(target_addr - 1), value2);
140  
141    } else {
142  #ifdef USE_ROCM
143      gpuAtomicAddNoReturn(
144          reinterpret_cast<at::Half*>(tensor) + index, static_cast<at::Half>(value));
145  #else
146      atomicAdd(
147          reinterpret_cast<__half*>(tensor) + index, static_cast<__half>(value));
148  #endif
149    }
150  #endif
151  }

159      scalar_t* tensor,

181      __nv_bfloat162 value2;
182      value2.x = NATIVE_ZERO_BF16;
183      value2.y = *reinterpret_cast<__nv_bfloat16*>(&value);
184      ATOMICADD(reinterpret_cast<__nv_bfloat162*>(target_addr - 1), value2);
185  
186    } else {
187  #ifdef USE_ROCM
188      gpuAtomicAddNoReturn(
189          reinterpret_cast<at::BFloat16*>(tensor) + index, static_cast<at::BFloat16>(value));
190  #else
191      atomicAdd(

194    }
195  #endif

212  template <class scalar_t, class index_t> 
213  __device__ __forceinline__ void fastAtomicAdd(
214      scalar_t* tensor,
215      index_t index,
216      const index_t numel,
217      scalar_t value,
218      bool fast_atomics) {
219    if (fast_atomics) {
220      fastSpecializedAtomicAdd(tensor, index, numel, value);
221    } else {
222      gpuAtomicAddNoReturn(tensor + index, value);
223    }
224  }

226  #if (defined(__gfx940__) || defined(__gfx941__) || defined(__gfx942__) || defined(__gfx950__))
227  // This function implements warp-level opportunistic fastatomics 
228  // To reduce contention on an atomicAdd, this replaces per-thread atomicAdd with a per-warp atomicAdd. 
229  // We identify all the threads within a warp that will perform an atomicAdd on the same destination 
230  // address and perform the addition on the CU. Each warp elects a leader thread which does the 
231  // atomicAdd to the destination address. 
232  template <class scalar_t, class index_t> 
233  __device__ __forceinline__ void opportunistic_fastAtomicAdd(
234      scalar_t* self_ptr,
235      index_t index,
236      const index_t numel,
237      scalar_t value) {
238  
239      scalar_t* dst = self_ptr + index;
240  
241      //pack coalseced bf16 and fp16 
242      if constexpr (std::is_same<scalar_t, c10::BFloat16>::value || std::is_same<scalar_t, c10::Half>::value)
243      {
244          typedef unsigned short __attribute__((ext_vector_type(2))) vec_short2;
245          union ill { unsigned int i[2]; int64_t il; };
246          ill iil_, ill_oneUpDst = {};
247          iil_.il = (int64_t)dst;
248          ill_oneUpDst.i[0] = __builtin_amdgcn_mov_dpp(iil_.i[0], 0x130, 0xf, 0xf, 0);
249          ill_oneUpDst.i[1] = __builtin_amdgcn_mov_dpp(iil_.i[1], 0x130, 0xf, 0xf, 0);
250          union bfi {scalar_t bf; short s; } bfi_ = { .bf = value  }; bfi bfi_oneUpVal;
251  
252          bfi_oneUpVal.s = __builtin_amdgcn_mov_dpp(bfi_.s, 0x130, 0xf, 0xf, 0);
253          auto oneUpVal = bfi_oneUpVal.bf;
254  
255          __half* target_addr = reinterpret_cast<__half*>(self_ptr + index);
256          bool low_byte = (reinterpret_cast<std::uintptr_t>(target_addr) % sizeof(__half2) == 0);
257          bool canCombnUp = (bool)(__activemask()&(1<<(threadIdx.x+1))) &&
258                                   (low_byte && index < (numel - 1)) &&
259                                   (ill_oneUpDst.il - reinterpret_cast<int64_t>(dst) == sizeof(scalar_t));
260          bool canCombnDn = (__builtin_amdgcn_mov_dpp(canCombnUp, 0x138, 0xf, 0xf, 0));
261  
262          if (__lane_id()%2==0)
263          {
264            if (canCombnUp) {
265              typedef _Float16 __attribute__((ext_vector_type(2))) vec_fp162;
266              union bfvs { scalar_t bf[2]; vec_short2 vs2; vec_fp162 df16;  };
267              bfvs bfvs_ = {};
268              bfvs_.bf[0] = value;
269              bfvs_.bf[1] = oneUpVal;
270              if constexpr (std::is_same<scalar_t, c10::BFloat16>::value)
271                __builtin_amdgcn_flat_atomic_fadd_v2bf16((vec_short2*)dst, bfvs_.vs2);
272              else
273                __builtin_amdgcn_flat_atomic_fadd_v2f16((__half2*)dst, bfvs_.df16);
274              return;
275            }
276          }
277          else
278          {
279            if (canCombnDn)
280              return;
281          }
282      }
283  
284      // not coalsced, so now let try to capture lane-matches... 
285      // __activemask() -- finds the set of threads in the warp that are about to perform atomicAdd 
286      // __match_any_sync() -- returns bit mask of the threads that have same dest addr 
287      auto mask = __match_any_sync(__activemask(), (int64_t)dst);
288  
289      // select a leader thread 
290      int leader = __ffsll(mask) - 1;
291  
292      scalar_t crnt_val = (scalar_t)0;
293      auto crnt_msk = mask >> (leader);
294      int crnt_idx = leader;
295  
296      // __shfl is limited in the dtypes it accepts 
297      // That's why, we need these if/else to correctly do the addition on the CU 
298      if constexpr(sizeof(scalar_t) <= sizeof(int)) {
299       union punner { int l; scalar_t s; };
300       punner pnr = {};
301       pnr.s = value;
302       while (crnt_msk != 0) {
303          if (crnt_msk & 1) {
304              punner add_val = {};
305              add_val.l = __shfl(pnr.l ,crnt_idx);
306              crnt_val += add_val.s;
307          }
308          crnt_idx++;
309          crnt_msk = crnt_msk >> 1;
310       }
311      }
312      else if constexpr(sizeof(scalar_t) <= sizeof(long)) {
313       union punner { long l; scalar_t s; };
314       punner pnr = {};
315       pnr.s = value;
316       while (crnt_msk != 0) {
317          if (crnt_msk & 1) {
318              punner add_val = {};
319              add_val.l = __shfl(pnr.l ,crnt_idx);
320              crnt_val += add_val.s;
321          }
322          crnt_idx++;
323          crnt_msk = crnt_msk >> 1;
324       }
325      }
326      else if constexpr(sizeof(scalar_t) <= sizeof(long long)) {
327       union punner { long long l; scalar_t s; };
328       punner pnr = {};
329       pnr.s = value;
330       while (crnt_msk != 0) {
331          if (crnt_msk & 1) {
332              punner add_val = {};
333              add_val.l = __shfl(pnr.l ,crnt_idx);
334              crnt_val += add_val.s;
335          }
336          crnt_idx++;
337          crnt_msk = crnt_msk >> 1;
338       }
339      }
340      else {
341       union punner { long long l[2]; scalar_t s; };
342       punner pnr = {};
343       pnr.s = value;
344       while (crnt_msk != 0) {
345          if (crnt_msk & 1) {
346              punner add_val = {};
347              add_val.l[0] = __shfl(pnr.l[0] ,crnt_idx);
348              add_val.l[1] = __shfl(pnr.l[1] ,crnt_idx);
349              crnt_val += add_val.s;
350          }
351          crnt_idx++;
352          crnt_msk = crnt_msk >> 1;
353       }
354      }
355  
356  
357      //Once the correct crnt_val is determined, only the leader thread does the update to the dest addr 
358      if (__lane_id() == leader) {
359        fastAtomicAdd(self_ptr, index, numel, crnt_val, true);
360      }
361  }
