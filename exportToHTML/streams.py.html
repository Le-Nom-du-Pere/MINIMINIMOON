14   class Stream(torch._C._CudaStreamBase):
15       r"""Wrapper around a CUDA stream. 
16    
17       A CUDA stream is a linear sequence of execution that belongs to a specific 
18       device, independent from other streams. It supports with statement as a 
19       context manager to ensure the operators within the with block are running 
20       on the corresponding stream.  See :ref:`cuda-semantics` for details. 
21    
22       Args: 
23           device(torch.device or int, optional): a device on which to allocate 
24               the stream. If :attr:`device` is ``None`` (default) or a negative 
25               integer, this will use the current device. 
26           priority(int, optional): priority of the stream, which can be positive, 0, or negative. 
27               A lower number indicates a higher priority. By default, the priority is set to 0. 
28               If the value falls outside of the allowed priority range, it will automatically be 
29               mapped to the nearest valid priority (lowest for large positive numbers or 
30               highest for large negative numbers). 
31    
32       """
33   
34       def __new__(cls, device=None, priority=0, **kwargs):
35           # setting device manager is expensive, so we avoid it unless necessary
36           if device is None or ("stream_id" in kwargs and "device_index" in kwargs):
37               return super().__new__(cls, priority=priority, **kwargs)
38           else:
39               with torch.cuda.device(device):
40                   return super().__new__(cls, priority=priority, **kwargs)
41   
42       def wait_event(self, event) -> None:
43           r"""Make all future work submitted to the stream wait for an event. 
44    
45           Args: 
46               event (torch.cuda.Event): an event to wait for. 
47    
48           .. note:: This is a wrapper around ``cudaStreamWaitEvent()``: see 
49              `CUDA Stream documentation`_ for more info. 
50    
51              This function returns without waiting for :attr:`event`: only future 
52              operations are affected. 
53    
54           .. _CUDA Stream documentation: 
55              https://docs.nvidia.com/cuda/cuda-runtime-api/group__CUDART__STREAM.html 
56           """
57           event.wait(self)
58   
59       def wait_stream(self, stream) -> None:
60           r"""Synchronize with another stream. 
61    
62           All future work submitted to this stream will wait until all kernels 
63           submitted to a given stream at the time of call complete. 
64    
65           Args: 
66               stream (Stream): a stream to synchronize. 
67    
68           .. note:: This function returns without waiting for currently enqueued 
69              kernels in :attr:`stream`: only future operations are affected. 
70           """
71           self.wait_event(stream.record_event())
72   
73       def record_event(self, event=None):
74           r"""Record an event. 
75    
76           Args: 
77               event (torch.cuda.Event, optional): event to record. If not given, a new one 
78                   will be allocated. 
79    
80           Returns: 
81               Recorded event. 
82           """
83           if event is None:
84               event = Event()
85           event.record(self)
86           return event
87   
88       def query(self) -> bool:
89           r"""Check if all the work submitted has been completed. 
90    
91           Returns: 
92               A boolean indicating if all kernels in this stream are completed. 
93           """
94           return super().query()
95   
96       def synchronize(self) -> None:
97           r"""Wait for all the kernels in this stream to complete. 
98    
99           .. note:: This is a wrapper around ``cudaStreamSynchronize()``: see 
100             `CUDA Stream documentation`_ for more info. 
101          """
102          super().synchronize()
103  
104      @property
105      def _as_parameter_(self):
106          return ctypes.c_void_p(self.cuda_stream)
107  
108      def __eq__(self, o) -> bool:
109          if isinstance(o, Stream):
110              return super().__eq__(o)
111          return False
112  
113      def __hash__(self):
114          return hash((self.cuda_stream, self.device))
115  
116      def __repr__(self):
117          return f"<torch.cuda.Stream device={self.device} cuda_stream={self.cuda_stream:#x}>"

120  class ExternalStream(Stream):
121      r"""Wrapper around an externally allocated CUDA stream. 
122   
123      This class is used to wrap streams allocated in other libraries in order 
124      to facilitate data exchange and multi-library interactions. 
125   
126      .. note:: This class doesn't manage the stream life-cycle, it is the user 
127         responsibility to keep the referenced stream alive while this class is 
128         being used. 
129   
130      Args: 
131          stream_ptr(int): Integer representation of the `cudaStream_t` value. 
132              allocated externally. 
133          device(torch.device or int, optional): the device where the stream 
134              was originally allocated. If device is specified incorrectly, 
135              subsequent launches using this stream may fail. 
136      """
137  
138      def __new__(cls, stream_ptr, device=None, **kwargs):
139          with torch.cuda.device(device):
140              return super().__new__(cls, stream_ptr=stream_ptr, **kwargs)

143  class Event(torch._C._CudaEventBase):
144      r"""Wrapper around a CUDA event. 
145   
146      CUDA events are synchronization markers that can be used to monitor the 
147      device's progress, to accurately measure timing, and to synchronize CUDA 
148      streams. 
149   
150      The underlying CUDA events are lazily initialized when the event is first 
151      recorded or exported to another process. After creation, only streams on the 
152      same device may record the event. However, streams on any device can wait on 
153      the event. 
154   
155      Args: 
156          enable_timing (bool, optional): indicates if the event should measure time 
157              (default: ``False``) 
158          blocking (bool, optional): if ``True``, :meth:`wait` will be blocking (default: ``False``) 
159          interprocess (bool): if ``True``, the event can be shared between processes 
160              (default: ``False``) 
161          external (bool, optional): indicates whether this event should create event record and event wait nodes, or create an internal cross-stream dependency, when captured in a cuda graph. See `cross-stream dependencies <https://docs.nvidia.com/cuda/archive/12.9.0/cuda-c-programming-guide/index.html#cross-stream-dependencies-and-events>`_, `cudaEventRecordExternal <https://docs.nvidia.com/cuda/archive/12.9.0/cuda-runtime-api/group__CUDART__TYPES.html#group__CUDART__TYPES_1g3457b81d1d32c6a00f6132fbc2693d47>`_, and `cudaEventWaitExternal <https://docs.nvidia.com/cuda/archive/12.9.0/cuda-runtime-api/group__CUDART__TYPES.html#group__CUDART__TYPES_1g0c23426b7252eaa9cef695859991304e>`_ for more information about internal vs. external events. (default: ``False``) 
162   
163      .. _CUDA Event Documentation: 
164         https://docs.nvidia.com/cuda/cuda-runtime-api/group__CUDART__EVENT.html 
165      """  # noqa: B950
166  
167      def __new__(
168          cls, enable_timing=False, blocking=False, interprocess=False, external=False
169      ):
170          return super().__new__(
171              cls,
172              enable_timing=enable_timing,
173              blocking=blocking,
174              interprocess=interprocess,
175              external=external,
176          )
177  
178      @classmethod
179      def from_ipc_handle(cls, device, handle):
180          r"""Reconstruct an event from an IPC handle on the given device."""
181          return super().from_ipc_handle(device, handle)
182  
183      def record(self, stream=None):
184          r"""Record the event in a given stream. 
185   
186          Uses ``torch.cuda.current_stream()`` if no stream is specified. The 
187          stream's device must match the event's device. 
188          """
189          if stream is None:
190              stream = torch.cuda.current_stream()
191          super().record(stream)
192  
193      def wait(self, stream=None) -> None:
194          r"""Make all future work submitted to the given stream wait for this event. 
195   
196          Use ``torch.cuda.current_stream()`` if no stream is specified. 
197   
198          .. note:: This is a wrapper around ``cudaStreamWaitEvent()``: see 
199              `CUDA Event documentation`_ for more info. 
200          """
201          if stream is None:
202              stream = torch.cuda.current_stream()
203          super().wait(stream)
204  
205      def query(self):
206          r"""Check if all work currently captured by event has completed. 
207   
208          Returns: 
209              A boolean indicating if all work currently captured by event has 
210              completed. 
211          """
212          return super().query()
213  
214      def elapsed_time(self, end_event):
215          r"""Return the time elapsed. 
216   
217          Time reported in milliseconds after the event was recorded and 
218          before the end_event was recorded. 
219          """
220          return super().elapsed_time(end_event)
221  
222      def synchronize(self) -> None:
223          r"""Wait for the event to complete. 
224   
225          Waits until the completion of all work currently captured in this event. 
226          This prevents the CPU thread from proceeding until the event completes. 
227   
228           .. note:: This is a wrapper around ``cudaEventSynchronize()``: see 
229              `CUDA Event documentation`_ for more info. 
230          """
231          super().synchronize()
232  
233      def ipc_handle(self):
234          r"""Return an IPC handle of this event. 
235   
236          If not recorded yet, the event will use the current device. 
237          """
238          return super().ipc_handle()
239  
240      @property
241      def _as_parameter_(self):
242          return ctypes.c_void_p(self.cuda_event)
243  
244      def __repr__(self) -> str:
245          if self.cuda_event:
246              return f"<torch.cuda.Event {self._as_parameter_.value:#x}>"
247          else:
248              return "<torch.cuda.Event uninitialized>"
