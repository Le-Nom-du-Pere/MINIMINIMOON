14   rand_ops = {
15       aten.dropout,
16       aten._fused_dropout,
17       aten._standard_gamma,
18       aten.bernoulli,
19       aten.multinomial,
20       aten.native_dropout,

24       aten.rrelu,

26       aten.rand,
27       aten.randint,
28       aten.randn,

33       aten.add_,
34       aten.sub_,
35       aten.mul_,
36       aten.div_,
37       aten.pow_,

45   @torch.fx._compatibility.compatibility(is_backward_compatible=False)
46   def get_CSE_banned_ops():
47       return rand_ops.union(inplace_ops)

50   @torch.fx._compatibility.compatibility(is_backward_compatible=False)
51   class CSEPass(PassBase):
52       def __init__(self, banned_ops=None):
53           """ 
54           This version of CSE Pass aims to be dialect agnostic, and it's implemented purely based on the connectivity between fx.Node. 
55    
56           For functional dialects, user would only need to specify the random ops in ban list. 
57    
58           Warning: CSE Pass cannot be safely applied on a FX graph in non-functional dialects. 
59           If your dialect contains stateful operators, please customized the banned_ops. 
60    
61           """
62           if banned_ops is None:
63               banned_ops = set()
64           self.banned_ops = banned_ops
65           super().__init__()
66   
67       def call(self, graph_module: GraphModule) -> PassResult:
68           """ 
69           Return a new copy of torch.fx.GraphModule with CSE applied to the input graph 
70    
71           Example usage: 
72    
73           from torch.fx.experimental.proxy_tensor import make_fx 
74           def f(a): 
75               b = a * a 
76               c = a * a 
77               return b+c 
78    
79           p = CSEPass() 
80           traced_graph = make_fx(f)(torch.tensor(1)) 
81           print(traced_graph) 
82           result = p(traced_graph) 
83           print(result.graph_module) 
84           """
85   
86           def get_aten_target(node):
87               if hasattr(node.target, "overloadpacket"):
88                   return node.target.overloadpacket
89               return node.target
90   
91           modified = False
92           new_graph = Graph()
93           env: dict[
94               Node, Node
95           ] = {}  # map from node in the old graph to node in the new graph
96           hash_env: dict[
97               tuple[torch._ops.OpOverload, int], Node
98           ] = {}  # map from hash to a node in the new graph
99           token_map: dict[
100              tuple[torch._ops.OpOverload, int], dict[str, Any]
101          ] = {}  # map from hash to token
102          for n in graph_module.graph.nodes:
103              # The placeholder, output, and get_attr nodes are copied to the new graph without change
104              # do not CSE away random operations
105              if (
106                  n.op == "placeholder"
107                  or n.op == "output"
108                  or n.op == "get_attr"
109                  or get_aten_target(n) in self.banned_ops
110              ):
111                  new_node = new_graph.node_copy(n, lambda x: env[x])
112                  env[n] = new_node
113              else:  # n.op == 'call_function', should never see n.op == 'call_module' or 'call_method'
114                  # substitute args and kwargs members to their mapping in env if exists
115                  # specs can be used to reconstruct nested list/dictionaries
116                  def substitute(arg_list):
117                      arg_list, spec = tree_flatten(arg_list)
118                      for i in range(len(arg_list)):
119                          v = arg_list[i]
120                          if isinstance(v, Node) and v in env:
121                              arg_list[i] = env[v]
122                      return tuple(arg_list), spec
123  
124                  args, args_spec = substitute(n.args)
125                  kwargs, kwargs_spec = substitute(n.kwargs)
126  
127                  # each token corresponds to a unique node
128                  # nodes with the same token can be substituted
129                  token = {
130                      "target": n.target,
131                      "args": args,
132                      "args_spec": args_spec,
133                      "kwargs": kwargs,
134                      "kwargs_spec": kwargs_spec,
135                  }
136  
137                  # hash substituted args to a number, do not hash specs because specs are not hashable
138                  hash_arg = hash((args, kwargs))
139                  hash_val = (n.target, hash_arg)
140  
141                  # check if a node has a substitute and can be eliminated
142                  hash_val_in_hash_env = hash_val in hash_env
143                  if hash_val_in_hash_env and token_map[hash_val] == token:
144                      modified = True  # substitution happens and the graph is modified
145                      env[n] = hash_env[hash_val]
146                      continue
147  
148                  new_node = new_graph.node_copy(n, lambda x: env[x])
149                  env[n] = new_node
150                  if not hash_val_in_hash_env:
151                      hash_env[hash_val] = new_node
152                      token_map[hash_val] = token
153  
154          csed_gm = GraphModule(graph_module, new_graph)
155          return PassResult(csed_gm, modified)
