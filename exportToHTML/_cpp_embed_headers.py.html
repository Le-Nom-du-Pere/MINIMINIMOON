12   def _embed_headers(
13       content: list[str], include_dirs: list[Path], processed_files: set[str]
14   ) -> str:
15       for line_idx, cur_line in enumerate(content):
16           # Eliminate warning: `#pragma once in main file`
17           if cur_line.startswith("#pragma once"):
18               content[line_idx] = ""
19               continue
20           m = _match('^\\s*#include\\s*[<"]([^>"]+)[>"]', cur_line)
21           if m is None:
22               continue
23           for include_dir in include_dirs:
24               path = include_dir / m[1]
25               if not path.exists():
26                   continue
27               if str(path) in processed_files:
28                   content[line_idx] = ""
29                   continue
30               processed_files.add(str(path))
31               content[line_idx] = _embed_headers(
32                   read_file(path), include_dirs, processed_files
33               )
34               break
35       return "".join(content)

38   def embed_headers(
39       fname: str, include_dirs: Optional[Union[Sequence[str], Sequence[Path], str]] = None
40   ) -> str:
41       if include_dirs is None:
42           base_dir = Path(__file__).parent.parent.parent
43           include_dirs = [base_dir, base_dir / "aten" / "src"]
44       elif isinstance(include_dirs, str):
45           include_dirs = [Path(include_dirs)]
46       else:
47           include_dirs = [Path(x) for x in include_dirs]
48   
49       return _embed_headers(read_file(fname), include_dirs, {fname})
