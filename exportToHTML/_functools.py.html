1    import functools
2    from typing import Callable, TypeVar
3    from typing_extensions import Concatenate, ParamSpec
4    
5    
6    _P = ParamSpec("_P")
7    _T = TypeVar("_T")
8    _C = TypeVar("_C")
9    
10   # Sentinel used to indicate that cache lookup failed.
11   _cache_sentinel = object()
12   
13   
14   def cache_method(
15       f: Callable[Concatenate[_C, _P], _T]
16   ) -> Callable[Concatenate[_C, _P], _T]:
17       """ 
18       Like `@functools.cache` but for methods. 
19    
20       `@functools.cache` (and similarly `@functools.lru_cache`) shouldn't be used 
21       on methods because it caches `self`, keeping it alive 
22       forever. `@cache_method` ignores `self` so won't keep `self` alive (assuming 
23       no cycles with `self` in the parameters). 
24    
25       Footgun warning: This decorator completely ignores self's properties so only 
26       use it when you know that self is frozen or won't change in a meaningful 
27       way (such as the wrapped function being pure). 
28       """
29       cache_name = "_cache_method_" + f.__name__
30   
31       @functools.wraps(f)
32       def wrap(self: _C, *args: _P.args, **kwargs: _P.kwargs) -> _T:
33           assert not kwargs
34           if not (cache := getattr(self, cache_name, None)):
35               cache = {}
36               setattr(self, cache_name, cache)
37           cached_value = cache.get(args, _cache_sentinel)
38           if cached_value is not _cache_sentinel:
39               return cached_value
40           value = f(self, *args, **kwargs)
41           cache[args] = value
42           return value
43   
44       return wrap
