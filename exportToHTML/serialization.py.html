21   def _create_tensor_proto_with_external_data(
22       tensor: torch.Tensor,
23       name: str,
24       location: str,
25       basepath: str,
26       dtype_override: onnx.TypeProto | None = None,  # type: ignore[name-defined]
27   ) -> onnx.TensorProto:  # type: ignore[name-defined]
28       """Create a TensorProto with external data from a PyTorch tensor. 
29       The external data is saved to os.path.join(basepath, location). 
30    
31       Args: 
32           tensor: Tensor to be saved. 
33           name: Name of the tensor (i.e., initializer name in ONNX graph). 
34           location: Relative location of the external data file 
35               (e.g., "/tmp/initializers/weight_0" when model is "/tmp/model_name.onnx"). 
36           basepath: Base path of the external data file (e.g., "/tmp/external_data" while model must be in "/tmp"). 
37    
38    
39       Reference for ONNX's external data format: 
40           How to load? 
41           https://github.com/onnx/onnx/blob/5dac81ac0707bdf88f56c35c0a5e8855d3534673/onnx/external_data_helper.py#L187 
42           How to save? 
43           https://github.com/onnx/onnx/blob/5dac81ac0707bdf88f56c35c0a5e8855d3534673/onnx/external_data_helper.py#L43 
44           How to set ONNX fields? 
45           https://github.com/onnx/onnx/blob/5dac81ac0707bdf88f56c35c0a5e8855d3534673/onnx/external_data_helper.py#L88 
46       """
47       # FIXME: Avoid importing onnx into torch.onnx.
48       import onnx
49   
50       scalar_type = (
51           jit_type_utils.JitScalarType.from_onnx_type(
52               dtype_override.tensor_type.elem_type
53           )
54           if dtype_override is not None
55           else jit_type_utils.JitScalarType.from_dtype(tensor.dtype)
56       )
57   
58       # Checkpoints can be stored with a different dtype as the model expects because
59       # the user script can explicitly cast the original type to something or maybe
60       # PyTorch's type promotion might do it
61       if dtype_override is not None and scalar_type.dtype() != tensor.dtype:
62           tensor = tensor.to(scalar_type.dtype())
63   
64       tensor_proto = onnx.TensorProto()  # type: ignore[attr-defined]
65       tensor_proto.name = name
66       tensor_proto.data_type = scalar_type.onnx_type()  # type: ignore[assignment]
67   
68       tensor_proto.dims.extend(tensor.shape)
69       tensor_proto.data_location = onnx.TensorProto.EXTERNAL  # type: ignore[attr-defined]
70   
71       # Settings for saving one tensor per file.
72       # Offset is zero because there is no other tensor in the same file.
73       key_value_pairs = {
74           "location": location,
75           "offset": 0,
76           "length": tensor.untyped_storage().nbytes(),
77       }
78       for k, v in key_value_pairs.items():
79           entry = tensor_proto.external_data.add()
80           entry.key = k
81           entry.value = str(v)
82   
83       # Actual path to write content of tensor.
84       external_data_file_path = os.path.join(basepath, location)
85       if os.path.exists(external_data_file_path):
86           os.remove(external_data_file_path)
87   
88       # Create external data's folder if not exists.
89       external_data_dir_path = os.path.dirname(external_data_file_path)
90       if not os.path.exists(external_data_dir_path):
91           # if the demo_folder directory is not present
92           # then create it.
93           os.makedirs(external_data_dir_path)
94   
95       # Create a fresh file.
96       with open(external_data_file_path, "xb") as data_file:
97           # No need to call "seek" because offset is 0.
98           # data_file.seek(0)
99           # Write tensor content to the file.
100          data_file.write(tensor.numpy(force=True).tobytes())
101  
102      return tensor_proto

105  def _convert_safetensors_to_torch_format(safetensors_file):
106      # It this function is called, safetensors is guaranteed to exist
107      # because the HF model with safetensors was already loaded and exported to ONNX
108      from safetensors import safe_open  # type: ignore[import-not-found, import-untyped]
109  
110      tensors = {}
111      with safe_open(safetensors_file, framework="pt", device="cpu") as f:  # type: ignore[attr-defined]
112          for k in f.keys():
113              tensors[k] = f.get_tensor(k).cpu()
114      return tensors

118  def save_model_with_external_data(
119      basepath: str,
120      model_location: str,
121      initializer_location: str,
122      torch_state_dicts: tuple[dict | FileLike, ...],
123      onnx_model: onnx.ModelProto,  # type: ignore[name-defined]
124      rename_initializer: bool = False,
125  ) -> None:
126      """Load PyTorch tensors from files and add to "onnx_model" as external initializers. 
127   
128      Output files: 
129          ONNX model file path: 
130          ONNX initializer folder: os.path.join(basepath, initializer_location) 
131   
132      After running this function, you can do 
133          ort_sess = onnxruntime.InferenceSession(os.path.join(basepath, model_location)) 
134      to execute the model. 
135   
136      Arguments: 
137          basepath: Base path of the ONNX external data file (e.g., "/path/to/large_model/"). 
138          model_location: Relative location of the ONNX model file. 
139              E.g., "model.onnx" so that the model file is saved to 
140              "<basepath>/model.onnx". 
141          initializer_location: Relative location of the ONNX initializer folder. 
142              E.g., "initializers" so that the initializers are saved to 
143              "<basepath>/initializers/". 
144              Note: When initializers are >2GB, must be the same as `model_location`. 
145          torch_state_dicts: Dictionaries or files which contain PyTorch tensors to be saved 
146              as ONNX initializers. For non-dict arguments, `torch.load` will be used to load them from file-like objects. 
147          onnx_model: ONNX model to be saved with external initializers. 
148              If an input name matches a tensor loaded from "torch_state_dicts", 
149              the tensor will be saved as that input's external initializer. 
150          rename_initializer: Replaces "." by "_" for all ONNX initializer names. 
151              Not needed by the official torch.onnx.dynamo_export. This is a hack 
152              for supporting `FXSymbolicTracer` tracer with fake tensor mode. 
153              In short, `FXSymbolicTracer` lifts FX parameters (self.linear_weight) 
154              as inputs (`def forward(self, linear_weight)`) and therefore, `.` cannot be used. 
155      """
156      # FIXME: Avoid importing onnx into torch.onnx.
157      import onnx
158  
159      initializers_to_be_deleted = {}  # Using dict because it is **ordered**
160      existing_initializers = {
161          k.name: idx for idx, k in enumerate(onnx_model.graph.initializer)
162      }
163      onnx_input_names = {input.name for input in onnx_model.graph.input}
164      for el in torch_state_dicts:
165          if isinstance(el, dict):
166              # Useful for when state_dict is loaded with torch.load(..., mmap=True, map_location="cpu") by the user
167              # Using torch.save wouldn't leverage mmap, leading to higher memory usage
168              state_dict = el
169          else:
170              if isinstance(el, (str, os.PathLike)) and os.fspath(el).endswith(
171                  ".safetensors"
172              ):
173                  state_dict = _convert_safetensors_to_torch_format(el)
174              else:
175                  try:
176                      # Loads checkpoint using memory-map on CPU to support really large models
177                      # The underlying torch.UntypedStorage is memory mapped, so state_dict is lazy loaded
178                      state_dict = torch.load(el, map_location="cpu", mmap=True)
179                  except (RuntimeError, ValueError) as e:
180                      if "mmap can only be used with files saved with" in str(e) or (
181                          isinstance(el, (io.IOBase, IO))
182                          and el.readable()
183                          and el.seekable()
184                      ):
185                          log.warning(
186                              "Failed to load the checkpoint with memory-map enabled, retrying without memory-map."
187                              "Consider updating the checkpoint with mmap by using torch.save() on PyTorch version >= 1.6."
188                          )
189                          if isinstance(el, (io.IOBase, IO)):
190                              el.seek(0)  # torch.load from `try:` has read the file.
191                          state_dict = torch.load(el, map_location="cpu")
192                      else:
193                          raise e
194  
195          for name, tensor in state_dict.items():
196              if rename_initializer:
197                  # Basically, "transformer.attention.self.query.weight" is mapped
198                  # to "transformer_attention_self_query_weight" for mimicking the
199                  # name-modifying code in FX-to-ONNX exporter.
200                  # See function _replace_get_attr_with_placeholder for details.
201                  name = name.replace(".", "_")
202  
203              # This block tries to match the onnx initializer name with torch parameter/buffer
204              #  e.g. A pytorch buffer 'transformer.h.0.attn.bias' can be named 'h.0.attn.bias' in a ONNX initializer
205              # For each PyTorch tensor name loaded by torch.load,
206              #  1.  Search its best match in ONNX model. E.g., the match of
207              #       "transformer_attention_weight" could be "attention_weight".
208              #  2.  Set "tensor" as the initializer of the matched ONNX input.
209              #      E.g., "tensor" is stored as the initializer of "attention_weight".
210              # Step 1 is required because sometimes, tensor names are stored with prefix the dictionary
211              # loaded by torch.load.
212              if name in onnx_input_names:
213                  # Same input name shouldn't be matched again
214                  onnx_input_names.remove(name)
215              else:
216                  for onnx_input_name in onnx_input_names:
217                      if onnx_input_name.endswith(name) or name.endswith(onnx_input_name):
218                          # Find a match. Change name to the matched ONNX input name, so that we
219                          # create initializer with the right ONNX name.
220                          name = onnx_input_name
221                          onnx_input_names.remove(onnx_input_name)
222                          break
223  
224              relative_tensor_file_path = os.path.join(initializer_location, name)
225              # Create one file per tensor.
226              # tensor_proto.raw_data is stored to external file at
227              # os.path.join(basepath, relative_tensor_file_path).
228              model_input_types = {k.name: k.type for k in onnx_model.graph.input}
229  
230              # Mark for deletion - a replacement will be appended next
231              if name in existing_initializers:
232                  initializers_to_be_deleted[existing_initializers[name]] = name
233              tensor_proto = _create_tensor_proto_with_external_data(
234                  tensor,
235                  name,
236                  relative_tensor_file_path,
237                  basepath,
238                  model_input_types.pop(name, None),
239              )
240              # Add the tensor_proto to the ONNX model as an initializer with external data.
241              onnx_model.graph.initializer.append(tensor_proto)
242      # Remove old duplicated initializers, if any. delete in desc order to not invalidate deletion indices
243      initializers_to_be_deleted = dict(
244          sorted(initializers_to_be_deleted.items(), reverse=True)
245      )
246      for idx in initializers_to_be_deleted.keys():
247          del onnx_model.graph.initializer[idx]
248  
249      # model_location should be a pure file name such as "file_name.onnx", not "folder/file_name.onnx".
250      onnx.save(onnx_model, os.path.join(basepath, model_location))  # type: ignore[attr-defined]
