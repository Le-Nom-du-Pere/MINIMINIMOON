54   @dataclass
55   class _MinimizerSettingBase:
56       """ 
57       Args: 
58       `accumulate_error`: Instead of using a's input for both converted module to verify 
59       , use the previous outputs of each converted module as input to accumulate the 
60       errors. 
61    
62       `traverse_method`: "sequential" or "binary" or "accumulate" 
63       Determine the way of traverse the nodes in FX module. 
64    
65       `find_all`: Minimizer will go through the entire model and return all problematic nodes. 
66    
67       `return_intermediate`: If true, when using `run_nodes()` function to run the 
68       model, intermediate results of all the ops will be returned as output. 
69    
70       `all_outputs`: If true, when using `_run_and_compare()` function, 
71       all the output nodes in the subgraph will be used for comparison. 
72       """
73   
74       accumulate_error: bool = False
75       traverse_method: str = "sequential"
76       find_all: bool = False
77       return_intermediate: bool = False
78       all_outputs: bool = False
79   
80       def __str__(self):
81           settings_str = "FX Minimizer Settings:\n"
82   
83           for k, v in vars(self).items():
84               settings_str += f"\t{k}: {v}\n"
85   
86           return settings_str

89   class _MinimizerBase:
90       """ 
91       This class is used to automatically find problematic nodes in a model. It takes a FX 
92       graphmodule and generate some submodules while traverse the graph. Then two functions 
93       `run_a` and `run_b` will be used to run the same submodule and a function `compare_fn` 
94       will be used to compare the results. 
95    
96       Currently we provides two ways to traverse the graph and generate submodules. 
97           1. Sequential traversal: this will traverse the graph node by node and generate 
98              one submodule with one sigle node. 
99           2. Binary searching: this will do a binary search style traversal on the graph. 
100   
101      For internal Users, a guide can be found here https://fb.quip.com/HDtuAgiKGfkP. 
102      """
103  
104      def __init__(
105          self,
106          module: torch.fx.GraphModule,
107          sample_input: Tensors,
108          compare_fn: Callable[
109              [TensorOrTensors, TensorOrTensors, Names], tuple[float, bool]
110          ],
111          settings: _MinimizerSettingBase,
112          module_exporter: Optional[
113              Callable[[Tensors, torch.fx.GraphModule, str], None]
114          ] = None,
115          exclusion_fn: Optional[Callable[[NodeList, int, int], None]] = None,
116      ):
117          assert isinstance(module, torch.fx.GraphModule)
118  
119          self.module = module
120          self.sample_input = sample_input
121          self.compare_fn = compare_fn
122          self.module_exporter = module_exporter
123          self.settings = settings
124          self.exclusion_fn = exclusion_fn
125  
126          # Stores outputs of run_a function
127          self.a_outputs: dict[str, Any] = {}
128  
129          # Stores outputs of run_b function
130          self.b_outputs: dict[str, Any] = {}
131  
132          # Stores the results of compare_fn
133          self.results: dict[Any, Any] = {}
134  
135          # Stores the report for the runs
136          self.reports: list[list[str]] = []
137  
138          # Current iteration
139          self.iteration: int = 0
140  
141          callable_nodes = {
142              node for node in self.module.graph.nodes if node.op in CALLABLE_NODE_OPS
143          }
144          self.run_shape_prop()
145          self.fusions = FxNetAccFusionsFinder(self.module, callable_nodes)()
146  
147          # Check if number of input in sample_input matches the number of placeholders
148          placeholders = [
149              node.name for node in self.module.graph.nodes if node.op == "placeholder"
150          ]
151          assert len(placeholders) == len(self.sample_input)
152  
153          # Store sample_input
154          for i, name in enumerate(placeholders):
155              self.a_outputs[name] = sample_input[i]
156              self.b_outputs[name] = sample_input[i]

183      def _store_outputs(
184          self,
185          a_result: TensorOrTensors,
186          b_result: TensorOrTensors,
187          submodule: torch.fx.GraphModule,
188      ):
189          """ 
190          Store the outputs of self.run_a() and self.run_b() into self.a_outputs and 
191          self.b_outputs, so that we can use them when execute preceding nodes that 
192          use those outputs as inputs. 
193   
194          Args: 
195              a_result: Output of self.run_a(). Could be a tensor or tensors. 
196              b_result: Output of self.run_b(). Could be a tensor or tensors. 
197              submodule: The module that generates a_result and b_result. 
198          """
199          output_node = next(
200              node for node in submodule.graph.nodes if node.op == "output"
201          )
202  
203          # Only one output
204          if isinstance(output_node.args[0], torch.fx.Node):
205              self.a_outputs[output_node.args[0].name] = a_result
206              self.b_outputs[output_node.args[0].name] = b_result
207          # Multiple outputs
208          else:
209              for i, arg in enumerate(output_node.args[0]):
210                  self.a_outputs[arg.name] = a_result[i]
211                  self.b_outputs[arg.name] = b_result[i]

213      def _get_submod_inputs(
214          self, main_module: torch.fx.GraphModule, submod_path: str
215      ) -> tuple[Tensors, Tensors]:
216          """ 
217          Try get submodule inputs from stored outputs. If not found then use 
218          torch_glow.get_submod_inputs to get the inputs. 
219   
220          If accumulate_error is False, use a_input for run_a() and run_b() 
221          otherwise use a_input for run_a and b_input for run_b. 
222   
223          Args: 
224              main_module: Top-levlel fx module. 
225              submod_path: Path to the submodule we want to run and compare results. 
226   
227          Returns: 
228              a_input: List of tensor(s) that will be used by run_a() as submodule inputs. 
229              b_input: List of tensor(s) that will be used by run_b() as submodule inputs. 
230          """
231          a_input = []
232          b_input = []
233          submodule = getattr(main_module, submod_path)
234          placeholders = [
235              node.name for node in submodule.graph.nodes if node.op == "placeholder"
236          ]
237  
238          # If all placeholder can be found in stored outputs, use stored
239          # outputs as inputs. Otherwise, use `torch_glow.get_submod_inputs`
240          # to get the inputs.
241          if set(placeholders) <= self.a_outputs.keys():
242              for name in placeholders:
243                  a_input.append(self.a_outputs[name])
244                  b_input.append(self.b_outputs[name])
245          else:
246              if self.settings.accumulate_error:
247                  print(f"Can't find previous stored outputs named {placeholders}!")
248  
249              def get_inputs(self: torch.nn.Module, inputs: Any):
250                  nonlocal a_input
251                  a_input = inputs
252  
253              # Use forward hook to get the inputs to the submodule
254              handle = submodule.register_forward_pre_hook(get_inputs)
255              main_module(*self.sample_input)
256              handle.remove()
257  
258              b_input = a_input
259  
260          if not self.settings.accumulate_error:
261              return a_input, a_input
262  
263          return a_input, b_input

265      def _tag_nodes(self, selected_nodes: NodeSet):
266          """ 
267          Tag selected nodes with tag "minimize". Nodes with the same tags will 
268          be split to the same submodule afterwards. 
269   
270          Args: 
271              selected_nodes: Nodes that we want to minimize. We will tag those nodes 
272                  with "minimize", all preceding nodes with "main_0" and all following 
273                  nodes with "main_1". 
274          """
275          for node in self.module.graph.nodes:
276              if node.op not in CALLABLE_NODE_OPS:
277                  continue
278  
279              if node in selected_nodes:
280                  node.tag = "minimize"
281              elif any(
282                  n.tag in {"minimize", "main_1"}
283                  for n in node.all_input_nodes
284                  if n.op in CALLABLE_NODE_OPS
285              ):
286                  node.tag = "main_1"
287              else:
288                  node.tag = "main_0"

290      def _build_submodule(self, nodes: NodeSet) -> tuple[torch.fx.GraphModule, str]:
291          """ 
292          Split self.module so that one submodule consists of `nodes` and only `nodes`. 
293   
294          Args: 
295              nodes: Nodes that we want to include in the minimize submodule. 
296   
297          Returns: 
298              split_module (torch.fx.GraphModule): the module after split. 
299              submodule_name (str): the name of the submodule that consists of `nodes`. 
300          """
301          # Color provided nodes
302          self._tag_nodes(nodes)
303  
304          # Split module based on coloring
305          split_module = split_by_tags(self.module, ["main_0", "minimize", "main_1"])
306  
307          # Find submodule containing colored nodes
308          submodule_name: str = ""
309          for child_name, _ in split_module.named_children():  # type: ignore[union-attr]
310              # Skip submodules we're not interested in at the moment
311              if "minimize" not in child_name:
312                  continue
313  
314              if submodule_name == "":
315                  submodule_name = child_name
316              else:
317                  raise FxNetMinimizerBadModuleError(
318                      f"Expected only one minimize submodule with nodes {nodes}"
319                  )
320  
321          if submodule_name == "":
322              raise FxNetMinimizerBadModuleError(
323                  f"Minimize submodule was not found with nodes {nodes}"
324              )
325  
326          return split_module, submodule_name  # type: ignore[return-value]

328      def _run_and_compare(
329          self,
330          split_module: torch.fx.GraphModule,
331          submod_name: str,
332          output_names: Names,
333          report_idx: int = -1,
334      ):
335          """ 
336          Run the submodule in `split_module` that has name `submod_name` 
337          using `self.run_a` and `self.run_b` and compare their results. 
338   
339          Args: 
340              split_module: Main module that contains the minimize submodule. 
341              submod_name: Name of the minimize submodule. 
342              output_names: Names of the node we want to output. If None, we 
343                  will use the original output. 
344          """
345          submodule = getattr(split_module, submod_name)
346          a_input, b_input = self._get_submod_inputs(split_module, submod_name)
347  
348          if len(self.reports) == 0:
349              self.reports.append([])
350              self.iteration = 1
351  
352          report = self.reports[report_idx if report_idx >= 0 else self.iteration - 1]
353          report.append("Run and compare ...")
354  
355          if output_names and not self.settings.all_outputs:
356              output_nodes: NodeList = []
357              for node in submodule.graph.nodes:
358                  if node.op == "output":
359                      submodule.graph.erase_node(node)
360  
361                  if node.name in output_names:
362                      output_nodes.append(node)
363  
364              submodule.graph.output(
365                  output_nodes[0] if len(output_nodes) == 1 else tuple(output_nodes)
366              )
367              submodule.graph.lint()
368              submodule.recompile()
369  
370          # Use name of args in output node as key to store comparison result
371          for node in submodule.graph.nodes:
372              if node.op == "output":
373                  result_key = map_arg(node.args, lambda x: x.name)
374  
375          try:
376              a_result = self.run_a(submodule, a_input, report_idx)
377              b_result = self.run_b(submodule, b_input, report_idx)
378              self._store_outputs(a_result, b_result, submodule)
379          except Exception as e:
380              report.append(f"Exception raised when running {submod_name}: {e}")
381              raise FxNetMinimizerRunFuncError(  # noqa: B904
382                  f"Exception raised when running {submod_name}: {e}"
383              )
384  
385          # Compare results
386          names: Names = output_names
387          if output_names is None:
388              names = [str(v) for v in result_key]  # type: ignore[possibly-undefined]
389  
390          numeric_result, bool_result = self.compare_fn(a_result, b_result, names)
391  
392          self.results[result_key] = numeric_result  # type: ignore[possibly-undefined]
393          report.append(f"Numerical accuracy = {numeric_result}")
394          if not bool_result:
395              report.append(f"Result mismatch for {result_key}")  # type: ignore[possibly-undefined]
396              if self.module_exporter:
397                  if isinstance(result_key, tuple):  # type: ignore[possibly-undefined]
398                      result_key = result_key[-1]
399                  # If the result is still a tuple (happens in non-sequential mode),
400                  # we only use the first element as name.
401                  if isinstance(result_key, tuple):  # type: ignore[possibly-undefined]
402                      result_key = str(result_key[0])
403                  # pyre-ignore[29]: not a function
404                  self.module_exporter(
405                      a_input,
406                      submodule,
407                      result_key + "_cpu",
408                  )
409                  # pyre-ignore[29]: not a function
410                  self.module_exporter(
411                      b_input,
412                      submodule,
413                      result_key + "_acc",
414                  )
415              raise FxNetMinimizerResultMismatchError(f"Result mismatch for {result_key}")  # type: ignore[possibly-undefined]

417      def _binary_search_impl(
418          self, all_nodes: NodeList, start_idx: int, end_idx: int
419      ) -> NodeSet:
420          """ 
421          Recursive binary search implementation. 
422          """
423          culprits: NodeSet = set()
424          nodes: NodeList = all_nodes[start_idx:end_idx]
425  
426          report: list[str] = []
427          if self.exclusion_fn is not None:
428              self.exclusion_fn(nodes, start_idx, end_idx)
429              if len(nodes) == 0:
430                  report = ["All nodes are excluded by user"]
431                  self.reports.append(report)
432                  return culprits
433  
434          first_node_name = nodes[0].name
435          output_node_name = nodes[-1].name
436          self.iteration += 1
437          self.reports.append(report)
438          report.append(f"Binary search iteration {self.iteration}")
439          report.append(
440              f"From node index {start_idx}:{first_node_name} to {end_idx - 1}:{output_node_name}. "
441              f"Size of the interested node list is {len(nodes)}"
442          )
443          cur_nodes: NodeSet = set(nodes)
444  
445          try:
446              split_module, submod_name = self._build_submodule(cur_nodes)
447              self._run_and_compare(split_module, submod_name, [output_node_name])
448  
449          except (FxNetMinimizerRunFuncError, FxNetMinimizerResultMismatchError):
450              if len(nodes) == 1:
451                  report.append(
452                      f"This is the last node in the sub-module. "
453                      f"Search in the current branch is successful with culprit = {cur_nodes}."
454                  )
455                  self.print_report(report)
456                  return cur_nodes
457  
458              report.append(
459                  "Proceed to split and lower the halves of the current "
460                  "sub-module individually."
461              )
462              self.print_report(report)
463  
464              mid = len(nodes) // 2
465              culprits = self._binary_search_impl(all_nodes, start_idx, start_idx + mid)
466  
467              if len(culprits) != 0 and not self.settings.find_all:
468                  return culprits
469  
470              culprits = self._binary_search_impl(all_nodes, start_idx + mid, end_idx)
471  
472              if len(culprits) == 0:
473                  report.append(
474                      f"Further split and lowering found no errors. "
475                      f"Unable to minimize the submodule with list of nodes: {nodes}"
476                  )
477                  self.print_report(report)
478  
479              return culprits
480          else:
481              report.append("No discrepancy found.")
482              self.print_report(report)
483              return set()
484  
485      def _binary_traverse(self, nodes: NodeList) -> NodeSet:
486          """ 
487          Binary search on `nodes` for culprit. 
488          """
489          return self._binary_search_impl(nodes, 0, len(nodes))

491      def _sequential_traverse(self, nodes: NodeList) -> NodeSet:
492          """ 
493          Traverse `nodes` one by one and determine if any of them is a culprit. 
494          """
495          culprits: NodeSet = set()
496  
497          for node in nodes:
498              report: list[str] = []
499              self.reports.append(report)
500              self.iteration += 1
501              report.append(f"Sequential traverse iteration {self.iteration}.")
502              report.append(f"Visit node: {node.name}")
503  
504              _LOGGER.info("Visit node: %s", node.name)
505              node_list: NodeList = [node]
506              if self.exclusion_fn is not None:
507                  self.exclusion_fn(node_list, -1, -1)
508                  if len(node_list) == 0:
509                      report.append(f"User exclusion : {node.name}")
510                      self.print_report(report)
511                      if not self.settings.find_all:
512                          return culprits
513                      else:
514                          continue
515  
516              cur_nodes: NodeSet = {node}
517  
518              if node in self.fusions:
519                  cur_nodes = self.fusions[node]
520  
521              try:
522                  split_module, submod_name = self._build_submodule(cur_nodes)
523                  self._run_and_compare(split_module, submod_name, [node.name])
524                  self.print_report(report)
525              except FxNetMinimizerResultMismatchError:
526                  culprits.add(node)
527                  report.append(f"Found culprit from numeric error: {node}")
528                  self.print_report(report)
529                  if not self.settings.find_all:
530                      return culprits
531              except FxNetMinimizerRunFuncError:
532                  culprits.update(cur_nodes)
533                  report.append(f"Found culprit from run error: {node}")
534                  self.print_report(report)
535                  if not self.settings.find_all:
536                      return culprits
537  
538          return culprits

540      def _block_traverse_impl(
541          self, nodes: NodeList, start_idx: int, end_idx: int, find_last_node: bool
542      ) -> Optional[int]:
543          """ 
544          Recursive block search implementation. 
545          find_last_node: If True, search for the last node which result in numerics difference 
546          if False: find first node in sorted node list 
547          """
548          report: list[str] = []
549  
550          mid = (start_idx + end_idx) // 2
551          cur_nodes_list: NodeList = nodes[: mid + 1] if find_last_node else nodes[mid:]
552  
553          if self.exclusion_fn:
554              self.exclusion_fn(cur_nodes_list, -1, -1)
555  
556          cur_nodes = set(cur_nodes_list)
557  
558          first_node_name = cur_nodes_list[0].name
559          last_node_name = cur_nodes_list[-1].name
560          target_node_name = last_node_name if find_last_node else first_node_name
561  
562          self.iteration += 1
563          self.reports.append(report)
564          report.extend(
565              [
566                  "=" * 30,
567                  f"Block search iteration {self.iteration}",
568              ]
569          )
570          report.extend(
571              [
572                  f"Search for {'last' if find_last_node else 'first'} node in culprits",
573                  f"From node index {start_idx}:{nodes[start_idx].name} to {end_idx}:{nodes[end_idx].name}. ",
574                  f"Subgraph constructed by {first_node_name} to {last_node_name}",
575                  f"Targeting node: {target_node_name}",
576                  f"Size of the interested node list is {end_idx - start_idx + 1}",
577              ]
578          )
579          report_idx = len(self.reports) - 1
580  
581          try:
582              split_module, submod_name = self._build_submodule(cur_nodes)
583              self._run_and_compare(
584                  split_module, submod_name, [last_node_name], report_idx
585              )
586          except (FxNetMinimizerResultMismatchError, FxNetMinimizerRunFuncError):
587              report.append(
588                  f"Culprits found from node {first_node_name} to {last_node_name}."
589              )
590  
591              if start_idx == mid == end_idx:
592                  report.extend(
593                      [
594                          "This is the last node in the sub-module. ",
595                          "Search in the current branch is successful with node :",
596                          f"{start_idx}, node name: {nodes[start_idx].name}.",
597                      ]
598                  )
599                  self.print_report(report)
600                  return start_idx
601  
602              report.append(
603                  "Proceed to split and lower the halves of the current "
604                  "sub-module individually."
605              )
606              self.print_report(report)
607  
608              if find_last_node:
609                  return self._block_traverse_impl(nodes, start_idx, mid, find_last_node)
610              else:
611                  return self._block_traverse_impl(
612                      nodes, mid + 1, end_idx, find_last_node
613                  )
614          else:
615              report.append(
616                  f"Culprits not found from node start to {mid}:{nodes[mid].name}."
617              )
618  
619              if start_idx == mid == end_idx:
620                  # We did not find anything if the pointers have not moved
621                  if (start_idx == 0 and not find_last_node) or (
622                      start_idx == len(nodes) - 1 and find_last_node
623                  ):
624                      report.append(
625                          f"At {'last' if find_last_node else 'first'} node, no culprits found."
626                      )
627                      self.print_report(report)
628                      return None
629  
630                  # Otherwise, we have converged on the border between discrepancy and valid
631                  return start_idx + (1 if find_last_node else -1)
632  
633              report.append(
634                  "Proceed to split and lower the halves of the current "
635                  "sub-module individually."
636              )
637              self.print_report(report)
638  
639              if find_last_node:
640                  return self._block_traverse_impl(
641                      nodes, mid + 1, end_idx, find_last_node
642                  )
643              else:
644                  return self._block_traverse_impl(nodes, start_idx, mid, find_last_node)

646      def _block_traverse(
647          self, nodes: NodeList, find_last_node: Optional[bool]
648      ) -> NodeSet:
649          """ 
650          Traverse topologically sorted node list 
651          Find minimium block (start_idx, end_idx) which contains the culprit 
652          1st pass: search for end_idx by finding the last node in culprit block 
653          where Numerical accuracy (0, end_idx) > threshold 
654          2nd pass: search for start_idx by finding the first node in culprit block 
655          where Numerical accuracy (start_idx, end_idx) < threshold 
656          Form minimum block by (start_idx - 1, end_idx) 
657          """
658          culprits: NodeSet = set()
659          first_node_name = nodes[0].name
660          last_node_name = nodes[-1].name
661          last_node_report = [f"Block search from {first_node_name} to {last_node_name}"]
662          last_node_report.append("*" * 50)
663          self.reports.append(last_node_report)
664  
665          start_idx = 0
666          end_idx = len(nodes) - 1
667  
668          final_start_idx: Optional[int] = start_idx
669          final_end_idx: Optional[int] = end_idx
670  
671          run_both = True if find_last_node is None else False
672  
673          # step 1: find (0, end_idx) of culprit block
674          if run_both or find_last_node:
675              last_node_report.append("Start searching for last node in culprit")
676              self.print_report(last_node_report)
677              final_end_idx = self._block_traverse_impl(nodes, start_idx, end_idx, True)
678  
679              if final_end_idx is None:
680                  last_node_report.append("No culprits found")
681                  self.print_report(last_node_report)
682                  return culprits
683  
684              last_node_report.extend(
685                  [
686                      "Finish Pass 1",
687                      f"Find end_idx = {final_end_idx}:{nodes[final_end_idx].name}",
688                  ]
689              )
690              self.print_report(last_node_report)
691  
692          # step 2: reduce culprit block to (start_idx, end_idx)
693          if run_both or not find_last_node:
694              first_node_report = ["Start searching for first node in culprit"]
695              self.print_report(first_node_report)
696              final_start_idx = self._block_traverse_impl(
697                  nodes[0 : end_idx + 1], start_idx, final_end_idx or end_idx, False
698              )
699  
700              if final_start_idx is None:
701                  last_node_report.append("No culprits found")
702                  self.print_report(last_node_report)
703                  return culprits
704  
705              first_node_report.append("*" * 50)
706              self.reports.append(first_node_report)
707              first_node_report.extend(
708                  [
709                      "Finish Pass 2",
710                      f"Find start_idx = {final_start_idx}:{nodes[final_start_idx].name}",
711                  ]
712              )
713              self.print_report(first_node_report)
714  
715          # step 3: form module with minimum culprits. These indexes are guaranteed to exist
716          range_start, range_end = cast(int, final_start_idx), cast(int, final_end_idx)
717          culprits.update(nodes[range_start : range_end + 1])
718          result_report = [
719              f"Finish searching, found minimum block ({nodes[range_start]},{nodes[range_end]})"
720          ]
721          self.reports.append(result_report)
722          self.print_report(result_report)
723          return culprits

725      def _defined_traverse(self, nodes: NodeList) -> NodeSet:
726          """ 
727          run user defined `nodes` and determine if it is a culprit. 
728          """
729          culprits: NodeSet = set()
730          if self.exclusion_fn is not None:
731              self.exclusion_fn(nodes, -1, -1)
732          if len(nodes) == 0:
733              report = ["All nodes are excluded by user"]
734              self.reports.append(report)
735              return culprits
736  
737          first_node_name = nodes[0].name
738          output_node_name = nodes[-1].name
739          report = [f"Defined graph from {first_node_name} to {output_node_name}"]
740          cur_nodes: NodeSet = set(nodes)
741          try:
742              split_module, submod_name = self._build_submodule(cur_nodes)
743              self._run_and_compare(split_module, submod_name, [output_node_name])
744              self.print_report(report)
745          except (FxNetMinimizerResultMismatchError, FxNetMinimizerRunFuncError):
746              report.append(f"Found culprit {cur_nodes}")
747              self.print_report(report)
748              return culprits
749  
750          return culprits

752      def _accumulate_traverse(self, nodes: NodeList) -> NodeSet:
753          culprits: NodeSet = set()
754          nodes_to_run: NodeSet = set()
755  
756          # find_all is not supported for accumulate traversal because all the
757          # ops run on NNPI. So we return after the first op that raises error.
758          if self.settings.find_all:
759              print("'Find All' mode is not supported in accumulate traversal.")
760              return culprits
761  
762          for node in nodes:
763              report: list[str] = []
764              self.reports.append(report)
765              self.iteration += 1
766              report.append(f"Accumulate traverse iteration {self.iteration}.")
767  
768              nodes_to_run.add(node)
769  
770              node_name = node.name
771              if node_name is not None and isinstance(node_name, tuple):
772                  node_name = node_name[0]
773              assert node_name is not None and isinstance(node_name, str), (
774                  f"minimize: node_name: {node_name}"
775              )
776  
777              report.append(f"Add node: {node_name}")
778  
779              try:
780                  split_module, submod_name = self._build_submodule(nodes_to_run)
781                  self._run_and_compare(split_module, submod_name, [node_name])
782                  self.print_report(report)
783              except (FxNetMinimizerResultMismatchError, FxNetMinimizerRunFuncError):
784                  culprits.add(node)
785                  report.append(f"Found culprit {node}")
786                  self.print_report(report)
787                  return culprits
788  
789          return culprits

791      def _skip_traverse_impl(
792          self, all_nodes: NodeList, start_idx: int, end_idx: int
793      ) -> NodeSet:
794          """ 
795          Skip certain nodes in graph based on settings 
796          """
797          culprits: NodeSet = set()
798          nodes: NodeList = all_nodes[start_idx:end_idx]
799          cur_nodes: NodeSet = set(nodes)
800          if self.exclusion_fn is not None:
801              self.exclusion_fn(nodes, start_idx, end_idx)
802              cur_nodes = set(nodes)
803          else:
804              for node in nodes:
805                  if node in self.fusions:
806                      cur_nodes.update(self.fusions[node])
807          report: list[str] = []
808          self.reports.append(report)
809          self.iteration += 1
810          report.append(f" Nodes block {self.iteration}.")
811          report.append(
812              f"From node index {start_idx} to {end_idx - 1}. "
813              f"Size of the interested node list is {len(nodes)}"
814          )
815  
816          try:
817              split_module, submod_name = self._build_submodule(cur_nodes)
818              self._run_and_compare(split_module, submod_name, [])
819          except FxNetMinimizerResultMismatchError:
820              culprits.update(cur_nodes)
821              report.append(f"Found culprit from numeric error: {cur_nodes}")
822              self.print_report(report)
823              return culprits
824          except FxNetMinimizerRunFuncError:
825              culprits.update(cur_nodes)
826              report.append(f"Found culprit from run error: {cur_nodes}")
827              self.print_report(report)
828              return culprits
829          else:
830              report.append("No discrepancy found.")
831              self.print_report(report)
832              return set()
833  
834      def _skip_traverse(self, all_nodes: NodeList, skip_nodes: list) -> NodeSet:
835          """ 
836          Skip certain nodes in graph based on settings 
837          """
838          start_idx = 0
839          num_nodes = len(all_nodes)
840          idx = 0
841          culprits = set()
842          while idx < num_nodes:
843              node = all_nodes[idx]
844              if node.name in skip_nodes:  # skip the node
845                  if idx > start_idx:
846                      culprits = self._skip_traverse_impl(all_nodes, start_idx, idx)
847                  start_idx = idx + 1
848              elif idx == num_nodes - 1 and start_idx <= idx:  # last node
849                  culprits = self._skip_traverse_impl(all_nodes, start_idx, idx + 1)
850              idx += 1
851  
852          return culprits

854      def _collect_nodes(self, start: Optional[str], end: Optional[str]) -> NodeList:
855          """ 
856          Collect nodes in the model that between nodes with name of `start` and `end`. 
857          These two nodes are also included. 
858          """
859          nodes: NodeList = []
860          add_node = start is None
861  
862          for node in self.module.graph.nodes:
863              if node.op not in CALLABLE_NODE_OPS:
864                  continue
865  
866              if node.name == start:
867                  add_node = True
868  
869              if add_node:
870                  nodes.append(node)
871  
872              if node.name == end:
873                  break
874  
875          return nodes

877      def run_nodes(self, start: Optional[str] = None, end: Optional[str] = None):
878          """ 
879          Run part of the model from `start` node to `end` node. If `start` is None 
880          then we start from the beginning of the model. If `end` is None then we 
881          stop at the end of the model. 
882   
883          Args: 
884              start: The name of the node which is the first node of the submodule 
885                  we want to run. If set to None, then we'll start with the first 
886                  node of the model. 
887              end: The name of the node which is the last node of the submodule we 
888                  want to run. If set to None, we'll end with the last node of the 
889                  model. 
890          """
891          nodes = self._collect_nodes(start, end)
892          cur_nodes = set(nodes)
893  
894          for node in nodes:
895              if node in self.fusions:
896                  cur_nodes.update(self.fusions[node])
897  
898          output_names = []
899          if self.settings.return_intermediate:
900              output_names = [node.name for node in nodes]
901  
902          try:
903              split_module, submod_name = self._build_submodule(cur_nodes)
904              self._run_and_compare(split_module, submod_name, output_names)
905          except (
906              FxNetMinimizerRunFuncError,
907              FxNetMinimizerResultMismatchError,
908          ) as e:
909              print(e)

922      def minimize(
923          self,
924          start: Optional[str] = None,
925          end: Optional[str] = None,
926          skip_nodes: Optional[list] = None,
927          find_last_node: Optional[bool] = None,
928      ) -> NodeSet:
929          """ 
930          Minimizing the model from node with name `start` to node with name `end` base 
931          on self.settings. Find culprits that causes FxNetMinimizerRunFuncError or 
932          FxNetMinimizerResultMismatchError errors. 
933   
934          Args: 
935              start: The name of the node where we want to start minimizing. If set 
936                  to None, then we'll start with the first node of the model. 
937              end: The name of the node where we want to terminate minimizing. If 
938                  set to None, we'll end with the last node of the model. 
939              skip_nodes: The names of nodes where we want to skip during minimizing. 
940                  It'll create subgraphs without these skip nodes under the hood. 
941                  Only applicable in mode "skip". 
942              find_last_node: True if only last_node of a culprits is needed in mode "block". 
943                  False if only the first_node of a culprits is needed. 
944                  Only applicable in mode "block". 
945   
946          Returns: 
947              nodes: A list of nodes that causes FxNetMinimizerRunFuncError or 
948                  FxNetMinimizerResultMismatchError errors during minimizing. 
949          """
950  
951          print(self.settings)
952          print(self.module.graph)
953  
954          nodes = self._collect_nodes(start, end)
955  
956          if self.settings.traverse_method == "sequential":
957              return self._sequential_traverse(nodes)
958  
959          if self.settings.traverse_method == "binary":
960              return self._binary_traverse(nodes)
961  
962          if self.settings.traverse_method == "accumulate":
963              return self._accumulate_traverse(nodes)
964  
965          if self.settings.traverse_method == "skip":
966              if skip_nodes is None:
967                  raise RuntimeError(
968                      "'skip_nodes' can't be None when 'traverse_method' is 'skip'."
969                  )
970              return self._skip_traverse(nodes, skip_nodes)
971  
972          if self.settings.traverse_method == "defined":
973              return self._defined_traverse(nodes)
974  
975          if self.settings.traverse_method == "block":
976              return self._block_traverse(nodes, find_last_node)
977  
978          raise RuntimeError(f"Unknown traverse method {self.settings.traverse_method}!")
