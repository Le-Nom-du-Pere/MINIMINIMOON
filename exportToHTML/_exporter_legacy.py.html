49   @dataclasses.dataclass
50   class ONNXFakeContext:
51       """A dataclass used to store context for model export using FakeTensor. 
52    
53       This dataclass stores the FakeTensorMode instance used to convert 
54       real tensors and model parameters into fake tensors. This :attr:`ONNXFakeContext.fake_mode` is 
55       reused internally during tracing of a :class:`torch.nn.Module` into a FX :class:`GraphModule`. 
56       """
57   
58       fake_mode: fake_tensor.FakeTensorMode
59       """The fake tensor mode used for tracing model using fake tensors and parameters."""
60   
61       state_dict_paths: tuple[str | io.BytesIO | dict[str, Any]] | None = None
62       """List of paths of files that contain the model :meth:`state_dict`"""

65   @deprecated(
66       "torch.onnx.dynamo_export is deprecated since 2.7.0. Please use torch.onnx.export(..., dynamo=True) instead.",
67   )
68   class OnnxRegistry:
69       """Registry for ONNX functions. 
70    
71       .. deprecated:: 2.7 
72           Please use ``torch.onnx.export(..., dynamo=True)`` instead. 
73    
74       The registry maintains a mapping from qualified names to symbolic functions under a 
75       fixed opset version. It supports registering custom onnx-script functions and for 
76       dispatcher to dispatch calls to the appropriate function. 
77    
78       """
79   
80       def __init__(self) -> None:
81           """Initializes the registry"""
82   
83           # NOTE: _registry is the registry maps OpNameto a list of ONNXFunctions. It is important
84           # not to directly modify this variable. Instead, access to it should be done through
85           # the public methods: register_custom_op, get_ops, and is_registered_op.
86           self._registry: dict[registration.OpName, list[registration.ONNXFunction]] = (
87               defaultdict(list)
88           )
89   
90           self._opset_version = _constants.TORCHLIB_OPSET
91           warnings.warn(
92               f"torch.onnx.dynamo_export only implements opset version {self._opset_version} for now. If you need to use a "
93               "different opset version, please register them with register_custom_op."
94           )
95   
96           self._initiate_registry_from_torchlib()
97   
98       @property
99       def opset_version(self) -> int:
100          """The ONNX opset version the exporter should target."""
101  
102          return self._opset_version
103  
104      def _initiate_registry_from_torchlib(self) -> None:
105          """Populates the registry with ATen functions from torchlib. 
106   
107          Args: 
108              torchlib_registry: The torchlib registry to use for populating the registry. 
109          """
110          for meta in onnxscript_apis.get_torchlib_ops():
111              internal_name_instance = registration.OpName.from_qualified_name(
112                  meta.qualified_name
113              )
114              symbolic_function = registration.ONNXFunction(
115                  onnx_function=meta.function,  # type: ignore[arg-type]
116                  op_full_name=internal_name_instance.qualified_name(),
117                  is_custom=False,
118                  is_complex=meta.is_complex,
119              )
120              self._register(internal_name_instance, symbolic_function)
121  
122      def _register(
123          self,
124          internal_qualified_name: registration.OpName,
125          symbolic_function: registration.ONNXFunction,
126      ) -> None:
127          """Registers a ONNXFunction to an operator. 
128   
129          Args: 
130              internal_qualified_name: The qualified name of the operator to register: OpName. 
131              symbolic_function: The ONNXFunction to register. 
132          """
133          self._registry[internal_qualified_name].append(symbolic_function)
134  
135      def register_op(
136          self,
137          function: onnxscript.OnnxFunction | onnxscript.TracedOnnxFunction,
138          namespace: str,
139          op_name: str,
140          overload: str | None = None,
141          is_complex: bool = False,
142      ) -> None:
143          """Registers a custom operator: torch.ops.<namespace>.<op_name>.<overload>. 
144   
145          Args: 
146              function: The onnx-sctip function to register. 
147              namespace: The namespace of the operator to register. 
148              op_name: The name of the operator to register. 
149              overload: The overload of the operator to register. If it's default overload, 
150                  leave it to None. 
151              is_complex: Whether the function is a function that handles complex valued inputs. 
152   
153          Raises: 
154              ValueError: If the name is not in the form of 'namespace::op'. 
155          """
156          internal_name_instance = registration.OpName.from_name_parts(
157              namespace=namespace, op_name=op_name, overload=overload
158          )
159          symbolic_function = registration.ONNXFunction(
160              onnx_function=function,
161              op_full_name=internal_name_instance.qualified_name(),
162              is_custom=True,
163              is_complex=is_complex,
164          )
165          self._register(internal_name_instance, symbolic_function)
166  
167      def get_op_functions(
168          self, namespace: str, op_name: str, overload: str | None = None
169      ) -> list[registration.ONNXFunction] | None:
170          """Returns a list of ONNXFunctions for the given op: torch.ops.<namespace>.<op_name>.<overload>. 
171   
172          The list is ordered by the time of registration. The custom operators should be 
173          in the second half of the list. 
174   
175          Args: 
176              namespace: The namespace of the operator to get. 
177              op_name: The name of the operator to get. 
178              overload: The overload of the operator to get. If it's default overload, 
179                  leave it to None. 
180          Returns: 
181              A list of ONNXFunctions corresponding to the given name, or None if 
182              the name is not in the registry. 
183          """
184          internal_name_instance = registration.OpName.from_name_parts(
185              namespace=namespace, op_name=op_name, overload=overload
186          )
187          return self._registry.get(internal_name_instance)
188  
189      def is_registered_op(
190          self, namespace: str, op_name: str, overload: str | None = None
191      ) -> bool:
192          """Returns whether the given op is registered: torch.ops.<namespace>.<op_name>.<overload>. 
193   
194          Args: 
195              namespace: The namespace of the operator to check. 
196              op_name: The name of the operator to check. 
197              overload: The overload of the operator to check. If it's default overload, 
198                  leave it to None. 
199   
200          Returns: 
201              True if the given op is registered, otherwise False. 
202          """
203          functions = self.get_op_functions(
204              namespace=namespace, op_name=op_name, overload=overload
205          )
206          return functions is not None
207  
208      def _all_registered_ops(self) -> set[str]:
209          """Returns the set of all registered function names."""
210          return {
211              op_name_class.qualified_name() for op_name_class in self._registry.keys()
212          }

278  @contextlib.contextmanager
279  def enable_fake_mode():
280      """Enable fake mode for the duration of the context. 
281   
282      Internally it instantiates a :class:`torch._subclasses.fake_tensor.FakeTensorMode` context manager 
283      that converts user input and model parameters into :class:`torch._subclasses.fake_tensor.FakeTensor`. 
284   
285      A :class:`torch._subclasses.fake_tensor.FakeTensor` 
286      is a :class:`torch.Tensor` with the ability to run PyTorch code without having to 
287      actually do computation through tensors allocated on a ``meta`` device. Because 
288      there is no actual data being allocated on the device, this API allows for 
289      initializing and exporting large models without the actual memory footprint needed for executing it. 
290   
291      It is highly recommended to initialize the model in fake mode when exporting models that 
292      are too large to fit into memory. 
293   
294      .. note:: 
295          This function does not support torch.onnx.export(..., dynamo=True, optimize=True). 
296          Please call ONNXProgram.optimize() outside of the function after the model is exported. 
297   
298      Example:: 
299   
300          # xdoctest: +REQUIRES(env:TORCH_DOCTEST_ONNX) 
301          >>> import torch 
302          >>> class MyModel(torch.nn.Module):  # Model with a parameter 
303          ...     def __init__(self) -> None: 
304          ...         super().__init__() 
305          ...         self.weight = torch.nn.Parameter(torch.tensor(42.0)) 
306          ...     def forward(self, x): 
307          ...         return self.weight + x 
308          >>> with torch.onnx.enable_fake_mode(): 
309          ...     # When initialized in fake mode, the model's parameters are fake tensors 
310          ...     # They do not take up memory so we can initialize large models 
311          ...     my_nn_module = MyModel() 
312          ...     arg1 = torch.randn(2, 2, 2) 
313          >>> onnx_program = torch.onnx.export(my_nn_module, (arg1,), dynamo=True, optimize=False) 
314          >>> # Saving model WITHOUT initializers (only the architecture) 
315          >>> onnx_program.save( 
316          ...     "my_model_without_initializers.onnx", 
317          ...     include_initializers=False, 
318          ...     keep_initializers_as_inputs=True, 
319          ... ) 
320          >>> # Saving model WITH initializers after applying concrete weights 
321          >>> onnx_program.apply_weights({"weight": torch.tensor(42.0)}) 
322          >>> onnx_program.save("my_model_with_initializers.onnx") 
323   
324      .. warning:: 
325          This API is experimental and is *NOT* backward-compatible. 
326   
327      """
328      from torch._subclasses import fake_tensor
329      from torch.fx.experimental.symbolic_shapes import ShapeEnv
330  
331      # This overrides the internal `FakeTensorMode` instance created by `torch._dynamo.export`[1].
332      # It is a good idea to keep them in sync (constructor args) to maintain the same default behavior
333      # [1] `torch/_dynamo/output_graph.py::InstructionTranslator::OutputGraph.__init__`
334      # Mixed fake/real tensors are only allowed when `torch.onnx.dynamo_export` is not called within `FakeTensorMode`
335      # This is needed because models can create new parameters during `forward(self, *args, **kwargs)` run
336      fake_mode = fake_tensor.FakeTensorMode(
337          allow_non_fake_inputs=not torch._guards.detect_fake_mode(),
338          shape_env=ShapeEnv(
339              allow_scalar_outputs=False, allow_dynamic_output_shape_ops=False
340          ),
341      )
342      # The patcher is needed for when user calls `fake_model.load_state_dict(...)` within fake mode
343      patcher_context = patcher.ONNXTorchPatcher()
344      fake_context = ONNXFakeContext(fake_mode=fake_mode)
345      with fake_mode, patcher_context:
346          yield fake_context
347      fake_context.state_dict_paths = tuple(
348          patcher_context.paths,
349      )  # type: ignore[assignment]

436  def common_pre_export_passes(
437      options: ResolvedExportOptions,
438      original_model: torch.nn.Module | Callable,
439      fx_module: torch.fx.GraphModule,
440      fx_module_args: Sequence[Any],
441  ):
442      # TODO: Import here to prevent circular dependency
443      from torch.onnx._internal.fx import passes
444  
445      # Apply decomposition table to the input graph.
446      module = passes.Decompose(
447          fx_module,
448          options.decomposition_table,  # type: ignore[arg-type]
449          enable_dynamic_axes=options.dynamic_shapes,
450          allow_fake_constant=options.fake_context is not None,
451      ).run(*fx_module_args)
452  
453      # ONNX does not support views and mutations.
454      # Functionalize to get a semantically equivalent graph without mutations.
455      module = passes.Functionalize(
456          module,
457          enable_dynamic_axes=options.dynamic_shapes,
458          allow_fake_constant=options.fake_context is not None,
459      ).run(*fx_module_args)
460  
461      # Input mutations are detected and distilled after `Functionalize` pass.
462      # Remove them since ONNX inference does not need them.
463      module = passes.RemoveInputMutation(module).run(*fx_module_args)
464  
465      # ONNX does not support concept of (implicit) type promotion.
466      # Insert type casts explicitly where needed.
467      module = passes.InsertTypePromotion(module).run()
468  
469      if isinstance(original_model, torch.nn.Module):
470          module = passes.RestoreParameterAndBufferNames(module, original_model).run()
471  
472      # ONNX does not support None inputs. During graph building, all None inputs
473      # are removed. Here we register this step to input adapter.
474      options.fx_tracer.input_adapter.append_step(io_adapter.RemoveNoneInputStep())
475  
476      # NOTE: temp workaround for https://github.com/pytorch/pytorch/issues/99534
477      # Dynamo doesn't support non-tensor inputs.
478      options.fx_tracer.input_adapter.append_step(io_adapter.RemoveNonTensorInputStep())
479  
480      # ONNX does not support complex inputs. During graph building, all complex inputs
481      # are converted to real representation inputs. Here we register this step to
482      # input/output adapter.
483      options.fx_tracer.input_adapter.append_step(
484          io_adapter.ConvertComplexToRealRepresentationInputStep()
485      )
486  
487      # ONNX can't represent collection types (e.g., dictionary, tuple of tuple of
488      # tensor, etc), we flatten the collection and register each element as output.
489      options.fx_tracer.output_adapter.append_step(io_adapter.FlattenOutputStep())
490  
491      # Output post-processing steps should happen after `FlattenOutputStep`.
492      options.fx_tracer.output_adapter.append_step(
493          io_adapter.ConvertComplexToRealRepresentationOutputStep()
494      )
495  
496      return module
