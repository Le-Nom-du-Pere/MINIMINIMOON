1    """Metrics collection and management system for Dynamo. 
2     
3    This module provides context managers for gathering and reporting metrics during 
4    compilation and runtime. 
5     
6    It includes two main components: 
7    - MetricsContext: A context manager for collecting metrics during compilation, supporting 
8      nested contexts and various metric types (counters, sets, key-value pairs) 
9    - RuntimeMetricsContext: A specialized context for runtime metrics collection that doesn't 
10     require explicit context management 
11    
12   The metrics system enables comprehensive monitoring and analysis of both compilation and 
13   execution performance. 
14   """
15   
16   import heapq
17   import logging
18   import time
19   from collections.abc import Iterator
20   from typing import Any, Callable, Optional
21   from typing_extensions import TypeAlias
22   
23   
24   log = logging.getLogger(__name__)
25   
26   
27   class TopN:
28       """ 
29       Helper to record a list of metrics, keeping only the top N "most expensive" elements. 
30       """
31   
32       def __init__(self, at_most: int = 25):
33           self.at_most = at_most
34           self.heap: list[tuple[int, Any]] = []
35   
36       def add(self, key: Any, val: int) -> None:
37           # Push if we haven't reached the max size, else push and pop the smallest
38           fn = heapq.heappush if len(self.heap) < self.at_most else heapq.heappushpop
39           fn(self.heap, (val, key))
40   
41       def __len__(self) -> int:
42           return len(self.heap)
43   
44       def __iter__(self) -> Iterator[tuple[Any, int]]:
45           return ((key, val) for val, key in sorted(self.heap, reverse=True))

48   OnExitType: TypeAlias = Callable[
49       [int, int, dict[str, Any], Optional[type[BaseException]], Optional[BaseException]],
50       None,
51   ]
52   
53   
54   class MetricsContext:
55       def __init__(self, on_exit: OnExitType):
56           """ 
57           Use this class as a contextmanager to create a context under which to accumulate 
58           a set of metrics, e.g., metrics gathered during a compilation. On exit of the 
59           contextmanager, call the provided 'on_exit' function and pass a dictionary of 
60           all metrics set during the lifetime of the contextmanager. 
61           """
62           self._on_exit = on_exit
63           self._metrics: dict[str, Any] = {}
64           self._start_time_ns: int = 0
65           self._level: int = 0
66   
67       def __enter__(self) -> "MetricsContext":
68           """ 
69           Initialize metrics recording. 
70           """
71           if self._level == 0:
72               # In case of recursion, track at the outermost context.
73               self._metrics = {}
74               self._start_time_ns = time.time_ns()
75   
76           self._level += 1
77           return self
78   
79       def __exit__(
80           self,
81           exc_type: Optional[type[BaseException]],
82           exc_value: Optional[BaseException],
83           _traceback: Any,
84       ) -> None:
85           """ 
86           At exit, call the provided on_exit function. 
87           """
88           self._level -= 1
89           assert self._level >= 0
90           if self._level == 0:
91               try:
92                   end_time_ns = time.time_ns()
93                   self._on_exit(
94                       self._start_time_ns, end_time_ns, self._metrics, exc_type, exc_value
95                   )
96               except Exception:
97                   log.exception("Unexpected exception logging compilation metrics")
98   
99       def in_progress(self) -> bool:
100          """ 
101          True if we've entered the context. 
102          """
103          return self._level > 0
104  
105      def increment(self, metric: str, value: int) -> None:
106          """ 
107          Increment a metric by a given amount. 
108          """
109          if self._level == 0:
110              raise RuntimeError(f"Cannot increment {metric} outside of a MetricsContext")
111          if metric not in self._metrics:
112              self._metrics[metric] = 0
113          self._metrics[metric] += value
114  
115      def set(self, metric: str, value: Any, overwrite: bool = False) -> None:
116          """ 
117          Set a metric to a given value. Raises if the metric has been assigned previously 
118          in the current context. 
119          """
120          if self._level == 0:
121              raise RuntimeError(f"Cannot set {metric} outside of a MetricsContext")
122          if metric in self._metrics and not overwrite:
123              raise RuntimeError(
124                  f"Metric '{metric}' has already been set in the current context"
125              )
126          self._metrics[metric] = value
127  
128      def set_key_value(self, metric: str, key: str, value: Any) -> None:
129          """ 
130          Treats a give metric as a dictionary and set the k and value within it. 
131          Note that the metric must be a dictionary or not present. 
132   
133          We allow this to be called multiple times (i.e. for features, it's not uncommon 
134          for them to be used multiple times within a single compilation). 
135          """
136          if self._level == 0:
137              raise RuntimeError(f"Cannot set {metric} outside of a MetricsContext")
138          if metric not in self._metrics:
139              self._metrics[metric] = {}
140          self._metrics[metric][key] = value
141  
142      def update(self, values: dict[str, Any], overwrite: bool = False) -> None:
143          """ 
144          Set multiple metrics directly. This method does NOT increment. Raises if any 
145          metric has been assigned previously in the current context and overwrite is 
146          not set to True. 
147          """
148          if self._level == 0:
149              raise RuntimeError("Cannot update metrics outside of a MetricsContext")
150          existing = self._metrics.keys() & values.keys()
151          if existing and not overwrite:
152              raise RuntimeError(
153                  f"Metric(s) {existing} have already been set in the current context"
154              )
155          self._metrics.update(values)
156  
157      def update_outer(self, values: dict[str, Any]) -> None:
158          """ 
159          Update, but only when at the outermost context. 
160          """
161          if self._level == 0:
162              raise RuntimeError("Cannot update metrics outside of a MetricsContext")
163          if self._level == 1:
164              self.update(values)
165  
166      def add_to_set(self, metric: str, value: Any) -> None:
167          """ 
168          Records a metric as a set() of values. 
169          """
170          if self._level == 0:
171              raise RuntimeError(f"Cannot add {metric} outside of a MetricsContext")
172          if metric not in self._metrics:
173              self._metrics[metric] = set()
174          self._metrics[metric].add(value)
175  
176      def add_top_n(self, metric: str, key: Any, val: int) -> None:
177          """ 
178          Records a metric as a TopN set of values. 
179          """
180          if self._level == 0:
181              return
182          if metric not in self._metrics:
183              self._metrics[metric] = TopN()
184          self._metrics[metric].add(key, val)
185

187  class RuntimeMetricsContext:
188      def __init__(self, on_exit: OnExitType):
189          """ 
190          Similar to MetricsContext, but used to gather the runtime metrics that are 
191          decoupled from compilation, where there's not a natural place to insert a 
192          context manager. 
193          """
194          self._on_exit = on_exit
195          self._metrics: dict[str, Any] = {}
196          self._start_time_ns: int = 0
197  
198      def increment(
199          self, metric: str, value: int, extra: Optional[dict[str, Any]] = None
200      ) -> None:
201          """ 
202          Increment a metric by a given amount. 
203          """
204          if not self._metrics:
205              # Start timing on the first entry
206              self._start_time_ns = time.time_ns()
207          if metric not in self._metrics:
208              self._metrics[metric] = 0
209          self._metrics[metric] += value
210  
211          if extra:
212              for k, v in extra.items():
213                  if k not in self._metrics and v is not None:
214                      self._metrics[k] = v
215  
216      def finish(self) -> None:
217          """ 
218          Call the on_exit function with the metrics gathered so far and reset. 
219          """
220          if self._metrics:
221              try:
222                  end_time_ns = time.time_ns()
223                  self._on_exit(
224                      self._start_time_ns, end_time_ns, self._metrics, None, None
225                  )
226              except Exception:
227                  log.exception("Unexpected exception logging runtime metrics")
228              finally:
229                  self._metrics = {}
