8    @given(n=st.integers(1, 10**10))
9    def test_tau_hypothesis(n):
10       div = divisors(n)
11       tau_n = len(div)
12       assert is_square(n) == (tau_n % 2 == 1)
13       sigmas = [divisor_sigma(i) for i in div]
14       totients = [totient(n // i) for i in div]
15       mul = [a * b for a, b in zip(sigmas, totients)]
16       assert n * tau_n == sum(mul)

19   @given(n=st.integers(1, 10**10))
20   def test_totient_hypothesis(n):
21       assert totient(n) <= n
22       div = divisors(n)
23       totients = [totient(i) for i in div]
24       assert n == sum(totients)
