34   @typing_extensions.deprecated(DEPRECATION_WARNING)
35   def prepare_pt2e(
36       model: GraphModule,
37       quantizer: Quantizer,
38   ) -> GraphModule:
39       """Prepare a model for post training quantization 
40    
41       Args: 
42         * `model` (torch.fx.GraphModule): a model captured by `torch.export.export_for_training` API. 
43         * `quantizer`: A backend specific quantizer that conveys how user want the 
44           model to be quantized. Tutorial for how to write a quantizer can be found here: 
45           https://pytorch.org/tutorials/prototype/pt2e_quantizer.html 
46    
47       Return: 
48         A GraphModule with observer (based on quantizer annotation), ready for calibration 
49    
50       Example:: 
51    
52           import torch 
53           from torch.ao.quantization.quantize_pt2e import prepare_pt2e 
54           from torch.ao.quantization.quantizer import ( 
55               XNNPACKQuantizer, 
56               get_symmetric_quantization_config, 
57           ) 
58    
59           class M(torch.nn.Module): 
60               def __init__(self) -> None: 
61                   super().__init__() 
62                   self.linear = torch.nn.Linear(5, 10) 
63    
64              def forward(self, x): 
65                  return self.linear(x) 
66    
67           # initialize a floating point model 
68           float_model = M().eval() 
69    
70           # define calibration function 
71           def calibrate(model, data_loader): 
72               model.eval() 
73               with torch.no_grad(): 
74                   for image, target in data_loader: 
75                       model(image) 
76    
77           # Step 1. program capture 
78           # NOTE: this API will be updated to torch.export API in the future, but the captured 
79           # result shoud mostly stay the same 
80           m = torch.export.export_for_training(m, *example_inputs).module() 
81           # we get a model with aten ops 
82    
83           # Step 2. quantization 
84           # backend developer will write their own Quantizer and expose methods to allow 
85           # users to express how they 
86           # want the model to be quantized 
87           quantizer = XNNPACKQuantizer().set_global(get_symmetric_quantization_config()) 
88           m = prepare_pt2e(m, quantizer) 
89    
90           # run calibration 
91           # calibrate(m, sample_inference_data) 
92       """
93       torch._C._log_api_usage_once("quantization_api.quantize_pt2e.prepare_pt2e")
94       original_graph_meta = model.meta
95       node_name_to_scope = _get_node_name_to_scope(model)
96       # TODO: check qconfig_mapping to make sure conv and bn are both configured
97       # to be quantized before fusion
98       # TODO: (maybe) rewrite this with subgraph_rewriter
99       _fuse_conv_bn_(model)
100      model = quantizer.transform_for_annotation(model)
101      quantizer.annotate(model)
102      quantizer.validate(model)
103      model = prepare(
104          model,
105          node_name_to_scope,
106          is_qat=False,
107          obs_or_fq_callback=quantizer.prepare_obs_or_fq_callback,
108      )
109      model.meta.update(original_graph_meta)
110      model = _disallow_eval_train(model)
111      return model

114  @typing_extensions.deprecated(DEPRECATION_WARNING)
115  def prepare_qat_pt2e(
116      model: GraphModule,
117      quantizer: Quantizer,
118  ) -> GraphModule:
119      """Prepare a model for quantization aware training 
120   
121      Args: 
122        * `model` (torch.fx.GraphModule): see :func:`~torch.ao.quantization.quantize_pt2e.prepare_pt2e` 
123        * `quantizer`: see :func:`~torch.ao.quantization.quantize_pt2e.prepare_pt2e` 
124   
125      Return: 
126        A GraphModule with fake quant modules (based on quantizer annotation), ready for 
127        quantization aware training 
128   
129      Example:: 
130          import torch 
131          from torch.ao.quantization.quantize_pt2e import prepare_qat_pt2e 
132          from torch.ao.quantization.quantizer import ( 
133              XNNPACKQuantizer, 
134              get_symmetric_quantization_config, 
135          ) 
136   
137          class M(torch.nn.Module): 
138              def __init__(self) -> None: 
139                  super().__init__() 
140                  self.linear = torch.nn.Linear(5, 10) 
141   
142             def forward(self, x): 
143                 return self.linear(x) 
144   
145          # initialize a floating point model 
146          float_model = M().eval() 
147   
148          # define the training loop for quantization aware training 
149          def train_loop(model, train_data): 
150              model.train() 
151              for image, target in data_loader: 
152                  ... 
153   
154          # Step 1. program capture 
155          # NOTE: this API will be updated to torch.export API in the future, but the captured 
156          # result shoud mostly stay the same 
157          m = torch.export.export_for_training(m, *example_inputs).module() 
158          # we get a model with aten ops 
159   
160          # Step 2. quantization 
161          # backend developer will write their own Quantizer and expose methods to allow 
162          # users to express how they 
163          # want the model to be quantized 
164          quantizer = XNNPACKQuantizer().set_global(get_symmetric_quantization_config()) 
165          m = prepare_qat_pt2e(m, quantizer) 
166   
167          # run quantization aware training 
168          train_loop(prepared_model, train_loop) 
169   
170      """
171      torch._C._log_api_usage_once("quantization_api.quantize_pt2e.prepare_qat_pt2e")
172      original_graph_meta = model.meta
173      node_name_to_scope = _get_node_name_to_scope(model)
174      model = quantizer.transform_for_annotation(model)
175      quantizer.annotate(model)
176      quantizer.validate(model)
177      # Perform fusion after annotate to avoid quantizing ops in the new
178      # subgraph that don't need to be quantized
179      # TODO: only fuse if conv and bn are both configured to be quantized
180      _fuse_conv_bn_qat(model)
181      model = prepare(
182          model,
183          node_name_to_scope,
184          is_qat=True,
185          obs_or_fq_callback=quantizer.prepare_obs_or_fq_callback,
186      )
187      model.meta.update(original_graph_meta)
188      model = _disallow_eval_train(model)
189      return model

200  def _quant_node_constraint(n: Node) -> bool:
201      """If there is any pure ops between get_attr and quantize op they will be const propagated 
202      e.g. get_attr(weight) -> transpose -> quantize -> dequantize* 
203      (Note: dequantize op is not going to be constant propagated) 
204   
205      This filter is added because we don't want to constant fold the things that are not 
206      related to quantization 
207      """
208      return n.op == "call_function" and n.target in _QUANT_OPS

211  @typing_extensions.deprecated(DEPRECATION_WARNING)
212  def convert_pt2e(
213      model: GraphModule,
214      use_reference_representation: bool = False,
215      fold_quantize: bool = True,
216  ) -> GraphModule:
217      """Convert a calibrated/trained model to a quantized model 
218   
219      Args: 
220        * `model` (torch.fx.GraphModule): calibrated/trained model 
221        * `use_reference_representation` (bool): boolean flag to indicate whether to produce referece representation or not 
222        * `fold_quantize` (bool): boolean flag for whether fold the quantize op or not 
223   
224      Returns: 
225          quantized model, either in q/dq representation or reference representation 
226   
227      Example:: 
228   
229          # prepared_model: the model produced by `prepare_pt2e`/`prepare_qat_pt2e` and calibration/training 
230          # `convert_pt2e` produces a quantized model that represents quantized computation with 
231          # quantize dequantize ops and fp32 ops by default. 
232          # Please refer to 
233          # https://pytorch.org/tutorials/prototype/pt2e_quant_ptq_static.html#convert-the-calibrated-model-to-a-quantized-model 
234          # for detailed explanation of output quantized model 
235          quantized_model = convert_pt2e(prepared_model) 
236   
237      """
238      torch._C._log_api_usage_once("quantization_api.quantize_pt2e.convert_pt2e")
239      if not isinstance(use_reference_representation, bool):
240          raise ValueError(
241              "Unexpected argument type for `use_reference_representation`, "
242              f"please make sure you intend to pass argument {use_reference_representation} to convert_pt2e"
243          )
244      original_graph_meta = model.meta
245      model = _convert_to_reference_decomposed_fx(model)
246      model = _fold_conv_bn_qat(model)
247  
248      pm = PassManager([DuplicateDQPass()])
249      model = pm(model).graph_module
250  
251      pm = PassManager([PortNodeMetaForQDQ()])
252      model = pm(model).graph_module
253  
254      if fold_quantize:
255          constant_fold(model, _quant_node_constraint)
256  
257      if use_reference_representation:
258          model = reference_representation_rewrite(model)
259  
260      model.meta.update(original_graph_meta)
261      model = _disallow_eval_train(model)
262      return model
