15   def test_transpose():
16       Sq = MatrixSymbol('Sq', n, n)
17   
18       assert transpose(A) == Transpose(A)
19       assert Transpose(A).shape == (m, n)
20       assert Transpose(A*B).shape == (l, n)
21       assert transpose(Transpose(A)) == A
22       assert isinstance(Transpose(Transpose(A)), Transpose)
23   
24       assert adjoint(Transpose(A)) == Adjoint(Transpose(A))
25       assert conjugate(Transpose(A)) == Adjoint(A)
26   
27       assert Transpose(eye(3)).doit() == eye(3)
28   
29       assert Transpose(S(5)).doit() == S(5)
30   
31       assert Transpose(Matrix([[1, 2], [3, 4]])).doit() == Matrix([[1, 3], [2, 4]])
32   
33       assert transpose(trace(Sq)) == trace(Sq)
34       assert trace(Transpose(Sq)) == trace(Sq)
35   
36       assert Transpose(Sq)[0, 1] == Sq[1, 0]
37   
38       assert Transpose(A*B).doit() == Transpose(B) * Transpose(A)
