12   // The real implementation of MPSHooksInterface
13   struct MPSHooks : public at::MPSHooksInterface {
14     MPSHooks(at::MPSHooksArgs) {}
15     void init() const override;
16   
17     // MPSDevice interface
18     bool hasMPS() const override;
19     bool isOnMacOSorNewer(unsigned major, unsigned minor) const override;
20   
21     Device getDeviceFromPtr(void* data) const override;
22   
23     // MPSGeneratorImpl interface
24     const Generator& getDefaultGenerator(
25         DeviceIndex device_index = -1) const override;
26     Generator getNewGenerator(DeviceIndex device_index = -1) const override;
27   
28     // MPSStream interface
29     void deviceSynchronize() const override;
30     void commitStream() const override;
31     void* getCommandBuffer() const override;
32     void* getDispatchQueue() const override;
33   
34     // MPSAllocator interface
35     Allocator* getMPSDeviceAllocator() const override;
36     void emptyCache() const override;
37     size_t getCurrentAllocatedMemory() const override;
38     size_t getDriverAllocatedMemory() const override;
39     size_t getRecommendedMaxMemory() const override;
40     void setMemoryFraction(double ratio) const override;
41     bool isPinnedPtr(const void* data) const override;
42     Allocator* getPinnedMemoryAllocator() const override;
43   
44     // MPSProfiler interface
45     void profilerStartTrace(const std::string& mode, bool waitUntilCompleted)
46         const override;
47     void profilerStopTrace() const override;
48   
49     // MPSEvent interface
50     uint32_t acquireEvent(bool enable_timing) const override;
51     void releaseEvent(uint32_t event_id) const override;
52     void recordEvent(uint32_t event_id) const override;
53     void waitForEvent(uint32_t event_id) const override;
54     void synchronizeEvent(uint32_t event_id) const override;
55     bool queryEvent(uint32_t event_id) const override;
56     double elapsedTimeOfEvents(uint32_t start_event_id, uint32_t end_event_id)
57         const override;
58   
59     bool isBuilt() const override {
60       return true;
61     }
62     bool isAvailable() const override {
63       return hasMPS();
64     }
65     bool hasPrimaryContext(DeviceIndex device_index) const override {
66       // When MPS is available, it is always in use for the one device.
67       return true;
68     }
69   };
