51   def parse_maxima(str, globals=None, name_dict={}):
52       str = str.strip()
53       str = str.rstrip('; ')
54   
55       for k, v in sub_dict.items():
56           str = v.sub(k, str)
57   
58       assign_var = None
59       var_match = var_name.search(str)
60       if var_match:
61           assign_var = var_match.group(1)
62           str = str[var_match.end():].strip()
63   
64       dct = MaximaHelpers.__dict__.copy()
65       dct.update(name_dict)
66       obj = sympify(str, locals=dct)
67   
68       if assign_var and globals:
69           globals[assign_var] = obj
70   
71       return obj
