63   def _window_function_checks(
64       function_name: str, M: int, dtype: torch.dtype, layout: torch.layout
65   ) -> None:
66       r"""Performs common checks for all the defined windows. 
67       This function should be called before computing any window. 
68    
69       Args: 
70           function_name (str): name of the window function. 
71           M (int): length of the window. 
72           dtype (:class:`torch.dtype`): the desired data type of returned tensor. 
73           layout (:class:`torch.layout`): the desired layout of returned tensor. 
74       """
75       if M < 0:
76           raise ValueError(
77               f"{function_name} requires non-negative window length, got M={M}"
78           )
79       if layout is not torch.strided:
80           raise ValueError(
81               f"{function_name} is implemented for strided tensors only, got: {layout}"
82           )
83       if dtype not in [torch.float32, torch.float64]:
84           raise ValueError(
85               f"{function_name} expects float32 or float64 dtypes, got: {dtype}"
86           )

89   @_add_docstr(
90       r""" 
91   Computes a window with an exponential waveform. 
92   Also known as Poisson window. 
93    
94   The exponential window is defined as follows: 
95    
96   .. math:: 
97       w_n = \exp{\left(-\frac{|n - c|}{\tau}\right)} 
98    
99   where `c` is the ``center`` of the window. 
100      """,
101      r""" 
102   
103  {normalization} 
104   
105  Args: 
106      {M} 
107   
108  Keyword args: 
109      center (float, optional): where the center of the window will be located. 
110          Default: `M / 2` if `sym` is `False`, else `(M - 1) / 2`. 
111      tau (float, optional): the decay value. 
112          Tau is generally associated with a percentage, that means, that the value should 
113          vary within the interval (0, 100]. If tau is 100, it is considered the uniform window. 
114          Default: 1.0. 
115      {sym} 
116      {dtype} 
117      {layout} 
118      {device} 
119      {requires_grad} 
120   
121  Examples:: 
122   
123      >>> # Generates a symmetric exponential window of size 10 and with a decay value of 1.0. 
124      >>> # The center will be at (M - 1) / 2, where M is 10. 
125      >>> torch.signal.windows.exponential(10) 
126      tensor([0.0111, 0.0302, 0.0821, 0.2231, 0.6065, 0.6065, 0.2231, 0.0821, 0.0302, 0.0111]) 
127   
128      >>> # Generates a periodic exponential window and decay factor equal to .5 
129      >>> torch.signal.windows.exponential(10, sym=False,tau=.5) 
130      tensor([4.5400e-05, 3.3546e-04, 2.4788e-03, 1.8316e-02, 1.3534e-01, 1.0000e+00, 1.3534e-01, 1.8316e-02, 2.4788e-03, 3.3546e-04]) 
131      """.format(
132          **window_common_args
133      ),
134  )
135  def exponential(
136      M: int,
137      *,
138      center: Optional[float] = None,
139      tau: float = 1.0,
140      sym: bool = True,
141      dtype: Optional[torch.dtype] = None,
142      layout: torch.layout = torch.strided,
143      device: Optional[torch.device] = None,
144      requires_grad: bool = False,
145  ) -> Tensor:
146      if dtype is None:
147          dtype = torch.get_default_dtype()
148  
149      _window_function_checks("exponential", M, dtype, layout)
150  
151      if tau <= 0:
152          raise ValueError(f"Tau must be positive, got: {tau} instead.")
153  
154      if sym and center is not None:
155          raise ValueError("Center must be None for symmetric windows")
156  
157      if M == 0:
158          return torch.empty(
159              (0,), dtype=dtype, layout=layout, device=device, requires_grad=requires_grad
160          )
161  
162      if center is None:
163          center = (M if not sym and M > 1 else M - 1) / 2.0
164  
165      constant = 1 / tau
166  
167      k = torch.linspace(
168          start=-center * constant,
169          end=(-center + (M - 1)) * constant,
170          steps=M,
171          dtype=dtype,
172          layout=layout,
173          device=device,
174          requires_grad=requires_grad,
175      )
176  
177      return torch.exp(-torch.abs(k))

183  This window is also known as the sine window. 
184   
185  The cosine window is defined as follows: 
186   
187  .. math:: 
188      w_n = \sin\left(\frac{\pi (n + 0.5)}{M}\right) 
189

191  which shifts the sample positions. This adjustment results in a window that starts and ends with non-zero values. 
192   
193  """,
194      r""" 
195   
196  {normalization} 
197   
198  Args: 
199      {M} 
200   
201  Keyword args: 
202      {sym} 
203      {dtype} 
204      {layout} 
205      {device} 
206      {requires_grad} 
207   
208  Examples:: 
209   
210      >>> # Generates a symmetric cosine window. 
211      >>> torch.signal.windows.cosine(10) 
212      tensor([0.1564, 0.4540, 0.7071, 0.8910, 0.9877, 0.9877, 0.8910, 0.7071, 0.4540, 0.1564]) 
213   
214      >>> # Generates a periodic cosine window. 
215      >>> torch.signal.windows.cosine(10, sym=False) 
216      tensor([0.1423, 0.4154, 0.6549, 0.8413, 0.9595, 1.0000, 0.9595, 0.8413, 0.6549, 0.4154]) 
217  """.format(
218          **window_common_args,
219      ),
220  )
221  def cosine(
222      M: int,
223      *,
224      sym: bool = True,
225      dtype: Optional[torch.dtype] = None,
226      layout: torch.layout = torch.strided,
227      device: Optional[torch.device] = None,
228      requires_grad: bool = False,
229  ) -> Tensor:
230      if dtype is None:
231          dtype = torch.get_default_dtype()
232  
233      _window_function_checks("cosine", M, dtype, layout)
234  
235      if M == 0:
236          return torch.empty(
237              (0,), dtype=dtype, layout=layout, device=device, requires_grad=requires_grad
238          )
239  
240      start = 0.5
241      constant = torch.pi / (M + 1 if not sym and M > 1 else M)
242  
243      k = torch.linspace(
244          start=start * constant,
245          end=(start + (M - 1)) * constant,
246          steps=M,
247          dtype=dtype,
248          layout=layout,
249          device=device,
250          requires_grad=requires_grad,
251      )
252  
253      return torch.sin(k)

290  """.format(
291          **window_common_args,
292      ),
293  )
294  def gaussian(
295      M: int,
296      *,
297      std: float = 1.0,
298      sym: bool = True,
299      dtype: Optional[torch.dtype] = None,
300      layout: torch.layout = torch.strided,
301      device: Optional[torch.device] = None,
302      requires_grad: bool = False,
303  ) -> Tensor:
304      if dtype is None:
305          dtype = torch.get_default_dtype()
306  
307      _window_function_checks("gaussian", M, dtype, layout)
308  
309      if std <= 0:
310          raise ValueError(f"Standard deviation must be positive, got: {std} instead.")
311

316  
317      start = -(M if not sym and M > 1 else M - 1) / 2.0
318

320

325          dtype=dtype,
326          layout=layout,
327          device=device,
328          requires_grad=requires_grad,

336  Computes the Kaiser window. 
337   
338  The Kaiser window is defined as follows: 
339   
340  .. math:: 
341      w_n = I_0 \left( \beta \sqrt{1 - \left( {\frac{n - N/2}{N/2}} \right) ^2 } \right) / I_0( \beta ) 
342   
343  where ``I_0`` is the zeroth order modified Bessel function of the first kind (see :func:`torch.special.i0`), and 
344  ``N = M - 1 if sym else M``. 
345      """,
346      r""" 
347   
348  {normalization} 
349   
350  Args: 
351      {M} 
352   
353  Keyword args: 
354      beta (float, optional): shape parameter for the window. Must be non-negative. Default: 12.0 
355      {sym} 
356      {dtype} 
357      {layout} 
358      {device} 
359      {requires_grad} 
360   
361  Examples:: 
362   
363      >>> # Generates a symmetric gaussian window with a standard deviation of 1.0. 
364      >>> torch.signal.windows.kaiser(5) 
365      tensor([4.0065e-05, 2.1875e-03, 4.3937e-02, 3.2465e-01, 8.8250e-01, 8.8250e-01, 3.2465e-01, 4.3937e-02, 2.1875e-03, 4.0065e-05]) 
366      >>> # Generates a periodic gaussian window and standard deviation equal to 0.9. 
367      >>> torch.signal.windows.kaiser(5, sym=False,std=0.9) 
368      tensor([1.9858e-07, 5.1365e-05, 3.8659e-03, 8.4658e-02, 5.3941e-01, 1.0000e+00, 5.3941e-01, 8.4658e-02, 3.8659e-03, 5.1365e-05]) 
369  """.format(
370          **window_common_args,
371      ),
372  )
373  def kaiser(
374      M: int,
375      *,
376      beta: float = 12.0,
377      sym: bool = True,
378      dtype: Optional[torch.dtype] = None,
379      layout: torch.layout = torch.strided,
380      device: Optional[torch.device] = None,
381      requires_grad: bool = False,
382  ) -> Tensor:
383      if dtype is None:
384          dtype = torch.get_default_dtype()
385  
386      _window_function_checks("kaiser", M, dtype, layout)
387  
388      if beta < 0:
389          raise ValueError(f"beta must be non-negative, got: {beta} instead.")
390  
391      if M == 0:
392          return torch.empty(
393              (0,), dtype=dtype, layout=layout, device=device, requires_grad=requires_grad
394          )
395  
396      if M == 1:
397          return torch.ones(
398              (1,), dtype=dtype, layout=layout, device=device, requires_grad=requires_grad
399          )
400  
401      # Avoid NaNs by casting `beta` to the appropriate dtype.
402      beta = torch.tensor(beta, dtype=dtype, device=device)
403  
404      start = -beta
405      constant = 2.0 * beta / (M if not sym else M - 1)
406      end = torch.minimum(beta, start + (M - 1) * constant)
407  
408      k = torch.linspace(
409          start=start,
410          end=end,
411          steps=M,
412          dtype=dtype,
413          layout=layout,
414          device=device,
415          requires_grad=requires_grad,
416      )
417  
418      return torch.i0(torch.sqrt(beta * beta - torch.pow(k, 2))) / torch.i0(beta)

616      {dtype} 
617      {layout} 
618      {device} 
619      {requires_grad} 
620

622   
623      >>> # Generates a symmetric Bartlett window. 
624      >>> torch.signal.windows.bartlett(10) 
625      tensor([0.0000, 0.2222, 0.4444, 0.6667, 0.8889, 0.8889, 0.6667, 0.4444, 0.2222, 0.0000]) 
626   
627      >>> # Generates a periodic Bartlett window. 
628      >>> torch.signal.windows.bartlett(10, sym=False) 
629      tensor([0.0000, 0.2000, 0.4000, 0.6000, 0.8000, 1.0000, 0.8000, 0.6000, 0.4000, 0.2000]) 
630  """.format(
631          **window_common_args
632      ),
633  )
634  def bartlett(
635      M: int,
636      *,
637      sym: bool = True,
638      dtype: Optional[torch.dtype] = None,
639      layout: torch.layout = torch.strided,
640      device: Optional[torch.device] = None,
641      requires_grad: bool = False,
642  ) -> Tensor:
643      if dtype is None:
644          dtype = torch.get_default_dtype()
645  
646      _window_function_checks("bartlett", M, dtype, layout)
647  
648      if M == 0:
649          return torch.empty(
650              (0,), dtype=dtype, layout=layout, device=device, requires_grad=requires_grad
651          )
652  
653      if M == 1:
654          return torch.ones(
655              (1,), dtype=dtype, layout=layout, device=device, requires_grad=requires_grad
656          )
657  
658      start = -1
659      constant = 2 / (M if not sym else M - 1)
660  
661      k = torch.linspace(
662          start=start,
663          end=start + (M - 1) * constant,
664          steps=M,
665          dtype=dtype,
666          layout=layout,
667          device=device,
668          requires_grad=requires_grad,
669      )
670  
671      return 1 - torch.abs(k)

707  """.format(
708          **window_common_args
709      ),
710  )
711  def general_cosine(
712      M,
713      *,
714      a: Iterable,
715      sym: bool = True,
716      dtype: Optional[torch.dtype] = None,
717      layout: torch.layout = torch.strided,
718      device: Optional[torch.device] = None,
719      requires_grad: bool = False,
720  ) -> Tensor:
721      if dtype is None:
722          dtype = torch.get_default_dtype()
723  
724      _window_function_checks("general_cosine", M, dtype, layout)
725  
726      if M == 0:
727          return torch.empty(
728              (0,), dtype=dtype, layout=layout, device=device, requires_grad=requires_grad
729          )
730  
731      if M == 1:
732          return torch.ones(
733              (1,), dtype=dtype, layout=layout, device=device, requires_grad=requires_grad
734          )
735  
736      if not isinstance(a, Iterable):
737          raise TypeError("Coefficients must be a list/tuple")
738  
739      if not a:
740          raise ValueError("Coefficients cannot be empty")
741  
742      constant = 2 * torch.pi / (M if not sym else M - 1)
743  
744      k = torch.linspace(
745          start=0,
746          end=(M - 1) * constant,
747          steps=M,
748          dtype=dtype,
749          layout=layout,
750          device=device,
751          requires_grad=requires_grad,
752      )
753  
754      a_i = torch.tensor(
755          [(-1) ** i * w for i, w in enumerate(a)],
756          device=device,
757          dtype=dtype,
758          requires_grad=requires_grad,
759      )
760      i = torch.arange(
761          a_i.shape[0],
762          dtype=a_i.dtype,
763          device=a_i.device,
764          requires_grad=a_i.requires_grad,
765      )
766      return (a_i.unsqueeze(-1) * torch.cos(i.unsqueeze(-1) * k)).sum(0)
