18   template<int N> 
19   static OffsetCalculator<N> make_input_offset_calculator(const TensorIteratorBase& iter) {
20     // array size can not be 0, this happens when N == 0 
21     constexpr int array_size = std::max<int>(N, 1);
22     TORCH_INTERNAL_ASSERT(N == iter.ntensors() - iter.noutputs());
23     std::array<const int64_t*, array_size> strides;
24     int64_t element_sizes[array_size];
25     for (int i = 0; i < N; i++) {
26       strides[i] = iter.strides(i + iter.noutputs()).data();
27       element_sizes[i] = iter.element_size(i + iter.noutputs());
28     }
29     return OffsetCalculator<N>(iter.ndim(), iter.shape().data(), strides.data(), element_sizes);
30   }
31   
32   template <int num_outputs = 1> 
33   static OffsetCalculator<num_outputs> make_output_offset_calculator(const TensorIteratorBase& iter) {
34     TORCH_INTERNAL_ASSERT(num_outputs == iter.noutputs());
35     std::array<const int64_t*, num_outputs> strides;
36     int64_t element_sizes[num_outputs];
37     for (int i = 0; i < num_outputs; i++) {
38       strides[i] = iter.strides(i).data();
39       element_sizes[i] = iter.element_size(i);
40     }
41     return OffsetCalculator<num_outputs>(iter.ndim(), iter.shape().data(), strides.data(), element_sizes);
42   }

44   template <bool reverted_idx = false, typename func_t, typename policy_t> 
45   __device__ inline void elementwise_kernel_helper(func_t f, policy_t policy) {
46     using traits = function_traits<func_t>;
47     using return_t = typename traits::result_type;
48     using args_t = typename traits::ArgsTuple;
49     constexpr int elems_per_thread = policy_t::tws;
50   
51     int idx = blockIdx.x;
52     if constexpr (reverted_idx)
53       idx = gridDim.x - blockIdx.x - 1;
54   
55     return_t results[elems_per_thread];
56     args_t args[elems_per_thread];
57   
58     // load 
59     policy.load(args, idx);
60   
61     // compute 
62     #pragma unroll
63     for (int i = 0; i < elems_per_thread; i++) {
64       if (policy.check_inbounds(i)) {
65         results[i] = c10::guts::apply(f, args[i]);
66       }
67     }
68   
69     // store 
70     policy.store(results, idx);
71   }

79   template <typename func_t> 
80   void gpu_kernel_nocast(TensorIteratorBase& iter, const func_t& f) {
81   
82     for (int arg = 0; arg < iter.ntensors(); arg++) {
83       TORCH_INTERNAL_ASSERT(
84         iter.device(arg).is_cuda(),
85         "argument ", arg, ": expected a CUDA device but found ", iter.device(arg));
86     }
87   
88     if (iter.numel() == 0) {
89       return;
90     }
91   
92     if (!iter.can_use_32bit_indexing()) {
93       for (auto& sub_iter : iter.with_32bit_indexing()) {
94         gpu_kernel_nocast(sub_iter, f);
95       }
96       return;
97     }
98   
99     gpu_kernel_impl_nocast(iter, f);
100  }

102  template <typename func_t> 
103  void gpu_kernel(TensorIteratorBase& iter, const func_t& f) {
104  
105    for (int arg = 0; arg < iter.ntensors(); arg++) {
106      TORCH_INTERNAL_ASSERT(
107        iter.device(arg).is_cuda(),
108        "argument ", arg, ": expected a CUDA device but found ", iter.device(arg));
109    }
110  
111    if (iter.numel() == 0) {
112      return;
113    }
114  
115    if (!iter.can_use_32bit_indexing()) {
116      for (auto& sub_iter : iter.with_32bit_indexing()) {
117        gpu_kernel(sub_iter, f);
118      }
119      return;
120    }
121  
122    gpu_kernel_impl(iter, f);
123  }

125  template<typename arg1_t, typename arg2_t, typename return_t, typename func_t> 
126  struct AUnaryFunctor {
127    using traits = function_traits<func_t>;
128    using opmath_arg1_t = typename traits::template arg<0>::type;
129    __device__ return_t operator()(arg2_t b) const {
130      return f(a, b);
131    }
132    // NB: scalar is stored in higher precision! 
133    AUnaryFunctor(func_t f_, opmath_arg1_t a_): f(f_), a(a_) {}
134    private:
135      func_t f;
136      opmath_arg1_t a;
137  };

139  template<typename arg1_t, typename arg2_t, typename return_t, typename func_t> 
140  struct BUnaryFunctor {
141    using traits = function_traits<func_t>;
142    using opmath_arg2_t = typename traits::template arg<1>::type;
143    __device__ return_t operator()(arg1_t a) const {
144      return f(a, b);
145    }
146    // NB: scalar is stored in higher precision! 
147    BUnaryFunctor(func_t f_, opmath_arg2_t b_): f(f_), b(b_) {}
148    private:
149      func_t f;
150      opmath_arg2_t b;
151  };

153  // Though seemingly noop, this inserts casts from arg1_t to func_t's type 
154  // (which may be higher precision), as well as casts to return_t 
155  template <typename arg1_t, typename arg2_t, typename return_t, typename func_t> 
156  struct BinaryFunctor {
157    __device__ return_t operator()(arg1_t a, arg2_t b) const {
158      return f(a, b);
159    }
160    BinaryFunctor(func_t f_): f(f_) {}
161    private:
162      func_t f;
163  };

165  // Unlike gpu_kernel_with_scalars, this allows you to pass a func_t which 
166  // accepts inputs at higher precision (typically opmath_t), but then 
167  // ensure that we load from memory at the correct precision (scalar_t) 
168  // to avoid expensive loads.  For the whole sordid story see 
169  // https://dev-discuss.pytorch.org/t/cuda-loops-case-study-code-generation-vs-templates/302 
170  template <typename arg1_t, typename arg2_t = arg1_t, typename return_t = arg1_t, typename func_t> 
171  void opmath_gpu_kernel_with_scalars(TensorIteratorBase& iter, const func_t& f) {
172    TORCH_INTERNAL_ASSERT(iter.ntensors() == 3);
173  
174    using traits = function_traits<func_t>;
175    using opmath_arg1_t = typename traits::template arg<0>::type;
176    using opmath_arg2_t = typename traits::template arg<1>::type;
177    static_assert(
178        traits::arity == 2,
179        "gpu_kernel_with_scalars only supports two input arguments");
180  
181    if (iter.is_cpu_scalar(1)) {
182      AUnaryFunctor<arg1_t, arg2_t, return_t, func_t> af(f, iter.scalar_value<opmath_arg1_t>(1));
183      iter.remove_operand(1);
184      // TODO: When all kernels that use gpu_kernel_with_scalars are 
185      // ported to structured, this device guard can be deleted.  This 
186      // works around incorrect device guard generation for pre-structured 
187      // kernels device guards, but structured kernels do it right and 
188      // we can assume the device is already set correctly 
189      const OptionalDeviceGuard device_guard(iter.device(1));
190      gpu_kernel(iter, af);
191    } else if (iter.is_cpu_scalar(2)) {
192      BUnaryFunctor<arg1_t, arg2_t, return_t, func_t> bf(f, iter.scalar_value<opmath_arg2_t>(2));
193      iter.remove_operand(2);
194      gpu_kernel(iter, bf);
195    } else {
196      gpu_kernel(iter, BinaryFunctor<arg1_t, arg2_t, return_t, func_t>(f));
197    }
198  }

200  template <typename scalar_t, typename return_t = scalar_t, typename func_t> 
201  void opmath_symmetric_gpu_kernel_with_scalars(TensorIteratorBase& iter, const func_t& f) {
202    // Use symmetric property of the functor to reduce number of kernels, 
203    // requires f(a, b) == f(b, a) 
204    TORCH_INTERNAL_ASSERT(iter.ntensors() == 3);
205  
206    using traits = function_traits<func_t>;
207    using opmath_arg_t = typename traits::template arg<0>::type;
208    static_assert(
209        traits::arity == 2,
210        "gpu_kernel_with_scalars only supports two input arguments");
211    static_assert(std::is_same_v<opmath_arg_t, typename traits::template arg<1>::type>, 
212                  "f is not symmetric");
213  
214    OptionalDeviceGuard device_guard;
215    opmath_arg_t scalar_val{};
216  
217    if (iter.is_cpu_scalar(1)) {
218      scalar_val = iter.scalar_value<opmath_arg_t>(1);
219      iter.remove_operand(1);
220  
221      // TODO: When all kernels that use gpu_kernel_with_scalars are 
222      // ported to structured, this device guard can be deleted.  This 
223      // works around incorrect device guard generation for pre-structured 
224      // kernels device guards, but structured kernels do it right and 
225      // we can assume the device is already set correctly 
226      device_guard.reset_device(iter.device(1));
227    } else if (iter.is_cpu_scalar(2)) {
228      scalar_val = iter.scalar_value<opmath_arg_t>(2);
229      iter.remove_operand(2);
230    }
231  
232    if (iter.ninputs() == 2) {
233      gpu_kernel(iter, BinaryFunctor<scalar_t, scalar_t, return_t, func_t>(f));
234    } else {
235      AUnaryFunctor<scalar_t, scalar_t, return_t, func_t> unary_f(f, scalar_val);
236      gpu_kernel(iter, unary_f);
237    }
238  }

240  // Legacy variant that assumes that func_t has the correct types 
241  // that we expect to load from memory 
242  template <typename func_t> 
243  void gpu_kernel_with_scalars(TensorIteratorBase& iter, const func_t& f) {
244    using traits = function_traits<func_t>;
245    static_assert(
246        traits::arity == 2,
247        "gpu_kernel_with_scalars only supports two input arguments");
248    using arg1_t = typename traits::template arg<0>::type;
249    using arg2_t = typename traits::template arg<1>::type;
250    using return_t = typename traits::result_type;
251    opmath_gpu_kernel_with_scalars<arg1_t, arg2_t, return_t, func_t>(iter, f);
252  }

261  template <int num_outputs, typename func_t, typename array_t, typename inp_calc_t, typename out_calc_t> 
262  C10_LAUNCH_BOUNDS_1(num_threads())
263  __global__ void unrolled_elementwise_kernel_for_multi_outputs(int N, func_t f, array_t data, inp_calc_t ic, out_calc_t oc) {
264    int remaining = N - block_work_size() * blockIdx.x;
265    elementwise_kernel_helper(f, memory::policies::multi_outputs_unroll<array_t, inp_calc_t, out_calc_t, num_outputs>(data, remaining, ic, oc));
266  }

268  template <int num_outputs, typename func_t, typename array_t, typename inp_calc_t, typename out_calc_t> 
269  static inline void launch_unrolled_kernel_for_multi_outputs(int64_t N, const func_t& f, array_t data, inp_calc_t ic, out_calc_t oc) {
270    TORCH_INTERNAL_ASSERT(N > 0 && N <= std::numeric_limits<int32_t>::max());
271    int64_t grid = (N + block_work_size() - 1) / block_work_size();
272    auto stream = at::cuda::getCurrentCUDAStream();
273    unrolled_elementwise_kernel_for_multi_outputs<num_outputs, func_t, array_t><<<grid, num_threads(), 0, stream>>>(N, f, data, ic, oc);
274    C10_CUDA_KERNEL_LAUNCH_CHECK();
275  }

277  template <typename func_t> 
278  void gpu_kernel_multiple_outputs_impl(TensorIteratorBase& iter, const func_t& f) {
279    using traits = function_traits<func_t>;
280    using output_t = typename traits::result_type;
281    static_assert(is_tuple<output_t>::value, "f's return type must be `thrust::tuple`");
282    constexpr int num_outputs = thrust::tuple_size<output_t>::value;
283    constexpr int num_inputs = traits::arity;
284    constexpr int ntensors = num_outputs + num_inputs;
285  
286    TORCH_INTERNAL_ASSERT(iter.can_use_32bit_indexing());
287    TORCH_INTERNAL_ASSERT(iter.ntensors() == ntensors);
288  
289    std::array<char*, ntensors> data;
290    for (int i = 0; i < ntensors; i++) {
291      data[i] = (char*)iter.data_ptr(i);
292    }
293  
294    int64_t numel = iter.numel();
295  
296    if (iter.is_contiguous()) {
297      auto input_calc = TrivialOffsetCalculator<num_inputs>();
298      auto output_calc = TrivialOffsetCalculator<num_outputs>();
299      launch_unrolled_kernel_for_multi_outputs<num_outputs>(numel, f, data, input_calc, output_calc);
300    } else {
301      auto input_calc = make_input_offset_calculator<num_inputs>(iter);
302      auto output_calc = make_output_offset_calculator<num_outputs>(iter);
303      launch_unrolled_kernel_for_multi_outputs<num_outputs>(numel, f, data, input_calc, output_calc);
304    }
305  }

308  template <typename func_t> 
309  void gpu_kernel_multiple_outputs(TensorIteratorBase& iter, const func_t& f) {
310    ASSERT_HOST_DEVICE_LAMBDA(func_t);
311  
312    for (int arg = 0; arg < iter.ntensors(); arg++) {
313      TORCH_INTERNAL_ASSERT(iter.device(arg).is_cuda());
314    }
315  
316    if (iter.numel() == 0) {
317      return;
318    }
319  
320    if (!iter.can_use_32bit_indexing()) {
321      for (auto& sub_iter : iter.with_32bit_indexing()) {
322        gpu_kernel_multiple_outputs(sub_iter, f);
323      }
324      return;
325    }
326  
327    gpu_kernel_multiple_outputs_impl(iter, f);
328  }
