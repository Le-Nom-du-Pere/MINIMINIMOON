17   def call_op(
18       op_type: str,
19       *args: ir.Value,
20       _num_outputs: int = 1,
21       _domain: str = "",
22       **kwargs: int | float | str | bool | ir.Graph | ir.TensorProtocol | Sequence[int],
23   ) -> Sequence[ir.Value]:
24       """Call an operator with the given arguments and keyword arguments. 
25    
26       Arguments are always inputs, while keyword arguments are attributes. 
27       """
28       # This is a wrapper around the IR node creation that hooks into the _builder.OpRecorder
29       # tracer so that all nodes created are recorded the same way as if we were to use
30       # onnxscript ops directly.
31       from onnxscript.ir import convenience as ir_convenience
32   
33       assert _core.current_tracer is not None
34       tracer = _core.current_tracer
35   
36       inputs = list(args)
37   
38       # If final inputs are None, strip them from the node inputs
39       for input in reversed(inputs):
40           if input is not None:
41               break
42           inputs.pop()
43   
44       # Construct and filter out None attributes
45       attributes = [
46           attr
47           for attr in ir_convenience.convert_attributes(kwargs)
48           if attr.value is not None  # type: ignore[union-attr]
49       ]
50       tracer.nodes.append(
51           node := ir.Node(
52               _domain,
53               op_type,
54               inputs=inputs,
55               attributes=attributes,
56               num_outputs=_num_outputs,
57               version=tracer.opset.version,
58           )
59       )
60       return node.outputs

63   @onnx_impl(torch.ops.higher_order.cond, no_compile=True)
64   def higher_order_cond(
65       cond: ir.Value,
66       true_func: ir.Function,
67       false_func: ir.Function,
68       inputs: Sequence[ir.Value],
69   ) -> Sequence[ir.Value]:
70       then_node = ir.Node(
71           true_func.domain, true_func.name, inputs, num_outputs=len(true_func.outputs)
72       )
73       else_node = ir.Node(
74           false_func.domain, false_func.name, inputs, num_outputs=len(false_func.outputs)
75       )
76   
77       # ONNX Runtime complains about duplicate output names if we don't rename them.
78       # But the doesn't seem to be an actual violation of SSA form without renaming.
79       for func_out, out in zip(true_func.outputs, then_node.outputs):
80           out.name = f"{func_out.name}_{true_func.name}"
81       for func_out, out in zip(false_func.outputs, else_node.outputs):
82           out.name = f"{func_out.name}_{false_func.name}"
83   
84       return call_op(
85           "If",
86           cond,
87           _num_outputs=len(true_func.outputs),
88           then_branch=ir.Graph(
89               (), then_node.outputs, nodes=[then_node], name=true_func.name
90           ),
91           else_branch=ir.Graph(
92               (), else_node.outputs, nodes=[else_node], name=false_func.name
93           ),
94       )

97   @onnx_impl(torch.ops.higher_order.scan, no_compile=True)
98   def higher_order_scan(
99       body_func: ir.Function,
100      scan_inits: Sequence[ir.Value],
101      scan_inputs: Sequence[ir.Value],
102      additional_inputs: Sequence[ir.Value] | None,
103      reverse: bool = False,
104  ) -> Sequence[ir.Value]:
105      """https://github.com/pytorch/pytorch/blob/66ac724b56e6c37a534f3e066423ef2f41d7477f/torch/_higher_order_ops/scan.py#L109"""
106      subgraph_inputs = [
107          *[
108              ir.Value(
109                  name=f"{inp.name}_{body_func.name}__subgraph_in",
110                  shape=inp.shape,
111                  type=ir.TensorType(inp.dtype),  # type: ignore[arg-type]
112              )
113              for inp in scan_inits
114          ],
115          *[
116              ir.Value(
117                  name=f"{inp.name}_{body_func.name}__subgraph_in",
118                  # The iterated element passed to the body subgraph does not have a sequence axis.
119                  # It will have a rank one less than the rank of the corresponding scan_input.
120                  shape=ir.Shape(inp.shape[1:]),  # type: ignore[index]
121                  type=ir.TensorType(inp.dtype),  # type: ignore[arg-type]
122              )
123              for inp in scan_inputs
124          ],
125      ]
126      # The one and only node in the Scan subgraph that calls the body_func
127      body_node = ir.Node(
128          body_func.domain,
129          body_func.name,
130          [
131              *subgraph_inputs,
132              *(additional_inputs or []),
133          ],
134          num_outputs=len(body_func.outputs),
135      )
136  
137      # ONNX Runtime complains about duplicate output names if we don't rename them.
138      # But the doesn't seem to be an actual violation of SSA form without renaming.
139      for func_out, out in zip(body_func.outputs, body_node.outputs):
140          out.name = f"{func_out.name}_{body_func.name}"
141  
142      n_outputs = len(body_func.outputs) - len(scan_inits)
143      return call_op(
144          "Scan",
145          *scan_inits,
146          *scan_inputs,
147          _num_outputs=len(body_func.outputs),
148          body=ir.Graph(
149              subgraph_inputs,
150              body_node.outputs,
151              nodes=[body_node],
152              name=body_func.name,
153          ),
154          num_scan_inputs=len(scan_inputs),
155          scan_input_directions=[(1 if reverse else 0) for _ in scan_inputs],
156          scan_output_directions=[(1 if reverse else 0) for _ in range(n_outputs)],
157      )
