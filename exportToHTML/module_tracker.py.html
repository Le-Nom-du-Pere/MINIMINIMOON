25   class ModuleTracker:
26       """ 
27       ``ModuleTracker`` is a context manager that tracks the nn.Module hierarchy during execution 
28       so that other system can query which Module is currently being executed (or its backward is being 
29       executed). 
30    
31       You can access the ``parents`` attribute on this context manager to get the set of all the 
32       Modules currently being executed via their fqn (fully qualified name, also used as the key within 
33       the state_dict). 
34       You can access the ``is_bw`` attribute to know if you are currently running in backward or not. 
35    
36       Note that ``parents`` is never empty and always contains the "Global" key. The ``is_bw`` flag 
37       will remain ``True`` after the forward until another Module is executed. If you need it to be 
38       more accurate, please submit an issue requesting this. Adding a map from fqn to the module instance 
39       is possible but not done yet, please submit an issue requesting this if you need it. 
40    
41       Example usage 
42    
43       .. code-block:: python 
44    
45           mod = torch.nn.Linear(2, 2) 
46    
47           with ModuleTracker() as tracker: 
48               # Access anything during the forward pass 
49               def my_linear(m1, m2, bias): 
50                   print(f"Current modules: {tracker.parents}") 
51                   return torch.mm(m1, m2.t()) + bias 
52               torch.nn.functional.linear = my_linear 
53    
54               mod(torch.rand(2, 2)) 
55    
56       """
57   
58       parents: set[str]
59       """ 
60       A Set containing the fqn for each module currently running their forward 
61       """
62   
63       def __init__(self) -> None:
64           self.parents = {"Global"}
65           self._known_modules: weakref.WeakKeyDictionary = weakref.WeakKeyDictionary()
66           self._seen_modules: weakref.WeakSet = weakref.WeakSet()
67           self._has_callback = False
68           self._hooks: list[RemovableHandle] = []
69   
70       def _maybe_set_engine_callback(self):
71           # This assumes no concurrent calls to backward
72           if self._has_callback:
73               return
74   
75           def callback():
76               self.parents = {"Global"}
77               self._has_callback = False
78   
79           torch.autograd.Variable._execution_engine.queue_callback(callback)
80           self._has_callback = True
81   
82       @property
83       def is_bw(self):
84           """ 
85           A boolean marking if this is currently running during the backward pass or not 
86           """
87           return torch._C._current_graph_task_id() != -1
88   
89       def _get_mod_name(self, mod):
90           if mod not in self._known_modules:
91               self._known_modules[mod] = type(mod).__name__
92           mod_name = self._known_modules[mod]
93           if mod not in self._seen_modules:
94               for name, submod in mod.named_children():
95                   self._known_modules[submod] = f"{mod_name}.{name}"
96                   self._get_mod_name(submod)
97               self._seen_modules.add(mod)
98           return mod_name
99   
100      def _get_append_fn(self, name, is_bw):
101          def fn(*args):
102              if is_bw:
103                  self._maybe_set_engine_callback()
104              if name in self.parents:
105                  logger.info(
106                      "The module hierarchy tracking seems to be broken as this Module was already entered. %s during %s",
107                      name,
108                      "backward" if is_bw else "forward",
109                  )
110              self.parents.add(name)
111  
112          return fn
113  
114      def _get_pop_fn(self, name, is_bw):
115          def fn(*args):
116              if name in self.parents:
117                  self.parents.remove(name)
118              else:
119                  logger.info(
120                      "The Module hierarchy tracking is confused as we're exiting a Module that was never entered. %s during %s",
121                      name,
122                      "backward" if is_bw else "forward",
123                  )
124  
125          return fn
126  
127      def _fw_pre_hook(self, mod, input):
128          name = self._get_mod_name(mod)
129          self._get_append_fn(name, False)()
130  
131          args, _ = tree_flatten(input)
132          tensors = [a for a in args if isinstance(a, torch.Tensor) and a.requires_grad]
133          if tensors:
134              self._hooks.append(
135                  register_multi_grad_hook(tensors, self._get_pop_fn(name, True))
136              )
137  
138      def _fw_post_hook(self, mod, input, output):
139          name = self._get_mod_name(mod)
140          self._get_pop_fn(name, False)()
141  
142          args, _ = tree_flatten(output)
143          tensors = [a for a in args if isinstance(a, torch.Tensor) and a.requires_grad]
144          if tensors:
145              self._hooks.append(
146                  register_multi_grad_hook(tensors, self._get_append_fn(name, True))
147              )
148  
149      def __enter__(self):
150          self._fw_pre_handle = register_module_forward_pre_hook(self._fw_pre_hook)
151          self._fw_post_handle = register_module_forward_hook(self._fw_post_hook)
152          return self
153  
154      def __exit__(self, *args):
155          self._fw_pre_handle.remove()
156          self._fw_post_handle.remove()
157          for hook in self._hooks:
158              hook.remove()
159          self._hooks.clear()
