11   // Note [Accelerator Concept]
12   // This file defines the top level Accelerator concept for PyTorch.
13   // A device is an accelerator per the definition here if:
14   // - It is mutually exclusive with all other accelerators
15   // - It performs asynchronous compute via a Stream/Event system
16   // - It provides a set of common APIs as defined by AcceleratorHooksInterface
17   //
18   // As of today, accelerator devices are (in no particular order):
19   // CUDA, MTIA, XPU, HIP, MPS, PrivateUse1
20   
21   // Ensures that only one accelerator is available (at
22   // compile time if possible) and return it.
23   // When checked is true, the returned optional always has a value.
24   TORCH_API std::optional<c10::DeviceType> getAccelerator(bool checked = false);
25   
26   // Check if the given device type is an accelerator.
27   TORCH_API bool isAccelerator(c10::DeviceType device_type);
28   
29   // Check if the given device type is an accelerator, not the excluded ones.
30   template <
31       typename... T,
32       typename = std::enable_if_t<(std::is_same_v<T, c10::DeviceType> && ...)>>
33   TORCH_API inline bool isAcceleratorExcluded(
34       c10::DeviceType device_type,
35       c10::DeviceType first_excluded,
36       T... rest_excluded) {
37     if constexpr (sizeof...(rest_excluded) > 0) {
38       return device_type != first_excluded &&
39           isAcceleratorExcluded(device_type, rest_excluded...);
40     } else {
41       return device_type != first_excluded && isAccelerator(device_type);
42     }
43   }
44   
45   // Return the number of the device available. Note that this is *REQUIRED* to
46   // not raise any exception.
47   TORCH_API c10::DeviceIndex deviceCount();
48   
49   // Set the current device index to the given device index.
50   TORCH_API void setDeviceIndex(c10::DeviceIndex device_index);
51   
52   // Get the current device index.
53   TORCH_API c10::DeviceIndex getDeviceIndex();
54   
55   // Set the current stream to a given stream. Note that this API doesn't change
56   // the current device index.
57   TORCH_API void setCurrentStream(c10::Stream stream);
58   
59   // Get the current stream of the given device index.
60   TORCH_API c10::Stream getCurrentStream(c10::DeviceIndex device_index);
61   
62   // Wait (by blocking the calling thread) until all the work previously enqueued
63   // on the given device index has been completed.
64   TORCH_API void synchronizeDevice(c10::DeviceIndex device_index);
65   
66   // Set the current device index to the given device_index and return the
67   // original device index that was active before the change.
68   TORCH_API c10::DeviceIndex exchangeDevice(c10::DeviceIndex device_index);
69   
70   // Set the current device index to the given device_index. Avoid creating a new
71   // context if the context for device_index is not initialized. Return the
72   // original device index that was active before the change.
73   TORCH_API c10::DeviceIndex maybeExchangeDevice(c10::DeviceIndex device_index);
