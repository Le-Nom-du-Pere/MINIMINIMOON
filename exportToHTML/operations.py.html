29   class AssocOp(Basic):
30       """ Associative operations, can separate noncommutative and 
31       commutative parts. 
32    
33       (a op b) op c == a op (b op c) == a op b op c. 
34    
35       Base class for Add and Mul. 
36    
37       This is an abstract base class, concrete derived classes must define 
38       the attribute `identity`. 
39    
40       .. deprecated:: 1.7 
41    
42          Using arguments that aren't subclasses of :class:`~.Expr` in core 
43          operators (:class:`~.Mul`, :class:`~.Add`, and :class:`~.Pow`) is 
44          deprecated. See :ref:`non-expr-args-deprecated` for details. 
45    
46       Parameters 
47       ========== 
48    
49       *args : 
50           Arguments which are operated 
51    
52       evaluate : bool, optional 
53           Evaluate the operation. If not passed, refer to ``global_parameters.evaluate``. 
54       """
55   
56       # for performance reason, we don't let is_commutative go to assumptions,
57       # and keep it right here
58       __slots__: tuple[str, ...] = ('is_commutative',)
59   
60       _args_type: type[Basic] | None = None
61   
62       @cacheit
63       def __new__(cls, *args, evaluate=None, _sympify=True):
64           # Allow faster processing by passing ``_sympify=False``, if all arguments
65           # are already sympified.
66           if _sympify:
67               args = list(map(_sympify_, args))
68   
69           # Disallow non-Expr args in Add/Mul
70           typ = cls._args_type
71           if typ is not None:
72               from .relational import Relational
73               if any(isinstance(arg, Relational) for arg in args):
74                   raise TypeError("Relational cannot be used in %s" % cls.__name__)
75   
76               # This should raise TypeError once deprecation period is over:
77               for arg in args:
78                   if not isinstance(arg, typ):
79                       sympy_deprecation_warning(
80                           f"""
81   
82   Using non-Expr arguments in {cls.__name__} is deprecated (in this case, one of
83   the arguments has type {type(arg).__name__!r}).
84   
85   If you really did intend to use a multiplication or addition operation with
86   this object, use the * or + operator instead.
87   
88                           """,
89                           deprecated_since_version="1.7",
90                           active_deprecations_target="non-expr-args-deprecated",
91                           stacklevel=4,
92                       )
93   
94           if evaluate is None:
95               evaluate = global_parameters.evaluate
96           if not evaluate:
97               obj = cls._from_args(args)
98               obj = cls._exec_constructor_postprocessors(obj)
99               return obj
100  
101          args = [a for a in args if a is not cls.identity]
102  
103          if len(args) == 0:
104              return cls.identity
105          if len(args) == 1:
106              return args[0]
107  
108          c_part, nc_part, order_symbols = cls.flatten(args)
109          is_commutative = not nc_part
110          obj = cls._from_args(c_part + nc_part, is_commutative)
111          obj = cls._exec_constructor_postprocessors(obj)
112  
113          if order_symbols is not None:
114              from sympy.series.order import Order
115              return Order(obj, *order_symbols)
116          return obj

118      @classmethod
119      def _from_args(cls, args, is_commutative=None):
120          """Create new instance with already-processed args. 
121          If the args are not in canonical order, then a non-canonical 
122          result will be returned, so use with caution. The order of 
123          args may change if the sign of the args is changed."""
124          if len(args) == 0:
125              return cls.identity
126          elif len(args) == 1:
127              return args[0]
128  
129          obj = super().__new__(cls, *args)
130          if is_commutative is None:
131              is_commutative = fuzzy_and(a.is_commutative for a in args)
132          obj.is_commutative = is_commutative
133          return obj

135      def _new_rawargs(self, *args, reeval=True, **kwargs):
136          """Create new instance of own class with args exactly as provided by 
137          caller but returning the self class identity if args is empty. 
138   
139          Examples 
140          ======== 
141   
142             This is handy when we want to optimize things, e.g. 
143   
144                 >>> from sympy import Mul, S 
145                 >>> from sympy.abc import x, y 
146                 >>> e = Mul(3, x, y) 
147                 >>> e.args 
148                 (3, x, y) 
149                 >>> Mul(*e.args[1:]) 
150                 x*y 
151                 >>> e._new_rawargs(*e.args[1:])  # the same as above, but faster 
152                 x*y 
153   
154             Note: use this with caution. There is no checking of arguments at 
155             all. This is best used when you are rebuilding an Add or Mul after 
156             simply removing one or more args. If, for example, modifications, 
157             result in extra 1s being inserted they will show up in the result: 
158   
159                 >>> m = (x*y)._new_rawargs(S.One, x); m 
160                 1*x 
161                 >>> m == x 
162                 False 
163                 >>> m.is_Mul 
164                 True 
165   
166             Another issue to be aware of is that the commutativity of the result 
167             is based on the commutativity of self. If you are rebuilding the 
168             terms that came from a commutative object then there will be no 
169             problem, but if self was non-commutative then what you are 
170             rebuilding may now be commutative. 
171   
172             Although this routine tries to do as little as possible with the 
173             input, getting the commutativity right is important, so this level 
174             of safety is enforced: commutativity will always be recomputed if 
175             self is non-commutative and kwarg `reeval=False` has not been 
176             passed. 
177          """
178          if reeval and self.is_commutative is False:
179              is_commutative = None
180          else:
181              is_commutative = self.is_commutative
182          return self._from_args(args, is_commutative)

184      @classmethod
185      def flatten(cls, seq):
186          """Return seq so that none of the elements are of type `cls`. This is 
187          the vanilla routine that will be used if a class derived from AssocOp 
188          does not define its own flatten routine."""
189          # apply associativity, no commutativity property is used
190          new_seq = []
191          while seq:
192              o = seq.pop()
193              if o.__class__ is cls:  # classes must match exactly
194                  seq.extend(o.args)
195              else:
196                  new_seq.append(o)
197          new_seq.reverse()
198  
199          # c_part, nc_part, order_symbols
200          return [], new_seq, None

202      def _matches_commutative(self, expr, repl_dict=None, old=False):
203          """ 
204          Matches Add/Mul "pattern" to an expression "expr". 
205   
206          repl_dict ... a dictionary of (wild: expression) pairs, that get 
207                        returned with the results 
208   
209          This function is the main workhorse for Add/Mul. 
210   
211          Examples 
212          ======== 
213   
214          >>> from sympy import symbols, Wild, sin 
215          >>> a = Wild("a") 
216          >>> b = Wild("b") 
217          >>> c = Wild("c") 
218          >>> x, y, z = symbols("x y z") 
219          >>> (a+sin(b)*c)._matches_commutative(x+sin(y)*z) 
220          {a_: x, b_: y, c_: z} 
221   
222          In the example above, "a+sin(b)*c" is the pattern, and "x+sin(y)*z" is 
223          the expression. 
224   
225          The repl_dict contains parts that were already matched. For example 
226          here: 
227   
228          >>> (x+sin(b)*c)._matches_commutative(x+sin(y)*z, repl_dict={a: x}) 
229          {a_: x, b_: y, c_: z} 
230   
231          the only function of the repl_dict is to return it in the 
232          result, e.g. if you omit it: 
233   
234          >>> (x+sin(b)*c)._matches_commutative(x+sin(y)*z) 
235          {b_: y, c_: z} 
236   
237          the "a: x" is not returned in the result, but otherwise it is 
238          equivalent. 
239   
240          """
241          from .function import _coeff_isneg
242          # make sure expr is Expr if pattern is Expr
243          from .expr import Expr
244          if isinstance(self, Expr) and not isinstance(expr, Expr):
245              return None
246  
247          if repl_dict is None:
248              repl_dict = {}
249  
250          # handle simple patterns
251          if self == expr:
252              return repl_dict
253  
254          d = self._matches_simple(expr, repl_dict)
255          if d is not None:
256              return d
257  
258          # eliminate exact part from pattern: (2+a+w1+w2).matches(expr) -> (w1+w2).matches(expr-a-2)
259          from .function import WildFunction
260          from .symbol import Wild
261          wild_part, exact_part = sift(self.args, lambda p:
262              p.has(Wild, WildFunction) and not expr.has(p),
263              binary=True)
264          if not exact_part:
265              wild_part = list(ordered(wild_part))
266              if self.is_Add:
267                  # in addition to normal ordered keys, impose
268                  # sorting on Muls with leading Number to put
269                  # them in order
270                  wild_part = sorted(wild_part, key=lambda x:
271                      x.args[0] if x.is_Mul and x.args[0].is_Number else
272                      0)
273          else:
274              exact = self._new_rawargs(*exact_part)
275              free = expr.free_symbols
276              if free and (exact.free_symbols - free):
277                  # there are symbols in the exact part that are not
278                  # in the expr; but if there are no free symbols, let
279                  # the matching continue
280                  return None
281              newexpr = self._combine_inverse(expr, exact)
282              if not old and (expr.is_Add or expr.is_Mul):
283                  check = newexpr
284                  if _coeff_isneg(check):
285                      check = -check
286                  if check.count_ops() > expr.count_ops():
287                      return None
288              newpattern = self._new_rawargs(*wild_part)
289              return newpattern.matches(newexpr, repl_dict)
290  
291          # now to real work ;)
292          i = 0
293          saw = set()
294          while expr not in saw:
295              saw.add(expr)
296              args = tuple(ordered(self.make_args(expr)))
297              if self.is_Add and expr.is_Add:
298                  # in addition to normal ordered keys, impose
299                  # sorting on Muls with leading Number to put
300                  # them in order
301                  args = tuple(sorted(args, key=lambda x:
302                      x.args[0] if x.is_Mul and x.args[0].is_Number else
303                      0))
304              expr_list = (self.identity,) + args
305              for last_op in reversed(expr_list):
306                  for w in reversed(wild_part):
307                      d1 = w.matches(last_op, repl_dict)
308                      if d1 is not None:
309                          d2 = self.xreplace(d1).matches(expr, d1)
310                          if d2 is not None:
311                              return d2
312  
313              if i == 0:
314                  if self.is_Mul:
315                      # make e**i look like Mul
316                      if expr.is_Pow and expr.exp.is_Integer:
317                          from .mul import Mul
318                          if expr.exp > 0:
319                              expr = Mul(*[expr.base, expr.base**(expr.exp - 1)], evaluate=False)
320                          else:
321                              expr = Mul(*[1/expr.base, expr.base**(expr.exp + 1)], evaluate=False)
322                          i += 1
323                          continue
324  
325                  elif self.is_Add:
326                      # make i*e look like Add
327                      c, e = expr.as_coeff_Mul()
328                      if abs(c) > 1:
329                          from .add import Add
330                          if c > 0:
331                              expr = Add(*[e, (c - 1)*e], evaluate=False)
332                          else:
333                              expr = Add(*[-e, (c + 1)*e], evaluate=False)
334                          i += 1
335                          continue
336  
337                      # try collection on non-Wild symbols
338                      from sympy.simplify.radsimp import collect
339                      was = expr
340                      did = set()
341                      for w in reversed(wild_part):
342                          c, w = w.as_coeff_mul(Wild)
343                          free = c.free_symbols - did
344                          if free:
345                              did.update(free)
346                              expr = collect(expr, free)
347                      if expr != was:
348                          i += 0
349                          continue
350  
351                  break  # if we didn't continue, there is nothing more to do
352  
353          return

355      def _has_matcher(self):
356          """Helper for .has() that checks for containment of 
357          subexpressions within an expr by using sets of args 
358          of similar nodes, e.g. x + 1 in x + y + 1 checks 
359          to see that {x, 1} & {x, y, 1} == {x, 1} 
360          """
361          def _ncsplit(expr):
362              # this is not the same as args_cnc because here
363              # we don't assume expr is a Mul -- hence deal with args --
364              # and always return a set.
365              cpart, ncpart = sift(expr.args,
366                  lambda arg: arg.is_commutative is True, binary=True)
367              return set(cpart), ncpart
368  
369          c, nc = _ncsplit(self)
370          cls = self.__class__
371  
372          def is_in(expr):
373              if isinstance(expr, cls):
374                  if expr == self:
375                      return True
376                  _c, _nc = _ncsplit(expr)
377                  if (c & _c) == c:
378                      if not nc:
379                          return True
380                      elif len(nc) <= len(_nc):
381                          for i in range(len(_nc) - len(nc) + 1):
382                              if _nc[i:i + len(nc)] == nc:
383                                  return True
384              return False
385          return is_in

387      def _eval_evalf(self, prec):
388          """ 
389          Evaluate the parts of self that are numbers; if the whole thing 
390          was a number with no functions it would have been evaluated, but 
391          it wasn't so we must judiciously extract the numbers and reconstruct 
392          the object. This is *not* simply replacing numbers with evaluated 
393          numbers. Numbers should be handled in the largest pure-number 
394          expression as possible. So the code below separates ``self`` into 
395          number and non-number parts and evaluates the number parts and 
396          walks the args of the non-number part recursively (doing the same 
397          thing). 
398          """
399          from .add import Add
400          from .mul import Mul
401          from .symbol import Symbol
402          from .function import AppliedUndef
403          if isinstance(self, (Mul, Add)):
404              x, tail = self.as_independent(Symbol, AppliedUndef)
405              # if x is an AssocOp Function then the _evalf below will
406              # call _eval_evalf (here) so we must break the recursion
407              if not (tail is self.identity or
408                      isinstance(x, AssocOp) and x.is_Function or
409                      x is self.identity and isinstance(tail, AssocOp)):
410                  # here, we have a number so we just call to _evalf with prec;
411                  # prec is not the same as n, it is the binary precision so
412                  # that's why we don't call to evalf.
413                  x = x._evalf(prec) if x is not self.identity else self.identity
414                  args = []
415                  tail_args = tuple(self.func.make_args(tail))
416                  for a in tail_args:
417                      # here we call to _eval_evalf since we don't know what we
418                      # are dealing with and all other _eval_evalf routines should
419                      # be doing the same thing (i.e. taking binary prec and
420                      # finding the evalf-able args)
421                      newa = a._eval_evalf(prec)
422                      if newa is None:
423                          args.append(a)
424                      else:
425                          args.append(newa)
426                  return self.func(x, *args)
427  
428          # this is the same as above, but there were no pure-number args to
429          # deal with
430          args = []
431          for a in self.args:
432              newa = a._eval_evalf(prec)
433              if newa is None:
434                  args.append(a)
435              else:
436                  args.append(newa)
437          return self.func(*args)

452      @classmethod
453      def make_args(cls: type[Basic], expr: Basic) -> tuple[Basic, ...]:
454          """ 
455          Return a sequence of elements `args` such that cls(*args) == expr 
456   
457          Examples 
458          ======== 
459   
460          >>> from sympy import Symbol, Mul, Add 
461          >>> x, y = map(Symbol, 'xy') 
462   
463          >>> Mul.make_args(x*y) 
464          (x, y) 
465          >>> Add.make_args(x*y)

468          True 
469   
470          """
471          if isinstance(expr, cls):
472              return expr.args
473          else:
474              return (sympify(expr),)

477          if hints.get('deep', True):
478              terms = [term.doit(**hints) for term in self.args]
479          else:
480              terms = self.args
481          return self.func(*terms, evaluate=True)

487  class LatticeOp(AssocOp):
488      """ 
489      Join/meet operations of an algebraic lattice[1]. 
490   
491      Explanation 
492      =========== 
493   
494      These binary operations are associative (op(op(a, b), c) = op(a, op(b, c))), 
495      commutative (op(a, b) = op(b, a)) and idempotent (op(a, a) = op(a) = a). 
496      Common examples are AND, OR, Union, Intersection, max or min. They have an 
497      identity element (op(identity, a) = a) and an absorbing element 
498      conventionally called zero (op(zero, a) = zero). 
499   
500      This is an abstract base class, concrete derived classes must declare 
501      attributes zero and identity. All defining properties are then respected. 
502   
503      Examples 
504      ======== 
505   
506      >>> from sympy import Integer 
507      >>> from sympy.core.operations import LatticeOp 
508      >>> class my_join(LatticeOp): 
509      ...     zero = Integer(0) 
510      ...     identity = Integer(1) 
511      >>> my_join(2, 3) == my_join(3, 2) 
512      True 
513      >>> my_join(2, my_join(3, 4)) == my_join(2, 3, 4) 
514      True 
515      >>> my_join(0, 1, 4, 2, 3, 4) 
516      0 
517      >>> my_join(1, 2) 
518      2 
519   
520      References 
521      ========== 
522   
523      .. [1] https://en.wikipedia.org/wiki/Lattice_%28order%29 
524      """
525  
526      is_commutative = True
527  
528      def __new__(cls, *args, **options):
529          args = (_sympify_(arg) for arg in args)
530  
531          try:
532              # /!\ args is a generator and _new_args_filter
533              # must be careful to handle as such; this
534              # is done so short-circuiting can be done
535              # without having to sympify all values
536              _args = frozenset(cls._new_args_filter(args))
537          except ShortCircuit:
538              return sympify(cls.zero)
539          if not _args:
540              return sympify(cls.identity)
541          elif len(_args) == 1:
542              return set(_args).pop()
543          else:
544              # XXX in almost every other case for __new__, *_args is
545              # passed along, but the expectation here is for _args
546              obj = super(AssocOp, cls).__new__(cls, *ordered(_args))
547              obj._argset = _args
548              return obj
549  
550      @classmethod
551      def _new_args_filter(cls, arg_sequence, call_cls=None):
552          """Generator filtering args"""
553          ncls = call_cls or cls
554          for arg in arg_sequence:
555              if arg == ncls.zero:
556                  raise ShortCircuit(arg)
557              elif arg == ncls.identity:
558                  continue
559              elif arg.func == ncls:
560                  yield from arg.args
561              else:
562                  yield arg
563  
564      @classmethod
565      def make_args(cls, expr):
566          """ 
567          Return a set of args such that cls(*arg_set) == expr. 
568          """
569          if isinstance(expr, cls):
570              return expr._argset
571          else:
572              return frozenset([sympify(expr)])

575  class AssocOpDispatcher:
576      """ 
577      Handler dispatcher for associative operators 
578   
579      .. notes:: 
580         This approach is experimental, and can be replaced or deleted in the future. 
581         See https://github.com/sympy/sympy/pull/19463. 
582   
583      Explanation 
584      =========== 
585   
586      If arguments of different types are passed, the classes which handle the operation for each type 
587      are collected. Then, a class which performs the operation is selected by recursive binary dispatching. 
588      Dispatching relation can be registered by ``register_handlerclass`` method. 
589   
590      Priority registration is unordered. You cannot make ``A*B`` and ``B*A`` refer to 
591      different handler classes. All logic dealing with the order of arguments must be implemented 
592      in the handler class. 
593   
594      Examples 
595      ======== 
596   
597      >>> from sympy import Add, Expr, Symbol 
598      >>> from sympy.core.add import add 
599   
600      >>> class NewExpr(Expr): 
601      ...     @property 
602      ...     def _add_handler(self): 
603      ...         return NewAdd 
604      >>> class NewAdd(NewExpr, Add): 
605      ...     pass 
606      >>> add.register_handlerclass((Add, NewAdd), NewAdd) 
607   
608      >>> a, b = Symbol('a'), NewExpr() 
609      >>> add(a, b) == NewAdd(a, b) 
610      True 
611   
612      """
613      def __init__(self, name, doc=None):
614          self.name = name
615          self.doc = doc
616          self.handlerattr = "_%s_handler" % name
617          self._handlergetter = attrgetter(self.handlerattr)
618          self._dispatcher = Dispatcher(name)
619  
620      def __repr__(self):
621          return "<dispatched %s>" % self.name
622  
623      def register_handlerclass(self, classes, typ, on_ambiguity=ambiguity_register_error_ignore_dup):
624          """ 
625          Register the handler class for two classes, in both straight and reversed order. 
626   
627          Paramteters 
628          =========== 
629   
630          classes : tuple of two types 
631              Classes who are compared with each other. 
632   
633          typ: 
634              Class which is registered to represent *cls1* and *cls2*. 
635              Handler method of *self* must be implemented in this class. 
636          """
637          if not len(classes) == 2:
638              raise RuntimeError(
639                  "Only binary dispatch is supported, but got %s types: <%s>." % (
640                  len(classes), str_signature(classes)
641              ))
642          if len(set(classes)) == 1:
643              raise RuntimeError(
644                  "Duplicate types <%s> cannot be dispatched." % str_signature(classes)
645              )
646          self._dispatcher.add(tuple(classes), typ, on_ambiguity=on_ambiguity)
647          self._dispatcher.add(tuple(reversed(classes)), typ, on_ambiguity=on_ambiguity)
648  
649      @cacheit
650      def __call__(self, *args, _sympify=True, **kwargs):
651          """ 
652          Parameters 
653          ========== 
654   
655          *args : 
656              Arguments which are operated 
657          """
658          if _sympify:
659              args = tuple(map(_sympify_, args))
660          handlers = frozenset(map(self._handlergetter, args))
661  
662          # no need to sympify again
663          return self.dispatch(handlers)(*args, _sympify=False, **kwargs)
664  
665      @cacheit
666      def dispatch(self, handlers):
667          """ 
668          Select the handler class, and return its handler method. 
669          """
670  
671          # Quick exit for the case where all handlers are same
672          if len(handlers) == 1:
673              h, = handlers
674              if not isinstance(h, type):
675                  raise RuntimeError("Handler {!r} is not a type.".format(h))
676              return h
677  
678          # Recursively select with registered binary priority
679          for i, typ in enumerate(handlers):
680  
681              if not isinstance(typ, type):
682                  raise RuntimeError("Handler {!r} is not a type.".format(typ))
683  
684              if i == 0:
685                  handler = typ
686              else:
687                  prev_handler = handler
688                  handler = self._dispatcher.dispatch(prev_handler, typ)
689  
690                  if not isinstance(handler, type):
691                      raise RuntimeError(
692                          "Dispatcher for {!r} and {!r} must return a type, but got {!r}".format(
693                          prev_handler, typ, handler
694                      ))
695  
696          # return handler class
697          return handler
698  
699      @property
700      def __doc__(self):
701          docs = [
702              "Multiply dispatched associative operator: %s" % self.name,
703              "Note that support for this is experimental, see the docs for :class:`AssocOpDispatcher` for details"
704          ]
705  
706          if self.doc:
707              docs.append(self.doc)
708  
709          s = "Registered handler classes\n"
710          s += '=' * len(s)
711          docs.append(s)
712  
713          amb_sigs = []
714  
715          typ_sigs = defaultdict(list)
716          for sigs in self._dispatcher.ordering[::-1]:
717              key = self._dispatcher.funcs[sigs]
718              typ_sigs[key].append(sigs)
719  
720          for typ, sigs in typ_sigs.items():
721  
722              sigs_str = ', '.join('<%s>' % str_signature(sig) for sig in sigs)
723  
724              if isinstance(typ, RaiseNotImplementedError):
725                  amb_sigs.append(sigs_str)
726                  continue
727  
728              s = 'Inputs: %s\n' % sigs_str
729              s += '-' * len(s) + '\n'
730              s += typ.__name__
731              docs.append(s)
732  
733          if amb_sigs:
734              s = "Ambiguous handler classes\n"
735              s += '=' * len(s)
736              docs.append(s)
737  
738              s = '\n'.join(amb_sigs)
739              docs.append(s)
740  
741          return '\n\n'.join(docs)
