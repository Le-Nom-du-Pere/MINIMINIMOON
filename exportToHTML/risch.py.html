1    """ 
2    The Risch Algorithm for transcendental function integration. 
3     
4    The core algorithms for the Risch algorithm are here.  The subproblem 
5    algorithms are in the rde.py and prde.py files for the Risch 
6    Differential Equation solver and the parametric problems solvers, 
7    respectively.  All important information concerning the differential extension 
8    for an integrand is stored in a DifferentialExtension object, which in the code 
9    is usually called DE.  Throughout the code and Inside the DifferentialExtension 
10   object, the conventions/attribute names are that the base domain is QQ and each 
11   differential extension is x, t0, t1, ..., tn-1 = DE.t. DE.x is the variable of 
12   integration (Dx == 1), DE.D is a list of the derivatives of 
13   x, t1, t2, ..., tn-1 = t, DE.T is the list [x, t1, t2, ..., tn-1], DE.t is the 
14   outer-most variable of the differential extension at the given level (the level 
15   can be adjusted using DE.increment_level() and DE.decrement_level()), 
16   k is the field C(x, t0, ..., tn-2), where C is the constant field.  The 
17   numerator of a fraction is denoted by a and the denominator by 
18   d.  If the fraction is named f, fa == numer(f) and fd == denom(f). 
19   Fractions are returned as tuples (fa, fd).  DE.d and DE.t are used to 
20   represent the topmost derivation and extension variable, respectively. 
21   The docstring of a function signifies whether an argument is in k[t], in 
22   which case it will just return a Poly in t, or in k(t), in which case it 
23   will return the fraction (fa, fd). Other variable names probably come 
24   from the names used in Bronstein's book. 
25   """

53   def integer_powers(exprs):
54       """ 
55       Rewrites a list of expressions as integer multiples of each other. 
56    
57       Explanation 
58       =========== 
59    
60       For example, if you have [x, x/2, x**2 + 1, 2*x/3], then you can rewrite 
61       this as [(x/6) * 6, (x/6) * 3, (x**2 + 1) * 1, (x/6) * 4]. This is useful 
62       in the Risch integration algorithm, where we must write exp(x) + exp(x/2) 
63       as (exp(x/2))**2 + exp(x/2), but not as exp(x) + sqrt(exp(x)) (this is 
64       because only the transcendental case is implemented and we therefore cannot 
65       integrate algebraic extensions). The integer multiples returned by this 
66       function for each term are the smallest possible (their content equals 1). 
67    
68       Returns a list of tuples where the first element is the base term and the 
69       second element is a list of `(item, factor)` terms, where `factor` is the 
70       integer multiplicative factor that must multiply the base term to obtain 
71       the original item. 
72    
73       The easiest way to understand this is to look at an example: 
74    
75       >>> from sympy.abc import x 
76       >>> from sympy.integrals.risch import integer_powers 
77       >>> integer_powers([x, x/2, x**2 + 1, 2*x/3]) 
78       [(x/6, [(x, 6), (x/2, 3), (2*x/3, 4)]), (x**2 + 1, [(x**2 + 1, 1)])] 
79    
80       We can see how this relates to the example at the beginning of the 
81       docstring.  It chose x/6 as the first base term.  Then, x can be written as 
82       (x/2) * 2, so we get (0, 2), and so on. Now only element (x**2 + 1) 
83       remains, and there are no other terms that can be written as a rational 
84       multiple of that, so we get that it can be written as (x**2 + 1) * 1. 
85    
86       """
87       # Here is the strategy:
88   
89       # First, go through each term and determine if it can be rewritten as a
90       # rational multiple of any of the terms gathered so far.
91       # cancel(a/b).is_Rational is sufficient for this.  If it is a multiple, we
92       # add its multiple to the dictionary.
93   
94       terms = {}
95       for term in exprs:
96           for trm, trm_list in terms.items():
97               a = cancel(term/trm)
98               if a.is_Rational:
99                   trm_list.append((term, a))
100                  break
101          else:
102              terms[term] = [(term, S.One)]
103  
104      # After we have done this, we have all the like terms together, so we just
105      # need to find a common denominator so that we can get the base term and
106      # integer multiples such that each term can be written as an integer
107      # multiple of the base term, and the content of the integers is 1.
108  
109      newterms = {}
110      for term, term_list in terms.items():
111          common_denom = reduce(ilcm, [i.as_numer_denom()[1] for _, i in
112              term_list])
113          newterm = term/common_denom
114          newmults = [(i, j*common_denom) for i, j in term_list]
115          newterms[newterm] = newmults
116  
117      return sorted(iter(newterms.items()), key=lambda item: item[0].sort_key())

120  class DifferentialExtension:
121      """ 
122      A container for all the information relating to a differential extension. 
123   
124      Explanation 
125      =========== 
126   
127      The attributes of this object are (see also the docstring of __init__): 
128   
129      - f: The original (Expr) integrand. 
130      - x: The variable of integration. 
131      - T: List of variables in the extension. 
132      - D: List of derivations in the extension; corresponds to the elements of T. 
133      - fa: Poly of the numerator of the integrand. 
134      - fd: Poly of the denominator of the integrand. 
135      - Tfuncs: Lambda() representations of each element of T (except for x). 
136        For back-substitution after integration. 
137      - backsubs: A (possibly empty) list of further substitutions to be made on 
138        the final integral to make it look more like the integrand. 
139      - exts: 
140      - extargs: 
141      - cases: List of string representations of the cases of T. 
142      - t: The top level extension variable, as defined by the current level 
143        (see level below). 
144      - d: The top level extension derivation, as defined by the current 
145        derivation (see level below). 
146      - case: The string representation of the case of self.d. 
147      (Note that self.T and self.D will always contain the complete extension, 
148      regardless of the level.  Therefore, you should ALWAYS use DE.t and DE.d 
149      instead of DE.T[-1] and DE.D[-1].  If you want to have a list of the 
150      derivations or variables only up to the current level, use 
151      DE.D[:len(DE.D) + DE.level + 1] and DE.T[:len(DE.T) + DE.level + 1].  Note 
152      that, in particular, the derivation() function does this.) 
153   
154      The following are also attributes, but will probably not be useful other 
155      than in internal use: 
156      - newf: Expr form of fa/fd. 
157      - level: The number (between -1 and -len(self.T)) such that 
158        self.T[self.level] == self.t and self.D[self.level] == self.d. 
159        Use the methods self.increment_level() and self.decrement_level() to change 
160        the current level. 
161      """
162      # __slots__ is defined mainly so we can iterate over all the attributes
163      # of the class easily (the memory use doesn't matter too much, since we
164      # only create one DifferentialExtension per integration).  Also, it's nice
165      # to have a safeguard when debugging.
166      __slots__ = ('f', 'x', 'T', 'D', 'fa', 'fd', 'Tfuncs', 'backsubs',
167          'exts', 'extargs', 'cases', 'case', 't', 'd', 'newf', 'level',
168          'ts', 'dummy')

170      def __init__(self, f=None, x=None, handle_first='log', dummy=False, extension=None, rewrite_complex=None):
171          """ 
172          Tries to build a transcendental extension tower from ``f`` with respect to ``x``. 
173   
174          Explanation 
175          =========== 
176   
177          If it is successful, creates a DifferentialExtension object with, among 
178          others, the attributes fa, fd, D, T, Tfuncs, and backsubs such that 
179          fa and fd are Polys in T[-1] with rational coefficients in T[:-1], 
180          fa/fd == f, and D[i] is a Poly in T[i] with rational coefficients in 
181          T[:i] representing the derivative of T[i] for each i from 1 to len(T). 
182          Tfuncs is a list of Lambda objects for back replacing the functions 
183          after integrating.  Lambda() is only used (instead of lambda) to make 
184          them easier to test and debug. Note that Tfuncs corresponds to the 
185          elements of T, except for T[0] == x, but they should be back-substituted 
186          in reverse order.  backsubs is a (possibly empty) back-substitution list 
187          that should be applied on the completed integral to make it look more 
188          like the original integrand. 
189   
190          If it is unsuccessful, it raises NotImplementedError. 
191   
192          You can also create an object by manually setting the attributes as a 
193          dictionary to the extension keyword argument.  You must include at least 
194          D.  Warning, any attribute that is not given will be set to None. The 
195          attributes T, t, d, cases, case, x, and level are set automatically and 
196          do not need to be given.  The functions in the Risch Algorithm will NOT 
197          check to see if an attribute is None before using it.  This also does not 
198          check to see if the extension is valid (non-algebraic) or even if it is 
199          self-consistent.  Therefore, this should only be used for 
200          testing/debugging purposes. 
201          """
202          # XXX: If you need to debug this function, set the break point here
203  
204          if extension:
205              if 'D' not in extension:
206                  raise ValueError("At least the key D must be included with "
207                      "the extension flag to DifferentialExtension.")
208              for attr in extension:
209                  setattr(self, attr, extension[attr])
210  
211              self._auto_attrs()
212  
213              return
214          elif f is None or x is None:
215              raise ValueError("Either both f and x or a manual extension must "
216              "be given.")
217  
218          if handle_first not in ('log', 'exp'):
219              raise ValueError("handle_first must be 'log' or 'exp', not %s." %
220                  str(handle_first))
221  
222          # f will be the original function, self.f might change if we reset
223          # (e.g., we pull out a constant from an exponential)
224          self.f = f
225          self.x = x
226          # setting the default value 'dummy'
227          self.dummy = dummy
228          self.reset()
229          exp_new_extension, log_new_extension = True, True
230  
231          # case of 'automatic' choosing
232          if rewrite_complex is None:
233              rewrite_complex = I in self.f.atoms()
234  
235          if rewrite_complex:
236              rewritables = {
237                  (sin, cos, cot, tan, sinh, cosh, coth, tanh): exp,
238                  (asin, acos, acot, atan): log,
239              }
240              # rewrite the trigonometric components
241              for candidates, rule in rewritables.items():
242                  self.newf = self.newf.rewrite(candidates, rule)
243              self.newf = cancel(self.newf)
244          else:
245              if any(i.has(x) for i in self.f.atoms(sin, cos, tan, atan, asin, acos)):
246                  raise NotImplementedError("Trigonometric extensions are not "
247                  "supported (yet!)")
248  
249          exps = set()
250          pows = set()
251          numpows = set()
252          sympows = set()
253          logs = set()
254          symlogs = set()
255  
256          while True:
257              if self.newf.is_rational_function(*self.T):
258                  break
259  
260              if not exp_new_extension and not log_new_extension:
261                  # We couldn't find a new extension on the last pass, so I guess
262                  # we can't do it.
263                  raise NotImplementedError("Couldn't find an elementary "
264                      "transcendental extension for %s.  Try using a " % str(f) +
265                      "manual extension with the extension flag.")
266  
267              exps, pows, numpows, sympows, log_new_extension = \
268                      self._rewrite_exps_pows(exps, pows, numpows, sympows, log_new_extension)
269  
270              logs, symlogs = self._rewrite_logs(logs, symlogs)
271  
272              if handle_first == 'exp' or not log_new_extension:
273                  exp_new_extension = self._exp_part(exps)
274                  if exp_new_extension is None:
275                      # reset and restart
276                      self.f = self.newf
277                      self.reset()
278                      exp_new_extension = True
279                      continue
280  
281              if handle_first == 'log' or not exp_new_extension:
282                  log_new_extension = self._log_part(logs)
283  
284          self.fa, self.fd = frac_in(self.newf, self.t)
285          self._auto_attrs()
286  
287          return
288

295      def _rewrite_exps_pows(self, exps, pows, numpows,
296              sympows, log_new_extension):
297          """ 
298          Rewrite exps/pows for better processing. 
299          """
300          from .prde import is_deriv_k
301  
302          # Pre-preparsing.
303          #################
304          # Get all exp arguments, so we can avoid ahead of time doing
305          # something like t1 = exp(x), t2 = exp(x/2) == sqrt(t1).
306  
307          # Things like sqrt(exp(x)) do not automatically simplify to
308          # exp(x/2), so they will be viewed as algebraic.  The easiest way
309          # to handle this is to convert all instances of exp(a)**Rational
310          # to exp(Rational*a) before doing anything else.  Note that the
311          # _exp_part code can generate terms of this form, so we do need to
312          # do this at each pass (or else modify it to not do that).
313  
314          ratpows = [i for i in self.newf.atoms(Pow)
315                     if (isinstance(i.base, exp) and i.exp.is_Rational)]
316  
317          ratpows_repl = [
318              (i, i.base.base**(i.exp*i.base.exp)) for i in ratpows]
319          self.backsubs += [(j, i) for i, j in ratpows_repl]
320          self.newf = self.newf.xreplace(dict(ratpows_repl))
321  
322          # To make the process deterministic, the args are sorted
323          # so that functions with smaller op-counts are processed first.
324          # Ties are broken with the default_sort_key.
325  
326          # XXX Although the method is deterministic no additional work
327          # has been done to guarantee that the simplest solution is
328          # returned and that it would be affected be using different
329          # variables. Though it is possible that this is the case
330          # one should know that it has not been done intentionally, so
331          # further improvements may be possible.
332  
333          # TODO: This probably doesn't need to be completely recomputed at
334          # each pass.
335          exps = update_sets(exps, self.newf.atoms(exp),
336              lambda i: i.exp.is_rational_function(*self.T) and
337              i.exp.has(*self.T))
338          pows = update_sets(pows, self.newf.atoms(Pow),
339              lambda i: i.exp.is_rational_function(*self.T) and
340              i.exp.has(*self.T))
341          numpows = update_sets(numpows, set(pows),
342              lambda i: not i.base.has(*self.T))
343          sympows = update_sets(sympows, set(pows) - set(numpows),
344              lambda i: i.base.is_rational_function(*self.T) and
345              not i.exp.is_Integer)
346  
347          # The easiest way to deal with non-base E powers is to convert them
348          # into base E, integrate, and then convert back.
349          for i in ordered(pows):
350              old = i
351              new = exp(i.exp*log(i.base))
352              # If exp is ever changed to automatically reduce exp(x*log(2))
353              # to 2**x, then this will break.  The solution is to not change
354              # exp to do that :)
355              if i in sympows:
356                  if i.exp.is_Rational:
357                      raise NotImplementedError("Algebraic extensions are "
358                          "not supported (%s)." % str(i))
359                  # We can add a**b only if log(a) in the extension, because
360                  # a**b == exp(b*log(a)).
361                  basea, based = frac_in(i.base, self.t)
362                  A = is_deriv_k(basea, based, self)
363                  if A is None:
364                      # Nonelementary monomial (so far)
365  
366                      # TODO: Would there ever be any benefit from just
367                      # adding log(base) as a new monomial?
368                      # ANSWER: Yes, otherwise we can't integrate x**x (or
369                      # rather prove that it has no elementary integral)
370                      # without first manually rewriting it as exp(x*log(x))
371                      self.newf = self.newf.xreplace({old: new})
372                      self.backsubs += [(new, old)]
373                      log_new_extension = self._log_part([log(i.base)])
374                      exps = update_sets(exps, self.newf.atoms(exp), lambda i:
375                          i.exp.is_rational_function(*self.T) and i.exp.has(*self.T))
376                      continue
377                  ans, u, const = A
378                  newterm = exp(i.exp*(log(const) + u))
379                  # Under the current implementation, exp kills terms
380                  # only if they are of the form a*log(x), where a is a
381                  # Number.  This case should have already been killed by the
382                  # above tests.  Again, if this changes to kill more than
383                  # that, this will break, which maybe is a sign that you
384                  # shouldn't be changing that.  Actually, if anything, this
385                  # auto-simplification should be removed.  See
386                  # https://groups.google.com/group/sympy/browse_thread/thread/a61d48235f16867f
387  
388                  self.newf = self.newf.xreplace({i: newterm})
389  
390              elif i not in numpows:
391                  continue
392              else:
393                  # i in numpows
394                  newterm = new
395              # TODO: Just put it in self.Tfuncs
396              self.backsubs.append((new, old))
397              self.newf = self.newf.xreplace({old: newterm})
398              exps.append(newterm)
399  
400          return exps, pows, numpows, sympows, log_new_extension

402      def _rewrite_logs(self, logs, symlogs):
403          """ 
404          Rewrite logs for better processing. 
405          """
406          atoms = self.newf.atoms(log)
407          logs = update_sets(logs, atoms,
408              lambda i: i.args[0].is_rational_function(*self.T) and
409              i.args[0].has(*self.T))
410          symlogs = update_sets(symlogs, atoms,
411              lambda i: i.has(*self.T) and i.args[0].is_Pow and
412              i.args[0].base.is_rational_function(*self.T) and
413              not i.args[0].exp.is_Integer)
414  
415          # We can handle things like log(x**y) by converting it to y*log(x)
416          # This will fix not only symbolic exponents of the argument, but any
417          # non-Integer exponent, like log(sqrt(x)).  The exponent can also
418          # depend on x, like log(x**x).
419          for i in ordered(symlogs):
420              # Unlike in the exponential case above, we do not ever
421              # potentially add new monomials (above we had to add log(a)).
422              # Therefore, there is no need to run any is_deriv functions
423              # here.  Just convert log(a**b) to b*log(a) and let
424              # log_new_extension() handle it from there.
425              lbase = log(i.args[0].base)
426              logs.append(lbase)
427              new = i.args[0].exp*lbase
428              self.newf = self.newf.xreplace({i: new})
429              self.backsubs.append((new, i))
430  
431          # remove any duplicates
432          logs = sorted(set(logs), key=default_sort_key)
433  
434          return logs, symlogs

436      def _auto_attrs(self):
437          """ 
438          Set attributes that are generated automatically. 
439          """
440          if not self.T:
441              # i.e., when using the extension flag and T isn't given
442              self.T = [i.gen for i in self.D]
443          if not self.x:
444              self.x = self.T[0]
445          self.cases = [get_case(d, t) for d, t in zip(self.D, self.T)]
446          self.level = -1
447          self.t = self.T[self.level]
448          self.d = self.D[self.level]
449          self.case = self.cases[self.level]

451      def _exp_part(self, exps):
452          """ 
453          Try to build an exponential extension. 
454   
455          Returns 
456          ======= 
457   
458          Returns True if there was a new extension, False if there was no new 
459          extension but it was able to rewrite the given exponentials in terms 
460          of the existing extension, and None if the entire extension building 
461          process should be restarted.  If the process fails because there is no 
462          way around an algebraic extension (e.g., exp(log(x)/2)), it will raise 
463          NotImplementedError. 
464          """
465          from .prde import is_log_deriv_k_t_radical
466          new_extension = False
467          restart = False
468          expargs = [i.exp for i in exps]
469          ip = integer_powers(expargs)
470          for arg, others in ip:
471              # Minimize potential problems with algebraic substitution
472              others.sort(key=lambda i: i[1])
473  
474              arga, argd = frac_in(arg, self.t)
475              A = is_log_deriv_k_t_radical(arga, argd, self)
476  
477              if A is not None:
478                  ans, u, n, const = A
479                  # if n is 1 or -1, it's algebraic, but we can handle it
480                  if n == -1:
481                      # This probably will never happen, because
482                      # Rational.as_numer_denom() returns the negative term in
483                      # the numerator.  But in case that changes, reduce it to
484                      # n == 1.
485                      n = 1
486                      u **= -1
487                      const *= -1
488                      ans = [(i, -j) for i, j in ans]
489  
490                  if n == 1:
491                      # Example: exp(x + x**2) over QQ(x, exp(x), exp(x**2))
492                      self.newf = self.newf.xreplace({exp(arg): exp(const)*Mul(*[
493                          u**power for u, power in ans])})
494                      self.newf = self.newf.xreplace({exp(p*exparg):
495                          exp(const*p) * Mul(*[u**power for u, power in ans])
496                          for exparg, p in others})
497                      # TODO: Add something to backsubs to put exp(const*p)
498                      # back together.
499  
500                      continue
501  
502                  else:
503                      # Bad news: we have an algebraic radical.  But maybe we
504                      # could still avoid it by choosing a different extension.
505                      # For example, integer_powers() won't handle exp(x/2 + 1)
506                      # over QQ(x, exp(x)), but if we pull out the exp(1), it
507                      # will.  Or maybe we have exp(x + x**2/2), over
508                      # QQ(x, exp(x), exp(x**2)), which is exp(x)*sqrt(exp(x**2)),
509                      # but if we use QQ(x, exp(x), exp(x**2/2)), then they will
510                      # all work.
511                      #
512                      # So here is what we do: If there is a non-zero const, pull
513                      # it out and retry.  Also, if len(ans) > 1, then rewrite
514                      # exp(arg) as the product of exponentials from ans, and
515                      # retry that.  If const == 0 and len(ans) == 1, then we
516                      # assume that it would have been handled by either
517                      # integer_powers() or n == 1 above if it could be handled,
518                      # so we give up at that point.  For example, you can never
519                      # handle exp(log(x)/2) because it equals sqrt(x).
520  
521                      if const or len(ans) > 1:
522                          rad = Mul(*[term**(power/n) for term, power in ans])
523                          self.newf = self.newf.xreplace({exp(p*exparg):
524                              exp(const*p)*rad for exparg, p in others})
525                          self.newf = self.newf.xreplace(dict(list(zip(reversed(self.T),
526                              reversed([f(self.x) for f in self.Tfuncs])))))
527                          restart = True
528                          break
529                      else:
530                          # TODO: give algebraic dependence in error string
531                          raise NotImplementedError("Cannot integrate over "
532                              "algebraic extensions.")
533  
534              else:
535                  arga, argd = frac_in(arg, self.t)
536                  darga = (argd*derivation(Poly(arga, self.t), self) -
537                      arga*derivation(Poly(argd, self.t), self))
538                  dargd = argd**2
539                  darga, dargd = darga.cancel(dargd, include=True)
540                  darg = darga.as_expr()/dargd.as_expr()
541                  self.t = next(self.ts)
542                  self.T.append(self.t)
543                  self.extargs.append(arg)
544                  self.exts.append('exp')
545                  self.D.append(darg.as_poly(self.t, expand=False)*Poly(self.t,
546                      self.t, expand=False))
547                  if self.dummy:
548                      i = Dummy("i")
549                  else:
550                      i = Symbol('i')
551                  self.Tfuncs += [Lambda(i, exp(arg.subs(self.x, i)))]
552                  self.newf = self.newf.xreplace(
553                          {exp(exparg): self.t**p for exparg, p in others})
554                  new_extension = True
555  
556          if restart:
557              return None
558          return new_extension

560      def _log_part(self, logs):
561          """ 
562          Try to build a logarithmic extension. 
563   
564          Returns 
565          ======= 
566   
567          Returns True if there was a new extension and False if there was no new 
568          extension but it was able to rewrite the given logarithms in terms 
569          of the existing extension.  Unlike with exponential extensions, there 
570          is no way that a logarithm is not transcendental over and cannot be 
571          rewritten in terms of an already existing extension in a non-algebraic 
572          way, so this function does not ever return None or raise 
573          NotImplementedError. 
574          """
575          from .prde import is_deriv_k
576          new_extension = False
577          logargs = [i.args[0] for i in logs]
578          for arg in ordered(logargs):
579              # The log case is easier, because whenever a logarithm is algebraic
580              # over the base field, it is of the form a1*t1 + ... an*tn + c,
581              # which is a polynomial, so we can just replace it with that.
582              # In other words, we don't have to worry about radicals.
583              arga, argd = frac_in(arg, self.t)
584              A = is_deriv_k(arga, argd, self)
585              if A is not None:
586                  ans, u, const = A
587                  newterm = log(const) + u
588                  self.newf = self.newf.xreplace({log(arg): newterm})
589                  continue
590  
591              else:
592                  arga, argd = frac_in(arg, self.t)
593                  darga = (argd*derivation(Poly(arga, self.t), self) -
594                      arga*derivation(Poly(argd, self.t), self))
595                  dargd = argd**2
596                  darg = darga.as_expr()/dargd.as_expr()
597                  self.t = next(self.ts)
598                  self.T.append(self.t)
599                  self.extargs.append(arg)
600                  self.exts.append('log')
601                  self.D.append(cancel(darg.as_expr()/arg).as_poly(self.t,
602                      expand=False))
603                  if self.dummy:
604                      i = Dummy("i")
605                  else:
606                      i = Symbol('i')
607                  self.Tfuncs += [Lambda(i, log(arg.subs(self.x, i)))]
608                  self.newf = self.newf.xreplace({log(arg): self.t})
609                  new_extension = True
610  
611          return new_extension

786  def gcdex_diophantine(a, b, c):
787      """ 
788      Extended Euclidean Algorithm, Diophantine version. 
789   
790      Explanation 
791      =========== 
792   
793      Given ``a``, ``b`` in K[x] and ``c`` in (a, b), the ideal generated by ``a`` and 
794      ``b``, return (s, t) such that s*a + t*b == c and either s == 0 or s.degree() 
795      < b.degree(). 
796      """
797      # Extended Euclidean Algorithm (Diophantine Version) pg. 13
798      # TODO: This should go in densetools.py.
799      # XXX: Better name?
800  
801      s, g = a.half_gcdex(b)
802      s *= c.exquo(g)  # Inexact division means c is not in (a, b)
803      if s and s.degree() >= b.degree():
804          _, s = s.div(b)
805      t = (c - s*a).exquo(b)
806      return (s, t)

809  def frac_in(f, t, *, cancel=False, **kwargs):
810      """ 
811      Returns the tuple (fa, fd), where fa and fd are Polys in t. 
812   
813      Explanation 
814      =========== 
815   
816      This is a common idiom in the Risch Algorithm functions, so we abstract 
817      it out here. ``f`` should be a basic expression, a Poly, or a tuple (fa, fd), 
818      where fa and fd are either basic expressions or Polys, and f == fa/fd. 
819      **kwargs are applied to Poly. 
820      """
821      if isinstance(f, tuple):
822          fa, fd = f
823          f = fa.as_expr()/fd.as_expr()
824      fa, fd = f.as_expr().as_numer_denom()
825      fa, fd = fa.as_poly(t, **kwargs), fd.as_poly(t, **kwargs)
826      if cancel:
827          fa, fd = fa.cancel(fd, include=True)
828      if fa is None or fd is None:
829          raise ValueError("Could not turn %s into a fraction in %s." % (f, t))
830      return (fa, fd)

879  def derivation(p, DE, coefficientD=False, basic=False):
880      """ 
881      Computes Dp. 
882   
883      Explanation 
884      =========== 
885   
886      Given the derivation D with D = d/dx and p is a polynomial in t over 
887      K(x), return Dp. 
888   
889      If coefficientD is True, it computes the derivation kD 
890      (kappaD), which is defined as kD(sum(ai*Xi**i, (i, 0, n))) == 
891      sum(Dai*Xi**i, (i, 1, n)) (Definition 3.2.2, page 80).  X in this case is 
892      T[-1], so coefficientD computes the derivative just with respect to T[:-1], 
893      with T[-1] treated as a constant. 
894   
895      If ``basic=True``, the returns a Basic expression.  Elements of D can still be 
896      instances of Poly. 
897      """
898      if basic:
899          r = 0
900      else:
901          r = Poly(0, DE.t)
902  
903      t = DE.t
904      if coefficientD:
905          if DE.level <= -len(DE.T):
906              # 'base' case, the answer is 0.
907              return r
908          DE.decrement_level()
909  
910      D = DE.D[:len(DE.D) + DE.level + 1]
911      T = DE.T[:len(DE.T) + DE.level + 1]
912  
913      for d, v in zip(D, T):
914          pv = p.as_poly(v)
915          if pv is None or basic:
916              pv = p.as_expr()
917  
918          if basic:
919              r += d.as_expr()*pv.diff(v)
920          else:
921              r += (d.as_expr()*pv.diff(v).as_expr()).as_poly(t)
922  
923      if basic:
924          r = cancel(r)
925      if coefficientD:
926          DE.increment_level()
927  
928      return r

932      """ 
933      Returns the type of the derivation d. 
934   
935      Returns one of {'exp', 'tan', 'base', 'primitive', 'other_linear', 
936      'other_nonlinear'}. 
937      """
938      if not d.expr.has(t):

948      return 'other_linear'

951  def splitfactor(p, DE, coefficientD=False, z=None):
952      """ 
953      Splitting factorization. 
954   
955      Explanation 
956      =========== 
957   
958      Given a derivation D on k[t] and ``p`` in k[t], return (p_n, p_s) in 
959      k[t] x k[t] such that p = p_n*p_s, p_s is special, and each square 
960      factor of p_n is normal. 
961   
962      Page. 100 
963      """
964      kinv = [1/x for x in DE.T[:DE.level]]
965      if z:
966          kinv.append(z)
967  
968      One = Poly(1, DE.t, domain=p.get_domain())
969      Dp = derivation(p, DE, coefficientD=coefficientD)
970      # XXX: Is this right?
971      if p.is_zero:
972          return (p, One)
973  
974      if not p.expr.has(DE.t):
975          s = p.as_poly(*kinv).gcd(Dp.as_poly(*kinv)).as_poly(DE.t)
976          n = p.exquo(s)
977          return (n, s)
978  
979      if not Dp.is_zero:
980          h = p.gcd(Dp).to_field()
981          g = p.gcd(p.diff(DE.t)).to_field()
982          s = h.exquo(g)
983  
984          if s.degree(DE.t) == 0:
985              return (p, One)
986  
987          q_split = splitfactor(p.exquo(s), DE, coefficientD=coefficientD)
988  
989          return (q_split[0], q_split[1]*s)
990      else:
991          return (p, One)

1058 def hermite_reduce(a, d, DE):
1059     """ 
1060     Hermite Reduction - Mack's Linear Version. 
1061  
1062     Given a derivation D on k(t) and f = a/d in k(t), returns g, h, r in 
1063     k(t) such that f = Dg + h + r, h is simple, and r is reduced. 
1064  
1065     """
1066     # Make d monic
1067     l = Poly(1/d.LC(), DE.t)
1068     a, d = a.mul(l), d.mul(l)
1069 
1070     fp, fs, fn = canonical_representation(a, d, DE)
1071     a, d = fn
1072     l = Poly(1/d.LC(), DE.t)
1073     a, d = a.mul(l), d.mul(l)
1074 
1075     ga = Poly(0, DE.t)
1076     gd = Poly(1, DE.t)
1077 
1078     dd = derivation(d, DE)
1079     dm = gcd(d.to_field(), dd.to_field()).as_poly(DE.t)
1080     ds, _ = d.div(dm)
1081 
1082     while dm.degree(DE.t) > 0:
1083 
1084         ddm = derivation(dm, DE)
1085         dm2 = gcd(dm.to_field(), ddm.to_field())
1086         dms, _ = dm.div(dm2)
1087         ds_ddm = ds.mul(ddm)
1088         ds_ddm_dm, _ = ds_ddm.div(dm)
1089 
1090         b, c = gcdex_diophantine(-ds_ddm_dm.as_poly(DE.t),
1091             dms.as_poly(DE.t), a.as_poly(DE.t))
1092         b, c = b.as_poly(DE.t), c.as_poly(DE.t)
1093 
1094         db = derivation(b, DE).as_poly(DE.t)
1095         ds_dms, _ = ds.div(dms)
1096         a = c.as_poly(DE.t) - db.mul(ds_dms).as_poly(DE.t)
1097 
1098         ga = ga*dm + b*gd
1099         gd = gd*dm
1100         ga, gd = ga.cancel(gd, include=True)
1101         dm = dm2
1102 
1103     q, r = a.div(ds)
1104     ga, gd = ga.cancel(gd, include=True)
1105 
1106     r, d = r.cancel(ds, include=True)
1107     rra = q*fs[1] + fp*fs[1] + fs[0]
1108     rrd = fs[1]
1109     rra, rrd = rra.cancel(rrd, include=True)
1110 
1111     return ((ga, gd), (r, d), (rra, rrd))

1116     Polynomial Reduction. 
1117  
1118     Explanation 
1119     =========== 
1120  
1121     Given a derivation D on k(t) and p in k[t] where t is a nonlinear 
1122     monomial over k, return q, r in k[t] such that p = Dq  + r, and

1136 def laurent_series(a, d, F, n, DE):
1137     """ 
1138     Contribution of ``F`` to the full partial fraction decomposition of A/D. 
1139  
1140     Explanation 
1141     =========== 
1142  
1143     Given a field K of characteristic 0 and ``A``,``D``,``F`` in K[x] with D monic, 
1144     nonzero, coprime with A, and ``F`` the factor of multiplicity n in the square- 
1145     free factorization of D, return the principal parts of the Laurent series of 
1146     A/D at all the zeros of ``F``. 
1147     """
1148     if F.degree()==0:
1149         return 0
1150     Z = _symbols('z', n)
1151     z = Symbol('z')
1152     Z.insert(0, z)
1153     delta_a = Poly(0, DE.t)
1154     delta_d = Poly(1, DE.t)
1155 
1156     E = d.quo(F**n)
1157     ha, hd = (a, E*Poly(z**n, DE.t))
1158     dF = derivation(F,DE)
1159     B, _ = gcdex_diophantine(E, F, Poly(1,DE.t))
1160     C, _ = gcdex_diophantine(dF, F, Poly(1,DE.t))
1161 
1162     # initialization
1163     F_store = F
1164     V, DE_D_list, H_list= [], [], []
1165 
1166     for j in range(0, n):
1167     # jth derivative of z would be substituted with dfnth/(j+1) where dfnth =(d^n)f/(dx)^n
1168         F_store = derivation(F_store, DE)
1169         v = (F_store.as_expr())/(j + 1)
1170         V.append(v)
1171         DE_D_list.append(Poly(Z[j + 1],Z[j]))
1172 
1173     DE_new = DifferentialExtension(extension = {'D': DE_D_list}) #a differential indeterminate
1174     for j in range(0, n):

1202 def recognize_derivative(a, d, DE, z=None):
1203     """ 
1204     Compute the squarefree factorization of the denominator of f 
1205     and for each Di the polynomial H in K[x] (see Theorem 2.7.1), using the 
1206     LaurentSeries algorithm. Write Di = GiEi where Gj = gcd(Hn, Di) and 
1207     gcd(Ei,Hn) = 1. Since the residues of f at the roots of Gj are all 0, and 
1208     the residue of f at a root alpha of Ei is Hi(a) != 0, f is the derivative of a 
1209     rational function if and only if Ei = 1 for each i, which is equivalent to 
1210     Di | H[-1] for each i. 
1211     """
1212     flag =True
1213     a, d = a.cancel(d, include=True)
1214     _, r = a.div(d)
1215     Np, Sp = splitfactor_sqf(d, DE, coefficientD=True, z=z)
1216 
1217     j = 1
1218     for s, _ in Sp:
1219         delta_a, delta_d, H = laurent_series(r, d, s, j, DE)
1220         g = gcd(d, H[-1]).as_poly()
1221         if g is not d:
1222             flag = False
1223             break
1224         j = j + 1
1225     return flag

1258 def residue_reduce(a, d, DE, z=None, invert=True):
1259     """ 
1260     Lazard-Rioboo-Rothstein-Trager resultant reduction. 
1261  
1262     Explanation 
1263     =========== 
1264  
1265     Given a derivation ``D`` on k(t) and f in k(t) simple, return g 
1266     elementary over k(t) and a Boolean b in {True, False} such that f - 
1267     Dg in k[t] if b == True or f + h and f + h - Dg do not have an 
1268     elementary integral over k(t) for any h in k<t> (reduced) if b == 
1269     False. 
1270  
1271     Returns (G, b), where G is a tuple of tuples of the form (s_i, S_i), 
1272     such that g = Add(*[RootSum(s_i, lambda z: z*log(S_i(z, t))) for 
1273     S_i, s_i in G]). f - Dg is the remaining integral, which is elementary 
1274     only if b == True, and hence the integral of f is elementary only if 
1275     b == True. 
1276  
1277     f - Dg is not calculated in this function because that would require 
1278     explicitly calculating the RootSum.  Use residue_reduce_derivation(). 
1279     """
1280     # TODO: Use log_to_atan() from rationaltools.py
1281     # If r = residue_reduce(...), then the logarithmic part is given by:
1282     # sum([RootSum(a[0].as_poly(z), lambda i: i*log(a[1].as_expr()).subs(z,
1283     # i)).subs(t, log(x)) for a in r[0]])
1284 
1285     z = z or Dummy('z')
1286     a, d = a.cancel(d, include=True)
1287     a, d = a.to_field().mul_ground(1/d.LC()), d.to_field().mul_ground(1/d.LC())
1288     kkinv = [1/x for x in DE.T[:DE.level]] + DE.T[:DE.level]
1289 
1290     if a.is_zero:
1291         return ([], True)
1292     _, a = a.div(d)
1293 
1294     pz = Poly(z, DE.t)
1295 
1296     Dd = derivation(d, DE)
1297     q = a - pz*Dd
1298 
1299     if Dd.degree(DE.t) <= d.degree(DE.t):
1300         r, R = d.resultant(q, includePRS=True)
1301     else:
1302         r, R = q.resultant(d, includePRS=True)
1303 
1304     R_map, H = {}, []
1305     for i in R:
1306         R_map[i.degree()] = i
1307 
1308     r = Poly(r, z)
1309     Np, Sp = splitfactor_sqf(r, DE, coefficientD=True, z=z)
1310 
1311     for s, i in Sp:
1312         if i == d.degree(DE.t):
1313             s = Poly(s, z).monic()
1314             H.append((s, d))
1315         else:
1316             h = R_map.get(i)
1317             if h is None:
1318                 continue
1319             h_lc = Poly(h.as_poly(DE.t).LC(), DE.t, field=True)
1320 
1321             h_lc_sqf = h_lc.sqf_list_include(all=True)
1322 
1323             for a, j in h_lc_sqf:
1324                 h = Poly(h, DE.t, field=True).exquo(Poly(gcd(a, s**j, *kkinv),
1325                     DE.t))
1326 
1327             s = Poly(s, z).monic()
1328 
1329             if invert:
1330                 h_lc = Poly(h.as_poly(DE.t).LC(), DE.t, field=True, expand=False)
1331                 inv, coeffs = h_lc.as_poly(z, field=True).invert(s), [S.One]
1332 
1333                 for coeff in h.coeffs()[1:]:
1334                     L = reduced(inv*coeff.as_poly(inv.gens), [s])[1]
1335                     coeffs.append(L.as_expr())
1336 
1337                 h = Poly(dict(list(zip(h.monoms(), coeffs))), DE.t)
1338 
1339             H.append((s, h))
1340 
1341     b = not any(cancel(i.as_expr()).has(DE.t, z) for i, _ in Np)
1342 
1343     return (H, b)

1649 class NonElementaryIntegral(Integral):
1650     """ 
1651     Represents a nonelementary Integral. 
1652  
1653     Explanation 
1654     =========== 
1655  
1656     If the result of integrate() is an instance of this class, it is 
1657     guaranteed to be nonelementary.  Note that integrate() by default will try 
1658     to find any closed-form solution, even in terms of special functions which 
1659     may themselves not be elementary.  To make integrate() only give 
1660     elementary solutions, or, in the cases where it can prove the integral to 
1661     be nonelementary, instances of this class, use integrate(risch=True). 
1662     In this case, integrate() may raise NotImplementedError if it cannot make 
1663     such a determination. 
1664  
1665     integrate() uses the deterministic Risch algorithm to integrate elementary 
1666     functions or prove that they have no elementary integral.  In some cases, 
1667     this algorithm can split an integral into an elementary and nonelementary 
1668     part, so that the result of integrate will be the sum of an elementary 
1669     expression and a NonElementaryIntegral. 
1670  
1671     Examples 
1672     ======== 
1673  
1674     >>> from sympy import integrate, exp, log, Integral 
1675     >>> from sympy.abc import x 
1676  
1677     >>> a = integrate(exp(-x**2), x, risch=True) 
1678     >>> print(a) 
1679     Integral(exp(-x**2), x) 
1680     >>> type(a) 
1681     <class 'sympy.integrals.risch.NonElementaryIntegral'> 
1682  
1683     >>> expr = (2*log(x)**2 - log(x) - x**2)/(log(x)**3 - x**2*log(x)) 
1684     >>> b = integrate(expr, x, risch=True)

1699 def risch_integrate(f, x, extension=None, handle_first='log',
1700                     separate_integral=False, rewrite_complex=None,
1701                     conds='piecewise'):
1702     r""" 
1703     The Risch Integration Algorithm. 
1704  
1705     Explanation 
1706     =========== 
1707  
1708     Only transcendental functions are supported.  Currently, only exponentials 
1709     and logarithms are supported, but support for trigonometric functions is 
1710     forthcoming. 
1711  
1712     If this function returns an unevaluated Integral in the result, it means 
1713     that it has proven that integral to be nonelementary.  Any errors will 
1714     result in raising NotImplementedError.  The unevaluated Integral will be 
1715     an instance of NonElementaryIntegral, a subclass of Integral. 
1716  
1717     handle_first may be either 'exp' or 'log'.  This changes the order in 
1718     which the extension is built, and may result in a different (but 
1719     equivalent) solution (for an example of this, see issue 5109).  It is also 
1720     possible that the integral may be computed with one but not the other, 
1721     because not all cases have been implemented yet.  It defaults to 'log' so 
1722     that the outer extension is exponential when possible, because more of the 
1723     exponential case has been implemented. 
1724  
1725     If ``separate_integral`` is ``True``, the result is returned as a tuple (ans, i), 
1726     where the integral is ans + i, ans is elementary, and i is either a 
1727     NonElementaryIntegral or 0.  This useful if you want to try further 
1728     integrating the NonElementaryIntegral part using other algorithms to 
1729     possibly get a solution in terms of special functions.  It is False by 
1730     default. 
1731  
1732     Examples 
1733     ======== 
1734  
1735     >>> from sympy.integrals.risch import risch_integrate 
1736     >>> from sympy import exp, log, pprint 
1737     >>> from sympy.abc import x 
1738  
1739     First, we try integrating exp(-x**2). Except for a constant factor of 
1740     2/sqrt(pi), this is the famous error function. 
1741  
1742     >>> pprint(risch_integrate(exp(-x**2), x)) 
1743       / 
1744      | 
1745      |    2 
1746      |  -x 
1747      | e    dx 
1748      | 
1749     / 
1750  
1751     The unevaluated Integral in the result means that risch_integrate() has 
1752     proven that exp(-x**2) does not have an elementary anti-derivative. 
1753  
1754     In many cases, risch_integrate() can split out the elementary 
1755     anti-derivative part from the nonelementary anti-derivative part. 
1756     For example, 
1757  
1758     >>> pprint(risch_integrate((2*log(x)**2 - log(x) - x**2)/(log(x)**3 - 
1759     ... x**2*log(x)), x)) 
1760                                              / 
1761                                             | 
1762       log(-x + log(x))   log(x + log(x))    |   1 
1763     - ---------------- + --------------- +  | ------ dx 
1764              2                  2           | log(x) 
1765                                             | 
1766                                            / 
1767  
1768     This means that it has proven that the integral of 1/log(x) is 
1769     nonelementary.  This function is also known as the logarithmic integral, 
1770     and is often denoted as Li(x). 
1771  
1772     risch_integrate() currently only accepts purely transcendental functions 
1773     with exponentials and logarithms, though note that this can include 
1774     nested exponentials and logarithms, as well as exponentials with bases 
1775     other than E. 
1776  
1777     >>> pprint(risch_integrate(exp(x)*exp(exp(x)), x)) 
1778      / x\ 
1779      \e / 
1780     e 
1781     >>> pprint(risch_integrate(exp(exp(x)), x)) 
1782       / 
1783      | 
1784      |  / x\ 
1785      |  \e / 
1786      | e     dx 
1787      | 
1788     / 
1789  
1790     >>> pprint(risch_integrate(x*x**x*log(x) + x**x + x*x**x, x)) 
1791        x 
1792     x*x 
1793     >>> pprint(risch_integrate(x**x, x)) 
1794       / 
1795      | 
1796      |  x 
1797      | x  dx 
1798      | 
1799     / 
1800  
1801     >>> pprint(risch_integrate(-1/(x*log(x)*log(log(x))**2), x)) 
1802          1 
1803     ----------- 
1804     log(log(x)) 
1805  
1806     """
1807     f = S(f)
1808 
1809     DE = extension or DifferentialExtension(f, x, handle_first=handle_first,
1810             dummy=True, rewrite_complex=rewrite_complex)
1811     fa, fd = DE.fa, DE.fd
1812 
1813     result = S.Zero
1814     for case in reversed(DE.cases):
1815         if not fa.expr.has(DE.t) and not fd.expr.has(DE.t) and not case == 'base':
1816             DE.decrement_level()
1817             fa, fd = frac_in((fa, fd), DE.t)
1818             continue
1819 
1820         fa, fd = fa.cancel(fd, include=True)
1821         if case == 'exp':
1822             ans, i, b = integrate_hyperexponential(fa, fd, DE, conds=conds)
1823         elif case == 'primitive':
1824             ans, i, b = integrate_primitive(fa, fd, DE)
1825         elif case == 'base':
1826             # XXX: We can't call ratint() directly here because it doesn't
1827             # handle polynomials correctly.
1828             ans = integrate(fa.as_expr()/fd.as_expr(), DE.x, risch=False)
1829             b = False
1830             i = S.Zero
1831         else:
1832             raise NotImplementedError("Only exponential and logarithmic "
1833             "extensions are currently supported.")
1834 
1835         result += ans
1836         if b:
1837             DE.decrement_level()
1838             fa, fd = frac_in(i, DE.t)
1839         else:
1840             result = result.subs(DE.backsubs)
1841             if not i.is_zero:
1842                 i = NonElementaryIntegral(i.function.subs(DE.backsubs),i.limits)
1843             if not separate_integral:
1844                 result += i
1845                 return result
1846             else:
1847 
1848                 if isinstance(i, NonElementaryIntegral):
1849                     return (result, i)
1850                 else:
1851                     return (result, 0)
