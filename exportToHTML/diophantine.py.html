40   class DiophantineSolutionSet(set):
41       """ 
42       Container for a set of solutions to a particular diophantine equation. 
43    
44       The base representation is a set of tuples representing each of the solutions. 
45    
46       Parameters 
47       ========== 
48    
49       symbols : list 
50           List of free symbols in the original equation. 
51       parameters: list 
52           List of parameters to be used in the solution. 
53    
54       Examples 
55       ======== 
56    
57       Adding solutions: 
58    
59           >>> from sympy.solvers.diophantine.diophantine import DiophantineSolutionSet 
60           >>> from sympy.abc import x, y, t, u 
61           >>> s1 = DiophantineSolutionSet([x, y], [t, u]) 
62           >>> s1 
63           set() 
64           >>> s1.add((2, 3)) 
65           >>> s1.add((-1, u)) 
66           >>> s1 
67           {(-1, u), (2, 3)} 
68           >>> s2 = DiophantineSolutionSet([x, y], [t, u]) 
69           >>> s2.add((3, 4)) 
70           >>> s1.update(*s2) 
71           >>> s1 
72           {(-1, u), (2, 3), (3, 4)} 
73    
74       Conversion of solutions into dicts: 
75    
76           >>> list(s1.dict_iterator()) 
77           [{x: -1, y: u}, {x: 2, y: 3}, {x: 3, y: 4}] 
78    
79       Substituting values: 
80    
81           >>> s3 = DiophantineSolutionSet([x, y], [t, u]) 
82           >>> s3.add((t**2, t + u)) 
83           >>> s3 
84           {(t**2, t + u)} 
85           >>> s3.subs({t: 2, u: 3}) 
86           {(4, 5)} 
87           >>> s3.subs(t, -1) 
88           {(1, u - 1)} 
89           >>> s3.subs(t, 3) 
90           {(9, u + 3)} 
91    
92       Evaluation at specific values. Positional arguments are given in the same order as the parameters: 
93    
94           >>> s3(-2, 3) 
95           {(4, 1)} 
96           >>> s3(5) 
97           {(25, u + 5)} 
98           >>> s3(None, 2) 
99           {(t**2, t + 2)} 
100      """
101  
102      def __init__(self, symbols_seq, parameters):
103          super().__init__()
104  
105          if not is_sequence(symbols_seq):
106              raise ValueError("Symbols must be given as a sequence.")
107  
108          if not is_sequence(parameters):
109              raise ValueError("Parameters must be given as a sequence.")
110  
111          self.symbols = tuple(symbols_seq)
112          self.parameters = tuple(parameters)
113  
114      def add(self, solution):
115          if len(solution) != len(self.symbols):
116              raise ValueError("Solution should have a length of %s, not %s" % (len(self.symbols), len(solution)))
117          # make solution canonical wrt sign (i.e. no -x unless x is also present as an arg)
118          args = set(solution)
119          for i in range(len(solution)):
120              x = solution[i]
121              if not type(x) is int and (-x).is_Symbol and -x not in args:
122                  solution = [_.subs(-x, x) for _ in solution]
123          super().add(Tuple(*solution))
124  
125      def update(self, *solutions):
126          for solution in solutions:
127              self.add(solution)
128  
129      def dict_iterator(self):
130          for solution in ordered(self):
131              yield dict(zip(self.symbols, solution))
132  
133      def subs(self, *args, **kwargs):
134          result = DiophantineSolutionSet(self.symbols, self.parameters)
135          for solution in self:
136              result.add(solution.subs(*args, **kwargs))
137          return result
138  
139      def __call__(self, *args):
140          if len(args) > len(self.parameters):
141              raise ValueError("Evaluation should have at most %s values, not %s" % (len(self.parameters), len(args)))
142          rep = {p: v for p, v in zip(self.parameters, args) if v is not None}
143          return self.subs(rep)

146  class DiophantineEquationType:
147      """ 
148      Internal representation of a particular diophantine equation type. 
149   
150      Parameters 
151      ========== 
152   
153      equation : 
154          The diophantine equation that is being solved. 
155      free_symbols : list (optional) 
156          The symbols being solved for. 
157   
158      Attributes 
159      ========== 
160   
161      total_degree : 
162          The maximum of the degrees of all terms in the equation 
163      homogeneous : 
164          Does the equation contain a term of degree 0 
165      homogeneous_order : 
166          Does the equation contain any coefficient that is in the symbols being solved for 
167      dimension : 
168          The number of symbols being solved for 
169      """
170      name: str
171  
172      def __init__(self, equation, free_symbols=None):
173          self.equation = _sympify(equation).expand(force=True)
174  
175          if free_symbols is not None:
176              self.free_symbols = free_symbols
177          else:
178              self.free_symbols = list(self.equation.free_symbols)
179              self.free_symbols.sort(key=default_sort_key)
180  
181          if not self.free_symbols:
182              raise ValueError('equation should have 1 or more free symbols')
183  
184          self.coeff = self.equation.as_coefficients_dict()
185          if not all(int_valued(c) for c in self.coeff.values()):
186              raise TypeError("Coefficients should be Integers")
187  
188          self.total_degree = Poly(self.equation).total_degree()
189          self.homogeneous = 1 not in self.coeff
190          self.homogeneous_order = not (set(self.coeff) & set(self.free_symbols))
191          self.dimension = len(self.free_symbols)
192          self._parameters = None
193  
194      def matches(self):
195          """ 
196          Determine whether the given equation can be matched to the particular equation type. 
197          """
198          return False
199  
200      @property
201      def n_parameters(self):
202          return self.dimension
203  
204      @property
205      def parameters(self):
206          if self._parameters is None:
207              self._parameters = symbols('t_:%i' % (self.n_parameters,), integer=True)
208          return self._parameters
209  
210      def solve(self, parameters=None, limit=None) -> DiophantineSolutionSet:
211          raise NotImplementedError('No solver has been written for %s.' % self.name)
212  
213      def pre_solve(self, parameters=None):
214          if not self.matches():
215              raise ValueError("This equation does not match the %s equation type." % self.name)
216  
217          if parameters is not None:
218              if len(parameters) != self.n_parameters:
219                  raise ValueError("Expected %s parameter(s) but got %s" % (self.n_parameters, len(parameters)))
220  
221          self._parameters = parameters

224  class Univariate(DiophantineEquationType):
225      """ 
226      Representation of a univariate diophantine equation. 
227   
228      A univariate diophantine equation is an equation of the form 
229      `a_{0} + a_{1}x + a_{2}x^2 + .. + a_{n}x^n = 0` where `a_{1}, a_{2}, ..a_{n}` are 
230      integer constants and `x` is an integer variable. 
231   
232      Examples 
233      ======== 
234   
235      >>> from sympy.solvers.diophantine.diophantine import Univariate 
236      >>> from sympy.abc import x 
237      >>> Univariate((x - 2)*(x - 3)**2).solve() # solves equation (x - 2)*(x - 3)**2 == 0 
238      {(2,), (3,)} 
239   
240      """
241  
242      name = 'univariate'
243  
244      def matches(self):
245          return self.dimension == 1
246  
247      def solve(self, parameters=None, limit=None):
248          self.pre_solve(parameters)
249  
250          result = DiophantineSolutionSet(self.free_symbols, parameters=self.parameters)
251          for i in solveset_real(self.equation, self.free_symbols[0]).intersect(S.Integers):
252              result.add((i,))
253          return result

256  class Linear(DiophantineEquationType):
257      """ 
258      Representation of a linear diophantine equation. 
259   
260      A linear diophantine equation is an equation of the form `a_{1}x_{1} + 
261      a_{2}x_{2} + .. + a_{n}x_{n} = 0` where `a_{1}, a_{2}, ..a_{n}` are 
262      integer constants and `x_{1}, x_{2}, ..x_{n}` are integer variables. 
263   
264      Examples 
265      ======== 
266   
267      >>> from sympy.solvers.diophantine.diophantine import Linear 
268      >>> from sympy.abc import x, y, z 
269      >>> l1 = Linear(2*x - 3*y - 5) 
270      >>> l1.matches() # is this equation linear 
271      True 
272      >>> l1.solve() # solves equation 2*x - 3*y - 5 == 0 
273      {(3*t_0 - 5, 2*t_0 - 5)} 
274   
275      Here x = -3*t_0 - 5 and y = -2*t_0 - 5 
276   
277      >>> Linear(2*x - 3*y - 4*z -3).solve() 
278      {(t_0, 2*t_0 + 4*t_1 + 3, -t_0 - 3*t_1 - 3)} 
279   
280      """
281  
282      name = 'linear'
283  
284      def matches(self):
285          return self.total_degree == 1
286  
287      def solve(self, parameters=None, limit=None):
288          self.pre_solve(parameters)
289  
290          coeff = self.coeff
291          var = self.free_symbols
292  
293          if 1 in coeff:
294              # negate coeff[] because input is of the form: ax + by + c ==  0
295              #                              but is used as: ax + by     == -c
296              c = -coeff[1]
297          else:
298              c = 0
299  
300          result = DiophantineSolutionSet(var, parameters=self.parameters)
301          params = result.parameters
302  
303          if len(var) == 1:
304              q, r = divmod(c, coeff[var[0]])
305              if not r:
306                  result.add((q,))
307              return result
308  
309          ''' 
310          base_solution_linear() can solve diophantine equations of the form: 
311   
312          a*x + b*y == c 
313   
314          We break down multivariate linear diophantine equations into a 
315          series of bivariate linear diophantine equations which can then 
316          be solved individually by base_solution_linear(). 
317   
318          Consider the following: 
319   
320          a_0*x_0 + a_1*x_1 + a_2*x_2 == c 
321   
322          which can be re-written as: 
323   
324          a_0*x_0 + g_0*y_0 == c 
325   
326          where 
327   
328          g_0 == gcd(a_1, a_2) 
329   
330          and 
331   
332          y == (a_1*x_1)/g_0 + (a_2*x_2)/g_0 
333   
334          This leaves us with two binary linear diophantine equations. 
335          For the first equation: 
336   
337          a == a_0 
338          b == g_0 
339          c == c 
340   
341          For the second: 
342   
343          a == a_1/g_0 
344          b == a_2/g_0 
345          c == the solution we find for y_0 in the first equation. 
346   
347          The arrays A and B are the arrays of integers used for 
348          'a' and 'b' in each of the n-1 bivariate equations we solve. 
349          '''
350  
351          A = [coeff[v] for v in var]
352          B = []
353          if len(var) > 2:
354              B.append(igcd(A[-2], A[-1]))
355              A[-2] = A[-2] // B[0]
356              A[-1] = A[-1] // B[0]
357              for i in range(len(A) - 3, 0, -1):
358                  gcd = igcd(B[0], A[i])
359                  B[0] = B[0] // gcd
360                  A[i] = A[i] // gcd
361                  B.insert(0, gcd)
362          B.append(A[-1])
363  
364          ''' 
365          Consider the trivariate linear equation: 
366   
367          4*x_0 + 6*x_1 + 3*x_2 == 2 
368   
369          This can be re-written as: 
370   
371          4*x_0 + 3*y_0 == 2 
372   
373          where 
374   
375          y_0 == 2*x_1 + x_2 
376          (Note that gcd(3, 6) == 3) 
377   
378          The complete integral solution to this equation is: 
379   
380          x_0 ==  2 + 3*t_0 
381          y_0 == -2 - 4*t_0 
382   
383          where 't_0' is any integer. 
384   
385          Now that we have a solution for 'x_0', find 'x_1' and 'x_2': 
386   
387          2*x_1 + x_2 == -2 - 4*t_0 
388   
389          We can then solve for '-2' and '-4' independently, 
390          and combine the results: 
391   
392          2*x_1a + x_2a == -2 
393          x_1a == 0 + t_0 
394          x_2a == -2 - 2*t_0 
395   
396          2*x_1b + x_2b == -4*t_0 
397          x_1b == 0*t_0 + t_1 
398          x_2b == -4*t_0 - 2*t_1 
399   
400          ==> 
401   
402          x_1 == t_0 + t_1 
403          x_2 == -2 - 6*t_0 - 2*t_1 
404   
405          where 't_0' and 't_1' are any integers. 
406   
407          Note that: 
408   
409          4*(2 + 3*t_0) + 6*(t_0 + t_1) + 3*(-2 - 6*t_0 - 2*t_1) == 2 
410   
411          for any integral values of 't_0', 't_1'; as required. 
412   
413          This method is generalised for many variables, below. 
414   
415          '''
416          solutions = []
417          for Ai, Bi in zip(A, B):
418              tot_x, tot_y = [], []
419  
420              for arg in Add.make_args(c):
421                  if arg.is_Integer:
422                      # example: 5 -> k = 5
423                      k, p = arg, S.One
424                      pnew = params[0]
425                  else:  # arg is a Mul or Symbol
426                      # example: 3*t_1 -> k = 3
427                      # example: t_0 -> k = 1
428                      k, p = arg.as_coeff_Mul()
429                      pnew = params[params.index(p) + 1]
430  
431                  sol = sol_x, sol_y = base_solution_linear(k, Ai, Bi, pnew)
432  
433                  if p is S.One:
434                      if None in sol:
435                          return result
436                  else:
437                      # convert a + b*pnew -> a*p + b*pnew
438                      if isinstance(sol_x, Add):
439                          sol_x = sol_x.args[0]*p + sol_x.args[1]
440                      if isinstance(sol_y, Add):
441                          sol_y = sol_y.args[0]*p + sol_y.args[1]
442  
443                  tot_x.append(sol_x)
444                  tot_y.append(sol_y)
445  
446              solutions.append(Add(*tot_x))
447              c = Add(*tot_y)
448  
449          solutions.append(c)
450          result.add(solutions)
451          return result

454  class BinaryQuadratic(DiophantineEquationType):
455      """ 
456      Representation of a binary quadratic diophantine equation. 
457   
458      A binary quadratic diophantine equation is an equation of the 
459      form `Ax^2 + Bxy + Cy^2 + Dx + Ey + F = 0`, where `A, B, C, D, E, 
460      F` are integer constants and `x` and `y` are integer variables. 
461   
462      Examples 
463      ======== 
464   
465      >>> from sympy.abc import x, y 
466      >>> from sympy.solvers.diophantine.diophantine import BinaryQuadratic 
467      >>> b1 = BinaryQuadratic(x**3 + y**2 + 1) 
468      >>> b1.matches() 
469      False 
470      >>> b2 = BinaryQuadratic(x**2 + y**2 + 2*x + 2*y + 2) 
471      >>> b2.matches() 
472      True 
473      >>> b2.solve() 
474      {(-1, -1)} 
475   
476      References 
477      ========== 
478   
479      .. [1] Methods to solve Ax^2 + Bxy + Cy^2 + Dx + Ey + F = 0, [online], 
480            Available: https://www.alpertron.com.ar/METHODS.HTM 
481      .. [2] Solving the equation ax^2+ bxy + cy^2 + dx + ey + f= 0, [online], 
482            Available: https://web.archive.org/web/20160323033111/http://www.jpr2718.org/ax2p.pdf 
483   
484      """
485  
486      name = 'binary_quadratic'
487  
488      def matches(self):
489          return self.total_degree == 2 and self.dimension == 2
490  
491      def solve(self, parameters=None, limit=None) -> DiophantineSolutionSet:
492          self.pre_solve(parameters)
493  
494          var = self.free_symbols
495          coeff = self.coeff
496  
497          x, y = var
498  
499          A = coeff[x**2]
500          B = coeff[x*y]
501          C = coeff[y**2]
502          D = coeff[x]
503          E = coeff[y]
504          F = coeff[S.One]
505  
506          A, B, C, D, E, F = [as_int(i) for i in _remove_gcd(A, B, C, D, E, F)]
507  
508          # (1) Simple-Hyperbolic case: A = C = 0, B != 0
509          # In this case equation can be converted to (Bx + E)(By + D) = DE - BF
510          # We consider two cases; DE - BF = 0 and DE - BF != 0
511          # More details, https://www.alpertron.com.ar/METHODS.HTM#SHyperb
512  
513          result = DiophantineSolutionSet(var, self.parameters)
514          t, u = result.parameters
515  
516          discr = B**2 - 4*A*C
517          if A == 0 and C == 0 and B != 0:
518  
519              if D*E - B*F == 0:
520                  q, r = divmod(E, B)
521                  if not r:
522                      result.add((-q, t))
523                  q, r = divmod(D, B)
524                  if not r:
525                      result.add((t, -q))
526              else:
527                  div = divisors(D*E - B*F)
528                  div = div + [-term for term in div]
529                  for d in div:
530                      x0, r = divmod(d - E, B)
531                      if not r:
532                          q, r = divmod(D*E - B*F, d)
533                          if not r:
534                              y0, r = divmod(q - D, B)
535                              if not r:
536                                  result.add((x0, y0))
537  
538          # (2) Parabolic case: B**2 - 4*A*C = 0
539          # There are two subcases to be considered in this case.
540          # sqrt(c)D - sqrt(a)E = 0 and sqrt(c)D - sqrt(a)E != 0
541          # More Details, https://www.alpertron.com.ar/METHODS.HTM#Parabol
542  
543          elif discr == 0:
544  
545              if A == 0:
546                  s = BinaryQuadratic(self.equation, free_symbols=[y, x]).solve(parameters=[t, u])
547                  for soln in s:
548                      result.add((soln[1], soln[0]))
549  
550              else:
551                  g = sign(A)*igcd(A, C)
552                  a = A // g
553                  c = C // g
554                  e = sign(B / A)
555  
556                  sqa = isqrt(a)
557                  sqc = isqrt(c)
558                  _c = e*sqc*D - sqa*E
559                  if not _c:
560                      z = Symbol("z", real=True)
561                      eq = sqa*g*z**2 + D*z + sqa*F
562                      roots = solveset_real(eq, z).intersect(S.Integers)
563                      for root in roots:
564                          ans = diop_solve(sqa*x + e*sqc*y - root)
565                          result.add((ans[0], ans[1]))
566  
567                  elif int_valued(c):
568                      solve_x = lambda u: -e*sqc*g*_c*t**2 - (E + 2*e*sqc*g*u)*t \
569                                          - (e*sqc*g*u**2 + E*u + e*sqc*F) // _c
570  
571                      solve_y = lambda u: sqa*g*_c*t**2 + (D + 2*sqa*g*u)*t \
572                                          + (sqa*g*u**2 + D*u + sqa*F) // _c
573  
574                      for z0 in range(0, abs(_c)):
575                          # Check if the coefficients of y and x obtained are integers or not
576                          if (divisible(sqa*g*z0**2 + D*z0 + sqa*F, _c) and
577                              divisible(e*sqc*g*z0**2 + E*z0 + e*sqc*F, _c)):
578                              result.add((solve_x(z0), solve_y(z0)))
579  
580          # (3) Method used when B**2 - 4*A*C is a square, is described in p. 6 of the below paper
581          # by John P. Robertson.
582          # https://web.archive.org/web/20160323033111/http://www.jpr2718.org/ax2p.pdf
583  
584          elif is_square(discr):
585              if A != 0:
586                  r = sqrt(discr)
587                  u, v = symbols("u, v", integer=True)
588                  eq = _mexpand(
589                      4*A*r*u*v + 4*A*D*(B*v + r*u + r*v - B*u) +
590                      2*A*4*A*E*(u - v) + 4*A*r*4*A*F)
591  
592                  solution = diop_solve(eq, t)
593  
594                  for s0, t0 in solution:
595  
596                      num = B*t0 + r*s0 + r*t0 - B*s0
597                      x_0 = S(num) / (4*A*r)
598                      y_0 = S(s0 - t0) / (2*r)
599                      if isinstance(s0, Symbol) or isinstance(t0, Symbol):
600                          if len(check_param(x_0, y_0, 4*A*r, parameters)) > 0:
601                              ans = check_param(x_0, y_0, 4*A*r, parameters)
602                              result.update(*ans)
603                      elif x_0.is_Integer and y_0.is_Integer:
604                          if is_solution_quad(var, coeff, x_0, y_0):
605                              result.add((x_0, y_0))
606  
607              else:
608                  s = BinaryQuadratic(self.equation, free_symbols=var[::-1]).solve(parameters=[t, u])  # Interchange x and y
609                  while s:
610                      result.add(s.pop()[::-1])  # and solution <--------+
611  
612          # (4) B**2 - 4*A*C > 0 and B**2 - 4*A*C not a square or B**2 - 4*A*C < 0
613  
614          else:
615  
616              P, Q = _transformation_to_DN(var, coeff)
617              D, N = _find_DN(var, coeff)
618              solns_pell = diop_DN(D, N)
619  
620              if D < 0:
621                  for x0, y0 in solns_pell:
622                      for x in [-x0, x0]:
623                          for y in [-y0, y0]:
624                              s = P*Matrix([x, y]) + Q
625                              try:
626                                  result.add([as_int(_) for _ in s])
627                              except ValueError:
628                                  pass
629              else:
630                  # In this case equation can be transformed into a Pell equation
631  
632                  solns_pell = set(solns_pell)
633                  solns_pell.update((-X, -Y) for X, Y in list(solns_pell))
634  
635                  a = diop_DN(D, 1)
636                  T = a[0][0]
637                  U = a[0][1]
638  
639                  if all(int_valued(_) for _ in P[:4] + Q[:2]):
640                      for r, s in solns_pell:
641                          _a = (r + s*sqrt(D))*(T + U*sqrt(D))**t
642                          _b = (r - s*sqrt(D))*(T - U*sqrt(D))**t
643                          x_n = _mexpand(S(_a + _b) / 2)
644                          y_n = _mexpand(S(_a - _b) / (2*sqrt(D)))
645                          s = P*Matrix([x_n, y_n]) + Q
646                          result.add(s)
647  
648                  else:
649                      L = ilcm(*[_.q for _ in P[:4] + Q[:2]])
650  
651                      k = 1
652  
653                      T_k = T
654                      U_k = U
655  
656                      while (T_k - 1) % L != 0 or U_k % L != 0:
657                          T_k, U_k = T_k*T + D*U_k*U, T_k*U + U_k*T
658                          k += 1
659  
660                      for X, Y in solns_pell:
661  
662                          for i in range(k):
663                              if all(int_valued(_) for _ in P*Matrix([X, Y]) + Q):
664                                  _a = (X + sqrt(D)*Y)*(T_k + sqrt(D)*U_k)**t
665                                  _b = (X - sqrt(D)*Y)*(T_k - sqrt(D)*U_k)**t
666                                  Xt = S(_a + _b) / 2
667                                  Yt = S(_a - _b) / (2*sqrt(D))
668                                  s = P*Matrix([Xt, Yt]) + Q
669                                  result.add(s)
670  
671                              X, Y = X*T + D*U*Y, X*U + Y*T
672  
673          return result

696      """ 
697      Representation of a homogeneous ternary quadratic normal diophantine equation. 
698   
699      Examples 
700      ======== 
701   
702      >>> from sympy.abc import x, y, z 
703      >>> from sympy.solvers.diophantine.diophantine import HomogeneousTernaryQuadraticNormal 
704      >>> HomogeneousTernaryQuadraticNormal(4*x**2 - 5*y**2 + z**2).solve() 
705      {(1, 2, 4)} 
706   
707      """
708  
709      name = 'homogeneous_ternary_quadratic_normal'
710  
711      def matches(self):
712          if not (self.total_degree == 2 and self.dimension == 3):
713              return False
714          if not self.homogeneous:
715              return False
716          if not self.homogeneous_order:
717              return False
718  
719          nonzero = [k for k in self.coeff if self.coeff[k]]
720          return len(nonzero) == 3 and all(i**2 in nonzero for i in self.free_symbols)
721  
722      def solve(self, parameters=None, limit=None) -> DiophantineSolutionSet:
723          self.pre_solve(parameters)
724  
725          var = self.free_symbols
726          coeff = self.coeff
727  
728          x, y, z = var
729  
730          a = coeff[x**2]
731          b = coeff[y**2]
732          c = coeff[z**2]
733  
734          (sqf_of_a, sqf_of_b, sqf_of_c), (a_1, b_1, c_1), (a_2, b_2, c_2) = \
735              sqf_normal(a, b, c, steps=True)
736  
737          A = -a_2*c_2
738          B = -b_2*c_2
739  
740          result = DiophantineSolutionSet(var, parameters=self.parameters)
741  
742          # If following two conditions are satisfied then there are no solutions
743          if A < 0 and B < 0:
744              return result
745  
746          if (
747              sqrt_mod(-b_2*c_2, a_2) is None or
748              sqrt_mod(-c_2*a_2, b_2) is None or
749              sqrt_mod(-a_2*b_2, c_2) is None):
750              return result
751  
752          z_0, x_0, y_0 = descent(A, B)
753  
754          z_0, q = _rational_pq(z_0, abs(c_2))
755          x_0 *= q
756          y_0 *= q
757  
758          x_0, y_0, z_0 = _remove_gcd(x_0, y_0, z_0)
759  
760          # Holzer reduction
761          if sign(a) == sign(b):
762              x_0, y_0, z_0 = holzer(x_0, y_0, z_0, abs(a_2), abs(b_2), abs(c_2))
763          elif sign(a) == sign(c):
764              x_0, z_0, y_0 = holzer(x_0, z_0, y_0, abs(a_2), abs(c_2), abs(b_2))
765          else:
766              y_0, z_0, x_0 = holzer(y_0, z_0, x_0, abs(b_2), abs(c_2), abs(a_2))
767  
768          x_0 = reconstruct(b_1, c_1, x_0)
769          y_0 = reconstruct(a_1, c_1, y_0)
770          z_0 = reconstruct(a_1, b_1, z_0)
771  
772          sq_lcm = ilcm(sqf_of_a, sqf_of_b, sqf_of_c)
773  
774          x_0 = abs(x_0*sq_lcm // sqf_of_a)
775          y_0 = abs(y_0*sq_lcm // sqf_of_b)
776          z_0 = abs(z_0*sq_lcm // sqf_of_c)
777  
778          result.add(_remove_gcd(x_0, y_0, z_0))
779          return result

782  class HomogeneousTernaryQuadratic(DiophantineEquationType):
783      """ 
784      Representation of a homogeneous ternary quadratic diophantine equation. 
785   
786      Examples 
787      ======== 
788   
789      >>> from sympy.abc import x, y, z 
790      >>> from sympy.solvers.diophantine.diophantine import HomogeneousTernaryQuadratic 
791      >>> HomogeneousTernaryQuadratic(x**2 + y**2 - 3*z**2 + x*y).solve() 
792      {(-1, 2, 1)} 
793      >>> HomogeneousTernaryQuadratic(3*x**2 + y**2 - 3*z**2 + 5*x*y + y*z).solve() 
794      {(3, 12, 13)} 
795   
796      """
797  
798      name = 'homogeneous_ternary_quadratic'
799  
800      def matches(self):
801          if not (self.total_degree == 2 and self.dimension == 3):
802              return False
803          if not self.homogeneous:
804              return False
805          if not self.homogeneous_order:
806              return False
807  
808          nonzero = [k for k in self.coeff if self.coeff[k]]
809          return not (len(nonzero) == 3 and all(i**2 in nonzero for i in self.free_symbols))
810  
811      def solve(self, parameters=None, limit=None):
812          self.pre_solve(parameters)
813  
814          _var = self.free_symbols
815          coeff = self.coeff
816  
817          x, y, z = _var
818          var = [x, y, z]
819  
820          # Equations of the form B*x*y + C*z*x + E*y*z = 0 and At least two of the
821          # coefficients A, B, C are non-zero.
822          # There are infinitely many solutions for the equation.
823          # Ex: (0, 0, t), (0, t, 0), (t, 0, 0)
824          # Equation can be re-written as y*(B*x + E*z) = -C*x*z and we can find rather
825          # unobvious solutions. Set y = -C and B*x + E*z = x*z. The latter can be solved by
826          # using methods for binary quadratic diophantine equations. Let's select the
827          # solution which minimizes |x| + |z|
828  
829          result = DiophantineSolutionSet(var, parameters=self.parameters)
830  
831          def unpack_sol(sol):
832              if len(sol) > 0:
833                  return list(sol)[0]
834              return None, None, None
835  
836          if not any(coeff[i**2] for i in var):
837              if coeff[x*z]:
838                  sols = diophantine(coeff[x*y]*x + coeff[y*z]*z - x*z)
839                  s = min(sols, key=lambda r: abs(r[0]) + abs(r[1]))
840                  result.add(_remove_gcd(s[0], -coeff[x*z], s[1]))
841                  return result
842  
843              var[0], var[1] = _var[1], _var[0]
844              y_0, x_0, z_0 = unpack_sol(_diop_ternary_quadratic(var, coeff))
845              if x_0 is not None:
846                  result.add((x_0, y_0, z_0))
847              return result
848  
849          if coeff[x**2] == 0:
850              # If the coefficient of x is zero change the variables
851              if coeff[y**2] == 0:
852                  var[0], var[2] = _var[2], _var[0]
853                  z_0, y_0, x_0 = unpack_sol(_diop_ternary_quadratic(var, coeff))
854  
855              else:
856                  var[0], var[1] = _var[1], _var[0]
857                  y_0, x_0, z_0 = unpack_sol(_diop_ternary_quadratic(var, coeff))
858  
859          else:
860              if coeff[x*y] or coeff[x*z]:
861                  # Apply the transformation x --> X - (B*y + C*z)/(2*A)
862                  A = coeff[x**2]
863                  B = coeff[x*y]
864                  C = coeff[x*z]
865                  D = coeff[y**2]
866                  E = coeff[y*z]
867                  F = coeff[z**2]
868  
869                  _coeff = {}
870  
871                  _coeff[x**2] = 4*A**2
872                  _coeff[y**2] = 4*A*D - B**2
873                  _coeff[z**2] = 4*A*F - C**2
874                  _coeff[y*z] = 4*A*E - 2*B*C
875                  _coeff[x*y] = 0
876                  _coeff[x*z] = 0
877  
878                  x_0, y_0, z_0 = unpack_sol(_diop_ternary_quadratic(var, _coeff))
879  
880                  if x_0 is None:
881                      return result
882  
883                  p, q = _rational_pq(B*y_0 + C*z_0, 2*A)
884                  x_0, y_0, z_0 = x_0*q - p, y_0*q, z_0*q
885  
886              elif coeff[z*y] != 0:
887                  if coeff[y**2] == 0:
888                      if coeff[z**2] == 0:
889                          # Equations of the form A*x**2 + E*yz = 0.
890                          A = coeff[x**2]
891                          E = coeff[y*z]
892  
893                          b, a = _rational_pq(-E, A)
894  
895                          x_0, y_0, z_0 = b, a, b
896  
897                      else:
898                          # Ax**2 + E*y*z + F*z**2  = 0
899                          var[0], var[2] = _var[2], _var[0]
900                          z_0, y_0, x_0 = unpack_sol(_diop_ternary_quadratic(var, coeff))
901  
902                  else:
903                      # A*x**2 + D*y**2 + E*y*z + F*z**2 = 0, C may be zero
904                      var[0], var[1] = _var[1], _var[0]
905                      y_0, x_0, z_0 = unpack_sol(_diop_ternary_quadratic(var, coeff))
906  
907              else:
908                  # Ax**2 + D*y**2 + F*z**2 = 0, C may be zero
909                  x_0, y_0, z_0 = unpack_sol(_diop_ternary_quadratic_normal(var, coeff))
910  
911          if x_0 is None:
912              return result
913  
914          result.add(_remove_gcd(x_0, y_0, z_0))
915          return result

1086

1216 def _remove_gcd(*x):
1217     try:
1218         g = igcd(*x)
1219     except ValueError:
1220         fx = list(filter(None, x))
1221         if len(fx) < 2:
1222             return x
1223         g = igcd(*[i.as_content_primitive()[0] for i in fx])
1224     except TypeError:
1225         raise TypeError('_remove_gcd(a,b,c) or _remove_gcd(*container)')
1226     if g == 1:
1227         return x
1228     return tuple([i//g for i in x])

1252 def diophantine(eq, param=symbols("t", integer=True), syms=None,
1253                 permute=False):
1254     """ 
1255     Simplify the solution procedure of diophantine equation ``eq`` by 
1256     converting it into a product of terms which should equal zero. 
1257  
1258     Explanation 
1259     =========== 
1260  
1261     For example, when solving, `x^2 - y^2 = 0` this is treated as 
1262     `(x + y)(x - y) = 0` and `x + y = 0` and `x - y = 0` are solved 
1263     independently and combined. Each term is solved by calling 
1264     ``diop_solve()``. (Although it is possible to call ``diop_solve()`` 
1265     directly, one must be careful to pass an equation in the correct 
1266     form and to interpret the output correctly; ``diophantine()`` is 
1267     the public-facing function to use in general.) 
1268  
1269     Output of ``diophantine()`` is a set of tuples. The elements of the 
1270     tuple are the solutions for each variable in the equation and 
1271     are arranged according to the alphabetic ordering of the variables. 
1272     e.g. For an equation with two variables, `a` and `b`, the first 
1273     element of the tuple is the solution for `a` and the second for `b`. 
1274  
1275     Usage 
1276     ===== 
1277  
1278     ``diophantine(eq, t, syms)``: Solve the diophantine 
1279     equation ``eq``. 
1280     ``t`` is the optional parameter to be used by ``diop_solve()``. 
1281     ``syms`` is an optional list of symbols which determines the 
1282     order of the elements in the returned tuple. 
1283  
1284     By default, only the base solution is returned. If ``permute`` is set to 
1285     True then permutations of the base solution and/or permutations of the 
1286     signs of the values will be returned when applicable. 
1287  
1288     Details 
1289     ======= 
1290  
1291     ``eq`` should be an expression which is assumed to be zero. 
1292     ``t`` is the parameter to be used in the solution. 
1293  
1294     Examples 
1295     ======== 
1296  
1297     >>> from sympy import diophantine 
1298     >>> from sympy.abc import a, b 
1299     >>> eq = a**4 + b**4 - (2**4 + 3**4) 
1300     >>> diophantine(eq) 
1301     {(2, 3)} 
1302     >>> diophantine(eq, permute=True) 
1303     {(-3, -2), (-3, 2), (-2, -3), (-2, 3), (2, -3), (2, 3), (3, -2), (3, 2)}

1307     {(t_0, -t_0), (t_0, t_0)}

1507 def merge_solution(var, var_t, solution):
1508     """ 
1509     This is used to construct the full solution from the solutions of sub 
1510     equations. 
1511  
1512     Explanation 
1513     =========== 
1514  
1515     For example when solving the equation `(x - y)(x^2 + y^2 - z^2) = 0`, 
1516     solutions for each of the equations `x - y = 0` and `x^2 + y^2 - z^2` are 
1517     found independently. Solutions for `x - y = 0` are `(x, y) = (t, t)`. But 
1518     we should introduce a value for z when we output the solution for the 
1519     original equation. This function converts `(t, t)` into `(t, t, n_{1})` 
1520     where `n_{1}` is an integer parameter. 
1521     """
1522     sol = []
1523 
1524     if None in solution:
1525         return ()
1526 
1527     solution = iter(solution)
1528     params = numbered_symbols("n", integer=True, start=1)
1529     for v in var:
1530         if v in var_t:
1531             sol.append(next(solution))
1532         else:
1533             sol.append(next(params))
1534 
1535     for val, symb in zip(sol, var):
1536         if check_assumptions(val, **symb.assumptions0) is False:
1537             return ()
1538 
1539     return tuple(sol)

1717     order as the sorted variables. 
1718  
1719     Details 
1720     ======= 
1721  
1722     ``eq`` is a linear diophantine equation which is assumed to be zero. 
1723     ``param`` is the parameter to be used in the solution.

1927     Solves the equation `x^2 - Dy^2 = N`. 
1928  
1929     Explanation 
1930     =========== 
1931  
1932     Mainly concerned with the case `D > 0, D` is not a perfect square, 
1933     which is the same as the generalized Pell equation. The LMM 
1934     algorithm [1]_ is used to solve this equation. 
1935  
1936     Returns one solution tuple, (`x, y)` for each class of the solutions. 
1937     Other solutions of the class can be constructed according to the 
1938     values of ``D`` and ``N``.

1941     =====

1944     ``t`` is the parameter to be used in the solutions.

1951

1953     ========

1955     >>> from sympy.solvers.diophantine.diophantine import diop_DN

1957     [(3, 1), (393, 109), (36, 10)] 
1958

1960     solutions to the equation `x^2 - 13y^2 = -4` given by (3, 1), (393, 109)

1962     that `x = 3` and `y = 1`.

1964     >>> diop_DN(986, 1) # Solves equation x**2 - 986*y**2 = 1 
1965     [(49299, 1570)]

1968     ======== 
1969  
1970     find_DN(), diop_bf_DN() 
1971  
1972     References 
1973     ========== 
1974  
1975     .. [1] Solving the generalized Pell equation x**2 - D*y**2 = N, John P.

1977         https://web.archive.org/web/20160323033128/http://www.jpr2718.org/pell.pdf

1992 
1993     if D == 0:

1995             return []

1998         sN, _exact = integer_nthroot(N, 2)

2000             return [(sN, t)]

2002 
2003     # D > 0

2007             return [(sD*t, t)]
2008

2013             except ValueError:
2014                 _exact = False
2015             if _exact:
2016                 sol.append((sq, y))
2017         return sol
2018 
2019     if 1 < N**2 < D:

2021         return _special_diop_DN(D, N)

2023     if N == 0:
2024         return [(0, 0)]
2025

2054                     if prev_G**2 - D*prev_B**2 == m:
2055                         sol.append((f*prev_G, f*prev_B))
2056                     elif a := diop_DN(D, -1):
2057                         sol.append((f*(prev_G*a[0][0] + prev_B*D*a[0][1]),

2141 def cornacchia(a:int, b:int, m:int) -> set[tuple[int, int]]:
2142     r""" 
2143     Solves `ax^2 + by^2 = m` where `\gcd(a, b) = 1 = gcd(a, m)` and `a, b > 0`. 
2144  
2145     Explanation 
2146     =========== 
2147  
2148     Uses the algorithm due to Cornacchia. The method only finds primitive 
2149     solutions, i.e. ones with `\gcd(x, y) = 1`. So this method cannot be used to 
2150     find the solutions of `x^2 + y^2 = 20` since the only solution to former is 
2151     `(x, y) = (4, 2)` and it is not primitive. When `a = b`, only the 
2152     solutions with `x \leq y` are found. For more details, see the References. 
2153  
2154     Examples 
2155     ======== 
2156  
2157     >>> from sympy.solvers.diophantine.diophantine import cornacchia 
2158     >>> cornacchia(2, 3, 35) # equation 2x**2 + 3y**2 = 35 
2159     {(2, 3), (4, 1)} 
2160     >>> cornacchia(1, 1, 25) # equation x**2 + y**2 = 25 
2161     {(4, 3)} 
2162  
2163     References 
2164     =========== 
2165  
2166     .. [1] A. Nitaj, "L'algorithme de Cornacchia" 
2167     .. [2] Solving the diophantine equation ax**2 + by**2 = m by Cornacchia's 
2168         method, [online], Available: 
2169         http://www.numbertheory.org/php/cornacchia.html 
2170  
2171     See Also 
2172     ======== 
2173  
2174     sympy.utilities.iterables.signed_permutations 
2175     """
2176     # Assume gcd(a, b) = gcd(a, m) = 1 and a, b > 0 but no error checking
2177     sols = set()
2178 
2179     if a + b > m:
2180         # xy = 0 must hold if there exists a solution
2181         if a == 1:
2182             # y = 0
2183             s, _exact = iroot(m // a, 2)
2184             if _exact:
2185                 sols.add((int(s), 0))
2186             if a == b:
2187                 # only keep one solution
2188                 return sols
2189         if m % b == 0:
2190             # x = 0
2191             s, _exact = iroot(m // b, 2)
2192             if _exact:
2193                 sols.add((0, int(s)))
2194         return sols
2195 
2196     # the original cornacchia
2197     for t in sqrt_mod_iter(-b*invert(a, m), m):
2198         if t < m // 2:
2199             continue
2200         u, r = m, t
2201         while (m1 := m - a*r**2) <= 0:
2202             u, r = r, u % r
2203         m1, _r = divmod(m1, b)
2204         if _r:
2205             continue
2206         s, _exact = iroot(m1, 2)
2207         if _exact:
2208             if a == b and r < s:
2209                 r, s = s, r
2210             sols.add((int(r), int(s)))
2211     return sols

2234

2441 def transformation_to_DN(eq):
2442     """ 
2443     This function transforms general quadratic, 
2444     `ax^2 + bxy + cy^2 + dx + ey + f = 0` 
2445     to more easy to deal with `X^2 - DY^2 = N` form. 
2446  
2447     Explanation 
2448     =========== 
2449  
2450     This is used to solve the general quadratic equation by transforming it to 
2451     the latter form. Refer to [1]_ for more detailed information on the 
2452     transformation. This function returns a tuple (A, B) where A is a 2 X 2 
2453     matrix and B is a 2 X 1 matrix such that, 
2454  
2455     Transpose([x y]) =  A * Transpose([X Y]) + B 
2456  
2457     Usage 
2458     ===== 
2459  
2460     ``transformation_to_DN(eq)``: where ``eq`` is the quadratic to be 
2461     transformed. 
2462  
2463     Examples 
2464     ======== 
2465  
2466     >>> from sympy.abc import x, y 
2467     >>> from sympy.solvers.diophantine.diophantine import transformation_to_DN 
2468     >>> A, B = transformation_to_DN(x**2 - 3*x*y - y**2 - 2*y + 1) 
2469     >>> A 
2470     Matrix([ 
2471     [1/26, 3/26], 
2472     [   0, 1/13]]) 
2473     >>> B 
2474     Matrix([ 
2475     [-6/13], 
2476     [-4/13]]) 
2477  
2478     A, B  returned are such that Transpose((x y)) =  A * Transpose((X Y)) + B. 
2479     Substituting these values for `x` and `y` and a bit of simplifying work 
2480     will give an equation of the form `x^2 - Dy^2 = N`. 
2481  
2482     >>> from sympy.abc import X, Y 
2483     >>> from sympy import Matrix, simplify 
2484     >>> u = (A*Matrix([X, Y]) + B)[0] # Transformation for x 
2485     >>> u 
2486     X/26 + 3*Y/26 - 6/13 
2487     >>> v = (A*Matrix([X, Y]) + B)[1] # Transformation for y 
2488     >>> v 
2489     Y/13 - 4/13 
2490  
2491     Next we will substitute these formulas for `x` and `y` and do 
2492     ``simplify()``. 
2493  
2494     >>> eq = simplify((x**2 - 3*x*y - y**2 - 2*y + 1).subs(zip((x, y), (u, v)))) 
2495     >>> eq 
2496     X**2/676 - Y**2/52 + 17/13 
2497  
2498     By multiplying the denominator appropriately, we can get a Pell equation 
2499     in the standard form. 
2500  
2501     >>> eq * 676 
2502     X**2 - 13*Y**2 + 884 
2503  
2504     If only the final equation is needed, ``find_DN()`` can be used. 
2505  
2506     See Also 
2507     ======== 
2508  
2509     find_DN() 
2510  
2511     References 
2512     ========== 
2513  
2514     .. [1] Solving the equation ax^2 + bxy + cy^2 + dx + ey + f = 0, 
2515            John P.Robertson, May 8, 2003, Page 7 - 11. 
2516            https://web.archive.org/web/20160323033111/http://www.jpr2718.org/ax2p.pdf 
2517     """
2518 
2519     var, coeff, diop_type = classify_diop(eq, _dict=False)
2520     if diop_type == BinaryQuadratic.name:
2521         return _transformation_to_DN(var, coeff)

2524 def _transformation_to_DN(var, coeff):
2525 
2526     x, y = var
2527 
2528     a = coeff[x**2]
2529     b = coeff[x*y]
2530     c = coeff[y**2]
2531     d = coeff[x]
2532     e = coeff[y]
2533     f = coeff[1]
2534 
2535     a, b, c, d, e, f = [as_int(i) for i in _remove_gcd(a, b, c, d, e, f)]
2536 
2537     X, Y = symbols("X, Y", integer=True)
2538 
2539     if b:
2540         B, C = _rational_pq(2*a, b)
2541         A, T = _rational_pq(a, B**2)
2542 
2543         # eq_1 = A*B*X**2 + B*(c*T - A*C**2)*Y**2 + d*T*X + (B*e*T - d*T*C)*Y + f*T*B
2544         coeff = {X**2: A*B, X*Y: 0, Y**2: B*(c*T - A*C**2), X: d*T, Y: B*e*T - d*T*C, 1: f*T*B}
2545         A_0, B_0 = _transformation_to_DN([X, Y], coeff)
2546         return Matrix(2, 2, [S.One/B, -S(C)/B, 0, 1])*A_0, Matrix(2, 2, [S.One/B, -S(C)/B, 0, 1])*B_0
2547 
2548     if d:
2549         B, C = _rational_pq(2*a, d)
2550         A, T = _rational_pq(a, B**2)
2551 
2552         # eq_2 = A*X**2 + c*T*Y**2 + e*T*Y + f*T - A*C**2
2553         coeff = {X**2: A, X*Y: 0, Y**2: c*T, X: 0, Y: e*T, 1: f*T - A*C**2}
2554         A_0, B_0 = _transformation_to_DN([X, Y], coeff)
2555         return Matrix(2, 2, [S.One/B, 0, 0, 1])*A_0, Matrix(2, 2, [S.One/B, 0, 0, 1])*B_0 + Matrix([-S(C)/B, 0])
2556 
2557     if e:
2558         B, C = _rational_pq(2*c, e)
2559         A, T = _rational_pq(c, B**2)
2560 
2561         # eq_3 = a*T*X**2 + A*Y**2 + f*T - A*C**2
2562         coeff = {X**2: a*T, X*Y: 0, Y**2: A, X: 0, Y: 0, 1: f*T - A*C**2}
2563         A_0, B_0 = _transformation_to_DN([X, Y], coeff)
2564         return Matrix(2, 2, [1, 0, 0, S.One/B])*A_0, Matrix(2, 2, [1, 0, 0, S.One/B])*B_0 + Matrix([0, -S(C)/B])
2565 
2566     # TODO: pre-simplification: Not necessary but may simplify
2567     # the equation.
2568     return Matrix(2, 2, [S.One/a, 0, 0, 1]), Matrix([0, 0])

2719 def transformation_to_normal(eq):
2720     """ 
2721     Returns the transformation Matrix that converts a general ternary 
2722     quadratic equation ``eq`` (`ax^2 + by^2 + cz^2 + dxy + eyz + fxz`) 
2723     to a form without cross terms: `ax^2 + by^2 + cz^2 = 0`. This is 
2724     not used in solving ternary quadratics; it is only implemented for 
2725     the sake of completeness. 
2726     """
2727     var, coeff, diop_type = classify_diop(eq, _dict=False)
2728 
2729     if diop_type in (
2730             "homogeneous_ternary_quadratic",
2731             "homogeneous_ternary_quadratic_normal"):
2732         return _transformation_to_normal(var, coeff)
2733 
2734 
2735 def _transformation_to_normal(var, coeff):
2736 
2737     _var = list(var)  # copy
2738     x, y, z = var
2739 
2740     if not any(coeff[i**2] for i in var):
2741         # https://math.stackexchange.com/questions/448051/transform-quadratic-ternary-form-to-normal-form/448065#448065
2742         a = coeff[x*y]
2743         b = coeff[y*z]
2744         c = coeff[x*z]
2745         swap = False
2746         if not a:  # b can't be 0 or else there aren't 3 vars
2747             swap = True
2748             a, b = b, a
2749         T = Matrix(((1, 1, -b/a), (1, -1, -c/a), (0, 0, 1)))
2750         if swap:
2751             T.row_swap(0, 1)
2752             T.col_swap(0, 1)
2753         return T
2754 
2755     if coeff[x**2] == 0:
2756         # If the coefficient of x is zero change the variables
2757         if coeff[y**2] == 0:
2758             _var[0], _var[2] = var[2], var[0]
2759             T = _transformation_to_normal(_var, coeff)
2760             T.row_swap(0, 2)
2761             T.col_swap(0, 2)
2762             return T
2763 
2764         _var[0], _var[1] = var[1], var[0]
2765         T = _transformation_to_normal(_var, coeff)
2766         T.row_swap(0, 1)
2767         T.col_swap(0, 1)
2768         return T
2769 
2770     # Apply the transformation x --> X - (B*Y + C*Z)/(2*A)
2771     if coeff[x*y] != 0 or coeff[x*z] != 0:
2772         A = coeff[x**2]
2773         B = coeff[x*y]
2774         C = coeff[x*z]
2775         D = coeff[y**2]
2776         E = coeff[y*z]
2777         F = coeff[z**2]
2778 
2779         _coeff = {}
2780 
2781         _coeff[x**2] = 4*A**2
2782         _coeff[y**2] = 4*A*D - B**2
2783         _coeff[z**2] = 4*A*F - C**2
2784         _coeff[y*z] = 4*A*E - 2*B*C
2785         _coeff[x*y] = 0
2786         _coeff[x*z] = 0
2787 
2788         T_0 = _transformation_to_normal(_var, _coeff)
2789         return Matrix(3, 3, [1, S(-B)/(2*A), S(-C)/(2*A), 0, 1, 0, 0, 0, 1])*T_0
2790 
2791     elif coeff[y*z] != 0:
2792         if coeff[y**2] == 0:
2793             if coeff[z**2] == 0:
2794                 # Equations of the form A*x**2 + E*yz = 0.
2795                 # Apply transformation y -> Y + Z ans z -> Y - Z
2796                 return Matrix(3, 3, [1, 0, 0, 0, 1, 1, 0, 1, -1])
2797 
2798             # Ax**2 + E*y*z + F*z**2  = 0
2799             _var[0], _var[2] = var[2], var[0]
2800             T = _transformation_to_normal(_var, coeff)
2801             T.row_swap(0, 2)
2802             T.col_swap(0, 2)
2803             return T
2804 
2805         # A*x**2 + D*y**2 + E*y*z + F*z**2 = 0, F may be zero
2806         _var[0], _var[1] = var[1], var[0]
2807         T = _transformation_to_normal(_var, coeff)
2808         T.row_swap(0, 1)
2809         T.col_swap(0, 1)
2810         return T
2811 
2812     return Matrix.eye(3)

2815 def parametrize_ternary_quadratic(eq):
2816     """ 
2817     Returns the parametrized general solution for the ternary quadratic 
2818     equation ``eq`` which has the form 
2819     `ax^2 + by^2 + cz^2 + fxy + gyz + hxz = 0`. 
2820  
2821     Examples 
2822     ======== 
2823  
2824     >>> from sympy import Tuple, ordered 
2825     >>> from sympy.abc import x, y, z 
2826     >>> from sympy.solvers.diophantine.diophantine import parametrize_ternary_quadratic 
2827  
2828     The parametrized solution may be returned with three parameters: 
2829  
2830     >>> parametrize_ternary_quadratic(2*x**2 + y**2 - 2*z**2) 
2831     (p**2 - 2*q**2, -2*p**2 + 4*p*q - 4*p*r - 4*q**2, p**2 - 4*p*q + 2*q**2 - 4*q*r) 
2832  
2833     There might also be only two parameters: 
2834  
2835     >>> parametrize_ternary_quadratic(4*x**2 + 2*y**2 - 3*z**2) 
2836     (2*p**2 - 3*q**2, -4*p**2 + 12*p*q - 6*q**2, 4*p**2 - 8*p*q + 6*q**2) 
2837  
2838     Notes 
2839     ===== 
2840  
2841     Consider ``p`` and ``q`` in the previous 2-parameter 
2842     solution and observe that more than one solution can be represented 
2843     by a given pair of parameters. If `p` and ``q`` are not coprime, this is 
2844     trivially true since the common factor will also be a common factor of the 
2845     solution values. But it may also be true even when ``p`` and 
2846     ``q`` are coprime: 
2847  
2848     >>> sol = Tuple(*_) 
2849     >>> p, q = ordered(sol.free_symbols) 
2850     >>> sol.subs([(p, 3), (q, 2)]) 
2851     (6, 12, 12) 
2852     >>> sol.subs([(q, 1), (p, 1)]) 
2853     (-1, 2, 2) 
2854     >>> sol.subs([(q, 0), (p, 1)]) 
2855     (2, -4, 4) 
2856     >>> sol.subs([(q, 1), (p, 0)]) 
2857     (-3, -6, 6) 
2858  
2859     Except for sign and a common factor, these are equivalent to 
2860     the solution of (1, 2, 2). 
2861  
2862     References 
2863     ========== 
2864  
2865     .. [1] The algorithmic resolution of Diophantine equations, Nigel P. Smart, 
2866            London Mathematical Society Student Texts 41, Cambridge University 
2867            Press, Cambridge, 1998. 
2868  
2869     """
2870     var, coeff, diop_type = classify_diop(eq, _dict=False)
2871 
2872     if diop_type in (
2873             "homogeneous_ternary_quadratic",
2874             "homogeneous_ternary_quadratic_normal"):
2875         x_0, y_0, z_0 = list(_diop_ternary_quadratic(var, coeff))[0]
2876         return _parametrize_ternary_quadratic(
2877             (x_0, y_0, z_0), var, coeff)
2878

2880 def _parametrize_ternary_quadratic(solution, _var, coeff):
2881     # called for a*x**2 + b*y**2 + c*z**2 + d*x*y + e*y*z + f*x*z = 0
2882     assert 1 not in coeff
2883 
2884     x_0, y_0, z_0 = solution
2885 
2886     v = list(_var)  # copy
2887 
2888     if x_0 is None:
2889         return (None, None, None)
2890 
2891     if solution.count(0) >= 2:
2892         # if there are 2 zeros the equation reduces
2893         # to k*X**2 == 0 where X is x, y, or z so X must
2894         # be zero, too. So there is only the trivial
2895         # solution.
2896         return (None, None, None)
2897 
2898     if x_0 == 0:
2899         v[0], v[1] = v[1], v[0]
2900         y_p, x_p, z_p = _parametrize_ternary_quadratic(
2901             (y_0, x_0, z_0), v, coeff)
2902         return x_p, y_p, z_p
2903 
2904     x, y, z = v
2905     r, p, q = symbols("r, p, q", integer=True)
2906 
2907     eq = sum(k*v for k, v in coeff.items())
2908     eq_1 = _mexpand(eq.subs(zip(
2909         (x, y, z), (r*x_0, r*y_0 + p, r*z_0 + q))))
2910     A, B = eq_1.as_independent(r, as_Add=True)
2911 
2912 
2913     x = A*x_0
2914     y = (A*y_0 - _mexpand(B/r*p))
2915     z = (A*z_0 - _mexpand(B/r*q))
2916 
2917     return _remove_gcd(x, y, z)

3189 def gaussian_reduce(w:int, a:int, b:int) -> tuple[int, int]:
3190     r"""

3192     `X^2 - aZ^2 \equiv 0 \pmod{b}` so that `x^2 + |a|z^2` is as small as possible.

3204         u \cdot v := (wu_1 + bu_2)(wv_1 + bv_2) + |a|u_1v_1

3254            https://doi.org/10.1090/S0025-5718-02-01480-1

3258         return u[0]*v[0] + a*u[1]*v[1]

3281     a new reduced solution `(x', y', z')` such that `z'^2 \leq \mid ab \mid`. 
3282

3284     on page 8 of Cremona and Rusin's paper [1]_ and the work of Mordell in 
3285     reference [2]_. 
3286  
3287     References 
3288     ========== 
3289  
3290     .. [1] Cremona, J. E., Rusin, D. (2003). Efficient Solution of Rational Conics. 
3291            Mathematics of Computation, 72(243), 1417-1441. 
3292            https://doi.org/10.1090/S0025-5718-02-01480-1 
3293     .. [2] Diophantine Equations, L. J. Mordell, page 48. 
3294  
3295     """

3297     if _odd(c):
3298         k = 2*c
3299     else:
3300         k = c//2
3301 
3302     small = a*b*c
3303     step = 0
3304     while True:
3305         t1, t2, t3 = a*x**2, b*y**2, c*z**2
3306         # check that it's a solution
3307         if t1 + t2 != t3:
3308             if step == 0:
3309                 raise ValueError('bad starting solution')
3310             break
3311         x_0, y_0, z_0 = x, y, z
3312         if max(t1, t2, t3) <= small:
3313             # Holzer condition
3314             break
3315 
3316         uv = u, v = base_solution_linear(k, y_0, -x_0)
3317         if None in uv:
3318             break

3320         p, q = -(a*u*x_0 + b*v*y_0), c*z_0
3321         r = Rational(p, q)
3322         if _even(c):

3324             assert abs(w - r) <= S.Half
3325         else:

3327             if _odd(a*u + b*v + c*w):
3328                 w += 1
3329             assert abs(w - r) <= S.One

3334         y = Rational(y_0*A - 2*v*B, k)
3335         z = Rational(z_0*A - 2*w*B, k)
3336         assert all(i.is_Integer for i in (x, y, z))
