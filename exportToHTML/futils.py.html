9    def render_as_module(definitions, name, declarations=(), printer_settings=None):
10       """ Creates a ``Module`` instance and renders it as a string. 
11    
12       This generates Fortran source code for a module with the correct ``use`` statements. 
13    
14       Parameters 
15       ========== 
16    
17       definitions : iterable 
18           Passed to :class:`sympy.codegen.fnodes.Module`. 
19       name : str 
20           Passed to :class:`sympy.codegen.fnodes.Module`. 
21       declarations : iterable 
22           Passed to :class:`sympy.codegen.fnodes.Module`. It will be extended with 
23           use statements, 'implicit none' and public list generated from ``definitions``. 
24       printer_settings : dict 
25           Passed to ``FCodePrinter`` (default: ``{'standard': 2003, 'source_format': 'free'}``). 
26    
27       """
28       printer_settings = printer_settings or {'standard': 2003, 'source_format': 'free'}
29       printer = FCodePrinter(printer_settings)
30       dummy = Dummy()
31       if isinstance(definitions, Module):
32           raise ValueError("This function expects to construct a module on its own.")
33       mod = Module(name, chain(declarations, [dummy]), definitions)
34       fstr = printer.doprint(mod)
35       module_use_str = '   %s\n' % '   \n'.join(['use %s, only: %s' % (k, ', '.join(v)) for
36                                                   k, v in printer.module_uses.items()])
37       module_use_str += '   implicit none\n'
38       module_use_str += '   private\n'
39       module_use_str += '   public %s\n' % ', '.join([str(node.name) for node in definitions if getattr(node, 'name', None)])
40       return fstr.replace(printer.doprint(dummy), module_use_str)
