71   class GdsFile:
72       r"""Wrapper around cuFile. 
73    
74       cuFile is a file-like interface to the GPUDirect Storage (GDS) API. 
75    
76       See the `cufile docs <https://docs.nvidia.com/gpudirect-storage/api-reference-guide/index.html#cufile-io-api>`_ 
77       for more details. 
78    
79       Args: 
80           filename (str): Name of the file to open. 
81           flags (int): Flags to pass to ``os.open`` when opening the file. ``os.O_DIRECT`` will 
82               be added automatically. 
83    
84       Example:: 
85    
86           >>> # xdoctest: +SKIP("gds filesystem requirements") 
87           >>> src1 = torch.randn(1024, device="cuda") 
88           >>> src2 = torch.randn(2, 1024, device="cuda") 
89           >>> file = torch.cuda.gds.GdsFile(f, os.O_CREAT | os.O_RDWR) 
90           >>> file.save_storage(src1.untyped_storage(), offset=0) 
91           >>> file.save_storage(src2.untyped_storage(), offset=src1.nbytes) 
92           >>> dest1 = torch.empty(1024, device="cuda") 
93           >>> dest2 = torch.empty(2, 1024, device="cuda") 
94           >>> file.load_storage(dest1.untyped_storage(), offset=0) 
95           >>> file.load_storage(dest2.untyped_storage(), offset=src1.nbytes) 
96           >>> torch.equal(src1, dest1) 
97           True 
98           >>> torch.equal(src2, dest2) 
99           True 
100   
101      """
102  
103      def __init__(self, filename: str, flags: int):
104          if sys.platform == "win32":
105              raise RuntimeError("GdsFile is not supported on this platform.")
106          self.filename = filename
107          self.flags = flags
108          self.fd = os.open(filename, flags | os.O_DIRECT)  # type: ignore[attr-defined]
109          self.handle: Optional[int] = None
110          self.register_handle()
111  
112      def __del__(self) -> None:
113          if self.handle is not None:
114              self.deregister_handle()
115          os.close(self.fd)
116  
117      def register_handle(self) -> None:
118          """Registers file descriptor to cuFile Driver. 
119   
120          This is a wrapper around ``cuFileHandleRegister``. 
121          """
122          assert (
123              self.handle is None
124          ), "Cannot register a handle that is already registered."
125          self.handle = torch._C._gds_register_handle(self.fd)
126  
127      def deregister_handle(self) -> None:
128          """Deregisters file descriptor from cuFile Driver. 
129   
130          This is a wrapper around ``cuFileHandleDeregister``. 
131          """
132          assert (
133              self.handle is not None
134          ), "Cannot deregister a handle that is not registered."
135          torch._C._gds_deregister_handle(self.handle)
136          self.handle = None
137  
138      def load_storage(self, storage: Storage, offset: int = 0) -> None:
139          """Loads data from the file into the storage. 
140   
141          This is a wrapper around ``cuFileRead``. ``storage.nbytes()`` of data 
142          will be loaded from the file at ``offset`` into the storage. 
143   
144          Args: 
145              storage (Storage): Storage to load data into. 
146              offset (int, optional): Offset into the file to start loading from. (Default: 0) 
147          """
148          assert (
149              self.handle is not None
150          ), "Cannot load data from a file that is not registered."
151          torch._C._gds_load_storage(self.handle, storage, offset)
152  
153      def save_storage(self, storage: Storage, offset: int = 0) -> None:
154          """Saves data from the storage into the file. 
155   
156          This is a wrapper around ``cuFileWrite``. All bytes of the storage 
157          will be written to the file at ``offset``. 
158   
159          Args: 
160              storage (Storage): Storage to save data from. 
161              offset (int, optional): Offset into the file to start saving to. (Default: 0) 
162          """
163          assert (
164              self.handle is not None
165          ), "Cannot save data to a file that is not registered."
166          torch._C._gds_save_storage(self.handle, storage, offset)
