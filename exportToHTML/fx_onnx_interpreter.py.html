48   def _retrieve_or_adapt_input_to_graph_set(
49       fx_node_arg: fx_type_utils.Argument,
50       fx_name_to_onnxscript_value: dict[
51           str,
52           onnxscript_graph_building.TorchScriptTensor
53           | tuple[onnxscript_graph_building.TorchScriptTensor, ...],
54       ],
55       tracer: onnxscript_graph_building.TorchScriptTracingEvaluator,
56   ):
57       """Map FX value to TorchScript value. 
58    
59       When creating TorchScript graph from FX graph, we need a mapping from FX variable 
60       to TorchScript variable. This function maps FX variable, fx_node_arg, to torch.jit.Value. 
61       """
62       from onnxscript import opset18 as op
63   
64       onnx_tensor = fx_node_arg
65       if isinstance(onnx_tensor, torch.fx.Node):
66           # 1. fx_node_arg is a torch.fx.Node, which means
67           #    fx_node_arg stands for the output of that torch.fx.Node.
68           # 2. fx_node_arg (variable in torch.fx.Graph) is be mapped to
69           #    torch.jit.Value, fx_name_to_onnxscript_value[fx_node_arg.name],
70           #    in TorchScript graph.
71           return fx_name_to_onnxscript_value[onnx_tensor.name]
72       elif isinstance(onnx_tensor, (tuple, list)) and any(
73           isinstance(node, torch.fx.Node)
74           and fx_type_utils.is_torch_symbolic_type(node.meta.get("val"))
75           for node in onnx_tensor
76       ):
77           # This intends to handle dynamic axes. for example, if the input size of op.Expand
78           # is dynamic, each dimension would be variable (i.e., sym variable in Pytorch
79           # FX graph. Note that sym variable is mapped to tensor in ONNX Script world)
80           # calculated by other operators.
81           sequence_mixed_elements: list[
82               onnxscript_graph_building.TorchScriptTensor
83               | tuple[onnxscript_graph_building.TorchScriptTensor, ...]
84               | list[int]
85           ] = []
86           # onnx_tensor contains a list of scalars which could be one of
87           #   - tensor with empty shape,
88           #   - tensor with tensor with shape (1,),
89           #   - torch.SymInt,
90           #   - int
91           #   - ...
92           # They should all be promoted to tensor with shape (1,)
93           # in order to call ONNX's Concat.
94           for tensor in onnx_tensor:
95               # Prepare `tensor` as input of ONNX's Concat.
96   
97               if isinstance(
98                   tensor, torch.fx.Node
99               ) and fx_type_utils.is_torch_symbolic_type(tensor.meta.get("val")):
100                  # In this case, tensor is a torch.SymInt from Dynamo's perspective.
101                  # It might be mapped to tensor with shape () or (1,) in ONNX.
102                  element_value = fx_name_to_onnxscript_value[tensor.name]
103                  if isinstance(
104                      element_value, onnxscript_graph_building.TorchScriptTensor
105                  ):
106                      # All elements sequence_mixed_elements will be send to onnx's Concat
107                      # as inputs. Therefore, they are required to have the same rank.
108                      # Since tensors with rank=0 (i.e., scalar) cannot be concated, all
109                      # scalars are promoted to tensors with shape (1,).
110                      with onnxscript.evaluator.default_as(tracer):
111                          element_value = op.Reshape(
112                              element_value,  # type: ignore[arg-type, type-var]
113                              [1],  # type: ignore[arg-type, type-var]
114                          )
115                  sequence_mixed_elements.append(element_value)
116              elif isinstance(tensor, int):
117                  # NOTE: op.Concat doesn't support scalar, so we need to wrap it with
118                  # dim, and onnx-script will promote it to tensor(int64)
119                  sequence_mixed_elements.append([tensor])
120              else:
121                  raise RuntimeError(
122                      f"Unsupported type in sequence_mixed_elements: {type(tensor)}"
123                  )
124          # Concat all the elements in the sequence.
125          # shapes are mapped to tensors in ONNX graph (TorchScriptGraph),
126          # so list of sym_ints is concatenated to a tensor before calling ONNX op.
127  
128          # For example:
129          #    inputs: [[2], [4], fx.Node(SymIntA), [1], fx.Node(SymIntB)]
130          #    outputs: op.Concat([op.Constant(2), op.Constant(4), TorchScriptTensor(A), op.Constant(1), TorchScriptTensor(B)])
131  
132          # onnx-script auto wraps python number with op.Constants,
133          # so we don't need to specifically process them.
134          with onnxscript.evaluator.default_as(tracer):
135              output = op.Concat(*sequence_mixed_elements, axis=0)  # type: ignore[type-var]
136          output.dtype = torch.int64  # type: ignore[union-attr]
137          output.shape = [len(sequence_mixed_elements)]  # type: ignore[union-attr]
138          return output
139      elif isinstance(onnx_tensor, (tuple, list)) and all(
140          isinstance(node, torch.fx.Node) or node is None for node in onnx_tensor
141      ):
142          sequence_elements: list[
143              onnxscript_graph_building.TorchScriptTensor
144              | None
145              | tuple[onnxscript_graph_building.TorchScriptTensor, ...]
146          ] = []
147          for tensor in onnx_tensor:
148              sequence_elements.append(
149                  fx_name_to_onnxscript_value[tensor.name] if tensor is not None else None  # type: ignore[index, union-attr]
150              )
151          return sequence_elements
152      if isinstance(onnx_tensor, torch.dtype):
153          onnx_tensor = int(  # type: ignore[call-overload]
154              jit_type_utils.JitScalarType.from_dtype(onnx_tensor).onnx_type()
155          )
156      # NOTE: if device is specified in kwargs (not consumed), it's free to ignored. But
157      # if it's in args, we need to set it to string for dispatcher to match schema.
158      if isinstance(onnx_tensor, torch.device):
159          # torch.device is not supported by onnxscript (no op). We turn it into
160          # a string.
161          return str(onnx_tensor)
162      # all other cases, we do nothing.
163      return onnx_tensor

166  def filter_incompatible_and_dtype_convert_kwargs(kwargs):
167      """Filter out kwargs that are not supported by onnxscript."""
168      filtered = {}
169      for key, value in kwargs.items():
170          if key in {
171              "layout",
172              "device",
173              "requires_grad",
174              "pin_memory",
175              "memory_format",
176              "implicit",
177          }:
178              continue
179          if key == "dtype":
180              if value is None:
181                  # We omit if dtype is not provided, because onnxscript handles the
182                  # default case.
183                  continue
184              else:
185                  value = int(jit_type_utils.JitScalarType.from_dtype(value).onnx_type())  # type: ignore[call-overload]
186          filtered[key] = value
187      return filtered

190  def _fill_tensor_shape_type(
191      onnxscript_values: onnxscript_graph_building.TorchScriptTensor
192      | tuple[onnxscript_graph_building.TorchScriptTensor, ...],
193      name: str,
194      expected_values: fx_type_utils.META_VALUE_TYPE
195      | list[fx_type_utils.META_VALUE_TYPE]
196      | tuple[fx_type_utils.META_VALUE_TYPE | None, ...],
197  ):
198      """Fill the meta information of onnxscript_values with that from the fx FakeTensor."""
199  
200      if isinstance(expected_values, (list, tuple)) and not isinstance(
201          onnxscript_values, (list, tuple)
202      ):
203          # ex: aten::split - in onnx_dtype: seq(tensor)
204          # onnxscript_values is a single tensor, but expected_values is a list of tensors.
205          return
206  
207      flat_onnxscript_values, _ = _pytree.tree_flatten(onnxscript_values)
208      flat_expected_values, _ = _pytree.tree_flatten(expected_values)
209      for i, (onnxscript_value, expected_value) in enumerate(
210          zip(flat_onnxscript_values, flat_expected_values)
211      ):
212          if expected_value is None:
213              # There is no shape/type from None.
214              # NOTE: according to https://github.com/pytorch/pytorch/blob/main/torch/_meta_registrations.py,
215              # None could be a valid value for return type, so we need to handle it.
216              # e.g. the function: meta__scaled_dot_product_flash() in cpu mode.
217              continue
218          elif fx_type_utils.is_torch_symbolic_type(expected_value):
219              # aten::sym_size output is a int, not a tensor, which stands
220              # for the size of one dim. We treat it as 1-D tensor.
221              onnxscript_value.dtype = fx_type_utils.from_sym_value_to_torch_dtype(
222                  expected_value
223              )
224              onnxscript_value.shape = torch.Size([1])
225          elif isinstance(expected_value, (int, float, bool)):
226              onnxscript_value.dtype = fx_type_utils.from_scalar_type_to_torch_dtype(
227                  type(expected_value)
228              )
229              onnxscript_value.shape = torch.Size([])
230          elif isinstance(expected_value, complex):
231              # From complex scalar to real representation
232              onnxscript_value_to_torch_dtype = (
233                  fx_type_utils.from_scalar_type_to_torch_dtype(type(expected_value))
234              )
235              onnxscript_value.dtype = (
236                  fx_type_utils.from_complex_to_float(onnxscript_value_to_torch_dtype)
237                  if onnxscript_value_to_torch_dtype is not None
238                  else None
239              )
240              onnxscript_value.shape = torch.Size([2])
241          elif fx_type_utils.is_torch_complex_dtype(expected_value.dtype):
242              # Like torch.view_as_real, we flatten complex tensors to real tensors with
243              # additional last dimension of 2
244              onnxscript_value.shape = torch.Size((*expected_value.size(), 2))
245              # complex64 -> float32, complex128 -> float64, etc.
246              onnxscript_value.dtype = fx_type_utils.from_complex_to_float(
247                  expected_value.dtype
248              )
249              # Dispatcher needs to know the value is complex
250              onnxscript_value.is_complex = True
251          else:
252              # We set node output sizes to be dynamic to continue the model conversion,
253              # and inputs are also set to be dynamic in add_input().
254              onnxscript_value.shape = expected_value.size()
255              onnxscript_value.dtype = expected_value.dtype
256  
257          # naming
258          if i > 0:
259              onnxscript_value.name = f"{name}_{i}"
260          else:
261              onnxscript_value.name = name

264  def _fill_in_default_kwargs(
265      node: torch.fx.Node,
266  ) -> tuple[list[fx_type_utils.Argument], dict[str, fx_type_utils.Argument]]:
267      """Find and Fill in the not provided kwargs with default values."""
268  
269      # TODO: aten::sym_size has overload, but fx graph is using
270      # overloadpacket for some reasons.
271      # https://github.com/pytorch/pytorch/issues/97201
272      # We manually assigned overload for aten::sym_size.
273      if hasattr(node.target, "_schema"):
274          node_schema = node.target._schema  # type: ignore[union-attr]
275      else:
276          node_schema = torch.ops.aten.sym_size.int._schema  # type: ignore[union-attr]
277  
278      # This function assumes the order of arguments in FX op is the
279      # same as the order of arguments in TorchScript op.
280      complete_args: list[fx_type_utils.Argument] = []
281      complete_kwargs: dict[str, fx_type_utils.Argument] = {}
282  
283      if inspect.isbuiltin(node.target):
284          complete_args = list(node.args)
285      else:
286          for i, expected_arg in enumerate(node_schema.arguments):
287              if i < len(node.args):
288                  complete_args.append(node.args[i])
289              elif expected_arg.name in node.kwargs:
290                  complete_kwargs[expected_arg.name] = node.kwargs[expected_arg.name]
291              else:
292                  # Get default from schema.
293                  complete_kwargs[expected_arg.name] = expected_arg.default_value
294  
295      return complete_args, complete_kwargs

298  def _wrap_fx_args_as_onnxscript_args(
299      complete_args: list[fx_type_utils.Argument],
300      complete_kwargs: dict[str, fx_type_utils.Argument],
301      fx_name_to_onnxscript_value: dict[
302          str,
303          onnxscript_graph_building.TorchScriptTensor
304          | tuple[onnxscript_graph_building.TorchScriptTensor, ...],
305      ],
306      tracer: onnxscript_graph_building.TorchScriptTracingEvaluator,
307  ) -> tuple[
308      Sequence[
309          onnxscript_graph_building.TorchScriptTensor
310          | str
311          | int
312          | float
313          | bool
314          | list
315          | complex
316          | None
317      ],
318      dict[str, fx_type_utils.Argument],
319  ]:
320      """Map all FX arguments of a node to arguments in TorchScript graph."""
321  
322      onnxscript_args = tuple(
323          _retrieve_or_adapt_input_to_graph_set(arg, fx_name_to_onnxscript_value, tracer)
324          for arg in complete_args
325      )
326      onnxscript_kwargs = filter_incompatible_and_dtype_convert_kwargs(complete_kwargs)
327  
328      return onnxscript_args, onnxscript_kwargs

331  class FxOnnxInterpreter:
332      """Stateless class to process FX graph Nodes and translate them into their ONNX counterparts. 
333   
334      All FX nodes described by [FX Graph](https://pytorch.org/docs/stable/fx.html#torch.fx.Graph) are supported. 
335      Similarly to [FX Interpreter pattern](https://pytorch.org/docs/stable/fx.html#torch.fx.Interpreter), each FX node 
336      must be implemented on its own method in this class. 
337   
338      Each operator's implementation returns either an `onnxscript.OnnxFunction` or 
339      `onnxscript.TracedOnnxFunction` instance based on the dispatch algorithm. They can 
340      also raise RuntimeError: If there are no overloaded functions available for the given FX node. 
341      """
342  
343      def run_node(
344          self,
345          node,
346          fx_graph_module: torch.fx.GraphModule,
347          onnxfunction_dispatcher: onnxfunction_dispatcher.OnnxFunctionDispatcher,
348          onnxscript_graph: onnxscript_graph_building.TorchScriptGraph,
349          onnxscript_tracer: onnxscript_graph_building.TorchScriptTracingEvaluator,
350          fx_name_to_onnxscript_value: dict[
351              str,
352              onnxscript_graph_building.TorchScriptTensor
353              | tuple[onnxscript_graph_building.TorchScriptTensor, ...],
354          ],
355      ):
356          """Execute a single FX node to produce its ONNX counterpart. 
357   
358          Args: 
359              node: The FX node to be translated. 
360              fx_graph_module: The FX graph module containing the node. 
361              onnxfunction_dispatcher: The dispatcher to find the best matched ONNX op. 
362              onnxscript_graph: The ONNX graph to be populated. 
363              onnxscript_tracer: The tracer to trace the ONNX graph. 
364              fx_name_to_onnxscript_value: The mapping from FX node name to ONNX Script value. 
365   
366          Raises: 
367              RuntimeError: When a node.op is not supported. 
368          """
369          if node.op == "placeholder":
370              self.placeholder(node, onnxscript_graph, fx_name_to_onnxscript_value)
371          elif node.op == "get_attr":
372              self.get_attr(
373                  node,
374                  onnxscript_graph,
375                  fx_name_to_onnxscript_value,
376                  fx_graph_module,
377              )
378          elif node.op == "call_function":
379              self.call_function(
380                  node,
381                  onnxscript_tracer,
382                  fx_name_to_onnxscript_value,
383                  onnxfunction_dispatcher,
384                  fx_graph_module,
385              )
386          elif node.op == "call_method":
387              self.call_method(node)
388          elif node.op == "call_module":
389              self.call_module(
390                  node,
391                  onnxscript_graph,
392                  fx_name_to_onnxscript_value,
393                  onnxscript_tracer,
394                  fx_graph_module,
395                  onnxfunction_dispatcher,
396              )
397          elif node.op == "output":
398              self.output(node, onnxscript_graph, fx_name_to_onnxscript_value)
399          else:
400              raise RuntimeError(f"Found node type not defined in torch.fx: {node.op}")
401  
402      def run(
403          self,
404          fx_graph_module: torch.fx.GraphModule,
405          onnxfunction_dispatcher: onnxfunction_dispatcher.OnnxFunctionDispatcher,
406          parent_onnxscript_graph: onnxscript_graph_building.TorchScriptGraph
407          | None = None,
408      ) -> onnxscript_graph_building.TorchScriptGraph:
409          """Analyze all FX nodes and trigger their ONNX translation. 
410   
411          Args: 
412              fx_graph_module: FX graph module to be translated. 
413              onnxfunction_dispatcher: ONNX function dispatcher. 
414              parent_onnxscript_graph: The parent TorchScript graph. Must be provided if 
415                  `fx_graph_module` is a submodule. If not provided, 
416                  `fx_graph_module` is assumed to be the root module. 
417          """
418          if parent_onnxscript_graph is not None:
419              # If parent_onnxscript_graph is provided, we assume fx_graph_module is a
420              # submodule representing a forward call of an nn.Module.
421              # Compose package and version where the nn.Module is defined as domain name
422              # for the local function.
423  
424              onnx_meta: _pass.GraphModuleOnnxMeta | None = fx_graph_module.meta.get(
425                  "onnx"
426              )
427              if onnx_meta is None:
428                  raise RuntimeError(
429                      f"ONNX meta is not found in submodule {fx_graph_module._get_name()}. "
430                      f"Only submodules produced by `Modularize` pass is supported in ONNX export."
431                  )
432  
433              onnx_domain = onnx_meta.package_info.to_onnx_domain_string()
434          else:
435              # Leave as default domain name for the root module.
436              onnx_domain = None
437  
438          onnxscript_graph = onnxscript_graph_building.TorchScriptGraph(
439              parent_onnxscript_graph, domain_name=onnx_domain
440          )
441          onnxscript_tracer = onnxscript_graph_building.TorchScriptTracingEvaluator(
442              onnxscript_graph
443          )
444          # In the following loop, a TorchScript graph is created to
445          # represent the input FX graph with ONNX symbols (e.g., onnx::add).
446          # To connect the values to nodes in the TorchScript graph, we maintain
447          # fx_name_to_onnxscript_value. Basically, we want to translate
448          #   fx_tensor_x (type: torch.fx.Node) -> fx_node_1 -> fx_tensor_y (type: torch.fx.Node)
449          # to
450          #   fx_name_to_onnxscript_value[fx_tensor_x.name] -> onnx_node_1 -> fx_name_to_onnxscript_value[fx_tensor_y.name]
451          fx_name_to_onnxscript_value: dict[
452              str,
453              onnxscript_graph_building.TorchScriptTensor
454              | tuple[onnxscript_graph_building.TorchScriptTensor, ...],
455          ] = {}
456  
457          # TODO: Fix FakeTensorMode limitation asap
458          # We want to pass list of ints and floats to TorchScript graph correctly
459          # in _export_fx_to_ts, so we must disable FakeTensorMode. Otherwise, graph may
460          # receive FakeTensor and results runtime error. In addition, TorchScript-based
461          # ONNX exporter used in _ts_graph_to_onnx_model_in_protobuf is not compatible
462          # with FakeTensorMode.
463          with torch.utils._mode_utils.no_dispatch():
464              for node in fx_graph_module.graph.nodes:
465                  self.run_node(
466                      node,
467                      fx_graph_module,
468                      onnxfunction_dispatcher,
469                      onnxscript_graph,
470                      onnxscript_tracer,
471                      fx_name_to_onnxscript_value,
472                  )
473  
474          return onnxscript_graph
475  
476      def placeholder(
477          self,
478          node: torch.fx.Node,
479          onnxscript_graph: onnxscript_graph_building.TorchScriptGraph,
480          fx_name_to_onnxscript_value: dict[
481              str,
482              onnxscript_graph_building.TorchScriptTensor
483              | tuple[onnxscript_graph_building.TorchScriptTensor, ...],
484          ],
485      ):
486          # Input of graph.
487          # The node.meta["val"] is generated by FakeTensorProp.
488          # NOTE: add_input() intends to create nodes with shape/type
489          fake_tensor = node.meta.get("val", None)
490          # NOTE: During the tracing, when inputs are constants, they are represented
491          # by nodes with node.meta['val'] being None (nn.Module to dynamo_export)
492          # or nodes with node.meta['val'] being a builtin value (ExportedProgram to dynamo_export).
493          # Nonethless, the nodes are not consumed by others, so we don't need to
494          # create a TorchScriptTensor for them.
495          if fake_tensor is None or isinstance(fake_tensor, (int, float, bool, str)):
496              output = onnxscript_graph.add_input(
497                  input_name=None,
498              )
499          elif isinstance(fake_tensor, torch.Tensor):
500              # NOTE: ONNX doesn't support tensor of complex64/complex128, so we
501              # convert them to float32/float64 with real representation.
502              if fx_type_utils.is_torch_complex_dtype(fake_tensor.dtype):
503                  fake_tensor = torch.view_as_real(fake_tensor.resolve_conj())
504              output = onnxscript_graph.add_input(
505                  input_name=node.name,
506                  shape=fake_tensor.shape,
507                  dtype=fake_tensor.dtype,
508              )
509  
510          elif fx_type_utils.is_torch_symbolic_type(fake_tensor):
511              output = onnxscript_graph.add_input(
512                  input_name=node.name,
513                  shape=torch.Size([]),
514                  dtype=fx_type_utils.from_sym_value_to_torch_dtype(fake_tensor),
515              )
516          else:
517              raise RuntimeError(
518                  f"Unsupported type(node.meta['val']) for placeholder: {type(fake_tensor)}"
519              )
520          assert output is not None, (
521              f"Node creates None with target={node.target} and name={node.name}"
522          )
523  
524          assert isinstance(output, onnxscript_graph_building.TorchScriptTensor)
525          assert isinstance(output, onnxscript.tensor.Tensor)
526  
527          fx_name_to_onnxscript_value[node.name] = output
528  
529      def call_function(
530          self,
531          node: torch.fx.Node,
532          onnxscript_tracer: onnxscript_graph_building.TorchScriptTracingEvaluator,
533          fx_name_to_onnxscript_value: dict[
534              str,
535              onnxscript_graph_building.TorchScriptTensor
536              | tuple[onnxscript_graph_building.TorchScriptTensor, ...],
537          ],
538          onnxfunction_dispatcher: onnxfunction_dispatcher.OnnxFunctionDispatcher,
539          fx_graph_module: torch.fx.GraphModule,
540      ):
541          # aten ops and other stateless functions.
542          if node.target == operator.getitem and isinstance(
543              fx_name_to_onnxscript_value[node.args[0].name],  # type: ignore[union-attr,index]
544              tuple,
545          ):
546              onnx_tensor_tuple = fx_name_to_onnxscript_value[node.args[0].name]  # type: ignore[union-attr,index]
547              index = node.args[1]
548              value = onnx_tensor_tuple[index]  # type: ignore[index]
549              assert value is not None, (
550                  f"Node creates None with target={node.target} and name={node.name}"
551              )
552              assert isinstance(
553                  value, (onnxscript_graph_building.TorchScriptTensor, tuple)
554              ), type(value)
555  
556              fx_name_to_onnxscript_value[node.name] = value
557              return
558  
559          # Map FX inputs to ONNX inputs and fill optional inputs with default values.
560          # torch_args and torch_kwargs are for op-level validation
561          fx_args, fx_kwargs = _fill_in_default_kwargs(node)
562  
563          onnx_args, onnx_kwargs = _wrap_fx_args_as_onnxscript_args(
564              fx_args,
565              fx_kwargs,
566              fx_name_to_onnxscript_value,
567              onnxscript_tracer,
568          )
569          # Dispatch to ONNX op through OpShema. The input argument dtypes are compared to
570          # function signature in OpSchema, and find the best matched overload.
571          symbolic_fn = onnxfunction_dispatcher.dispatch(
572              node=node,
573              onnx_args=onnx_args,  # type: ignore[arg-type]
574              onnx_kwargs=onnx_kwargs,
575          )
576          with onnxscript.evaluator.default_as(onnxscript_tracer):
577              output: (
578                  onnxscript_graph_building.TorchScriptTensor
579                  | tuple[onnxscript_graph_building.TorchScriptTensor, ...]
580              ) = symbolic_fn(*onnx_args, **onnx_kwargs)
581          assert output is not None, (
582              f"Node creates None with target={node.target}, name={node.name}, args={onnx_args}, kwargs={onnx_kwargs}"
583          )
584          # Assign type and shape from fx graph.
585          _fill_tensor_shape_type(output, node.name, node.meta["val"])
586          # One fx node could produce multiple outputs (e.g., tuple of tensors); in
587          # that case, v is a tuple of TorchScriptTensors.
588          assert isinstance(
589              output, (onnxscript_graph_building.TorchScriptTensor, tuple)
590          ), type(output)
591          fx_name_to_onnxscript_value[node.name] = output
592  
593      def output(
594          self,
595          node: torch.fx.Node,
596          onnxscript_graph: onnxscript_graph_building.TorchScriptGraph,
597          fx_name_to_onnxscript_value: dict[
598              str,
599              onnxscript_graph_building.TorchScriptTensor
600              | tuple[onnxscript_graph_building.TorchScriptTensor, ...],
601          ],
602      ):
603          if isinstance(node.args[0], torch.fx.Node):
604              onnx_tensor_or_tensor_tuple = fx_name_to_onnxscript_value[node.args[0].name]
605              onnxscript_graph.register_outputs(onnx_tensor_or_tensor_tuple)
606          else:
607              # ONNX can't represent collection types (e.g., dictionary, tuple of tuple of
608              # tensor, etc), we flatten the collection and register each element as output.
609              flat_args, _ = _pytree.tree_flatten(node.args[0])
610              for arg in flat_args:
611                  assert isinstance(arg, torch.fx.Node), (
612                      f"arg must be a torch.fx.Node, not {type(arg)}"
613                  )
614                  onnx_tensor_or_tensor_tuple = fx_name_to_onnxscript_value[arg.name]
615                  onnxscript_graph.register_outputs(onnx_tensor_or_tensor_tuple)

621      def call_module(
622          self,
623          node: torch.fx.Node,
624          parent_onnxscript_graph: onnxscript_graph_building.TorchScriptGraph,
625          fx_name_to_onnxscript_value: dict[
626              str,
627              onnxscript_graph_building.TorchScriptTensor
628              | tuple[onnxscript_graph_building.TorchScriptTensor, ...],
629          ],
630          tracer: onnxscript_graph_building.TorchScriptTracingEvaluator,
631          root_fx_graph_module: torch.fx.GraphModule,
632          onnxfunction_dispatcher: onnxfunction_dispatcher.OnnxFunctionDispatcher,
633      ) -> None:
634          """Export a fx.GraphModule submodule to ONNXScript graph. 
635   
636          The export process specifically targets `call_module` nodes that are created by 
637          the exporter's `Modularize` pass. Each `call_module` node has an associated fx.GraphModule 
638          by `node.target` underneath the root fx.GraphModule. These `call_module` nodes are exported as ONNX 
639          function nodes. The related `sub_module` is then exported as an ONNX model local function, 
640          which is represented by another `TorchScriptGraph`. This `TorchScriptGraph` sets the current 
641          `onnxscript_graph` as its parent. 
642   
643          Args: 
644              node: The call_module node in the FX graph that represents the submodule call. 
645              parent_onnxscript_graph: The parent ONNXScript graph to which the ONNX function and 
646                  function node belong. 
647              fx_name_to_onnxscript_value: The mapping from FX node name to ONNXScript value. 
648              tracer: The tracer used to trace the ONNXScript graph. 
649              root_fx_graph_module: The root FX module. 
650              onnxfunction_dispatcher: The dispatcher. 
651          """
652          assert isinstance(node.target, str), (
653              f"node.target must be a str, not {type(node.target)} for node {node}."
654          )
655  
656          sub_module = root_fx_graph_module.get_submodule(node.target)
657  
658          assert isinstance(sub_module, torch.fx.GraphModule), (
659              f"sub_module must be a torch.fx.GraphModule, not {type(sub_module)} for node {node}."
660          )
661  
662          sub_onnxscript_graph = self.run(
663              sub_module, onnxfunction_dispatcher, parent_onnxscript_graph
664          )
665  
666          onnx_args, _ = _wrap_fx_args_as_onnxscript_args(
667              list(node.args), {}, fx_name_to_onnxscript_value, tracer
668          )
669  
670          # TODO: We may want to consider other naming styles. The goal is to be stable and
671          # unique such that it can be easily identified in case of kernel substitution.
672          # Example for current style is combination of qualified module class name and
673          # module attribute name: `torch_nn_modules_conv_Conv2d_conv1`.
674          # Other naming styles such as qualified module class name made unique can also
675          # be considered.
676          unique_module_name = f"{sub_module._get_name()}_{node.target}"
677  
678          outputs: (
679              onnxscript_graph_building.TorchScriptTensor
680              | tuple[onnxscript_graph_building.TorchScriptTensor, ...]
681          ) = parent_onnxscript_graph.add_module_call(  # type: ignore[assignment]
682              unique_module_name, sub_onnxscript_graph, onnx_args
683          )
684  
685          assert isinstance(
686              outputs, (onnxscript_graph_building.TorchScriptTensor, tuple)
687          ), f"Unexpected outputs type {type(outputs)} for node {node}."
688  
689          _fill_tensor_shape_type(outputs, node.name, node.meta["val"])
690          fx_name_to_onnxscript_value[node.name] = outputs
691  
692          # Skip op_level_validation for call_module. Subgraph nodes are validated individually.
693  
694      def get_attr(
695          self,
696          node: torch.fx.Node,
697          onnxscript_graph: onnxscript_graph_building.TorchScriptGraph,
698          fx_name_to_onnxscript_value: dict[
699              str,
700              onnxscript_graph_building.TorchScriptTensor
701              | tuple[onnxscript_graph_building.TorchScriptTensor, ...],
702          ],
703          fx_graph_module: torch.fx.GraphModule,
704      ):
705          assert isinstance(node.target, str), f"node.target {node.target} is not a str."
706          attr_tensor = getattr(fx_graph_module, node.target)
707          assert isinstance(attr_tensor, torch.Tensor), f"{attr_tensor} is not a tensor."
708  
709          # Parameter/buffer name cannot contain "."
710          # Revert from "/" to restore namespace formatting.
711          input_ = onnxscript_graph.add_initializer(
712              name=node.target.replace("/", "."),
713              value=attr_tensor,
714          )
715  
716          assert isinstance(input_, onnxscript_graph_building.TorchScriptTensor)
717          assert isinstance(input_, onnxscript.tensor.Tensor)
718          fx_name_to_onnxscript_value[node.name] = input_
