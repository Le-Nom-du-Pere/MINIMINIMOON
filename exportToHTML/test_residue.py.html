320  def test_binomial_p_pow():
321      n, binomials, binomial = 1000, [1], 1
322      for i in range(1, n + 1):
323          binomial *= n - i + 1
324          binomial //= i
325          binomials.append(binomial)
326  
327      # Test powers of two, which the algorithm treats slightly differently
328      trials_2 = 100
329      for _ in range(trials_2):
330          m, power = randint(0, n), randint(1, 20)
331          assert _binomial_mod_prime_power(n, m, 2, power) == binomials[m] % 2**power
332  
333      # Test against other prime powers
334      primes = list(sieve.primerange(2*n))
335      trials = 1000
336      for _ in range(trials):
337          m, prime, power = randint(0, n), choice(primes), randint(1, 10)
338          assert _binomial_mod_prime_power(n, m, prime, power) == binomials[m] % prime**power
