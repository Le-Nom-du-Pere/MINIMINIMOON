50   @compatibility(is_backward_compatible=True)
51   class ProxyableClassMeta(type):
52       """ 
53       ProxyableClassMeta allows you to make construction of a given Python class 
54       symbolically traceable. For example:: 
55    
56           import torch 
57           import torch.fx 
58    
59    
60           class TensorPair(metaclass=torch.fx.ProxyableClassMeta): 
61               def __init__(self, left, right): 
62                   self.left, self.right = left, right 
63    
64               def add(self, other): 
65                   l = self.left + other.left 
66                   r = self.right + other.right 
67                   return TensorPair(l, r) 
68    
69               def mul(self, other): 
70                   l = self.left * other.left 
71                   r = self.right * other.right 
72                   return TensorPair(l, r) 
73    
74    
75           def use_tensor_pair_ctor(x: TensorPair, y: torch.Tensor): 
76               s = x.add(TensorPair(y, y)) 
77               return s.mul(x) 
78    
79    
80           x = TensorPair(torch.randn(5, 3), torch.randn(5, 3)) 
81           y = torch.randn(5, 3) 
82           ref_out = use_tensor_pair_ctor(x, y) 
83    
84           traced = torch.fx.symbolic_trace(use_tensor_pair_ctor) 
85           print(traced.code) 
86           ''' 
87           def forward(self, x : __main___TensorPair, y : torch.Tensor): 
88               tensor_pair = __main___TensorPair(y, y);  y = None 
89               add = x.add(tensor_pair);  tensor_pair = None 
90               mul = add.mul(x);  add = x = None 
91               return mul 
92           ''' 
93    
94       From this example, we can see that construction of a class (``TensorPair``) 
95       defined with ``ProxyableClassMeta`` as metaclass can be recorded in symbolic 
96       tracing. 
97       """
98   
99       def __init__(cls, name, bases, attrs):
100          _proxyable_classes.setdefault(cls)
101          super().__init__(name, bases, attrs)
102  
103      def __call__(cls, *args, **kwargs):
104          instance = cls.__new__(cls)  # type: ignore[call-overload]
105  
106          if not is_fx_tracing():
107              cls.__init__(instance, *args, **kwargs)  # type: ignore[misc]
108              return instance
109  
110          found_proxies = []
111  
112          def check_proxy(a):
113              if isinstance(a, Proxy):
114                  found_proxies.append(a)
115  
116          map_aggregate(args, check_proxy)
117          map_aggregate(kwargs, check_proxy)
118  
119          if len(found_proxies) != 0:
120              tracer = found_proxies[0].tracer
121              return tracer.create_proxy("call_function", cls, args, kwargs)
122          else:
123              cls.__init__(instance, *args, **kwargs)  # type: ignore[misc]
124              return instance

227      for attr_name in dir(fr):
228          attr_val = getattr(fr, attr_name)
229          if (
230              not callable(attr_val)
231              and not attr_name.startswith("__")
232              and not hasattr(to, attr_name)
233          ):

240      # The first line of this docstring overrides the one Sphinx generates for the

244      """Tracer(autowrap_modules=(math,), autowrap_functions=()) 
245   
246      ``Tracer`` is the class that implements the symbolic tracing functionality 
247      of ``torch.fx.symbolic_trace``. A call to ``symbolic_trace(m)`` is equivalent 
248      to ``Tracer().trace(m)``. 
249   
250      Tracer can be subclassed to override various behaviors of the tracing 
251      process. The different behaviors that can be overridden are described 
252      in the docstrings of the methods on this class. 
253      """

259      def __init__(

262          autowrap_functions: tuple[Callable, ...] = (),
263          param_shapes_constant: bool = False,

265          # This method's signature is overridden by the first line of this class'
266          # docstring. If this method's signature is modified, the signature that
267          # overrides it also should be modified accordingly.

269          """ 
270          Construct a Tracer object. 
271   
272          Args: 
273   
274              autowrap_modules (Tuple[ModuleType]): defaults to `(math, )`, 
275                  Python modules whose functions should be wrapped automatically 
276                  without needing to use fx.wrap(). Backward-compatibility for 
277                  this parameter is guaranteed. 
278   
279              autowrap_functions (Tuple[Callable, ...]): defaults to `()`, 
280                  Python functions that should be wrapped automatically without 
281                  needing to use fx.wrap(). Backward compatibility for this 
282                  parameter is guaranteed. 
283   
284              param_shapes_constant (bool): When this flag is set,  calls to shape,

288                  is guaranteed.

301          }

321      @compatibility(is_backward_compatible=True)
322      def get_fresh_qualname(self, prefix: str) -> str:
323          """ 
324          Gets a fresh name for a prefix and returns it. This function ensures 
325          that it will not clash with an existing attribute on the graph. 
326          """
327          # The idea here is that if the module doesn't have this prefix at all we
328          # should reset the counter to start from the beginning
329          # It's a ... little bit hacky (doesn't cover all cases) but the precise
330          # naming of the prefixes isn't a correctness issue, just a niceness
331          # issue
332          qualname = f"{prefix}0"
333          if not hasattr(self.root, qualname):
334              self._qualname_counter[prefix] = 0
335              return qualname
336  
337          i = self._qualname_counter[prefix]
338          while True:
339              qualname = f"{prefix}{i}"
340              i += 1
341              if not hasattr(self.root, qualname):
342                  break
343          self._qualname_counter[prefix] = i
344  
345          return qualname

347      @compatibility(is_backward_compatible=True)
348      def create_arg(self, a: Any) -> "Argument":
349          """ 
350          A method to specify the behavior of tracing when preparing values to 
351          be used as arguments to nodes in the ``Graph``. 
352   
353          By default, the behavior includes: 
354   
355          #. Iterate through collection types (e.g. tuple, list, dict) and recursively 
356             call ``create_args`` on the elements. 
357          #. Given a Proxy object, return a reference to the underlying IR ``Node`` 
358          #. Given a non-Proxy Tensor object, emit IR for various cases: 
359   
360              * For a Parameter, emit a ``get_attr`` node referring to that Parameter 
361              * For a non-Parameter Tensor, store the Tensor away in a special 
362                attribute referring to that attribute. 
363   
364          This method can be overridden to support more types. 
365   
366          Args: 
367   
368              a (Any): The value to be emitted as an ``Argument`` in the ``Graph``. 
369   
370   
371          Returns: 
372   
373              The value ``a`` converted into the appropriate ``Argument`` 
374          """
375          # The base tracer is used to construct Graphs when there is no associated
376          # module hierarchy, so it can never create parameter references.
377          # The default tracer adds the ability to refer to parameters when
378          # tracing modules.
379          if isinstance(a, torch.nn.Parameter):
380              for n, p in self.root.named_parameters():
381                  if a is p:
382                      return self.create_node("get_attr", n, (), {})
383              raise NameError("parameter is not a member of this module")
384          elif isinstance(a, torch.Tensor):
385              for n_, p_ in self.root.named_buffers():
386                  if a is p_:
387                      return self.create_node("get_attr", n_, (), {})
388          elif isinstance(a, torch.nn.Module):
389              for n_, p_ in self.root.named_modules():
390                  if a is p_:
391                      return self.create_node("get_attr", n_, (), {})
392          # For NamedTuple instances that appear literally as args, we emit
393          # a node to construct the NamedTuple and use that Node as the argument.
394          if isinstance(a, tuple) and hasattr(a, "_fields"):
395              args = tuple(self.create_arg(elem) for elem in a)
396              return self.create_node("call_function", a.__class__, args, {})
397  
398          # Tensors do not have a reliable string repr() from which they can be
399          # constructed (and we probably don't want to rely on that, either), so
400          # for any constant Tensor values we encounter, first search for if they
401          # are an attribute of some module in the module hierarchy. If so, emit
402          # a get_attr to retrieve that tensor. Otherwise, we'll store away the
403          # tensor value into a special attribute on the Module s.t. we can
404          # retrieve it with a get_attr.
405          if isinstance(a, _constant_attribute_types):
406              qualname: Optional[str] = self.tensor_attrs.get(a)
407  
408              # Tensor was not found in the Module hierarchy, stow it away in a
409              # special attribute and set the qualname to refer to that
410              if not qualname:
411                  if isinstance(a, torch.Tensor):
412                      base_name = "_tensor_constant"
413                  elif isinstance(a, (FakeScriptObject, ScriptObject)):
414                      base_name = "_torchbind_obj"
415                  elif isinstance(a, pytree.TreeSpec):
416                      base_name = "_tree_spec_constant"
417                  else:
418                      raise RuntimeError(
419                          f"cannot create constant arg for {a} of type {type(a)}."
420                      )
421                  qualname = self.get_fresh_qualname(base_name)
422                  assert isinstance(qualname, str)
423                  self.tensor_attrs[a] = qualname
424                  setattr(self.root, qualname, a)
425  
426              return self.create_node("get_attr", qualname, (), {})
427  
428          if type(a) in _proxyable_classes:
429              # This is an instance of a proxyable class for which we did not
430              # witness its construction. Intern this as a constant attribute
431  
432              # TODO: binary search
433              qualname = self.get_fresh_qualname(f"_{a.__class__.__name__}_constant_")
434              assert isinstance(qualname, str)
435              setattr(self.root, qualname, a)
436  
437              return self.create_node("get_attr", qualname, (), {})
438  
439          return super().create_arg(a)

441      @compatibility(is_backward_compatible=True)
442      def is_leaf_module(self, m: torch.nn.Module, module_qualified_name: str) -> bool:
443          """ 
444          A method to specify whether a given ``nn.Module`` is a "leaf" module. 
445   
446          Leaf modules are the atomic units that appear in 
447          the IR, referenced by ``call_module`` calls. By default, 
448          Modules in the PyTorch standard library namespace (torch.nn) 
449          are leaf modules. All other modules are traced through and 
450          their constituent ops are recorded, unless specified otherwise 
451          via this parameter. 
452   
453          Args: 
454   
455              m (Module): The module being queried about 
456              module_qualified_name (str): The path to root of this module. For example, 
457                  if you have a module hierarchy where submodule ``foo`` contains 
458                  submodule ``bar``, which contains submodule ``baz``, that module will 
459                  appear with the qualified name ``foo.bar.baz`` here. 
460          """
461          return (
462              m.__module__.startswith("torch.nn")
463              or m.__module__.startswith("torch.ao.nn")
464          ) and not isinstance(m, torch.nn.Sequential)
465  
466      @compatibility(is_backward_compatible=True)
467      def path_of_module(self, mod: torch.nn.Module) -> str:
468          """ 
469          Helper method to find the qualified name of ``mod`` in the Module hierarchy 
470          of ``root``. For example, if ``root`` has a submodule named ``foo``, which has 
471          a submodule named ``bar``, passing ``bar`` into this function will return 
472          the string "foo.bar". 
473   
474          Args: 
475   
476              mod (str): The ``Module`` to retrieve the qualified name for. 
477          """
478          # Prefer the O(1) algorithm
479          if self.submodule_paths:
480              path = self.submodule_paths.get(mod)
481              if path is None:
482                  raise NameError("module is not installed as a submodule")
483              assert isinstance(path, str)
484              return path
485          # O(N^2) fallback in the case that we didn't store the submodule
486          # paths.
487          else:
488              for n, p in self.root.named_modules():
489                  if mod is p:
490                      return n
491              raise NameError("module is not installed as a submodule")

493      @compatibility(is_backward_compatible=True)
494      def call_module(
495          self,
496          m: torch.nn.Module,
497          forward: Callable[..., Any],
498          args: tuple[Any, ...],
499          kwargs: dict[str, Any],
500      ) -> Any:
501          """ 
502          Method that specifies the behavior of this ``Tracer`` when it encounters 
503          a call to an ``nn.Module`` instance. 
504   
505          By default, the behavior is to check if the called module is a leaf module 
506          via ``is_leaf_module``. If it is, emit a ``call_module`` node referring to 
507          ``m`` in the ``Graph``. Otherwise, call the ``Module`` normally, tracing through 
508          the operations in its ``forward`` function. 
509   
510          This method can be overridden to--for example--create nested traced 
511          GraphModules, or any other behavior you would want while tracing across 
512          ``Module`` boundaries. 
513   
514          Args: 
515   
516              m (Module): The module for which a call is being emitted 
517              forward (Callable): The forward() method of the ``Module`` to be invoked 
518              args (Tuple): args of the module callsite 
519              kwargs (Dict): kwargs of the module callsite 
520   
521          Return: 
522   
523              The return value from the Module call. In the case that a ``call_module`` 
524              node was emitted, this is a ``Proxy`` value. Otherwise, it is whatever 
525              value was returned from the ``Module`` invocation. 
526          """
527          module_qualified_name = self.path_of_module(m)
528          with ScopeContextManager(
529              self.scope, Scope(module_qualified_name, type(m))
530          ) as _scope:
531              # module_stack is an ordered dict so writing then deleting the
532              # entry is equivalent to push/pop on a list
533              num_calls = self.num_calls.get(module_qualified_name, 0)
534              module_key = (
535                  f"{_scope.module_path}@{num_calls}"
536                  if num_calls > 0
537                  else _scope.module_path
538              )
539              self.module_stack[module_key] = (module_qualified_name, _scope.module_type)
540              self.num_calls[module_qualified_name] = num_calls + 1
541              if not self.is_leaf_module(m, module_qualified_name):
542                  ret_val = forward(*args, **kwargs)
543              else:
544                  ret_val = self.create_proxy(
545                      "call_module", module_qualified_name, args, kwargs
546                  )
547              key, _ = self.module_stack.popitem(last=True)
548              assert key == module_key, f" Unexpected key {key}"
549  
550          return ret_val

552      @compatibility(is_backward_compatible=False)
553      def getattr(self, attr: str, attr_val: Any, parameter_proxy_cache: dict[str, Any]):
554          """ 
555          Method that specifies the behavior of this ``Tracer`` when we call getattr 
556          on a call to an ``nn.Module`` instance. 
557   
558          By default, the behavior is to return a proxy value for the attribute. It 
559          also stores the proxy value in the ``parameter_proxy_cache``, so that future 
560          calls will reuse the proxy rather than creating a new one. 
561   
562          This method can be overridden to --for example-- not return proxies when 
563          querying parameters. 
564   
565          Args: 
566   
567              attr (str): The name of the attribute being queried 
568              attr_val (Any): The value of the attribute 
569              parameter_proxy_cache (Dict[str, Any]): A cache of attr names to proxies 
570   
571          Return: 
572   
573              The return value from the getattr call. 
574          """
575  
576          def maybe_get_proxy_for_attr(
577              attr_val, collection_to_search, parameter_proxy_cache
578          ):
579              for n, p in collection_to_search:
580                  if attr_val is p:
581                      if n not in parameter_proxy_cache:
582                          kwargs = {}
583                          if (
584                              "proxy_factory_fn"
585                              in inspect.signature(self.create_proxy).parameters
586                          ):
587                              kwargs["proxy_factory_fn"] = (
588                                  None
589                                  if not self.param_shapes_constant
590                                  else lambda node: ParameterProxy(
591                                      self, node, n, attr_val
592                                  )
593                              )
594                          val_proxy = self.create_proxy("get_attr", n, (), {}, **kwargs)  # type: ignore[arg-type]
595                          parameter_proxy_cache[n] = val_proxy
596                      return parameter_proxy_cache[n]
597              return None
598  
599          if isinstance(attr_val, torch.nn.Parameter):
600              maybe_parameter_proxy = maybe_get_proxy_for_attr(
601                  attr_val, self.root.named_parameters(), parameter_proxy_cache
602              )
603              if maybe_parameter_proxy is not None:
604                  return maybe_parameter_proxy
605  
606          if self.proxy_buffer_attributes and isinstance(attr_val, torch.Tensor):
607              maybe_buffer_proxy = maybe_get_proxy_for_attr(
608                  attr_val, self.root.named_buffers(), parameter_proxy_cache
609              )
610              if maybe_buffer_proxy is not None:
611                  return maybe_buffer_proxy
612  
613          return attr_val

615      # This method will be refactored
616      @compatibility(is_backward_compatible=False)
617      def create_args_for_root(self, root_fn, is_module, concrete_args=None):
618          """ 
619          Create ``placeholder`` nodes corresponding to the signature of the ``root`` 
620          Module. This method introspects root's signature and emits those 
621          nodes accordingly, also supporting ``*args`` and ``**kwargs``. 
622          """
623          # In some cases, a function or method has been decorated with a wrapper
624          # defined via ``functools.wraps``. In this case, the outer code object
625          # will likely not contain the actual parameters we care about, so unwrap
626          # the function to get to the innermost callable.
627          fn_for_analysis = inspect.unwrap(root_fn)
628          co = fn_for_analysis.__code__
629          total_args = co.co_argcount + co.co_kwonlyargcount
630          orig_args = list(co.co_varnames)
631          names_iter = iter(co.co_varnames)
632          args: list[Any] = []
633          skip_arg_idx = 0
634          if is_module:
635              if total_args == 0:
636                  raise RuntimeError(
637                      "``self`` argument cannot be part of *args expansion!"
638                  )
639              skip_arg_idx = 1
640              next(names_iter)  # skip self
641              args.append(self.root)
642  
643          sig = inspect.signature(fn_for_analysis)
644  
645          # This covers the very specific case where we are passing in flat
646          # concrete_args as a tuple, but our traced fn takes (*args, **kwargs).
647          # In this case, just take the concrete_args and pass them through.
648          name_idx = 0
649          if (
650              isinstance(concrete_args, tuple)
651              and len(concrete_args) > 0
652              and (co.co_flags & HAS_VARSTUFF)
653              and total_args == 1
654          ):
655              for concrete_arg in concrete_args:
656                  out = self.create_proxy("placeholder", f"input_{name_idx}", (), {})
657                  if isinstance(concrete_arg, PHBase):
658                      if concrete_arg != PH:
659                          # Transfer attrs in the case where you're using a placeholder other
660                          # than the singleton PH (PH has no attributes to transfer).
661                          # Proxies were created out of the placeholders.
662                          # Transfer any metadata (put on the placeholders in the form of
663                          # attributes set by the user) from the placeholder to the
664                          # underlying nodes (the proxy is unwrapped by the user, but
665                          # the metadata should hold).
666                          _transfer_attrs(fr=concrete_arg, to=out.node)
667                  args.append(out)
668                  name_idx += 1
669              return root_fn, args
670  
671          arg_names = [next(names_iter) for idx in range(skip_arg_idx, total_args)]
672          if isinstance(concrete_args, tuple):
673              if len(arg_names) != len(concrete_args):
674                  raise RuntimeError(
675                      f"Tracing expected {len(arg_names)} arguments but got {len(concrete_args)} concrete arguments"
676                  )
677              concrete_args = dict(zip(arg_names, concrete_args))
678  
679          def proxy_placeholder(name):
680              return self._proxy_placeholder(name, concrete_args, sig, fn_for_analysis)
681  
682          args.extend(proxy_placeholder(names) for names in arg_names)
683  
684          if co.co_kwonlyargcount > 0 or co.co_flags & HAS_VARSTUFF:
685              # TODO: type annotations for *args and **kwargs
686              if co.co_flags & inspect.CO_VARARGS:
687                  args.append(proxy_placeholder("*" + next(names_iter)))
688              if co.co_flags & inspect.CO_VARKEYWORDS:
689                  args.append(proxy_placeholder("**" + next(names_iter)))
690              root_fn = _patch_function(root_fn, len(args))
691  
692          flat_args, in_spec = pytree.tree_flatten(tuple(args))
693          if not all(child.is_leaf() for child in in_spec.children_specs):
694              # In the case that we have pytree-flattened inputs in
695              # `concrete_args`, generate a flattening wrapper around the
696              # original root function and return that.
697              self.graph._codegen = _PyTreeCodeGen(
698                  _PyTreeInfo(orig_args[:total_args], in_spec, None)
699              )
700  
701              def flatten_fn(*args):
702                  tree_args = pytree.tree_unflatten(list(args), in_spec)
703                  tree_out = root_fn(*tree_args)
704                  out_args, out_spec = pytree.tree_flatten(tree_out)
705                  assert isinstance(self.graph._codegen, _PyTreeCodeGen)
706                  self.graph._codegen.pytree_info = (
707                      self.graph._codegen.pytree_info._replace(out_spec=out_spec)
708                  )
709                  return out_args
710  
711              return flatten_fn, flat_args
712          return root_fn, args
713  
714      @compatibility(is_backward_compatible=True)
715      def trace(
716          self,
717          root: Union[torch.nn.Module, Callable[..., Any]],
718          concrete_args: Optional[dict[str, Any]] = None,
719      ) -> Graph:
720          """ 
721          Trace ``root`` and return the corresponding FX ``Graph`` representation. ``root`` 
722          can either be an ``nn.Module`` instance or a Python callable. 
723   
724          Note that after this call, ``self.root`` may be different from the ``root`` passed 
725          in here. For example, when a free function is passed to ``trace()``, we will 
726          create an ``nn.Module`` instance to use as the root and add embedded constants 
727          to. 
728   
729   
730          Args: 
731   
732              root (Union[Module, Callable]): Either a ``Module`` or a function to be 
733                  traced through. Backwards-compatibility for this parameter is 
734                  guaranteed. 
735              concrete_args (Optional[Dict[str, any]]): Concrete arguments that should 
736                  not be treated as Proxies. This parameter is experimental and 
737                  its backwards-compatibility is *NOT* guaranteed. 
738   
739          Returns: 
740   
741              A ``Graph`` representing the semantics of the passed-in ``root``. 
742          """
743          global _is_fx_tracing_flag
744          old_is_fx_tracing_flag = _is_fx_tracing_flag
745          _is_fx_tracing_flag = True
746          try:
747              if isinstance(root, torch.nn.Module):
748                  # do real recompilation for _LazyGraphModule before retracing since the trace
749                  # method can not trace the _lazy_forward method. Got error:
750                  #   https://gist.github.com/shunting314/75549c2e82ae07ac1139c94a3583d259
751                  # without this.
752                  from torch.fx._lazy_graph_module import _LazyGraphModule
753  
754                  _LazyGraphModule.force_recompile(root)
755  
756                  self.root = root
757  
758                  assert hasattr(type(root), self.traced_func_name), (
759                      f"traced_func_name={self.traced_func_name} doesn't exist in {type(root).__name__}"
760                  )
761  
762                  fn = getattr(type(root), self.traced_func_name)
763                  self.root_module_name = root._get_name()
764                  self.submodule_paths = {mod: name for name, mod in root.named_modules()}
765              else:
766                  self.root = torch.nn.Module()
767                  fn = root
768  
769              tracer_cls: Optional[type[Tracer]] = getattr(self, "__class__", None)
770              self.graph = Graph(tracer_cls=tracer_cls)
771              if hasattr(fn, "__code__"):
772                  code = fn.__code__
773                  self.graph._co_fields = {
774                      "co_name": code.co_name,
775                      "co_filename": code.co_filename,
776                      "co_firstlineno": code.co_firstlineno,
777                  }
778  
779              # When we encounter a Tensor value that's not a parameter, we look if it
780              # is some other attribute on the model. Construct a dict mapping Tensor
781              # values to the qualified name here for efficiency. This is used downstream
782              # in create_arg
783              self.tensor_attrs: dict[
784                  _ConstantAttributeType,
785                  str,
786              ] = {}
787  
788              def collect_tensor_attrs(m: torch.nn.Module, prefix_atoms: list[str]):
789                  for k, v in m.__dict__.items():
790                      if isinstance(v, _constant_attribute_types):
791                          self.tensor_attrs[v] = ".".join(prefix_atoms + [k])
792                  for k, v in m.named_children():
793                      collect_tensor_attrs(v, prefix_atoms + [k])
794  
795              collect_tensor_attrs(self.root, [])
796  
797              assert isinstance(fn, FunctionType)
798  
799              fn_globals = fn.__globals__  # run before it gets patched
800              fn, args = self.create_args_for_root(
801                  fn, isinstance(root, torch.nn.Module), concrete_args
802              )
803  
804              parameter_proxy_cache: dict[
805                  str, Proxy
806              ] = {}  # Reduce number of get_attr calls
807  
808              # Method dispatch on parameters is not recorded unless it's directly used.
809              # Thus, we need to insert a proxy when __getattr__ requests a parameter.
810              @functools.wraps(_orig_module_getattr)
811              def module_getattr_wrapper(mod, attr):
812                  attr_val = _orig_module_getattr(mod, attr)
813                  return self.getattr(attr, attr_val, parameter_proxy_cache)
814  
815              @functools.wraps(_orig_module_call)
816              def module_call_wrapper(mod, *args, **kwargs):
817                  def forward(*args, **kwargs):
818                      return _orig_module_call(mod, *args, **kwargs)
819  
820                  _autowrap_check(
821                      patcher,  # type: ignore[has-type]
822                      getattr(getattr(mod, "forward", mod), "__globals__", {}),
823                      self._autowrap_function_ids,
824                  )
825                  return self.call_module(mod, forward, args, kwargs)
826  
827              with _new_patcher() as patcher:
828                  # allow duplicate patches to support the case of nested calls
829                  patcher.patch_method(
830                      torch.nn.Module,
831                      "__getattr__",
832                      module_getattr_wrapper,
833                      deduplicate=False,
834                  )
835                  patcher.patch_method(
836                      torch.nn.Module,
837                      "__call__",
838                      module_call_wrapper,
839                      deduplicate=False,
840                  )
841                  _patch_wrapped_functions(patcher)
842                  _autowrap_check(patcher, fn_globals, self._autowrap_function_ids)
843                  for module in self._autowrap_search:
844                      _autowrap_check(
845                          patcher, module.__dict__, self._autowrap_function_ids
846                      )
847                  self.create_node(
848                      "output",
849                      "output",
850                      (self.create_arg(fn(*args)),),
851                      {},
852                      type_expr=fn.__annotations__.get("return", None),
853                  )
854  
855              self.submodule_paths = None
856          except RuntimeError as e:
857              if isinstance(e.args[0], str) and "data-dependent" in e.args[0]:
858                  partial_fx_graph = self.graph.python_code(
859                      root_module="self",
860                      verbose=True,
861                  ).src
862                  e.partial_fx_graph = partial_fx_graph  # type: ignore[attr-defined]
863                  raise
864  
865              raise
866          finally:
867              _is_fx_tracing_flag = old_is_fx_tracing_flag
868          return self.graph

870      def __deepcopy__(self, memo):
871          # _autowrap_search contains modules, which cannot be deepcopied.
872          new_tracer = Tracer.__new__(Tracer)
873  
874          for k, v in self.__dict__.items():
875              if k in {"_autowrap_search"}:
876                  new_obj = copy.copy(v)
877              else:
878                  new_obj = copy.deepcopy(v, memo)
879  
880              new_tracer.__dict__[k] = new_obj
881  
882          return new_tracer

884      def _proxy_placeholder(self, name, concrete_args, sig, fn_for_analysis):
885          if concrete_args is not None and name in concrete_args:
886              cnt = 0
887  
888              def replace_ph(x):
889                  nonlocal cnt
890                  cnt += 1
891                  param = sig.parameters[name]
892                  default: tuple[Any, ...] = (
893                      () if param.default is inspect.Parameter.empty else (param.default,)
894                  )
895                  out = self.create_proxy(
896                      "placeholder", f"{name}_{str(cnt)}", default, {}
897                  )
898                  if isinstance(x, PHBase):
899                      if x != PH:
900                          # Transfer attrs in the case where you're using a placeholder other
901                          # than the singleton PH (PH has no attributes to transfer).
902                          # Proxies were created out of the placeholders.
903                          # Transfer any metadata (put on the placeholders in the form of
904                          # attributes set by the user) from the placeholder to the
905                          # underlying nodes (the proxy is unwrapped by the user, but
906                          # the metadata should hold).
907                          _transfer_attrs(fr=x, to=out.node)
908  
909                      return out
910                  # Union[int, bool] == bool in Python <= 3.6
911                  if type(x) == bool or type(x) in base_types and type(x) != torch.Tensor:
912                      torch._assert(
913                          out == x,
914                          f"{name} has been specialized to have value {x} but got another value",
915                      )
916                  elif x is None:
917                      args = (
918                          out,
919                          f"{name} has been specialized to have value None but got another value",
920                      )
921                      self.create_proxy("call_function", _assert_is_none, args, {})
922                  else:
923                      warnings.warn(
924                          f"Was not able to add assertion to guarantee correct input {name} to "
925                          f"specialized function. It is up to the user to make sure that your inputs match the "
926                          f"inputs you specialized the function with."
927                      )
928  
929                  return x
930  
931              return pytree.tree_map(replace_ph, concrete_args[name])
932          if name[0] == "*":
933              default: tuple[Any, ...] = ()
934          else:
935              param = sig.parameters[name]
936              default = (  # type: ignore[assignment]
937                  () if param.default is inspect.Parameter.empty else (param.default,)
938              )
939          return self.create_proxy(
940              "placeholder",
941              name,
942              default,
943              {},
944              type_expr=fn_for_analysis.__annotations__.get(name, None),
945          )
946  
947  
948  # Dictionary of (id(globals dict), function name) => globals_dict to patch for
949  # the purposes of the wrap() API.
950  # We key by the globals dict id and function name to ensure we're wrapping a given
951  # function only once.
952  _wrapped_fns_to_patch: dict[tuple[int, str], dict] = {}
953  
954  # List of methods on classes to wrap (class type, function name)
955  # this currently only works for Tensor.* methods that aren't traced properly
956  _wrapped_methods_to_patch: list[tuple[type, str]] = []
957  
958  if os.environ.get("FX_PATCH_GETITEM") == "1":
959      # This change is needed to trace models like PositionalEmbedding from BERT:
960      # https://github.com/pytorch/benchmark/blob/master/torchbenchmark/models/BERT_pytorch/bert_pytorch/model/embedding/position.py
961      # but causes issues in quantization documented here:
962      # https://github.com/pytorch/pytorch/issues/50710
963      # once that is fixed we can make this the default behavior.
964      _wrapped_methods_to_patch.append((torch.Tensor, "__getitem__"))

968      """ 
969      Recursively search a data structure for a Proxy() and return it, 
970      return None if not found. 
971      """
972      proxy = None
973  
974      def find_proxy(x):
975          nonlocal proxy
976          if isinstance(x, Proxy):
977              proxy = x

983  def _create_wrapped_func(orig_fn):
984      @functools.wraps(orig_fn)
985      def wrapped(*args, **kwargs):
986          """ 
987          Given an closed-over ``orig_function`` to invoke, search the args and kwargs for 
988          a Proxy object. If there is one, emit a ``call_function`` node to preserve the 
989          call to this leaf function directly. Otherwise, just return the results of 
990          this function call, as this function is not being traced. 
991          """
992          proxy = _find_proxy(args, kwargs)
993          if proxy is not None:
994              return_proxy = proxy.tracer.create_proxy(
995                  "call_function", orig_fn, args, kwargs
996              )
997              return_proxy.node.meta["is_wrapped"] = True
998              return return_proxy
999          return orig_fn(*args, **kwargs)
1000 
1001     return wrapped

1004 def _create_wrapped_method(cls, name):
1005     orig_fn = getattr(cls, name)
1006 
1007     @functools.wraps(orig_fn)
1008     def wrapped(*args, **kwargs):
1009         """ 
1010         Search the args and kwargs for a Proxy object. If there is one, 
1011         emit a ``call_method`` node to preserve the call to this method 
1012         directly. Otherwise, just return the results of this function 
1013         call, as this function is not being traced. 
1014         """
1015         proxy = _find_proxy(args, kwargs)
1016         if proxy is not None:
1017             return proxy.tracer.create_proxy("call_method", name, args, kwargs)
1018         return orig_fn(*args, **kwargs)
1019

1060 class _Patcher:
1061     def __init__(self) -> None:
1062         super().__init__()
1063         self.patches_made: list[_PatchedFn] = []
1064         self.visited: set[int] = set()
1065 
1066     def patch(
1067         self,
1068         frame_dict: dict[str, Any],
1069         name: str,
1070         new_fn: Callable,
1071         deduplicate: bool = True,
1072     ):
1073         """ 
1074         Replace frame_dict[name] with new_fn until we exit the context manager. 
1075         """
1076         new_fn.__fx_already_patched = deduplicate  # type: ignore[attr-defined]
1077         if name not in frame_dict and hasattr(builtins, name):
1078             self.patches_made.append(_PatchedFnDel(frame_dict, name, None, new_fn))
1079             self.patches_made[-1].patch()
1080         elif getattr(frame_dict[name], "__fx_already_patched", False):
1081             return  # already patched, no need to do it again
1082         else:
1083             self.patches_made.append(
1084                 _PatchedFnSetItem(frame_dict, name, frame_dict[name], new_fn)
1085             )
1086             self.patches_made[-1].patch()
1087 
1088     def patch_method(
1089         self, cls: type, name: str, new_fn: Callable, deduplicate: bool = True
1090     ):
1091         """ 
1092         Replace object_or_dict.name with new_fn until we exit the context manager. 
1093         """
1094         new_fn.__fx_already_patched = deduplicate  # type: ignore[attr-defined]
1095         orig_fn = getattr(cls, name)
1096         if getattr(orig_fn, "__fx_already_patched", False):
1097             return  # already patched, no need to do it again
1098         self.patches_made.append(_PatchedFnSetAttr(cls, name, orig_fn, new_fn))
1099         self.patches_made[-1].patch()

1101     def visit_once(self, thing: Any):
1102         """Return True on the first call to with thing, otherwise false"""
1103         idx = id(thing)
1104         if idx in self.visited:
1105             return False
1106         self.visited.add(idx)
1107         return True
1108 
1109     def revert_all_patches(self):
1110         """ 
1111         Remove all the stored patcheds. It doesn't modify patches_made. 
1112         """
1113         for patch in self.patches_made:
1114             patch.revert()
1115         return self.patches_made
1116 
1117     def reapply_all_patches(self):
1118         """ 
1119         Patch all the stored patcheds. It doesn't modify patches_made. 
1120         """
1121         for patch in self.patches_made:
1122             patch.patch()
1123         return self.patches_made
1124 
1125     def __enter__(self):
1126         return self
1127 
1128     def __exit__(self, exc_type, exc_val, exc_tb):
1129         """ 
1130         Undo all the changes made via self.patch() and self.patch_method() 
1131         """
1132         while self.patches_made:
1133             # unpatch in reverse order to handle duplicates correctly
1134             self.patches_made.pop().revert()
1135         self.visited.clear()

1141 @contextlib.contextmanager

1143     global CURRENT_PATCHER

1172 def _patch_wrapped_functions(patcher: _Patcher):
1173     """ 
1174     Go through ``_wrapped_fn_patch_table`` and, for each frame object, wrap 
1175     the listed global functions in the `_create_wrapped_func` wrapper. 
1176     """
1177     for (_, name), frame_dict in _wrapped_fns_to_patch.copy().items():
1178         if name not in frame_dict and hasattr(builtins, name):
1179             orig_fn = getattr(builtins, name)
1180         else:
1181             orig_fn = frame_dict[name]

1188 def _autowrap_check(
1189     patcher: _Patcher, frame_dict: dict[str, Any], function_ids: set[int]
1190 ):
1191     """ 
1192     Some methods, like `math.sqrt` are common enough we want to automatically wrap them as we see them. 
1193     This method searches a scope for them and patches them if found. 
1194     """
1195     if patcher.visit_once(frame_dict):
1196         for name, value in frame_dict.items():
1197             if (
1198                 not name.startswith("_")
1199                 and callable(value)
1200                 and id(value) in function_ids
1201             ):
1202                 patcher.patch(frame_dict, name, _create_wrapped_func(value))

1205 @compatibility(is_backward_compatible=True)
1206 def wrap(fn_or_name: Union[str, Callable]):
1207     """ 
1208     This function can be called at module-level scope to register fn_or_name as a "leaf function". 
1209     A "leaf function" will be preserved as a CallFunction node in the FX trace instead of being 
1210     traced through:: 
1211  
1212         # foo/bar/baz.py 
1213         def my_custom_function(x, y): 
1214             return x * x + y * y 
1215  
1216  
1217         torch.fx.wrap("my_custom_function") 
1218  
1219  
1220         def fn_to_be_traced(x, y): 
1221             # When symbolic tracing, the below call to my_custom_function will be inserted into 
1222             # the graph rather than tracing it. 
1223             return my_custom_function(x, y) 
1224  
1225     This function can also equivalently be used as a decorator:: 
1226  
1227         # foo/bar/baz.py 
1228         @torch.fx.wrap 
1229         def my_custom_function(x, y): 
1230             return x * x + y * y 
1231  
1232     A wrapped function can be thought of a "leaf function", analogous to the concept of 
1233     "leaf modules", that is, they are functions that are left as calls in the FX trace 
1234     rather than traced through. 
1235  
1236     Args: 
1237  
1238         fn_or_name (Union[str, Callable]): The function or name of the global function to insert into the 
1239             graph when it's called 
1240     """
1241     if not callable(fn_or_name) and not isinstance(fn_or_name, str):
1242         raise RuntimeError(
1243             "Unsupported type for global function! Must be either a callable or "
1244             "string name"
1245         )
1246 
1247     if callable(fn_or_name):
1248         assert not isinstance(fn_or_name, str)  # to make mypy happy
1249         fn_name = fn_or_name.__name__
1250     else:
1251         assert isinstance(fn_or_name, str), (
1252             "fn_or_name must be a global function or string name"
1253         )
1254         fn_name = fn_or_name
1255 
1256     currentframe = inspect.currentframe()
1257     assert currentframe is not None
1258     f = currentframe.f_back
1259     assert f is not None
1260     if f.f_code.co_name != "<module>":
1261         raise NotImplementedError("wrap must be called at the top level of a module")

1269 @compatibility(is_backward_compatible=True)
1270 def symbolic_trace(
1271     root: Union[torch.nn.Module, Callable[..., Any]],
1272     concrete_args: Optional[dict[str, Any]] = None,
1273 ) -> GraphModule:
1274     """ 
1275     Symbolic tracing API 
1276  
1277     Given an ``nn.Module`` or function instance ``root``, this function will return a ``GraphModule`` 
1278     constructed by recording operations seen while tracing through ``root``. 
1279  
1280     ``concrete_args`` allows you to partially specialize your function, whether it's to remove control flow or data structures. 
1281  
1282     For example:: 
1283  
1284         def f(a, b): 
1285             if b == True: 
1286                 return a 
1287             else: 
1288                 return a * 2 
1289  
1290     FX can typically not trace through this due to the presence of control 
1291     flow. However, we can use `concrete_args` to specialize on the value of 
1292     `b` to trace through this:: 
1293  
1294         f = fx.symbolic_trace(f, concrete_args={"b": False}) 
1295         assert f(3, False) == 6 
1296  
1297     Note that although you can still pass in different values of `b`, they will be ignored. 
1298  
1299     We can also use `concrete_args` to eliminate data-structure handling from 
1300     our function. This will use pytrees to flatten your input. To avoid 
1301     overspecializing, pass in `fx.PH` for values that shouldn't be 
1302     specialized. For example:: 
1303  
1304         def f(x): 
1305             out = 0 
1306             for v in x.values(): 
1307                 out += v 
1308             return out 
1309  
1310  
1311         f = fx.symbolic_trace( 
1312             f, concrete_args={"x": {"a": fx.PH, "b": fx.PH, "c": fx.PH}} 
1313         ) 
1314         assert f({"a": 1, "b": 2, "c": 4}) == 7 
1315  
1316  
1317     Args: 
1318         root (Union[torch.nn.Module, Callable]): Module or function to be traced and converted 
1319             into a Graph representation. 
1320         concrete_args (Optional[Dict[str, any]]): Inputs to be partially specialized 
1321  
1322     Returns: 
1323         GraphModule: a Module created from the recorded operations from ``root``. 
1324     """
1325     tracer = Tracer()
1326     graph = tracer.trace(root, concrete_args)
1327     name = (
1328         root.__class__.__name__ if isinstance(root, torch.nn.Module) else root.__name__
1329     )
1330     return _make_graph_module(tracer.root, graph, name)
