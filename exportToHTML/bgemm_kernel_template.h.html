24   template <ck::index_t... Is>
25   using S = ck::Sequence<Is...>;
26   
27   using BF16 = ck::bhalf_t;
28   using F32  = float;
29   
30   using AccDataType      = F32;
31   using DsDataType       = ck::Tuple<>;
32   using CDataType        = BF16;
33   using CShuffleDataType = BF16;
34   using DsLayout         = ck::Tuple<>;
35   using CLayout          = Row;
36   
37   using PassThrough = ck::tensor_operation::element_wise::PassThrough;
38   using AElementOp   = PassThrough;
39   using BElementOp   = PassThrough;
40   using CDEElementOp = PassThrough;
41   
42   using Row = ck::tensor_layout::gemm::RowMajor;
43   using Col = ck::tensor_layout::gemm::ColumnMajor;
44   
45   template <
46       typename A_DATA_TYPE,
47       typename B_DATA_TYPE,
48       int BLOCK_SIZE,
49       int MBLOCK,
50       int NBLOCK,
51       int KBLOCK,
52       int AK1,
53       int BK1,
54       int WAVE_TILE_M,
55       int WAVE_TILE_N,
56       int WAVE_MAP_M,
57       int WAVE_MAP_N,
58       typename ABLOCK_TRANSFER,
59       int ABLOCK_TRANSFER_SSPV,
60       int ABLOCK_TRANSFER_DSPV_K1,
61       typename BBLOCK_TRANSFER,
62       int BBLOCK_TRANSFER_SSPV,
63       int BBLOCK_TRANSFER_SSPV_K1,
64       int CSHUFFLE_MXDL_PWPS,
65       int CSHUFFLE_NXDL_PWPS,
66       typename CSHUFFLEBLOCK_TRANSFER,
67       typename CDESHUFFLEBLOCK_TRANSFER,
68       ck::BlockGemmPipelineScheduler LOOP_SCHED,
69       ck::BlockGemmPipelineVersion PIPELINE_VERSION,
70       ck::tensor_operation::device::GemmSpecialization GEMM_SPEC =
71           ck::tensor_operation::device::GemmSpecialization::MNPadding,
72       bool TRANSA = false,
73       bool TRANSB = false>
74   void bgemm_kernel_impl(CUDABLAS_BGEMM_ARGTYPES(at::BFloat16)) {
75   
76     using ADataType = typename CkMathType<A_DATA_TYPE>::dtype;
77     using BDataType = typename CkMathType<B_DATA_TYPE>::dtype;
78   
79     using ALayout = typename CkTensorLayout<TRANSA, TRANSB>::a_layout;
80     using BLayout = typename CkTensorLayout<TRANSA, TRANSB>::b_layout;
81   
82     auto a_element_op = AElementOp{};
83     auto b_element_op = BElementOp{};
84     auto cde_element_op = CDEElementOp{};
85   
86     auto gemm = ck::tensor_operation::device::DeviceBatchedGemmMultiD_Xdl_CShuffle_V3<
87         ALayout,                  // ALayout
88         BLayout,                  // BLayout
89         DsLayout,                 // DsLayout
90         CLayout,                  // CLayout
91         ADataType,                // ADataType
92         BDataType,                // BDataType
93         DsDataType,               // DsDataType
94         CDataType,                // CDataType
95         AccDataType,              // AccDataType
96         CShuffleDataType,         // CshuffleType
97         AElementOp,               // AElementwiseOperation
98         BElementOp,               // BElementwiseOperation
99         CDEElementOp,             // CElementwiseOperation
100        GEMM_SPEC,                // GEMMSpecialization
101        BLOCK_SIZE,               // BlockSize
102        MBLOCK,                   // MPerBlock
103        NBLOCK,                   // NPerBlock
104        KBLOCK,                   // KPerBlock
105        AK1,                      // AK1
106        BK1,                      // BK1
107        WAVE_TILE_M,              // MPerXDL
108        WAVE_TILE_N,              // NPerXDL
109        WAVE_MAP_M,               // MXdlPerWave
110        WAVE_MAP_N,               // NXdlPerWave
111        ABLOCK_TRANSFER,          // ABlockTransferThreadClusterLengths_AK0_M_AK1
112        S<1, 0, 2>,               // ABlockTransferThreadClusterArrangeOrder
113        S<1, 0, 2>,               // ABlockTransferSrcAccessOrder
114        2,                        // ABlockTransferSrcVectorDim
115        ABLOCK_TRANSFER_SSPV,     // ABlockTransferSrcScalarPerVector
116        ABLOCK_TRANSFER_DSPV_K1,  // ABlockTransferDstScalarPerVector_AK1
117        0,                        // ABlockLdsExtraM
118        BBLOCK_TRANSFER,          // BBlockTransferThreadClusterLengths_BK0_N_BK1
119        S<1, 0, 2>,               // BBlockTransferThreadClusterArrangeOrder
120        S<1, 0, 2>,               // BBlockTransferSrcAccessOrder
121        2,                        // BBlockTransferSrcVectorDim
122        BBLOCK_TRANSFER_SSPV,     // BBlockTransferSrcScalarPerVector
123        BBLOCK_TRANSFER_SSPV_K1,  // BBlockTransferDstScalarPerVector_BK1
124        0,                        // BBlockLdsAddExtraN
125        CSHUFFLE_MXDL_PWPS,       // CShuffleMXdlPerWavePerShuffle
126        CSHUFFLE_NXDL_PWPS,       // CShuffleNXdlPerWavePerShuffle
127        CSHUFFLEBLOCK_TRANSFER,   // CShuffleBlockTransferClusterLengths_MBlock_MPerBlock_NBlock_NPerBlock
128        CDESHUFFLEBLOCK_TRANSFER, // CDEShuffleBlockTransferScalarPerVectors
129        LOOP_SCHED,               // BlockGemmPipelineScheduler
130        PIPELINE_VERSION          // BlockGemmPipelineVersion
131        >{};
132    auto invoker = gemm.MakeInvoker();
133    auto argument = gemm.MakeArgument(
134      b, // A and B are swapped for CK
135      a,
136      {},
137      c,
138      n,
139      m,
140      k,
141      num_batches,
142      ldb,
143      lda,
144      {},
145      ldc,
146      n * k,  // batch_stride_a
147      m * k,  // batch_stride_b
148      {},
149      m * n,  // batch_stride_c
150      a_element_op,
151      b_element_op,
152      cde_element_op
153    );
154    if(!gemm.IsSupportedArgument(argument))
155    {
156        throw std::runtime_error(
157            "wrong! device_gemm with the specified compilation parameters does "
158            "not support this GEMM problem");
159    }
160    auto stream = at::cuda::getCurrentHIPStream().stream();
161    invoker.Run(argument, StreamConfig{stream, false});
162  }
