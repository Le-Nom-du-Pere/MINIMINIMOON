1    from sympy.sandbox.indexed_integrals import IndexedIntegral
2    from sympy.core.symbol import symbols
3    from sympy.functions.elementary.trigonometric import (cos, sin)
4    from sympy.tensor.indexed import (Idx, IndexedBase)
5    
6    
7    def test_indexed_integrals():
8        A = IndexedBase('A')
9        i, j = symbols('i j', integer=True)
10       a1, a2 = symbols('a1:3', cls=Idx)
11       assert isinstance(a1, Idx)
12   
13       assert IndexedIntegral(1, A[i]).doit() == A[i]
14       assert IndexedIntegral(A[i], A[i]).doit() == A[i] ** 2 / 2
15       assert IndexedIntegral(A[j], A[i]).doit() == A[i] * A[j]
16       assert IndexedIntegral(A[i] * A[j], A[i]).doit() == A[i] ** 2 * A[j] / 2
17       assert IndexedIntegral(sin(A[i]), A[i]).doit() == -cos(A[i])
18       assert IndexedIntegral(sin(A[j]), A[i]).doit() == sin(A[j]) * A[i]
19   
20       assert IndexedIntegral(1, A[a1]).doit() == A[a1]
21       assert IndexedIntegral(A[a1], A[a1]).doit() == A[a1] ** 2 / 2
22       assert IndexedIntegral(A[a2], A[a1]).doit() == A[a1] * A[a2]
23       assert IndexedIntegral(A[a1] * A[a2], A[a1]).doit() == A[a1] ** 2 * A[a2] / 2
24       assert IndexedIntegral(sin(A[a1]), A[a1]).doit() == -cos(A[a1])
25       assert IndexedIntegral(sin(A[a2]), A[a1]).doit() == sin(A[a2]) * A[a1]
