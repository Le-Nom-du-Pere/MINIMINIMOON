31   class _ModuleMeta:
32       """Meta information about a module. 
33    
34       This class is used to represent the module information in a more structured way. 
35       It parses raw module information from a single item from 
36       `node.meta["nn_module_stack"].items()`. 
37    
38       See the uses of `from_raw_meta`, `from_fx_tracer_produced_raw_meta`, and 
39       `from_dynamo_produced_raw_meta` for how to create an instance. 
40    
41       Attributes: 
42           _module_class: The class of the module. E.g. `torch.nn.module.sparse.Embedding`. 
43           _module_name: The name of the module. E.g. `L__self___h_1_mlp_c_proj`. 
44           _raw_meta: The raw meta '(module_name, node.meta["nn_module_stack"][module_name])'. 
45       """
46   
47       _module_class: Final[type | str | None]  # type: ignore[misc]
48       _module_name: Final[str]  # type: ignore[misc]
49       _raw_meta: Final[tuple[Any, Any]]  # type: ignore[misc]
50   
51       def __init__(
52           self,
53           module_name: str,
54           module_class: type | str | None,
55           raw_meta: tuple[Any, Any],
56       ):
57           self._module_name = module_name
58           self._module_class = module_class
59           self._raw_meta = raw_meta
60   
61       @property
62       def module_display_name(self) -> str:
63           """The display name of the module. 
64    
65           E.g. `h_1_mlp_c_proj`. 
66           """
67           # E.g., from 'L__self___h_1_mlp_c_proj' to 'h_1_mlp_c_proj'.
68           name = self.module_name
69           name = name.removeprefix("L__self___")
70           return name
71   
72       @property
73       def qualified_module_class_name(self) -> str:
74           """Qualified name of the module class. 
75    
76           E.g. `torch_nn_module_sparse_Embedding`. 
77           """
78           if self._module_class is None:
79               return ""
80           mod_cls = self._module_class
81           if isinstance(mod_cls, type):
82               mod_cls = mod_cls.__module__ + "." + mod_cls.__qualname__
83           return mod_cls.replace(".", "_")
84   
85       @property
86       def module_class_name(self) -> str:
87           """Name of the module class. 
88    
89           E.g. `Embedding`. 
90           """
91           if self._module_class is None:
92               return ""
93           if isinstance(self._module_class, type):
94               return self._module_class.__name__
95           return self._module_class
96   
97       @property
98       def module_name(self) -> str:
99           """Name of the module. 
100   
101          E.g. `L__self___h_1_mlp_c_proj`. 
102          """
103          return self._module_name
104  
105      @property
106      def raw_meta(self) -> tuple[Any, Any]:
107          """Returns the raw module meta data. 
108   
109          I.e. (module_name, node.meta['nn_module_stack'][module_name]). 
110          """
111          return self._raw_meta
112  
113      def __eq__(self, other: object, /) -> bool:
114          if not isinstance(other, _ModuleMeta):
115              return False
116          return (
117              self._module_name == other._module_name
118              and self._module_class == other._module_class
119          )
120  
121      def __hash__(self) -> int:
122          return hash((self._module_name, self._module_class))
123  
124      def __repr__(self) -> str:
125          return f"ModuleMeta(name={self._module_name}, class={self._module_class})"
126  
127      @classmethod
128      def create_root(cls) -> _ModuleMeta:
129          """Create an empty module meta representing root module."""
130          return _ModuleMeta("", None, ("", None))
131  
132      @classmethod
133      def from_fx_tracer_produced_raw_meta(
134          cls, raw_meta: _FX_TRACER_NN_MODULE_META_TYPE
135      ) -> _ModuleMeta:
136          """Create a module meta from raw meta produced by FX symbolic tracer."""
137          module_name, module_class = raw_meta
138          return _ModuleMeta(module_name, module_class, raw_meta)
139  
140      @classmethod
141      def from_dynamo_produced_raw_meta(
142          cls, raw_meta: _DYNAMO_NN_MODULE_META_TYPE
143      ) -> _ModuleMeta:
144          """Create a module meta from raw meta produced by FX dynamo tracer."""
145          module_name, (_qualified_name, module_class) = raw_meta
146          return _ModuleMeta(module_name.split("@")[0], module_class, raw_meta)
147  
148      @classmethod
149      def from_raw_meta(
150          cls,
151          raw_meta: _FX_TRACER_NN_MODULE_META_TYPE | _DYNAMO_NN_MODULE_META_TYPE,
152      ) -> _ModuleMeta:
153          if (
154              isinstance(raw_meta, tuple)
155              and len(raw_meta) == 2
156              and isinstance(raw_meta[1], type)
157          ):
158              # Trying to do `instance(raw_meta, _FX_TRACER_NN_MODULE_META_TYPE)`
159              return _ModuleMeta.from_fx_tracer_produced_raw_meta(raw_meta)
160          if (
161              isinstance(raw_meta, tuple)
162              and len(raw_meta) == 2
163              and isinstance(raw_meta[1], tuple)
164          ):
165              # Trying to do `instance(raw_meta, _DYNAMO_NN_MODULE_META_TYPE)`
166              return _ModuleMeta.from_dynamo_produced_raw_meta(raw_meta)
167          raise TypeError(
168              f"Unknown type of raw meta item from node.meta['nn_module_stack'].items(): {type(raw_meta)}"
169          )

172  class _ModuleStackMeta:
173      """Meta information about the module call stack. 
174   
175      This class is used to represent the module call stack information in a more 
176      structured way. It parses raw module stack information from `node.meta["nn_module_stack"]`. 
177   
178      Example of raw module stack information: 
179   
180          If produced by dynamo: 
181   
182              { 
183                  'L__self___h_1': ( 
184                      "L['self'].h[1]", 
185                      <class 'transformers.models.gpt2.modeling_gpt2.GPT2Block'> 
186                  ), 
187                  'L__self___h_1_attn': ( 
188                      "L['self'].h[1].attn", 
189                      <class 'transformers.models.gpt2.modeling_gpt2.GPT2Attention'> 
190                  ) 
191              } 
192   
193          If produced by fx.symbolic_trace: 
194   
195              { 
196                  'h.1': <class 'transformers.models.gpt2.modeling_gpt2.GPT2Block'>, 
197                  'h.1.attn': <class 'transformers.models.gpt2.modeling_gpt2.GPT2Attention'> 
198              } 
199      """
200  
201      _module_stack: Final[list[_ModuleMeta]]  # type: ignore[misc]
202  
203      def __init__(
204          self,
205          nn_module_stack_meta: _FX_TRACER_NN_MODULE_STACK_META_TYPE
206          | _DYNAMO_NN_MODULE_STACK_META_TYPE
207          | None,
208          is_exported_program: bool = True,
209      ):
210          self._module_stack = []
211          if nn_module_stack_meta is None:
212              return
213          raw_meta = copy.copy(nn_module_stack_meta)
214          for item in raw_meta.items():
215              # If produced by torch.export.export, there is another call stack layer
216              # that we need to skip
217              if is_exported_program:
218                  is_exported_program = False
219                  continue
220              self.push(_ModuleMeta.from_raw_meta(item))  # type: ignore[arg-type]
221  
222      def __len__(self) -> int:
223          return len(self._module_stack)
224  
225      def __getitem__(self, index: int) -> _ModuleMeta:
226          return self._module_stack[index]
227  
228      def __iter__(self) -> Iterator[_ModuleMeta]:
229          return iter(self._module_stack)
230  
231      def is_empty_or_root(self) -> bool:
232          return len(self._module_stack) == 0
233  
234      def top(self) -> _ModuleMeta:
235          """Returns the top module meta in the stack. I.e., the meta for leaf module. 
236   
237          Example: 
238   
239              Consider the following module stack: 
240   
241              stack = [GPT, block1, Attention_1, MLP] 
242   
243              stack.top() == MLP 
244          """
245          if self.is_empty_or_root():
246              return _ModuleMeta.create_root()
247          return self._module_stack[-1]
248  
249      def is_superset_of(
250          self,
251          module_stack: _ModuleStackMeta,
252      ) -> bool:
253          """Determines if self is a superset of the provided module stack. 
254   
255          I.e., If self includes all elements from the provided module stack, plus additional 
256          elements on top. If self is empty or root, this method always return False. 
257   
258          Example: 
259   
260              Consider the following module stack: 
261   
262              stack_1 = [GPT, block1, Attention_1, MLP] 
263              stack_2 = [GPT, block1] 
264   
265              stack_1.is_superset_of(stack_2) == True 
266              stack_2.is_superset_of(stack_1) == False 
267   
268              stack_3 = [GPT, block2, Attention_1] 
269   
270              stack_1.is_superset_of(stack_3) == False 
271              stack_3.is_superset_of(stack_1) == False 
272          """
273          if self.is_empty_or_root():
274              return False
275  
276          if module_stack.is_empty_or_root() is None:
277              return True
278  
279          if len(self) <= len(module_stack):
280              return False
281  
282          for i, parent_key in enumerate(module_stack):
283              if self[i] != parent_key:
284                  return False
285  
286          return True
287  
288      def push(self, module_meta: _ModuleMeta) -> None:
289          """Pushes a module meta to the stack."""
290          self._module_stack.append(module_meta)
291  
292      def __eq__(self, other: object, /) -> bool:
293          if not isinstance(other, _ModuleStackMeta):
294              return False
295          return self._module_stack == other._module_stack
296  
297      @property
298      def raw_meta(self) -> dict[str, tuple[str, type]] | None:
299          """Returns the raw module stack meta data, i.e. node.meta['nn_module_stack']."""
300          return {
301              module_meta.raw_meta[0]: module_meta.raw_meta[1]
302              for module_meta in self._module_stack
303          }
304  
305      def __repr__(self) -> str:
306          return f"ModuleStackMeta({self._module_stack})"
307  
308      @property
309      def module_display_name(self) -> str:
310          """Returns the module display name of the top module."""
311          return self.top().module_display_name
312  
313      @property
314      def qualified_module_class_name(self) -> str:
315          """Returns the qualified module class name of the top module."""
316          return self.top().qualified_module_class_name
317  
318      @property
319      def module_class(self) -> type | str | None:
320          """Returns the module class of the top module."""
321          return self.top()._module_class

338  class _IRNode(abc.ABC):
339      """Base class for IR nodes. 
340   
341      IR nodes are used for Modularize pass only. They add a layer of abstraction on top of 
342      torch.fx.Node. 
343   
344      [NOTE: Modularize Pass Implementation] 
345      The main job of the pass is to group `fx.Node`s that belong to the same `nn.Module` 
346      forward call, and then create `call_module` node and sub `fx.GraphModule` from them. 
347      Each `fx.Node` possesses an `nn_module_stack` meta data that contains information 
348      about the module call stack. See `_ModuleStackMeta` for examples. 
349   
350      Analysis step 
351      ------------- 
352   
353      Each module call is identified by a set of base stack layers. For each module call, 
354      the pass creates a `_ModuleNode` and groups the sequence of nodes that shares the 
355      same base stack layers. 
356   
357      For example, 
358   
359          stack_of_node_0 = [GPT, block0] 
360          stack_of_node_1 = [GPT, block1] 
361          stack_of_node_2 = [GPT, block1, Attention1, MLP] 
362          stack_of_node_3 = [GPT, block1, Attention1] 
363          stack_of_node_4 = [GPT, block2] 
364   
365      All nodes belong to the `GPT` module call, since they share the base stack layers [GPT]. 
366      [node_1, node_2, node_3] are grouped for `GPT.block1`, because they share the base 
367      stack layers [GPT, block1]. And [node_2, node_3] for `GPT.block1.Attention1`, [node_0] 
368      for `GPT.block0`, and [node_4] for `GPT.block2` respectfully. 
369   
370      After the analysis step, a hierarchical representation is generated. 
371   
372      For above example, the representation is: 
373   
374          _ModuleNode(GPT) 
375              _ModuleNode(block0) 
376                  _LeafNode(node_0) 
377              _ModuleNode(block1) 
378                  _LeafNode(node_1) 
379                  _ModuleNode(Attention1) 
380                      _ModuleNode(MLP) 
381                          _LeafNode(node_2) 
382                  _LeafNode(node_3) 
383              _ModuleNode(block2) 
384                  _LeafNode(node_4) 
385   
386      Construction step 
387      ----------------- 
388   
389      The second step is to build the actual `call_module` node and the sub `fx.GraphModule`. 
390      This is done recursively from the leaf `_ModuleNode` to the root. 
391   
392      For example, the first submodule to be built is `GPT.block1.Attention1.MLP`. Below pair 
393      is generated from `_ModuleNode(MLP)`. 
394   
395          fx.GraphModule(GPT.block1.Attention1.MLP) 
396              graph: 
397                  node_2 
398   
399          new_mlp_node = `call_module[GPT.block1.Attention1.MLP](...)` 
400   
401      Next, the `GPT.block1.Attention1` submodule is built. Below is generated from 
402      `_ModuleNode(Attention1)`. 
403   
404          fx.GraphModule(GPT.block1.Attention1) 
405              graph: 
406                  new_mlp_node 
407                  node_3 
408   
409          new_attention1_node = `call_module[GPT.block1.Attention1](...)` 
410   
411      Until every submodule is built, the new modularized `fx.GraphModule` is generated. 
412   
413      Alternatives 
414      ------------ 
415   
416      The current algorithm adopts a top down approach. A bottom up approach is similar. 
417      In contrast to these two, an alternative flat order approach is also possible, where 
418      each node is traversed and copied to the corresponding submodule. 
419   
420      The advantage of the current approach lies in the encapsulation of the fx.GraphModule 
421      construction for each individual submodule within a single `build_module` method, which 
422      can be called separately once the analysis phase is completed, making debugging more 
423      convenient. 
424   
425      Regarding construction step, an alternative implementation is to utilize `fx.Interpreter` 
426      for traversing all the nodes under the flattened root module and copying the nodes 
427      into their respective submodule under construction. This approach is not adopted because 
428   
429          1. It uses the flat order approach discussed above. This means one cannot individually 
430      construct a submodule and examine it while debugging. 
431   
432          2. The graph execution functionality of `fx.Interpreter` is not necessary for the 
433      purpose of this pass. Ignoring that, `fx.Interpreter.run` achieves the same effect 
434      as a for loop over all the nodes. 
435      """
436

450  class _ModuleNode(_IRNode):
451      """Representing a sequence of fx.Nodes to be formed into a fx.GraphModule. 
452   
453      This class encapsulates metadata and provides building block methods to construct this 
454      layered abstraction from a sequence of flat fx.Nodes. 
455   
456      Attributes: 
457      - _stack_meta: Metadata of the module stack. 
458      - _nodes: List of IR nodes in the module. 
459      - _reference_root_module: Reference to the root flat fx.GraphModule instance. 
460      """
461  
462      def __init__(
463          self, reference_root_module: torch.fx.GraphModule, stack_meta: _ModuleStackMeta
464      ):
465          self._stack_meta = stack_meta
466          self._nodes: list[_IRNode] = []
467          self._reference_module = reference_root_module
468  
469      @property
470      def stack_meta(self) -> _ModuleStackMeta:
471          return self._stack_meta
472  
473      @property
474      def stack_trace(self) -> str | None:
475          assert self._nodes
476          return self._nodes[0].stack_trace
477  
478      def __str__(self) -> str:
479          return f"ModuleNode({self._stack_meta})"
480  
481      def is_same_module_as(self, node: _IRNode) -> bool:
482          """Determines if the provided node pertains to the same module as this node."""
483          return self.stack_meta == node.stack_meta
484  
485      def is_parent_module_of(self, node: _IRNode) -> bool:
486          """Determines if this node represents a parent module of the provided node."""
487          return node.stack_meta.is_superset_of(self.stack_meta)
488  
489      def add_leaf_node(self, leaf_node: _LeafNode) -> None:
490          """Adds a leaf node to the module. 
491   
492          The leaf node must belong to the same or a child module. This method will recursively 
493          construct _ModuleNode instance based on the stack_meta information of the leaf node. 
494          """
495          if self.is_same_module_as(leaf_node) or leaf_node.fx_op == "call_module":
496              self._nodes.append(leaf_node)
497          elif leaf_node.fx_op == "placeholder":
498              # Although the original placeholder has empty nn_module_stack, the placeholder lifted
499              # from get_attr nodes by exported program has their original nn_module_stack. Here
500              # we need to avoid them building submodule.
501              self._nodes.append(leaf_node)
502          elif self.is_parent_module_of(leaf_node):
503              # This node belongs in a submodule.
504              # Check if the last node is a submodule and if it is the parent of this node.
505              last_node = self._nodes[-1] if self._nodes else None
506              if isinstance(last_node, _ModuleNode) and (
507                  last_node.is_parent_module_of(leaf_node)
508                  or last_node.is_same_module_as(leaf_node)
509              ):
510                  # This node belongs to the last_node.
511                  last_node.add_leaf_node(leaf_node)
512              else:
513                  # Create a new SubmoduleNode for the immediate child module of the current
514                  # module. The leaf node may be a grandchild of the current module.
515                  # Example:
516                  #   self.stack_meta = [A, B, C]
517                  #   leaf_node.stack_meta = [A, B, C, D, E, F]
518                  # Create a new ModuleNode with stack_meta = [A, B, C, D] and add leaf_node to it.
519                  stack_meta = copy.deepcopy(self.stack_meta)
520                  stack_meta.push(leaf_node.stack_meta[len(self.stack_meta)])
521                  last_node = _ModuleNode(
522                      self._reference_module,
523                      stack_meta,
524                  )
525                  self._nodes.append(last_node)
526                  last_node.add_leaf_node(leaf_node)
527          else:
528              raise AssertionError(
529                  f"Node {leaf_node} ({leaf_node.stack_meta}) does not belong to module "
530                  f"{self._stack_meta}."
531              )
532  
533      def fx_nodes(self) -> Generator[torch.fx.Node, None, None]:
534          """Returns an iterator for the sequence of fx nodes this instance holds."""
535          for node in self._nodes:
536              if isinstance(node, _ModuleNode):
537                  yield from node.fx_nodes()
538              else:
539                  assert isinstance(node, _LeafNode)
540                  yield node.fx_node

542      def module_inputs(self) -> Sequence[torch.fx.Node]:
543          """Extract module inputs from the sequence of fx nodes this instance holds. 
544   
545          All node args that are produced by nodes outside of the module are considered module 
546          inputs. The order of returned module inputs is the same as the their use order. 
547   
548          ### Known limitations 
549   
550          The original ordering of module inputs is not preserved. There is no meta information 
551          to be found from the `fx.GraphModule` that can be used to recover the original ordering. 
552   
553          Returns: 
554              Sequence of module inputs. 
555          """
556          nodes = list(self.fx_nodes())
557          assert len(nodes) > 0, "Cannot extract module inputs from empty nodes."
558          module_inputs: dict[torch.fx.Node, None] = {}
559          node_set: set[torch.fx.Node] = set(nodes)
560  
561          def _extract_arg_if_node_outside_module(arg: Any):
562              if isinstance(arg, torch.fx.Node) and arg not in node_set:
563                  module_inputs[arg] = None
564  
565          for node in nodes:
566              pytree.tree_map(_extract_arg_if_node_outside_module, node.args)
567              pytree.tree_map(_extract_arg_if_node_outside_module, node.kwargs)
568          return list(module_inputs.keys())
569  
570      def module_outputs(self) -> Sequence[torch.fx.Node]:
571          """Extract module outputs from the sequence of fx nodes this instance holds. 
572   
573          All nodes that are used by nodes outside of the module are considered module 
574          outputs. The order of returned module outputs is the same as the their creation order. 
575   
576          ### Known limitations 
577   
578          The original ordering of module outputs is not preserved. There is no meta information 
579          to be found from the `fx.GraphModule` that can be used to recover the original ordering. 
580   
581          Returns: 
582              Sequence of module outputs. 
583          """
584          nodes = list(self.fx_nodes())
585          assert len(nodes) > 0, "Cannot extract module inputs from empty nodes."
586          # Need ordered set. Emulate with dict.
587          module_outputs: dict[torch.fx.Node, None] = {}
588          node_set: set[torch.fx.Node] = set(nodes)
589  
590          for node in nodes:
591              if any(user not in node_set for user in node.users):
592                  module_outputs[node] = None
593          return list(module_outputs.keys())
594  
595      def build_module(self, module_names: dict[str, int]) -> torch.fx.GraphModule:
596          """ 
597          Constructs the fx.GraphModule for this node, registering submodules as necessary. 
598   
599          Args: 
600              module_names: A dictionary of module names and their counts. This is used to 
601                  generate unique module names for submodules. This should be an empty 
602                  dictionary when the method is called on a root module. 
603          """
604          module_class_name = self._stack_meta.qualified_module_class_name
605          fx_graph = torch.fx.Graph()
606          copy_env: dict[torch.fx.Node, torch.fx.Node] = {}
607  
608          def _arg_transform(node: torch.fx.Node) -> torch.fx.Node:
609              return copy_env[node]
610  
611          ref_inputs = self.module_inputs()
612          for node in ref_inputs:
613              copy_env[node] = fx_graph.placeholder(node.name, node.type)
614              copy_env[node].meta = copy.copy(node.meta)
615  
616          for ir_node in self._nodes:
617              if isinstance(ir_node, _LeafNode):
618                  fx_node = ir_node.fx_node
619                  copy_env[fx_node] = fx_graph.node_copy(
620                      fx_node, arg_transform=_arg_transform
621                  )
622                  continue
623  
624              assert isinstance(ir_node, _ModuleNode)
625              # Create fx.GraphModule for child submodule.
626              submodule = ir_node.build_module(module_names)
627              ref_submodule_inputs = ir_node.module_inputs()
628              ref_submodule_outputs = ir_node.module_outputs()
629              unique_submodule_name = _get_unique_module_name(
630                  module_names, ir_node.stack_meta.module_display_name
631              )
632              # Link the newly generated sub fx.GraphModule with the root reference module.
633              # This step is essential to meet the needs of the subsequent fx.GraphModule initialization
634              # for the fx.GraphModule being created by this method.
635              # The initialization of fx.GraphModule will replicate all necessary attributes from a reference
636              # fx.GraphModule for the fx.Graph. While the root reference module possesses all
637              # parameters and buffers, it does not include the newly created sub fx.GraphModule.
638              # Therefore, it's necessary to register it under the root reference at this stage.
639              self._reference_module.add_submodule(unique_submodule_name, submodule)
640  
641              # create call_module fx.Node
642              submodule_node = fx_graph.call_module(
643                  unique_submodule_name,
644                  tuple(_arg_transform(node) for node in ref_submodule_inputs),
645              )
646              if len(ref_submodule_outputs) > 1:
647                  # Module node has multiple output. Create 'getitem' node for each output.
648                  submodule_node.meta["val"] = tuple(
649                      ref_output.meta.get("val") for ref_output in ref_submodule_outputs
650                  )
651                  for i, ref_output in enumerate(ref_submodule_outputs):
652                      getitem_node = fx_graph.call_function(
653                          operator.getitem,
654                          args=(submodule_node, i),
655                          type_expr=ref_output.type,
656                      )
657                      getitem_node.meta = copy.copy(ref_output.meta)
658                      # Make a copy for "nn_module_stack" since the current module will be
659                      # popped from the stack for this 'getitem' node.
660                      getitem_node.meta["nn_module_stack"] = copy.copy(
661                          ref_output.meta["nn_module_stack"]
662                      )
663                      # The node is associated with the parent module.
664                      getitem_node.meta["nn_module_stack"].popitem()
665                      copy_env[ref_output] = getitem_node
666              else:
667                  # Module node has single output. Use module node directly.
668                  copy_env[ref_submodule_outputs[0]] = submodule_node
669                  submodule_node.meta = copy.copy(ref_submodule_outputs[0].meta)
670  
671              # Update meta for new call_module node.
672              if (stack_trace := ir_node.stack_trace) is not None:
673                  submodule_node.meta["stack_trace"] = stack_trace
674              raw_module_stack_meta = ir_node.stack_meta.raw_meta
675              assert raw_module_stack_meta is not None
676              submodule_node.meta["nn_module_stack"] = copy.copy(raw_module_stack_meta)
677              # The node is associated with the parent module.
678              submodule_node.meta["nn_module_stack"].popitem()
679  
680          new_nodes = fx_graph.nodes
681          # Skip if the last node is already 'output'. This is the case for root module.
682          # Otherwise create an 'output' node for the inferred outputs.
683          if next(iter(reversed(new_nodes))).op != "output":
684              ref_submodule_outputs = self.module_outputs()
685              new_outputs = [copy_env[ref_output] for ref_output in self.module_outputs()]
686              node = fx_graph.output(
687                  new_outputs[0] if len(new_outputs) == 1 else new_outputs
688              )
689  
690          graph_module = torch.fx.GraphModule(
691              self._reference_module, fx_graph, module_class_name
692          )
693          if (module_class := self._stack_meta.module_class) is not None:
694              graph_module.meta["onnx"] = _pass.GraphModuleOnnxMeta(
695                  _pass.PackageInfo.from_python_class(module_class)
696              )
697          return graph_module

733  class Modularize(_pass.Transform):
734      """Transforms a flattened `fx.GraphModule` into a modular structure. 
735   
736      In the flattened `fx.GraphModule`, each `nn.Module` forward call has been traced as 
737      a sequence of `fx.Node`s. All these `fx.Node`s are flattened and reside in the same 
738      `fx.GraphModule`. `fx.GraphModule` could be from `torch.export.ExportedProgram` or 
739      directly generated by `torch._dynamo.export` with torch.nn.Module. 
740   
741      This pass generates a new `fx.GraphModule`. It groups the flattened `fx.Node`s that belong 
742      to the same `nn.Module` forward call into a sub `fx.GraphModule`. It then replaces the 
743      sequence of flattened `fx.Node`s with a single `call_module` node, which is linked with 
744      the sub `fx.GraphModule` by `node.target`. The sub `fx.GraphModule` is registered as a 
745      submodule of the new `fx.GraphModule`. 
746   
747      The process is done based on information from the `nn_module_stack` metadata of each node, i.e. 
748      `node.meta["nn_module_stack"]`. For more implementation details, see [NOTE: Modularize Pass Implementation]. 
749   
750      An fx submodule under this context can typically be interpreted in three different ways: 
751   
752          1. As an embodiment of an nn.Module class, which is considered stateless. 
753          Its execution path can vary depending on the configuration of module initialization, 
754          which should also be part of the inputs. 
755   
756          2. As a representation of an nn.Module instance. It maintains the state initialized in the module. 
757          The execution path can vary based on actual input data. 
758   
759          3. As a captured call of an nn.Module instance, where the execution path 
760          is set. 
761   
762      The generality decreases along this list. Within the scope of this function, the pass 
763      creates fx submodules according to the third interpretation. 
764   
765      The first interpretation is the most general case. It requires complex analysis and additional 
766      metadata and code information to construct its general form. Consider an example nn.Module 
767      that generates arbitrary submodules based on an initialization configuration file. It's impractical 
768      to extract this logic for the generated fx submodule to function with arbitrary configuration. 
769   
770      The second interpretation demands less analysis and is sturdier than the 
771      first. In most use cases, it's equivalent to the third. It only differs in exceptional situations 
772      where a complex nn.Module instance is called multiple times, each with a different set of inputs 
773      leading to a unique execution branching path. 
774   
775      The third interpretation is the most specific scenario. It necessitates the minimum 
776      analysis and creates the most stable representation. The drawback is that it 
777      generates more redundancy than the other two methods. If needed, a subsequent post-processing 
778      pass can be applied to consolidate completely identical functions and reduce duplication. 
779   
780      ### Known constraints 
781      Two successive calls to the same module instance will be conflated. They are indistinguishable. 
782      This is due to limitations of the current fx metadata "nn_module_stack". 
783   
784      [NOTE: Modularize pass ordering] 
785      This pass groups fx nodes into subgraphs that reside within the `call_module` fx node. 
786      Other fx passes (including some outside the exporter) might not recognize `call_module`. 
787      They may assume that all nodes are flattened. Hence it is recommended to invoke this pass 
788      as the last pre onnx export fx pass. If not for this consideration, this operation could 
789      potentially be relocated anywhere earlier in the pipeline. 
790   
791      Example: 
792   
793          >>> # xdoctest: +REQUIRES(env:TORCH_DOCTEST_ONNX) 
794          >>> import torch 
795          >>> from torch.onnx._internal.fx import passes 
796          >>> 
797          >>> class CustomModule(torch.nn.Module): 
798          >>>     def __init__(self) -> None: 
799          >>>         super().__init__() 
800          >>>         self.embedding = torch.nn.Embedding(10, 32) 
801          >>>         self.relu = torch.nn.ReLU() 
802          >>> 
803          >>>     def forward(self, x): 
804          >>>         out = self.embedding(x) 
805          >>>         out = self.relu(out) 
806          >>>         return out 
807          >>> 
808          >>> class TestModule(torch.nn.Module): 
809          >>>     def __init__(self) -> None: 
810          >>>         super().__init__() 
811          >>>         self.layer = CustomModule() 
812          >>>         self.linear = torch.nn.Linear(32, 10) 
813          >>> 
814          >>>     def forward(self, x): 
815          >>>         out = self.layer(x) 
816          >>>         out = self.linear(out) 
817          >>>         return out 
818          >>> 
819          >>> gm, _ = torch._dynamo.export(TestModule(), aten_graph=True)( 
820          ...     torch.tensor([0, 1, 2]) 
821          ... ) 
822          >>> gm.print_readable() 
823   
824          >>> gm = passes.Modularize( 
825          ...     gm, 
826          ... ).run() 
827          >>> gm.print_readable() 
828   
829      """
830  
831      def __init__(
832          self,
833          module: torch.fx.GraphModule,
834          is_exported_program: bool = False,
835      ):
836          super().__init__(module)
837          self.module = module
838          self.is_exported_program = is_exported_program
839  
840      def _run(self) -> torch.fx.GraphModule:
841          # DCE to remove unused nodes.
842          # If a submodule is unused, it is hard to analyze which nodes constitutes the submodule
843          # outputs. But since it is unused, we can just remove it.
844          self.module.graph.eliminate_dead_code()
845  
846          reference_module = torch.fx.GraphModule(self.module, self.module.graph)
847          root_module_node = _ModuleNode(
848              reference_module,
849              _ModuleStackMeta(
850                  nn_module_stack_meta=None, is_exported_program=self.is_exported_program
851              ),
852          )
853          for fx_node in self.module.graph.nodes:
854              root_module_node.add_leaf_node(
855                  _LeafNode(fx_node, is_exported_program=self.is_exported_program)
856              )
857          return root_module_node.build_module({})
