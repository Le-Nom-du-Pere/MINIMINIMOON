1    # This code is backported from python 3.10 dataclasses. Once 3.10 becomes the
2    # minimum supported we should use dataclass(slots=True) instead.

7    import itertools
8    from typing import TYPE_CHECKING, TypeVar

12       from collections.abc import Generator

17   __all__ = ["dataclass_slots"]

19   _T = TypeVar("_T", bound="DataclassInstance")
20   
21   
22   def dataclass_slots(cls: type[_T]) -> type[DataclassInstance]:
23       assert dataclasses.is_dataclass(cls), "Can only be used on dataclasses."
24   
25       def _get_slots(cls: type[DataclassInstance]) -> Generator[str, None, None]:
26           slots = cls.__dict__.get("__slots__")
27           # `__dictoffset__` and `__weakrefoffset__` can tell us whether
28           # the base type has dict/weakref slots, in a way that works correctly
29           # for both Python classes and C extension types. Extension types
30           # don't use `__slots__` for slot creation
31           if slots is None:
32               slots = []
33               if getattr(cls, "__weakrefoffset__", -1) != 0:
34                   slots.append("__weakref__")
35               if getattr(cls, "__dictrefoffset__", -1) != 0:
36                   slots.append("__dict__")
37               yield from slots
38           elif isinstance(slots, str):
39               yield slots
40           # Slots may be any iterable, but we cannot handle an iterator
41           # because it will already be (partially) consumed.
42           elif not hasattr(cls, "__next__"):
43               yield from slots
44           else:
45               raise TypeError(f"Slots of '{cls.__name__}' cannot be determined")
46   
47       def _add_slots(
48           cls: type[DataclassInstance], is_frozen: bool, weakref_slot: bool
49       ) -> type[DataclassInstance]:
50           # Need to create a new class, since we can't set __slots__
51           #  after a class has been created.
52   
53           # Make sure __slots__ isn't already set.
54           if "__slots__" in cls.__dict__:
55               raise TypeError(f"{cls.__name__} already specifies __slots__")
56   
57           # Create a new dict for our new class.
58           cls_dict = dict(cls.__dict__)
59           field_names = tuple(f.name for f in dataclasses.fields(cls))
60           # Make sure slots don't overlap with those in base classes.
61           inherited_slots = set(
62               itertools.chain.from_iterable(map(_get_slots, cls.__mro__[1:-1]))
63           )
64           # The slots for our class.  Remove slots from our base classes.  Add
65           # '__weakref__' if weakref_slot was given, unless it is already present.
66           cls_dict["__slots__"] = tuple(
67               itertools.filterfalse(
68                   inherited_slots.__contains__,
69                   itertools.chain(
70                       # gh-93521: '__weakref__' also needs to be filtered out if
71                       # already present in inherited_slots
72                       field_names,
73                       ("__weakref__",) if weakref_slot else (),
74                   ),
75               ),
76           )
77   
78           for field_name in field_names:
79               # Remove our attributes, if present. They'll still be
80               #  available in _MARKER.
81               cls_dict.pop(field_name, None)
82   
83           # Remove __dict__ itself.
84           cls_dict.pop("__dict__", None)
85   
86           # Clear existing `__weakref__` descriptor, it belongs to a previous type:
87           cls_dict.pop("__weakref__", None)  # gh-102069
88   
89           # And finally create the class.
90           qualname = getattr(cls, "__qualname__", None)
91           cls = type(cls.__name__, cls.__bases__, cls_dict)
92           if qualname is not None:
93               cls.__qualname__ = qualname
94   
95           def _dataclass_getstate(self: _T) -> object:
96               fields = dataclasses.fields(self)
97               return [getattr(self, f.name) for f in fields]
98   
99           def _dataclass_setstate(self: _T, state: list[object]) -> None:
100              fields = dataclasses.fields(self)
101              for field, value in zip(fields, state):
102                  # use setattr because dataclass may be frozen
103                  object.__setattr__(self, field.name, value)
104  
105          if is_frozen:
106              # Need this for pickling frozen classes with slots.
107              if "__getstate__" not in cls_dict:
108                  cls.__getstate__ = _dataclass_getstate  # type: ignore[method-assign, assignment]
109              if "__setstate__" not in cls_dict:
110                  cls.__setstate__ = _dataclass_setstate  # type: ignore[attr-defined]
111  
112          return cls
113  
114      params = getattr(cls, dataclasses._PARAMS)  # type: ignore[attr-defined]
115      weakref_slot = getattr(params, "weakref_slot", False)
116      return _add_slots(cls, params.frozen, weakref_slot)
