36   @compatibility(is_backward_compatible=False)
37   class OperatorSupportBase(abc.ABC):
38       """Interface for determining if a fx.Node is supported by a backend"""
39   
40       @abc.abstractmethod
41       def is_node_supported(
42           self, submodules: t.Mapping[str, torch.nn.Module], node: torch.fx.Node
43       ) -> bool:
44           raise NotImplementedError

47   @compatibility(is_backward_compatible=False)
48   class OperatorSupport(OperatorSupportBase):
49       """ 
50       `_support_dict` maps node.target typename to supported inputs dtypes. 
51    
52       node.target typename is retrieved using helper function `get_node_target()` 
53    
54       If supported inputs dtypes is None, it means any dtype is supported, else 
55       we should see a tuple like (([dtypes], ...), {"name":[dtypes], ...}). 
56    
57       The first tuple ([dtypes], ...) indicates what dtypes are supported for 
58       inputs in node.args and the second dict {"name": [dtypes], ...} indicates 
59       what dtypes are supported for inputs in node.kwargs. 
60    
61       For inputs in args, if we don't want to check it, we can put None there, 
62       e.g. (None, [torch.float]) indicates that we don't care about the type of 
63       the first input in args. And for inputs in kwargs, if not listed, will not 
64       be checked. 
65       """
66   
67       _support_dict: SupportDict
68   
69       def __init__(self, support_dict: t.Optional[SupportDict] = None):
70           self._support_dict = support_dict or {}
71   
72       def is_node_supported(
73           self, submodules: t.Mapping[str, torch.nn.Module], node: torch.fx.Node
74       ) -> bool:
75           """ 
76           Args: 
77               `submodules`: mapping from module name to the module. This can be 
78                             retrieved by calling model.named_modules(). 
79    
80               `node`: a Fx node that we want to determine whether it's supported. 
81    
82           Returns: 
83               `is_supported`: whether the arg `node` is supported. 
84           """
85           if node.op not in CALLABLE_NODE_OPS:
86               return True
87   
88           target = get_node_target(submodules, node)
89   
90           # Target not found in _support_dict meaning that we don't support this op at all
91           if target not in self._support_dict:
92               return False
93   
94           # The rule for target is None meaning that we accept any dtype
95           if self._support_dict[target] is None:
96               return True
97   
98           args_dtypes, kwargs_dtypes = self._support_dict[target]  # type: ignore[misc]
99   
100          # Check args dtypes
101          for i, dtypes in enumerate(args_dtypes):
102              if len(node.args) <= i:
103                  break
104  
105              # None indicates we don't care about the dtype of args[i]
106              if dtypes is None:
107                  continue
108  
109              # If arg is not a node then we don't check it
110              if not isinstance(node.args[i], torch.fx.Node):
111                  continue
112  
113              arg_dtype = _get_arg_dtype(node.args[i])  # type: ignore[arg-type]
114              if arg_dtype not in dtypes:
115                  return False
116  
117          # Check kwargs dtypes
118          for k, dtypes in kwargs_dtypes.items():
119              if k not in node.kwargs:
120                  continue
121  
122              # If arg is not a node then we don't check it
123              if not isinstance(node.kwargs[k], torch.fx.Node):
124                  continue
125  
126              kwarg_dtype = _get_arg_dtype(node.kwargs[k])  # type: ignore[arg-type]
127              if kwarg_dtype not in dtypes:
128                  return False
129  
130          return True

141  @compatibility(is_backward_compatible=False)
142  def create_op_support(is_node_supported: IsNodeSupported) -> OperatorSupportBase:
143      """Wraps a `IsNodeSupported` function into an `OperatorSupportBase` instance 
144   
145      `IsNodeSupported` has the same call signature as 
146      `OperatorSupportBase.is_node_supported` 
147      """
148  
149      class FunctionalOperatorSupport(OperatorSupportBase):
150          def is_node_supported(
151              self, submodules: t.Mapping[str, torch.nn.Module], node: torch.fx.Node
152          ) -> bool:
153              return is_node_supported(submodules, node)
154  
155      return FunctionalOperatorSupport()

158  @compatibility(is_backward_compatible=False)
159  def chain(*op_support: OperatorSupportBase) -> OperatorSupportBase:
160      """Combines a sequence of `OperatorSupportBase` instances to form a single `OperatorSupportBase` 
161      instance by evaluating each input `OperatorSupportBase` instance, and returns False if 
162      any of it reports False. 
163      """
164  
165      def _chain(submods, node) -> bool:
166          return all(x.is_node_supported(submods, node) for x in op_support)
167  
168      return create_op_support(_chain)
169

171  @compatibility(is_backward_compatible=False)
172  def any_chain(*op_support: OperatorSupportBase) -> OperatorSupportBase:
173      """Combines a sequence of `OperatorSupportBase` instances to form a single `OperatorSupportBase` 
174      instance by evaluating each input `OperatorSupportBase` instance, and returns True if 
175      any of it reports True. 
176      """
177  
178      def _any_chain(submods, node) -> bool:
179          return any(x.is_node_supported(submods, node) for x in op_support)
180  
181      return create_op_support(_any_chain)

184  @compatibility(is_backward_compatible=False)
185  class OpSupports:
186      """A set of atomic `OperatorSupportBase` instances that can be combined together 
187      to form more complex operator support logic. 
188      """
189  
190      @classmethod
191      def decline_if_input_dtype(cls, dtype: torch.dtype) -> OperatorSupportBase:
192          """Report a node as non-supported, if any of its arguments is of dtype"""
193  
194          def _decline_if_input_dtype(
195              submodules: t.Mapping[str, torch.nn.Module],
196              node: torch.fx.Node,
197          ) -> bool:
198              for arg in node.all_input_nodes:
199                  arg_dtype = _get_arg_dtype(arg)
200                  if arg_dtype == dtype:
201                      return False
202              return True
203  
204          return create_op_support(_decline_if_input_dtype)
205  
206      @classmethod
207      def decline_if_node_in_names(cls, disallow_set: set[str]) -> OperatorSupportBase:
208          """ 
209          If a node has a name that is in the disallow set, reported it as non-supported. 
210          """
211  
212          def _decline_if_node_in_names(
213              submodules: t.Mapping[str, torch.nn.Module],
214              node: torch.fx.Node,
215          ) -> bool:
216              return node.name not in disallow_set
217  
218          return create_op_support(_decline_if_node_in_names)

221  def _get_arg_dtype(arg: torch.fx.Node) -> t.Any:
222      assert isinstance(arg, torch.fx.Node)
223      tensor_meta = arg.meta.get("tensor_meta")  # type: ignore[union-attr]
224      dtype = (
225          tensor_meta.dtype
226          if isinstance(tensor_meta, TensorMetadata)
227          else arg.meta["type"]
228      )
229      return dtype
