47   def is_onnxrt_backend_supported() -> bool:
48       """Returns ``True`` if ONNX Runtime dependencies are installed and usable 
49       to support TorchDynamo backend integration; ``False`` otherwise. 
50    
51       Example:: 
52    
53           # xdoctest: +REQUIRES(env:TORCH_DOCTEST_ONNX) 
54           >>> import torch 
55           >>> if torch.onnx.is_onnxrt_backend_supported(): 
56           ...     @torch.compile(backend="onnxrt") 
57           ...     def f(x): 
58           ...             return x * x 
59           ...     print(f(torch.randn(10))) 
60           ... else: 
61           ...     print("pip install onnx onnxscript onnxruntime") 
62           ... 
63       """
64       global _SUPPORT_ONNXRT
65   
66       if _SUPPORT_ONNXRT is None:
67           # `onnxruntime` might import a lot of other runtime packages,
68           # e.g. apex, deepspeed, transformers.
69           # So lazy-importing onnxruntime to avoid possible circular import.
70           try:
71               importlib.import_module("onnxruntime")
72               importlib.import_module("onnxruntime.capi._pybind_state")
73   
74               # This is not use directly in DORT but needed by underlying exporter,
75               # so we still need to check if it exists.
76               importlib.import_module("onnxscript")
77   
78               import torch.onnx  # noqa: F401
79               import torch.onnx._internal  # noqa: F401
80               import torch.onnx._internal._exporter_legacy  # noqa: F401
81               from torch.onnx._internal.fx import (  # noqa: F401
82                   decomposition_table,
83                   fx_onnx_interpreter,
84                   passes,
85                   type_utils,
86               )
87   
88               _SUPPORT_ONNXRT = True
89           except ImportError:
90               _SUPPORT_ONNXRT = False
91   
92       return _SUPPORT_ONNXRT

98   def _dump_onnx_model(
99       model_string: bytes, graph_module: Optional[torch.fx.GraphModule] = None
100  ) -> str:
101      """Stores the onnx model into a file. 
102      The name is "{ONNXRT_DUMP_PATH}{N}.onnx" 
103      where *N* is the number of files already stored with 
104      this prefix. 
105      If graph_module is not None, the graph is stored as a string with 
106      the same filename except the extension (.txt). 
107      """
108      prefix = os.environ.get("ONNXRT_DUMP_PATH", None)
109      if not prefix:
110          return ""
111      n = _dumped_onnx_model.get(prefix, -1) + 1
112      filename = f"{prefix}{n}.onnx"
113      with open(filename, "wb") as f:
114          f.write(model_string)
115      _dumped_onnx_model[prefix] = n
116      if graph_module is not None:
117          filename_txt = f"{prefix}{n}.txt"
118          with open(filename_txt, "w", encoding="utf-8") as f:
119              f.write(str(graph_module.graph))
120      return filename

147  def _get_ort_device_type(device_type: str):
148      from onnxruntime.capi import _pybind_state as ORTC
149  
150      if device_type == "cuda":
151          return ORTC.OrtDevice.cuda()
152      if device_type == "cpu":
153          return ORTC.OrtDevice.cpu()
154      # ort pytorch device is mapped to NPU OrtDevice type
155      if device_type == "maia":
156          return ORTC.OrtDevice.npu()
157      raise ValueError("Unsupported device type: " + device_type)

166  class OrtOperatorSupport(OperatorSupport):
167      """Operator support for ONNXRuntime backend. 
168   
169      It has two-level of support decision. One is via support_dict and the other one 
170      is via extra_support_dict. The logic of using support_dict is implemented in 
171      OrtOperatorSupport and extra_support_dict is used by OperatorSupport.is_node_supported. 
172      """
173  
174      def __init__(self, support_dict: set[Any], extra_support_dict: dict[str, Any]):
175          # Use extra_support_dict[op_name] = None to indicate
176          # we support op_name with all input types. Otherwise,
177          # see support_dict (type: SupportDict) in operator_support.py
178          # for specifying supported types.
179          super().__init__(extra_support_dict)
180          self._onnx_support_dict = support_dict
181  
182      def is_node_supported(
183          self, submodules: Mapping[str, torch.nn.Module], node: torch.fx.Node
184      ) -> bool:
185          # OperatorSupport.is_node_supported returns True for non-callable nodes.
186          # Since ORT can't execute them, we return False here to override the base
187          # behavior.
188          if node.op not in CALLABLE_NODE_OPS:
189              return False
190          # This is the and the only place to decide if aten op is supported.
191          if node.op == "call_function" and node.target in self._onnx_support_dict:
192              logger.info(
193                  "support_dict supports node.target: %s (type: %s)",
194                  node.target,
195                  type(node.target),
196              )
197              return True
198          # If node.target is not in support_dict, we still want to check if torch.jit.script
199          # can convert it to ONNX equivalence. Let's use base mechanism to do this.
200          # See extra_support_dict  for supported ops.
201          if super().is_node_supported(submodules, node):
202              logger.info(
203                  "extra_support_dict supports node.target: %s (type: %s)",
204                  node.target,
205                  type(node.target),
206              )
207              return True
208          logger.warning(
209              "support_dict and extra_support_dict don't support node.target: %s (type: %s)",
210              node.target,
211              type(node.target),
212          )
213          return False

216  def _move_placeholder_to_front(graph_module: torch.fx.GraphModule) -> None:
217      """ 
218      In torch.fx.Graph, placeholder is a special assignment node. If it's not 
219      executed in the beginning, it could overwrite values computed by upstream 
220      nodes. 
221      """
222  
223      graph = graph_module.graph
224      placeholders = []
225      first_not_placeholder = None
226      for node in graph.nodes:
227          if node.op == "placeholder":
228              placeholders.append(node)
229          if first_not_placeholder is None and node.op != "placeholder":
230              first_not_placeholder = node
231      if first_not_placeholder is None:
232          return
233      for placeholder in placeholders:
234          first_not_placeholder.prepend(placeholder)

237  def _infer_ep_from_device(*args) -> tuple[str, ...]:
238      """Return the first valid device (i.e., GPU or CPU) in argument list."""
239      eps = []
240      for arg in args:
241          if hasattr(arg, "device"):
242              device = arg.device
243              if device.type == "cuda":
244                  eps.append("CUDAExecutionProvider")
245              elif device.type == "cpu":
246                  eps.append("CPUExecutionProvider")
247      return tuple(eps)

250  def _extract_graph_module_inputs(graph_module: torch.fx.GraphModule) -> tuple[Any, ...]:
251      placeholders = []
252      for node in graph_module.graph.nodes:
253          if node.op == "placeholder":
254              if hasattr(node, "meta") and "val" in node.meta:
255                  assert isinstance(node.meta["val"], torch.Tensor)
256              placeholders.append(node)
257      return tuple(placeholders)

260  def _extract_graph_module_outputs(graph_module: torch.fx.GraphModule) -> Any:
261      """Collect "val" fields from outputs metadata in this torch.fx.GraphModule."""
262      for node in graph_module.graph.nodes:
263          if node.op == "output":
264              # Output node is unique. Let's retrieve output values from
265              # this node's input list. And then just return.
266              return node.args[0]
267      raise ValueError("No output node found in this torch.fx.GraphModule.")

270  def _infer_ep_from_graph_module(graph_module: torch.fx.GraphModule) -> tuple[str, ...]:
271      """Return the all valid devices (i.e., GPU or CPU) among outputs of this torch.fx.GraphModule."""
272      flattened_output_args, _ = _pytree.tree_flatten(
273          _extract_graph_module_outputs(graph_module)
274      )
275      # Output arguments with example value (type: torch.Tensor) in the `graph_module`.
276      selected_output_args = [
277          output_arg.meta["val"]
278          for output_arg in flattened_output_args
279          # output_arg must have tensor for its device information.
280          # Otherwise, skip it.
281          if (hasattr(output_arg, "meta") and "val" in output_arg.meta)
282      ]
283      return _infer_ep_from_device(*selected_output_args)

286  def _sort_eps(eps: tuple[str, ...]) -> tuple[str, ...]:
287      """Sort execution providers in eps based on pre-set priority."""
288  
289      def get_execution_provider_priority(ep: str) -> int:
290          if ep == "CPUExecutionProvider":
291              # Lowest priority.
292              return 2
293          if ep == "CUDAExecutionProvider":
294              # Higher priority than CPU but lower than
295              # other specialized EPs.
296              return 1
297          # Highest priority.
298          return 0
299  
300      unique_eps = set(eps)
301      return tuple(sorted(unique_eps, key=get_execution_provider_priority, reverse=True))

304  def _get_onnx_devices(
305      values: tuple[
306          Union[
307              torch.Tensor, torch.SymInt, int, torch.SymFloat, float, torch.SymBool, bool
308          ],
309          ...,
310      ],
311  ) -> tuple["ORTC.OrtDevice", ...]:
312      from onnxruntime.capi import _pybind_state as ORTC
313  
314      def _device_id_or_zero(device_id: int) -> int:
315          return device_id or 0
316  
317      def _map_tensor_or_sym_to_device(
318          value: Union[
319              torch.Tensor, torch.SymInt, int, torch.SymFloat, float, torch.SymBool, bool
320          ],
321      ) -> int:
322          if isinstance(value, torch.Tensor):
323              return ORTC.OrtDevice(
324                  _get_ort_device_type(value.device.type),
325                  ORTC.OrtDevice.default_memory(),
326                  _device_id_or_zero(value.device.index),
327              )
328          elif isinstance(
329              value, (torch.SymInt, int, torch.SymFloat, float, torch.SymBool, bool)
330          ):
331              return ORTC.OrtDevice(
332                  _get_ort_device_type("cpu"), ORTC.OrtDevice.default_memory(), 0
333              )
334          else:
335              raise ValueError("Unsupported value type: " + str(type(value)))
336  
337      if len(values) > 0:
338          ort_devices = tuple(_map_tensor_or_sym_to_device(value) for value in values)
339          return ort_devices
340      else:
341          return (_map_tensor_or_sym_to_device(1),)

344  def _get_ortvalues_from_torch_tensors(
345      tensors: tuple[torch.Tensor, ...], devices: tuple["ORTC.OrtDevice", ...]
346  ) -> tuple[torch.Tensor, ...]:
347      # TODO(justinchuby): Refactor this function
348      import numpy as np
349      from onnxruntime.capi import _pybind_state as ORTC
350  
351      torch_dtype_to_numpy_dtype = {
352          torch.float16: np.float16,
353          torch.float32: np.float32,
354          torch.float64: np.float64,
355          torch.uint8: np.uint8,
356          torch.int8: np.int8,
357          torch.int16: np.int16,
358          torch.int32: np.int32,
359          torch.int64: np.longlong,
360          torch.bool: np.bool_,
361      }
362      ortvalues = ORTC.OrtValueVector()
363      ortvalues.reserve(len(tensors))
364      dtypes = []
365      shapes = []
366      data_ptrs = []
367  
368      for tensor in tensors:
369          dtypes.append(torch_dtype_to_numpy_dtype[tensor.dtype])
370          shapes.append(tensor.size())
371          data_ptrs.append(tensor.data_ptr())
372      ortvalues.push_back_batch(tensors, data_ptrs, dtypes, shapes, devices)
373      return ortvalues

376  def _to_real_tensor(tensor: FakeTensor) -> torch.Tensor:
377      if tensor.is_sparse:
378          raise ValueError("sparse tensor is not yet supported.")
379      out = torch.empty(tensor.size(), dtype=tensor.dtype, device=tensor.device)
380      return out

383  def _adjust_scalar_from_fx_to_onnx(
384      dynamo_value: Union[
385          torch.Tensor,
386          int,
387          float,
388          bool,
389      ],
390      value_info: "onnx.ValueInfoProto",  # type: ignore[name-defined]
391  ) -> torch.Tensor:
392      """Helper function to wrap PyTorch variables as torch.Tensor"""
393      if (
394          isinstance(dynamo_value, torch.Tensor)
395          and len(value_info.type.tensor_type.shape.dim) == 0
396          and dynamo_value.shape == (1,)
397      ):
398          # ONNX expect a scalar with empty shape.
399          # In contrast, PyTorch usually allows implicit
400          # conversion between shape=() and shape=(1,).
401          #
402          # Below, PyTorch's shape (1,) is reshaped to ().
403          return torch.squeeze(dynamo_value)
404      elif isinstance(dynamo_value, int):
405          return torch.tensor(dynamo_value, dtype=torch.int64)
406      elif isinstance(dynamo_value, float):
407          return torch.tensor(dynamo_value, dtype=torch.float32)
408      elif isinstance(dynamo_value, bool):
409          return torch.tensor(dynamo_value, dtype=torch.bool)
410      else:
411          assert isinstance(dynamo_value, torch.Tensor)
412          return dynamo_value.contiguous()

415  def _adjust_scalar_from_onnx_to_fx(
416      tensor: torch.Tensor,
417      prim_value: Union[
418          torch.Tensor,
419          torch.SymInt,
420          int,
421          torch.SymFloat,
422          float,
423          torch.SymBool,
424          bool,
425      ],
426  ) -> Union[
427      torch.Tensor,
428      int,
429      float,
430      bool,
431  ]:
432      """Helper function to wrap ORT-produced torch.Tensor as PyTorch variables"""
433      assert isinstance(tensor, torch.Tensor), "ORT's output must be tensor."
434      if isinstance(
435          prim_value,
436          (torch.SymInt, int, torch.SymFloat, float, torch.SymBool, bool),
437      ):
438          # Convert tensor back to scalar to match Dynamo's expectation.
439          return tensor.item()
440      return tensor

443  def _run_onnx_session_with_ortvaluevector(
444      sess: "onnxruntime.InferenceSession",
445      input_names: tuple[str, ...],
446      inputs: tuple[torch.Tensor, ...],
447      input_devices: tuple["ORTC.OrtDevice", ...],
448      output_names: tuple[str, ...],
449      outputs: tuple[torch.Tensor, ...],
450      output_devices: tuple["ORTC.OrtDevice", ...],
451      preallocate_output: bool,
452      input_value_infos: tuple["onnx.ValueInfoProto", ...],  # type: ignore[name-defined]
453      normalized_prim_outputs: tuple[
454          Union[
455              torch.Tensor, torch.SymInt, int, torch.SymFloat, float, torch.SymBool, bool
456          ],
457          ...,
458      ],
459  ) -> tuple[Union[torch.Tensor, int, float, bool], ...]:
460      import onnxruntime
461      from onnxruntime.capi import _pybind_state as ORTC
462  
463      _nvtx_range_push("contiguous")
464      inputs = tuple(
465          _adjust_scalar_from_fx_to_onnx(arg, value_info)
466          for arg, value_info in zip(inputs, input_value_infos)
467      )
468      _nvtx_range_pop()
469  
470      _nvtx_range_push("push_back_batch")
471      ort_inputs = _get_ortvalues_from_torch_tensors(inputs, input_devices)
472  
473      # preallocate output pytorch Tensors and use the buffers affined to the torch device for the output ortvalue.
474      # Because the output ortvalue is not allocated and owned by ort, it does not need to convert the output ortvalue
475      # to torch Tensor transferring the ownership.
476      if preallocate_output:
477          pth_outputs = tuple(
478              _to_real_tensor(t) if isinstance(t, FakeTensor) else t for t in outputs
479          )
480          ort_outputs = _get_ortvalues_from_torch_tensors(pth_outputs, output_devices)
481      else:
482          ort_outputs = ORTC.OrtValueVector()
483      _nvtx_range_pop()
484  
485      _nvtx_range_push("run_with_ortvaluevector")
486      run_options = onnxruntime.RunOptions()
487      run_options.add_run_config_entry("disable_synchronize_execution_providers", "1")
488      sess.run_with_ortvaluevector(
489          run_options, input_names, ort_inputs, output_names, ort_outputs, output_devices
490      )
491      _nvtx_range_pop()
492  
493      # Post-processing step:
494      #  wrap ORT's outputs to the schema represented by
495      #  `prim_output` (obtained by running the original
496      #  torch.fx.GraphModule).
497      if preallocate_output:
498          # Profile the ORT-to-PyTorch type cast below
499          _nvtx_range_push("after run_with_ortvaluevector")
500          # Outputs are stored on pre-allocated torch.Tensors' memory,
501          # so this case doesn't need to convert ORTValue to torch.Tensor.
502          pth_outputs = tuple(
503              _adjust_scalar_from_onnx_to_fx(onnx_output, prim_output)  # type: ignore[misc]
504              for onnx_output, prim_output in zip(pth_outputs, normalized_prim_outputs)
505          )
506          _nvtx_range_pop()
507          return pth_outputs
508      else:
509          import onnxruntime.training
510  
511          # Profile the two ORT-to-PyTorch type casts below
512          _nvtx_range_push("after run_with_ortvaluevector")
513          # Map ORTValue to torch.Tensor.
514          pth_outputs = onnxruntime.training.ortmodule._utils._ortvalues_to_torch_tensor(
515              ort_outputs
516          )
517          # Change some torch.Tensor to int, float, bool.
518          pth_outputs = tuple(
519              _adjust_scalar_from_onnx_to_fx(onnx_output, prim_output)  # type: ignore[misc]
520              for onnx_output, prim_output in zip(pth_outputs, normalized_prim_outputs)
521          )
522          _nvtx_range_pop()
523          return pth_outputs

526  def _run_onnx_session_with_fetch(
527      sess: "onnxruntime.InferenceSession",
528      input_names: tuple[str, ...],
529      inputs: tuple[torch.Tensor, ...],
530      input_devices: tuple["ORTC.OrtDevice", ...],
531      output_names: tuple[str, ...],
532      outputs: tuple[torch.Tensor, ...],
533      output_devices: tuple["ORTC.OrtDevice", ...],
534      preallocate_output: bool,
535      input_value_infos: tuple["onnx.ValueInfoProto", ...],  # type: ignore[name-defined]
536      normalized_prim_outputs: tuple[
537          Union[
538              torch.Tensor, torch.SymInt, int, torch.SymFloat, float, torch.SymBool, bool
539          ],
540          ...,
541      ],
542  ) -> tuple[Union[torch.Tensor, int, float, bool], ...]:
543      import onnxruntime
544  
545      inputs = tuple(
546          _adjust_scalar_from_fx_to_onnx(arg, value_info)
547          for arg, value_info in zip(inputs, input_value_infos)
548      )
549      feed = {
550          name: onnxruntime.OrtValue.ortvalue_from_numpy(tensor.cpu().numpy())
551          for name, tensor in zip(input_names, inputs)
552      }
553      ort_outputs = sess.run(output_names, feed)
554      pth_outputs = tuple(
555          _adjust_scalar_from_onnx_to_fx(
556              torch.from_numpy(value),
557              prim_output,
558          )
559          for value, prim_output in zip(ort_outputs, normalized_prim_outputs)
560      )
561      return pth_outputs

564  def _from_python_type_to_onnx_tensor_element_type(type: type):
565      """ 
566      Converts a Python type to the corresponding ONNX tensor element type. 
567      For example, `_from_python_type_to_onnx_tensor_element_type(float)` returns 
568      `onnx.TensorProto.FLOAT`. 
569   
570      Args: 
571        type (type): The Python type to convert. 
572   
573      Returns: 
574        int: The corresponding ONNX tensor element type. 
575   
576      """
577      import onnx
578  
579      _PYTHON_TYPE_TO_ONNX_TENSOR_ELEMENT_TYPE = {
580          float: onnx.TensorProto.FLOAT,  # type: ignore[attr-defined]
581          int: onnx.TensorProto.INT64,  # type: ignore[attr-defined]
582          bool: onnx.TensorProto.BOOL,  # type: ignore[attr-defined]
583      }
584      return _PYTHON_TYPE_TO_ONNX_TENSOR_ELEMENT_TYPE.get(type)

587  class OrtExecutionInfoPerSession:
588      """Information required to execute torch.fx.GraphModule using onnxruntime.InferenceSession"""
589  
590      def __init__(
591          self,
592          session: "onnxruntime.InferenceSession",
593          input_names: tuple[str, ...],
594          input_value_infos: tuple["onnx.ValueInfoProto", ...],  # type: ignore[name-defined]
595          output_names: tuple[str, ...],
596          output_value_infos: tuple["onnx.ValueInfoProto", ...],  # type: ignore[name-defined]
597          input_devices: tuple["ORTC.OrtDevice", ...],
598          output_devices: tuple["ORTC.OrtDevice", ...],
599          example_outputs: Union[tuple[torch.Tensor, ...], torch.Tensor],
600      ):
601          # Carrier of ONNX model and its executor.
602          self.session: onnxruntime.InferenceSession = session
603          # For the ONNX model stored in self.session, self.input_names[i] is the
604          # name of the i-th positional input.
605          self.input_names: tuple[str, ...] = input_names
606          # self.input_name[i]'s type information is stored in self.input_value_infos[i].
607          self.input_value_infos: tuple[onnx.ValueInfoProto, ...] = input_value_infos  # type: ignore[name-defined]
608          # Similar to self.input_names, but for outputs.
609          self.output_names: tuple[str, ...] = output_names
610          # Similar to self.input_value_infos but for outputs.
611          self.output_value_infos: tuple[onnx.ValueInfoProto, ...] = output_value_infos  # type: ignore[name-defined]
612          # For the ONNX model stored in self.session, self.input_devices[i] is the
613          # i-th positional input's device.
614          self.input_devices: tuple[ORTC.OrtDevice, ...] = input_devices
615          # Similar to self.input_devices, but for outputs.
616          self.output_devices: tuple[ORTC.OrtDevice, ...] = output_devices
617          # This is the outputs of executing the original torch.fx.GraphModule with example inputs
618          # (i.e., args passed into OrtBackend._ort_acclerated_call).
619          self.example_outputs: Union[tuple[torch.Tensor, ...], torch.Tensor] = (
620              example_outputs
621          )

623      def is_supported(self, *args):
624          # TODO(justinchuby): Simplify
625          import onnx
626  
627          _onnx_tensor_element_type_to_torch_dtype = {
628              onnx.TensorProto.FLOAT: torch.float32,  # type: ignore[attr-defined]
629              onnx.TensorProto.FLOAT16: torch.float16,  # type: ignore[attr-defined]
630              onnx.TensorProto.FLOAT8E5M2: torch.float8_e5m2,  # type: ignore[attr-defined]
631              onnx.TensorProto.FLOAT8E5M2FNUZ: torch.float8_e5m2fnuz,  # type: ignore[attr-defined]
632              onnx.TensorProto.FLOAT8E4M3FN: torch.float8_e4m3fn,  # type: ignore[attr-defined]
633              onnx.TensorProto.FLOAT8E4M3FNUZ: torch.float8_e4m3fnuz,  # type: ignore[attr-defined]
634              onnx.TensorProto.DOUBLE: torch.float64,  # type: ignore[attr-defined]
635              onnx.TensorProto.BOOL: torch.bool,  # type: ignore[attr-defined]
636              onnx.TensorProto.UINT8: torch.uint8,  # type: ignore[attr-defined]
637              onnx.TensorProto.INT8: torch.int8,  # type: ignore[attr-defined]
638              onnx.TensorProto.INT16: torch.int16,  # type: ignore[attr-defined]
639              onnx.TensorProto.INT32: torch.int32,  # type: ignore[attr-defined]
640              onnx.TensorProto.INT64: torch.int64,  # type: ignore[attr-defined]
641          }
642          _torch_dtype_to_onnx_tensor_element_type = {
643              value: key
644              for key, value in _onnx_tensor_element_type_to_torch_dtype.items()
645          }
646  
647          # Compare the args and the input schema in ONNX model and
648          # return the first match.
649          if len(args) != len(self.input_value_infos):
650              return False
651          for arg, value_info in zip(args, self.input_value_infos):
652              if not isinstance(arg, (torch.Tensor, float, int)):
653                  return False
654  
655              # Check Python scalars such as int, float, and bool.
656              if isinstance(arg, (int, float, bool)):
657                  # Map, e.g., float to onnx.TensorProto.FLOAT.
658                  onnx_dtype = _from_python_type_to_onnx_tensor_element_type(type(arg))
659                  if onnx_dtype != value_info.type.tensor_type.elem_type:
660                      return False
661                  if len(value_info.type.tensor_type.shape.dim) != 0:
662                      return False
663                  continue
664  
665              # Check tensor.
666              onnx_dtype = _torch_dtype_to_onnx_tensor_element_type[arg.dtype]
667              if onnx_dtype != value_info.type.tensor_type.elem_type:
668                  return False
669              for dim, onnx_dim in zip(arg.shape, value_info.type.tensor_type.shape.dim):
670                  if isinstance(dim, int) and (
671                      onnx_dim.dim_value == dim or onnx_dim.dim_param
672                  ):
673                      continue
674                  elif isinstance(dim, torch.SymInt) and onnx_dim.dim_param:
675                      continue
676                  else:
677                      return False
678          return True

682  class OrtExecutionInfoForAllGraphModules:
683      def __init__(self) -> None:
684          # All sessions (and their related information) created by exporting the same GraphModule
685          # with different inputs.
686          self.execution_info_per_graph_module: dict[
687              torch.fx.GraphModule, list[OrtExecutionInfoPerSession]
688          ] = {}
689  
690      def search_reusable_session_execution_info(
691          self, graph_module: torch.fx.GraphModule, *args
692      ):
693          if graph_module not in self.execution_info_per_graph_module:
694              return None
695          # All execution information for ONNX models exported from the same `graph_module`
696          # with different inputs.
697          candidates = self.execution_info_per_graph_module[graph_module]
698  
699          for candidate in candidates:
700              if candidate.is_supported(*args):
701                  # Returns the first session that accepts this input schema.
702                  return candidate
703          # No reusable session found.
704          return None

706      def cache_session_execution_info(
707          self, graph_module: torch.fx.GraphModule, info: OrtExecutionInfoPerSession
708      ):
709          if graph_module not in self.execution_info_per_graph_module:
710              self.execution_info_per_graph_module[graph_module] = [info]
711          else:
712              self.execution_info_per_graph_module[graph_module].append(info)

790  @compatibility(is_backward_compatible=False)
791  class OrtBackend:
792      """A backend compiles (sub-)graphs in torch.fx.GraphModule to onnxruntime.InferenceSession calls. 
793   
794      The compiler entry point is OrtBackend.compile, which 
795          1. partitions the original graph into supported sub-graphs (type: torch.fx.GraphModule) and unsupported 
796             sub-graphs. 
797          2. For each supported sub-graph, it replaces its _wrapped_call function with _ort_accelerated_call. 
798          3. Inside _ort_accelerated_call, it creates onnxruntime.InferenceSession and calls it to execute the sub-graph. 
799      """
800  
801      def __init__(self, options: Optional[OrtBackendOptions] = None):
802          from onnxruntime.capi import _pybind_state as ORTC
803  
804          import torch.onnx
805          import torch.onnx._internal._exporter_legacy
806          import torch.onnx._internal.fx.decomposition_table
807  
808          self._options: Final = OrtBackendOptions() if options is None else options
809  
810          # options.export_options contains information shared between exporter and DORT.
811          # For example, they should use the same decomposition table when
812          #  1. capturing FX graph in torch.compile (see how we create aot_ort in register_backend.py)
813          #  2. call exporter's API to convert `torch.fx.GraphModule` to ONNX model
814          #     (see onnxfunction_dispatcher passed to FxOnnxInterpreter.run below).
815          #
816          # Convert user-facing option to internal option used by ONNX exporter
817          # to access required information.
818          # Some useful fields:
819          # - Decomposition table for decomposing FX operators in exporter is
820          #   self._resolved_onnx_exporter_options.decomposition_table.
821          # - self._resolved_onnx_exporter_options.onnx_registry records what
822          #   aten/prim ops are supported by exporter and their exporters (type: callable).
823          self._resolved_onnx_exporter_options = (
824              torch.onnx._internal._exporter_legacy.ResolvedExportOptions()
825          )
826  
827          #  Given DORT's computation flow:
828          #   1. OrtOperatorSupport uses support_dict and extra_support_dict to select operators
829          #      and send them to DORT.
830          #   2. Then, DORT exports the selected sub-graphs into ONNX.
831          #   3. Finally DORT calls ORT to do the computation.
832          #  OrtOperatorSupport and create_onnx_friendly_decomposition_table(...)
833          #  must use the same support_dict. If the support_dict here contains something not
834          #  supported by exporter, exporter will fails in step 2 since the selected graphs may
835          #  contains unsupported operators such as aten::_who_you_are.
836          #  This restriction is automatically done since DORT and exporter shares the same
837          #  self._resolved_onnx_exporter_options.
838          support_dict = torch.onnx._internal.fx.decomposition_table._create_onnx_supports_op_overload_table(
839              self._resolved_onnx_exporter_options.onnx_registry
840          )
841  
842          extra_support_dict: dict[str, Any] = {
843              "getattr": None,
844              # To send operator.getitem to ORT, add the corresponding string
845              # recognized by PyTorch's OperatorSupport class.
846              "_operator.getitem": None,
847              # To send operator.mul to ORT, add the corresponding string
848              # recognized by PyTorch's OperatorSupport class.
849              "_operator.mul": None,
850              "_operator.add": None,
851              "_operator.sub": None,
852          }
853  
854          self._supported_ops = OrtOperatorSupport(support_dict, extra_support_dict)
855          # TODO(wschin): this is a naive implementation of cache without proper guard
856          # See https://github.com/pytorch/pytorch/issues/106868.
857          self._partitioner_cache: dict[torch.fx.GraphModule, torch.fx.GraphModule] = {}
858          # Conceptually, this filed is a 2-layer dictionary
859          #   GraphModule 0
860          #     ONNX Model 0 (with ORT InferenceSession and related information. type: OrtExecutionInfoPerSession)
861          #     ONNX Model 1
862          #     ...
863          #   GraphModule 1
864          #     ONNX Model 2 (with ORT InferenceSession and related information. type: OrtExecutionInfoPerSession)
865          #     ONNX Model 3
866          #     ...
867          #   ...
868          # , which caches all previous compilation result so that we can reuse them.
869          # ONNX Model 0 and 1 are exported from the same GraphModule 0 but with different inputs
870          # (e.g., tensors with different ranks). GraphModule 0 and GraphModule 1 are different
871          # graphs captured by Dynamo and sent to OrtBackend.compile.
872          self._all_ort_execution_info = OrtExecutionInfoForAllGraphModules()
873  
874          self._assert_allclose_to_baseline = False
875  
876          self.execution_count = 0
877  
878          # Function which invokes ORT do to the real computation.
879          self.run = (
880              _run_onnx_session_with_ortvaluevector
881              if hasattr(ORTC.OrtValueVector, "push_back_batch")
882              else _run_onnx_session_with_fetch
883          )

885      def _select_eps(
886          self, graph_module: torch.fx.GraphModule, *args
887      ) -> Sequence[tuple[str, Mapping[str, Any]]]:
888          inferred_eps: tuple[str, ...] = ()
889          if self._options.infer_execution_providers:
890              if eps_from_args := _infer_ep_from_device(*args):
891                  # If user feeds CUDA tensor as input argument,
892                  # we want to use CUDA EP.
893                  # Thus, `eps_from_args` (deduced from input arguments)
894                  # has highest priority.
895                  inferred_eps = eps_from_args
896              elif eps_from_graph_module := _infer_ep_from_graph_module(graph_module):
897                  # If there is no EP in input arguments, we deduce EP from
898                  # graph_module's outputs. Those outputs may come from
899                  # FakeTensorProp or Dynamo's built-in symbolic shape inference.
900                  inferred_eps = eps_from_graph_module
901  
902          selected_eps = []
903  
904          for ep in (
905              *(self._options.preferred_execution_providers or []),
906              *_sort_eps(inferred_eps),
907              *(self._options.default_execution_providers or _infer_default_eps()),
908          ):
909              if isinstance(ep, str):
910                  ep = (ep, {})
911              elif isinstance(ep, tuple) and ep[1] is None:
912                  ep = (ep[0], {})
913              if ep is not None and ep not in selected_eps:
914                  selected_eps.append(ep)
915  
916          return selected_eps

918      def _ort_acclerated_call(self, graph_module: torch.fx.GraphModule, *args, **kwargs):
919          """This function replaces GraphModule._wrapped_call in compiled model. 
920   
921          The _wrapped_call is the underlying implementation of forward method. Replacing 
922          it means we delegate the computation to _ort_acclerated_call and therefore 
923          onnxruntime.InferenceSession. 
924          """
925          import onnxruntime
926  
927          from torch.onnx._internal.fx import fx_onnx_interpreter, passes
928  
929          cached_execution_info_per_session = (
930              self._all_ort_execution_info.search_reusable_session_execution_info(
931                  graph_module, *args
932              )
933          )
934          if cached_execution_info_per_session:
935              onnx_session = cached_execution_info_per_session.session
936              input_names = cached_execution_info_per_session.input_names
937              output_names = cached_execution_info_per_session.output_names
938              input_value_infos = cached_execution_info_per_session.input_value_infos
939              output_value_infos = cached_execution_info_per_session.output_value_infos
940              input_devices = cached_execution_info_per_session.input_devices
941              output_devices = cached_execution_info_per_session.output_devices
942              prim_outputs = cached_execution_info_per_session.example_outputs
943          else:
944              # It's first time seeing such as graph. Let's make a new session
945              # (type: onnxruntime.InferenceSession) for it.
946  
947              graph_module = passes.MovePlaceholderToFront(
948                  graph_module,
949              ).run()
950              # Generate reference outputs. They are used to indicate output
951              # tensors' types and devices when calling ORT.
952              #
953              # WARNING: The downstream code should not change prim_outputs and
954              # this backend should always produces output with schema identical to prim_outputs'.
955  
956              if self._resolved_onnx_exporter_options.dynamic_shapes:
957                  # No pre-allocation when dynamic shape is enabled.
958                  self.preallocate_output = False
959                  extracted_outputs = _extract_graph_module_outputs(graph_module)
960  
961                  def maybe_map_to_meta_val(value):
962                      if hasattr(value, "meta") and "val" in value.meta:
963                          # Select outputs with "val" information. Without "val",
964                          # it's not possible access output_arg.meta["val"].device.
965                          return value.meta["val"]
966                      else:
967                          return value
968  
969                  prim_outputs = _pytree.tree_map(
970                      maybe_map_to_meta_val, extracted_outputs
971                  )
972              else:
973                  try:
974                      prim_outputs = FakeTensorProp(graph_module).propagate(
975                          *args, **kwargs
976                      )
977                  except Exception:
978                      logger.warning("FakeTensorProb failed for %s", graph_module)
979                      # When FakeTensorProp fails, it is not possible to preallocate output buffers
980                      # because the output shapes are not inferred.
981                      self.preallocate_output = False
982  
983                      # rethrow FakeTensorProb failure because it is not yet currently handled.
984                      raise
985  
986              # Create the object to iterate through the nodes in graph one-by-one
987              # and calls the corresponding ONNX exporter for each node.
988              fx_interpreter = fx_onnx_interpreter.FxOnnxInterpreter()
989              # Cast FX variables if they will result schema-mismatch when searching
990              # for ONNX operator. E.g., add(double_tensor, int_tensor) is fine in PyTorch,
991              # but ONNX expects add(double_tensor, double_tensor).
992              graph_module = passes.InsertTypePromotion(graph_module).run()
993              # Start the per-node exporting process. It's conceptually a for loop
994              # scanning through the nodes in the graph.
995              exported = fx_interpreter.run(
996                  fx_graph_module=graph_module,
997                  onnxfunction_dispatcher=self._resolved_onnx_exporter_options.onnxfunction_dispatcher,
998              )
999              # Convert the exported result to ONNX ModelProto.
1000             onnx_model = exported.to_model_proto(
1001                 opset_version=self._resolved_onnx_exporter_options.onnx_registry.opset_version,
1002             )
1003 
1004             # Modify ONNX model using pre-registered graph transforms.
1005             # They are in-place modifications for avoiding unnecessary
1006             # copy of ONNX initializers.
1007             if self._options.pre_ort_model_transforms:
1008                 for transform in self._options.pre_ort_model_transforms:
1009                     transform(onnx_model)
1010 
1011             onnx_model_bytes = onnx_model.SerializeToString()
1012             if os.environ.get("ONNXRT_DUMP_PATH", None):
1013                 # If not empty, environment variable ONNXRT_DUMP_PATH defined the path
1014                 # where generated onnx files should be stored.
1015                 # This module keeps a global variables keeping track of the
1016                 # stored models.
1017                 # If ONNXRT_DUMP_PATH="dumped/dumped_model_"
1018                 # The first file name will be 'dumped/dumped_model_0.onnx'.
1019                 # For every dumped model, a text file 'dumped/dumped_model_0.txt'
1020                 # is created as well to contain the string representing the graph_module.
1021                 _dump_onnx_model(onnx_model_bytes, graph_module=graph_module)
1022 
1023             # Initialize a ORT session to execute this ONNX model.
1024             # Note that TorchDynamo assumes all inputs/outputs are on the
1025             # same device, but it's subject to change (very likely with
1026             # dynamic shape support), so we add execution providers
1027             # based on the logic in _select_eps: (explicitly preferred EPs,
1028             # EPs inferred from inputs or graph, and the fallback default EP)/
1029             #
1030             # TODO(wschin): enable external allocators.
1031             # See https://github.com/pytorch/pytorch/issues/106867
1032             onnx_session = onnxruntime.InferenceSession(
1033                 path_or_bytes=onnx_model_bytes,
1034                 sess_options=self._options.ort_session_options,
1035                 providers=self._select_eps(graph_module, *args),
1036             )
1037 
1038             # Cache ORT session. It's reused for the same "graph_module".
1039             # Generate ONNX model and extract its input and output names.
1040             input_names = tuple(input.name for input in onnx_model.graph.input)
1041             output_names = tuple(output.name for output in onnx_model.graph.output)
1042             input_devices = _get_onnx_devices(args)
1043             # Cache devices for inputs and outputs. They are used to invoke
1044             # ORT session. Output devices indicate where (e.g., GPU or CPU)
1045             # to store outputs
1046             if isinstance(prim_outputs, tuple):
1047                 output_devices = _get_onnx_devices(prim_outputs)
1048             else:
1049                 output_devices = _get_onnx_devices((prim_outputs,))
1050 
1051             input_value_infos = tuple(input for input in onnx_model.graph.input)
1052             output_value_infos = tuple(output for output in onnx_model.graph.output)
1053 
1054             execution_info_per_session = OrtExecutionInfoPerSession(
1055                 session=onnx_session,
1056                 input_names=input_names,
1057                 input_value_infos=input_value_infos,
1058                 output_names=output_names,
1059                 output_value_infos=output_value_infos,
1060                 input_devices=input_devices,
1061                 output_devices=output_devices,
1062                 example_outputs=prim_outputs,
1063             )
1064 
1065             self._all_ort_execution_info.cache_session_execution_info(
1066                 graph_module, execution_info_per_session
1067             )
1068 
1069         self.execution_count += 1
1070 
1071         # ORT always returns a tuple of outputs. If the original output is a tensor,
1072         # ORT output's first element must be extracted and returned. Otherwise, type
1073         # mismatch may happen in downstream computation.
1074         is_single_tensor_output = isinstance(prim_outputs, torch.Tensor)
1075         normalized_prim_outputs = (
1076             (prim_outputs,) if is_single_tensor_output else prim_outputs
1077         )
1078         assert isinstance(normalized_prim_outputs, tuple)
1079         assert all(
1080             isinstance(elem, (torch.Tensor, torch.SymInt, int))
1081             for elem in normalized_prim_outputs
1082         )
1083 
1084         _nvtx_range_push("run_onnx_session_with_ortvaluevector")
1085         onnx_outputs = self.run(
1086             onnx_session,
1087             input_names,
1088             args,
1089             input_devices,
1090             output_names,
1091             normalized_prim_outputs,
1092             output_devices,
1093             self._options.preallocate_output,
1094             input_value_infos,
1095             normalized_prim_outputs,
1096         )
1097         _nvtx_range_pop()
1098 
1099         if self._assert_allclose_to_baseline:
1100             # Compute baseline.
1101             baseline_outputs = torch._prims.executor.execute(
1102                 graph_module, *args, executor="aten"
1103             )
1104             normalized_baseline_ouptuts = (
1105                 (baseline_outputs,) if is_single_tensor_output else baseline_outputs
1106             )
1107             # Ensure every output tensor is close to the corresponding baseline.
1108             for onnx_output, baseline_output in zip(
1109                 onnx_outputs, normalized_baseline_ouptuts
1110             ):
1111                 torch.testing.assert_close(onnx_output, baseline_output)
1112         return onnx_outputs[0] if is_single_tensor_output else onnx_outputs

1114     def compile(self, graph_module: torch.fx.GraphModule, args) -> torch.fx.GraphModule:
1115         # Deferred import since CapabilityBasedPartitioner is not decorated with
1116         # @compatibility; importing it at the module level will result in the test
1117         # failing: pytest test/test_fx.py -k test_public_api_surface
1118         # because this module is imported into torch.onnx.
1119         from torch.fx.passes.infra.partitioner import CapabilityBasedPartitioner
1120 
1121         # FX graph based partitioning based on ONNX supported ops.
1122         # Given a graph module
1123         #  GraphModule0
1124         #   node_0
1125         #   node_1
1126         #   node_2
1127         #   node_3
1128         #   node_4
1129         # If only node_2 is not supported by ONNX, this graph module will be partitioned into
1130         #  GraphModule0
1131         #   GraphModule1
1132         #    node_0
1133         #    node_1
1134         #   node_2
1135         #   GraphModule2
1136         #    node_3
1137         #    node_4
1138         # by calling CapabilityBasedPartitioner.partition_and_fuse.
1139         # Then, GraphModule1's and GraphModule2's forward method (GraphModule._wrapped_call)
1140         # will be replaced by OrtBackend._ort_accelerated_call to delegate computation to ORT.
1141         if graph_module in self._partitioner_cache:
1142             partitioned_prim_graph_module = self._partitioner_cache[graph_module]
1143         else:
1144             prim_graph_module = graph_module
1145             partitioner = CapabilityBasedPartitioner(
1146                 prim_graph_module,
1147                 self._supported_ops,
1148                 allows_single_node_partition=True,
1149             )
1150             partitioned_prim_graph_module = partitioner.partition_and_fuse()
1151             self._partitioner_cache[graph_module] = partitioned_prim_graph_module
1152 
1153             # Overriding fused_module's __call__() function with ort_acclerated_call()
1154             # This loop goes through all graph partitions (each of them is an ONNX-representable graph)
1155             # and override their _wrapped_call function with _ort_accelerated_call.
1156             # Inside _ort_accelerated_call, the partition's graph is exported into ONNX and executed by ORT.
1157             for node in partitioned_prim_graph_module.graph.nodes:
1158                 # TODO(wschin): use a better way to identify fused submodule
1159                 # See https://github.com/pytorch/pytorch/issues/106872.
1160                 if node.op == "call_module" and "fused_" in node.name:
1161                     fused_module = getattr(partitioned_prim_graph_module, node.name)
1162                     # self.ort_acclerated_call is responsible for exporting graph to ONNX,
1163                     # creating ORT session, and running ORT session.
1164                     fused_module._wrapped_call = self._ort_acclerated_call
1165 
1166         return partitioned_prim_graph_module
1167 
1168     def __call__(
1169         self, graph_module: torch.fx.GraphModule, args
1170     ) -> torch.fx.GraphModule:
1171         """If ``OrtBackendOptions.use_aot_autograd`` is ``True``, the `auto_autograd` compiler 
1172         will be invoked, wrapping this ``OrtBackend`` instance's ``compile`` method. Otherwise, 
1173         the ``compile`` method is invoked directly."""
1174         if self._options.use_aot_autograd:
1175             from functorch.compile import min_cut_rematerialization_partition
1176             from torch._dynamo.backends.common import aot_autograd
1177 
1178             return aot_autograd(
1179                 fw_compiler=self.compile,
1180                 partition_fn=min_cut_rematerialization_partition,
1181                 decompositions=self._resolved_onnx_exporter_options.decomposition_table,
1182             )(graph_module, args)
1183 
1184         return self.compile(graph_module, args)

1189     @staticmethod
1190     def get_cached_instance_for_options(
1191         options: Optional[Union[OrtBackendOptions, Mapping[str, Any]]] = None,
1192     ) -> "OrtBackend":
1193         """Returns a possibly cached instance of an ``OrtBackend``. If an existing 
1194         backend was created previously through this function with the same options, 
1195         it will be returned. Otherwise a new backend will be created, cached, and 
1196         returned. 
1197  
1198         Note: if ``options`` sets ``ort_session_options``, a new ``OrtBackend`` 
1199         will always be returned, since ``onnxruntime.SessionOptions`` cannot 
1200         participate in caching."""
1201 
1202         def reusable(a: OrtBackendOptions, b: OrtBackendOptions):
1203             if (
1204                 a.preferred_execution_providers != b.preferred_execution_providers
1205                 or a.infer_execution_providers != b.infer_execution_providers
1206                 or a.default_execution_providers != b.default_execution_providers
1207                 or a.preallocate_output != b.preallocate_output
1208                 or a.use_aot_autograd != b.use_aot_autograd
1209                 or a.pre_ort_model_transforms != b.pre_ort_model_transforms
1210             ):
1211                 return False
1212 
1213             # onnxruntime.SessionOptions is a pybind11 object, cannot be pickled,
1214             # and holds too much potential state to reasonably check manually;
1215             # ort_session_options is provided at all, the backend does not participate
1216             # in caching.
1217             if a.ort_session_options is not None or b.ort_session_options is not None:
1218                 return False
1219 
1220             return True
1221 
1222         if not isinstance(options, OrtBackendOptions):
1223             options = OrtBackendOptions(**(options or {}))
1224 
1225         backend = next(
1226             (b for b in OrtBackend.__instance_cache if reusable(b._options, options)),
1227             None,
1228         )
1229 
1230         if backend is None:
1231             assert (
1232                 len(OrtBackend.__instance_cache) < OrtBackend.__instance_cache_max_count
1233             ), (
1234                 f"No more than {OrtBackend.__instance_cache_max_count} instances of "
1235                 f"{OrtBackend} allowed. Please instantiate `{OrtBackend}` explicitly "
1236                 "to pass to `torch.compile`. "
1237                 "See https://github.com/pytorch/pytorch/pull/107973#discussion_r1306144795 "
1238                 "for discussion."
1239             )
1240             OrtBackend.__instance_cache.append(backend := OrtBackend(options))
1241 
1242         return backend
1243

1246         OrtBackend.__instance_cache.clear()
1247

1250         return tuple(OrtBackend.__instance_cache)
