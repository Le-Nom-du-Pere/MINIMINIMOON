29   // A storage represents the underlying backing data buffer for a
30   // tensor.  This concept was inherited from the original Torch7
31   // codebase; we'd kind of like to get rid of the concept
32   // (see https://github.com/pytorch/pytorch/issues/14797) but
33   // it's hard work and no one has gotten around to doing it.
34   //
35   // NB: storage is supposed to uniquely own a data pointer; e.g.,
36   // two non-null data pointers alias if and only if they are from
37   // the same storage.  Technically you can violate this invariant
38   // (e.g., you can create a non-owning StorageImpl with at::from_blob)
39   // but a lot of things won't work correctly, including:
40   //
41   // - An ordinary deleter on such a storage is wrong, because normal deleters
42   //   assume unique ownership, but if you have two storages at the same data,
43   //   that implies there is some sort of shared ownership. So your deleter would
44   //   have to actually be internally doing some sort of refcount thing
45   // - Deepcopy in Python side relies on storage equality and not data pointer
46   //   equality; so if there are two separate storages pointing to the same data,
47   //   the data will actually get duplicated in that case (one data ptr before,
48   //   two data ptrs after)
49   // - Version counts won't work correctly, because we do all VC tracking at the
50   //   level of storages (unless you explicitly disconnect the VC with detach);
51   //   mutation because data pointers are the same are totally untracked
52   struct C10_API StorageImpl : public c10::intrusive_ptr_target {
53    public:
54     struct use_byte_size_t {};
55   
56     StorageImpl(
57         use_byte_size_t /*use_byte_size*/,
58         SymInt size_bytes,
59         at::DataPtr data_ptr,
60         at::Allocator* allocator,
61         bool resizable)
62         : data_ptr_(std::move(data_ptr)),
63           size_bytes_(std::move(size_bytes)),
64           size_bytes_is_heap_allocated_(size_bytes_.is_heap_allocated()),
65           resizable_(resizable),
66           received_cuda_(false),
67           allocator_(allocator) {
68       if (resizable) {
69         TORCH_INTERNAL_ASSERT(
70             allocator_, "For resizable storage, allocator must be provided");
71       }
72       refresh_has_data_ptr_check();
73     }
74   
75     StorageImpl(
76         use_byte_size_t /*use_byte_size*/,
77         const SymInt& size_bytes,
78         at::Allocator* allocator,
79         bool resizable)
80         : StorageImpl(
81               use_byte_size_t(),
82               size_bytes,
83               size_bytes.is_heap_allocated()
84                   ? allocator->allocate(0)
85                   : allocator->allocate(size_bytes.as_int_unchecked()),
86               allocator,
87               resizable) {}
88   
89     StorageImpl& operator=(StorageImpl&& other) = delete;
90     StorageImpl& operator=(const StorageImpl&) = delete;
91     StorageImpl() = delete;
92     StorageImpl(StorageImpl&& other) = delete;
93     StorageImpl(const StorageImpl&) = delete;
94     ~StorageImpl() override = default;
95   
96     void reset() {
97       data_ptr_.clear();
98       size_bytes_ = 0;
99       size_bytes_is_heap_allocated_ = false;
100    }
101  
102    // Destructor doesn't call release_resources because it's
103    // unnecessary; don't forget to change that if needed!
104    void release_resources() override {
105      data_ptr_.clear();
106    }
107  
108    size_t nbytes() const {
109      // OK to do this instead of maybe_as_int as nbytes is guaranteed positive
110      TORCH_CHECK(!size_bytes_is_heap_allocated_);
111      return size_bytes_.as_int_unchecked();
112    }
113  
114    SymInt sym_nbytes() const {
115      return size_bytes_;
116    }
117  
118    // TODO: remove later
119    void set_nbytes(size_t size_bytes) {
120      size_bytes_ = static_cast<int64_t>(size_bytes);
121      size_bytes_is_heap_allocated_ = false;
122    }
123  
124    void unsafe_set_nbytes(size_t size_bytes) {
125      TORCH_INTERNAL_ASSERT_DEBUG_ONLY(!size_bytes_is_heap_allocated_);
126      size_bytes_.unsafe_set_data(size_bytes);
127    }
128  
129    void set_nbytes(c10::SymInt size_bytes) {
130      size_bytes_ = std::move(size_bytes);
131    }
132  
133    bool resizable() const {
134      return resizable_;
135    }
136  
137    const at::DataPtr& data_ptr() const {
138      if (C10_UNLIKELY(throw_on_immutable_data_ptr_)) {
139        throw_data_ptr_access_error();
140      }
141      return data_ptr_;
142    }
143  
144    at::DataPtr& mutable_data_ptr() {
145      if (C10_UNLIKELY(has_mutable_data_ptr_check_)) {
146        if (throw_on_immutable_data_ptr_) {
147          throw_data_ptr_access_error();
148        }
149        if (throw_on_mutable_data_ptr_) {
150          throwNullDataPtrError();
151        }
152        if (warn_deprecated_on_mutable_data_ptr_) {
153          warnDeprecatedDataPtr();
154        }
155        maybe_materialize_cow();
156      }
157      return data_ptr_;
158    }
159  
160    // Returns the data_ptr. Bypasses all checks.
161    at::DataPtr& _mutable_data_ptr_no_checks() {
162      return data_ptr_;
163    }
164  
165    // Returns the previous data_ptr
166    at::DataPtr set_data_ptr(at::DataPtr&& data_ptr) {
167      // We need to materialize the old COW DataPtr because it is
168      // being returned as mutable.
169      maybe_materialize_cow();
170      return set_data_ptr_no_materialize_cow(std::move(data_ptr));
171    }
172  
173    void set_data_ptr_noswap(at::DataPtr&& data_ptr) {
174      data_ptr_ = std::move(data_ptr);
175      refresh_has_data_ptr_check();
176    }
177  
178    const void* data() const {
179      if (C10_UNLIKELY(throw_on_immutable_data_ptr_)) {
180        throw_data_ptr_access_error();
181      }
182      return data_ptr_.get();
183    }
184  
185    void* mutable_data() {
186      if (C10_UNLIKELY(has_mutable_data_ptr_check_)) {
187        if (throw_on_immutable_data_ptr_) {
188          throw_data_ptr_access_error();
189        }
190        if (throw_on_mutable_data_ptr_) {
191          throwNullDataPtrError();
192        }
193        if (warn_deprecated_on_mutable_data_ptr_) {
194          warnDeprecatedDataPtr();
195        }
196        maybe_materialize_cow();
197      }
198      return data_ptr_.mutable_get();
199    }
200  
201    at::DeviceType device_type() const {
202      return data_ptr_.device().type();
203    }
204  
205    at::Allocator* allocator() {
206      return allocator_;
207    }
208  
209    const at::Allocator* allocator() const {
210      return allocator_;
211    }
212  
213    // You generally shouldn't use this method, but it is occasionally
214    // useful if you want to override how a tensor will be reallocated,
215    // after it was already allocated (and its initial allocator was
216    // set)
217    void set_allocator(at::Allocator* allocator) {
218      allocator_ = allocator;
219    }
220  
221    Device device() const {
222      return data_ptr_.device();
223    }
224  
225    void set_resizable(bool resizable) {
226      if (resizable) {
227        // We need an allocator to be resizable
228        AT_ASSERT(allocator_);
229      }
230      resizable_ = resizable;
231    }

238        size_t size_bytes,

240      UniqueStorageShareExternalPointer(
241          at::DataPtr(src, src, d, data_ptr_.device()), size_bytes);

244    /** 
245     * Can only be called when use_count is 1 
246     */
247    void UniqueStorageShareExternalPointer(
248        at::DataPtr&& data_ptr,
249        size_t size_bytes) {
250      data_ptr_ = std::move(data_ptr);
251      size_bytes_ = static_cast<int64_t>(size_bytes);
252      size_bytes_is_heap_allocated_ = false;
253      allocator_ = nullptr;
254      resizable_ = false;
255    }

261    }
262  
263    bool received_cuda() {
264      return received_cuda_;

268      return &pyobj_slot_;

270  
271    const impl::PyObjectSlot* pyobj_slot() const {
272      return &pyobj_slot_;
273    }

275    StorageExtraMeta& get_extra_meta() {
276      if (!extra_meta_) {
277        extra_meta_ = std::make_unique<StorageExtraMeta>();
278      }
279      return *extra_meta_;
280    }
281  
282    [[noreturn]] void throw_data_ptr_access_error() const;
283  
284    void release_data_and_set_meta_custom_data_ptr_error_msg_(
285        std::optional<std::string> s) {
286      throw_on_immutable_data_ptr_ = true;
287      get_extra_meta().custom_data_ptr_error_msg_ = std::move(s);
288      refresh_has_data_ptr_check();
289    }
290  
291    void set_throw_on_mutable_data_ptr() {
292      throw_on_mutable_data_ptr_ = true;
293      refresh_has_data_ptr_check();
294    }
295  
296    void set_warn_deprecated_on_mutable_data_ptr() {
297      warn_deprecated_on_mutable_data_ptr_ = true;
298      refresh_has_data_ptr_check();
299    }
300  
301   protected:
302    // materialize_cow_storage needs to call set_data_ptr_no_materlize_cow
303    friend void c10::impl::cow::materialize_cow_storage(StorageImpl& storage);
304  
305    // Returns the previous data_ptr. If the old data_ptr was COW,
306    // this avoids materializing it
307    at::DataPtr set_data_ptr_no_materialize_cow(at::DataPtr&& data_ptr) {
308      at::DataPtr old_data_ptr(std::move(data_ptr_));
309      data_ptr_ = std::move(data_ptr);
310      refresh_has_data_ptr_check();
311      return old_data_ptr;
312    }
313  
314   private:
315    void refresh_has_data_ptr_check() {
316      has_mutable_data_ptr_check_ = is_cow() || throw_on_mutable_data_ptr_ ||
317          warn_deprecated_on_mutable_data_ptr_ || throw_on_immutable_data_ptr_;
318    }
319  
320    inline bool is_cow() const {
321      return c10::impl::cow::is_cow_data_ptr(data_ptr_);
322    }
323  
324    // Triggers a copy if this is a copy-on-write tensor.
325    void maybe_materialize_cow() {
326      if (is_cow()) {
327        impl::cow::materialize_cow_storage(*this);
328      }
329    }
330  
331    DataPtr data_ptr_;
332    SymInt size_bytes_;
333    bool size_bytes_is_heap_allocated_;
334    bool resizable_;
335    // Identifies that Storage was received from another process and doesn't have
336    // local to process cuda memory allocation
337    bool received_cuda_;
338    // All special checks in data/data_ptr calls are guarded behind this single
339    // boolean. This is for performance: .data/.data_ptr calls are commonly in the
340    // hot-path.
341    bool has_mutable_data_ptr_check_ = false;
342    // If we should throw when mutable_data_ptr() or mutable_data() is called.
343    bool throw_on_mutable_data_ptr_ = false;
344    // If we should throw when data_ptr() or data() is called.
345    bool throw_on_immutable_data_ptr_ = false;
346    // If we warn when mutable_data_ptr() or mutable_data() is called.
347    bool warn_deprecated_on_mutable_data_ptr_ = false;
348    Allocator* allocator_;
349    impl::PyObjectSlot pyobj_slot_;
350    std::unique_ptr<StorageExtraMeta> extra_meta_ = nullptr;
351  };
