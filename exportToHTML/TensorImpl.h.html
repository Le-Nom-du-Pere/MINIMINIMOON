75    * Return product of all dimensions starting from k 
76    */
77   inline int64_t size_from_dim_(int k, IntArrayRef dims) {
78     int64_t r = 1;
79     for (const auto i : c10::irange(k, dims.size())) {
80       r *= dims[i];
81     }
82     return r;
83   }
84   
85   // Product of all dims up to k (not including dims[k])
86   inline int64_t size_to_dim_(int k, IntArrayRef dims) {
87     TORCH_CHECK(k >= 0 && static_cast<size_t>(k) <= dims.size());
88     int64_t r = 1;
89     for (const auto i : c10::irange(k)) {
90       r *= dims[i];
91     }
92     return r;
93   }
94   
95   // Product of all dims between k and l (not including dims[k] and dims[l])
96   inline int64_t size_between_dim_(int k, int l, IntArrayRef dims) {
97     TORCH_CHECK((unsigned)l < dims.size() && (unsigned)k < dims.size());
98     int64_t r = 1;
99     if (k < l) {
100      for (int i = k + 1; i < l; ++i) {
101        r *= dims[i];
102      }
103    } else {
104      for (int i = l + 1; i < k; ++i) {
105        r *= dims[i];
106      }
107    }
108    return r;
109  }

286  // NOTE [ Version Counter Sharing ]
287  //
288  // Every Tensor has a version counter. Version counters are incremented whenever
289  // the data or size of a tensor changes through in-place Variable operations.
290  // Version counters are used to detect modifications to saved variables which
291  // would result in incorrect gradient calculations. Version counters may be
292  // shared between Variables:
293  //
294  // 1. A view shares the version counter of the base Variable,
295  // 2. `x.detach()` shares the version counter of `x`,
296  // 3. Unpacked saved variables share the version counter of the source.
297  //
298  // Version counters are not shared in these scenarios:
299  //
300  // 1. When we replace a `Variable`'s underlying `Tensor` by calling
301  // `set_data(...)`,
302  // 2. `x.data` does not share the version counter of `x`. (See discussion at
303  // https://github.com/pytorch/pytorch/issues/5396)
304  //
305  // Question: Why do we put the version counter in TensorImpl instead of
306  // AutogradMeta?
307  //
308  // Answer: After the Variable/Tensor merge, a tensor will not have AutogradMeta
309  // when its `requires_grad_` is false, but when we use this tensor in the
310  // forward pass of a function that requires saving this tensor for backward, we
311  // need to keep track of this tensor's version to make sure it's always valid in
312  // the autograd graph.
313  //
314  // To achieve this goal, we put the version counter in TensorImpl instead of
315  // AutogradMeta, and have it always be available. This allows us to have the
316  // optimization of not carrying AutogradMeta when a tensor doesn't require
317  // gradient.
318  //
319  // A hypothetical alternative way to achieve this goal is to initialize
320  // AutogradMeta and create the version counter for the non-requires-grad tensor
321  // only when it's saved for backward. However, since saving a tensor for
322  // backward happens in the forward pass, and our invariant is that forward pass
323  // needs to be thread-safe, lazy-initializing AutogradMeta when saving a tensor
324  // can introduce race conditions when we are running the forward pass in
325  // multi-thread scenarios, thus making the forward pass not thread-safe anymore,
326  // which breaks the invariant.
327  struct C10_API VariableVersion {
328   private:
329    struct VersionCounter : intrusive_ptr_target {
330      VersionCounter(uint32_t version) : version_(version) {}
331      std::atomic<uint32_t> version_;
332    };
333    c10::intrusive_ptr<VersionCounter> version_counter_;
334  
335   public:
336    // Note [Disabled VariableVersion]
337    // VariableVersion struct has an intrusive_ptr pointing VersionCounter struct
338    // with an atomic variable. Thus `VariableVersion(/*version=*/0)` is not as
339    // cheap as we expected. In some cases constructing a VariableVersion with
340    // version 0 is not necessary so we add a cheap constructor which
341    // doesn't allocate the intrusive_ptr.
342    // Example use cases are:
343    //  - Inference tensors don't track version counter, so they'll just always
344    //    have disabled VariableVersion.
345    //  - In SavedVariable class we override version_counter_ inside its
346    //  constructor
347    //    so that we can use the cheap constructor there.
348    enum Disabled { DISABLED };
349    // It's okay to return true even for inference tensor which
350    // doesn't have version counter enabled.
351    // We want to be permissive here since in many cases (e.g. make_variable)
352    // we can std::move a TensorImpl if there's no other uses which saves us
353    // an additional TensorImpl allocation.
354    bool unique() const {
355      return version_counter_ ? 1 == version_counter_.use_count() : true;
356    }
357    // NOTE: As of C++11 and 14, default-constructing a std::atomic variable
358    // leaves it in a persistently undefined state. See
359    // https://cplusplus.github.io/LWG/issue2334.
360    VariableVersion(uint32_t version)
361        : version_counter_(c10::make_intrusive<VersionCounter>(version)) {}
362    VariableVersion(Disabled = DISABLED) {}
363  
364    bool enabled() const {
365      return version_counter_;
366    }
367  
368    // Note [Inplace update inference tensor]
369    // 1. Inplace update to inference tensor is forbidden in normal mode.
370    //   For example:
371    //     inference_tensor.copy_(normal_tensor_requires_grad)
372    //   This inplace makes inference_tensor have requires_grad=True and
373    //   have a grad_fn.  This is bad because views of `inference_tensor`
374    //   created in InferenceMode won't be able to know the grad_fn since
375    //   their ViewMeta were not recorded. To match NoGradMode behavior
376    //   that "inplace update to a view created in NoGradMode raise an error",
377    //   we just ban inplace update to inference tensor since we can't tell
378    //   if an inference tensor is a view created in InferenceMode.
379    //
380    //   Note that views of normal tensor created in InferenceMode has proper
381    //   ViewMeta so that they're aware of the grad_fn correctly.
382    //
383    // 2. Inplace update to inference tensor in inference tensor doesn't bump
384    //    version counter.
385    //    * It either doesn't call bump() by skipping ADInplaceOrView kernel,
386    //      - e.g. inference_tensor.add_(1)
387    //    * or bump() is a no-op for inference tensor.
388    //      - e.g. inference_tensor.add_(normal_tensor)
389    void bump() {
390      // TODO: Replace the link to the documentation once it's available.
391      TORCH_CHECK(
392          version_counter_ || InferenceMode::is_enabled(),
393          "Inplace update to inference tensor outside InferenceMode is not allowed."
394          "You can make a clone to get a normal tensor before doing inplace update."
395          "See https://github.com/pytorch/rfcs/pull/17 for more details.");
396      if (version_counter_) {
397        ++version_counter_->version_;
398      }
399    }
400  
401    void set_version(int64_t i) {
402      TORCH_CHECK(
403          version_counter_,
404          "Tried to call torch.autograd._unsafe_set_version() on a tensor "
405          "that does not have a version counter. Was it created in inference mode?");
406      TORCH_CHECK(i >= 0, "Cannot set a version_counter to a value below 0: ", i);
407      version_counter_->version_ = i;
408    }
409  
410    // Inference tensor doesn't have version counter so it shouldn't be
411    // accessed.
412    uint32_t current_version() const {
413      TORCH_CHECK(
414          version_counter_, "Inference tensors do not track version counter.");
415      return version_counter_->version_;
416    }
417  };

438  /** 
439   * The low-level representation of a tensor, which contains a pointer 
440   * to a storage (which contains the actual data) and metadata (e.g., sizes and 
441   * strides) describing this particular view of the data as a tensor. 
442   * 
443   * Some basic characteristics about our in-memory representation of 
444   * tensors: 
445   * 
446   *  - It contains a pointer to a storage struct (Storage/StorageImpl) 
447   *    which contains the pointer to the actual data and records the 
448   *    data type and device of the view.  This allows multiple tensors 
449   *    to alias the same underlying data, which allows to efficiently 
450   *    implement differing *views* on a tensor. 
451   * 
452   *  - The tensor struct itself records view-specific metadata about 
453   *    the tensor, e.g., sizes, strides and offset into storage. 
454   *    Each view of a storage can have a different size or offset. 
455   * 
456   *  - This class is intrusively refcounted.  It is refcounted so that 
457   *    we can support prompt deallocation of large tensors; it is 
458   *    intrusively refcounted so that we can still perform reference 
459   *    counted operations on raw pointers, which is often more convenient 
460   *    when passing tensors across language boundaries. 
461   * 
462   *  - For backwards-compatibility reasons, a tensor may be in an 
463   *    uninitialized state.  A tensor may be uninitialized in the following 
464   *    two ways: 
465   * 
466   *      - A tensor may be DTYPE UNINITIALIZED.  A tensor of this 
467   *        form has an uninitialized dtype.  This situation most 
468   *        frequently arises when a user writes Tensor x(CPU).  The dtype 
469   *        is subsequently initialized when mutable_data<T>() is 
470   *        invoked for the first time. 
471   * 
472   *      - A tensor may be STORAGE UNINITIALIZED.  A tensor of this form 
473   *        has non-zero size, but has a storage with a null data pointer. 
474   *        This situation most frequently arises when a user calls 
475   *        Resize() or FreeMemory().  This is because Caffe2 historically 
476   *        does lazy allocation: allocation of data doesn't occur until 
477   *        mutable_data<T>() is invoked.  A tensor with zero size is 
478   *        always storage initialized, because no allocation is necessary 
479   *        in this case. 
480   * 
481   *    All combinations of these two uninitialized states are possible. 
482   *    Consider the following transcript in idiomatic Caffe2 API: 
483   * 
484   *      Tensor x(CPU); // x is storage-initialized, dtype-UNINITIALIZED 
485   *      x.Resize(4); // x is storage-UNINITIALIZED, dtype-UNINITIALIZED 
486   *      x.mutable_data<float>(); // x is storage-initialized, dtype-initialized 
487   *      x.FreeMemory(); // x is storage-UNINITIALIZED, dtype-initialized. 
488   * 
489   *    All other fields on tensor are always initialized.  In particular, 
490   *    size is always valid. (Historically, a tensor declared as Tensor x(CPU) 
491   *    also had uninitialized size, encoded as numel == -1, but we have now 
492   *    decided to default to zero size, resulting in numel == 0). 
493   * 
494   *    Uninitialized storages MUST be uniquely owned, to keep our model 
495   *    simple.  Thus, we will reject operations which could cause an 
496   *    uninitialized storage to become shared (or a shared storage to 
497   *    become uninitialized, e.g., from FreeMemory). 
498   * 
499   *    In practice, tensors which are storage-UNINITIALIZED and 
500   *    dtype-UNINITIALIZED are *extremely* ephemeral: essentially, 
501   *    after you do a Resize(), you basically always call mutable_data() 
502   *    immediately afterwards.  Most functions are not designed to 
503   *    work if given a storage-UNINITIALIZED, dtype-UNINITIALIZED tensor. 
504   * 
505   *    We intend to eliminate all uninitialized states, so that every 
506   *    tensor is fully initialized in all fields.  Please do not write new code 
507   *    that depends on these uninitialized states. 
508   */

596    // NOTE: The general recipe for customizable methods is that the fastpath
597    // function (e.g., sizes()) does an unlikely policy test, and if doesn't
598    // trigger, it does the fast path implementation with no checks and going
599    // directly to on-TensorImpl fields.  In particular, you never need to
600    // check ExtraMeta if the policy doesn't trigger, as non-trivial ExtraMeta
601    // implies the policy will always match.
602    //
603    // The default implementations of methods are "safe": they do extra tests
604    // to make sure the internal state is consistent no matter if you are
605    // doing symbolic shapes or not.  If you don't want the tests, directly
606    // override the custom method (e.g., custom_sizes()) to do your preferred
607    // behavior.

913     * if necessary. 
914     * 
915     * NOTE: if you know wrapping is unnecessary, do sizes()[d] instead; it will 
916     * be faster 
917     */
918    int64_t stride(int64_t d) const {
919      d = maybe_wrap_dim(d, dim(), false);
920      if (C10_UNLIKELY(matches_policy(SizesStridesPolicy::CustomStrides))) {
921        // TODO: provide stride_custom, symmetrically with size_custom.
922        // There is presently no user for it; only NestedTensor is using
923        // size_custom overrideability
924        return strides_custom()[d]; // unchecked (maybe_wrap_dim enforces bounds)

926      // Intentionally don't call default, which also handles symbolic
927      return sizes_and_strides_.stride_at_unchecked(d);

930    enum class SizesStridesPolicy : uint8_t {
931      // Default behavior, e.g., dense tensor.
932      //
933      // Can override: nothing
934      Default = 0,
935      // Customizable strides behavior, e.g., sparse tensor,
936      // mkldnn tensor.
937      //
938      // Can override: strides(), is_contiguous()
939      CustomStrides = 1,
940      // Customizable sizes behavior, e.g., nested tensor
941      //
942      // Can override: strides(), is_contiguous(), sizes(), dim(), numel()
943      CustomSizes = 2
944    };

948      return sizes_strides_policy_ >= static_cast<uint8_t>(policy);
949    }

952      return custom_sizes_strides_ >= static_cast<uint8_t>(policy);

956      auto r = python_custom_sizes_strides_ >= static_cast<uint8_t>(policy);

1841   /** 
1842    * Set the sizes and strides of a tensor. 
1843    * 
1844    * WARNING: This function does not check if the requested 
1845    * sizes/strides are in bounds for the storage that is allocated; 
1846    * this is the responsibility of the caller 
1847    */
1848   void set_sizes_and_strides(
1849       IntArrayRef new_size,
1850       IntArrayRef new_stride,
1851       std::optional<int64_t> storage_offset = std::nullopt) {
1852     TORCH_CHECK(
1853         allow_tensor_metadata_change(),
1854         "set_sizes_and_strides ",
1855         err_msg_tensor_metadata_change_not_allowed);
1856     TORCH_CHECK(
1857         !has_symbolic_sizes_strides_,
1858         "set_sizes_and_strides() called on tensor with symbolic shape")
1859     TORCH_CHECK(
1860         new_size.size() == new_stride.size(),
1861         "dimensionality of sizes (",
1862         new_size.size(),
1863         ") must match dimensionality of strides (",
1864         new_stride.size(),
1865         ")");
1866     const auto new_dim = new_size.size();
1867     bool overflowed = false;
1868     sizes_and_strides_.set_sizes(new_size);
1869 
1870     if (new_dim > 0) {
1871       for (size_t dim = new_dim - 1;; dim--) {
1872         if (new_stride[dim] >= 0) {
1873           sizes_and_strides_.stride_at_unchecked(dim) = new_stride[dim];
1874         } else {
1875           // XXX: This behavior is surprising and may need to be removed to
1876           // support negative strides. Some pytorch functions rely on it:
1877           // for example, torch.cat (run TestTorch.test_cat_empty).
1878           if (dim == new_dim - 1) {
1879             sizes_and_strides_.stride_at_unchecked(dim) = 1;
1880           } else {
1881             // Keep stride monotonically increasing to match NumPy.
1882             overflowed |= c10::mul_overflows(
1883                 sizes_and_strides_.stride_at_unchecked(dim + 1),
1884                 std::max<int64_t>(
1885                     sizes_and_strides_.size_at_unchecked(dim + 1), 1),
1886                 std::addressof(sizes_and_strides_.stride_at_unchecked(dim)));
1887           }
1888         }
1889         if (dim == 0)
1890           break;
1891       }
1892       TORCH_CHECK(!overflowed, "Stride calculation overflowed");
1893     }
1894 
1895     refresh_numel();
1896     refresh_contiguous();
1897 
1898     if (storage_offset.has_value()) {
1899       storage_offset_ = *storage_offset;
1900     }
1901   }

1995   // NOTE [ TensorImpl Shallow-Copying ]
1996   //
1997   // TensorImpl shallow-copying is used when we want to have two Variables share
1998   // the same tensor metadata (e.g. sizes / strides / storage pointer /
1999   // storage_offset), but each with a different autograd history. Example call
2000   // sites:
2001   //
2002   // 1. `var_detached = var.detach()` uses `shallow_copy_and_detach()` to create
2003   // `var_detached` that shares the same tensor metadata with `var`, but with a
2004   // completely new autograd history.
2005   // 2. `var.set_data(tensor)` uses `shallow_copy_from()` to copy tensor
2006   // metadata from `tensor` into `var`, while keeping `var`'s original
2007   // AutogradMeta.
2008   //
2009   // Functions that shallow-copy a TensorImpl (such as
2010   // `shallow_copy_and_detach()` / `shallow_copy_from()` /
2011   // `copy_tensor_metadata()`) copy the tensor metadata fields (e.g. sizes /
2012   // strides / storage pointer / storage_offset) by value. However, the
2013   // following fields are not copied:
2014   //
2015   // 1. the AutogradMeta pointer, because it is unique for each Variable.
2016   // 2. the version counter, because the destination TensorImpl's version
2017   // counter is either set to the passed-in `version_counter` (in
2018   // `shallow_copy_and_detach()` and `copy_tensor_metadata()`), or it is kept
2019   // intact (in `shallow_copy_from()`). See NOTE [ Version Counter Sharing ] for
2020   // details.
2021   //
2022   // In `shallow_copy_and_detach()` and `copy_tensor_metadata()`, the passed-in
2023   // `allow_tensor_metadata_change` determines whether the TensorImpl
2024   // shallow-copy allows changes to its metadata (e.g. sizes / strides / storage
2025   // / storage_offset). See NOTE [ Metadata Change for a Detached Tensor ] for
2026   // details.
2027   //
2028   // In `shallow_copy_from()`, we don't check the destination TensorImpl's
2029   // `allow_tensor_metadata_change_`, because `shallow_copy_from()` is used for
2030   // implementing functions such as `var.set_data(tensor)`, which changes
2031   // `var`'s tensor metadata and expects its `allow_tensor_metadata_change_` to
2032   // be ignored.
2033 
2034   /** 
2035    * One TensorImpl can be copied to another TensorImpl if they have the same 
2036    * DispatchKeySet. The only two special cases (for legacy reason) are: 
2037    * CPU is compatible with CUDA and SparseCPU is 
2038    * compatible with SparseCUDA. 
2039    */
2040   inline bool has_compatible_shallow_copy_type(DispatchKeySet from) {
2041     auto is_dense = [](DispatchKeySet ts) {
2042       constexpr auto dense_backends = DispatchKeySet(
2043           {BackendComponent::CPUBit,
2044            BackendComponent::CUDABit,
2045            BackendComponent::MPSBit,
2046            BackendComponent::HIPBit,
2047            BackendComponent::XPUBit,
2048            BackendComponent::HPUBit,
2049            BackendComponent::MTIABit});
2050       constexpr auto dense_k = DispatchKeySet(DispatchKey::Dense);
2051       return ts.has_any(dense_k) && ts.has_any(dense_backends);
2052     };
2053     auto is_sparse = [](DispatchKeySet ts) {
2054       constexpr auto sparse_backends = DispatchKeySet(
2055           {BackendComponent::CPUBit,
2056            BackendComponent::CUDABit,
2057            BackendComponent::HIPBit,
2058            BackendComponent::XPUBit});
2059       constexpr auto sparse_k = DispatchKeySet(DispatchKey::Sparse);
2060       return ts.has_any(sparse_k) && ts.has_any(sparse_backends);
2061     };
2062     auto is_sparse_compressed = [](DispatchKeySet ts) {
2063       constexpr auto sparse_compressed_k =
2064           DispatchKeySet(DispatchKey::SparseCsr);
2065       return ts.has_any(sparse_compressed_k);
2066     };
2067     return (key_set_ == from) || (is_dense(key_set_) && is_dense(from)) ||
2068         (is_sparse(key_set_) && is_sparse(from)) ||
2069         (is_sparse_compressed(key_set_) && is_sparse_compressed(from));
2070     ;
2071   }

2251   /** 
2252    * Returns a mutable raw pointer of the underlying storage. Since we will need 
2253    * to know the type of the data for allocation, a TypeMeta object is passed in 
2254    * to specify the necessary information. This is conceptually equivalent of 
2255    * calling mutable_data<T>() where the TypeMeta parameter meta is derived from 
2256    * the type T. This function differs from mutable_data<T>() in the sense that 
2257    * the type T can be specified during runtime via the TypeMeta object. 
2258    * 
2259    * If the existing data does not match the desired type, it will be deleted 
2260    * and a new storage will be created. 
2261    */
2262   inline void* raw_mutable_data(const caffe2::TypeMeta& meta) {
2263     // For 0-size tensors it's fine to return any pointer (including nullptr)
2264     if (data_type_ == meta && storage_initialized()) {
2265       return static_cast<void*>(
2266           static_cast<char*>(storage_.mutable_data()) +
2267           storage_offset_ * meta.itemsize());
2268     } else {
2269       bool had_special_dtor = data_type_.placementDelete() != nullptr;
2270       storage_offset_ = 0;
2271       data_type_ = meta;
2272       // NB: device is not changed
2273 
2274       // We can reuse the existing buffer if the current data does not have
2275       // a special destructor and the new data doesn't have a special
2276       // constructor.
2277       if (numel_ == 0 ||
2278           (meta.placementNew() == nullptr && !had_special_dtor &&
2279            (storage_.nbytes() >= (numel_ * data_type_.itemsize())))) {
2280         TORCH_INTERNAL_ASSERT(
2281             storage_offset_ == 0); // because we just reallocated
2282         return storage_.mutable_data();
2283       }
2284       Allocator* allocator = storage_.allocator();
2285       // Storage might have nullptr allocator in rare cases, for example, if
2286       // an external memory segment has been wrapped with Tensor and we don't
2287       // know how to reallocate it. However, in order to preserve legacy C2
2288       // behavior, we allow reallocating the memory using default allocator.
2289       if (allocator == nullptr) {
2290         allocator = GetAllocator(storage_.device_type());
2291       }
2292       if (meta.placementNew()) {
2293         // For types that need placement new, we will call it, as well as
2294         // making sure that when the data is freed, it calls the right
2295         // destruction procedure.
2296         auto size = numel_;
2297         auto dtor = data_type_.placementDelete();
2298         auto data_ptr = allocator->allocate(numel_ * data_type_.itemsize());
2299         storage_.set_data_ptr_noswap(PlacementDeleteContext::makeDataPtr(
2300             std::move(data_ptr), dtor, size, storage_.device()));
2301         data_type_.placementNew()(storage_.mutable_data(), numel_);
2302       } else {
2303         // For fundamental type, new and delete is easier.
2304         storage_.set_data_ptr_noswap(
2305             allocator->allocate(numel_ * data_type_.itemsize()));
2306       }
2307       storage_.set_nbytes(numel_ * data_type_.itemsize());
2308       TORCH_INTERNAL_ASSERT(
2309           storage_offset_ == 0); // because we just reallocated
2310       device_opt_ = storage_.device();
2311       return storage_.mutable_data();
2312     }
2313   }

2372   /** 
2373    * Set the strides of the tensor to match memory_format 
2374    * 
2375    * WARNING: This function doesn't rearrange data and assumes tensor is a 
2376    * memory contiguous 
2377    */
2378   void empty_tensor_restride(MemoryFormat memory_format) {
2379     if (has_symbolic_sizes_strides_) {
2380       empty_tensor_restride_symint(memory_format);
2381       return;
2382     }
2383 #ifdef DEBUG
2384     TORCH_INTERNAL_ASSERT(
2385         compute_numel() == numel_,
2386         "If you are seeing this error, that means empty_tensor_restride was "
2387         "called before setting correct numel");
2388 #endif
2389     switch (memory_format) {
2390       case MemoryFormat::Contiguous: {
2391         // dim_ is a virtual call, don't repeat it
2392         const auto dim_ = dim();
2393         sizes_and_strides_.resize(dim_);
2394         if (dim_ > 0) {
2395           bool overflowed = false;
2396           const auto last_idx = dim_ - 1;
2397           sizes_and_strides_.stride_at_unchecked(last_idx) = 1;
2398           for (auto i = last_idx - 1; i >= 0; --i) {
2399             overflowed |= c10::mul_overflows(
2400                 sizes_and_strides_.stride_at_unchecked(i + 1),
2401                 std::max<int64_t>(
2402                     sizes_and_strides_.size_at_unchecked(i + 1), 1),
2403                 std::addressof(sizes_and_strides_.stride_at_unchecked(i)));
2404           }
2405           TORCH_CHECK(!overflowed, "Stride calculation overflowed");
2406         }
2407         break;
2408       }
2409       case MemoryFormat::ChannelsLast: {
2410         TORCH_CHECK(
2411             dim() == 4, "required rank 4 tensor to use channels_last format");
2412         set_sizes_and_strides(sizes(), get_channels_last_strides_2d(sizes()));
2413         break;
2414       }
2415       case MemoryFormat::ChannelsLast3d: {
2416         TORCH_CHECK(
2417             dim() == 5,
2418             "required rank 5 tensor to use channels_last_3d format");
2419         set_sizes_and_strides(sizes(), get_channels_last_strides_3d(sizes()));
2420         break;

2467   // well as variadic with Resize(2, 2).  These overloads provide all of the
2468   // supported calling configurations, while being overloads (and not templates)
2469   // so that implicit conversions still work.
2470   //
2471   // SetDims on ArrayRef is internally implemented as a template, so we can
2472   // handle both ArrayRefs of different types (there are some uses of
2473   // Resize in Caffe2 which pass in int, not int64_t.)

2475   template <
2476       typename T,
2477       typename = typename std::enable_if_t<std::is_integral_v<T>>>

2479     TORCH_CHECK(
2480         !has_symbolic_sizes_strides_,
2481         "SetDims() called on tensor with symbolic shape")
2482

2486     for (const auto i : c10::irange(src.size())) {
2487       new_numel *= src[i];
2488       sizes_and_strides_.size_at_unchecked(i) = src[i];
2489     }
2490     numel_ = new_numel;
2491     empty_tensor_restride(MemoryFormat::Contiguous);
2492     return numel_ != old_numel;
2493   }

2547   /** 
2548    * Compute the number of elements based on the sizes of a 
2549    * tensor. Catches integer overflow that may occur when a tensor 
2550    * using a sparse layout has multiple dimensions with large sizes. 
2551    */
2552   int64_t safe_compute_numel() const {
2553     TORCH_INTERNAL_ASSERT_DEBUG_ONLY(!has_symbolic_sizes_strides_);
2554     uint64_t n = 1;
2555     bool overflows =
2556         c10::safe_multiplies_u64(sizes_and_strides_.sizes_arrayref(), &n);
2557     constexpr auto numel_max = std::min(
2558         static_cast<uint64_t>(std::numeric_limits<int64_t>::max()),
2559         static_cast<uint64_t>(std::numeric_limits<size_t>::max()));
2560 
2561     overflows |= (n > numel_max);
2562     TORCH_CHECK(!overflows, "numel: integer multiplication overflow");
2563     return static_cast<int64_t>(n);
2564   }

2690         _set_is_channels_last_3d(false);

2692             compute_is_non_overlapping_and_dense_dim4());

2694       }
2695       case 5: {
2696         _set_is_contiguous(compute_contiguous());
2697         _set_is_channels_last_contiguous(compute_channels_last_contiguous_2d());
2698         _set_is_channels_last_3d_contiguous(

2707         // is_channels_last_ and is_channels_last_3d_ are suggested
2708         // memory_format. Being channels_last_contiguous doesn't necessarily
