4    """ 
5    This module implements variable tracking for TorchScript objects during Dynamo tracing. 
6     
7    The TorchScriptObjectVariable class provides specialized handling for TorchScript 
8    objects with strong safety guarantees by: 
9    - Enforcing method-call-only access to prevent unsafe attribute manipulation 
10   - Converting graph breaks into hard errors via _raise_hard_error_if_graph_break 
11   - Proper proxy and source tracking for TorchScript method calls 
12   - Integration with higher-order operators for method call handling 
13    
14   Key safety features: 
15   - Strict validation that only method calls are allowed (no direct attribute access) 
16   - Immediate error reporting for potentially unsafe operations 
17   - Proper source tracking for debugging and guard installation 
18   - Safe handling of TorchScript object method calls through torchbind 
19    
20   The module ensures that TorchScript objects are handled safely during tracing 
21   by limiting operations to known-safe patterns and failing fast for unsafe usage. 
22   """
23   
24   import functools
25   
26   import torch
27   
28   from ..exc import unimplemented, UnsafeScriptObjectError, Unsupported
29   from .base import VariableTracker
30   from .user_defined import UserDefinedObjectVariable
31   
32   
33   def _raise_hard_error_if_graph_break(reason):
34       def deco(fn):
35           @functools.wraps(fn)
36           def graph_break_as_hard_error(*args, **kwargs):
37               try:
38                   return fn(*args, **kwargs)
39               except Unsupported as e:
40                   raise UnsafeScriptObjectError(e.msg) from e
41   
42           return graph_break_as_hard_error
43   
44       return deco
45   
46   
47   class TorchScriptObjectVariable(UserDefinedObjectVariable):
48       _fake_script_object_cache: dict[int, "TorchScriptObjectVariable"] = {}
49   
50       @classmethod
51       def is_matching_cls(cls, user_cls: type):
52           return issubclass(user_cls, torch.ScriptObject)
53   
54       @staticmethod
55       def create(proxy, value, **options):
56           return TorchScriptObjectVariable(proxy, value, **options)
57   
58       def __init__(self, proxy, value, source, **kwargs) -> None:
59           super().__init__(value, **kwargs)
60           self.proxy = proxy
61           self.proxy.node.meta["example_value"] = value
62           self.source = source
63   
64       def as_proxy(self):
65           return self.proxy
66   
67       @_raise_hard_error_if_graph_break(
68           "Dynamo cannot safely trace script object due to graph break."
69       )
70       def var_getattr(self, tx, name: str) -> VariableTracker:
71           from torch._higher_order_ops.torchbind import call_torchbind
72   
73           from ..source import AttrSource
74           from .higher_order_ops import TorchHigherOrderOperatorVariable
75   
76           method = getattr(self.value, name, None)
77           if method is None:
78               unimplemented(
79                   f"FakeScriptObject doesn't define method {name}. Did you forget to implement it in the fake class?"
80               )
81   
82           if not callable(method):
83               unimplemented(
84                   "Only method calls on TorchScript objects can be supported safely."
85                   " Please use method calls instead of attribute access."
86               )
87   
88           return TorchHigherOrderOperatorVariable.make(
89               call_torchbind,
90               source=AttrSource(self.source, name),
91               script_obj_var=self,
92               method_name=name,
93           )
94   
95       # We only support method calls on script objects. Interpreting the bytecodes
96       # should go through var_getattr then call_function instead of call_method.
97       #
98       # However, it's possible for call_method to be used directly e.g. for __setattr__.
99       @_raise_hard_error_if_graph_break(
100          "Dynamo cannot safely trace script object due to graph break."
101      )
102      def call_method(self, tx, name, args, kwargs):
103          unimplemented(f"call method {name} on script object is not safe.")
