91   /// Runs the matrix multiplication microkernel followed by a clamp operation.
92   ///
93   /// The pointer of each buffers (LHS, packed RHS and output) needs to be added with offset
94   /// calculated using the following functions:
95   ///
96   ///   * LHS: @ref kai_get_lhs_offset_matmul_clamp_qai8_qai8_qsi8cxp2vlx4sb_1x16vl_sme2_dot.
97   ///   * Packed RHS: @ref kai_get_rhs_packed_offset_matmul_clamp_qai8_qai8_qsi8cxp2vlx4sb_1x16vl_sme2_dot.
98   ///   * Output: @ref kai_get_dst_offset_matmul_clamp_qai8_qai8_qsi8cxp2vlx4sb_1x16vl_sme2_dot.
99   ///
100  /// @param[in]  m Number of output rows to be computed. This must be 1.
101  /// @param[in]  n Number of output columns to be computed.
102  /// @param[in]  k Common dimension of the LHS and RHS operand.
103  /// @param[in]  lhs LHS matrix buffer.
104  /// @param[in]  rhs_packed Packed RHS matrix buffer.
105  /// @param[out] dst Output matrix buffer.
106  /// @param[in]  dst_stride_row Row stride in bytes of the output matrix. Currently, an unused parameter.
107  /// @param[in]  dst_stride_col Column stride in bytes of the output matrix. Currently, an unused parameter.
108  /// @param[in]  params Quantization parameters
109  void kai_run_matmul_clamp_qai8_qai8_qsi8cxp2vlx4sb_1x16vl_sme2_dot(
110      size_t m, size_t n, size_t k, const void* lhs, const void* rhs_packed, void* dst, size_t dst_stride_row,
111      size_t dst_stride_col, const struct kai_matmul_requantize32_params* params);
