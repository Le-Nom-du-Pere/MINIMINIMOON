11   C10_ALWAYS_INLINE void _check_rms_norm_inputs_symint(
12       const Tensor& input,
13       c10::SymIntArrayRef normalized_shape,
14       const Tensor& weight /* optional */) {
15   
16     const int normalized_ndim = normalized_shape.size();
17     TORCH_CHECK(
18         normalized_ndim >= 1,
19         "Expected normalized_shape to be at least 1-dimensional, i.e., ",
20         "containing at least one element, but got normalized_shape = ",
21         normalized_shape);
22     TORCH_CHECK(
23         !weight.defined() || weight.sym_sizes().equals(normalized_shape),
24         "Expected weight to be of same shape as normalized_shape, but got ",
25         "weight of shape ",
26         weight.sym_sizes(),
27         " and normalized_shape = ",
28         normalized_shape);
29   
30     const auto input_ndim = input.dim();
31     const auto input_shape = input.sym_sizes();
32     if (input_ndim < normalized_ndim ||
33         !input_shape.slice(input_ndim - normalized_ndim)
34              .equals(normalized_shape)) {
35       std::stringstream ss;
36       ss << "Given normalized_shape=" << normalized_shape
37          << ", expected input with shape [*";
38       for (auto size : normalized_shape) {
39         ss << ", " << size;
40       }
41       ss << "], but got input of size" << input_shape;
42       TORCH_CHECK(false, ss.str());
43     }
44   }

46   C10_ALWAYS_INLINE std::pair<int64_t, int64_t> _check_layer_norm_inputs(
47       const Tensor& input,
48       IntArrayRef normalized_shape,
49       const Tensor& weight /* optional */,
50       const Tensor& bias /* optional */) {
51   
52     const int normalized_ndim = normalized_shape.size();
53     TORCH_CHECK(
54         normalized_ndim >= 1,
55         "Expected normalized_shape to be at least 1-dimensional, i.e., ",
56         "containing at least one element, but got normalized_shape = ",
57         normalized_shape);
58     TORCH_CHECK(
59         !weight.defined() || weight.sizes().equals(normalized_shape),
60         "Expected weight to be of same shape as normalized_shape, but got ",
61         "weight of shape ",
62         weight.sizes(),
63         " and normalized_shape = ",
64         normalized_shape);
65     TORCH_CHECK(
66         !bias.defined() || bias.sizes().equals(normalized_shape),
67         "Expected bias to be of same shape as normalized_shape, but got ",
68         "bias of shape ",
69         bias.sizes(),
70         " and normalized_shape = ",
71         normalized_shape);
72   
73     const auto input_shape = input.sizes();
74     const auto input_ndim = input.dim();
75   
76     if (input_ndim < normalized_ndim ||
77         !input_shape.slice(input_ndim - normalized_ndim)
78              .equals(normalized_shape)) {
79       std::stringstream ss;
80       ss << "Given normalized_shape=" << normalized_shape
81          << ", expected input with shape [*";
82       for (auto size : normalized_shape) {
83         ss << ", " << size;
84       }
85       ss << "], but got input of size" << input_shape;
86       TORCH_CHECK(false, ss.str());
87     }
88   
89     const int axis = input_ndim - normalized_ndim;
90     const int64_t M =
91         c10::multiply_integers(input_shape.cbegin(), input_shape.cbegin() + axis);
92     const int64_t N =
93         c10::multiply_integers(input_shape.cbegin() + axis, input_shape.cend());
94   
95     return std::make_pair(M, N);
96   }
