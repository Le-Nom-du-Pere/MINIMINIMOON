103  def test_rewrite():
104      assert SingularityFunction(x, 4, 5).rewrite(Piecewise) == (
105          Piecewise(((x - 4)**5, x - 4 >= 0), (0, True)))
106      assert SingularityFunction(x, -10, 0).rewrite(Piecewise) == (
107          Piecewise((1, x + 10 >= 0), (0, True)))
108      assert SingularityFunction(x, 2, -1).rewrite(Piecewise) == (
109          Piecewise((oo, Eq(x - 2, 0)), (0, True)))
110      assert SingularityFunction(x, 0, -2).rewrite(Piecewise) == (
111          Piecewise((oo, Eq(x, 0)), (0, True)))
112  
113      n = Symbol('n', nonnegative=True)
114      p = SingularityFunction(x, a, n).rewrite(Piecewise)
115      assert p == (
116          Piecewise(((x - a)**n, x - a >= 0), (0, True)))
117      assert p.subs(x, a).subs(n, 0) == 1
118  
119      expr_in = SingularityFunction(x, 4, 5) + SingularityFunction(x, -3, -1) - SingularityFunction(x, 0, -2)
120      expr_out = (x - 4)**5*Heaviside(x - 4, 1) + DiracDelta(x + 3) - DiracDelta(x, 1)
121      assert expr_in.rewrite(Heaviside) == expr_out
122      assert expr_in.rewrite(DiracDelta) == expr_out
123      assert expr_in.rewrite('HeavisideDiracDelta') == expr_out
124  
125      expr_in = SingularityFunction(x, a, n) + SingularityFunction(x, a, -1) - SingularityFunction(x, a, -2)
126      expr_out = (x - a)**n*Heaviside(x - a, 1) + DiracDelta(x - a) + DiracDelta(a - x, 1)
127      assert expr_in.rewrite(Heaviside) == expr_out
128      assert expr_in.rewrite(DiracDelta) == expr_out
129      assert expr_in.rewrite('HeavisideDiracDelta') == expr_out
