18   def _remove_effect_tokens_from_graph_helper(
19       ep, num_tokens, input_token_names, output_token_names
20   ):
21       inputs_to_lifted_custom_objs = ep.graph_signature.inputs_to_lifted_custom_objs
22   
23       output_node = None
24       with_effect_nodes: list[torch.fx.Node] = []
25   
26       # Output node need to check its args agianst output_token_names (collected from output_spec)
27       # Therefore, we only need to find the top-levele output node
28       output_node = next(reversed(ep.graph_module.graph.find_nodes(op="output")))
29       for module in ep.graph_module.modules():
30           if not isinstance(module, torch.fx.GraphModule):
31               continue
32   
33           for node in module.graph.nodes:
34               if not (node.op == "call_function" and node.target is with_effects):
35                   continue
36   
37               with_effect_nodes.append(node)
38   
39       # Remove tokens from outputs
40       assert output_node is not None
41       output_args = output_node.args[0]
42       assert len(output_args) >= num_tokens
43       out_token_nodes = output_args[:num_tokens]
44       output_node.args = (tuple(output_args[num_tokens:]),)
45       for out_token in out_token_nodes:
46           assert out_token.name in output_token_names
47           out_token.users.clear()
48           ep.graph.erase_node(out_token)
49   
50       # Replace with_effects(token, func, args) with just func(args)
51       for node in reversed(with_effect_nodes):
52           func = node.args[1]
53           assert isinstance(func, (torch._ops.OpOverload, torch._ops.HigherOrderOperator))
54   
55           if func == torch.ops.higher_order.call_torchbind:
56               custom_obj_meta = node.args[2].meta["val"]  # type: ignore[union-attr]
57               assert isinstance(custom_obj_meta, CustomObjArgument)
58               if custom_obj_meta.fake_val:
59                   custom_obj = custom_obj_meta.fake_val
60               elif node.args[2].name in inputs_to_lifted_custom_objs:  # type: ignore[union-attr]
61                   custom_obj = ep.constants[
62                       inputs_to_lifted_custom_objs[node.args[2].name]  # type: ignore[union-attr]
63                   ]
64               else:
65                   raise RuntimeError(f"Unable to find custom obj for node {node}")
66               schema = _get_schema(func, (custom_obj,) + node.args[3:])
67           else:
68               schema = _get_schema(func, node.args[2:])
69   
70           with ep.graph.inserting_before(node):
71               new_node = ep.graph.call_function(func, node.args[2:], node.kwargs)
72           for k, v in node.meta.items():
73               new_node.meta[k] = v
74               if k == "unbacked_bindings":
75                   # Remove the extra layer for effect token
76                   old_bindings = new_node.meta[k]
77                   new_bindings = {
78                       k: path[1:] if path else path for k, path in old_bindings.items()
79                   }
80                   new_node.meta[k] = new_bindings
81   
82           node.replace_all_uses_with(new_node)
83   
84           # Update user getitem nodes
85           for user in list(new_node.users.keys()):
86               assert user.target == operator.getitem
87               # getitem(with_effects, 0) == token
88               if user.args[1] == 0:
89                   ep.graph.erase_node(user)
90   
91           if len(schema.returns) == 1:
92               # If the function has 1 return then it will just directly return the
93               # result -- we don't need a getitem. So we can replace all the
94               # getitem(with_effects, 1) with just the note itself.
95               for user in list(new_node.users.keys()):
96                   assert user.args[1] == 1
97                   user.replace_all_uses_with(new_node)
98   
99               new_node.meta["val"] = node.meta["val"][1]
100          elif len(schema.returns) > 1:
101              # If the function has more than 1 return then since we got rid of
102              # the 1st return value (the token), we need to bump all the other
103              # getitem calls by 1 down
104              for user in list(new_node.users.keys()):
105                  assert user.args[1] >= 1
106                  user.args = (user.args[0], user.args[1] - 1)
107  
108              new_node.meta["val"] = node.meta["val"][1:]
109          else:
110              assert len(schema.returns) == 0
111              assert len(new_node.users) == 0
112              new_node.meta["val"] = None
113  
114          ep.graph.erase_node(node)
115  
116      # Remove tokens from inputs
117      placeholders = [node for node in ep.graph.nodes if node.op == "placeholder"]
118      assert len(placeholders) >= num_tokens
119      inp_token_nodes = placeholders[:num_tokens]
120      for inp_token in inp_token_nodes:
121          assert inp_token.name in input_token_names
122          ep.graph.erase_node(inp_token)
123  
124      ep.graph.eliminate_dead_code()
125

127  def _remove_effect_tokens(ep: ExportedProgram) -> ExportedProgram:
128      """ 
129      Removes the existance of tokens from the exported program, including: 
130      - Removes the input and output tokens 
131      - Replaces with_effects(token, func, args) with just func(args) 
132   
133      This function does an inplace modification on the given ExportedProgram. 
134      """
135      num_tokens: int = 0
136      input_token_names: list[str] = []
137      new_input_specs: list[InputSpec] = []
138      for inp in ep.graph_signature.input_specs:
139          if inp.kind == InputKind.TOKEN:
140              num_tokens += 1
141              assert isinstance(inp.arg, TokenArgument)
142              input_token_names.append(inp.arg.name)
143          else:
144              new_input_specs.append(inp)
145  
146      num_out_tokens: int = 0
147      new_output_specs: list[OutputSpec] = []
148      output_token_names: list[OutputSpec] = []
149      for out in ep.graph_signature.output_specs:
150          if out.kind == OutputKind.TOKEN:
151              num_out_tokens += 1
152              output_token_names.append(out.arg.name)
153          else:
154              new_output_specs.append(out)
155  
156      # Update graph signature
157      ep.graph_signature.input_specs = new_input_specs
158      ep.graph_signature.output_specs = new_output_specs
159  
160      assert num_tokens == num_out_tokens
161  
162      with ep.graph_module._set_replace_hook(ep.graph_signature.get_replace_hook()):
163          _remove_effect_tokens_from_graph_helper(
164              ep, num_tokens, input_token_names, output_token_names
165          )
166  
167      return ep
