85   class TheanoPrinter(Printer):
86       """ Code printer which creates Theano symbolic expression graphs. 
87    
88       Parameters 
89       ========== 
90    
91       cache : dict 
92           Cache dictionary to use. If None (default) will use 
93           the global cache. To create a printer which does not depend on or alter 
94           global state pass an empty dictionary. Note: the dictionary is not 
95           copied on initialization of the printer and will be updated in-place, 
96           so using the same dict object when creating multiple printers or making 
97           multiple calls to :func:`.theano_code` or :func:`.theano_function` means 
98           the cache is shared between all these applications. 
99    
100      Attributes 
101      ========== 
102   
103      cache : dict 
104          A cache of Theano variables which have been created for SymPy 
105          symbol-like objects (e.g. :class:`sympy.core.symbol.Symbol` or 
106          :class:`sympy.matrices.expressions.MatrixSymbol`). This is used to 
107          ensure that all references to a given symbol in an expression (or 
108          multiple expressions) are printed as the same Theano variable, which is 
109          created only once. Symbols are differentiated only by name and type. The 
110          format of the cache's contents should be considered opaque to the user. 
111      """
112      printmethod = "_theano"
113  
114      def __init__(self, *args, **kwargs):
115          self.cache = kwargs.pop('cache', {})
116          super().__init__(*args, **kwargs)
117  
118      def _get_key(self, s, name=None, dtype=None, broadcastable=None):
119          """ Get the cache key for a SymPy object. 
120   
121          Parameters 
122          ========== 
123   
124          s : sympy.core.basic.Basic 
125              SymPy object to get key for. 
126   
127          name : str 
128              Name of object, if it does not have a ``name`` attribute. 
129          """
130  
131          if name is None:
132              name = s.name
133  
134          return (name, type(s), s.args, dtype, broadcastable)
135  
136      def _get_or_create(self, s, name=None, dtype=None, broadcastable=None):
137          """ 
138          Get the Theano variable for a SymPy symbol from the cache, or create it 
139          if it does not exist. 
140          """
141  
142          # Defaults
143          if name is None:
144              name = s.name
145          if dtype is None:
146              dtype = 'floatX'
147          if broadcastable is None:
148              broadcastable = ()
149  
150          key = self._get_key(s, name, dtype=dtype, broadcastable=broadcastable)
151  
152          if key in self.cache:
153              return self.cache[key]
154  
155          value = tt.tensor(name=name, dtype=dtype, broadcastable=broadcastable)
156          self.cache[key] = value
157          return value
158  
159      def _print_Symbol(self, s, **kwargs):
160          dtype = kwargs.get('dtypes', {}).get(s)
161          bc = kwargs.get('broadcastables', {}).get(s)
162          return self._get_or_create(s, dtype=dtype, broadcastable=bc)
163  
164      def _print_AppliedUndef(self, s, **kwargs):
165          name = str(type(s)) + '_' + str(s.args[0])
166          dtype = kwargs.get('dtypes', {}).get(s)
167          bc = kwargs.get('broadcastables', {}).get(s)
168          return self._get_or_create(s, name=name, dtype=dtype, broadcastable=bc)
169  
170      def _print_Basic(self, expr, **kwargs):
171          op = mapping[type(expr)]
172          children = [self._print(arg, **kwargs) for arg in expr.args]
173          return op(*children)
174  
175      def _print_Number(self, n, **kwargs):
176          # Integers already taken care of below, interpret as float
177          return float(n.evalf())
178  
179      def _print_MatrixSymbol(self, X, **kwargs):
180          dtype = kwargs.get('dtypes', {}).get(X)
181          return self._get_or_create(X, dtype=dtype, broadcastable=(None, None))
182  
183      def _print_DenseMatrix(self, X, **kwargs):
184          if not hasattr(tt, 'stacklists'):
185              raise NotImplementedError(
186                 "Matrix translation not yet supported in this version of Theano")
187  
188          return tt.stacklists([
189              [self._print(arg, **kwargs) for arg in L]
190              for L in X.tolist()
191          ])
192  
193      _print_ImmutableMatrix = _print_ImmutableDenseMatrix = _print_DenseMatrix
194  
195      def _print_MatMul(self, expr, **kwargs):
196          children = [self._print(arg, **kwargs) for arg in expr.args]
197          result = children[0]
198          for child in children[1:]:
199              result = tt.dot(result, child)
200          return result
201  
202      def _print_MatPow(self, expr, **kwargs):
203          children = [self._print(arg, **kwargs) for arg in expr.args]
204          result = 1
205          if isinstance(children[1], int) and children[1] > 0:
206              for i in range(children[1]):
207                  result = tt.dot(result, children[0])
208          else:
209              raise NotImplementedError('''Only non-negative integer 
210             powers of matrices can be handled by Theano at the moment''')
211          return result
212  
213      def _print_MatrixSlice(self, expr, **kwargs):
214          parent = self._print(expr.parent, **kwargs)
215          rowslice = self._print(slice(*expr.rowslice), **kwargs)
216          colslice = self._print(slice(*expr.colslice), **kwargs)
217          return parent[rowslice, colslice]
218  
219      def _print_BlockMatrix(self, expr, **kwargs):
220          nrows, ncols = expr.blocks.shape
221          blocks = [[self._print(expr.blocks[r, c], **kwargs)
222                          for c in range(ncols)]
223                          for r in range(nrows)]
224          return tt.join(0, *[tt.join(1, *row) for row in blocks])
225  
226  
227      def _print_slice(self, expr, **kwargs):
228          return slice(*[self._print(i, **kwargs)
229                          if isinstance(i, sympy.Basic) else i
230                          for i in (expr.start, expr.stop, expr.step)])
231  
232      def _print_Pi(self, expr, **kwargs):
233          return math.pi
234  
235      def _print_Exp1(self, expr, **kwargs):
236          return ts.exp(1)
237  
238      def _print_Piecewise(self, expr, **kwargs):
239          import numpy as np
240          e, cond = expr.args[0].args  # First condition and corresponding value
241  
242          # Print conditional expression and value for first condition
243          p_cond = self._print(cond, **kwargs)
244          p_e = self._print(e, **kwargs)
245  
246          # One condition only
247          if len(expr.args) == 1:
248              # Return value if condition else NaN
249              return tt.switch(p_cond, p_e, np.nan)
250  
251          # Return value_1 if condition_1 else evaluate remaining conditions
252          p_remaining = self._print(sympy.Piecewise(*expr.args[1:]), **kwargs)
253          return tt.switch(p_cond, p_e, p_remaining)
254  
255      def _print_Rational(self, expr, **kwargs):
256          return tt.true_div(self._print(expr.p, **kwargs),
257                             self._print(expr.q, **kwargs))
258  
259      def _print_Integer(self, expr, **kwargs):
260          return expr.p
261  
262      def _print_factorial(self, expr, **kwargs):
263          return self._print(sympy.gamma(expr.args[0] + 1), **kwargs)
264  
265      def _print_Derivative(self, deriv, **kwargs):
266          rv = self._print(deriv.expr, **kwargs)
267          for var in deriv.variables:
268              var = self._print(var, **kwargs)
269              rv = tt.Rop(rv, var, tt.ones_like(var))
270          return rv
271  
272      def emptyPrinter(self, expr):
273          return expr

275      def doprint(self, expr, dtypes=None, broadcastables=None):
276          """ Convert a SymPy expression to a Theano graph variable. 
277   
278          The ``dtypes`` and ``broadcastables`` arguments are used to specify the 
279          data type, dimension, and broadcasting behavior of the Theano variables 
280          corresponding to the free symbols in ``expr``. Each is a mapping from 
281          SymPy symbols to the value of the corresponding argument to 
282          ``theano.tensor.Tensor``. 
283   
284          See the corresponding `documentation page`__ for more information on 
285          broadcasting in Theano. 
286   
287          .. __: http://deeplearning.net/software/theano/tutorial/broadcasting.html 
288   
289          Parameters 
290          ========== 
291   
292          expr : sympy.core.expr.Expr 
293              SymPy expression to print. 
294   
295          dtypes : dict 
296              Mapping from SymPy symbols to Theano datatypes to use when creating 
297              new Theano variables for those symbols. Corresponds to the ``dtype`` 
298              argument to ``theano.tensor.Tensor``. Defaults to ``'floatX'`` 
299              for symbols not included in the mapping. 
300   
301          broadcastables : dict 
302              Mapping from SymPy symbols to the value of the ``broadcastable`` 
303              argument to ``theano.tensor.Tensor`` to use when creating Theano 
304              variables for those symbols. Defaults to the empty tuple for symbols 
305              not included in the mapping (resulting in a scalar). 
306   
307          Returns 
308          ======= 
309   
310          theano.gof.graph.Variable 
311              A variable corresponding to the expression's value in a Theano 
312              symbolic expression graph. 
313   
314          """
315          if dtypes is None:
316              dtypes = {}
317          if broadcastables is None:
318              broadcastables = {}
319  
320          return self._print(expr, dtypes=dtypes, broadcastables=broadcastables)
321

326  def theano_code(expr, cache=None, **kwargs):
327      """ 
328      Convert a SymPy expression into a Theano graph variable. 
329   
330      .. deprecated:: 1.8 
331   
332        ``sympy.printing.theanocode`` is deprecated. Theano has been renamed to 
333        Aesara. Use ``sympy.printing.aesaracode`` instead. See 
334        :ref:`theanocode-deprecated` for more information. 
335   
336      Parameters 
337      ========== 
338   
339      expr : sympy.core.expr.Expr 
340          SymPy expression object to convert. 
341   
342      cache : dict 
343          Cached Theano variables (see :class:`TheanoPrinter.cache 
344          <TheanoPrinter>`). Defaults to the module-level global cache. 
345   
346      dtypes : dict 
347          Passed to :meth:`.TheanoPrinter.doprint`. 
348   
349      broadcastables : dict 
350          Passed to :meth:`.TheanoPrinter.doprint`. 
351   
352      Returns 
353      ======= 
354   
355      theano.gof.graph.Variable 
356          A variable corresponding to the expression's value in a Theano symbolic 
357          expression graph. 
358   
359      """
360      sympy_deprecation_warning(
361          """ 
362          sympy.printing.theanocode is deprecated. Theano has been renamed to 
363          Aesara. Use sympy.printing.aesaracode instead.""",
364          deprecated_since_version="1.8",
365      active_deprecations_target='theanocode-deprecated')
366  
367      if not theano:
368          raise ImportError("theano is required for theano_code")
369  
370      if cache is None:
371          cache = global_cache
372  
373      return TheanoPrinter(cache=cache, settings={}).doprint(expr, **kwargs)

376  def dim_handling(inputs, dim=None, dims=None, broadcastables=None):
377      r""" 
378      Get value of ``broadcastables`` argument to :func:`.theano_code` from 
379      keyword arguments to :func:`.theano_function`. 
380   
381      Included for backwards compatibility. 
382   
383      Parameters 
384      ========== 
385   
386      inputs 
387          Sequence of input symbols. 
388   
389      dim : int 
390          Common number of dimensions for all inputs. Overrides other arguments 
391          if given. 
392   
393      dims : dict 
394          Mapping from input symbols to number of dimensions. Overrides 
395          ``broadcastables`` argument if given. 
396   
397      broadcastables : dict 
398          Explicit value of ``broadcastables`` argument to 
399          :meth:`.TheanoPrinter.doprint`. If not None function will return this value unchanged. 
400   
401      Returns 
402      ======= 
403      dict 
404          Dictionary mapping elements of ``inputs`` to their "broadcastable" 
405          values (tuple of ``bool``\ s). 
406      """
407      if dim is not None:
408          return dict.fromkeys(inputs, (False,) * dim)
409  
410      if dims is not None:
411          maxdim = max(dims.values())
412          return {
413              s: (False,) * d + (True,) * (maxdim - d)
414              for s, d in dims.items()
415          }
416  
417      if broadcastables is not None:
418          return broadcastables
419  
420      return {}

423  @doctest_depends_on(modules=('theano',))
424  def theano_function(inputs, outputs, scalar=False, *,
425          dim=None, dims=None, broadcastables=None, **kwargs):
426      """ 
427      Create a Theano function from SymPy expressions. 
428   
429      .. deprecated:: 1.8 
430   
431        ``sympy.printing.theanocode`` is deprecated. Theano has been renamed to 
432        Aesara. Use ``sympy.printing.aesaracode`` instead. See 
433        :ref:`theanocode-deprecated` for more information. 
434   
435      The inputs and outputs are converted to Theano variables using 
436      :func:`.theano_code` and then passed to ``theano.function``. 
437   
438      Parameters 
439      ========== 
440   
441      inputs 
442          Sequence of symbols which constitute the inputs of the function. 
443   
444      outputs 
445          Sequence of expressions which constitute the outputs(s) of the 
446          function. The free symbols of each expression must be a subset of 
447          ``inputs``. 
448   
449      scalar : bool 
450          Convert 0-dimensional arrays in output to scalars. This will return a 
451          Python wrapper function around the Theano function object. 
452   
453      cache : dict 
454          Cached Theano variables (see :class:`TheanoPrinter.cache 
455          <TheanoPrinter>`). Defaults to the module-level global cache. 
456   
457      dtypes : dict 
458          Passed to :meth:`.TheanoPrinter.doprint`. 
459   
460      broadcastables : dict 
461          Passed to :meth:`.TheanoPrinter.doprint`. 
462   
463      dims : dict 
464          Alternative to ``broadcastables`` argument. Mapping from elements of 
465          ``inputs`` to integers indicating the dimension of their associated 
466          arrays/tensors. Overrides ``broadcastables`` argument if given. 
467   
468      dim : int 
469          Another alternative to the ``broadcastables`` argument. Common number of 
470          dimensions to use for all arrays/tensors. 
471          ``theano_function([x, y], [...], dim=2)`` is equivalent to using 
472          ``broadcastables={x: (False, False), y: (False, False)}``. 
473   
474      Returns 
475      ======= 
476      callable 
477          A callable object which takes values of ``inputs`` as positional 
478          arguments and returns an output array for each of the expressions 
479          in ``outputs``. If ``outputs`` is a single expression the function will 
480          return a Numpy array, if it is a list of multiple expressions the 
481          function will return a list of arrays. See description of the ``squeeze`` 
482          argument above for the behavior when a single output is passed in a list. 
483          The returned object will either be an instance of 
484          ``theano.compile.function_module.Function`` or a Python wrapper 
485          function around one. In both cases, the returned value will have a 
486          ``theano_function`` attribute which points to the return value of 
487          ``theano.function``. 
488   
489      Examples 
490      ======== 
491   
492      >>> from sympy.abc import x, y, z 
493      >>> from sympy.printing.theanocode import theano_function 
494   
495      A simple function with one input and one output: 
496   
497      >>> f1 = theano_function([x], [x**2 - 1], scalar=True) 
498      >>> f1(3) 
499      8.0 
500   
501      A function with multiple inputs and one output: 
502   
503      >>> f2 = theano_function([x, y, z], [(x**z + y**z)**(1/z)], scalar=True) 
504      >>> f2(3, 4, 2) 
505      5.0 
506   
507      A function with multiple inputs and multiple outputs: 
508   
509      >>> f3 = theano_function([x, y], [x**2 + y**2, x**2 - y**2], scalar=True) 
510      >>> f3(2, 3) 
511      [13.0, -5.0] 
512   
513      See also 
514      ======== 
515   
516      dim_handling 
517   
518      """
519      sympy_deprecation_warning(
520          """ 
521          sympy.printing.theanocode is deprecated. Theano has been renamed to Aesara. Use sympy.printing.aesaracode instead""",
522          deprecated_since_version="1.8",
523      active_deprecations_target='theanocode-deprecated')
524  
525      if not theano:
526          raise ImportError("theano is required for theano_function")
527  
528      # Pop off non-theano keyword args
529      cache = kwargs.pop('cache', {})
530      dtypes = kwargs.pop('dtypes', {})
531  
532      broadcastables = dim_handling(
533          inputs, dim=dim, dims=dims, broadcastables=broadcastables,
534      )
535  
536      # Print inputs/outputs
537      code = partial(theano_code, cache=cache, dtypes=dtypes,
538                     broadcastables=broadcastables)
539      tinputs = list(map(code, inputs))
540      toutputs = list(map(code, outputs))
541  
542      #fix constant expressions as variables
543      toutputs = [output if isinstance(output, theano.Variable) else tt.as_tensor_variable(output) for output in toutputs]
544  
545      if len(toutputs) == 1:
546          toutputs = toutputs[0]
547  
548      # Compile theano func
549      func = theano.function(tinputs, toutputs, **kwargs)
550  
551      is_0d = [len(o.variable.broadcastable) == 0 for o in func.outputs]
552  
553      # No wrapper required
554      if not scalar or not any(is_0d):
555          func.theano_function = func
556          return func
557  
558      # Create wrapper to convert 0-dimensional outputs to scalars
559      def wrapper(*args):
560          out = func(*args)
561          # out can be array(1.0) or [array(1.0), array(2.0)]
562  
563          if is_sequence(out):
564              return [o[()] if is_0d[i] else o for i, o in enumerate(out)]
565          else:
566              return out[()]
567  
568      wrapper.__wrapped__ = func
569      wrapper.__doc__ = func.__doc__
570      wrapper.theano_function = func
571      return wrapper
