9    template <typename key_t, int value_size> 
10   void radix_sort_pairs_impl(
11       const key_t* keys_in,
12       key_t* keys_out,
13       const OpaqueType<value_size>* values_in,
14       OpaqueType<value_size>* values_out,
15       int64_t n,
16       bool descending,
17       int64_t begin_bit,
18       int64_t end_bit) {
19     TORCH_CHECK(
20         n <= std::numeric_limits<int>::max(),
21         "cub sort does not support sorting more than INT_MAX elements");
22     using key_t_ = typename detail::cuda_type<key_t>::type;
23   
24     auto allocator = c10::cuda::CUDACachingAllocator::get();
25     c10::DataPtr keys_out_owner;
26   
27     if (keys_out == nullptr) {
28       keys_out_owner = allocator->allocate(n * sizeof(key_t));
29       keys_out = reinterpret_cast<key_t*>(keys_out_owner.get());
30     }
31   
32     const key_t_* keys_in_ = reinterpret_cast<const key_t_*>(keys_in);
33     key_t_* keys_out_ = reinterpret_cast<key_t_*>(keys_out);
34   
35     if (descending) {
36       CUB_WRAPPER(
37           NO_ROCM(at_cuda_detail)::cub::DeviceRadixSort::SortPairsDescending,
38           keys_in_,
39           keys_out_,
40           values_in,
41           values_out,
42           n,
43           begin_bit,
44           end_bit,
45           c10::cuda::getCurrentCUDAStream());
46     } else {
47       CUB_WRAPPER(
48           NO_ROCM(at_cuda_detail)::cub::DeviceRadixSort::SortPairs,
49           keys_in_,
50           keys_out_,
51           values_in,
52           values_out,
53           n,
54           begin_bit,
55           end_bit,
56           c10::cuda::getCurrentCUDAStream());
57     }
58   }
