1    from sympy.series import approximants
2    from sympy.core.symbol import symbols
3    from sympy.functions.combinatorial.factorials import binomial
4    from sympy.functions.combinatorial.numbers import (fibonacci, lucas)
5    
6    
7    def test_approximants():
8        x, t = symbols("x,t")
9        g = [lucas(k) for k in range(16)]
10       assert list(approximants(g)) == (
11           [2, -4/(x - 2), (5*x - 2)/(3*x - 1), (x - 2)/(x**2 + x - 1)] )
12       g = [lucas(k)+fibonacci(k+2) for k in range(16)]
13       assert list(approximants(g)) == (
14           [3, -3/(x - 1), (3*x - 3)/(2*x - 1), -3/(x**2 + x - 1)] )
15       g = [lucas(k)**2 for k in range(16)]
16       assert list(approximants(g)) == (
17           [4, -16/(x - 4), (35*x - 4)/(9*x - 1), (37*x - 28)/(13*x**2 + 11*x - 7),
18           (50*x**2 + 63*x - 52)/(37*x**2 + 19*x - 13),
19           (-x**2 - 7*x + 4)/(x**3 - 2*x**2 - 2*x + 1)] )
20       p = [sum(binomial(k,i)*x**i for i in range(k+1)) for k in range(16)]
21       y = approximants(p, t, simplify=True)
22       assert next(y) == 1
23       assert next(y) == -1/(t*(x + 1) - 1)
