61   template <class T, class NullType>
62   c10::intrusive_ptr<T, NullType> IValue::moveToIntrusivePtr() {
63     auto t = c10::intrusive_ptr<T, NullType>::reclaim(
64         payload.u.as_intrusive_ptr == c10::UndefinedTensorImpl::singleton()
65         ? NullType::singleton()
66         : static_cast<T*>(payload.u.as_intrusive_ptr));
67     clearToNone();
68     return t;
69   }
70   template <typename T, class NullType>
71   c10::intrusive_ptr<T, NullType> IValue::toIntrusivePtr() const {
72     if (payload.u.as_intrusive_ptr == c10::UndefinedTensorImpl::singleton()) {
73       return c10::intrusive_ptr<T, NullType>();
74     }
75     c10::raw::intrusive_ptr::incref(payload.u.as_intrusive_ptr);
76     return c10::intrusive_ptr<T, NullType>::reclaim(
77         static_cast<T*>(payload.u.as_intrusive_ptr));
78   }

166    // As far as I can tell, omitting the usual explicit destructor call

169    // effectively an intrusive_ptr in the null state, so we don't need
170    // the behavior for correctness reasons either. Leaving this
171    // explanatory comment, including commented-out destructor call, to
172    // make this abundantly clear.

174    // payload.as_tensor.~Tensor();
175    clearToNone();
176    return result;

326  struct TORCH_API TupleElements {
327   private:
328    size_t inlineSize_;
329    // We represent TupleElements this way to save doing a heap
330    // allocation in the common (at least for unpickling) case where we
331    // have only 3 elements. We have our own union instead of
332    // c10::SmallVector<IValue> because c10::SmallVector<IValue> always
333    // stores the begin/end/capacity pointers, which would be a waste of
334    // space in our use case.
335    union {
336      std::vector<IValue> elementsVector_;
337      // Don't want to declare a std::array because the convenient
338      // iteration and size members are a footgun in this case -- the
339      // actual size of the array may be smaller than 3!
340      // NOLINTNEXTLINE(*c-arrays*)
341      IValue elementsInline_[3];
342    };
343  
344    void destroyInline() {
345     for (const auto ii : c10::irange(inlineSize_)) {
346       elementsInline_[ii].~IValue();
347     }
348    }
349   public:
350  
351    using iterator = IValue*;
352    using const_iterator = const IValue*;
353  
354    TupleElements() : inlineSize_(0) {
355      new (&elementsVector_) std::vector<IValue>();
356    }
357  
358    explicit TupleElements(std::vector<IValue> elements)
359    : inlineSize_(0), elementsVector_(std::move(elements)) {}
360  
361    explicit TupleElements(c10::ArrayRef<IValue> elements)
362    : inlineSize_(elements.size() <= 3 ? elements.size() : 0) {
363      switch (inlineSize_) {
364        case 3:
365          new (&elementsInline_[2]) IValue(elements[2]);
366          [[fallthrough]];
367        case 2:
368          new (&elementsInline_[1]) IValue(elements[1]);
369          [[fallthrough]];
370        case 1:
371          new (&elementsInline_[0]) IValue(elements[0]);
372          break;
373        case 0:
374          new (&elementsVector_) std::vector<IValue>(elements.begin(), elements.end());
375          break;
376      }
377    }
378  
379    explicit TupleElements(IValue&& e1)
380    : inlineSize_(1) {
381      new (&elementsInline_[0]) IValue(std::move(e1));
382    }
383  
384    explicit TupleElements(IValue&& e1, IValue&& e2)
385    : inlineSize_(2) {
386      new (&elementsInline_[0]) IValue(std::move(e1));
387      new (&elementsInline_[1]) IValue(std::move(e2));
388    }
389  
390    explicit TupleElements(IValue&& e1, IValue&& e2, IValue&& e3)
391    : inlineSize_(3) {
392      new (&elementsInline_[0]) IValue(std::move(e1));
393      new (&elementsInline_[1]) IValue(std::move(e2));
394      new (&elementsInline_[2]) IValue(std::move(e3));
395    }
396  
397    ~TupleElements() {
398      if (inlineSize_) {
399        destroyInline();
400      } else {
401        elementsVector_.~vector();
402      }
403    }
404  
405    // It would be nice to make this noncopyable to prevent people from
406    // writing code like `auto output =
407    // forward(...).toTupleRef().elements()` (which does refcount bumps on
408    // each element, unlike the more efficient but verbose
409    // ```
410    // auto outputIntrusivePtr = forward(...).toTuple();
411    // const auto& output = outputIntrusivePtr->elements();
412    // ```
413    // ), but there is simply an overwhelming amount of code that does
414    // it the inefficient way.
415    // See also operator std::vector below.
416    TupleElements(const TupleElements& rhs)
417    : inlineSize_(rhs.inlineSize_) {
418      if (rhs.inlineSize_) {
419        for (const auto  ii : c10::irange(inlineSize_)) {
420          new (&elementsInline_[ii]) IValue(rhs.elementsInline_[ii]);
421        }
422      } else {
423        new (&elementsVector_) std::vector<IValue>(rhs.elementsVector_);
424      }
425    }
426  
427    TupleElements& operator=(const TupleElements& rhs) {
428      if (inlineSize_) {
429        if (rhs.inlineSize_) {
430          for (const auto ii : c10::irange(std::min(inlineSize_, rhs.inlineSize_))) {
431            elementsInline_[ii] = rhs.elementsInline_[ii];
432          }
433          if (rhs.inlineSize_ > inlineSize_) {
434            for (const auto ii : c10::irange(inlineSize_, rhs.inlineSize_)) {
435              new (&elementsInline_[ii]) IValue(rhs.elementsInline_[ii]);
436            }
437          } else {
438            for (const auto ii : c10::irange(rhs.inlineSize_, inlineSize_)) {
439              elementsInline_[ii].~IValue();
440            }
441          }
442        } else {
443          destroyInline();
444          new (&elementsVector_) std::vector<IValue>(rhs.elementsVector_);
445        }
446      } else {
447        if (rhs.inlineSize_) {
448          elementsVector_.~vector();
449          for (const auto ii : c10::irange(rhs.inlineSize_)) {
450            new (&elementsInline_[ii]) IValue(rhs.elementsInline_[ii]);
451          }
452        } else {
453          elementsVector_ = rhs.elementsVector_;
454        }
455      }
456      inlineSize_ = rhs.inlineSize_;
457      return *this;
458    }
459  
460    TupleElements(TupleElements&& rhs) noexcept
461    : inlineSize_(rhs.inlineSize_) {
462      if (inlineSize_) {
463        for (const auto ii : c10::irange(inlineSize_)) {
464          new (&elementsInline_[ii]) IValue(std::move(rhs.elementsInline_[ii]));
465        }
466      } else {
467        new (&elementsVector_) std::vector<IValue>(std::move(rhs.elementsVector_));
468      }
469    }
470  
471    TupleElements& operator=(TupleElements&& rhs) noexcept {
472      if (inlineSize_) {
473        if (rhs.inlineSize_) {
474          for (const auto ii : c10::irange(std::min(inlineSize_, rhs.inlineSize_))) {
475            elementsInline_[ii] = std::move(rhs.elementsInline_[ii]);
476          }
477          if (rhs.inlineSize_ > inlineSize_) {
478            for (const auto ii : c10::irange(inlineSize_, rhs.inlineSize_)) {
479              new (&elementsInline_[ii]) IValue(std::move(rhs.elementsInline_[ii]));
480            }
481          } else {
482            for (const auto ii : c10::irange(rhs.inlineSize_, inlineSize_)) {
483              elementsInline_[ii].~IValue();
484            }
485          }
486        } else {
487          destroyInline();
488          new (&elementsVector_) std::vector<IValue>(std::move(rhs.elementsVector_));
489        }
490      } else {
491        if (rhs.inlineSize_) {
492          elementsVector_.~vector();
493          for (const auto ii : c10::irange(rhs.inlineSize_)) {
494            new (&elementsInline_[ii]) IValue(std::move(rhs.elementsInline_[ii]));
495          }
496        } else {
497          elementsVector_ = std::move(rhs.elementsVector_);
498        }
499      }
500      inlineSize_ = rhs.inlineSize_;
501      return *this;
502    }
503  
504    [[nodiscard]] c10::ArrayRef<IValue> asArrayRef() const {
505      if (inlineSize_) {
506        return c10::ArrayRef<IValue>(elementsInline_, inlineSize_);
507      } else {
508        return elementsVector_;
509      }
510    }

515    }

522      if (inlineSize_) {

526      } else {
527        elementsVector_ = std::move(contents);
528      }
529    }

914    /** 
915     * Explicitly mark the future as completed with the output value. Optionally, 
916     * the storages for all tensors in IValue can be passed as well. The DataPtrs 
917     * of these storages are used to synchronize CUDA streams. If storages isn't 
918     * given we will attempt to extract it from the value, if we need to (this 
919     * happens if a non-empty set of devices was given to the constructor). Thus 
920     * one only needs to provide storages when 1) they cannot be extracted through 
921     * IValue::getSubValues() or through pickling in case of Python object; or 
922     * when 2) customized storage extraction is more efficient. 
923     */
924    using WeakStorage = c10::weak_intrusive_ptr<c10::StorageImpl>;
925    void markCompleted(
926        IValue value,
927        std::optional<std::vector<WeakStorage>> storages = std::nullopt) {
928      // Start by performing all steps that can throw, before setting any field.
929      // Do this before even acquiring the mutex, because extractStorages might
930      // acquire the GIL, which could lead to a lock inversion with our mutex.
931      // See https://github.com/pytorch/pytorch/issues/58239.
932      std::vector<WeakStorage> actualStorages;
933      std::vector<c10::Device> usedDevices;
934      try {
935        // FIXME We should always extract DataPtrs, in order to catch the case of
936        // users using CUDA values but forgetting to set devices, which currently
937        // leads to a silent synchronization/correctness issue. However, as this
938        // might worsen perf in CPU-only cases, we should only do so after careful
939        // benchmarks.
940        if (impl_.type() != c10::kCPU) {
941          actualStorages =
942              storages.has_value() ? std::move(*storages) : extractStorages(value);
943          usedDevices = getDevicesOfStorages(impl_, actualStorages);
944          ensureIsSubsetOfDevices(usedDevices, devices_);
945        }
946      } catch (const std::exception&) {
947        setError(std::current_exception());
948        return;
949      }
950  
951      std::unique_lock<std::mutex> lock(mutex_);
952      TORCH_CHECK(
953          !completed(),
954          "Attempting to mark a completed Future as complete again. Note that "
955          "a Future can only be marked completed once.");
956  
957      // Only set value_ and completed_ flag once all checks and preparation steps
958      // have returned successfully to allow for proper error propagation.
959      value_ = std::move(value);
960      completed_ = true;
961  
962      currentDevice_ = impl_.getDevice();
963      storages_ = std::move(actualStorages);
964      for (const c10::Device& device : usedDevices) {
965        c10::Event event(impl_.type());
966        event.record(impl_.getStream(device));
967        events_.push_back(std::move(event));
968      }
969  
970      std::vector<FutureCallback> cbs;
971      cbs.swap(callbacks_);
972      lock.unlock();
973  
974      finished_cv_.notify_all();
975      for (const auto& callback : cbs) {
976        invokeCallback(callback.callback, callback.uses_future);
977      }
978    }

1065   /** 
1066    * Add a callback to the future, and return another Future to hold the return 
1067    * value of the callback. This is necessary when the callback provider needs 
1068    * to know for sure when the callback has finished. 
1069    */
1070   template <typename T>
1071   c10::intrusive_ptr<Future> then(T callback, TypePtr type) {
1072     using IValueWithStorages = std::tuple<IValue, std::vector<WeakStorage>>;
1073     static_assert(
1074         std::disjunction_v<
1075             std::is_invocable_r<IValue, T, Future&>,
1076             std::is_invocable_r<IValueWithStorages, T, Future&>>,
1077         "The callback must have signature IValue(Future&) or "
1078         "std::tuple<IValue, std::vector<Storage>>(Future&)");
1079 
1080     auto childFut = createInstance(::std::move(type));
1081     addCallback([childFut,
1082                  cb = std::move(callback)](Future& parentFut) {
1083       try {
1084         if constexpr (::std::is_convertible_v<typename std::invoke_result_t<T &&, Future&>, IValueWithStorages>) {
1085           auto [ivalue, storages] = cb(parentFut);
1086           childFut->markCompleted(::std::move(ivalue), ::std::move(storages));
1087         } else {
1088           childFut->markCompleted(cb(parentFut));
1089         }
1090       } catch (std::exception&) {
1091         childFut->setError(std::current_exception());
1092       }
1093     });
1094     return childFut;
1095   }

1174   // how/when that happens) as it will ensure that the proper "environment" is
1175   // set up before running the callback, as in, it will set up the CUDA streams,
1176   // synchronize them with the value, and so on (if needed).
1177   template<typename T>
1178   void invokeCallback(T& callback, bool uses_future) {
1179     static_assert(
1180         std::is_invocable_r_v<void, T, Future&>,
1181         "The callback must have signature void(Future&)");
1182 
1183     // The synchronization performed below shouldn't be needed when the future
1184     // is not used by the callback.
1185     if (uses_future) {
1186       c10::OptionalDeviceGuard deviceGuard(currentDevice_);
1187 
1188       std::vector<c10::Stream> streams;
1189       streams.reserve(devices_.size());
1190       for (const c10::Device& device : devices_) {
1191         streams.push_back(impl_.getStreamFromGlobalPool(device));
1192       }
1193       c10::MultiStreamGuard streamGuard(streams);
1194       synchronizeWithCurrentStreams();
1195       callback(*this);
1196     } else {
1197       callback(*this);
1198     }

1201   // This method should be called before this future's value is used, as it
1202   // ensures that the CUDA streams that are "current" at the callsite properly
1203   // synchronize with the value.
1204   void synchronizeWithCurrentStreams() {
1205     for (c10::Event& event : events_) {
1206       event.block(impl_.getStream(event.device()));
1207     }
1208 
1209     for (const WeakStorage& weak_storage : storages_) {
1210       c10::intrusive_ptr<c10::StorageImpl> storage = weak_storage.lock();
1211       if (!storage) {
1212         continue;
1213       }
1214       if (!storage->device().is_cpu()) {
1215         impl_.recordDataPtrOnStream(
1216             storage->data_ptr(), impl_.getStream(storage->device()));
1217       }
1218     }
1219   }

1261       const std::vector<WeakStorage>& storages) {
1262     c10::DeviceIndex deviceCount = impl.deviceCount();
1263     std::vector<bool> isDeviceUsed(deviceCount, false);
1264     for (const WeakStorage& weak_storage : storages) {
1265       c10::intrusive_ptr<c10::StorageImpl> storage = weak_storage.lock();
1266       if (!storage) {
1267         continue;
1268       }
1269       c10::Device device = storage->device();
1270       if (!device.is_cpu()) {
1271         TORCH_CHECK_VALUE(
1272             device.type() == impl.type(),
1273             "Expected all data ptrs to be on a device of type ",
1274             impl.type(),
1275             ", got one on device ",
1276             device);
1277         isDeviceUsed[device.index()] = true;
1278       }
1279     }
1280     std::vector<c10::Device> devices;
1281     for (c10::DeviceIndex idx = 0; idx < deviceCount; idx++) {
1282       if (isDeviceUsed[idx]) {
1283         devices.emplace_back(impl.type(), idx);
1284       }
1285     }
1286     return devices;
1287   }

1324   // We need devices to be sorted in order to use ensureIsSubsetOfDevices.
1325   static std::vector<c10::Device> sortAndDeduplicateDevices(
1326       const c10::impl::VirtualGuardImpl& /*impl*/,
1327       std::vector<c10::Device> devices) {
1328     std::sort(
1329       devices.begin(), devices.end(),
1330       [](const c10::Device& a, const c10::Device& b) { return a.index() < b.index(); });
1331     // Deduplicate by compacting.
1332     size_t targetIdx = 0;
1333     for (const auto sourceIdx : c10::irange(devices.size())) {
1334       TORCH_CHECK_VALUE(
1335           devices[sourceIdx].has_index(),
1336           "Expected devices to have indices, got ", devices[sourceIdx]);
1337       if (targetIdx > 0 && devices[targetIdx - 1].index() == devices[sourceIdx].index()) {
1338         // It's a duplicate, skip it.
1339         continue;
1340       }
1341       if (sourceIdx != targetIdx) {
1342         devices[targetIdx] = devices[sourceIdx];
1343       }
1344       targetIdx++;
1345     }
1346     // If there were duplicates there's now a gap at the end: trim it. Resizing
1347     // requires the item type to be default-constructible (which c10::Device is
1348     // not) because in principle it could be required to create new items. Since
1349     // we know we'll shrink the vector, we provide a custom dummy value instead.
1350     devices.resize(targetIdx, c10::Device(c10::kCPU));
1351     return devices;
1352   }

1357     // We assume the devices in both vectors have the same consistent type, and
1358     // their indices are unique and sorted.

1361         subset.begin(),
1362         subset.end(),
1363         superset.begin(),

1366         [](const c10::Device& a, const c10::Device& b) { return a.index() < b.index(); });
1367     TORCH_CHECK_VALUE(
1368         excessDevices.empty(),
1369         "The result contained tensors residing on device(s) ",
1370         formatSetOfDevices(excessDevices),
1371         " which are not among the expected device(s) ",
1372         formatSetOfDevices(superset));
1373   }

2442 inline bool IValue::isSameIdentity(const IValue& rhs) const {
2443   // We choose to not use memcmp for payload check due to potential random
2444   // padding characters on union type
2445 
2446   // Semantics:
2447   // 1. Immutable primitive values of the same type (Int, Double, None, Bool,
2448   // Str) return value equality
2449   // 2. If it is a tensor type, we need to take undefined tensor into account
2450   // 3. Undefined_tensor is None and vice versa should be true
2451   // 4. If it is a reference type (i.e. isIntrusivePtr()), then is True when
2452   // the pointed-to object is the same.
2453   // 5. False for all other comparisons.
2454   if (this->isNone() && rhs.isNone()) {
2455     return true;
2456   } else if (this->isBool() && rhs.isBool()) {
2457     // for bool type, do equality check
2458     return this->toBool() == rhs.toBool();
2459   } else if (this->isTensor() && rhs.isTensor()) {
2460     return this->payload.as_tensor.is_same(rhs.payload.as_tensor);
2461   } else if (this->isTensor() && rhs.isNone()) {
2462     // special case: undefined tensor and None are the same identity
2463     return !this->payload.as_tensor.defined();
2464   } else if (this->isNone() && rhs.isTensor()) {
2465     // special case: undefined tensor and None are the same identity
2466     return !rhs.payload.as_tensor.defined();
2467   } else if (this->isInt() && rhs.isInt()) {
2468     return this->toInt() == rhs.toInt();
2469   } else if (this->isDouble() && rhs.isDouble()) {
2470     return this->toDouble() == rhs.toDouble();
2471   } else if (this->isString() && rhs.isString()) {
2472     return this->toStringRef() == rhs.toStringRef();
2473   } else {
2474     // for objects holding in IValue, do shallow compare on pointer address to
2475     // testify the identity
2476     return this->isIntrusivePtr() && rhs.isIntrusivePtr() &&
2477         this->payload.u.as_intrusive_ptr == rhs.payload.u.as_intrusive_ptr;
2478   }
2479 }
