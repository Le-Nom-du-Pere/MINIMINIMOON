102  inline constexpr uint64_t crc64impl(
103      uint64_t accumulator,
104      const char* data,
105      size_t size) {
106    for (size_t i = 0; i < size; ++i) {
107      accumulator =
108          crc64_table[(accumulator ^ data[i]) & 0xFF] ^ (accumulator >> 8);
109    }
110    return accumulator;
111  }

121  // CRC64 with Jones coefficients and an init value of 0.
122  inline constexpr crc64_t crc64(const char* str, size_t size) {
123    return crc64_t{detail::crc64impl(0, str, size)};
124  }
