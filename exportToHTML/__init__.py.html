20   class Future(torch._C.Future, Generic[T], metaclass=_PyFutureMeta):
21       r""" 
22       Wrapper around a ``torch._C.Future`` which encapsulates an asynchronous 
23       execution of a callable, e.g. :meth:`~torch.distributed.rpc.rpc_async`. It 
24       also exposes a set of APIs to add callback functions and set results. 
25    
26       .. warning:: GPU support is a beta feature, subject to changes. 
27       """
28   
29       def __init__(
30           self, *, devices: Optional[list[Union[int, str, torch.device]]] = None
31       ):
32           r""" 
33           Create an empty unset ``Future``. If the future is intended to hold 
34           values containing CUDA tensors, (a superset of) their CUDA devices must 
35           be specified at construction. (This is only supported if 
36           ``torch.cuda.is_available()`` returns ``True``). This is needed to 
37           ensure proper CUDA stream synchronization. The child futures, returned 
38           by the ``then`` method, will inherit these devices. 
39    
40           Args: 
41               devices(``List[Union[int, str, torch.device]]``, optional): the set 
42                   of devices on which tensors contained in this future's value are 
43                   allowed to reside and on which callbacks are allowed to operate. 
44           """
45           if devices is None:
46               devices = []
47           super().__init__([torch.device(d) for d in devices])
48   
49       def done(self) -> bool:
50           r""" 
51           Return ``True`` if this ``Future`` is done. A ``Future`` is done if it 
52           has a result or an exception. 
53    
54           If the value contains tensors that reside on GPUs, ``Future.done()`` 
55           will return ``True`` even if the asynchronous kernels that are 
56           populating those tensors haven't yet completed running on the device, 
57           because at such stage the result is already usable, provided one 
58           performs the appropriate synchronizations (see :meth:`wait`). 
59           """
60           return super().done()
61   
62       def wait(self) -> T:
63           r""" 
64           Block until the value of this ``Future`` is ready. 
65    
66           If the value contains tensors that reside on GPUs, then an additional 
67           synchronization is performed with the kernels (executing on the device) 
68           which may be asynchronously populating those tensors. Such sync is 
69           non-blocking, which means that ``wait()`` will insert the necessary 
70           instructions in the current streams to ensure that further operations 
71           enqueued on those streams will be properly scheduled after the async 
72           kernels but, once that is done, ``wait()`` will return, even if those 
73           kernels are still running. No further synchronization is required when 
74           accessing and using the values, as long as one doesn't change streams. 
75    
76           Returns: 
77               The value held by this ``Future``. If the function (callback or RPC) 
78               creating the value has thrown an error, this ``wait`` method will 
79               also throw an error. 
80           """
81           return super().wait()
82   
83       def value(self) -> T:
84           r""" 
85           Obtain the value of an already-completed future. 
86    
87           This method should only be called after a call to :meth:`wait` has 
88           completed, or inside a callback function passed to :meth:`then`. In 
89           other cases this ``Future`` may not yet hold a value and calling 
90           ``value()`` could fail. 
91    
92           If the value contains tensors that reside on GPUs, then this method will 
93           *not* perform any additional synchronization. This should be done 
94           beforehand, separately, through a call to :meth:`wait` (except within 
95           callbacks, for which it's already being taken care of by :meth:`then`). 
96    
97           Returns: 
98               The value held by this ``Future``. If the function (callback or RPC) 
99               creating the value has thrown an error, this ``value()`` method will 
100              also throw an error. 
101          """
102          return super().value()
103  
104      def then(self, callback: Callable[[Future[T]], S]) -> Future[S]:
105          r""" 
106          Append the given callback function to this ``Future``, which will be run 
107          when the ``Future`` is completed.  Multiple callbacks can be added to 
108          the same ``Future``, but the order in which they will be executed cannot 
109          be guaranteed (to enforce a certain order consider chaining: 
110          ``fut.then(cb1).then(cb2)``). The callback must take one argument, which 
111          is the reference to this ``Future``. The callback function can use the 
112          :meth:`value` method to get the value. Note that if this ``Future`` is 
113          already completed, the given callback will be run immediately inline. 
114   
115          If the ``Future``'s value contains tensors that reside on GPUs, the 
116          callback might be invoked while the async kernels that are populating 
117          those tensors haven't yet finished executing on the device. However, the 
118          callback will be invoked with some dedicated streams set as current 
119          (fetched from a global pool) which will be synchronized with those 
120          kernels. Hence any operation performed by the callback on these tensors 
121          will be scheduled on the device after the kernels complete. In other 
122          words, as long as the callback doesn't switch streams, it can safely 
123          manipulate the result without any additional synchronization. This is 
124          similar to the non-blocking behavior of :meth:`wait`. 
125   
126          Similarly, if the callback returns a value that contains tensors that 
127          reside on a GPU, it can do so even if the kernels that are producing 
128          these tensors are still running on the device, as long as the callback 
129          didn't change streams during its execution. If one wants to change 
130          streams, one must be careful to re-synchronize them with the original 
131          streams, that is, those that were current when the callback was invoked. 
132   
133          Args: 
134              callback(``Callable``): a ``Callable`` that takes this ``Future`` as 
135                                      the only argument. 
136   
137          Returns: 
138              A new ``Future`` object that holds the return value of the 
139              ``callback`` and will be marked as completed when the given 
140              ``callback`` finishes. 
141   
142          .. note:: Note that if the callback function throws, either 
143              through the original future being completed with an exception and 
144              calling ``fut.wait()``, or through other code in the callback, the 
145              future returned by ``then`` will be marked appropriately with the 
146              encountered error. However, if this callback later completes 
147              additional futures, those futures are not marked as completed with 
148              an error and the user is responsible for handling completion/waiting 
149              on those futures independently. 
150   
151          Example:: 
152   
153              >>> # xdoctest: +REQUIRES(env:TORCH_DOCTEST_FUTURES) 
154              >>> def callback(fut): 
155              ...     print(f"RPC return value is {fut.wait()}.") 
156              >>> fut = torch.futures.Future() 
157              >>> # The inserted callback will print the return value when 
158              >>> # receiving the response from "worker1" 
159              >>> cb_fut = fut.then(callback) 
160              >>> chain_cb_fut = cb_fut.then( 
161              ...     lambda x : print(f"Chained cb done. {x.wait()}") 
162              ... ) 
163              >>> fut.set_result(5) 
164              RPC return value is 5. 
165              Chained cb done. None 
166          """
167          return cast(Future[S], super().then(callback))

169      def add_done_callback(self, callback: Callable[[Future[T]], None]) -> None:
170          r""" 
171          Append the given callback function to this ``Future``, which will be run 
172          when the ``Future`` is completed.  Multiple callbacks can be added to 
173          the same ``Future``, but the order in which they will be executed cannot 
174          be guaranteed. The callback must take one argument, which is the 
175          reference to this ``Future``. The callback function can use the 
176          :meth:`value` method to get the value. Note that if this ``Future`` is 
177          already completed, the given callback will be run inline. 
178   
179          We recommend that you use the :meth:`then` method as it provides a way 
180          to synchronize after your callback has completed. ``add_done_callback`` 
181          can be cheaper if your callback does not return anything. But both 
182          :meth:`then` and ``add_done_callback`` use the same callback 
183          registration API under the hood. 
184   
185          With respect to GPU tensors, this method behaves in the same way as 
186          :meth:`then`. 
187   
188          Args: 
189              callback(``Future``): a ``Callable`` that takes in one argument, 
190                  which is the reference to this ``Future``. 
191   
192          .. note:: Note that if the callback function throws, either 
193              through the original future being completed with an exception and 
194              calling ``fut.wait()``, or through other code in the callback, 
195              error handling must be carefully taken care of. For example, if 
196              this callback later completes additional futures, those futures are 
197              not marked as completed with an error and the user is responsible 
198              for handling completion/waiting on those futures independently. 
199   
200          Example:: 
201   
202              >>> # xdoctest: +REQUIRES(env:TORCH_DOCTEST_FUTURES) 
203              >>> def callback(fut): 
204              ...     print("This will run after the future has finished.") 
205              ...     print(fut.wait()) 
206              >>> fut = torch.futures.Future() 
207              >>> fut.add_done_callback(callback) 
208              >>> fut.set_result(5) 
209              This will run after the future has finished. 
210              5 
211          """
212          super().add_done_callback(callback)

214      def set_result(self, result: T) -> None:
215          r""" 
216          Set the result for this ``Future``, which will mark this ``Future`` as 
217          completed and trigger all attached callbacks. Note that a ``Future`` 
218          cannot be marked completed twice. 
219   
220          If the result contains tensors that reside on GPUs, this method can be 
221          called even if the asynchronous kernels that are populating those 
222          tensors haven't yet completed running on the device, provided that the 
223          streams on which those kernels were enqueued are set as the current ones 
224          when this method is called. Put simply, it's safe to call this method 
225          immediately after launching those kernels, without any additional 
226          synchronization, as long as one doesn't change streams in between. This 
227          method will record events on all the relevant current streams and will 
228          use them to ensure proper scheduling for all the consumers of this 
229          ``Future``. 
230   
231          Args: 
232              result (object): the result object of this ``Future``. 
233   
234          Example:: 
235   
236              >>> # xdoctest: +REQUIRES(env:TORCH_DOCTEST_FUTURES) 
237              >>> import threading 
238              >>> import time 
239              >>> def slow_set_future(fut, value): 
240              ...     time.sleep(0.5) 
241              ...     fut.set_result(value) 
242              >>> fut = torch.futures.Future() 
243              >>> t = threading.Thread( 
244              ...     target=slow_set_future, 
245              ...     args=(fut, torch.ones(2) * 3) 
246              ... ) 
247              >>> t.start() 
248              >>> print(fut.wait()) 
249              tensor([3., 3.]) 
250              >>> t.join() 
251          """
252          super().set_result(result)
253  
254      def set_exception(self, result: T) -> None:
255          r""" 
256          Set an exception for this ``Future``, which will mark this ``Future`` as 
257          completed with an error and trigger all attached callbacks. Note that 
258          when calling wait()/value() on this ``Future``, the exception set here 
259          will be raised inline. 
260   
261          Args: 
262              result (BaseException): the exception for this ``Future``. 
263   
264          Example:: 
265   
266              >>> # xdoctest: +REQUIRES(env:TORCH_DOCTEST_FUTURES) 
267              >>> fut = torch.futures.Future() 
268              >>> fut.set_exception(ValueError("foo")) 
269              >>> fut.wait() 
270              Traceback (most recent call last): 
271              ... 
272              ValueError: foo 
273          """
274          assert isinstance(result, Exception), (
275              f"{result} is of type {type(result)}, not an Exception."
276          )
277  
278          def raise_error(fut_result):
279              raise fut_result
280  
281          super()._set_unwrap_func(raise_error)
282          self.set_result(result)  # type: ignore[arg-type]

285  def collect_all(futures: list[Future]) -> Future[list[Future]]:
286      r""" 
287      Collects the provided :class:`~torch.futures.Future` objects into a single 
288      combined :class:`~torch.futures.Future` that is completed when all of the 
289      sub-futures are completed. 
290   
291      Args: 
292          futures (list): a list of :class:`~torch.futures.Future` objects. 
293   
294      Returns: 
295          Returns a :class:`~torch.futures.Future` object to a list of the passed 
296          in Futures. 
297   
298      Example:: 
299          >>> # xdoctest: +REQUIRES(env:TORCH_DOCTEST_FUTURES) 
300          >>> fut0 = torch.futures.Future() 
301          >>> fut1 = torch.futures.Future() 
302          >>> fut = torch.futures.collect_all([fut0, fut1]) 
303          >>> fut0.set_result(0) 
304          >>> fut1.set_result(1) 
305          >>> fut_list = fut.wait() 
306          >>> print(f"fut0 result = {fut_list[0].wait()}") 
307          fut0 result = 0 
308          >>> print(f"fut1 result = {fut_list[1].wait()}") 
309          fut1 result = 1 
310      """
311      return cast(
312          Future[list[Future]],
313          torch._C._collect_all(cast(list[torch._C.Future], futures)),
314      )

317  def wait_all(futures: list[Future]) -> list:
318      r""" 
319      Waits for all provided futures to be complete, and returns 
320      the list of completed values. If any of the futures encounters an error, 
321      the method will exit early and report the error not waiting for other 
322      futures to complete. 
323   
324      Args: 
325          futures (list): a list of :class:`~torch.futures.Future` object. 
326   
327      Returns: 
328          A list of the completed :class:`~torch.futures.Future` results. This 
329          method will throw an error if ``wait`` on any 
330          :class:`~torch.futures.Future` throws. 
331      """
332      return [
333          fut.wait()
334          for fut in torch._C._collect_all(cast(list[torch._C.Future], futures)).wait()
335      ]
