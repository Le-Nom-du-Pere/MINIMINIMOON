7    def figure_to_image(figures, close=True):
8        """Render matplotlib figure to numpy format. 
9     
10       Note that this requires the ``matplotlib`` package. 
11    
12       Args: 
13           figures (matplotlib.pyplot.figure or list of figures): figure or a list of figures 
14           close (bool): Flag to automatically close the figure 
15    
16       Returns: 
17           numpy.array: image in [CHW] order 
18       """
19       import matplotlib.pyplot as plt
20       import matplotlib.backends.backend_agg as plt_backend_agg
21   
22       def render_to_rgb(figure):
23           canvas = plt_backend_agg.FigureCanvasAgg(figure)
24           canvas.draw()
25           data: npt.NDArray = np.frombuffer(canvas.buffer_rgba(), dtype=np.uint8)
26           w, h = figure.canvas.get_width_height()
27           image_hwc = data.reshape([h, w, 4])[:, :, 0:3]
28           image_chw = np.moveaxis(image_hwc, source=2, destination=0)
29           if close:
30               plt.close(figure)
31           return image_chw
32   
33       if isinstance(figures, list):
34           images = [render_to_rgb(figure) for figure in figures]
35           return np.stack(images)
36       else:
37           image = render_to_rgb(figures)
38           return image

41   def _prepare_video(V):
42       """ 
43       Convert a 5D tensor into 4D tensor. 
44    
45       Convesrion is done from [batchsize, time(frame), channel(color), height, width]  (5D tensor) 
46       to [time(frame), new_width, new_height, channel] (4D tensor). 
47    
48       A batch of images are spreaded to a grid, which forms a frame. 
49       e.g. Video with batchsize 16 will have a 4x4 grid. 
50       """
51       b, t, c, h, w = V.shape
52   
53       if V.dtype == np.uint8:
54           V = np.float32(V) / 255.0
55   
56       def is_power2(num):
57           return num != 0 and ((num & (num - 1)) == 0)
58   
59       # pad to nearest power of 2, all at once
60       if not is_power2(V.shape[0]):
61           len_addition = int(2 ** V.shape[0].bit_length() - V.shape[0])
62           V = np.concatenate((V, np.zeros(shape=(len_addition, t, c, h, w))), axis=0)
63   
64       n_rows = 2 ** ((b.bit_length() - 1) // 2)
65       n_cols = V.shape[0] // n_rows
66   
67       V = np.reshape(V, newshape=(n_rows, n_cols, t, c, h, w))
68       V = np.transpose(V, axes=(2, 0, 4, 1, 5, 3))
69       V = np.reshape(V, newshape=(t, n_rows * h, n_cols * w, c))
70   
71       return V

74   def make_grid(I, ncols=8):
75       # I: N1HW or N3HW
76       assert isinstance(I, np.ndarray), "plugin error, should pass numpy array here"
77       if I.shape[1] == 1:
78           I = np.concatenate([I, I, I], 1)
79       assert I.ndim == 4 and I.shape[1] == 3
80       nimg = I.shape[0]
81       H = I.shape[2]
82       W = I.shape[3]
83       ncols = min(nimg, ncols)
84       nrows = int(np.ceil(float(nimg) / ncols))
85       canvas = np.zeros((3, H * nrows, W * ncols), dtype=I.dtype)
86       i = 0
87       for y in range(nrows):
88           for x in range(ncols):
89               if i >= nimg:
90                   break
91               canvas[:, y * H : (y + 1) * H, x * W : (x + 1) * W] = I[i]
92               i = i + 1
93       return canvas

100  def convert_to_HWC(tensor, input_format):  # tensor: numpy array
101      assert len(set(input_format)) == len(
102          input_format
103      ), f"You can not use the same dimension shordhand twice.         input_format: {input_format}"
104      assert len(tensor.shape) == len(
105          input_format
106      ), f"size of input tensor and input format are different. \ 
107          tensor shape: {tensor.shape}, input_format: {input_format}"
108      input_format = input_format.upper()
109  
110      if len(input_format) == 4:
111          index = [input_format.find(c) for c in "NCHW"]
112          tensor_NCHW = tensor.transpose(index)
113          tensor_CHW = make_grid(tensor_NCHW)
114          return tensor_CHW.transpose(1, 2, 0)
115  
116      if len(input_format) == 3:
117          index = [input_format.find(c) for c in "HWC"]
118          tensor_HWC = tensor.transpose(index)
119          if tensor_HWC.shape[2] == 1:
120              tensor_HWC = np.concatenate([tensor_HWC, tensor_HWC, tensor_HWC], 2)
121          return tensor_HWC
122  
123      if len(input_format) == 2:
124          index = [input_format.find(c) for c in "HW"]
125          tensor = tensor.transpose(index)
126          tensor = np.stack([tensor, tensor, tensor], 2)
127          return tensor
