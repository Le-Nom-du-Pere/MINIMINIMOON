natural.py 
1    """ 
2    Naturalunit system. 
3     
4    The natural system comes from "setting c = 1, hbar = 1". From the computer 
5    point of view it means that we use velocity and action instead of length and 
6    time. Moreover instead of mass we use energy. 
7    """
8    
9    from sympy.physics.units import DimensionSystem
10   from sympy.physics.units.definitions import c, eV, hbar
11   from sympy.physics.units.definitions.dimension_definitions import (
12       action, energy, force, frequency, length, mass, momentum,
13       power, time, velocity)
14   from sympy.physics.units.prefixes import PREFIXES, prefix_unit
15   from sympy.physics.units.unitsystem import UnitSystem
16   
17   
18   # dimension system
19   _natural_dim = DimensionSystem(
20       base_dims=(action, energy, velocity),
21       derived_dims=(length, mass, time, momentum, force, power, frequency)
22   )
23   
24   units = prefix_unit(eV, PREFIXES)
25   
26   # unit system
27   natural = UnitSystem(base_units=(hbar, eV, c), units=units, name="Natural system")
