106  /// Runs the matrix multiplication (matmul) micro-kernel followed by a clamp (min-max) operation.
107  ///
108  /// LHS matrix: Quantized Asymmetric Signed 8-bit with per-row quantization (qai8dx) and packed.
109  /// RHS matrix: Quantized Symmetric Signed 4-bit with per-block (32) quantization (qsi4c32) and packed.
110  /// Output tile: (rows x cols) = m_step x n_step.
111  ///
112  /// Note: Please refer to the get functions for m_step and n_step for the exact values.
113  ///
114  /// Features used: i8mm
115  ///
116  /// @param[in]  m              The number of output rows written.
117  /// @param[in]  n              The number of output columns written.
118  /// @param[in]  k              The number of channels. The common dimension between the LHS and RHS matrix.
119  ///                            It must be a multiple of the block length (bl).
120  /// @param[in]  bl             Block length. Block length. It must be a multiple of 32.
121  /// @param[in]  lhs_packed     The LHS packed matrix. The micro-kernel to pack the native LHS matrix is reported at the
122  /// top of this file.
123  /// @param[in]  rhs_packed     The RHS packed matrix. The micro-kernel to pack the native RHS matrix is reported at the
124  /// top of this file.
125  /// @param[out] dst            The DST matrix.
126  /// @param[in]  dst_stride_row Stride in bytes between two rows of the DST matrix.
127  /// @param[in]  dst_stride_col Stride in bytes between two columns of the DST matrix. It must be sizeof(float) bytes.
128  /// @param[in]  scalar_min     Min value used to clamp the final result.
129  /// @param[in]  scalar_max     Max value used to clamp the final result.
130  void kai_run_matmul_clamp_f32_qai8dxp4x8_qsi4c32p4x8_16x4x32_neon_i8mm(
131      size_t m,                //
132      size_t n,                //
133      size_t k,                //
134      size_t bl,               //
135      const void* lhs_packed,  //
136      const void* rhs_packed,  //
137      float* dst,              //
138      size_t dst_stride_row,   //
139      size_t dst_stride_col,   //
140      float scalar_min,        //
141      float scalar_max);       //
