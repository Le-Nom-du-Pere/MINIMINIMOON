16   
17   def masked_tensor(
18       data: object, mask: object, requires_grad: bool = False
19   ) -> MaskedTensor:
20       return MaskedTensor(data, mask, requires_grad)

22   
23   def as_masked_tensor(data: object, mask: object) -> MaskedTensor:
