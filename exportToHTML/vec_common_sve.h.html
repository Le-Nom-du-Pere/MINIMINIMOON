33   #define DEFINE_SVE_CAST(t1_t, t1_prefix, t2_t, t2_prefix)                 \
34     template <>                                                             \
35     inline Vectorized<t1_t> cast<t1_t, t2_t>(const Vectorized<t2_t>& src) { \
36       return svreinterpret_##t1_prefix##_##t2_prefix(src);                  \
37     }                                                                       \
38     template <>                                                             \
39     inline Vectorized<t2_t> cast<t2_t, t1_t>(const Vectorized<t1_t>& src) { \
40       return svreinterpret_##t2_prefix##_##t1_prefix(src);                  \
41     }

59   template <int64_t scale = 1>
60   std::enable_if_t<
61       scale == 1 || scale == 2 || scale == 4 || scale == 8,
62       Vectorized<
63           double>> inline gather(const double* base_addr, const Vectorized<int64_t>& vindex_) {
64     svint64_t vindex =
65         svasrd_n_s64_x(ptrue, svmul_s64_x(ptrue, vindex_, svdup_n_s64(scale)), 3);
66     return svld1_gather_s64index_f64(ptrue, base_addr, vindex);
67   }
68   
69   template <int64_t scale = 1>
70   std::enable_if_t<
71       scale == 1 || scale == 2 || scale == 4 || scale == 8,
72       Vectorized<
73           float>> inline gather(const float* base_addr, const Vectorized<int32_t>& vindex_) {
74     svint32_t vindex =
75         svasrd_n_s32_x(ptrue, svmul_s32_x(ptrue, vindex_, svdup_n_s32(scale)), 2);
76     return svld1_gather_s32index_f32(ptrue, base_addr, vindex);
77   }

79   // ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ MASK GATHER ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~
80   
81   template <int64_t scale = 1>
82   std::
83       enable_if_t<scale == 1 || scale == 2 || scale == 4 || scale == 8, Vectorized<double>> inline mask_gather(
84           const Vectorized<double>& src,
85           const double* base_addr,
86           const Vectorized<int64_t>& vindex_,
87           const Vectorized<double>& mask_) {
88     svbool_t mask =
89         svcmpeq_s64(ptrue, svreinterpret_s64_f64(mask_), ALL_S64_TRUE_MASK);
90     svint64_t vindex =
91         svasrd_n_s64_x(ptrue, svmul_s64_x(ptrue, vindex_, svdup_n_s64(scale)), 3);
92     return svsel_f64(
93         mask, svld1_gather_s64index_f64(mask, base_addr, vindex), src);
94   }
95   
96   template <int64_t scale = 1>
97   std::
98       enable_if_t<scale == 1 || scale == 2 || scale == 4 || scale == 8, Vectorized<float>> inline mask_gather(
99           const Vectorized<float>& src,
100          const float* base_addr,
101          const Vectorized<int32_t>& vindex_,
102          const Vectorized<float>& mask_) {
103    svbool_t mask =
104        svcmpeq_s32(ptrue, svreinterpret_s32_f32(mask_), ALL_S32_TRUE_MASK);
105    svint32_t vindex =
106        svasrd_n_s32_x(ptrue, svmul_s32_x(ptrue, vindex_, svdup_n_s32(scale)), 2);
107    return svsel_f32(
108        mask, svld1_gather_s32index_f32(mask, base_addr, vindex), src);
109  }

113  // Only works for inputs in the range: [-2^51, 2^51]
114  // From: https://stackoverflow.com/a/41148578
115  template <>
116  Vectorized<int64_t> inline convert_to_int_of_same_size<double>(
117      const Vectorized<double>& src) {
118    svfloat64_t x = svadd_f64_x(ptrue, src, svdup_n_f64(0x0018000000000000));
119    return svsub_s64_x(
120        ptrue,
121        svreinterpret_s64_f64(x),
122        svreinterpret_s64_f64(svdup_n_f64(0x0018000000000000)));
123  }
124  
125  template <>
126  Vectorized<int32_t> inline convert_to_int_of_same_size<float>(
127      const Vectorized<float>& src) {
128    return svcvt_s32_f32_x(ptrue, src);
129  }

133  template <>
134  std::pair<Vectorized<double>, Vectorized<double>> inline interleave2<double>(
135      const Vectorized<double>& a,
136      const Vectorized<double>& b) {
137    // inputs:
138    //   a = {a0, a1, a3, a3}
139    //   b = {b0, b1, b2, b3}
140    // group cols crossing lanes:
141    //   return {a0, b0, a1, b1}
142    //          {a2, b2, a3, b3}
143    return std::make_pair(
144        Vectorized<double>(svzip1_f64(a, b)),
145        Vectorized<double>(svzip2_f64(a, b)));
146  }
147  
148  template <>
149  std::pair<Vectorized<float>, Vectorized<float>> inline interleave2<float>(
150      const Vectorized<float>& a,
151      const Vectorized<float>& b) {
152    // inputs:
153    //   a = {a0, a1, a2, a3, a4, a5, a6, a7}
154    //   b = {b0, b1, b2, b3, b4, b5, b6, b7}
155    // group cols crossing lanes:
156    //   return {a0, b0, a1, b1, a2, b2, a3, b3}
157    //          {a4, b4, a5, b5, a6, b6, a7, b7}
158    return std::make_pair(
159        Vectorized<float>(svzip1_f32(a, b)), Vectorized<float>(svzip2_f32(a, b)));
160  }

183  
184  template <>
185  std::pair<Vectorized<double>, Vectorized<double>> inline deinterleave2<double>(
186      const Vectorized<double>& a,
187      const Vectorized<double>& b) {
188    // inputs:
189    //   a = {a0, b0, a1, b1}
190    //   b = {a2, b2, a3, b3}
191    // swap lanes:
192    //   return {a0, a1, a2, a3}
193    //          {b0, b1, b2, b3}
194    return std::make_pair(
195        Vectorized<double>(svuzp1_f64(a, b)),
196        Vectorized<double>(svuzp2_f64(a, b)));
197  }

199  template <>
200  std::pair<Vectorized<float>, Vectorized<float>> inline deinterleave2<float>(
201      const Vectorized<float>& a,
202      const Vectorized<float>& b) {
203    // inputs:
204    //   a = {a0, b0, a1, b1, a2, b2, a3, b3}
205    //   b = {a4, b4, a5, b5, a6, b6, a7, b7}
206    // swap lanes:
207    //   return {a0, a1, a2, a3, a4, a5, a6, a7}
208    //          {b0, b1, b2, b3, b4, b5, b6, b7}
209    return std::make_pair(
210        Vectorized<float>(svuzp1_f32(a, b)), Vectorized<float>(svuzp2_f32(a, b)));
211  }
