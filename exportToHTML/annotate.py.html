2    from torch.fx.proxy import Proxy

4    from ._compatibility import compatibility

7    @compatibility(is_backward_compatible=False)
8    def annotate(val, type):
9        """ 
10       Annotates a Proxy object with a given type. 
11    
12       This function annotates a val with a given type if a type of the val is a torch.fx.Proxy object 
13       Args: 
14           val (object): An object to be annotated if its type is torch.fx.Proxy. 
15           type (object): A type to be assigned to a given proxy object as val. 
16       Returns: 
17           The given val. 
18       Raises: 
19           RuntimeError: If a val already has a type in its node. 
20       """
21       if isinstance(val, Proxy):
22           if val.node.type:
23               raise RuntimeError(
24                   f"Tried to annotate a value that already had a type on it!"
25                   f" Existing type is {val.node.type} "
26                   f"and new type is {type}. "
27                   f"This could happen if you tried to annotate a function parameter "
28                   f"value (in which case you should use the type slot "
29                   f"on the function signature) or you called "
30                   f"annotate on the same value twice"
31               )
32           else:
33               val.node.type = type
34           return val
35       else:
36           return val
