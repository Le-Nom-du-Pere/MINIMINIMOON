56   def AZ(s=None):
57       """Return the letters of ``s`` in uppercase. In case more than 
58       one string is passed, each of them will be processed and a list 
59       of upper case strings will be returned. 
60    
61       Examples 
62       ======== 
63    
64       >>> from sympy.crypto.crypto import AZ 
65       >>> AZ('Hello, world!') 
66       'HELLOWORLD' 
67       >>> AZ('Hello, world!'.split()) 
68       ['HELLO', 'WORLD'] 
69    
70       See Also 
71       ======== 
72    
73       check_and_join 
74    
75       """
76       if not s:
77           return uppercase
78       t = isinstance(s, str)
79       if t:
80           s = [s]
81       rv = [check_and_join(i.upper().split(), uppercase, filter=True)
82           for i in s]
83       if t:
84           return rv[0]
85       return rv

92   def padded_key(key, symbols):
93       """Return a string of the distinct characters of ``symbols`` with 
94       those of ``key`` appearing first. A ValueError is raised if 
95       a) there are duplicate characters in ``symbols`` or 
96       b) there are characters in ``key`` that are  not in ``symbols``. 
97    
98       Examples 
99       ======== 
100   
101      >>> from sympy.crypto.crypto import padded_key 
102      >>> padded_key('PUPPY', 'OPQRSTUVWXY') 
103      'PUYOQRSTVWX' 
104      >>> padded_key('RSA', 'ARTIST') 
105      Traceback (most recent call last): 
106      ... 
107      ValueError: duplicate characters in symbols: T 
108   
109      """
110      syms = list(uniq(symbols))
111      if len(syms) != len(symbols):
112          extra = ''.join(sorted({
113              i for i in symbols if symbols.count(i) > 1}))
114          raise ValueError('duplicate characters in symbols: %s' % extra)
115      extra = set(key) - set(syms)
116      if extra:
117          raise ValueError(
118              'characters in key but not symbols: %s' % ''.join(
119              sorted(extra)))
120      key0 = ''.join(list(uniq(key)))
121      # remove from syms characters in key0
122      return key0 + translate(''.join(syms), None, key0)

125  def check_and_join(phrase, symbols=None, filter=None):
126      """ 
127      Joins characters of ``phrase`` and if ``symbols`` is given, raises 
128      an error if any character in ``phrase`` is not in ``symbols``. 
129   
130      Parameters 
131      ========== 
132   
133      phrase 
134          String or list of strings to be returned as a string. 
135   
136      symbols 
137          Iterable of characters allowed in ``phrase``. 
138   
139          If ``symbols`` is ``None``, no checking is performed. 
140   
141      Examples 
142      ======== 
143   
144      >>> from sympy.crypto.crypto import check_and_join 
145      >>> check_and_join('a phrase') 
146      'a phrase' 
147      >>> check_and_join('a phrase'.upper().split()) 
148      'APHRASE' 
149      >>> check_and_join('a phrase!'.upper().split(), 'ARE', filter=True) 
150      'ARAE' 
151      >>> check_and_join('a phrase!'.upper().split(), 'ARE') 
152      Traceback (most recent call last): 
153      ... 
154      ValueError: characters in phrase but not symbols: "!HPS" 
155   
156      """
157      rv = ''.join(''.join(phrase))
158      if symbols is not None:
159          symbols = check_and_join(symbols)
160          missing = ''.join(sorted(set(rv) - set(symbols)))
161          if missing:
162              if not filter:
163                  raise ValueError(
164                      'characters in phrase but not symbols: "%s"' % missing)
165              rv = translate(rv, None, missing)
166      return rv

169  def _prep(msg, key, alp, default=None):
170      if not alp:
171          if not default:
172              alp = AZ()
173              msg = AZ(msg)
174              key = AZ(key)
175          else:
176              alp = default
177      else:
178          alp = ''.join(alp)
179      key = check_and_join(key, alp, filter=True)
180      msg = check_and_join(msg, alp, filter=True)
181      return msg, key, alp

184  def cycle_list(k, n):
185      """ 
186      Returns the elements of the list ``range(n)`` shifted to the 
187      left by ``k`` (so the list starts with ``k`` (mod ``n``)). 
188   
189      Examples 
190      ======== 
191   
192      >>> from sympy.crypto.crypto import cycle_list 
193      >>> cycle_list(3, 10) 
194      [3, 4, 5, 6, 7, 8, 9, 0, 1, 2] 
195   
196      """
197      k = k % n
198      return list(range(k, n)) + list(range(k))

204  def encipher_shift(msg, key, symbols=None):
205      """ 
206      Performs shift cipher encryption on plaintext msg, and returns the 
207      ciphertext. 
208   
209      Parameters 
210      ========== 
211   
212      key : int 
213          The secret key. 
214   
215      msg : str 
216          Plaintext of upper-case letters. 
217   
218      Returns 
219      ======= 
220   
221      str 
222          Ciphertext of upper-case letters. 
223   
224      Examples 
225      ======== 
226   
227      >>> from sympy.crypto.crypto import encipher_shift, decipher_shift 
228      >>> msg = "GONAVYBEATARMY" 
229      >>> ct = encipher_shift(msg, 1); ct 
230      'HPOBWZCFBUBSNZ' 
231   
232      To decipher the shifted text, change the sign of the key: 
233   
234      >>> encipher_shift(ct, -1) 
235      'GONAVYBEATARMY' 
236   
237      There is also a convenience function that does this with the 
238      original key: 
239   
240      >>> decipher_shift(ct, 1) 
241      'GONAVYBEATARMY' 
242   
243      Notes 
244      ===== 
245   
246      ALGORITHM: 
247   
248          STEPS: 
249              0. Number the letters of the alphabet from 0, ..., N 
250              1. Compute from the string ``msg`` a list ``L1`` of 
251                 corresponding integers. 
252              2. Compute from the list ``L1`` a new list ``L2``, given by 
253                 adding ``(k mod 26)`` to each element in ``L1``. 
254              3. Compute from the list ``L2`` a string ``ct`` of 
255                 corresponding letters. 
256   
257      The shift cipher is also called the Caesar cipher, after 
258      Julius Caesar, who, according to Suetonius, used it with a 
259      shift of three to protect messages of military significance. 
260      Caesar's nephew Augustus reportedly used a similar cipher, but 
261      with a right shift of 1. 
262   
263      References 
264      ========== 
265   
266      .. [1] https://en.wikipedia.org/wiki/Caesar_cipher 
267      .. [2] https://mathworld.wolfram.com/CaesarsMethod.html 
268   
269      See Also 
270      ======== 
271   
272      decipher_shift 
273   
274      """
275      msg, _, A = _prep(msg, '', symbols)
276      shift = len(A) - key % len(A)
277      key = A[shift:] + A[:shift]
278      return translate(msg, key, A)

298   
299      Or use this function with the original key: 
300   
301      >>> decipher_shift(ct, 1) 
302      'GONAVYBEATARMY' 
303   
304      """

307  def encipher_rot13(msg, symbols=None):
308      """ 
309      Performs the ROT13 encryption on a given plaintext ``msg``. 
310   
311      Explanation 
312      =========== 
313   
314      ROT13 is a substitution cipher which substitutes each letter 
315      in the plaintext message for the letter furthest away from it 
316      in the English alphabet. 
317   
318      Equivalently, it is just a Caeser (shift) cipher with a shift 
319      key of 13 (midway point of the alphabet). 
320   
321      References 
322      ========== 
323   
324      .. [1] https://en.wikipedia.org/wiki/ROT13 
325   
326      See Also 
327      ======== 
328   
329      decipher_rot13 
330      encipher_shift 
331   
332      """
333      return encipher_shift(msg, 13, symbols)

343      ``decipher_shift`` with a key of 13 and ``encipher_shift`` key with a

368  def encipher_affine(msg, key, symbols=None, _inverse=False):
369      r""" 
370      Performs the affine cipher encryption on plaintext ``msg``, and 
371      returns the ciphertext. 
372   
373      Explanation 
374      =========== 
375   
376      Encryption is based on the map `x \rightarrow ax+b` (mod `N`) 
377      where ``N`` is the number of characters in the alphabet. 
378      Decryption is based on the map `x \rightarrow cx+d` (mod `N`), 
379      where `c = a^{-1}` (mod `N`) and `d = -a^{-1}b` (mod `N`). 
380      In particular, for the map to be invertible, we need 
381      `\mathrm{gcd}(a, N) = 1` and an error will be raised if this is 
382      not true. 
383   
384      Parameters 
385      ========== 
386   
387      msg : str 
388          Characters that appear in ``symbols``. 
389   
390      a, b : int, int 
391          A pair integers, with ``gcd(a, N) = 1`` (the secret key). 
392   
393      symbols 
394          String of characters (default = uppercase letters). 
395   
396          When no symbols are given, ``msg`` is converted to upper case 
397          letters and all other characters are ignored. 
398   
399      Returns 
400      ======= 
401   
402      ct 
403          String of characters (the ciphertext message) 
404   
405      Notes 
406      ===== 
407   
408      ALGORITHM: 
409   
410          STEPS: 
411              0. Number the letters of the alphabet from 0, ..., N 
412              1. Compute from the string ``msg`` a list ``L1`` of 
413                 corresponding integers. 
414              2. Compute from the list ``L1`` a new list ``L2``, given by 
415                 replacing ``x`` by ``a*x + b (mod N)``, for each element 
416                 ``x`` in ``L1``. 
417              3. Compute from the list ``L2`` a string ``ct`` of 
418                 corresponding letters. 
419   
420      This is a straightforward generalization of the shift cipher with 
421      the added complexity of requiring 2 characters to be deciphered in 
422      order to recover the key. 
423   
424      References 
425      ========== 
426   
427      .. [1] https://en.wikipedia.org/wiki/Affine_cipher 
428   
429      See Also 
430      ======== 
431   
432      decipher_affine 
433   
434      """
435      msg, _, A = _prep(msg, '', symbols)
436      N = len(A)
437      a, b = key
438      assert gcd(a, N) == 1
439      if _inverse:
440          c = invert(a, N)
441          d = -b*c
442          a, b = c, d
443      B = ''.join([A[(a*i + b) % N] for i in range(N)])
444      return translate(msg, A, B)

539  def encipher_substitution(msg, old, new=None):
540      r""" 
541      Returns the ciphertext obtained by replacing each character that 
542      appears in ``old`` with the corresponding character in ``new``. 
543      If ``old`` is a mapping, then new is ignored and the replacements 
544      defined by ``old`` are used. 
545   
546      Explanation 
547      =========== 
548   
549      This is a more general than the affine cipher in that the key can 
550      only be recovered by determining the mapping for each symbol. 
551      Though in practice, once a few symbols are recognized the mappings 
552      for other characters can be quickly guessed. 
553   
554      Examples 
555      ======== 
556   
557      >>> from sympy.crypto.crypto import encipher_substitution, AZ 
558      >>> old = 'OEYAG' 
559      >>> new = '034^6' 
560      >>> msg = AZ("go navy! beat army!") 
561      >>> ct = encipher_substitution(msg, old, new); ct 
562      '60N^V4B3^T^RM4' 
563   
564      To decrypt a substitution, reverse the last two arguments: 
565   
566      >>> encipher_substitution(ct, new, old) 
567      'GONAVYBEATARMY' 
568   
569      In the special case where ``old`` and ``new`` are a permutation of 
570      order 2 (representing a transposition of characters) their order 
571      is immaterial: 
572   
573      >>> old = 'NAVY' 
574      >>> new = 'ANYV' 
575      >>> encipher = lambda x: encipher_substitution(x, old, new) 
576      >>> encipher('NAVY') 
577      'ANYV' 
578      >>> encipher(_) 
579      'NAVY' 
580   
581      The substitution cipher, in general, is a method 
582      whereby "units" (not necessarily single characters) of plaintext 
583      are replaced with ciphertext according to a regular system. 
584   
585      >>> ords = dict(zip('abc', ['\\%i' % ord(i) for i in 'abc'])) 
586      >>> print(encipher_substitution('abc', ords)) 
587      \97\98\99 
588   
589      References 
590      ========== 
591   
592      .. [1] https://en.wikipedia.org/wiki/Substitution_cipher 
593   
594      """
595      return translate(msg, old, new)

602  def encipher_vigenere(msg, key, symbols=None):
603      """ 
604      Performs the Vigenere cipher encryption on plaintext ``msg``, and 
605      returns the ciphertext. 
606   
607      Examples 
608      ======== 
609   
610      >>> from sympy.crypto.crypto import encipher_vigenere, AZ 
611      >>> key = "encrypt" 
612      >>> msg = "meet me on monday" 
613      >>> encipher_vigenere(msg, key) 
614      'QRGKKTHRZQEBPR' 
615   
616      Section 1 of the Kryptos sculpture at the CIA headquarters 
617      uses this cipher and also changes the order of the 
618      alphabet [2]_. Here is the first line of that section of 
619      the sculpture: 
620   
621      >>> from sympy.crypto.crypto import decipher_vigenere, padded_key 
622      >>> alp = padded_key('KRYPTOS', AZ()) 
623      >>> key = 'PALIMPSEST' 
624      >>> msg = 'EMUFPHZLRFAXYUSDJKZLDKRNSHGNFIVJ' 
625      >>> decipher_vigenere(msg, key, alp) 
626      'BETWEENSUBTLESHADINGANDTHEABSENC' 
627   
628      Explanation 
629      =========== 
630   
631      The Vigenere cipher is named after Blaise de Vigenere, a sixteenth 
632      century diplomat and cryptographer, by a historical accident. 
633      Vigenere actually invented a different and more complicated cipher. 
634      The so-called *Vigenere cipher* was actually invented 
635      by Giovan Batista Belaso in 1553. 
636   
637      This cipher was used in the 1800's, for example, during the American 
638      Civil War. The Confederacy used a brass cipher disk to implement the 
639      Vigenere cipher (now on display in the NSA Museum in Fort 
640      Meade) [1]_. 
641   
642      The Vigenere cipher is a generalization of the shift cipher. 
643      Whereas the shift cipher shifts each letter by the same amount 
644      (that amount being the key of the shift cipher) the Vigenere 
645      cipher shifts a letter by an amount determined by the key (which is 
646      a word or phrase known only to the sender and receiver). 
647   
648      For example, if the key was a single letter, such as "C", then the 
649      so-called Vigenere cipher is actually a shift cipher with a 
650      shift of `2` (since "C" is the 2nd letter of the alphabet, if 
651      you start counting at `0`). If the key was a word with two 
652      letters, such as "CA", then the so-called Vigenere cipher will 
653      shift letters in even positions by `2` and letters in odd positions 
654      are left alone (shifted by `0`, since "A" is the 0th letter, if 
655      you start counting at `0`). 
656   
657   
658      ALGORITHM: 
659   
660          INPUT: 
661   
662              ``msg``: string of characters that appear in ``symbols`` 
663              (the plaintext) 
664   
665              ``key``: a string of characters that appear in ``symbols`` 
666              (the secret key) 
667   
668              ``symbols``: a string of letters defining the alphabet 
669   
670   
671          OUTPUT: 
672   
673              ``ct``: string of characters (the ciphertext message) 
674   
675          STEPS: 
676              0. Number the letters of the alphabet from 0, ..., N 
677              1. Compute from the string ``key`` a list ``L1`` of 
678                 corresponding integers. Let ``n1 = len(L1)``. 
679              2. Compute from the string ``msg`` a list ``L2`` of 
680                 corresponding integers. Let ``n2 = len(L2)``. 
681              3. Break ``L2`` up sequentially into sublists of size 
682                 ``n1``; the last sublist may be smaller than ``n1`` 
683              4. For each of these sublists ``L`` of ``L2``, compute a 
684                 new list ``C`` given by ``C[i] = L[i] + L1[i] (mod N)`` 
685                 to the ``i``-th element in the sublist, for each ``i``. 
686              5. Assemble these lists ``C`` by concatenation into a new 
687                 list of length ``n2``. 
688              6. Compute from the new list a string ``ct`` of 
689                 corresponding letters. 
690   
691      Once it is known that the key is, say, `n` characters long, 
692      frequency analysis can be applied to every `n`-th letter of 
693      the ciphertext to determine the plaintext. This method is 
694      called *Kasiski examination* (although it was first discovered 
695      by Babbage). If they key is as long as the message and is 
696      comprised of randomly selected characters -- a one-time pad -- the 
697      message is theoretically unbreakable. 
698   
699      The cipher Vigenere actually discovered is an "auto-key" cipher 
700      described as follows. 
701   
702      ALGORITHM: 
703   
704          INPUT: 
705   
706            ``key``: a string of letters (the secret key) 
707   
708            ``msg``: string of letters (the plaintext message) 
709   
710          OUTPUT: 
711   
712            ``ct``: string of upper-case letters (the ciphertext message) 
713   
714          STEPS: 
715              0. Number the letters of the alphabet from 0, ..., N 
716              1. Compute from the string ``msg`` a list ``L2`` of 
717                 corresponding integers. Let ``n2 = len(L2)``. 
718              2. Let ``n1`` be the length of the key. Append to the 
719                 string ``key`` the first ``n2 - n1`` characters of 
720                 the plaintext message. Compute from this string (also of 
721                 length ``n2``) a list ``L1`` of integers corresponding 
722                 to the letter numbers in the first step. 
723              3. Compute a new list ``C`` given by 
724                 ``C[i] = L1[i] + L2[i] (mod N)``. 
725              4. Compute from the new list a string ``ct`` of letters 
726                 corresponding to the new integers. 
727   
728      To decipher the auto-key ciphertext, the key is used to decipher 
729      the first ``n1`` characters and then those characters become the 
730      key to  decipher the next ``n1`` characters, etc...: 
731   
732      >>> m = AZ('go navy, beat army! yes you can'); m 
733      'GONAVYBEATARMYYESYOUCAN' 
734      >>> key = AZ('gold bug'); n1 = len(key); n2 = len(m) 
735      >>> auto_key = key + m[:n2 - n1]; auto_key 
736      'GOLDBUGGONAVYBEATARMYYE' 
737      >>> ct = encipher_vigenere(m, auto_key); ct 
738      'MCYDWSHKOGAMKZCELYFGAYR' 
739      >>> n1 = len(key) 
740      >>> pt = [] 
741      >>> while ct: 
742      ...     part, ct = ct[:n1], ct[n1:] 
743      ...     pt.append(decipher_vigenere(part, key)) 
744      ...     key = pt[-1] 
745      ... 
746      >>> ''.join(pt) == m 
747      True 
748   
749      References 
750      ========== 
751   
752      .. [1] https://en.wikipedia.org/wiki/Vigenere_cipher 
753      .. [2] https://web.archive.org/web/20071116100808/https://filebox.vt.edu/users/batman/kryptos.html 
754         (short URL: https://goo.gl/ijr22d) 
755   
756      """
757      msg, key, A = _prep(msg, key, symbols)
758      map = {c: i for i, c in enumerate(A)}
759      key = [map[c] for c in key]
760      N = len(map)
761      k = len(key)
762      rv = []
763      for i, m in enumerate(msg):
764          rv.append(A[(map[m] + key[i % k]) % N])
765      rv = ''.join(rv)
766      return rv

771      Decode using the Vigenere cipher. 
772   
773      Examples

796  def encipher_hill(msg, key, symbols=None, pad="Q"):
797      r""" 
798      Return the Hill cipher encryption of ``msg``. 
799   
800      Explanation 
801      =========== 
802   
803      The Hill cipher [1]_, invented by Lester S. Hill in the 1920's [2]_, 
804      was the first polygraphic cipher in which it was practical 
805      (though barely) to operate on more than three symbols at once. 
806      The following discussion assumes an elementary knowledge of 
807      matrices. 
808   
809      First, each letter is first encoded as a number starting with 0. 
810      Suppose your message `msg` consists of `n` capital letters, with no 
811      spaces. This may be regarded an `n`-tuple M of elements of 
812      `Z_{26}` (if the letters are those of the English alphabet). A key 
813      in the Hill cipher is a `k x k` matrix `K`, all of whose entries 
814      are in `Z_{26}`, such that the matrix `K` is invertible (i.e., the 
815      linear transformation `K: Z_{N}^k \rightarrow Z_{N}^k` 
816      is one-to-one). 
817   
818   
819      Parameters 
820      ========== 
821   
822      msg 
823          Plaintext message of `n` upper-case letters. 
824   
825      key 
826          A `k \times k` invertible matrix `K`, all of whose entries are 
827          in `Z_{26}` (or whatever number of symbols are being used). 
828   
829      pad 
830          Character (default "Q") to use to make length of text be a 
831          multiple of ``k``. 
832   
833      Returns 
834      ======= 
835   
836      ct 
837          Ciphertext of upper-case letters. 
838   
839      Notes 
840      ===== 
841   
842      ALGORITHM: 
843   
844          STEPS: 
845              0. Number the letters of the alphabet from 0, ..., N 
846              1. Compute from the string ``msg`` a list ``L`` of 
847                 corresponding integers. Let ``n = len(L)``. 
848              2. Break the list ``L`` up into ``t = ceiling(n/k)`` 
849                 sublists ``L_1``, ..., ``L_t`` of size ``k`` (with 
850                 the last list "padded" to ensure its size is 
851                 ``k``). 
852              3. Compute new list ``C_1``, ..., ``C_t`` given by 
853                 ``C[i] = K*L_i`` (arithmetic is done mod N), for each 
854                 ``i``. 
855              4. Concatenate these into a list ``C = C_1 + ... + C_t``. 
856              5. Compute from ``C`` a string ``ct`` of corresponding 
857                 letters. This has length ``k*t``. 
858   
859      References 
860      ========== 
861   
862      .. [1] https://en.wikipedia.org/wiki/Hill_cipher 
863      .. [2] Lester S. Hill, Cryptography in an Algebraic Alphabet, 
864         The American Mathematical Monthly Vol.36, June-July 1929, 
865         pp.306-312. 
866   
867      See Also 
868      ======== 
869   
870      decipher_hill 
871   
872      """
873      assert key.is_square
874      assert len(pad) == 1
875      msg, pad, A = _prep(msg, pad, symbols)
876      map = {c: i for i, c in enumerate(A)}
877      P = [map[c] for c in msg]
878      N = len(A)
879      k = key.cols
880      n = len(P)
881      m, r = divmod(n, k)
882      if r:
883          P = P + [map[pad]]*(k - r)
884          m += 1
885      rv = ''.join([A[c % N] for j in range(m) for c in
886          list(key*Matrix(k, 1, [P[i]
887          for i in range(k*j, k*(j + 1))]))])
888      return rv
889

891  def decipher_hill(msg, key, symbols=None):
892      """ 
893      Deciphering is the same as enciphering but using the inverse of the 
894      key matrix. 
895   
896      Examples 
897      ======== 
898   
899      >>> from sympy.crypto.crypto import encipher_hill, decipher_hill 
900      >>> from sympy import Matrix 
901   
902      >>> key = Matrix([[1, 2], [3, 5]]) 
903      >>> encipher_hill("meet me on monday", key) 
904      'UEQDUEODOCTCWQ' 
905      >>> decipher_hill(_, key) 
906      'MEETMEONMONDAY' 
907   
908      When the length of the plaintext (stripped of invalid characters) 
909      is not a multiple of the key dimension, extra characters will 
910      appear at the end of the enciphered and deciphered text. In order to 
911      decipher the text, those characters must be included in the text to 
912      be deciphered. In the following, the key has a dimension of 4 but 
913      the text is 2 short of being a multiple of 4 so two characters will 
914      be added. 
915   
916      >>> key = Matrix([[1, 1, 1, 2], [0, 1, 1, 0], 
917      ...               [2, 2, 3, 4], [1, 1, 0, 1]]) 
918      >>> msg = "ST" 
919      >>> encipher_hill(msg, key) 
920      'HJEB' 
921      >>> decipher_hill(_, key) 
922      'STQQ' 
923      >>> encipher_hill(msg, key, pad="Z") 
924      'ISPK' 
925      >>> decipher_hill(_, key) 
926      'STZZ' 
927   
928      If the last two characters of the ciphertext were ignored in 
929      either case, the wrong plaintext would be recovered: 
930   
931      >>> decipher_hill("HD", key) 
932      'ORMV' 
933      >>> decipher_hill("IS", key) 
934      'UIKY' 
935   
936      See Also 
937      ======== 
938   
939      encipher_hill 
940   
941      """
942      assert key.is_square
943      msg, _, A = _prep(msg, '', symbols)
944      map = {c: i for i, c in enumerate(A)}
945      C = [map[c] for c in msg]
946      N = len(A)
947      k = key.cols
948      n = len(C)
949      m, r = divmod(n, k)
950      if r:
951          C = C + [0]*(k - r)
952          m += 1
953      key_inv = key.inv_mod(N)
954      rv = ''.join([A[p % N] for j in range(m) for p in
955          list(key_inv*Matrix(
956          k, 1, [C[i] for i in range(k*j, k*(j + 1))]))])
957      return rv

963  def encipher_bifid(msg, key, symbols=None):
964      r""" 
965      Performs the Bifid cipher encryption on plaintext ``msg``, and 
966      returns the ciphertext. 
967   
968      This is the version of the Bifid cipher that uses an `n \times n` 
969      Polybius square. 
970   
971      Parameters 
972      ========== 
973   
974      msg 
975          Plaintext string. 
976   
977      key 
978          Short string for key. 
979   
980          Duplicate characters are ignored and then it is padded with the 
981          characters in ``symbols`` that were not in the short key. 
982   
983      symbols 
984          `n \times n` characters defining the alphabet. 
985   
986          (default is string.printable) 
987   
988      Returns 
989      ======= 
990   
991      ciphertext 
992          Ciphertext using Bifid5 cipher without spaces. 
993   
994      See Also 
995      ======== 
996   
997      decipher_bifid, encipher_bifid5, encipher_bifid6 
998   
999      References 
1000     ========== 
1001  
1002     .. [1] https://en.wikipedia.org/wiki/Bifid_cipher 
1003  
1004     """
1005     msg, key, A = _prep(msg, key, symbols, bifid10)
1006     long_key = ''.join(uniq(key)) or A
1007 
1008     n = len(A)**.5
1009     if n != int(n):
1010         raise ValueError(
1011             'Length of alphabet (%s) is not a square number.' % len(A))
1012     N = int(n)
1013     if len(long_key) < N**2:
1014         long_key = list(long_key) + [x for x in A if x not in long_key]
1015 
1016     # the fractionalization
1017     row_col = {ch: divmod(i, N) for i, ch in enumerate(long_key)}
1018     r, c = zip(*[row_col[x] for x in msg])
1019     rc = r + c
1020     ch = {i: ch for ch, i in row_col.items()}
1021     rv = ''.join(ch[i] for i in zip(rc[::2], rc[1::2]))
1022     return rv

1025 def decipher_bifid(msg, key, symbols=None):
1026     r""" 
1027     Performs the Bifid cipher decryption on ciphertext ``msg``, and 
1028     returns the plaintext. 
1029  
1030     This is the version of the Bifid cipher that uses the `n \times n` 
1031     Polybius square. 
1032  
1033     Parameters 
1034     ========== 
1035  
1036     msg 
1037         Ciphertext string. 
1038  
1039     key 
1040         Short string for key. 
1041  
1042         Duplicate characters are ignored and then it is padded with the 
1043         characters in symbols that were not in the short key. 
1044  
1045     symbols 
1046         `n \times n` characters defining the alphabet. 
1047  
1048         (default=string.printable, a `10 \times 10` matrix) 
1049  
1050     Returns 
1051     ======= 
1052  
1053     deciphered 
1054         Deciphered text. 
1055  
1056     Examples 
1057     ======== 
1058  
1059     >>> from sympy.crypto.crypto import ( 
1060     ...     encipher_bifid, decipher_bifid, AZ) 
1061  
1062     Do an encryption using the bifid5 alphabet: 
1063  
1064     >>> alp = AZ().replace('J', '') 
1065     >>> ct = AZ("meet me on monday!") 
1066     >>> key = AZ("gold bug") 
1067     >>> encipher_bifid(ct, key, alp) 
1068     'IEILHHFSTSFQYE' 
1069  
1070     When entering the text or ciphertext, spaces are ignored so it 
1071     can be formatted as desired. Re-entering the ciphertext from the 
1072     preceding, putting 4 characters per line and padding with an extra 
1073     J, does not cause problems for the deciphering: 
1074  
1075     >>> decipher_bifid(''' 
1076     ... IEILH 
1077     ... HFSTS 
1078     ... FQYEJ''', key, alp) 
1079     'MEETMEONMONDAY' 
1080  
1081     When no alphabet is given, all 100 printable characters will be 
1082     used: 
1083  
1084     >>> key = '' 
1085     >>> encipher_bifid('hello world!', key) 
1086     'bmtwmg-bIo*w' 
1087     >>> decipher_bifid(_, key) 
1088     'hello world!' 
1089  
1090     If the key is changed, a different encryption is obtained: 
1091  
1092     >>> key = 'gold bug' 
1093     >>> encipher_bifid('hello world!', 'gold_bug') 
1094     'hg2sfuei7t}w' 
1095  
1096     And if the key used to decrypt the message is not exact, the 
1097     original text will not be perfectly obtained: 
1098  
1099     >>> decipher_bifid(_, 'gold pug') 
1100     'heldo~wor6d!' 
1101  
1102     """
1103     msg, _, A = _prep(msg, '', symbols, bifid10)
1104     long_key = ''.join(uniq(key)) or A
1105 
1106     n = len(A)**.5
1107     if n != int(n):
1108         raise ValueError(
1109             'Length of alphabet (%s) is not a square number.' % len(A))
1110     N = int(n)
1111     if len(long_key) < N**2:
1112         long_key = list(long_key) + [x for x in A if x not in long_key]
1113 
1114     # the reverse fractionalization
1115     row_col = {
1116         ch: divmod(i, N) for i, ch in enumerate(long_key)}
1117     rc = [i for c in msg for i in row_col[c]]
1118     n = len(msg)
1119     rc = zip(*(rc[:n], rc[n:]))
1120     ch = {i: ch for ch, i in row_col.items()}
1121     rv = ''.join(ch[i] for i in rc)
1122     return rv

1166 def encipher_bifid5(msg, key):
1167     r""" 
1168     Performs the Bifid cipher encryption on plaintext ``msg``, and 
1169     returns the ciphertext. 
1170  
1171     Explanation 
1172     =========== 
1173  
1174     This is the version of the Bifid cipher that uses the `5 \times 5` 
1175     Polybius square. The letter "J" is ignored so it must be replaced 
1176     with something else (traditionally an "I") before encryption. 
1177  
1178     ALGORITHM: (5x5 case) 
1179  
1180         STEPS: 
1181             0. Create the `5 \times 5` Polybius square ``S`` associated 
1182                to ``key`` as follows: 
1183  
1184                 a) moving from left-to-right, top-to-bottom, 
1185                    place the letters of the key into a `5 \times 5` 
1186                    matrix, 
1187                 b) if the key has less than 25 letters, add the 
1188                    letters of the alphabet not in the key until the 
1189                    `5 \times 5` square is filled. 
1190  
1191             1. Create a list ``P`` of pairs of numbers which are the 
1192                coordinates in the Polybius square of the letters in 
1193                ``msg``. 
1194             2. Let ``L1`` be the list of all first coordinates of ``P`` 
1195                (length of ``L1 = n``), let ``L2`` be the list of all 
1196                second coordinates of ``P`` (so the length of ``L2`` 
1197                is also ``n``). 
1198             3. Let ``L`` be the concatenation of ``L1`` and ``L2`` 
1199                (length ``L = 2*n``), except that consecutive numbers 
1200                are paired ``(L[2*i], L[2*i + 1])``. You can regard 
1201                ``L`` as a list of pairs of length ``n``. 
1202             4. Let ``C`` be the list of all letters which are of the 
1203                form ``S[i, j]``, for all ``(i, j)`` in ``L``. As a 
1204                string, this is the ciphertext of ``msg``. 
1205  
1206     Parameters 
1207     ========== 
1208  
1209     msg : str 
1210         Plaintext string. 
1211  
1212         Converted to upper case and filtered of anything but all letters 
1213         except J. 
1214  
1215     key 
1216         Short string for key; non-alphabetic letters, J and duplicated 
1217         characters are ignored and then, if the length is less than 25 
1218         characters, it is padded with other letters of the alphabet 
1219         (in alphabetical order). 
1220  
1221     Returns 
1222     ======= 
1223  
1224     ct 
1225         Ciphertext (all caps, no spaces). 
1226  
1227     Examples 
1228     ======== 
1229  
1230     >>> from sympy.crypto.crypto import ( 
1231     ...     encipher_bifid5, decipher_bifid5) 
1232  
1233     "J" will be omitted unless it is replaced with something else: 
1234  
1235     >>> round_trip = lambda m, k: \ 
1236     ...     decipher_bifid5(encipher_bifid5(m, k), k) 
1237     >>> key = 'a' 
1238     >>> msg = "JOSIE" 
1239     >>> round_trip(msg, key) 
1240     'OSIE' 
1241     >>> round_trip(msg.replace("J", "I"), key) 
1242     'IOSIE' 
1243     >>> j = "QIQ" 
1244     >>> round_trip(msg.replace("J", j), key).replace(j, "J") 
1245     'JOSIE' 
1246  
1247  
1248     Notes 
1249     ===== 
1250  
1251     The Bifid cipher was invented around 1901 by Felix Delastelle. 
1252     It is a *fractional substitution* cipher, where letters are 
1253     replaced by pairs of symbols from a smaller alphabet. The 
1254     cipher uses a `5 \times 5` square filled with some ordering of the 
1255     alphabet, except that "J" is replaced with "I" (this is a so-called 
1256     Polybius square; there is a `6 \times 6` analog if you add back in 
1257     "J" and also append onto the usual 26 letter alphabet, the digits 
1258     0, 1, ..., 9). 
1259     According to Helen Gaines' book *Cryptanalysis*, this type of cipher 
1260     was used in the field by the German Army during World War I. 
1261  
1262     See Also 
1263     ======== 
1264  
1265     decipher_bifid5, encipher_bifid 
1266  
1267     """
1268     msg, key, _ = _prep(msg.upper(), key.upper(), None, bifid5)
1269     key = padded_key(key, bifid5)
1270     return encipher_bifid(msg, '', key)
1271

1459 def _decipher_rsa_crt(i, d, factors):
1460     """Decipher RSA using chinese remainder theorem from the information 
1461     of the relatively-prime factors of the modulus. 
1462  
1463     Parameters 
1464     ========== 
1465  
1466     i : integer 
1467         Ciphertext 
1468  
1469     d : integer 
1470         The exponent component. 
1471  
1472     factors : list of relatively-prime integers 
1473         The integers given must be coprime and the product must equal 
1474         the modulus component of the original RSA key. 
1475  
1476     Examples 
1477     ======== 
1478  
1479     How to decrypt RSA with CRT: 
1480  
1481     >>> from sympy.crypto.crypto import rsa_public_key, rsa_private_key 
1482     >>> primes = [61, 53] 
1483     >>> e = 17 
1484     >>> args = primes + [e] 
1485     >>> puk = rsa_public_key(*args) 
1486     >>> prk = rsa_private_key(*args) 
1487  
1488     >>> from sympy.crypto.crypto import encipher_rsa, _decipher_rsa_crt 
1489     >>> msg = 65 
1490     >>> crt_primes = primes 
1491     >>> encrypted = encipher_rsa(msg, puk) 
1492     >>> decrypted = _decipher_rsa_crt(encrypted, prk[1], primes) 
1493     >>> decrypted 
1494     65 
1495     """
1496     moduluses = [pow(i, d, p) for p in factors]
1497 
1498     result = crt(factors, moduluses)
1499     if not result:
1500         raise ValueError("CRT failed")
1501     return result[0]

1504 def _rsa_key(*args, public=True, private=True, totient='Euler', index=None, multipower=None):
1505     r"""A private subroutine to generate RSA key 
1506  
1507     Parameters 
1508     ========== 
1509  
1510     public, private : bool, optional 
1511         Flag to generate either a public key, a private key. 
1512  
1513     totient : 'Euler' or 'Carmichael' 
1514         Different notation used for totient. 
1515  
1516     multipower : bool, optional 
1517         Flag to bypass warning for multipower RSA. 
1518     """
1519 
1520     if len(args) < 2:
1521         return False
1522 
1523     if totient not in ('Euler', 'Carmichael'):
1524         raise ValueError(
1525             "The argument totient={} should either be " \
1526             "'Euler', 'Carmichalel'." \
1527             .format(totient))
1528 
1529     if totient == 'Euler':
1530         _totient = _euler
1531     else:
1532         _totient = _carmichael
1533 
1534     if index is not None:
1535         index = as_int(index)
1536         if totient != 'Carmichael':
1537             raise ValueError(
1538                 "Setting the 'index' keyword argument requires totient"
1539                 "notation to be specified as 'Carmichael'.")
1540 
1541     primes, e = args[:-1], args[-1]
1542 
1543     if not all(isprime(p) for p in primes):
1544         new_primes = []
1545         for i in primes:
1546             new_primes.extend(factorint(i, multiple=True))
1547         primes = new_primes
1548 
1549     n = reduce(lambda i, j: i*j, primes)
1550 
1551     tally = multiset(primes)
1552     if all(v == 1 for v in tally.values()):
1553         phi = int(_totient(tally))
1554 
1555     else:
1556         if not multipower:
1557             NonInvertibleCipherWarning(
1558                 'Non-distinctive primes found in the factors {}. '
1559                 'The cipher may not be decryptable for some numbers '
1560                 'in the complete residue system Z[{}], but the cipher '
1561                 'can still be valid if you restrict the domain to be '
1562                 'the reduced residue system Z*[{}]. You can pass '
1563                 'the flag multipower=True if you want to suppress this '
1564                 'warning.'
1565                 .format(primes, n, n)
1566                 # stacklevel=4 because most users will call a function that
1567                 # calls this function
1568                 ).warn(stacklevel=4)
1569         phi = int(_totient(tally))
1570 
1571     if gcd(e, phi) == 1:
1572         if public and not private:
1573             if isinstance(index, int):
1574                 e = e % phi
1575                 e += index * phi
1576             return n, e
1577 
1578         if private and not public:
1579             d = invert(e, phi)
1580             if isinstance(index, int):
1581                 d += index * phi
1582             return n, d
1583 
1584     return False

1587 def rsa_public_key(*args, **kwargs):
1588     r"""Return the RSA *public key* pair, `(n, e)` 
1589  
1590     Parameters 
1591     ========== 
1592  
1593     args : naturals 
1594         If specified as `p, q, e` where `p` and `q` are distinct primes 
1595         and `e` is a desired public exponent of the RSA, `n = p q` and 
1596         `e` will be verified against the totient 
1597         `\phi(n)` (Euler totient) or `\lambda(n)` (Carmichael totient) 
1598         to be `\gcd(e, \phi(n)) = 1` or `\gcd(e, \lambda(n)) = 1`. 
1599  
1600         If specified as `p_1, p_2, \dots, p_n, e` where 
1601         `p_1, p_2, \dots, p_n` are specified as primes, 
1602         and `e` is specified as a desired public exponent of the RSA, 
1603         it will be able to form a multi-prime RSA, which is a more 
1604         generalized form of the popular 2-prime RSA. 
1605  
1606         It can also be possible to form a single-prime RSA by specifying 
1607         the argument as `p, e`, which can be considered a trivial case 
1608         of a multiprime RSA. 
1609  
1610         Furthermore, it can be possible to form a multi-power RSA by 
1611         specifying two or more pairs of the primes to be same. 
1612         However, unlike the two-distinct prime RSA or multi-prime 
1613         RSA, not every numbers in the complete residue system 
1614         (`\mathbb{Z}_n`) will be decryptable since the mapping 
1615         `\mathbb{Z}_{n} \rightarrow \mathbb{Z}_{n}` 
1616         will not be bijective. 
1617         (Only except for the trivial case when 
1618         `e = 1` 
1619         or more generally, 
1620  
1621         .. math:: 
1622             e \in \left \{ 1 + k \lambda(n) 
1623             \mid k \in \mathbb{Z} \land k \geq 0 \right \} 
1624  
1625         when RSA reduces to the identity.) 
1626         However, the RSA can still be decryptable for the numbers in the 
1627         reduced residue system (`\mathbb{Z}_n^{\times}`), since the 
1628         mapping 
1629         `\mathbb{Z}_{n}^{\times} \rightarrow \mathbb{Z}_{n}^{\times}` 
1630         can still be bijective. 
1631  
1632         If you pass a non-prime integer to the arguments 
1633         `p_1, p_2, \dots, p_n`, the particular number will be 
1634         prime-factored and it will become either a multi-prime RSA or a 
1635         multi-power RSA in its canonical form, depending on whether the 
1636         product equals its radical or not. 
1637         `p_1 p_2 \dots p_n = \text{rad}(p_1 p_2 \dots p_n)` 
1638  
1639     totient : bool, optional 
1640         If ``'Euler'``, it uses Euler's totient `\phi(n)` which is 
1641         :meth:`sympy.functions.combinatorial.numbers.totient` in SymPy. 
1642  
1643         If ``'Carmichael'``, it uses Carmichael's totient `\lambda(n)` 
1644         which is :meth:`sympy.functions.combinatorial.numbers.reduced_totient` in SymPy. 
1645  
1646         Unlike private key generation, this is a trivial keyword for 
1647         public key generation because 
1648         `\gcd(e, \phi(n)) = 1 \iff \gcd(e, \lambda(n)) = 1`. 
1649  
1650     index : nonnegative integer, optional 
1651         Returns an arbitrary solution of a RSA public key at the index 
1652         specified at `0, 1, 2, \dots`. This parameter needs to be 
1653         specified along with ``totient='Carmichael'``. 
1654  
1655         Similarly to the non-uniquenss of a RSA private key as described 
1656         in the ``index`` parameter documentation in 
1657         :meth:`rsa_private_key`, RSA public key is also not unique and 
1658         there is an infinite number of RSA public exponents which 
1659         can behave in the same manner. 
1660  
1661         From any given RSA public exponent `e`, there are can be an 
1662         another RSA public exponent `e + k \lambda(n)` where `k` is an 
1663         integer, `\lambda` is a Carmichael's totient function. 
1664  
1665         However, considering only the positive cases, there can be 
1666         a principal solution of a RSA public exponent `e_0` in 
1667         `0 < e_0 < \lambda(n)`, and all the other solutions 
1668         can be canonicalzed in a form of `e_0 + k \lambda(n)`. 
1669  
1670         ``index`` specifies the `k` notation to yield any possible value 
1671         an RSA public key can have. 
1672  
1673         An example of computing any arbitrary RSA public key: 
1674  
1675         >>> from sympy.crypto.crypto import rsa_public_key 
1676         >>> rsa_public_key(61, 53, 17, totient='Carmichael', index=0) 
1677         (3233, 17) 
1678         >>> rsa_public_key(61, 53, 17, totient='Carmichael', index=1) 
1679         (3233, 797) 
1680         >>> rsa_public_key(61, 53, 17, totient='Carmichael', index=2) 
1681         (3233, 1577) 
1682  
1683     multipower : bool, optional 
1684         Any pair of non-distinct primes found in the RSA specification 
1685         will restrict the domain of the cryptosystem, as noted in the 
1686         explanation of the parameter ``args``. 
1687  
1688         SymPy RSA key generator may give a warning before dispatching it 
1689         as a multi-power RSA, however, you can disable the warning if 
1690         you pass ``True`` to this keyword. 
1691  
1692     Returns 
1693     ======= 
1694  
1695     (n, e) : int, int 
1696         `n` is a product of any arbitrary number of primes given as 
1697         the argument. 
1698  
1699         `e` is relatively prime (coprime) to the Euler totient 
1700         `\phi(n)`. 
1701  
1702     False 
1703         Returned if less than two arguments are given, or `e` is 
1704         not relatively prime to the modulus. 
1705  
1706     Examples 
1707     ======== 
1708  
1709     >>> from sympy.crypto.crypto import rsa_public_key 
1710  
1711     A public key of a two-prime RSA: 
1712  
1713     >>> p, q, e = 3, 5, 7 
1714     >>> rsa_public_key(p, q, e) 
1715     (15, 7) 
1716     >>> rsa_public_key(p, q, 30) 
1717     False 
1718  
1719     A public key of a multiprime RSA: 
1720  
1721     >>> primes = [2, 3, 5, 7, 11, 13] 
1722     >>> e = 7 
1723     >>> args = primes + [e] 
1724     >>> rsa_public_key(*args) 
1725     (30030, 7) 
1726  
1727     Notes 
1728     ===== 
1729  
1730     Although the RSA can be generalized over any modulus `n`, using 
1731     two large primes had became the most popular specification because a 
1732     product of two large primes is usually the hardest to factor 
1733     relatively to the digits of `n` can have. 
1734  
1735     However, it may need further understanding of the time complexities 
1736     of each prime-factoring algorithms to verify the claim. 
1737  
1738     See Also 
1739     ======== 
1740  
1741     rsa_private_key 
1742     encipher_rsa 
1743     decipher_rsa 
1744  
1745     References 
1746     ========== 
1747  
1748     .. [1] https://en.wikipedia.org/wiki/RSA_%28cryptosystem%29 
1749  
1750     .. [2] https://cacr.uwaterloo.ca/techreports/2006/cacr2006-16.pdf 
1751  
1752     .. [3] https://link.springer.com/content/pdf/10.1007/BFb0055738.pdf 
1753  
1754     .. [4] https://www.itiis.org/digital-library/manuscript/1381 
1755     """
1756     return _rsa_key(*args, public=True, private=False, **kwargs)

1901 def encipher_rsa(i, key, factors=None):
1902     r"""Encrypt the plaintext with RSA. 
1903  
1904     Parameters 
1905     ========== 
1906  
1907     i : integer 
1908         The plaintext to be encrypted for. 
1909  
1910     key : (n, e) where n, e are integers 
1911         `n` is the modulus of the key and `e` is the exponent of the 
1912         key. The encryption is computed by `i^e \bmod n`. 
1913  
1914         The key can either be a public key or a private key, however, 
1915         the message encrypted by a public key can only be decrypted by 
1916         a private key, and vice versa, as RSA is an asymmetric 
1917         cryptography system. 
1918  
1919     factors : list of coprime integers 
1920         This is identical to the keyword ``factors`` in 
1921         :meth:`decipher_rsa`. 
1922  
1923     Notes 
1924     ===== 
1925  
1926     Some specifications may make the RSA not cryptographically 
1927     meaningful. 
1928  
1929     For example, `0`, `1` will remain always same after taking any 
1930     number of exponentiation, thus, should be avoided. 
1931  
1932     Furthermore, if `i^e < n`, `i` may easily be figured out by taking 
1933     `e` th root. 
1934  
1935     And also, specifying the exponent as `1` or in more generalized form 
1936     as `1 + k \lambda(n)` where `k` is an nonnegative integer, 
1937     `\lambda` is a carmichael totient, the RSA becomes an identity 
1938     mapping. 
1939  
1940     Examples 
1941     ======== 
1942  
1943     >>> from sympy.crypto.crypto import encipher_rsa 
1944     >>> from sympy.crypto.crypto import rsa_public_key, rsa_private_key 
1945  
1946     Public Key Encryption: 
1947  
1948     >>> p, q, e = 3, 5, 7 
1949     >>> puk = rsa_public_key(p, q, e) 
1950     >>> msg = 12 
1951     >>> encipher_rsa(msg, puk) 
1952     3 
1953  
1954     Private Key Encryption: 
1955  
1956     >>> p, q, e = 3, 5, 7 
1957     >>> prk = rsa_private_key(p, q, e) 
1958     >>> msg = 12 
1959     >>> encipher_rsa(msg, prk) 
1960     3 
1961  
1962     Encryption using chinese remainder theorem: 
1963  
1964     >>> encipher_rsa(msg, prk, factors=[p, q]) 
1965     3 
1966     """
1967     return _encipher_decipher_rsa(i, key, factors=factors)

1970 def decipher_rsa(i, key, factors=None):
1971     r"""Decrypt the ciphertext with RSA. 
1972  
1973     Parameters 
1974     ========== 
1975  
1976     i : integer 
1977         The ciphertext to be decrypted for. 
1978  
1979     key : (n, d) where n, d are integers 
1980         `n` is the modulus of the key and `d` is the exponent of the 
1981         key. The decryption is computed by `i^d \bmod n`. 
1982  
1983         The key can either be a public key or a private key, however, 
1984         the message encrypted by a public key can only be decrypted by 
1985         a private key, and vice versa, as RSA is an asymmetric 
1986         cryptography system. 
1987  
1988     factors : list of coprime integers 
1989         As the modulus `n` created from RSA key generation is composed 
1990         of arbitrary prime factors 
1991         `n = {p_1}^{k_1}{p_2}^{k_2}\dots{p_n}^{k_n}` where 
1992         `p_1, p_2, \dots, p_n` are distinct primes and 
1993         `k_1, k_2, \dots, k_n` are positive integers, chinese remainder 
1994         theorem can be used to compute `i^d \bmod n` from the 
1995         fragmented modulo operations like 
1996  
1997         .. math:: 
1998             i^d \bmod {p_1}^{k_1}, i^d \bmod {p_2}^{k_2}, \dots, 
1999             i^d \bmod {p_n}^{k_n} 
2000  
2001         or like 
2002  
2003         .. math:: 
2004             i^d \bmod {p_1}^{k_1}{p_2}^{k_2}, 
2005             i^d \bmod {p_3}^{k_3}, \dots , 
2006             i^d \bmod {p_n}^{k_n} 
2007  
2008         as long as every moduli does not share any common divisor each 
2009         other. 
2010  
2011         The raw primes used in generating the RSA key pair can be a good 
2012         option. 
2013  
2014         Note that the speed advantage of using this is only viable for 
2015         very large cases (Like 2048-bit RSA keys) since the 
2016         overhead of using pure Python implementation of 
2017         :meth:`sympy.ntheory.modular.crt` may overcompensate the 
2018         theoretical speed advantage. 
2019  
2020     Notes 
2021     ===== 
2022  
2023     See the ``Notes`` section in the documentation of 
2024     :meth:`encipher_rsa` 
2025  
2026     Examples 
2027     ======== 
2028  
2029     >>> from sympy.crypto.crypto import decipher_rsa, encipher_rsa 
2030     >>> from sympy.crypto.crypto import rsa_public_key, rsa_private_key 
2031  
2032     Public Key Encryption and Decryption: 
2033  
2034     >>> p, q, e = 3, 5, 7 
2035     >>> prk = rsa_private_key(p, q, e) 
2036     >>> puk = rsa_public_key(p, q, e) 
2037     >>> msg = 12 
2038     >>> new_msg = encipher_rsa(msg, prk) 
2039     >>> new_msg 
2040     3 
2041     >>> decipher_rsa(new_msg, puk) 
2042     12 
2043  
2044     Private Key Encryption and Decryption: 
2045  
2046     >>> p, q, e = 3, 5, 7 
2047     >>> prk = rsa_private_key(p, q, e) 
2048     >>> puk = rsa_public_key(p, q, e) 
2049     >>> msg = 12 
2050     >>> new_msg = encipher_rsa(msg, puk)

2294 def lfsr_sequence(key, fill, n):
2295     r""" 
2296     This function creates an LFSR sequence. 
2297  
2298     Parameters 
2299     ========== 
2300  
2301     key : list 
2302         A list of finite field elements, `[c_0, c_1, \ldots, c_k].` 
2303  
2304     fill : list 
2305         The list of the initial terms of the LFSR sequence, 
2306         `[x_0, x_1, \ldots, x_k].` 
2307  
2308     n 
2309         Number of terms of the sequence that the function returns. 
2310  
2311     Returns 
2312     ======= 
2313  
2314     L 
2315         The LFSR sequence defined by 
2316         `x_{n+1} = c_k x_n + \ldots + c_0 x_{n-k}`, for 
2317         `n \leq k`. 
2318  
2319     Notes 
2320     ===== 
2321  
2322     S. Golomb [G]_ gives a list of three statistical properties a 
2323     sequence of numbers `a = \{a_n\}_{n=1}^\infty`, 
2324     `a_n \in \{0,1\}`, should display to be considered 
2325     "random". Define the autocorrelation of `a` to be 
2326  
2327     .. math:: 
2328  
2329         C(k) = C(k,a) = \lim_{N\rightarrow \infty} {1\over N}\sum_{n=1}^N (-1)^{a_n + a_{n+k}}. 
2330  
2331     In the case where `a` is periodic with period 
2332     `P` then this reduces to 
2333  
2334     .. math:: 
2335  
2336         C(k) = {1\over P}\sum_{n=1}^P (-1)^{a_n + a_{n+k}}. 
2337  
2338     Assume `a` is periodic with period `P`. 
2339  
2340     - balance: 
2341  
2342       .. math:: 
2343  
2344         \left|\sum_{n=1}^P(-1)^{a_n}\right| \leq 1. 
2345  
2346     - low autocorrelation: 
2347  
2348        .. math:: 
2349  
2350          C(k) = \left\{ \begin{array}{cc} 1,& k = 0,\\ \epsilon, & k \ne 0. \end{array} \right. 
2351  
2352       (For sequences satisfying these first two properties, it is known 
2353       that `\epsilon = -1/P` must hold.) 
2354  
2355     - proportional runs property: In each period, half the runs have 
2356       length `1`, one-fourth have length `2`, etc. 
2357       Moreover, there are as many runs of `1`'s as there are of 
2358       `0`'s. 
2359  
2360     Examples 
2361     ======== 
2362  
2363     >>> from sympy.crypto.crypto import lfsr_sequence 
2364     >>> from sympy.polys.domains import FF 
2365     >>> F = FF(2) 
2366     >>> fill = [F(1), F(1), F(0), F(1)] 
2367     >>> key = [F(1), F(0), F(0), F(1)] 
2368     >>> lfsr_sequence(key, fill, 10) 
2369     [1 mod 2, 1 mod 2, 0 mod 2, 1 mod 2, 0 mod 2, 
2370     1 mod 2, 1 mod 2, 0 mod 2, 0 mod 2, 1 mod 2] 
2371  
2372     References 
2373     ========== 
2374  
2375     .. [G] Solomon Golomb, Shift register sequences, Aegean Park Press, 
2376        Laguna Hills, Ca, 1967 
2377  
2378     """
2379     if not isinstance(key, list):
2380         raise TypeError("key must be a list")
2381     if not isinstance(fill, list):
2382         raise TypeError("fill must be a list")
2383     p = key[0].modulus()
2384     F = FF(p)
2385     s = fill
2386     k = len(fill)
2387     L = []
2388     for i in range(n):
2389         s0 = s[:]
2390         L.append(s[0])
2391         s = s[1:k]
2392         x = sum(int(key[i]*s0[i]) for i in range(k))
2393         s.append(F(x))
2394     return L       # use [int(x) for x in L] for int version

2718     >>> pub = elgamal_public_key(pri); pub

2733 def dh_private_key(digit=10, seed=None):
2734     r""" 
2735     Return three integer tuple as private key. 
2736  
2737     Explanation 
2738     =========== 
2739  
2740     Diffie-Hellman key exchange is based on the mathematical problem 
2741     called the Discrete Logarithm Problem (see ElGamal). 
2742  
2743     Diffie-Hellman key exchange is divided into the following steps: 
2744  
2745     *   Alice and Bob agree on a base that consist of a prime ``p`` 
2746         and a primitive root of ``p`` called ``g`` 
2747     *   Alice choses a number ``a`` and Bob choses a number ``b`` where 
2748         ``a`` and ``b`` are random numbers in range `[2, p)`. These are 
2749         their private keys. 
2750     *   Alice then publicly sends Bob `g^{a} \pmod p` while Bob sends 
2751         Alice `g^{b} \pmod p` 
2752     *   They both raise the received value to their secretly chosen 
2753         number (``a`` or ``b``) and now have both as their shared key 
2754         `g^{ab} \pmod p`

2757     ========== 
2758  
2759     digit

2761

2776     the output of this routine. See sympy.core.random._randrange. 
2777  
2778     Examples 
2779     ======== 
2780

2782     >>> from sympy.ntheory import isprime, is_primitive_root

2791     >>> is_primitive_root(g, p)

2796     g = primitive_root(p)

2926 def gm_private_key(p, q, a=None):
2927     r""" 
2928     Check if ``p`` and ``q`` can be used as private keys for 
2929     the Goldwasser-Micali encryption. The method works 
2930     roughly as follows. 
2931  
2932     Explanation 
2933     =========== 
2934  
2935     #. Pick two large primes $p$ and $q$. 
2936     #. Call their product $N$. 
2937     #. Given a message as an integer $i$, write $i$ in its bit representation $b_0, \dots, b_n$. 
2938     #. For each $k$, 
2939  
2940      if $b_k = 0$: 
2941         let $a_k$ be a random square 
2942         (quadratic residue) modulo $p q$ 
2943         such that ``jacobi_symbol(a, p*q) = 1`` 
2944      if $b_k = 1$: 
2945         let $a_k$ be a random non-square 
2946         (non-quadratic residue) modulo $p q$ 
2947         such that ``jacobi_symbol(a, p*q) = 1`` 
2948  
2949     returns $\left[a_1, a_2, \dots\right]$ 
2950  
2951     $b_k$ can be recovered by checking whether or not 
2952     $a_k$ is a residue. And from the $b_k$'s, the message 
2953     can be reconstructed. 
2954  
2955     The idea is that, while ``jacobi_symbol(a, p*q)`` 
2956     can be easily computed (and when it is equal to $-1$ will 
2957     tell you that $a$ is not a square mod $p q$), quadratic 
2958     residuosity modulo a composite number is hard to compute 
2959     without knowing its factorization. 
2960  
2961     Moreover, approximately half the numbers coprime to $p q$ have 
2962     :func:`~.jacobi_symbol` equal to $1$ . And among those, approximately half 
2963     are residues and approximately half are not. This maximizes the 
2964     entropy of the code. 
2965  
2966     Parameters 
2967     ========== 
2968  
2969     p, q, a 
2970         Initialization variables. 
2971  
2972     Returns 
2973     ======= 
2974  
2975     tuple : (p, q) 
2976         The input value ``p`` and ``q``. 
2977  
2978     Raises 
2979     ====== 
2980  
2981     ValueError 
2982         If ``p`` and ``q`` are not distinct odd primes. 
2983  
2984     """
2985     if p == q:
2986         raise ValueError("expected distinct primes, "
2987                          "got two copies of %i" % p)
2988     elif not isprime(p) or not isprime(q):
2989         raise ValueError("first two arguments must be prime, "
2990                          "got %i of %i" % (p, q))
2991     elif p == 2 or q == 2:
2992         raise ValueError("first two arguments must not be even, "
2993                          "got %i of %i" % (p, q))
2994     return p, q

2998     """ 
2999     Compute public keys for ``p`` and ``q``. 
3000     Note that in Goldwasser-Micali Encryption, 
3001     public keys are randomly selected. 
3002  
3003     Parameters 
3004     ========== 
3005  
3006     p, q, a : int, int, int 
3007         Initialization variables. 
3008  
3009     Returns 
3010     ======= 
3011  
3012     tuple : (a, N) 
3013         ``a`` is the input ``a`` if it is not ``None`` otherwise 
3014         some random integer coprime to ``p`` and ``q``. 
3015  
3016         ``N`` is the product of ``p`` and ``q``.

3248 def encipher_bg(i, key, seed=None):
3249     """

3251  
3252     Explanation 
3253     =========== 
3254  
3255     ALGORITHM: 
3256         1. Encodes i as a string of L bits, m. 
3257         2. Select a random element r, where 1 < r < key, and computes 
3258            x = r^2 mod key. 
3259         3. Use BBS pseudo-random number generator to generate L random bits, b, 
3260         using the initial seed as x. 
3261         4. Encrypted message, c_i = m_i XOR b_i, 1 <= i <= L. 
3262         5. x_L = x^(2^L) mod key. 
3263         6. Return (c, x_L)

3266     ========== 
3267  
3268     i 
3269         Message, a non-negative integer 
3270  
3271     key

3273

3279  
3280     Raises 
3281     ====== 
3282  
3283     ValueError 
3284         If i is negative. 
3285  
3286     """

3290             "message must be a non-negative "
3291             "integer: got %d instead" % i)

3294     while i > 0:
3295         enc_msg.append(i % 2)
3296         i //= 2

3313 def decipher_bg(message, key):
3314     """ 
3315     Decrypts the message using private keys. 
3316  
3317     Explanation 
3318     =========== 
3319  
3320     ALGORITHM: 
3321         1. Let, c be the encrypted message, y the second number received, 
3322         and p and q be the private keys. 
3323         2. Compute, r_p = y^((p+1)/4 ^ L) mod p and 
3324         r_q = y^((q+1)/4 ^ L) mod q. 
3325         3. Compute x_0 = (q(q^-1 mod p)r_p + p(p^-1 mod q)r_q) mod N. 
3326         4. From, recompute the bits using the BBS generator, as in the 
3327         encryption algorithm. 
3328         5. Compute original message by XORing c and b. 
3329  
3330     Parameters 
3331     ========== 
3332  
3333     message 
3334         Tuple of encrypted message and a non-negative integer. 
3335  
3336     key 
3337         Tuple of private keys. 
3338  
3339     Returns 
3340     ======= 
3341  
3342     orig_msg 
3343         The original message 
3344  
3345     """
3346 
3347     p, q = key
3348     encrypt_msg, y = message
3349     public_key = p * q
3350     L = len(encrypt_msg)
3351     p_t = ((p + 1)/4)**L
3352     q_t = ((q + 1)/4)**L
3353     r_p = pow(int(y), int(p_t), int(p))
3354     r_q = pow(int(y), int(q_t), int(q))
3355 
3356     x = (q * invert(q, p) * r_p + p * invert(p, q) * r_q) % public_key
3357 
3358     orig_bits = []
3359     for _ in range(L):
3360         orig_bits.append(x % 2)
3361         x = x**2 % public_key
3362 
3363     orig_msg = 0
3364     for (m, b) in zip(encrypt_msg, orig_bits):
3365         orig_msg = orig_msg * 2
3366         orig_msg += (m ^ b)
3367 
3368     return orig_msg
