50       def _print_Float(self, expr: sympy.Expr) -> str:
51           if expr._prec == 53:
52               # IEEE-754 double precision have 53 bits. SymPy prints them with
53               # 15 digits, but we need 17 for round-trip correctness
54               return str(sympy.Float(expr, dps=17))
55           else:
56               # We don't use other precisions in pytorch
57               return str(expr)

59       # This must be implemented because sympy will collect x * x into Pow(x, 2), without
60       # any explicit intervention.  We print it just like x * x, notably, we
61       # never generate sympy.Pow with floats.
62       #
63       # NB: this pow by natural, you should never have used builtin sympy.pow
64       # for FloatPow, and a symbolic exponent should be PowByNatural.  These
65       # means exp is guaranteed to be integer.
66       def _print_Pow(self, expr: sympy.Expr) -> str:
67           base, exp = expr.args
68           assert exp == int(exp), exp
69           exp = int(exp)
70           assert exp >= 0
71           if exp > 0:
72               return self.stringify([base] * exp, "*", PRECEDENCE["Mul"])
73           return "1"

130  class PythonPrinter(ExprPrinter):
131      def _print_ToFloat(self, expr: sympy.Expr) -> str:
132          assert len(expr.args) == 1
133          # NB: We use sym_float here because the printer is used for cache
134          # serialization, and cache guards get evaluated with SymInt to
135          # propagate guards to the parent ShapeEnv.  However, this comes at a
136          # runtime cost for guards involving float.  If this is unacceptable
137          # overhead, what you want to do is have two separate printers for
138          # SymInt, one for when the inputs are guaranteed to be int, and
139          # another for when they could be SymInt.
140          #
141          # NB: sym_min/sym_max also have this problem, but I chose not to fix
142          # those.
143          #
144          # See https://github.com/pytorch/pytorch/issues/142507 for more
145          # context.
146          return f"torch.sym_float({self._print(expr.args[0])})"

282  class CppPrinter(ExprPrinter):
283      def _print_Integer(self, expr: sympy.Expr) -> str:
284          suffix = "LL" if sys.platform in ["darwin", "win32"] else "L"
285          i = int(expr)
286          if i > INDEX_TYPE_MAX or i < INDEX_TYPE_MIN:
287              raise OverflowError(f"{i} too big to convert to {INDEX_TYPE}")
288          elif i == INDEX_TYPE_MIN:
289              assert i == (-1) << 63
290              # Writing -9223372036854775808L makes the value overflow
291              # as it is parsed as -(9223372036854775808L) by the C/C++ compiler
292              return f"(-1{suffix} << 63)"
293          return f"{i}{suffix}"

296          c, p, q = (
297              self.parenthesize(arg, PRECEDENCE["Atom"] - 0.5) for arg in expr.args
298          )
299          return f"{c} ? {p} : {q}"

301      def _print_ModularIndexing(self, expr: sympy.Expr) -> str:
302          x, div, mod = expr.args
303          x = self.doprint(x)
304          if div != 1:
305              div = self.doprint(div)
306              if expr.is_integer:
307                  x = f"c10::div_floor_integer(static_cast<int64_t>({x}), static_cast<int64_t>({div}))"
308              else:
309                  x = f"c10::div_floor_floating(static_cast<double>({x}), static_cast<double>({div}))"
310          mod = self.doprint(mod)
311          return f"(static_cast<{INDEX_TYPE}>({x}) % static_cast<{INDEX_TYPE}>({mod}))"

370      def _print_Pow(self, expr: sympy.Expr) -> str:
371          # Uses float constants to perform FP div
372          base, exp = expr.args
373  
374          if exp == 0.5 or exp == -0.5:
375              base = self._print(base)
376              return f"std::sqrt({base})" if exp == 0.5 else f"1.0/std::sqrt({base})"
377          if exp.is_integer:
378              exp = int(exp)
379              if exp > 0:
380                  r = self.stringify([base] * exp, "*", PRECEDENCE["Mul"])
381              elif exp < -1:
382                  r = (
383                      "1.0/("
384                      + self.stringify([base] * abs(exp), "*", PRECEDENCE["Mul"])
385                      + ")"
386                  )
387              elif exp == -1:
388                  r = "1.0/" + self._print(base)
389              else:  # exp == 0
390                  r = "1.0"
391  
392              return f"static_cast<{INDEX_TYPE}>({r})" if expr.is_integer else r
393          else:
394              # TODO: float vs double
395              return f"std::pow({base}, {float(exp)})"
