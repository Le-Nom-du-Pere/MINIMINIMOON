10   // If element_sizes is nullptr, then the strides will be in bytes, otherwise 
11   // the strides will be in # of elements. 
12   // Operands that share the same shape, but may have different strides. 
13   // OffsetCalculator iterates the tensor in a column-major order 
14   
15   #if defined(USE_ROCM)
16   constexpr int MAX_DIMS = 16;
17   #else
18   constexpr int MAX_DIMS = 25;
19   #endif
20   
21   template <int NARGS, typename index_t = uint32_t, bool signed_strides = false> 
22   struct OffsetCalculator {
23     // We allow having negative strides to implement some operations like torch.flip 
24     using stride_t = std::conditional_t<signed_strides,
25                                         std::make_signed_t<index_t>,
26                                         index_t>;
27     // The offset for each argument. Wrapper around fixed-size array. 
28     // On CUDA, zero sized array is not allowed, so when we are handling nullary 
29     // operators, we need to create a size 1 offset to avoid compiler failure. 
30     // This size 1 offset is just a placeholder, and we will not use it. 
31     using offset_type = std::array<stride_t, std::max<int>(NARGS, 1)>;
32   
33     // if element_sizes is nullptr, then the strides will be in bytes, otherwise 
34     // the strides will be in # of elements. 
35     OffsetCalculator(int dims, const int64_t* sizes, const int64_t* const* strides, const int64_t* element_sizes=nullptr) : dims(dims) {
36       TORCH_CHECK(dims <= MAX_DIMS, "tensor has too many (>", MAX_DIMS, ") dims");
37       for (int i=0; i < dims; i++){
38         sizes_[i] = at::cuda::detail::IntDivider<index_t>(sizes[i]);
39         for (int arg = 0; arg < NARGS; arg++) {
40           int64_t element_size = (element_sizes == nullptr ? 1LL : element_sizes[arg]);
41           strides_[i][arg] = strides[arg][i] / element_size;
42         }
43       }
44     }
45   
46     C10_HOST_DEVICE offset_type get(index_t linear_idx) const {
47       offset_type offsets;
48       #pragma unroll
49       for (int arg = 0; arg < NARGS; arg++) {
50         offsets[arg] = 0;
51       }
52   
53       #pragma unroll
54       for (int dim = 0; dim < MAX_DIMS; ++dim) {
55         if (dim == dims) {
56           break;
57         }
58         auto divmod = sizes_[dim].divmod(linear_idx);
59         linear_idx = divmod.div;
60   
61         #pragma unroll
62         for (int arg = 0; arg < NARGS; arg++) {
63           offsets[arg] += divmod.mod * strides_[dim][arg];
64         }
65   
66       }
67       return offsets;
68     }
69   
70     int dims;
71     at::cuda::detail::IntDivider<index_t> sizes_[MAX_DIMS];
72     stride_t strides_[MAX_DIMS][std::max<int>(NARGS, 1)];
73   };

75   template <int NARGS, typename index_t = uint32_t> 
76   struct TrivialOffsetCalculator {
77     // The offset for each argument. Wrapper around fixed-size array. 
78     // The offsets are in # of elements, not in bytes. 
79     // On CUDA, zero sized array is not allowed, so when we are handling nullary 
80     // operators, we need to create a size 1 offset to avoid compiler failure. 
81     // This size 1 offset is just a placeholder, and we will not use it. 
82     using offset_type = std::array<index_t, std::max<int>(NARGS, 1)>;
83   
84     C10_HOST_DEVICE offset_type get(index_t linear_idx) const {
85       offset_type offsets;
86       #pragma unroll
87       for (int arg = 0; arg < NARGS; arg++) {
88         offsets[arg] = linear_idx;
89       }
90       return offsets;
91     }
92   };

94   // Make an OffsetCalculator with byte offsets 
95   template<int N, bool signed_strides = false> 
96   static OffsetCalculator<N, uint32_t, signed_strides> make_offset_calculator(const at::TensorIteratorBase& iter) {
97     TORCH_INTERNAL_ASSERT(N <= iter.ntensors());
98     std::array<const int64_t*, N> strides;
99     for (int i = 0; i < N; i++) {
100      strides[i] = iter.strides(i).data();
101    }
102    return OffsetCalculator<N, uint32_t, signed_strides>(iter.ndim(), iter.shape().data(), strides.data());
103  }
104  
105  // Make an OffsetCalculator with element offsets 
106  template<int N, bool signed_strides = false> 
107  static OffsetCalculator<N, uint32_t, signed_strides> make_element_offset_calculator(
108      const at::TensorIteratorBase& iter) {
109    TORCH_INTERNAL_ASSERT(N <= iter.ntensors());
110    std::array<const int64_t*, N> strides;
111    std::array<int64_t, N> element_sizes;
112    for (int i = 0; i < N; i++) {
113      strides[i] = iter.strides(i).data();
114      element_sizes[i] = iter.element_size(i);
115    }
116    return OffsetCalculator<N, uint32_t, signed_strides>(
117        iter.ndim(), iter.shape().data(), strides.data(), element_sizes.data());
118  }
