75   class Library:
76       """ 
77       A class to create libraries that can be used to register new operators or 
78       override operators in existing libraries from Python. 
79       A user can optionally pass in a dispatch keyname if they only want to register 
80       kernels corresponding to only one specific dispatch key. 
81    
82       To create a library to override operators in an existing library (with name ns), set the kind to "IMPL". 
83       To create a new library (with name ns) to register new operators, set the kind to "DEF". 
84       To create a fragment of a possibly existing library to register operators (and bypass 
85       the limitation that there is only one library for a given namespace), set the kind to 
86       "FRAGMENT". 
87    
88       Args: 
89           ns: library name 
90           kind: "DEF", "IMPL", "FRAGMENT" 
91           dispatch_key: PyTorch dispatch key (default: "") 
92       """
93   
94       def __init__(self, ns, kind, dispatch_key=""):
95           from torch.fx.operator_schemas import _SCHEMA_TO_SIGNATURE_CACHE
96   
97           if kind not in ("IMPL", "DEF", "FRAGMENT"):
98               raise ValueError("Unsupported kind: ", kind)
99   
100          if ns in _reserved_namespaces and (kind == "DEF" or kind == "FRAGMENT"):
101              raise ValueError(
102                  ns,
103                  " is a reserved namespace. Please try creating a library with another name.",
104              )
105          if torch._running_with_deploy():
106              _library.utils.warn_deploy()
107              return
108  
109          frame = traceback.extract_stack(limit=3)[0]
110          filename, lineno = frame.filename, frame.lineno
111          self.m: Optional[Any] = torch._C._dispatch_library(
112              kind, ns, dispatch_key, filename, lineno
113          )
114          self.ns = ns
115          self._op_defs: set[str] = set()
116          self._op_impls: set[str] = set()
117          self._registration_handles: list[torch._library.utils.RegistrationHandle] = []
118          self.kind = kind
119          self.dispatch_key = dispatch_key
120          # Use a finalizer to setup the "destructor" instead of __del__.
121          # Python __del__ can lead to weird things (globals and locals may already
122          # be gone when __del__ actually gets called!). finalizers help the
123          # situation because it lets us capture references and keeps them alive
124          weakref.finalize(
125              self,
126              _del_library,
127              _impls,
128              self._op_impls,
129              _defs,
130              self._op_defs,
131              self._registration_handles,
132              self.m,
133              _SCHEMA_TO_SIGNATURE_CACHE,
134          )

137          return f"Library(kind={self.kind}, ns={self.ns}, dispatch_key={self.dispatch_key})>"

139      def define(self, schema, alias_analysis="", *, tags=()):
140          r"""Defines a new operator and its semantics in the ns namespace. 
141   
142          Args: 
143              schema: function schema to define a new operator. 
144              alias_analysis (optional): Indicates if the aliasing properties of the operator arguments can be 
145                                         inferred from the schema (default behavior) or not ("CONSERVATIVE"). 
146              tags (Tag | Sequence[Tag]): one or more torch.Tag to apply to this 
147                                         operator. Tagging an operator changes the operator's behavior 
148                                         under various PyTorch subsystems; please read the docs for the 
149                                         torch.Tag carefully before applying it. 
150   
151          Returns: 
152              name of the operator as inferred from the schema. 
153   
154          Example:: 
155   
156              >>> my_lib = Library("mylib", "DEF") 
157              >>> my_lib.define("sum(Tensor self) -> Tensor") 
158          """
159          if torch._running_with_deploy():
160              _library.utils.warn_deploy()
161              return
162  
163          # This is added because we also want to disallow PURE_FUNCTION alias analysis which is a valid
164          # AliasAnalysis type in C++
165          if alias_analysis not in ["", "FROM_SCHEMA", "CONSERVATIVE"]:
166              raise RuntimeError(f"Invalid alias_analysis type {alias_analysis}")
167          assert self.m is not None
168          if isinstance(tags, torch.Tag):
169              tags = (tags,)
170  
171          name = schema.split("(")[0]
172          packet_name = name.split(".")[0] if "." in name else name
173          has_preexisting_packet = hasattr(torch.ops, self.ns) and hasattr(
174              getattr(torch.ops, self.ns), packet_name
175          )
176  
177          result = self.m.define(schema, alias_analysis, tuple(tags))
178          name = schema.split("(")[0]
179          qualname = self.ns + "::" + name
180  
181          # If the OpOverloadPacket exists already, then this means we're adding a
182          # new OpOverload for it. Refresh the packet to include the new OpOverload.
183          if has_preexisting_packet:
184              ns = getattr(torch.ops, self.ns)
185              packet = getattr(ns, packet_name)
186              torch._ops._refresh_packet(packet)
187  
188          self._op_defs.add(qualname)
189          _defs.add(qualname)
190          return result

192      def _register_fake(self, op_name, fn, _stacklevel=1, *, allow_override=False):
193          r"""Registers the fake impl for an operator defined in the library."""
194          if torch._running_with_deploy():
195              _library.utils.warn_deploy()
196              return
197  
198          source = torch._library.utils.get_source(_stacklevel + 1)
199          frame = sys._getframe(_stacklevel)
200          caller_module = inspect.getmodule(frame)
201          # Can be none if you call register_fake from somewhere there isn't a module
202          # (e.g. __main__)
203          caller_module_name = None if caller_module is None else caller_module.__name__
204  
205          # TODO(rzou): We're gonna need to stage this change with torchvision,
206          # since torchvision is github first.
207          if caller_module_name is not None and caller_module_name.startswith(
208              "torchvision."
209          ):
210              caller_module_name = None
211  
212          qualname = f"{self.ns}::{op_name}"
213          entry = torch._library.simple_registry.singleton.find(qualname)
214          if caller_module_name is not None:
215              func_to_register = _check_pystubs_once(fn, qualname, caller_module_name)
216          else:
217              func_to_register = fn
218  
219          handle = entry.fake_impl.register(
220              func_to_register, source, lib=self, allow_override=allow_override
221          )
222          self._registration_handles.append(handle)

224      def _register_torch_dispatch_rule(self, op_name, torch_dispatch_class, fn):
225          r"""Registers a torch_dispatch rule for the given operator and torch_dispatch_class. 
226   
227          This allows for open registration to specify the behavior between the operator 
228          and the torch_dispatch_class without needing to modify the torch_dispatch_class 
229          or the operator directly. 
230   
231          The torch_dispatch_class is either a Tensor subclass with `__torch_dispatch__` or a 
232          TorchDispatchMode. 
233   
234          If it is a Tensor subclass, we expect fn to have the following signature: 
235          (cls, func: OpOverload, types: Tuple[type, ...], args, kwargs) -> Any 
236   
237          If it is a TorchDispatchMode, we expect fn to have the following signature: 
238          (mode, func: OpOverload, types: Tuple[type, ...], args, kwargs) -> Any 
239          """
240          if torch._running_with_deploy():
241              _library.utils.warn_deploy()
242              return
243  
244          qualname = f"{self.ns}::{op_name}"
245          entry = torch._library.simple_registry.singleton.find(qualname)
246          handle = entry.torch_dispatch_rules.register(torch_dispatch_class, fn)
247          self._registration_handles.append(handle)

249      def _impl_with_aoti_compile(self, op_name, dispatch_key=""):
250          r"""Register the operator to use the AOTI-compiled implementation. 
251   
252          Args: 
253              op_name: operator name (along with the overload) or OpOverload object. 
254              dispatch_key: dispatch key that the input function should be registered for. By default, it uses 
255                            the dispatch key that the library was created with. 
256   
257          Example:: 
258   
259              >>> my_lib = Library("aten", "IMPL") 
260              >>> my_lib._impl_with_aoti_compile("div.Tensor", "CPU") 
261          """
262          if torch._running_with_deploy():
263              _library.utils.warn_deploy()
264              return
265  
266          if dispatch_key == "":
267              dispatch_key = self.dispatch_key
268          assert torch.DispatchKeySet(dispatch_key).has(torch._C.DispatchKey.Dense)
269  
270          if isinstance(op_name, str):
271              name = op_name
272          elif isinstance(op_name, OpOverload):
273              name = op_name._schema.name
274              overload_name = op_name._schema.overload_name
275              if overload_name != "":
276                  name = name + "." + overload_name
277          else:
278              raise RuntimeError(
279                  "_impl_with_aoti_compile should be passed either a name or an OpOverload object "
280                  "as the first argument"
281              )
282  
283          key = self.ns + "/" + name.split("::")[-1] + "/" + dispatch_key
284          if key in _impls:
285              # TODO: in future, add more info about where the existing function is registered (this info is
286              # today already returned by the C++ warning when _impl_with_aoti_compile is called but we error out before that)
287              raise RuntimeError(
288                  "This is not allowed since there's already a kernel registered from python overriding {}"
289                  "'s behavior for {} dispatch key and {} namespace.".format(
290                      name.split("::")[-1], dispatch_key, self.ns
291                  )
292              )
293  
294          assert self.m is not None
295          impl_fn: Callable = self.m.impl_with_aoti_compile
296          impl_fn(self.ns, name.split("::")[-1], dispatch_key)
297  
298          _impls.add(key)
299          self._op_impls.add(key)

301      def impl(
302          self, op_name, fn, dispatch_key="", *, with_keyset=False, allow_override=False
303      ):
304          r"""Registers the function implementation for an operator defined in the library. 
305   
306          Args: 
307              op_name: operator name (along with the overload) or OpOverload object. 
308              fn: function that's the operator implementation for the input dispatch key or :func:`~fallthrough_kernel` 
309                  to register a fallthrough. 
310              dispatch_key: dispatch key that the input function should be registered for. By default, it uses 
311                            the dispatch key that the library was created with. 
312              with_keyset: flag controlling if the current dispatcher call keyset should be passed as the first argument 
313                           to :attr:`fn` when calling. This should be used to create the appropriate keyset for redispatch calls. 
314              allow_override: Flag controlling if we want to override an 
315                           existing registered kernel implementation. This is by 
316                           default off, and will error you're trying to register a 
317                           kernel to a dispatch key with a kernel already 
318                           registered. 
319   
320          Example:: 
321   
322              >>> my_lib = Library("aten", "IMPL") 
323              >>> def div_cpu(self, other): 
324              >>>     return self * (1 / other) 
325              >>> my_lib.impl("div.Tensor", div_cpu, "CPU") 
326          """
327          if torch._running_with_deploy():
328              _library.utils.warn_deploy()
329              return
330  
331          if not callable(fn):
332              raise TypeError(
333                  f"Input function is required to be a callable but found type {type(fn)}"
334              )
335          if dispatch_key == "":
336              dispatch_key = self.dispatch_key
337  
338          if isinstance(op_name, str):
339              name = op_name
340          elif isinstance(op_name, OpOverload):
341              name = op_name._schema.name
342              overload_name = op_name._schema.overload_name
343              if overload_name != "":
344                  name = name + "." + overload_name
345          else:
346              raise RuntimeError(
347                  "impl should be passed either a name or an OpOverload object as the first argument"
348              )
349  
350          key = self.ns + "/" + name.split("::")[-1] + "/" + dispatch_key
351          if (not allow_override) and key in _impls:
352              # TODO: in future, add more info about where the existing function is registered (this info is
353              # today already returned by the C++ warning when impl is called but we error out before that)
354              raise RuntimeError(
355                  "This is not allowed since there's already a kernel registered from python overriding {}"
356                  "'s behavior for {} dispatch key and {} namespace.".format(
357                      name.split("::")[-1], dispatch_key, self.ns
358                  )
359              )
360  
361          if dispatch_key == "Meta":
362              dispatcher_op_name = name
363              if "::" not in dispatcher_op_name:
364                  dispatcher_op_name = f"{self.ns}::{dispatcher_op_name}"
365  
366              # Internally, we shouldn't be registering meta kernels for any operators that
367              # have CompositeImplicitAutograd kernels.
368              # Instead, we should be letting those decompositions run, and writing meta kernels
369              # only for the base operators.
370              if torch._C._dispatch_has_kernel_for_dispatch_key(
371                  dispatcher_op_name, "CompositeImplicitAutograd"
372              ):
373                  raise RuntimeError(
374                      f"We should not register a meta kernel directly to the operator '{name}',"
375                      " because it has a CompositeImplicitAutograd kernel in core."
376                      " Instead we should let the operator decompose, and ensure that we have meta kernels"
377                      " for the base ops that it decomposes into."
378                  )
379  
380          assert self.m is not None
381          self.m.impl(
382              name,
383              dispatch_key if dispatch_key != "" else "CompositeImplicitAutograd",
384              fn,
385              with_keyset,
386          )
387  
388          _impls.add(key)
389          self._op_impls.add(key)

391      def fallback(self, fn, dispatch_key="", *, with_keyset=False):
392          r"""Registers the function implementation as the fallback for the given key. 
393   
394          This function only works for a library with global namespace ("_"). 
395   
396          Args: 
397              fn: function used as fallback for the given dispatch key or :func:`~fallthrough_kernel` 
398                  to register a fallthrough. 
399              dispatch_key: dispatch key that the input function should be registered for. By default, it uses 
400                            the dispatch key that the library was created with. 
401              with_keyset: flag controlling if the current dispatcher call keyset should be passed as the first argument 
402                           to :attr:`fn` when calling. This should be used to create the appropriate keyset for redispatch calls. 
403   
404          Example:: 
405   
406              >>> my_lib = Library("_", "IMPL") 
407              >>> def fallback_kernel(op, *args, **kwargs): 
408              >>>     # Handle all autocast ops generically 
409              >>>     # ... 
410              >>> my_lib.fallback(fallback_kernel, "Autocast") 
411          """
412          if torch._running_with_deploy():
413              _library.utils.warn_deploy()
414              return
415  
416          if dispatch_key == "":
417              dispatch_key = self.dispatch_key
418  
419          if self.ns != "_":
420              raise RuntimeError(
421                  f"""Fallback can only be registered using libary fragment on the global namespace "_" but it is {self.ns}"""
422              )
423  
424          assert dispatch_key != ""
425          assert self.m is not None
426  
427          self.m.fallback(dispatch_key, fn, with_keyset)

429      def _destroy(self):
430          if self.m is not None:
431              self.m.reset()
432          self.m = None
433          for handle in self._registration_handles:
434              handle.destroy()
435          self._registration_handles.clear()
436          global _impls
437          _impls -= self._op_impls
438          for name in self._op_defs:
439              # Delete the cached torch.ops.ns.foo if it was registered.
440              # Otherwise, accessing it leads to a segfault.
441              # It's possible that we only registered an overload in this Library
442              # and another library owns an alive overload.
443              # That's OK - the next time torch.ops.ns.foo gets called, it'll be
444              # recomputed to point at the right collection of overloads.
445              ns, name_with_overload = name.split("::")
446              name = name_with_overload.split(".")[0]
447              if not hasattr(torch.ops, ns):
448                  continue
449              namespace = getattr(torch.ops, ns)
450              if not hasattr(namespace, name):
451                  continue
452              delattr(namespace, name)
453              namespace._dir.remove(name)

456  def _del_library(
457      captured_impls,
458      op_impls,
459      captured_defs,
460      op_defs,
461      registration_handles,
462      m,
463      schema_to_signature_cache,
464  ):
465      for op_def in op_defs:
466          name = op_def
467          overload_name = ""
468          if "." in op_def:
469              name, overload_name = op_def.split(".")
470          if (
471              name,
472              overload_name,
473          ) in schema_to_signature_cache:
474              del schema_to_signature_cache[(name, overload_name)]
475  
476      captured_impls -= op_impls
477      captured_defs -= op_defs
478      for handle in registration_handles:
479          handle.destroy()
480  
481      if m is not None:
482          m.reset()

500  @functools.singledispatch
501  def define(qualname, schema, *, lib=None, tags=()):
502      r"""Defines a new operator. 
503   
504      In PyTorch, defining an op (short for "operator") is a two step-process: 
505      - we need to define the op (by providing an operator name and schema) 
506      - we need to implement behavior for how the operator interacts with 
507      various PyTorch subsystems, like CPU/CUDA Tensors, Autograd, etc. 
508   
509      This entrypoint defines the custom operator (the first step) 
510      you must then perform the second step by calling various 
511      ``impl_*`` APIs, like :func:`torch.library.impl` or 
512      :func:`torch.library.register_fake`. 
513   
514      Args: 
515          qualname (str): The qualified name for the operator. Should be 
516              a string that looks like "namespace::name", e.g. "aten::sin". 
517              Operators in PyTorch need a namespace to 
518              avoid name collisions; a given operator may only be created once. 
519              If you are writing a Python library, we recommend the namespace to 
520              be the name of your top-level module. 
521          schema (str): The schema of the operator. E.g. "(Tensor x) -> Tensor" 
522              for an op that accepts one Tensor and returns one Tensor. It does 
523              not contain the operator name (that is passed in ``qualname``). 
524          lib (Optional[Library]): If provided, the lifetime of this operator 
525              will be tied to the lifetime of the Library object. 
526          tags (Tag | Sequence[Tag]): one or more torch.Tag to apply to this 
527              operator. Tagging an operator changes the operator's behavior 
528              under various PyTorch subsystems; please read the docs for the 
529              torch.Tag carefully before applying it. 
530   
531      Example:: 
532          >>> import torch 
533          >>> import numpy as np 
534          >>> 
535          >>> # Define the operator 
536          >>> torch.library.define("mylib::sin", "(Tensor x) -> Tensor") 
537          >>> 
538          >>> # Add implementations for the operator 
539          >>> @torch.library.impl("mylib::sin", "cpu") 
540          >>> def f(x): 
541          >>>     return torch.from_numpy(np.sin(x.numpy())) 
542          >>> 
543          >>> # Call the new operator from torch.ops. 
544          >>> x = torch.randn(3) 
545          >>> y = torch.ops.mylib.sin(x) 
546          >>> assert torch.allclose(y, x.sin()) 
547   
548      """
549      if not isinstance(qualname, str):
550          raise ValueError(
551              f"define(qualname, schema): expected qualname "
552              f"to be instance of str, got {type(qualname)}"
553          )
554      namespace, name = torch._library.utils.parse_namespace(qualname)
555      if lib is None:
556          lib = Library(namespace, "FRAGMENT")
557          _keep_alive.append(lib)
558      if not NAMELESS_SCHEMA.fullmatch(schema):
559          raise ValueError(
560              f"define(qualname, schema, ...): expected schema "
561              f'to look like e.g. "(Tensor x) -> Tensor" but '
562              f'got "{schema}"'
563          )
564      lib.define(name + schema, alias_analysis="", tags=tags)

611  def impl(
612      qualname: str,
613      types: Union[str, Sequence[str]],
614      func: Optional[Callable[_P, _T]] = None,
615      *,
616      lib: Optional[Library] = None,
617  ) -> object:
618      """Register an implementation for a device type for this operator. 
619   
620      You may pass "default" for ``types`` to register this implementation as the 
621      default implementation for ALL device types. 
622      Please only use this if the implementation truly supports all device types; 
623      for example, this is true if it is a composition of built-in PyTorch operators. 
624   
625      This API may be used as a decorator. You can use nested decorators 
626      with this API provided they return a function and are placed inside 
627      this API (see Example 2). 
628   
629      Some valid types are: "cpu", "cuda", "xla", "mps", "ipu", "xpu". 
630   
631      Args: 
632          qualname (str): Should be a string that looks like "namespace::operator_name". 
633          types (str | Sequence[str]): The device types to register an impl to. 
634          lib (Optional[Library]): If provided, the lifetime of this registration 
635              will be tied to the lifetime of the Library object. 
636   
637      Examples: 
638          >>> import torch 
639          >>> import numpy as np 
640          >>> # Example 1: Register function. 
641          >>> # Define the operator 
642          >>> torch.library.define("mylib::mysin", "(Tensor x) -> Tensor") 
643          >>> 
644          >>> # Add implementations for the cpu device 
645          >>> @torch.library.impl("mylib::mysin", "cpu") 
646          >>> def f(x): 
647          >>>     return torch.from_numpy(np.sin(x.numpy())) 
648          >>> 
649          >>> x = torch.randn(3) 
650          >>> y = torch.ops.mylib.mysin(x) 
651          >>> assert torch.allclose(y, x.sin()) 
652          >>> 
653          >>> # Example 2: Register function with decorator. 
654          >>> def custom_decorator(func): 
655          >>>     def wrapper(*args, **kwargs):

715  def _impl(
716      qualname: str,
717      types: Union[str, Sequence[str]],
718      func: Optional[Callable[..., object]] = None,
719      *,
720      lib: Optional[Library] = None,
721      disable_dynamo: bool = False,
722  ) -> Optional[Callable[[Callable[..., object]], None]]:
723      # See impl()
724      if isinstance(types, str):
725          types = (types,)
726      keys = set({})
727      for typ in types:
728          is_dispatch_key = torch._C._parse_dispatch_key(typ)
729          if is_dispatch_key:
730              # We also support passing a DispatchKey to impl. Please prefer using
731              # the higher-level torch.library APIs and only pass DispatchKey to
732              # torch.library.impl with caution (or even better, don't use this
733              # option and file an issue on GitHub for what you need).
734              # We don't advertise this to users because
735              # it is very easy to shoot yourself in the foot.
736              keys.add(typ)
737          else:
738              keys.add(_device_type_to_key(typ))
739  
740      def register_(func: Callable[..., object]) -> None:
741          namespace, _ = torch._library.utils.parse_namespace(qualname)
742  
743          if lib is None:
744              use_lib = Library(namespace, "FRAGMENT")
745              _keep_alive.append(use_lib)
746          else:
747              use_lib = lib
748          if disable_dynamo:
749  
750              @torch._disable_dynamo
751              def func_no_dynamo(*args, **kwargs):
752                  return func(*args, **kwargs)
753  
754              for key in keys:
755                  use_lib.impl(qualname, func_no_dynamo, key)
756          else:
757              for key in keys:
758                  use_lib.impl(qualname, func, key)
759  
760      if func is None:
761          return register_
762      else:
763          register_(func)
764          return None

805

807      This API may be used as a decorator.

833          >>> @torch.library.register_kernel("mylib::numpy_sin", "cuda")

835          >>>     x_np = x.cpu().numpy()

855      if opdef is not None:
856          return opdef.register_kernel(device_types, func)

865      op: _op_identifier,
866      device_type: str,
867      cast_inputs: _dtype,
868      /,
869      *,
870      lib: Optional[Library] = None,
871  ):
872      r"""Register an autocast dispatch rule for this custom op. 
873   
874      Valid `device_type` include: "cpu" and "cuda". 
875   
876      Args: 
877          op (str | OpOverload): The operator to register an autocast dispatch rule to. 
878          device_type(str):  Device type to use. 'cuda' or 'cpu'. 
879              The type is the same as the `type` attribute of a :class:`torch.device`. 
880              Thus, you may obtain the device type of a tensor using `Tensor.device.type`. 
881          cast_inputs (:class:`torch.dtype`): When custom op runs in an autocast-enabled region, 
882              casts incoming floating-point Tensors to the target dtype (non-floating-point Tensors 
883              are not affected), then executes custom op with autocast disabled. 
884          lib (Optional[Library]): If provided, the lifetime of this registration 
885   
886      Examples:: 
887          >>> # xdoctest: +REQUIRES(env:TORCH_DOCTEST_CUDA) 
888          >>> import torch 
889          >>> from torch import Tensor 
890          >>> from torch.library import custom_op 
891          >>> 
892          >>> # Create a custom op that works on cuda 
893          >>> @torch.library.custom_op("mylib::my_sin", mutates_args=()) 
894          >>> def my_sin(x: Tensor) -> Tensor: 
895          >>>     return torch.sin(x) 
896          >>> 
897          >>> # Register autocast dispatch rule for the cuda device 
898          >>> torch.library.register_autocast("mylib::my_sin", "cuda", torch.float16) 
899          >>> 
900          >>> x = torch.randn(3, dtype=torch.float32, device="cuda") 
901          >>> with torch.autocast("cuda", dtype=torch.float16): 
902          >>>     y = torch.ops.mylib.my_sin(x) 
903          >>> assert y.dtype == torch.float16 
904   
905      """
906      if not isinstance(
907          op, (str, torch._ops.OpOverload, torch._library.custom_ops.CustomOpDef)
908      ):
909          raise ValueError(
910              f"register_autocast({op}): got unexpected type for op: {type(op)}"
911          )
912      if device_type not in ["cpu", "cuda"]:
913          raise ValueError(f"Unknown device type: {device_type}")
914  
915      if isinstance(op, torch._ops.OpOverload):
916          op = op._name
917      opdef = _maybe_get_opdef(op)
918      if opdef is not None:
919          return opdef.register_autocast(device_type, cast_inputs)
920  
921      assert isinstance(op, str)
922      qualname = op
923      _op = torch._library.utils.lookup_op(qualname)
924  
925      namespace, opname = torch._library.utils.parse_namespace(qualname)
926      if lib is None:
927          lib = Library(namespace, "FRAGMENT")
928          _keep_alive.append(lib)
929  
930      def kernel(_, *args, **kwargs):
931          assert len(kwargs) == 0, "Custom ops do not support kwargs yet."
932          autocast_keyset = torch._C.DispatchKeySet(
933              torch._C.DispatchKey.AutocastCPU
934          ) | torch._C.DispatchKeySet(torch._C.DispatchKey.AutocastCUDA)
935          with torch._C._ExcludeDispatchKeyGuard(autocast_keyset):
936              return _op(*_cast(args, device_type, cast_inputs))
937  
938      if device_type == "cuda":
939          return lib.impl(opname, kernel, "AutocastCUDA", with_keyset=True)
940      else:
941          # device_type is "cpu"
942          return lib.impl(opname, kernel, "AutocastCPU", with_keyset=True)

945  def register_fake(
946      op: _op_identifier,
947      func: Optional[Callable] = None,
948      /,
949      *,
950      lib: Optional[Library] = None,
951      _stacklevel: int = 1,
952      allow_override: bool = False,
953  ):
954      r"""Register a FakeTensor implementation ("fake impl") for this operator. 
955   
956      Also sometimes known as a "meta kernel", "abstract impl". 
957   
958      An "FakeTensor implementation" specifies the behavior of this operator on 
959      Tensors that carry no data ("FakeTensor"). Given some input Tensors with 
960      certain properties (sizes/strides/storage_offset/device), it specifies 
961      what the properties of the output Tensors are. 
962   
963      The FakeTensor implementation has the same signature as the operator. 
964      It is run for both FakeTensors and meta tensors. To write a FakeTensor 
965      implementation, assume that all Tensor inputs to the operator are 
966      regular CPU/CUDA/Meta tensors, but they do not have storage, and 
967      you are trying to return regular CPU/CUDA/Meta tensor(s) as output. 
968      The FakeTensor implementation must consist of only PyTorch operations 
969      (and may not directly access the storage or data of any input or 
970      intermediate Tensors). 
971   
972      This API may be used as a decorator (see examples). 
973   
974      For a detailed guide on custom ops, please see 
975      https://pytorch.org/tutorials/advanced/custom_ops_landing_page.html 
976   
977      Args: 
978          op_name: Operator name (along with the overload) or OpOverload object. 
979          func: Fake tensor implementation. 
980          lib (Optional[Library]): Library to register the fake tensor to. 
981          allow_override: Flag controlling if we want to override an 
982                          existing registered fake impl. This is by default off, 
983                          and will error you're trying to register a fake impl to 
984                          an operator that already has a fake impl. This also only 
985                          applies if the custom operator was not created via 
986                          torch.library.custom_op, as overriding and existing fake 
987                          impl is already allowed. 
988   
989      Examples: 
990          >>> import torch 
991          >>> import numpy as np 
992          >>> from torch import Tensor 
993          >>> 
994          >>> # Example 1: an operator without data-dependent output shape 
995          >>> @torch.library.custom_op("mylib::custom_linear", mutates_args=()) 
996          >>> def custom_linear(x: Tensor, weight: Tensor, bias: Tensor) -> Tensor: 
997          >>>     raise NotImplementedError("Implementation goes here") 
998          >>> 
999          >>> @torch.library.register_fake("mylib::custom_linear") 
1000         >>> def _(x, weight, bias): 
1001         >>>     assert x.dim() == 2 
1002         >>>     assert weight.dim() == 2 
1003         >>>     assert bias.dim() == 1 
1004         >>>     assert x.shape[1] == weight.shape[1] 
1005         >>>     assert weight.shape[0] == bias.shape[0] 
1006         >>>     assert x.device == weight.device 
1007         >>> 
1008         >>>     return (x @ weight.t()) + bias 
1009         >>> 
1010         >>> with torch._subclasses.fake_tensor.FakeTensorMode(): 
1011         >>>     x = torch.randn(2, 3) 
1012         >>>     w = torch.randn(3, 3) 
1013         >>>     b = torch.randn(3) 
1014         >>>     y = torch.ops.mylib.custom_linear(x, w, b) 
1015         >>> 
1016         >>> assert y.shape == (2, 3) 
1017         >>> 
1018         >>> # Example 2: an operator with data-dependent output shape 
1019         >>> @torch.library.custom_op("mylib::custom_nonzero", mutates_args=()) 
1020         >>> def custom_nonzero(x: Tensor) -> Tensor: 
1021         >>>     x_np = x.numpy(force=True) 
1022         >>>     res = np.stack(np.nonzero(x_np), axis=1) 
1023         >>>     return torch.tensor(res, device=x.device) 
1024         >>> 
1025         >>> @torch.library.register_fake("mylib::custom_nonzero") 
1026         >>> def _(x): 
1027         >>> # Number of nonzero-elements is data-dependent. 
1028         >>> # Since we cannot peek at the data in an fake impl, 
1029         >>> # we use the ctx object to construct a new symint that 
1030         >>> # represents the data-dependent size. 
1031         >>>     ctx = torch.library.get_ctx() 
1032         >>>     nnz = ctx.new_dynamic_size() 
1033         >>>     shape = [nnz, x.dim()] 
1034         >>>     result = x.new_empty(shape, dtype=torch.int64) 
1035         >>>     return result 
1036         >>> 
1037         >>> from torch.fx.experimental.proxy_tensor import make_fx 
1038         >>> 
1039         >>> x = torch.tensor([0, 1, 2, 3, 4, 0]) 
1040         >>> trace = make_fx(torch.ops.mylib.custom_nonzero, tracing_mode="symbolic")(x) 
1041         >>> trace.print_readable() 
1042         >>> 
1043         >>> assert torch.allclose(trace(x), torch.ops.mylib.custom_nonzero(x)) 
1044  
1045     """
1046     if not isinstance(
1047         op, (str, torch._ops.OpOverload, torch._library.custom_ops.CustomOpDef)
1048     ):
1049         raise ValueError(f"register_fake({op}): got unexpected type for op: {type(op)}")
1050     if isinstance(op, torch._ops.OpOverload):
1051         op = op._name
1052     opdef = _maybe_get_opdef(op)
1053     if opdef is not None:
1054         if func is None:
1055             return opdef.register_fake
1056         else:
1057             return opdef.register_fake(func)
1058     assert isinstance(op, str)
1059 
1060     stacklevel = _stacklevel
1061 
1062     def register(func):
1063         namespace, op_name = torch._library.utils.parse_namespace(op)
1064         if lib is None:
1065             use_lib = Library(namespace, "FRAGMENT")
1066             _keep_alive.append(use_lib)
1067         else:
1068             use_lib = lib
1069         use_lib._register_fake(
1070             op_name, func, _stacklevel=stacklevel + 1, allow_override=allow_override
1071         )
1072         return func
1073 
1074     if func is None:
1075         return register
1076     else:
1077         stacklevel += 1
1078         return register(func)

1081 def register_autograd(
1082     op: _op_identifier,
1083     backward: Callable,
1084     /,
1085     *,
1086     setup_context: Optional[Callable] = None,
1087     lib=None,
1088 ) -> None:
1089     r"""Register a backward formula for this custom op. 
1090  
1091     In order for an operator to work with autograd, you need to register 
1092     a backward formula: 
1093     1. You must tell us how to compute gradients during the backward pass 
1094     by providing us a "backward" function. 
1095     2. If you need any values from the forward to compute gradients, you can 
1096     use `setup_context` to save values for backward. 
1097  
1098     ``backward`` runs during the backward pass. It accepts ``(ctx, *grads)``: 
1099     - ``grads`` is one or more gradients. The number of gradients matches 
1100     the number of outputs of the operator. 
1101     The ``ctx`` object is `the same ctx object <context_method_mixins>`_ used by 
1102     :class:`torch.autograd.Function`. The semantics of ``backward_fn`` are the 
1103     same as :meth:`torch.autograd.Function.backward`. 
1104  
1105     ``setup_context(ctx, inputs, output)`` runs during the forward pass. 
1106     Please save quantities needed for backward onto the ``ctx`` object via 
1107     either :meth:`torch.autograd.function.FunctionCtx.save_for_backward` 
1108     or assigning them as attributes of ``ctx``. If your custom op has 
1109     kwarg-only arguments, we expect the signature of ``setup_context`` 
1110     to be ``setup_context(ctx, inputs, keyword_only_inputs, output)``. 
1111  
1112     Both ``setup_context_fn`` and ``backward_fn`` must be traceable. That is, 
1113     they may not directly access :meth:`torch.Tensor.data_ptr` and they must 
1114     not depend on or mutate global state. If you need a non-traceable backward, 
1115     you can make it a separate custom_op that you call inside ``backward_fn``. 
1116  
1117     If you need different autograd behavior on different devices, then we 
1118     recommend creating two different custom operators, one for each device 
1119     that needs different behavior, and switching between them at runtime. 
1120  
1121     Examples: 
1122         >>> import torch 
1123         >>> import numpy as np 
1124         >>> from torch import Tensor 
1125         >>> 
1126         >>> @torch.library.custom_op("mylib::numpy_sin", mutates_args=()) 
1127         >>> def numpy_sin(x: Tensor) -> Tensor: 
1128         >>>     x_np = x.cpu().numpy() 
1129         >>>     y_np = np.sin(x_np) 
1130         >>>     return torch.from_numpy(y_np).to(device=x.device) 
1131         >>> 
1132         >>> def setup_context(ctx, inputs, output) -> Tensor: 
1133         >>>     x, = inputs 
1134         >>>     ctx.save_for_backward(x) 
1135         >>> 
1136         >>> def backward(ctx, grad): 
1137         >>>     x, = ctx.saved_tensors 
1138         >>>     return grad * x.cos() 
1139         >>> 
1140         >>> torch.library.register_autograd( 
1141         ...     "mylib::numpy_sin", backward, setup_context=setup_context 
1142         ... ) 
1143         >>> 
1144         >>> x = torch.randn(3, requires_grad=True) 
1145         >>> y = numpy_sin(x) 
1146         >>> (grad_x,) = torch.autograd.grad(y, x, torch.ones_like(y)) 
1147         >>> assert torch.allclose(grad_x, x.cos()) 
1148         >>> 
1149         >>> # Example with a keyword-only arg 
1150         >>> @torch.library.custom_op("mylib::numpy_mul", mutates_args=()) 
1151         >>> def numpy_mul(x: Tensor, *, val: float) -> Tensor: 
1152         >>>     x_np = x.cpu().numpy() 
1153         >>>     y_np = x_np * val 
1154         >>>     return torch.from_numpy(y_np).to(device=x.device) 
1155         >>> 
1156         >>> def setup_context(ctx, inputs, keyword_only_inputs, output) -> Tensor: 
1157         >>>     ctx.val = keyword_only_inputs["val"] 
1158         >>> 
1159         >>> def backward(ctx, grad): 
1160         >>>     return grad * ctx.val 
1161         >>> 
1162         >>> torch.library.register_autograd( 
1163         ...     "mylib::numpy_mul", backward, setup_context=setup_context 
1164         ... ) 
1165         >>> 
1166         >>> x = torch.randn(3, requires_grad=True) 
1167         >>> y = numpy_mul(x, val=3.14) 
1168         >>> (grad_x,) = torch.autograd.grad(y, x, torch.ones_like(y)) 
1169         >>> assert torch.allclose(grad_x, torch.full_like(x, 3.14)) 
1170  
1171     """
1172     if not isinstance(
1173         op, (str, torch._ops.OpOverload, torch._library.custom_ops.CustomOpDef)
1174     ):
1175         raise ValueError(
1176             f"register_autograd({op}): got unexpected type for op: {type(op)}"
1177         )
1178     if isinstance(op, torch._ops.OpOverload):
1179         op = op._name
1180     opdef = _maybe_get_opdef(op)
1181     if opdef is not None:
1182         opdef.register_autograd(backward, setup_context=setup_context)
1183         return
1184 
1185     assert isinstance(op, str)
1186     qualname = op
1187     op = torch._library.utils.lookup_op(qualname)
1188     schema = op._schema
1189     if not _library.utils.is_functional_schema(schema):
1190         raise RuntimeError(
1191             f"Cannot register autograd formula for non-functional operator "
1192             f"{op} with schema {schema}. Please create "
1193             f"a functional operator and register an autograd formula for that."
1194         )
1195     if _library.utils.has_kwarg_only_tensors(schema):
1196         raise NotImplementedError(
1197             f"register_autograd with kwarg-only Tensor args. In the original "
1198             f"definition of the op, please make your tensors not kwarg-only. "
1199             f"Got: {schema}"
1200         )
1201 
1202     info = _library.autograd.Info(backward, setup_context)
1203     autograd_kernel = _library.autograd.make_autograd_impl(op, info)
1204     namespace, opname = torch._library.utils.parse_namespace(qualname)
1205     if lib is None:
1206         lib = Library(namespace, "FRAGMENT")
1207         _keep_alive.append(lib)
1208     lib.impl(opname, autograd_kernel, "Autograd", with_keyset=True)

1299     r"""Register a vmap implementation to support :func:`torch.vmap` for this custom op. 
1300  
1301     This API may be used as a decorator (see examples). 
1302  
1303     In order for an operator to work with :func:`torch.vmap`, you may need to register a 
1304     vmap implementation in the following signature: 
1305  
1306         ``vmap_func(info, in_dims: Tuple[Optional[int]], *args, **kwargs)``, 
1307  
1308     where ``*args`` and ``**kwargs`` are the arguments and kwargs for ``op``. 
1309     We do not support kwarg-only Tensor args. 
1310  
1311     It specifies how do we compute the batched version of ``op`` given inputs with an additional 
1312     dimension (specified by ``in_dims``). 
1313  
1314     For each arg in ``args``, ``in_dims`` has a corresponding ``Optional[int]``. It is ``None`` 
1315     if the arg is not a Tensor or if the arg is not being vmapped over, otherwise, it is an integer 
1316     specifying what dimension of the Tensor is being vmapped over. 
1317  
1318     ``info`` is a collection of additional metadata that may be helpful: 
1319     ``info.batch_size`` specifies the size of the dimension being vmapped over, while 
1320     ``info.randomness`` is the ``randomness`` option that was passed to :func:`torch.vmap`. 
1321  
1322     The return of the function ``func`` is a tuple of ``(output, out_dims)``. Similar to ``in_dims``, 
1323     ``out_dims`` should be of the same structure as ``output`` and contain one ``out_dim`` 
1324     per output that specifies if the output has the vmapped dimension and what index it is in. 
1325  
1326     Examples: 
1327         >>> import torch 
1328         >>> import numpy as np 
1329         >>> from torch import Tensor 
1330         >>> from typing import Tuple 
1331         >>> 
1332         >>> def to_numpy(tensor): 
1333         >>>     return tensor.cpu().numpy() 
1334         >>> 
1335         >>> lib = torch.library.Library("mylib", "FRAGMENT") 
1336         >>> @torch.library.custom_op("mylib::numpy_cube", mutates_args=()) 
1337         >>> def numpy_cube(x: Tensor) -> Tuple[Tensor, Tensor]: 
1338         >>>     x_np = to_numpy(x) 
1339         >>>     dx = torch.tensor(3 * x_np ** 2, device=x.device) 
1340         >>>     return torch.tensor(x_np ** 3, device=x.device), dx 
1341         >>> 
1342         >>> def numpy_cube_vmap(info, in_dims, x): 
1343         >>>     result = numpy_cube(x) 
1344         >>>     return result, (in_dims[0], in_dims[0]) 
1345         >>> 
1346         >>> torch.library.register_vmap(numpy_cube, numpy_cube_vmap) 
1347         >>> 
1348         >>> x = torch.randn(3) 
1349         >>> torch.vmap(numpy_cube)(x) 
1350         >>> 
1351         >>> @torch.library.custom_op("mylib::numpy_mul", mutates_args=()) 
1352         >>> def numpy_mul(x: Tensor, y: Tensor) -> Tensor: 
1353         >>>     return torch.tensor(to_numpy(x) * to_numpy(y), device=x.device) 
1354         >>> 
1355         >>> @torch.library.register_vmap("mylib::numpy_mul") 
1356         >>> def numpy_mul_vmap(info, in_dims, x, y): 
1357         >>>     x_bdim, y_bdim = in_dims 
1358         >>>     x = x.movedim(x_bdim, -1) if x_bdim is not None else x.unsqueeze(-1) 
1359         >>>     y = y.movedim(y_bdim, -1) if y_bdim is not None else y.unsqueeze(-1) 
1360         >>>     result = x * y 
1361         >>>     result = result.movedim(-1, 0) 
1362         >>>     return result, 0 
1363         >>> 
1364         >>> 
1365         >>> x = torch.randn(3) 
1366         >>> y = torch.randn(3) 
1367         >>> torch.vmap(numpy_mul)(x, y) 
1368  
1369     .. note:: 
1370         The vmap function should aim to preserve the semantics of the entire custom operator. 
1371         That is, ``grad(vmap(op))`` should be replaceable with a ``grad(map(op))``. 
1372  
1373         If your custom operator has any custom behavior in the backward pass, please 
1374         keep this in mind.

1378         op, (str, torch._ops.OpOverload, torch._library.custom_ops.CustomOpDef)

1381     if isinstance(op, torch._ops.OpOverload):

1493 def opcheck(
1494     op: Union[torch._ops.OpOverload, torch._ops.OpOverloadPacket, CustomOpDef],
1495     args: tuple[Any, ...],
1496     kwargs: Optional[dict[str, Any]] = None,
1497     *,
1498     test_utils: Union[str, Sequence[str]] = _OPCHECK_DEFAULT_UTILS,
1499     raise_exception: bool = True,
1500     atol=None,
1501     rtol=None,
1502 ) -> dict[str, str]:
1503     """Given an operator and some sample arguments, tests if the operator is 
1504     registered correctly. 
1505  
1506     That is, when you use the torch.library/TORCH_LIBRARY APIs to create a 
1507     custom op, you specified metadata (e.g. mutability info) about the custom op 
1508     and these APIs require that the functions you pass them satisfy certain 
1509     properties (e.g. no data pointer access in the fake/meta/abstract kernel) 
1510     ``opcheck`` tests these metadata and properties. 
1511  
1512     Concretely, we test the following: 
1513  
1514     - test_schema: If the schema matches the implementation of 
1515       the operator. For example: if the schema specifies a Tensor is mutated, 
1516       then we check the implementation mutates the Tensor. If the schema 
1517       specifies that we return a new Tensor, then we check that the 
1518       implementation returns a new Tensor (instead of an existing one or 
1519       a view of an existing one). 
1520     - test_autograd_registration: If the operator supports training 
1521       (autograd): we check that its autograd formula is registered via 
1522       torch.library.register_autograd or a manual registration to one 
1523       or more DispatchKey::Autograd keys. Any other DispatchKey-based 
1524       registrations may lead to undefined behavior. 
1525     - test_faketensor: If the operator has a FakeTensor kernel 
1526       (and if it is correct). The FakeTensor kernel is necessary ( 
1527       but not sufficient) for the operator to work with PyTorch compilation 
1528       APIs (torch.compile/export/FX). We check that a FakeTensor kernel 
1529       (also sometimes known as a meta kernel) was registered for the 
1530       operator and that it is correct. This test takes the result of 
1531       running the operator on real tensors and the result of running 
1532       the operator on FakeTensors and checks that they have the same 
1533       Tensor metadata (sizes/strides/dtype/device/etc). 
1534     - test_aot_dispatch_dynamic: If the operator has correct behavior 
1535       with PyTorch compilation APIs (torch.compile/export/FX). 
1536       This checks that the outputs (and gradients, if applicable) are the 
1537       same under eager-mode PyTorch and torch.compile. 
1538       This test is a superset of ``test_faketensor`` and is an e2e test; 
1539       other things it tests are that the operator supports 
1540       functionalization and that the backward pass (if it exists) also 
1541       supports FakeTensor and functionalization. 
1542  
1543     For best results, please call ``opcheck`` multiple times with a 
1544     representative set of inputs. If your operator supports 
1545     autograd, please use ``opcheck`` with inputs with ``requires_grad = True``; 
1546     if your operator supports multiple devices (e.g. CPU and CUDA), please 
1547     use ``opcheck`` with inputs on all supported devices. 
1548  
1549     Args: 
1550         op: The operator. Must either be a function decorated with 
1551             :func:`torch.library.custom_op` or an OpOverload/OpOverloadPacket 
1552             found in torch.ops.* (e.g. torch.ops.aten.sin, torch.ops.mylib.foo) 
1553         args: The args to the operator 
1554         kwargs: The kwargs to the operator 
1555         test_utils: Tests that we should run. Default: all of them. 
1556             Example: ("test_schema", "test_faketensor") 
1557         raise_exception: If we should raise an exception on the first 
1558             error. If False, we will return a dict with information 
1559             on if each test passed or not. 
1560         rtol (Optional[float]): Relative tolerance for floating point comparisons. 
1561             If specified ``atol`` must also be specified. 
1562             If omitted, default values based on the ``dtype`` are selected 
1563             (see the table in :func:`torch.testing.assert_close`). 
1564         atol (Optional[float]): Absolute tolerance for floating point comparisons. 
1565             If specified ``rtol`` must also be specified. 
1566             If omitted, default values based on the ``dtype`` are selected 
1567             (see the table in :func:`torch.testing.assert_close`). 
1568  
1569     .. warning:: 
1570  
1571         opcheck and :func:`torch.autograd.gradcheck` test different things; 
1572         opcheck tests if your usage of torch.library APIs is correct while 
1573         :func:`torch.autograd.gradcheck` tests if your autograd formula is 
1574         mathematically correct. Use both to test custom ops that support 
1575         gradient computation. 
1576  
1577     Example: 
1578  
1579         >>> # xdoctest: +REQUIRES(env:TORCH_DOCTEST_CUDA) 
1580         >>> @torch.library.custom_op("mylib::numpy_mul", mutates_args=()) 
1581         >>> def numpy_mul(x: Tensor, y: float) -> Tensor: 
1582         >>>     x_np = x.numpy(force=True) 
1583         >>>     z_np = x_np * y 
1584         >>>     return torch.from_numpy(z_np).to(x.device) 
1585         >>> 
1586         >>> @numpy_mul.register_fake 
1587         >>> def _(x, y): 
1588         >>>     return torch.empty_like(x) 
1589         >>> 
1590         >>> def setup_context(ctx, inputs, output): 
1591         >>>     y, = inputs 
1592         >>>     ctx.y = y 
1593         >>> 
1594         >>> def backward(ctx, grad): 
1595         >>>     return grad * ctx.y, None 
1596         >>> 
1597         >>> numpy_mul.register_autograd(backward, setup_context=setup_context) 
1598         >>> 
1599         >>> sample_inputs = [ 
1600         >>>     (torch.randn(3), 3.14), 
1601         >>>     (torch.randn(2, 3, device='cuda'), 2.718), 
1602         >>>     (torch.randn(1, 10, requires_grad=True), 1.234), 
1603         >>>     (torch.randn(64, 64, device='cuda', requires_grad=True), 90.18), 
1604         >>> ] 
1605         >>> 
1606         >>> for args in sample_inputs: 
1607         >>>     torch.library.opcheck(numpy_mul, args) 
1608  
1609     """
1610     import torch.testing._internal.optests as optests
1611 
1612     return optests.opcheck(
1613         op,
1614         args,
1615         kwargs,
1616         test_utils=test_utils,
1617         raise_exception=raise_exception,
1618         rtol=rtol,
1619         atol=atol,
