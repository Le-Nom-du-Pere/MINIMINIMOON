27   #define C10_CUDA_CHECK(EXPR)                                        \
28     do {                                                              \
29       const cudaError_t __err = EXPR;                                 \
30       c10::cuda::c10_cuda_check_implementation(                       \
31           static_cast<int32_t>(__err),                                \
32           __FILE__,                                                   \
33           __func__, /* Line number data type not well-defined between \ 
34                         compilers, so we perform an explicit cast */  \
35           static_cast<uint32_t>(__LINE__),                            \
36           true);                                                      \
37     } while (0)
38   
39   #define C10_CUDA_CHECK_WARN(EXPR)                              \
40     do {                                                         \
41       const cudaError_t __err = EXPR;                            \
42       if (C10_UNLIKELY(__err != cudaSuccess)) {                  \
43         [[maybe_unused]] auto error_unused = cudaGetLastError(); \
44         TORCH_WARN("CUDA warning: ", cudaGetErrorString(__err)); \
45       }                                                          \
46     } while (0)

48   // Indicates that a CUDA error is handled in a non-standard way
49   #define C10_CUDA_ERROR_HANDLED(EXPR) EXPR
50   
51   // Intentionally ignore a CUDA error
52   #define C10_CUDA_IGNORE_ERROR(EXPR)                                   \
53     do {                                                                \
54       const cudaError_t __err = EXPR;                                   \
55       if (C10_UNLIKELY(__err != cudaSuccess)) {                         \
56         [[maybe_unused]] cudaError_t error_unused = cudaGetLastError(); \
57       }                                                                 \
58     } while (0)
59   
60   // Clear the last CUDA error
61   #define C10_CUDA_CLEAR_ERROR()                                      \
62     do {                                                              \
63       [[maybe_unused]] cudaError_t error_unused = cudaGetLastError(); \
64     } while (0)

66   // This should be used directly after every kernel launch to ensure
67   // the launch happened correctly and provide an early, close-to-source
68   // diagnostic if it didn't.
69   #define C10_CUDA_KERNEL_LAUNCH_CHECK() C10_CUDA_CHECK(cudaGetLastError())

71   /// Launches a CUDA kernel appending to it all the information need to handle
72   /// device-side assertion failures. Checks that the launch was successful.
73   #define TORCH_DSA_KERNEL_LAUNCH(                                      \
74       kernel, blocks, threads, shared_mem, stream, ...)                 \
75     do {                                                                \
76       auto& launch_registry =                                           \
77           c10::cuda::CUDAKernelLaunchRegistry::get_singleton_ref();     \
78       kernel<<<blocks, threads, shared_mem, stream>>>(                  \
79           __VA_ARGS__,                                                  \
80           launch_registry.get_uvm_assertions_ptr_for_current_device(),  \
81           launch_registry.insert(                                       \
82               __FILE__, __FUNCTION__, __LINE__, #kernel, stream.id())); \
83       C10_CUDA_KERNEL_LAUNCH_CHECK();                                   \
84     } while (0)
