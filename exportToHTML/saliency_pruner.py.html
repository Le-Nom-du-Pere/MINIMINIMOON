5    class SaliencyPruner(BaseStructuredSparsifier):
6        """ 
7        Prune rows based on the saliency (L1 norm) of each row. 
8     
9        This pruner works on N-Dimensional weight tensors. 
10       For each row, we will calculate the saliency, whic is the sum the L1 norm of all weights in that row. 
11       We expect that the resulting saliency vector has the same shape as our mask. 
12       We then pick elements to remove until we reach the target sparsity_level. 
13       """
14   
15       def update_mask(self, module, tensor_name, **kwargs):
16           # tensor_name will give you the FQN, all other entries in sparse config is present in kwargs
17           weights = getattr(module, tensor_name)
18           mask = getattr(module.parametrizations, tensor_name)[0].mask
19   
20           # use negative weights so we can use topk (we prune out the smallest)
21           if weights.dim() <= 1:
22               raise Exception(  # noqa: TRY002
23                   "Structured pruning can only be applied to a 2+dim weight tensor!"
24               )
25           saliency = -weights.norm(dim=tuple(range(1, weights.dim())), p=1)
26           assert saliency.shape == mask.shape
27   
28           num_to_pick = int(len(mask) * kwargs["sparsity_level"])
29           prune = saliency.topk(num_to_pick).indices
30   
31           # Set the mask to be false for the rows we want to prune
32           mask.data[prune] = False
