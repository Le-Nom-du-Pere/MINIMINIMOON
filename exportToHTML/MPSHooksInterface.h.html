16   struct TORCH_API MPSHooksInterface : AcceleratorHooksInterface {
17     // this fails the implementation if MPSHooks functions are called, but
18     // MPS backend is not present.
19     #define FAIL_MPSHOOKS_FUNC(func) \
20       TORCH_CHECK(false, "Cannot execute ", func, "() without MPS backend.");
21   
22     ~MPSHooksInterface() override = default;
23   
24     // Initialize the MPS library state
25     void init() const override {
26       FAIL_MPSHOOKS_FUNC(__func__);
27     }
28     virtual bool hasMPS() const {
29       return false;
30     }
31     virtual bool isOnMacOSorNewer(unsigned major = 13, unsigned minor = 0) const {
32       FAIL_MPSHOOKS_FUNC(__func__);
33     }
34     const Generator& getDefaultGenerator(
35         [[maybe_unused]] DeviceIndex device_index = -1) const override {
36       FAIL_MPSHOOKS_FUNC(__func__);
37     }
38     Generator getNewGenerator(
39         [[maybe_unused]] DeviceIndex device_index) const override {
40       FAIL_MPSHOOKS_FUNC(__func__);
41     }
42     virtual Allocator* getMPSDeviceAllocator() const {
43       FAIL_MPSHOOKS_FUNC(__func__);
44     }
45     virtual void deviceSynchronize() const {
46       FAIL_MPSHOOKS_FUNC(__func__);
47     }
48     virtual void commitStream() const {
49       FAIL_MPSHOOKS_FUNC(__func__);
50     }
51     virtual void* getCommandBuffer() const {
52       FAIL_MPSHOOKS_FUNC(__func__);
53     }
54     virtual void* getDispatchQueue() const {
55       FAIL_MPSHOOKS_FUNC(__func__);
56     }
57     virtual void emptyCache() const {
58       FAIL_MPSHOOKS_FUNC(__func__);
59     }
60     virtual size_t getCurrentAllocatedMemory() const {
61       FAIL_MPSHOOKS_FUNC(__func__);
62     }
63     virtual size_t getDriverAllocatedMemory() const {
64       FAIL_MPSHOOKS_FUNC(__func__);
65     }
66     virtual size_t getRecommendedMaxMemory() const {
67       FAIL_MPSHOOKS_FUNC(__func__);
68     }
69     virtual void setMemoryFraction(double /*ratio*/) const {
70       FAIL_MPSHOOKS_FUNC(__func__);
71     }
72     virtual void profilerStartTrace(const std::string& mode, bool waitUntilCompleted) const {
73       FAIL_MPSHOOKS_FUNC(__func__);
74     }
75     virtual void profilerStopTrace() const {
76       FAIL_MPSHOOKS_FUNC(__func__);
77     }
78     virtual uint32_t acquireEvent(bool enable_timing) const {
79       FAIL_MPSHOOKS_FUNC(__func__);
80     }
81     Device getDeviceFromPtr(void* data) const override {
82       TORCH_CHECK(false, "Cannot get device of pointer on MPS without ATen_mps library. ");
83     }
84     virtual void releaseEvent(uint32_t event_id) const {
85       FAIL_MPSHOOKS_FUNC(__func__);
86     }
87     virtual void recordEvent(uint32_t event_id) const {
88       FAIL_MPSHOOKS_FUNC(__func__);
89     }
90     virtual void waitForEvent(uint32_t event_id) const {
91       FAIL_MPSHOOKS_FUNC(__func__);
92     }
93     virtual void synchronizeEvent(uint32_t event_id) const {
94       FAIL_MPSHOOKS_FUNC(__func__);
95     }
96     virtual bool queryEvent(uint32_t event_id) const {
97       FAIL_MPSHOOKS_FUNC(__func__);
98     }
99     virtual double elapsedTimeOfEvents(uint32_t start_event_id, uint32_t end_event_id) const {
100      FAIL_MPSHOOKS_FUNC(__func__);
101    }
102    bool hasPrimaryContext(DeviceIndex device_index) const override {
103      FAIL_MPSHOOKS_FUNC(__func__);
104    }
105    bool isPinnedPtr(const void* data) const override {
106      return false;
107    }
108    Allocator* getPinnedMemoryAllocator() const override {
109      FAIL_MPSHOOKS_FUNC(__func__);
110    }
111    #undef FAIL_MPSHOOKS_FUNC
112  };
