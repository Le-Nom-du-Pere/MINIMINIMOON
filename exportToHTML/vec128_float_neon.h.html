46     static float32x4_t impl(
47         const float32x4_t& a,
48         const float32x4_t& b,
49         float32x4_t& res);

52   template <int index>
53   struct BlendRegs<index, true> {
54     static float32x4_t impl(
55         const float32x4_t& a,
56         const float32x4_t& b,
57         float32x4_t& res) {
58       return vsetq_lane_f32(vgetq_lane_f32(b, index), res, index);
59     }
60   };

95     template <int64_t mask>
96     static Vectorized<float> blend(
97         const Vectorized<float>& a,
98         const Vectorized<float>& b) {
99       Vectorized<float> vec;
100      vec.values = BlendRegs < 0,
101      (mask & 0x01) != 0 > ::impl(a.values, b.values, vec.values);
102      vec.values = BlendRegs < 1,
103      (mask & 0x02) != 0 > ::impl(a.values, b.values, vec.values);
104      vec.values = BlendRegs < 2,
105      (mask & 0x04) != 0 > ::impl(a.values, b.values, vec.values);
106      vec.values = BlendRegs < 3,
107      (mask & 0x08) != 0 > ::impl(a.values, b.values, vec.values);
108      return vec;
109    }
110    static Vectorized<float> blendv(
111        const Vectorized<float>& a,
112        const Vectorized<float>& b,
113        const Vectorized<float>& mask) {
114      // TODO
115      // NB: This requires that each value, i.e., each uint value,
116      // of the mask either all be zeros or all be 1s.
117      // We perhaps need some kind of an assert?
118      // But that will affect performance.
119      Vectorized<float> vec(mask.values);
120      vec.values =
121          vbslq_f32(vreinterpretq_u32_f32(vec.values), b.values, a.values);
122      return vec;
123    }
124    template <typename step_t>
125    static Vectorized<float> arange(
126        float base = 0.f,
127        step_t step = static_cast<step_t>(1)) {
128      const Vectorized<float> base_vec(base);
129      const Vectorized<float> step_vec(step);
130      const Vectorized<float> step_sizes(0, 1, 2, 3);
131      return fmadd(step_sizes, step_vec, base_vec);
132    }
133    static Vectorized<float> set(
134        const Vectorized<float>& a,
135        const Vectorized<float>& b,
136        int64_t count = size()) {
137      switch (count) {
138        case 0:
139          return a;
140        case 1: {
141          Vectorized<float> vec;
142          static uint32x4_t mask_low = {0xFFFFFFFF, 0x0, 0x0, 0x0};
143          vec.values = vreinterpretq_f32_u32(mask_low);
144          vec.values =
145              vbslq_f32(vreinterpretq_u32_f32(vec.values), b.values, a.values);
146          return vec;
147        }
148        case 2: {
149          Vectorized<float> vec;
150          static uint32x4_t mask_low = {0xFFFFFFFF, 0xFFFFFFFF, 0x0, 0x0};
151          vec.values = vreinterpretq_f32_u32(mask_low);
152          vec.values =
153              vbslq_f32(vreinterpretq_u32_f32(vec.values), b.values, a.values);
154          return vec;
155        }
156        case 3: {
157          Vectorized<float> vec;
158          static uint32x4_t mask_low = {0xFFFFFFFF, 0xFFFFFFFF, 0xFFFFFFFF, 0x0};
159          vec.values = vreinterpretq_f32_u32(mask_low);
160          vec.values =
161              vbslq_f32(vreinterpretq_u32_f32(vec.values), b.values, a.values);
162          return vec;
163        }
164      }
165      return b;
166    }
167    static Vectorized<float> loadu(const void* ptr, int64_t count = size()) {
168      if (count == size()) {
169        return vld1q_f32(reinterpret_cast<const float*>(ptr));
170      } else {
171        __at_align__ float tmp_values[size()];
172        for (const auto i : c10::irange(size())) {
173          tmp_values[i] = 0.0;
174        }
175        std::memcpy(
176            tmp_values,
177            reinterpret_cast<const float*>(ptr),
178            count * sizeof(float));
179        return vld1q_f32(reinterpret_cast<const float*>(tmp_values));
180      }
181    }
182    void store(void* ptr, int64_t count = size()) const {
183      if (count == size()) {
184        vst1q_f32(reinterpret_cast<float*>(ptr), values);
185      } else {
186        float tmp_values[size()];
187        vst1q_f32(reinterpret_cast<float*>(tmp_values), values);
188        std::memcpy(ptr, tmp_values, count * sizeof(float));
189      }
190    }

205    // For boolean version where we want to if any 1/all zero
206    // etc. can be done faster in a different way.
207    int zero_mask() const {
208      __at_align__ float tmp[size()];
209      store(tmp);
210      int mask = 0;
211      for (int i = 0; i < size(); ++i) {
212        if (tmp[i] == 0.f) {
213          mask |= (1 << i);
214        }
215      }
216      return mask;
217    }
218    Vectorized<float> isnan() const {
219      return vreinterpretq_f32_u32(vmvnq_u32(vceqq_f32(values, values)));
220    }
221    bool has_inf_nan() const {
222      __at_align__ float tmp[size()];
223      store(tmp);
224      for (const auto i : c10::irange(size())) {
225        if (_isnan(tmp[i]) || _isinf(tmp[i])) {
226          return true;
227        }
228      }
229      return false;
230    }
231    Vectorized<float> map(float (*const f)(float)) const {
232      __at_align__ float tmp[size()];
233      store(tmp);
234      for (const auto i : c10::irange(size())) {
235        tmp[i] = f(tmp[i]);
236      }
237      return loadu(tmp);
238    }
239    Vectorized<float> map2(
240        const Vectorized<float>& second,
241        float (*const f)(float, float)) const {
242      __at_align__ float tmp[size()];
243      __at_align__ float tmp_second[size()];
244      store(tmp);
245      second.store(tmp_second);
246      for (const auto i : c10::irange(size())) {
247        tmp[i] = f(tmp[i], tmp_second[i]);
248      }
249      return loadu(tmp);
250    }

254    Vectorized<float> angle() const {
255      auto zero = Vectorized<float>(0);
256      auto pi = Vectorized<float>(c10::pi<float>);
257      auto tmp = blendv(zero, pi, *this < zero);
258      return blendv(tmp, *this, isnan());

542  template <>
543  inline void convert(const float* src, int32_t* dst, int64_t n) {
544    int64_t i;
545  #ifndef __msvc_cl__
546  #pragma unroll
547  #endif
548    for (i = 0; i <= (n - Vectorized<float>::size());
549         i += Vectorized<float>::size()) {
550      vst1q_s32(dst + i, vcvtq_s32_f32(vld1q_f32(src + i)));
551    }
552  #ifndef __msvc_cl__
553  #pragma unroll
554  #endif
555    for (; i < n; i++) {
556      dst[i] = static_cast<int32_t>(src[i]);
557    }
558  }

602    const Vectorized<float> p4(-1.453152027f);
603    const Vectorized<float> p5(1.061405429f);
604    // sign(x)
605    auto sign_mask = neg_zero_vec & *this;
606    auto abs_vec = this->abs();
607    // t = 1 / (p * abs(x) + 1)
608    auto tmp0 = fmadd(p, abs_vec, one_vec);
609    auto t = one_vec / tmp0;
610    // r = p5 * t ^ 4 + p4 * t ^ 3 + p3 * t ^ 2 + p2 * t + p1
611    auto tmp1 = fmadd(p5, t, p4);
612    auto tmp2 = fmadd(tmp1, t, p3);
613    auto tmp3 = fmadd(tmp2, t, p2);
614    auto r = fmadd(tmp3, t, p1);
615    // - exp(- x * x)
616    auto pow_2 = (*this) * (*this);
617    auto neg_pow_2 = pow_2 ^ neg_zero_vec;
618    auto tmp4 = neg_pow_2.map(
619        std::exp); // This can be swapped for a faster implementation of exp.
620    auto tmp5 = tmp4 ^ neg_zero_vec;
621    // erf(x) = sign(x) * (1 - r * t * exp(- x * x))
622    auto tmp6 = t * tmp5;
623    auto tmp7 = fmadd(tmp6, r, one_vec);
624    return tmp7 ^ sign_mask;
625  }

627  #undef DEFINE_SLEEF_COMPATIBLE_UNARY_ELEMENTWISE_FUNC
628  #endif /* defined(aarch64) */
