9    def extract_ir(filename: str) -> list[str]:
10       BEGIN = "<GRAPH_EXPORT>"
11       END = "</GRAPH_EXPORT>"
12       pfx = None
13       graphs = []
14       with open(filename) as f:
15           split_strs = f.read().split(BEGIN)
16           for i, split_str in enumerate(split_strs):
17               if i == 0:
18                   continue
19               end_loc = split_str.find(END)
20               if end_loc == -1:
21                   continue
22               s = split_str[:end_loc]
23               pfx = split_strs[i - 1].splitlines()[-1]
24               lines = [x[len(pfx):] for x in s.splitlines(keepends=True)]
25               graphs.append(''.join(lines))
26   
27       return graphs

30   def make_tensor_from_type(inp_type: torch._C.TensorType):
31       size = inp_type.sizes()
32       stride = inp_type.strides()
33       device = inp_type.device()
34       dtype = inp_type.dtype()
35       assert size is not None
36       assert stride is not None
37       assert device is not None
38       assert dtype is not None
39       return torch.empty_strided(size=size, stride=stride, device=device, dtype=dtype)

41   def load_graph_and_inputs(ir: str) -> tuple[Any, list[Any]]:
42       graph = torch._C.parse_ir(ir, parse_tensor_constants=True)
43       graph.makeMultiOutputIntoTuple()
44       inputs = []
45       for inp in graph.inputs():
46           if isinstance(inp.type(), torch._C.FloatType):
47               inputs.append(random.uniform(.1, 100))
48           elif isinstance(inp.type(), torch._C.IntType):
49               inputs.append(random.randint(1, 100))
50           elif isinstance(inp.type(), torch._C.TensorType):
51               tensorType = cast(torch._C.TensorType, inp.type())
52               inputs.append(make_tensor_from_type(tensorType))
53           elif isinstance(inp.type(), torch._C.BoolType):
54               inputs.append(random.randint(0, 1) == 1)
55           else:
56               raise NotImplementedError(f"A default value is not implemented for type {inp.type()}")
57   
58       func = torch._C._create_function_from_graph("forward", graph)
59       torch._C._jit_pass_erase_shape_information(func.graph)
60       return (func, inputs)

62   def time_cuda(fn, inputs, test_runs):
63       t = Timer(stmt="fn(*inputs)", globals={"fn": fn, "inputs" : inputs})
64       times = t.blocked_autorange()
65       return times.median * 1000  # time in ms

67   def time_cpu(fn, inputs, test_runs):
68       s = time.perf_counter()
69       for _ in range(test_runs):
70           fn(*inputs)
71       e = time.perf_counter()
72       return (e - s) / test_runs * 1000  # time in ms

74   def run_test(ir, inputs, *, warmup_runs=10, test_runs=20) -> float:
75       graph, _ = load_graph_and_inputs(ir)
76       for _ in range(warmup_runs):
77           graph(*inputs)
78   
79       is_cpu = None
80       for input in inputs:
81           if isinstance(input, torch.Tensor):
82               is_cpu = input.device.type == "cpu"
83               break
84       assert is_cpu is not None
85   
86       out = time_cpu(graph, inputs, test_runs) if is_cpu else time_cuda(graph, inputs, test_runs)
87       return out

89   @contextmanager
90   def no_fuser(*args, **kwargs):
91       old_optimize = torch._C._get_graph_executor_optimize(False)
92       try:
93           yield
94       finally:
95           torch._C._get_graph_executor_optimize(old_optimize)

97   def run_baseline_no_fusion(ir, inputs) -> float:
98       with no_fuser():
99           return run_test(ir, inputs)

102  def run_nnc(ir, inputs, dynamic) -> float:
103      try:
104          strat = [("DYNAMIC", 10)] if dynamic else [("STATIC", 10)]
105          old_strat = torch.jit.set_fusion_strategy(strat)
106          with torch.jit.fuser("fuser1"):
107              return run_test(ir, inputs)
108      finally:
109          torch.jit.set_fusion_strategy(old_strat)

111  def run_nvfuser(ir, inputs) -> float:
112      with torch.jit.fuser("fuser2"):
