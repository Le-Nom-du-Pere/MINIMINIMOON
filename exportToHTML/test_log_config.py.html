8    class TestLogConfig(unittest.TestCase):
9        """Test LOG_LEVEL environment variable support."""
10       
11       def setUp(self):
12           """Reset logging configuration before each test."""
13           # Reset logging configuration
14           for handler in logging.root.handlers[:]:
15               logging.root.removeHandler(handler)
16           logging.root.setLevel(logging.WARNING)
17       
18       def test_get_log_level_valid_values(self):
19           """Test that valid log level strings return correct constants."""
20           test_cases = [
21               ('DEBUG', logging.DEBUG),
22               ('debug', logging.DEBUG),
23               ('INFO', logging.INFO),
24               ('info', logging.INFO),
25               ('WARNING', logging.WARNING),
26               ('warning', logging.WARNING),
27               ('WARN', logging.WARNING),
28               ('warn', logging.WARNING),
29               ('ERROR', logging.ERROR),
30               ('error', logging.ERROR),
31               ('CRITICAL', logging.CRITICAL),
32               ('critical', logging.CRITICAL),
33               ('FATAL', logging.CRITICAL),
34               ('fatal', logging.CRITICAL),
35           ]
36           
37           for level_string, expected_level in test_cases:
38               with self.subTest(level_string=level_string):
39                   result = get_log_level(level_string)
40                   self.assertEqual(result, expected_level)
41       
42       def test_get_log_level_invalid_values(self):
43           """Test that invalid log level strings fall back to INFO."""
44           invalid_levels = ['INVALID', 'TRACE', '123', '', 'foo']
45           
46           for level_string in invalid_levels:
47               with self.subTest(level_string=level_string):
48                   result = get_log_level(level_string)
49                   self.assertEqual(result, logging.INFO)

51       @patch.dict(os.environ, {'LOG_LEVEL': 'DEBUG'})
52       def test_configure_logging_from_env_var(self):
53           """Test that LOG_LEVEL environment variable is respected."""
54           configure_logging()
55           
56           # Check that root logger level is set to DEBUG
57           self.assertEqual(logging.root.level, logging.DEBUG)
58           
59           # Test that debug messages are actually logged
60           with self.assertLogs(level='DEBUG') as log:
61               logger = logging.getLogger('test')
62               logger.debug('Test debug message')
63               self.assertIn('Test debug message', log.output[0])

81       @patch.dict(os.environ, {'LOG_LEVEL': 'INVALID'})
82       def test_configure_logging_invalid_env_var(self):

84           with self.assertLogs('log_config', level='WARNING') as log:
85               configure_logging()
86               
87               # Check that warning was logged about invalid value
88               self.assertTrue(any('Invalid LOG_LEVEL value' in msg for msg in log.output))

94           """Test that log_level parameter overrides environment variable."""
95           with patch.dict(os.environ, {'LOG_LEVEL': 'ERROR'}):
96               configure_logging(log_level='DEBUG')

98               # Check that parameter override works

101      def test_debug_messages_visible_with_debug_level(self):

103          configure_logging(log_level='DEBUG')
104

106          
107          with self.assertLogs(logger, level='DEBUG') as log:
108              logger.debug('This is a debug message')
109              logger.info('This is an info message')
110              logger.warning('This is a warning message')
111              
112              # All three messages should be captured
113              self.assertEqual(len(log.output), 3)
114              self.assertIn('debug message', log.output[0])
115              self.assertIn('info message', log.output[1])
116              self.assertIn('warning message', log.output[2])
