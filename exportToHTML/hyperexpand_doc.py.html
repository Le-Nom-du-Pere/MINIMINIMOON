hyperexpand_doc.py 
1    """ This module cooks up a docstring when imported. Its only purpose is to 
2        be displayed in the sphinx documentation. """
3    
4    from sympy.core.relational import Eq
5    from sympy.functions.special.hyper import hyper
6    from sympy.printing.latex import latex
7    from sympy.simplify.hyperexpand import FormulaCollection
8    
9    c = FormulaCollection()
10   
11   doc = ""
12   
13   for f in c.formulae:
14       obj = Eq(hyper(f.func.ap, f.func.bq, f.z),
15                f.closed_form.rewrite('nonrepsmall'))
16       doc += ".. math::\n  %s\n" % latex(obj)
17   
18   __doc__ = doc
