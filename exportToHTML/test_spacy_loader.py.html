24           """Test model loading with automatic download"""
25           with patch('spacy.load') as mock_load, \
26                patch('spacy.cli.download') as mock_download:
27

92       def test_degraded_mode(self):
93           """Test processor in degraded mode"""
94           with patch('spacy.load') as mock_load, \
95                patch('spacy.cli.download') as mock_download:
96               
97               # Model loading and download both fail
98               mock_load.side_effect = OSError("Model not found")
99               mock_download.side_effect = Exception("Download failed")
100              
101              processor = SafeSpacyProcessor()
102              result = processor.process_text("Test sentence. Another sentence.")
103              
104              self.assertEqual(result['processing_mode'], 'degraded')
105              self.assertEqual(result['tokens'], ['Test', 'sentence.', 'Another', 'sentence.'])
106              self.assertEqual(result['sentences'], ['Test sentence', ' Another sentence', ''])
107              self.assertFalse(processor.is_fully_functional())
