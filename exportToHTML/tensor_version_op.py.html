3    """This module implements tensor version operations for Dynamo tracing. 
4     
5    It provides primitives for handling tensor versioning during tracing, particularly in the 
6    context of functionalization where version operations are handled eagerly on fake tensors. 
7     
8    When we functionalize _tensor_version + _unsafe_set_version_counter, the ops disappear from 
9    the traced graph. We run them eagerly on the fake tensors used for tracing, in order to get 
10   past asserts that would fail in autograd. 
11    
12   Why is this ok? 
13   1) Versions on functional tensors do not make any sense since you cannot mutate a functional 
14      tensor. 
15   2) The whole point of version munging is to trick autograd into doing what we want, and after 
16      AotAutograd there is no longer any need for these ops. 
17    
18   Note this is similar to how no_grad is handled. 
19   """
20   
21   import torch
22   from torch._prims import _make_prim, RETURN_TYPE
23   from torch._subclasses import FakeTensorMode
24   from torch._subclasses.functional_tensor import FunctionalTensorMode
25   
26   
27   _tensor_version = _make_prim(
28       schema="_tensor_version(Tensor self) -> SymInt",
29       return_type=RETURN_TYPE.NEW,
30       meta=torch.ops.aten._version.default,
31       impl_aten=torch.ops.aten._version.default,
32       doc="Tracable unbacked SymInt version of torch.Tensor._version",
33   )
34   
35   
36   @_tensor_version.py_impl(FakeTensorMode)
37   def _tensor_version_fake(fake_mode, self_tensor):
38       """ 
39       The initial dynamo capture of _tensor_version + _unsafe_set_version_counter turns the 
40       `._version` into an unbacked SymInt so that we don't need to specialize on the `._version` 
41       of input tensors to the graph. 
42       """
43       return fake_mode.shape_env.create_unbacked_symint()
44   
45   
46   _unsafe_set_version_counter = _make_prim(
47       schema="_unsafe_set_version_counter(Tensor[] tensors, SymInt[] versions) -> ()",
48       return_type=RETURN_TYPE.NEW,
49       meta=lambda self, version: None,
50       impl_aten=torch._C._autograd._unsafe_set_version_counter,
51       doc="Tracable+SymInt version of torch._C._autograd._unsafe_set_version_counter",
52   )
53   torch.fx.node.has_side_effect(_unsafe_set_version_counter)
54   
55   
56   @_tensor_version.py_impl(FunctionalTensorMode)
57   def _tensor_version_functional(mode, self):
58       return self._version
59   
60   
61   @_unsafe_set_version_counter.py_impl(FunctionalTensorMode)
62   def _unsafe_set_version_counter_functional(ctx, tensors, versions):
63       torch._C._autograd._unsafe_set_version_counter(tensors, versions)
