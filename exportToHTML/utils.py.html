64   def _collect_and_set_constant_attrs(
65       graph_signature, constants, mod
66   ) -> "ConstantAttrMap":
67       # the exported module will store constants & non-persistent buffers such that
68       # retracing treats them as persistent buffers, so we inform the constants lifting pass
69       # and overwrite the new graph signature using the previous program. This is intended to only be used
70       # in run_decompositions where we still have access to original EP.
71       from torch._export.passes.lift_constants_pass import ConstantAttrMap
72   
73       constant_attrs = ConstantAttrMap()
74       non_persistent_buffers = {
75           spec.target
76           for spec in graph_signature.input_specs
77           if spec.kind == InputKind.BUFFER and not spec.persistent
78       }
79       for name, value in constants.items():
80           if name in non_persistent_buffers:
81               continue
82           # recursive getattr
83           _mod = mod
84           *atoms, attr = name.split(".")
85           for atom in atoms:
86               _mod = getattr(_mod, atom)
87           # remove as buffer, reassign as constant/non-persistent buffer
88           _mod._buffers.pop(attr, None)
89           setattr(_mod, attr, value)
90           constant_attrs.add(value, name)
91       return constant_attrs

94   def _register_constants_as_buffers(
95       mod: torch.fx.GraphModule, state_dict, non_persistent_buffers
96   ):
97       # TODO some annoying circular dependency issue
98       from torch.export.unflatten import _assign_attr, _AttrKind
99   
100      temp_registered_constants = set()
101  
102      for node in mod.graph.nodes:
103          if node.op == "get_attr":
104              target = torch.fx.graph_module._get_attr(mod, node.target)
105              if isinstance(target, torch.Tensor):
106                  # Make sure we also check if the original buffer is
107                  # non persistent as well.
108                  if (node.target not in state_dict) and (
109                      node.target not in non_persistent_buffers
110                  ):
111                      torch.fx.graph_module._del_attr(mod, node.target)
112                      _assign_attr(target, mod, node.target, _AttrKind.BUFFER, False)
113                      temp_registered_constants.add(node.target)
114  
115      mod.recompile()
116  
117      return temp_registered_constants

120  def _override_graph_signature_for_temp_registered_constants(
121      sig: "ExportGraphSignature", temp_registered_constants
122  ):
123      for spec in sig.input_specs:
124          if spec.target in temp_registered_constants:
125              spec.kind = InputKind.CONSTANT_TENSOR
126              spec.persistent = None
127  
128      for spec in sig.output_specs:
129          if (
130              spec.kind == OutputKind.BUFFER_MUTATION
131              and spec.target in temp_registered_constants
132          ):
133              raise RuntimeError(
134                  f"Constant {spec.target} is mutated in the forward method. Pls register it as buffer"
135              )
136  
137      return sig

140  def _overwrite_signature_for_non_persistent_buffers(
141      old_sig: "ExportGraphSignature", new_sig: "ExportGraphSignature"
142  ):
143      # overwrite signature for non-persistent buffers
144      non_persistent_buffers = {
145          spec.target
146          for spec in old_sig.input_specs
147          if spec.kind == InputKind.BUFFER and not spec.persistent
148      }
149  
150      for spec in new_sig.input_specs:
151          if spec.kind == InputKind.BUFFER and spec.target in non_persistent_buffers:
152              spec.persistent = False
153      return new_sig

156  def _collect_param_buffer_metadata(mod: torch.fx.GraphModule) -> dict[str, Any]:
157      """ 
158      Param/buffer metadata needs to be saved before lowering to aten IR 
159      because aten IR lifts them, as a result, automatic preservation doesn't work. 
160      This is intended to be called on the strict mode tracing right before lowering to 
161      aten IR OR run_decomposition pass. 
162      """
163      params_buffers_to_node_meta = {}
164  
165      def _getattr(model: torch.fx.GraphModule, attr_name: str):
166          *prefix, field = attr_name.split(".")
167          t = model
168          for item in prefix:
169              t = getattr(t, item, None)  # type: ignore[assignment]
170              assert t is not None
171  
172          return getattr(t, field)
173  
174      for node in mod.graph.nodes:
175          target = node.target
176          meta = node.meta
177          if node.op == "call_module":
178              submodule = _getattr(mod, target)
179              if isinstance(submodule, torch.nn.Module):
180                  for name, _ in submodule.named_parameters(
181                      recurse=True, remove_duplicate=False
182                  ):
183                      params_buffers_to_node_meta[target + "." + name] = meta
184  
185                  for name, _ in submodule.named_buffers(
186                      recurse=True, remove_duplicate=False
187                  ):
188                      params_buffers_to_node_meta[target + "." + name] = meta
189  
190          if node.op == "get_attr":
191              submodule = _getattr(mod, target)
192              if not isinstance(submodule, torch.fx.GraphModule):
193                  params_buffers_to_node_meta[target] = meta
194  
195          # If the call_function uses param as input, we also need to update params' meta
196          # with this call_function node's meta.
197          # This is basically the same flow as torch.fx.traceback.preserve_meta()
198          if node.op == "call_function" and not isinstance(
199              node.target, torch._ops.HigherOrderOperator
200          ):
201              for arg in node._input_nodes:
202                  if arg.op == "get_attr":
203                      for entry in torch.fx.proxy._COPY_META_FIELDS:
204                          #  the custom field should not be copied
205                          if entry == "custom":
206                              continue
207                          if entry in meta:
208                              params_buffers_to_node_meta[arg.target][entry] = meta[entry]
209  
210      return params_buffers_to_node_meta

213  def _populate_param_buffer_metadata_to_new_gm(
214      params_buffers_to_node_meta: dict[str, Any],
215      gm: torch.fx.GraphModule,
216      new_sig: "ExportGraphSignature",
217  ) -> None:
218      """ 
219      Given that we collected param'buffer metadata before, we put them back in 
220      newly traced graph module 
221      """
222      # Don't copy over nn_module_stack, stack_trace metadata for params/buffers nodes
223      for metadata in params_buffers_to_node_meta.values():
224          metadata.pop("nn_module_stack", None)
225          metadata.pop("stack_trace", None)
226  
227      for node in gm.graph.nodes:
228          if node.op == "placeholder":
229              if node.target in new_sig.inputs_to_parameters:
230                  param_name = new_sig.inputs_to_parameters[node.target]
231                  if param_name in params_buffers_to_node_meta:
232                      for k, v in params_buffers_to_node_meta[param_name].items():
233                          node.meta[k] = v
234              if node.target in new_sig.inputs_to_buffers:
235                  buffer_name = new_sig.inputs_to_buffers[node.target]
236                  if buffer_name in params_buffers_to_node_meta:
237                      for k, v in params_buffers_to_node_meta[buffer_name].items():
238                          node.meta[k] = v

256  def _rename_without_collisions(
257      name_map: dict[str, str],
258      orig_name: str,
259      name: str,
260      is_placeholder: bool = False,
261  ):
262      """ 
263      Renames nodes to avoid name collisions, with suffixing. 
264      name_map: map from original name to new name 
265      orig_name: mapping key 
266      name: candidate name (potentially suffixed, e.g. mul_2) 
267      is_placeholder: if the node is a placeholder, avoid detecting suffix 
268      """
269      if name in name_map.values():
270          # non-placeholder nodes may be suffixed with the count
271          # instead of adding another suffix, we will try to increment it
272          match = re.match(r"(.*)_(\d+)", name)
273          if match and not is_placeholder:
274              name, n = match.group(1), int(match.group(2))
275          else:
276              n = 0
277          while (dup_name := f"{name}_{n + 1}") in name_map.values():
278              n += 1
279          name_map[orig_name] = dup_name
280      else:
281          name_map[orig_name] = name
282      return name_map[orig_name]

285  def get_keystr(key_path: KeyPath) -> str:
286      """For a given index into the flat_args, return a human readable string 
287      describing how to access it, e.g. "*args["foo"][0].bar" 
288      """
289      # Prefix the keypath with "*args" or "**kwargs" to make it clearer where
290      # the arguments come from. Ultimately we ought to serialize the
291      # original arg names for the best error message here.
292      args_kwargs_key_path = key_path[0]
293      assert isinstance(args_kwargs_key_path, SequenceKey)
294      if args_kwargs_key_path.idx == 0:
295          return f"*args{keystr(key_path[1:])}"
296      else:
297          kwarg_key = key_path[1]
298          assert isinstance(kwarg_key, MappingKey)
299          name = str(kwarg_key)[1:-1]  # get rid of the enclosed []
300          return f"{name}{keystr(key_path[2:])}"

303  def _check_symint(
304      symint: Union[int, torch.SymInt],
305      arg: int,
306      range_constraints,
307      unification_map,
308      keypath: KeyPath,
309      i: Optional[int] = None,
310  ) -> None:
311      from torch.export.dynamic_shapes import _IntWrapper
312  
313      if (
314          isinstance(arg, torch.SymInt)
315          and not arg.node.expr.is_number
316          or isinstance(arg, _IntWrapper)
317      ):
318          # This can happen when, say, arg is a fake tensor.
319          # We do not run checks on symbolic shapes of fake inputs as
320          # such checks can affect the shape env.
321          return
322  
323      import sympy
324  
325      from torch._export.passes.add_runtime_assertions_for_constraints_pass import (
326          _convert_range_to_int,
327      )
328      from torch.utils._sympy.solve import try_solve
329  
330      if isinstance(symint, torch.SymInt) and len(symint.node.expr.free_symbols) == 1:
331          symbol = next(iter(symint.node.expr.free_symbols))
332          if symbol in unification_map:
333              existing_dim = symint.node.expr.subs(unification_map)
334              if arg != existing_dim:
335                  path = get_keystr(keypath)
336                  if i is not None:
337                      path += f".shape[{i}]"
338                  raise RuntimeError(
339                      f"Expected input at {path} to be equal to {existing_dim}, but got {arg}",
340                  )
341          else:
342              if isinstance(symint.node.expr, sympy.Symbol):
343                  # Short cut for try_solve below. Also useful in cases where
344                  # sympy.Eq(symint.node.expr, arg) would evaluate to False
345                  # purely because symbol is constrained to be size-like,
346                  # e.g., when symint.node.expr = symbol and arg = 0.
347                  unification_map[symbol] = int(arg)
348              else:
349                  solution = try_solve(sympy.Eq(symint.node.expr, arg), symbol)
350                  if solution is None:
351                      path = get_keystr(keypath)
352                      if i is not None:
353                          path += f".shape[{i}]"
354                      raise RuntimeError(  # noqa: B904
355                          f"Expected input {path} = {arg} to be "
356                          f"of the form {symint.node.expr}, where {symbol} is an integer"
357                      )
358                  else:
359                      unification_map[symbol] = int(solution[1])
360  
361          if symint.node.expr in range_constraints:
362              min_val, max_val = _convert_range_to_int(
363                  range_constraints[symint.node.expr]
364              )
365              # NOTE: we allow dimensions to be 0/1 at runtime
366              if min_val > 2:
367                  if arg < min_val:
368                      path = get_keystr(keypath)
369                      if i is not None:
370                          path += f".shape[{i}]"
371                      raise RuntimeError(
372                          f"Expected input at {path} to be >= {min_val}, but got {arg}",
373                      )
374              if max_val < math.inf:
375                  if arg > max_val:
376                      path = get_keystr(keypath)
377                      if i is not None:
378                          path += f".shape[{i}]"
379                      raise RuntimeError(
380                          f"Expected input at {path} to be <= {max_val}, but got {arg}",
381                      )
382      elif isinstance(symint, torch.SymInt) and not symint.node.expr.is_number:
383          # this means we deferred a guard from export analysis to runtime, let this pass
384          # we'll add a runtime assert checking equality to this replacement expression
385          pass
386      elif arg != symint:
387          path = get_keystr(keypath)
388          if i is not None:
389              path += f".shape[{i}]"
390          raise RuntimeError(
391              f"Expected input at {path} to be equal to {symint}, but got {arg}. "
392              "If you meant for this dimension to be dynamic, please re-export and specify dynamic_shapes "
393              "(e.g. with Dim.DYNAMIC)"
394          )

397  def _check_input_constraints_for_graph(
398      input_placeholders: list[torch.fx.Node], flat_args_with_path, range_constraints
399  ) -> None:
400      import sympy  # noqa: TC002
401  
402      if len(flat_args_with_path) != len(input_placeholders):
403          raise RuntimeError(
404              "Unexpected number of inputs "
405              f"(expected {len(input_placeholders)}, got {len(flat_args_with_path)})"
406          )
407      # NOTE: export already guarantees that the same symbol is used in metadata
408      # for all InputDims related by equality constraints, so we can just unify
409      # symbols with given input dimension values to check equality constraints.
410      unification_map: dict[sympy.Symbol, Any] = {}
411      for (key_path, arg), node in zip(flat_args_with_path, input_placeholders):
412          node_val = node.meta.get("val")
413          if isinstance(node_val, FakeTensor):
414              if not isinstance(arg, torch.Tensor):
415                  raise RuntimeError(
416                      f"Expected input at {get_keystr(key_path)} to be a tensor, but got {type(arg)}",
417                  )
418  
419              if len(node_val.shape) != len(arg.shape):
420                  raise RuntimeError(
421                      f"Unexpected number of dimensions in input at {get_keystr(key_path)}.shape "
422                      f"(expected {node_val.shape}, got {arg.shape})"
423                  )
424  
425              for j, (arg_dim, node_dim) in enumerate(zip(arg.shape, node_val.shape)):
426                  _check_symint(
427                      node_dim, arg_dim, range_constraints, unification_map, key_path, j
428                  )
429  
430          elif isinstance(node_val, (int, float, str)):
431              if type(arg) != type(node_val) or arg != node_val:
432                  raise RuntimeError(
433                      f"Expected input at {get_keystr(key_path)} to be equal to {node_val}, but got {arg}",
434                  )
435          elif isinstance(node_val, torch.SymInt):
436              _check_symint(
437                  node_val, arg, range_constraints, unification_map, key_path, None
438              )

441  def register_dataclass_as_pytree_node(
442      cls: type[Any],
443      flatten_fn: Optional[FlattenFunc] = None,
444      unflatten_fn: Optional[UnflattenFunc] = None,
445      *,
446      serialized_type_name: Optional[str] = None,
447      to_dumpable_context: Optional[ToDumpableContextFn] = None,
448      from_dumpable_context: Optional[FromDumpableContextFn] = None,
449      return_none_fields: bool = False,
450  ) -> None:
451      assert dataclasses.is_dataclass(
452          cls
453      ), f"Only dataclasses can be registered with this function: {cls}"
454  
455      def default_flatten_fn(obj: Any) -> tuple[list[Any], Context]:
456          flattened = []
457          flat_names = []
458          none_names = []
459          for f in dataclasses.fields(obj):
460              name, val = f.name, getattr(obj, f.name)
461              if val is not None or return_none_fields:
462                  flattened.append(val)
463                  flat_names.append(name)
464              else:
465                  none_names.append(name)
466          return flattened, [flat_names, none_names]
467  
468      def default_unflatten_fn(values: Iterable[Any], context: Context) -> Any:
469          flat_names, none_names = context
470          return cls(**dict(zip(flat_names, values)), **dict.fromkeys(none_names))
471  
472      def default_flatten_fn_with_keys(obj: Any) -> tuple[list[Any], Context]:
473          flattened, (flat_names, _none_names) = flatten_fn(obj)  # type: ignore[misc]
474          return [(MappingKey(k), v) for k, v in zip(flat_names, flattened)], flat_names
475  
476      flatten_fn = flatten_fn if flatten_fn is not None else default_flatten_fn
477      unflatten_fn = unflatten_fn if unflatten_fn is not None else default_unflatten_fn
478  
479      if (to_dumpable_context is None) ^ (from_dumpable_context is None):
480          raise ValueError(
481              f"Both to_dumpable_context and from_dumpable_context for {cls} must "
482              "be None or registered."
483          )
484  
485      _register_pytree_node(
486          cls,
487          flatten_fn,
488          unflatten_fn,
489          serialized_type_name=serialized_type_name,
490          flatten_with_keys_fn=default_flatten_fn_with_keys,
491          to_dumpable_context=to_dumpable_context,
492          from_dumpable_context=from_dumpable_context,
493      )

576  def sequential_split(
577      gm: torch.fx.GraphModule,
578      node_call_back: Callable[[torch.fx.Node], Union[torch.fx.Node, bool]],
579  ) -> torch.fx.GraphModule:
580      """ 
581      sequential_split creates a new graph module that splits the input graph module into multiple submodules 
582      based on the node_call_back. It doesn't mutate the input graph module. The node_call_back should return 
583      True if the node is a delimiter.  Delimiter will be the first node in the next submodule. 
584      """
585      from torch.fx.passes.split_module import split_module
586  
587      split_map = {}
588      split_id = 0
589      for node in gm.graph.nodes:
590          if node_call_back(node):
591              split_id += 1
592          split_map[node] = split_id
593  
594      new_gm = split_module(
595          gm,
596          gm,
597          lambda node: split_map[node],
598          keep_original_order=True,
599          keep_original_node_name=True,
600      )
601      # Keep the codegen from original graph module to preserve e.g. pytree info.
602      new_gm.graph._codegen = gm.graph._codegen
603      new_gm.recompile()
604      return new_gm

623  def _insert_aten_to_metadata_assert_pass(gm: torch.fx.GraphModule) -> None:
624      from torch._export.passes._node_metadata_hook import (
625          _node_metadata_hook,
626          _set_node_metadata_hook,
627      )
628  
629      if _DISABLE_ATEN_TO_ASSERTION_PASS:
630          return
631  
632      aten_to_variants = [
633          torch.ops.aten.to.device,
634          torch.ops.aten.to.dtype,
635          torch.ops.aten.to.dtype_layout,
636      ]
637      for node in gm.graph.nodes:

744  def node_inline_(call_mod_node: torch.fx.Node) -> Optional[torch.fx.GraphModule]:
745      """ 
746      Inline the submodule of the given node into the parent module. 
747      Note: we only support the case where submodule takes tensors inputs. 
748      """
749      assert call_mod_node.op == "call_module"
750      gm = call_mod_node.graph.owning_module
751      assert gm is not None
752  
753      assert isinstance(call_mod_node.target, str)
754      sub_gm = getattr(gm, call_mod_node.target)
755  
756      phs = (node for node in sub_gm.graph.nodes if node.op == "placeholder")
757      body = (
758          node for node in sub_gm.graph.nodes if node.op not in ("placeholder", "output")
759      )
760      output = [node for node in sub_gm.graph.nodes if node.op == "output"]
761  
762      for ph, arg in zip(phs, call_mod_node.args):
763          assert isinstance(arg, torch.fx.Node)
764          node_replace_(ph, arg)
765  
766      with gm.graph.inserting_before(call_mod_node):
767          for node in body:
768              new_node = gm.graph.node_copy(node)
769              if node.op == "get_attr":
770                  new_target_name = new_node.target
771                  if hasattr(gm, new_target_name):
772                      # Loop through and find the "submod_{i}" that have no name collision
773                      i = 1
774                      new_target_name = f"submod_{i}"
775                      while hasattr(gm, new_target_name):
776                          i += 1
777                          new_target_name = f"submod_{i}"
778                  new_node.target = new_target_name
779                  setattr(gm, new_node.target, getattr(sub_gm, node.target))
780              node_replace_(node, new_node)
781  
782          if len(output) > 0:
783              assert len(output) == 1 and len(output[0].args) == 1
784              new_output = output[0].args[0]
785  
786              if isinstance(new_output, torch.fx.Node):
787                  # Clear the users of the output node and set
788                  # the users to be the users of original call_module node.
789                  new_output.users.clear()
790                  node_replace_(call_mod_node, new_output)
791              elif isinstance(new_output, (list, tuple)):
792                  # Pop subgraph output node from users.
793                  for node in new_output:
794                      node.users.pop(output[0])
795  
796                  # Inline the get_item calls for the output node.
797                  get_item_users = nodes_filter(
798                      list(call_mod_node.users.keys()),
799                      lambda node: node.op == "call_function"
800                      and node.target == operator.getitem,
801                  )
802                  # get_item_node.args[1] is the idx referring to new_output[idx]
803                  nodes_map(
804                      get_item_users,
805                      lambda get_item_node: node_replace_(
806                          get_item_node,
807                          new_output[get_item_node.args[1]],
808                      ),
809                  )
810                  call_mod_node.graph.erase_node(call_mod_node)
811              else:
812                  raise NotImplementedError(
813                      f"Unsupported output type {type(new_output)}. Expect it to be a Node or a list/tuple of Nodes."
814                  )
815          else:
816              call_mod_node.graph.erase_node(call_mod_node)
817  
818      gm.delete_all_unused_submodules()
819      gm.recompile()
820      return gm

823  def _get_torch_jit_trace_forward_signature(mod: torch.nn.Module) -> inspect.Signature:
824      """ 
825      Get source code and parse argument names using AST. The function returns 
826      a signature of the forward() function. 
827   
828      # TODO: Directly provide inspect.signature compatible TS-d module. 
829      """
830      ast_mod = ast.parse(mod.code)  # type: ignore[call-overload]
831      ast_func_def: ast.FunctionDef = ast_mod.body[0]
832  
833      # FIXME(jiashenc): TorchScript should only allow positional or keywords arguments.
834      arg_type_map = {"args": Parameter.POSITIONAL_OR_KEYWORD}
835  
836      # Traverse all argument types in AST tree and create associated parameters.
837      param_list = []
838      for arg_type, param_type in arg_type_map.items():
839          arg_name_list = [a.arg for a in getattr(ast_func_def.args, arg_type)]
840          for arg_name in arg_name_list:
841              if arg_name == "self":
842                  continue  # Skip self argument.
843              param_list.append(inspect.Parameter(arg_name, param_type))
844  
845      return inspect.Signature(parameters=param_list)
846

848  def _bind_signature_to_inputs(mod, fake_args, fake_kwargs):
849      if isinstance(mod, (torch.jit.ScriptModule, torch.jit.TracedModule)):
850          sig = _get_torch_jit_trace_forward_signature(mod)
851  
852          # Sanity check for placeholder names coming from TorchScript.
853          assert len(sig.parameters) == len(fake_args) + len(fake_kwargs), (
854              "Arguments other than POSITIONAL_OR_KEYWORD kinds in forward() "
855              "are not supported in _get_torch_jit_trace_forward_signature"
856          )
857      else:
858          sig = inspect.signature(mod.forward)
859  
860      # Rather than binding both fake_args and fake_kwargs to sig names, we
861      # (partially) bind only fake_args, while reusing fake_kwarg names. This
862      # ensures that fake_kwargs do not get reordered, which is important to
863      # match flattened user inputs.
864      return {**sig.bind_partial(*fake_args).arguments, **fake_kwargs}

867  def _name_hoo_subgraph_placeholders(gm: torch.fx.GraphModule) -> None:
868      """ 
869      Propagate placeholder names from the top-level graph into HigherOrderOp subgraphs, 
870      and handle collisions with non-placeholders by count suffixing. 
871      Different HOO subgraph types have different input schemas, so we first enumerate them 
872      and gather the top-level named placeholder nodes. 
873      """
874      # gather all HOO subgraphs and their top-level named placeholder nodes
875      subgraph_ph_tuples: list[tuple[torch.fx.GraphModule, list[torch.fx.Node]]] = []
876      for node in gm.graph.nodes:
877          if node.op == "call_function" and isinstance(
878              node.target, torch._ops.HigherOrderOperator
879          ):
880              # HOO subgraphs have varying input schemas, so we enumerate them there
881              if node.target._name == "cond":
882                  _, true_graph, false_graph, cond_args = node._args
883                  subgraph_ph_tuples.append((getattr(gm, true_graph.target), cond_args))
884                  subgraph_ph_tuples.append((getattr(gm, false_graph.target), cond_args))
885              elif node.target._name == "wrap_with_set_grad_enabled":
886                  subgraph, phs = node._args[1], node._args[2:]
887                  subgraph_ph_tuples.append((getattr(gm, subgraph.target), phs))
888              elif node.target._name == "map_impl":
889                  body_graph, array, args = node._args
890                  subgraph_ph_tuples.append(
891                      (getattr(gm, body_graph.target), array + args)
892                  )
893  
894      # propagate names
895      for subgraph, hoo_phs in subgraph_ph_tuples:
896          name_map: dict[str, str] = {}
897          for i, node in enumerate(subgraph.graph.nodes):
898              if i < len(hoo_phs):  # placeholder, retain name
899                  name_map[node.name] = hoo_phs[i].name
900                  node.name = node.target = hoo_phs[i].name
901              else:  # non-placeholder, check for collisions
902                  node.name = _rename_without_collisions(name_map, node.name, node.name)
903  
904          # recurse and recompile
905          _name_hoo_subgraph_placeholders(subgraph)
906          subgraph.recompile()

909  def placeholder_naming_pass(
910      gm: torch.fx.GraphModule,
911      export_graph_signature: "ExportGraphSignature",
912      mod: torch.nn.Module,
913      fake_args,
914      fake_kwargs,
915      fake_params_buffers,
916      constants: dict[str, Any],
917  ) -> None:
918      """ 
919      This pass is run at the end of _export_non_strict() to assign better placeholder node names: 
920          - User inputs: 
921              These follow the signature of mod.forward(), e.g. forward(x, y) produces nodes x, y. 
922              For nested inputs from dictionaries, lists, tuples, or dataclasses, 
923              the names are a concatenation of the path to the tensor. 
924                  e.g. x = { 
925                      'a': torch.randn(), 
926                      'b': [torch.randn(), torch.randn()] 
927                  } 
928              produces nodes x_a, x_b_0, x_b_1. 
929          - Parameters/buffers/constants/custom objects: 
930              These follow the FQN of the object, prefixed by "p", "b", "c", "obj" respectively. 
931                  e.g. self.bar.l0.weight produces "p_bar_l0_weight". 
932          - Effect tokens: 
933              These are named token, token_1, ... 
934      """
935  
936      custom_meta: dict[str, Any] = {}
937      if isinstance(mod, torch.fx.GraphModule):
938          for node in mod.graph.nodes:
939              if "custom" in node.meta:
940                  custom_meta[node.name] = node.meta["custom"]
941  
942      def _strip_name(x):
943          if x.startswith("L__self___"):
944              x = x[len("L__self___") :]
945          elif x.startswith("self_"):
946              x = x[len("self_") :]
947          x = re.sub(r"[^a-zA-Z0-9]", "_", x)
948          return x
949  
950      def _extract_pytree_key(x):
951          if isinstance(x, MappingKey):
952              x = re.sub(r"[^a-zA-Z0-9]", "_", str(x.key))
953              return x
954          elif isinstance(x, SequenceKey):
955              return str(x.idx)
956          elif isinstance(x, GetAttrKey):
957              return x.name
958          else:
959              raise RuntimeError(f"Pytree key of type {type(x)} not handled for {x}")
960  
961      name_map: dict[str, str] = {}
962  
963      # map user input names with mod.forward() signature
964      combined_args = _bind_signature_to_inputs(mod, fake_args, fake_kwargs)
965  
966      flat_args_with_path, _ = tree_flatten_with_path(combined_args)
967      user_input_names = [
968          spec.arg.name
969          for spec in export_graph_signature.input_specs
970          if spec.kind == InputKind.USER_INPUT
971      ]
972  
973      # use pytree path to name nested user inputs
974      for (arg_path, _arg), user_input_name in zip(flat_args_with_path, user_input_names):
975          if user_input_name:
976              _rename_without_collisions(
977                  name_map,
978                  user_input_name,
979                  placeholder_prefixes[InputKind.USER_INPUT]
980                  + "_".join(_extract_pytree_key(x).lower() for x in arg_path),
981                  is_placeholder=True,
982              )
983  
984      # use graph signature input specs to map param/buffer/constant names
985      # name effect tokens as token, token_1, ... (these aren't visible to user)
986      for spec in export_graph_signature.input_specs:
987          if spec.kind == InputKind.USER_INPUT:
988              continue
989          if spec.kind == InputKind.TOKEN:
990              base_name = ""
991          else:
992              base_name = _strip_name(spec.target).lower()
993          base_name = re.sub(r"[^a-zA-Z0-9]", "_", base_name)
994  
995          _rename_without_collisions(
996              name_map,
997              spec.arg.name,
998              placeholder_prefixes[spec.kind] + base_name,
999              is_placeholder=True,
1000         )
1001         if base_name in custom_meta:
1002             # the keys in custom_meta are node names from `mod`,
1003             # which is the base_name here.
1004             # we need the re-mapped name for lookup later
1005             custom_meta[name_map[spec.arg.name]] = custom_meta[base_name]
1006             del custom_meta[base_name]
1007 
1008     # handle naming collisions with call_function/get_attr inputs.
1009     # here, we want to prioritize user input names over call_function names
1010     # e.g. not have forward(self, mul): lead to a placeholder node called mul_13,
1011     # so we increment the suffix of call_function nodes as needed
1012     for node in gm.graph.nodes:
1013         if node.op == "placeholder":
1014             continue
1015         _rename_without_collisions(name_map, node.name, node.name)
1016 
1017     # assign new node names
1018     for node in gm.graph.nodes:
1019         if node.op == "placeholder":
1020             assert node.name in name_map
1021             node.name = node.target = name_map[node.name]
1022             if node.name in custom_meta:
1023                 if node.meta.get("custom") is None:
1024                     node.meta["custom"] = custom_meta[node.name]
1025                 else:
1026                     assert node.meta["custom"] == custom_meta[node.name]
1027             # if the constant obj is an input, we also need to update meta["val"]
1028             # because this is created before the placeholder naming pass
1029             if isinstance(node.meta["val"], CustomObjArgument):
1030                 node.meta["val"].name = node.name
1031         elif node.name in name_map:
1032             node.name = name_map[node.name]
1033 
1034     # propagate names to higher order op subgraphs
1035     _name_hoo_subgraph_placeholders(gm)
1036 
1037     # re-generate graph module code
1038     gm.recompile()
1039 
1040     # modify graph signature (input specs, output specs, user input mutations)
1041     for spec in export_graph_signature.input_specs:
1042         assert spec.arg.name in name_map
1043         spec.arg.name = name_map[spec.arg.name]
1044         if (  # handle targets for custom objects
1045             spec.kind == InputKind.CUSTOM_OBJ and spec.target in name_map
1046         ):
1047             spec.target = name_map[spec.target][4:]  # strip obj_ prefix
1048 
1049     for spec in export_graph_signature.output_specs:
1050         if spec.arg.name in name_map:
1051             spec.arg.name = name_map[spec.arg.name]
1052         if spec.kind == OutputKind.USER_INPUT_MUTATION and spec.target in name_map:
1053             spec.target = name_map[spec.target]
1054 
1055     # rename keys in constants dict for custom objects
1056     for name in list(constants.keys()):
1057         constant = constants[name]
1058         if name in name_map and not isinstance(
1059             constant, torch.Tensor
1060         ):  # rename custom objects with generic names
1061             new_name = name_map[name]
1062             if (
1063                 new_name != name
1064                 and re.match(r"arg(\d+)_1", name)
1065                 and new_name != placeholder_prefixes[InputKind.CUSTOM_OBJ] + name
1066             ):
1067                 constants[new_name] = constant
1068                 del constants[name]

1071 def remove_proxy_from_state_dict(state_dict: dict, in_place: bool) -> dict:
1072     """ 
1073     If `in_place` is false, return a new copy of `state_dict` with "proxy" removed from `v.__dict__`. 
1074     `v` is the values in the dictionary. 
1075     If `in_place` is true, modify `state_dict` in place. 
1076     """
1077     if in_place:
1078         for k, v in state_dict.items():
1079             if hasattr(v, "proxy"):
1080                 delattr(state_dict[k], "proxy")
1081         return state_dict
1082     else:
1083         new_state_dict = {}
1084         for k, v in state_dict.items():
1085             if hasattr(v, "proxy"):
1086                 new_state_dict[k] = v.detach().clone()
1087             else:
1088                 new_state_dict[k] = v
1089         return new_state_dict

1092 def _detect_fake_mode_from_gm(
1093     gm: torch.fx.GraphModule,
1094 ) -> torch._subclasses.fake_tensor.FakeTensorMode:
1095     """ 
1096     For a given graph module, we look at the "val" of placeholder nodes to find the fake inputs. 
1097     Additionally, if gm doesn't have placeholders, we further look at the "example_value" or "val" of other nodes. 
1098     If no fake mode is found, we return None for fake_mode. 
1099     """
1100 
1101     fake_inps: list[torch.Tensor] = []
1102     fake_vals: list[torch.Tensor] = []
1103     for node in gm.graph.nodes:
1104         if node.op == "placeholder" and "val" in node.meta:
1105             fake_val = node.meta["val"]
1106             if fake_val is not None and isinstance(fake_val, torch.Tensor):

1109             "example_value" in node.meta or "val" in node.meta
1110         ):
1111             fake_val = None
1112             if "example_value" in node.meta:
1113                 fake_val = node.meta["example_value"]
1114             elif "val" in node.meta:
1115                 fake_val = node.meta["val"]
1116             if fake_val is not None and isinstance(fake_val, torch.Tensor):
1117                 fake_vals.append(fake_val)
1118 
1119     return detect_fake_mode(fake_inps + fake_vals)

1181 def _special_op_to_preserve_cia(*args, **kwargs):
1182     """ 
1183     This is an special marker that tells our infra that we shouldn't decompose this op. 
1184     """
1185     return NotImplemented

1188 # Our strategy for deciding if we can preserve a op is following:
1189 # 1. The op should be known statically that it is functional
1190 # 2. If it is maybe aliasing, we decompose because we must know if an op
1191 #    is mutating or aliasing.
1192 def _check_valid_to_preserve(op_overload: "OperatorBase"):
1193     from torch._decomp import _should_decompose_because_unsafe_op
1194 
1195     if _should_decompose_because_unsafe_op(op_overload):
1196         return False
1197     if op_overload in FunctionalTensor.metadata_fns:
1198         return False
1199 
1200     if not hasattr(op_overload, "_schema"):
1201         return False
1202 
1203     alias_info = len(
1204         [i for i in op_overload._schema.arguments if i.alias_info is not None]
1205     )
1206 
1207     is_mutating_or_aliasing = alias_info != 0 or op_overload._schema.is_mutable
1208 
1209     if is_mutating_or_aliasing:
1210         return False

1240 def _collect_all_valid_cia_ops() -> set["OperatorBase"]:
1241     """ 
1242     This is an util function that gets the all CIA functional ops. 
1243  
1244     The algorithm is in 2 steps: 
1245       1. We first query C++ dispatcher to get the list of CIA ops 
1246          and then we call getattr on torch.ops.aten to lazily populate 
1247          them. 
1248  
1249       2. Sometimes, handful of ops have CIA registered in python dispatcher 
1250          but not on the C++ side, these can't be caught at the first step. 
1251          So we walk again to get the final list. 
1252  
1253     Note that the output of this function should never be modified 
1254     """
1255     cia_ops = set()
1256     for op_namespace_name in torch.ops._dir:
1257         # The reason we split here is because aten ops are safe to cache.
1258         if op_namespace_name != "aten":
1259             assert hasattr(torch.ops, op_namespace_name)
1260             op_namespace = getattr(torch.ops, op_namespace_name)
1261             if isinstance(op_namespace, torch._ops._OpNamespace):
1262                 cia_ops |= _collect_all_valid_cia_ops_for_namespace(op_namespace)
1263         else:
1264             cia_ops |= _collect_all_valid_cia_ops_for_aten_namespace()
1265     return cia_ops

1268 def _get_decomp_for_cia(op: "OperatorBase"):
1269     # [NOTE] Seperating out func.decompose
1270     # Ideally we should be able to just register func.decompose but
1271     # we can't as this decomp is gonna be registered to the py_impl.
1272     # As a result it will infinitely recurse. So we first check if the op
1273     # has py_impl entry for CIA and if it is we use that first. If not,
1274     # we register C++ query to py_impl.
1275     dk = torch._C.DispatchKey.CompositeImplicitAutograd
1276     if dk in op.py_kernels and not isinstance(op.py_kernels[dk], torch._C.DispatchKey):
1277         return op.py_kernels[dk]
1278 
1279     def _special_op_to_decompose_cia(*args, **kwargs):
1280         kernel = kwargs["kernel"]
1281         del kwargs["kernel"]
1282         # Can't call kernel.decompose due to infinite recursion as
1283         # we register this kernel to py_impl directly
1284         dk = torch._C.DispatchKey.CompositeImplicitAutograd
1285         if torch._C._dispatch_has_kernel_for_dispatch_key(
1286             kernel.name(), torch._C.DispatchKey.CompositeImplicitAutograd
1287         ):
1288             return kernel._op_dk(dk, *args, **kwargs)
1289         else:
1290             raise AssertionError(
1291                 f"Expected {kernel} to have CompositeImplicitAutograd kernel"
1292             )
1293 
1294     return functools.partial(_special_op_to_decompose_cia, kernel=op)

1331 def register_module_as_pytree_input_node(cls: type[torch.nn.Module]) -> None:
1332     """ 
1333     Registers a module as a valid input type for :func:`torch.export.export`. 
1334  
1335     Args: 
1336         mod: the module instance 
1337         serialized_type_name: The serialized name for the module. This is 
1338         required if you want to serialize the pytree TreeSpec containing this 
1339         module. 
1340  
1341     Example:: 
1342  
1343         import torch 
1344  
1345         class Module(torch.nn.Module): 
1346             def __init__(self): 
1347                 super().__init__() 
1348                 self.linear = torch.nn.Linear(3, 3) 
1349  
1350             def forward(self, x): 
1351                 return self.linear(x) 
1352  
1353         torch._export.utils.register_module_as_pytree_node(InputDataClass) 
1354  
1355         class Mod(torch.nn.Module): 
1356             def forward(self, x, m): 
1357                 return m(x) + x 
1358  
1359         ep = torch.export.export(Mod(), (torch.randn(3), Module())) 
1360         print(ep) 
1361  
1362     """
1363     assert issubclass(cls, torch.nn.Module)
1364 
1365     import weakref
1366 
1367     class PrototypeModule(weakref.ref):
1368         def __init__(self, m, *args, **kwargs):
1369             super().__init__(m, *args, **kwargs)  # type: ignore[call-arg]
1370             assert isinstance(m, torch.nn.Module)
1371             assert not hasattr(self, "_proto_cls")
1372             self._proto_cls = cls
1373 
1374         def __eq__(self, other):
1375             return self._proto_cls == other._proto_cls
1376 
1377         def __deepcopy__(self, memo):
1378             return PrototypeModule(self())
1379 
1380     def default_flatten_fn(obj: Any) -> tuple[list[Any], Context]:
1381         named_parameters = dict(obj.named_parameters())
1382         named_buffers = dict(obj.named_buffers())
1383         params_buffers = {**named_parameters, **named_buffers}
1384         return list(params_buffers.values()), [
1385             list(params_buffers.keys()),
1386             PrototypeModule(obj),
1387         ]
1388 
1389     def default_unflatten_fn(values: Iterable[Any], context: Context) -> Any:
1390         flat_names, ref = context
1391         if ref is None or ref() is None:
1392             raise RuntimeError("Module has been garbage collected")
1393         obj = ref()
1394         assert flatten_fn is not None
1395         flattened, _ = flatten_fn(obj)
1396 
1397         # NOTE: This helper function will replicate an nn.Module in the exactly same
1398         #       structure to be used together with _reparametrize_module. This will
1399         #       create a clone of the module with the new parameters and buffers without
1400         #       affecting the original module.
1401         def copy_module(mod: torch.nn.Module):
1402             ret = copy.copy(mod)
1403             ret.__dict__ = {copy.copy(k): copy.copy(v) for k, v in mod.__dict__.items()}
1404             for name, child in ret.named_children():
1405                 setattr(ret, name, copy_module(child))
1406             return ret
1407 
1408         if any(v is not o for v, o in zip(values, flattened)):
1409             with torch.nn.utils.stateless._reparametrize_module(
1410                 obj, dict(zip(flat_names, values)), tie_weights=True, strict=True
1411             ):
1412                 ret = copy_module(obj)
1413         else:
1414             ret = obj
1415         return ret
1416 
1417     def default_flatten_fn_with_keys(obj: Any) -> tuple[list[Any], Context]:
1418         flattened, [flat_names, *args] = flatten_fn(obj)  # type: ignore[misc]
1419         return [(MappingKey(k), v) for k, v in zip(flat_names, flattened)], [
1420             flat_names,
1421             *args,
1422         ]
1423 
1424     flatten_fn = default_flatten_fn
1425     unflatten_fn = default_unflatten_fn
1426 
1427     serialized_type_name = cls.__module__ + "." + cls.__qualname__
1428 
1429     def to_dumpable_context(context):
1430         keys, *_ = context
1431         return json.dumps([keys, *([None] * len(_))])
1432 
1433     def from_dumpable_context(dumpable):
1434         s = json.loads(dumpable)
1435         s[1] = PrototypeModule(torch.nn.Module())
1436         return s
1437 
1438     _register_pytree_node(
1439         cls,
1440         flatten_fn,
1441         unflatten_fn,
1442         serialized_type_name=serialized_type_name,
1443         flatten_with_keys_fn=default_flatten_fn_with_keys,
1444         to_dumpable_context=to_dumpable_context,
1445         from_dumpable_context=from_dumpable_context,
1446     )
1447 
1448     def default_flatten_fn_spec(obj, spec) -> list[Any]:
1449         flats, context = flatten_fn(obj)
1450         assert context == spec.context
1451         return flats
1452 
1453     register_pytree_flatten_spec(
1454         cls,
1455         default_flatten_fn_spec,
1456     )

1464 def _sync_state(src, dst):
1465     assert isinstance(
1466         src,
1467         torch.nn.Module,
1468     ), f"Expected {src} to be a nn.Module"
1469     assert isinstance(
1470         dst,
1471         torch.nn.Module,
1472     ), f"Expected {dst} to be a nn.Module"
1473     # Share state (params, buffers) between modules.
1474     # This ensures that state mutations are visible across them.
1475     # Since tensor constants are not mutable, copying (without sharing) is OK.
1476     # Also, primitive constants are specialized, so copying (without sharing) is OK.
1477     dst._parameters = src._parameters
1478     dst._buffers = src._buffers

1481 def sync_state(*wrapped_method_modules):
1482     """ 
1483     Sync state between exported modules corresponding to wrapped methods. 
1484     This might be necessary after serializing/deserializing due to copying. 
1485     """
1486     if wrapped_method_modules:
1487         m, *other_ms = wrapped_method_modules
1488         for other_m in other_ms:
1489             _sync_state(m, other_m)
