22   def _cdiv(
23       x: Union[int, float, torch.Tensor], multiple: Union[int, float, torch.Tensor]
24   ):
25       return (x + multiple - 1) // multiple

28   class PagedAttention:
29       """ 
30       PagedAttention supports flex attention inference with a large batch size. 
31       With PagedAttention, a batch of key/value tensors with varying kv length 
32       is splitted into tensor blocks of fixed length and cached in a compact way. 
33       Thus we can avoid redundant memory consumption due to varying kv length and 
34       support a larger batch size. 
35       """
36   
37       def __init__(
38           self,
39           n_pages: int,
40           page_size: int,
41           max_batch_size: int,
42           device: str = "cuda",
43       ):
44           # number of pages
45           self.n_pages = n_pages
46   
47           # number of tokens per page
48           self.page_size = page_size
49   
50           # page table: [batch, logical_block_idx] -> physical_page_idx
51           self.page_table = -torch.ones(
52               (max_batch_size, self.n_pages), dtype=torch.int64, device=device
53           )
54   
55           # capacity: batch_idx -> allocated sequence length
56           self.capacity = torch.zeros(max_batch_size, dtype=torch.int64, device=device)
57   
58           # index of empty pages that is available for allocation
59           self.empty_pages = list(range(n_pages - 1, -1, -1))
60   
61           # mapping from physical page index to logical page index
62           self.physical_to_logical = -torch.ones(
63               (max_batch_size, n_pages), dtype=torch.int64, device=device
64           )
65   
66       def reserve(self, batch_idx: torch.Tensor, seq_len: torch.Tensor) -> None:
67           """ 
68           Requests the capacity of a given batch to be at least enough to 
69           hold `seq_len` elements. 
70    
71           Args: 
72               batch_idx (Tensor): batch index to be reserved; shape :math:`(1)`. 
73               seq_len (Tensor): minimum capacity for the given batch; shape :math:`(1)`. 
74           """
75   
76           if seq_len <= self.capacity[batch_idx]:
77               return
78   
79           num_pages_to_allocate = _cdiv(
80               seq_len - self.capacity[batch_idx], self.page_size
81           )
82   
83           assert len(self.empty_pages) >= num_pages_to_allocate, (
84               f"requested {num_pages_to_allocate.item()} pages "
85               f"but there are only {len(self.empty_pages)} empty pages"
86           )
87   
88           start_page_idx = self.capacity[batch_idx] // self.page_size
89           end_page_idx = start_page_idx + num_pages_to_allocate
90   
91           # find empty physical pages
92           allocated_pages = torch.tensor(
93               self.empty_pages[-num_pages_to_allocate:],
94               device=num_pages_to_allocate.device,
95           )
96           self.empty_pages = self.empty_pages[:-num_pages_to_allocate]
97   
98           # update page table
99           self.page_table[
100              batch_idx,
101              start_page_idx:end_page_idx,
102          ] = allocated_pages
103  
104          # update metadata
105          self.physical_to_logical[batch_idx, allocated_pages] = torch.arange(
106              start_page_idx.item(),
107              end_page_idx.item(),
108              device=num_pages_to_allocate.device,
109          )
110          self.capacity[batch_idx] += num_pages_to_allocate * self.page_size
111  
112      def erase(self, batch_idx: torch.Tensor) -> None:
113          """ 
114          Removes a single batch from paged attention. 
115   
116          Args: 
117              batch_idx (Tensor): batch index to be removed; shape :math:`(1)`. 
118          """
119  
120          # find allocated pages
121          allocated_page_idx = self.page_table[batch_idx] != -1
122          allocated_pages = self.page_table[batch_idx][allocated_page_idx]
123  
124          # clean metadata
125          self.capacity[batch_idx] = 0
126          self.empty_pages += allocated_pages.tolist()
127          self.physical_to_logical[batch_idx][:, allocated_pages] = -1
128          self.page_table[batch_idx] = -1

130      def assign(
131          self,
132          batch_idx: torch.Tensor,
133          input_pos: torch.Tensor,
134          k_val: torch.Tensor,
135          v_val: torch.Tensor,
136          k_cache: torch.Tensor,
137          v_cache: torch.Tensor,
138      ) -> None:
139          """ 
140          Assigns new contents `val` to the storage `cache` at the location 
141          `batch_idx` and `input_pos`. 
142   
143          Args: 
144              batch_idx (Tensor): batch index; shape :math:`(B)`. 
145              input_pos (Tensor): input positions to be assigned for the given batch; shape :math:`(B, S)`. 
146              val (Tensor): value to be assigned; shape :math:`(B, H, S, D)` 
147              cache (Tensor): the cache to store the values; shape:`(1, H, MAX_S, D)` 
148          """
149          if k_val.requires_grad:
150              raise RuntimeError("val must not require gradient")
151  
152          B, H, S, K_D = k_val.shape
153          V_D = v_val.shape[3]
154          if B != batch_idx.shape[0]:
155              raise RuntimeError(
156                  f"Expect val and batch_idx have the same batch size "
157                  f"but got B={B} and B={batch_idx.shape[0]}."
158              )
159          if H != k_cache.shape[1]:
160              raise RuntimeError(
161                  f"Expect val and cache has the same number of heads "
162                  f"but got H={H} and H={k_cache.shape[1]}."
163              )
164          if S != input_pos.shape[1]:
165              raise RuntimeError(
166                  f"Expect val and input_pos has the same length "
167                  f"but got S={S} and S={input_pos.shape[0]}."
168              )
169          if K_D != k_cache.shape[3]:
170              raise RuntimeError(
171                  f"Expect k_val and k_cache has the same hidden dim "
172                  f"but got D={K_D} and D={k_cache.shape[3]}."
173              )
174          if V_D != v_cache.shape[3]:
175              raise RuntimeError(
176                  f"Expect v_val and v_cache has the same hidden dim "
177                  f"but got D={V_D} and D={v_cache.shape[3]}."
178              )
179  
180          # find address
181          logical_block_idx = input_pos // self.page_size  # [B, S]
182          logical_block_offset = input_pos % self.page_size  # [B, S]
183          physical_block_idx = torch.gather(
184              self.page_table[batch_idx], 1, logical_block_idx.to(torch.int64)
185          ).to(torch.int32)  # [B, S]
186  
187          addr = (physical_block_idx * self.page_size + logical_block_offset).view(
188              -1
189          )  # [B*S]
190  
191          k_val = k_val.permute(1, 0, 2, 3).contiguous().view(1, H, B * S, K_D)
192          v_val = v_val.permute(1, 0, 2, 3).contiguous().view(1, H, B * S, V_D)
193  
194          k_cache[:, :, addr, :] = k_val
195          v_cache[:, :, addr, :] = v_val

197      def convert_logical_block_mask(
198          self,
199          block_mask: BlockMask,
200          batch_idx: Optional[torch.Tensor] = None,
201      ) -> BlockMask:
202          """ 
203          Converts a logical block mask by mapping its logical kv indices to the corresponding 
204          physical kv indices. 
205   
206          Args: 
207              block_mask (BlockMask): logical block mask; 
208                  kv_indices shape :math:`(B, H, ROWS, MAX_BLOCKS_IN_COL)`. 
209              batch_idx (Tensor): batch index corresponding to the block_mask 
210                  batch dimension. This provides flexibility to convert a 
211                  block mask with smaller batch size than the page table; 
212                  shape :math:`(B)`. 
213          """
214          B, H, ROWS, MAX_BLOCKS_IN_COL = block_mask.kv_indices.shape
215  
216          if block_mask.BLOCK_SIZE[1] != self.page_size:
217              raise RuntimeError(
218                  f"Expect block_mask has the same column block size as page_size"
219                  f"but got size={block_mask.BLOCK_SIZE[1]} and size={self.page_size}"
220              )
221  
222          # Increase the num columns of converted block mask from logical block mask's
223          # num columns to n_pages, since a) the converted block mask
224          # may have larger indices values; and b) `_ordered_to_dense` realizes
225          # a dense tensor with these converted indices. There would be an IndexError
226          # if using the logical block mask's num columns.
227  
228          device = block_mask.kv_num_blocks.device
229  
230          if batch_idx is None:
231              batch_idx = torch.arange(B, device=device)
232          page_table = self.page_table[batch_idx]
233  
234          new_kv_num_blocks = block_mask.kv_num_blocks.clone()
235  
236          new_kv_indices = torch.zeros(
237              (B, H, ROWS, self.n_pages), dtype=torch.int32, device=device
238          )
239          new_kv_indices[:, :, :, :MAX_BLOCKS_IN_COL] = (
240              torch.gather(
241                  page_table, 1, block_mask.kv_indices.view(B, -1).to(torch.int64)
242              )
243              .view(block_mask.kv_indices.shape)
244              .to(torch.int32)
245          )
246  
247          new_full_kv_indices, new_full_kv_num_blocks = None, None
248          if block_mask.full_kv_num_blocks is not None:
249              assert block_mask.full_kv_indices is not None
250              new_full_kv_num_blocks = block_mask.full_kv_num_blocks.clone()
251              new_full_kv_indices = torch.zeros(
252                  (B, H, ROWS, self.n_pages), dtype=torch.int32, device=device
253              )
254              new_full_kv_indices[:, :, :, :MAX_BLOCKS_IN_COL] = (
255                  torch.gather(
256                      page_table,
257                      1,
258                      block_mask.full_kv_indices.view(B, -1).to(torch.int64),
259                  )
260                  .view(block_mask.full_kv_indices.shape)
261                  .to(torch.int32)
262              )
263  
264          new_mask_mod = self.get_mask_mod(block_mask.mask_mod)
265  
266          seq_lengths = (block_mask.seq_lengths[0], self.n_pages * self.page_size)
267          return BlockMask.from_kv_blocks(
268              new_kv_num_blocks,
269              new_kv_indices,
270              new_full_kv_num_blocks,
271              new_full_kv_indices,
272              block_mask.BLOCK_SIZE,
273              new_mask_mod,
274              seq_lengths=seq_lengths,
275          )

277      def get_mask_mod(
278          self, mask_mod: Optional[_mask_mod_signature]
279      ) -> _mask_mod_signature:
280          """ 
281          Converts a mask_mod based on mapping from the physical block index to the logical 
282          block index. 
283   
284          Args: 
285              mask_mod (_mask_mod_signature): mask_mod based on the logical block index. 
286          """
287          if mask_mod is None:
288              mask_mod = noop_mask
289  
290          def new_mask_mod(
291              b: torch.Tensor,
292              h: torch.Tensor,
293              q_idx: torch.Tensor,
294              physical_kv_idx: torch.Tensor,
295          ):
296              physical_kv_block = physical_kv_idx // self.page_size
297              physical_kv_offset = physical_kv_idx % self.page_size
298              logical_block_idx = self.physical_to_logical[b, physical_kv_block]
299              logical_kv_idx = logical_block_idx * self.page_size + physical_kv_offset
300              return torch.where(
301                  logical_block_idx >= 0, mask_mod(b, h, q_idx, logical_kv_idx), False
302              )
303  
304          return new_mask_mod
305  
306      def get_score_mod(
307          self, score_mod: Optional[_score_mod_signature]
308      ) -> _score_mod_signature:
309          """ 
310          Converts a score_mod based on mapping from the physical block index to the logical 
311          block index. 
312   
313          Args: 
314              score_mod (_score_mod_signature): score_mod based on the logical block index. 
315          """
316          if score_mod is None:
317              score_mod = _identity
318  
319          def new_score_mod(
320              score: torch.Tensor,
321              b: torch.Tensor,
322              h: torch.Tensor,
323              q_idx: torch.Tensor,
324              physical_kv_idx: torch.Tensor,
325          ):
326              physical_kv_block = physical_kv_idx // self.page_size
327              physical_kv_offset = physical_kv_idx % self.page_size
328              logical_block_idx = self.physical_to_logical[b, physical_kv_block]
329              logical_kv_idx = logical_block_idx * self.page_size + physical_kv_offset
330              return torch.where(
331                  logical_block_idx >= 0,
332                  score_mod(score, b, h, q_idx, logical_kv_idx),
333                  float("-inf"),
334              )
335  
336          return new_score_mod
