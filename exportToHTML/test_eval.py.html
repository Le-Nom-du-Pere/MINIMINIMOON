11   def test_add_eval():
12       a = Symbol("a")
13       b = Symbol("b")
14       c = Rational(1)
15       p = Rational(5)
16       assert a*b + c + p == a*b + 6
17       assert c + a + p == a + 6
18       assert c + a - p == a + (-4)
19       assert a + a == 2*a
20       assert a + p + a == 2*a + 5
21       assert c + p == Rational(6)
22       assert b + a - b == a

36   def test_pow_eval():
37       # XXX Pow does not fully support conversion of negative numbers

39   
40       assert sqrt(-1) == I
41   
42       assert sqrt(-4) == 2*I

45       assert (-8)**Rational(1, 3) == 2*((-1)**Rational(1, 3))
