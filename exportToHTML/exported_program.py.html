128  def _fx_collection_equivalence_fn(
129      spec1_type: Optional[type],
130      spec1_context: pytree.Context,
131      spec2_type: Optional[type],
132      spec2_context: pytree.Context,
133  ) -> bool:
134      """Treat containers and their immutable variants as the same type. Otherwise 
135      compare as normal. 
136      """
137      if spec1_type is None or spec2_type is None:
138          return spec1_type is spec2_type and spec1_context == spec2_context
139  
140      if issubclass(spec1_type, (dict, immutable_dict)) and issubclass(
141          spec2_type, (dict, immutable_dict)
142      ):
143          return spec1_context == spec2_context
144  
145      if issubclass(spec1_type, (list, immutable_list)) and issubclass(
146          spec2_type, (list, immutable_list)
147      ):
148          return spec1_context == spec2_context
149  
150      return spec1_type is spec2_type and spec1_context == spec2_context

190  @contextmanager
191  def _override_composite_implicit_decomp(cia_ops_to_callable):
192      # This function overrides CompositeImplicitAutograd decomp for
193      # functional composite ops that user specified. Ideally we want to not-decompose
194      # ALL composite ops but today's C++ functinalization relies on
195      # the fact that it is working with the opset after decomp is run.
196      # Hence we can only do it for functional ops. One caveat is that
197      # there are some composite ops that lie about their schema (claimed to be
198      # functional but not really aka dropout), for these cases, we just decompose.
199      saved_tables = {}
200      patched_ops = set()
201      for op_overload, decomp_callable in cia_ops_to_callable.items():
202          saved_tables[op_overload] = op_overload.py_kernels.copy()
203          patched_ops.add(op_overload)
204          for override_dispatch_key in _AUTOGRAD_ALIAS_BACKEND_KEYS_TO_OVERRIDE:
205              if override_dispatch_key not in op_overload.py_kernels:
206                  # TODO (tmanlaibaatar)https://github.com/pytorch/pytorch/issues/129430
207                  op_overload.py_impl(override_dispatch_key)(
208                      autograd_not_implemented(op_overload, deferred_error=True)
209                  )
210          # See NOTE: Registering old CIA to Backend kernel
211          # It is important that we cache this before we override py_kernels.
212          orig_cia_callable = _get_decomp_for_cia(op_overload)
213          if torch._C.DispatchKey.CompositeImplicitAutograd in op_overload.py_kernels:
214              del op_overload.py_kernels[torch._C.DispatchKey.CompositeImplicitAutograd]
215  
216          op_overload.py_impl(torch._C.DispatchKey.CompositeImplicitAutograd)(
217              decomp_callable
218          )
219  
220          # [NOTE] Directly registering fake tensor rule to CIA ops
221          # The problem we are facing here is if your CIA custom rule
222          # says we want to preserve the op, we will return NotImplemented.
223          # Unfortunately, this will invoke meta device tracing in fake tensor
224          # resulting in divergent behaviour for CIA kernels that has device based
225          # branching (one case is torch.ops.aten.scaled_dot_product.attention)
226          # To get around this issue, we register direct fake impl so that we
227          # run the kernel before we actually try to decompose the op in FakeTensorMode.
228          # Note that is a no-op in most cases, because:
229          #   1) In post dispatch tracing, CIA would have already decomposed
230          #   2) Most CIA impl are device agnostic.
231          def _force_dispatch_to_orig_cia_callable(fake_tensor_mode, op, *args, **kwargs):
232              orig_cia_callable = kwargs["original_callable"]
233              del kwargs["original_callable"]
234              with fake_tensor_mode:
235                  return orig_cia_callable(*args, **kwargs)
236  
237          if not _is_op_registered_to_fake_rule(op_overload):
238              register_op_impl(op_overload)(
239                  functools.partial(
240                      _force_dispatch_to_orig_cia_callable,
241                      original_callable=orig_cia_callable,
242                  )
243              )
244  
245          for key in _BACKEND_KEYS_TO_OVERRIDE:
246              if key not in op_overload.py_kernels:
247                  # [NOTE] Registering old CIA to Backend kernel
248                  # We always register original CIA behavior to the backend keys kernel
249                  # The reason is when we are fake tensor prop-ing or executing real kernel,
250                  # we end up calling an operator on respective backend, which in python dispatcher,
251                  # will resolve into CIA key. (see resolve_key in torch/_ops.py)
252                  # As a result, this CIA now will call into the custom user defined
253                  # CIA which can cause a problem.
254                  # To make it more concrete, the case we are handling is:
255                  #  (1) there is a tensor constant we are performing constant propagation
256                  #      on during tracing
257                  #  (2) we invoke an op underneath autograd (either because we are below autograd,
258                  #      or we are tracing in inference mode), so one of the backend keys gets hit
259                  #  (3) the op we are invoking has a CIA impl that normally runs in eager mode
260                  #      (and the user wants to tweak this CIA impl during tracing, but during
261                  #      const-prop we want the original CIA to run
262                  op_overload.py_impl(key)(orig_cia_callable)
263  
264      try:
265          yield
266      finally:
267          for op in patched_ops:
268              op.py_kernels.clear()
269              op.py_kernels.update(saved_tables[op])
270              op._dispatch_cache.clear()
271              _deregister_op_impl(op)

274  def _split_decomp_table_to_cia_and_python_decomp(
275      decomp_table: dict[torch._ops.OperatorBase, Callable],
276  ) -> tuple[dict[torch._ops.OperatorBase, Callable], ...]:
277      all_preservable_cia_ops = set(_collect_all_valid_cia_ops())
278      cia_ops_to_callable = {}
279  
280      for op in list(decomp_table.keys()):
281          # TODO we are silently allowing non-safe(non-functional) ops through a crack
282          # due to core aten decomp table having non-functional entries. Once we have
283          # a tigher check around core aten decomp, we should warn users about them.
284          # Tracking issue: (https://github.com/pytorch/pytorch/issues/135759)
285  
286          # if it is a valid CIA op we can mess with in export, we check if it is:
287          #  1. Has been marked as to be decomposed. Example:
288          #        decomp_table = decomp_table_to_core_aten()
289          #        del decomp_table[aten.linear]
290          #     In this case, user says decompose everything except for aten.linear
291          #  2. Has been marked with custom decomp behavour. Example:
292          #        decomp_table = {aten.linear: some_op}
293          # For (1), we want to remove all the CIA ops that weren't handled by user as
294          # it suggests they are safe to decompose, so we should remove from preservable_list.
295          # for (2), we just plumb the custom decomp to AOTDIspatcher.
296          # In both cases, we want to remove this CIA op from the decomp_table as it is special
297          # handled.
298          if op in all_preservable_cia_ops:
299              cia_ops_to_callable[op] = decomp_table[op]
300              all_preservable_cia_ops.remove(op)
301              del decomp_table[op]
302          # If it is a custom op, we want to still preserve or do whatever
303          # with it if it is a functional CIA. The reason we don't remove
304          # from CIA list is because we don't query custom ops.
305          elif _is_preservable_cia_op(op):
306              op_name = op.name()
307              assert not op_name.startswith("aten"), "This should be a custom op"
308              cia_ops_to_callable[op] = decomp_table[op]
309  
310      # If we reached here, it means user intentionally deleted these CIA ops from
311      # decomp table.
312      for k in all_preservable_cia_ops:
313          cia_ops_to_callable[k] = _special_op_to_preserve_cia
314  
315      return cia_ops_to_callable, decomp_table

327  def _decompose_and_get_gm_with_new_signature_constants(
328      ep,
329      *,
330      cia_to_decomp: dict[torch._ops.OperatorBase, Callable],
331      python_decomp_table: dict[torch._ops.OperatorBase, Callable],
332      joint_loss_index: Optional[int],
333      decompose_custom_triton_ops,
334  ):
335      from torch._export.passes.lift_constants_pass import _materialize_and_lift_constants
336      from torch._functorch.aot_autograd import aot_export_module
337      from torch.export._trace import (
338          _disable_custom_triton_op_functional_decomposition,
339          _export_to_aten_ir,
340          _ignore_backend_decomps,
341          _verify_nn_module_stack,
342          _verify_placeholder_names,
343          _verify_stack_trace,
344      )
345      from torch.fx.experimental.symbolic_shapes import ShapeEnv
346  
347      def _is_joint_ir_decomp(ep, joint_loss_index):
348          return (
349              joint_loss_index is not None
350              or ep.graph_signature.backward_signature is not None
351          )
352  
353      if not _is_joint_ir_decomp(ep, joint_loss_index):
354          mod = ep.module()
355  
356          wrapped_params_buffers = {
357              **dict(mod.named_parameters(remove_duplicate=False)),
358              **dict(mod.named_buffers(remove_duplicate=False)),
359          }
360  
361          from torch._functorch._aot_autograd.subclass_parametrization import (
362              unwrap_tensor_subclass_parameters,
363          )
364  
365          # [NOTE] Unwrapping subclasses AOT
366          # In torch.compile, the subclass unwrapping/wrapping happen at runtime
367          # but at export, this is impossible as it is intented to be run on
368          # C++ environment. As a result, we unwrap subclass parameters AOT. After this,
369          # ExportedProgram state_dict won't be same as eager model because eager model
370          # could have subclass weights while ExportedProgram will have desugared versions.
371          # This is fine because run_decompositions is supposed to specialize to post-autograd
372          # graph where the subclass desugaring is supposed to happen.
373          unwrap_tensor_subclass_parameters(mod)
374          unwrapped_params_buffers = {
375              **dict(mod.named_parameters(remove_duplicate=False)),
376              **dict(mod.named_buffers(remove_duplicate=False)),
377          }
378  
379          # TODO T204030333
380          fake_mode = _detect_fake_mode_from_gm(ep.graph_module)
381          if fake_mode is None:
382              fake_mode = FakeTensorMode(shape_env=ShapeEnv(), export=True)
383  
384          # Fix the graph output signature to be tuple if scalar
385          out_spec = mod._out_spec
386  
387          orig_arg_names = mod.graph._codegen.pytree_info.orig_args
388  
389          # aot_export expect the return type to always be a tuple.
390          if out_spec.type not in (list, tuple):
391              out_spec = pytree.TreeSpec(tuple, None, [out_spec])
392  
393          mod.graph._codegen = _PyTreeCodeGen(
394              _PyTreeInfo(
395                  orig_arg_names,
396                  mod._in_spec,
397                  out_spec,
398              )
399          )
400  
401          mod.recompile()
402  
403          # the exported module will store constants & non-persistent buffers such that
404          # retracing treats them as persistent buffers, so we inform the constants lifting pass
405          # and overwrite the new graph signature using the previous program.
406          _collect_and_set_constant_attrs(ep.graph_signature, ep.constants, mod)
407  
408          # When we have a module with constant attributes, AotDispatcher doesn't actually
409          # wrap them as functional tensors, because dynamo would have already made it buffer.
410          # In non-strict case, however, AotDispatcher can intercept constants, causing it to not
411          # functionalize the operators that are operating on constant tensors. Since dynamo already
412          # wraps constants as buffers, we temporarily register the constants as buffers and undo this
413          # operation after AOTDispatcher is done.
414          temp_registered_constants = _register_constants_as_buffers(
415              mod, ep.state_dict, ep.graph_signature.non_persistent_buffers
416          )
417  
418          # get params & buffers after excluding constants
419          fake_params_buffers = _fakify_params_buffers(fake_mode, mod)
420  
421          params_buffers_to_node_meta = _collect_param_buffer_metadata(mod)
422  
423          # TODO (tmanlaibaatar) Ideally run_decomp should just call _non_strict_export
424          # but due to special handling of constants as non-persistent buffers make it little
425          # diffucult. But we should unify this code path together. T206837815
426          from torch._export.non_strict_utils import (
427              _enable_graph_inputs_of_type_nn_module,
428              _fakify_script_objects,
429          )
430  
431          retracing_args = []
432          for node in mod.graph.nodes:
433              if node.op == "placeholder":
434                  if isinstance(node.meta["val"], CustomObjArgument):
435                      real_script_obj = None
436                      if node.meta["val"].fake_val is None:
437                          real_script_obj = ep.constants[node.meta["val"].name]
438                      else:
439                          real_script_obj = node.meta["val"].fake_val.real_obj
440                      retracing_args.append(real_script_obj)
441                  else:
442                      retracing_args.append(node.meta["val"])
443  
444          tx = TracingContext(fake_mode)
445  
446          with (
447              fake_mode,
448              _override_composite_implicit_decomp(
449                  cia_to_decomp,
450              ),
451              _enable_graph_inputs_of_type_nn_module(ep.example_inputs),
452              tracing(tx),
453          ):
454              retracing_args_unwrapped = pytree.tree_unflatten(
455                  retracing_args, mod._in_spec
456              )
457              # this requires empty kwargs, but not in pytree.flattened format
458              with _fakify_script_objects(
459                  mod,
460                  (
461                      *retracing_args_unwrapped[0],
462                      *retracing_args_unwrapped[1].values(),
463                  ),
464                  {},
465                  fake_mode,
466              ) as (
467                  patched_mod,
468                  new_fake_args,
469                  new_fake_kwargs,
470                  new_fake_constant_attrs,
471                  map_fake_to_real,
472              ):
473                  aten_export_artifact = _export_to_aten_ir(
474                      patched_mod,
475                      new_fake_args,
476                      new_fake_kwargs,
477                      fake_params_buffers,
478                      new_fake_constant_attrs,
479                      decomp_table=python_decomp_table,
480                      _check_autograd_state=False,
481                      _prettify_placeholder_names=False,
482                      decompose_custom_triton_ops=decompose_custom_triton_ops,
483                  )
484  
485                  # aten_export_artifact.constants contains only fake script objects, we need to map them back
486                  aten_export_artifact.constants = {
487                      fqn: (
488                          map_fake_to_real[obj]
489                          if isinstance(obj, FakeScriptObject)
490                          else obj
491                      )
492                      for fqn, obj in aten_export_artifact.constants.items()
493                  }
494  
495                  gm = aten_export_artifact.gm
496                  new_graph_signature = aten_export_artifact.sig
497  
498                  # In the previous step, we assume constants as buffers for AOTDispatcher to
499                  # functianalize properly, so undo that here
500                  new_graph_signature = (
501                      _override_graph_signature_for_temp_registered_constants(
502                          new_graph_signature, temp_registered_constants
503                      )
504                  )
505  
506                  _populate_param_buffer_metadata_to_new_gm(
507                      params_buffers_to_node_meta, gm, new_graph_signature
508                  )
509  
510                  # overwrite signature for non-persistent buffers
511                  new_graph_signature = _overwrite_signature_for_non_persistent_buffers(
512                      ep.graph_signature, new_graph_signature
513                  )
514  
515                  constants = _materialize_and_lift_constants(
516                      gm, new_graph_signature, new_fake_constant_attrs
517                  )
518  
519                  placeholder_naming_pass(
520                      gm,
521                      new_graph_signature,
522                      patched_mod,
523                      new_fake_args,
524                      new_fake_kwargs,
525                      fake_params_buffers,
526                      constants,
527                  )
528  
529          _verify_nn_module_stack(gm)
530          _verify_stack_trace(gm)
531          _verify_placeholder_names(gm, new_graph_signature)
532  
533          gm, new_graph_signature = _remove_unneccessary_copy_op_pass(
534              gm, new_graph_signature
535          )
536  
537          # When we apply parameterixzation rule to unwrap
538          # subclasses, the state dict will now have different
539          # desugared parameters. We need to manually filter those
540          # and update the ep.state_dict. Ideally, we should just return
541          # the state dict of ep.module but ep.module only stores params
542          # buffers that participate in forward. If we undo this behaviour,
543          # it would break some downstream users.
544          new_state_dict = {
545              **ep.state_dict,
546              **{
547                  name: p
548                  for name, p in unwrapped_params_buffers.items()
549                  if name not in wrapped_params_buffers
550              },
551          }
552  
553          for name, p in wrapped_params_buffers.items():
554              # Buffers can be persistent/non-persistent
555              if name not in new_state_dict:
556                  assert not isinstance(p, torch.nn.Parameter)
557  
558              if name in new_state_dict:
559                  if name not in unwrapped_params_buffers:
560                      new_state_dict.pop(name)
561  
562          return gm, new_graph_signature, new_state_dict
563  
564      old_placeholders = [
565          node for node in ep.graph_module.graph.nodes if node.op == "placeholder"
566      ]
567      fake_args = [node.meta["val"] for node in old_placeholders]
568  
569      buffers_to_remove = [name for name, _ in ep.graph_module.named_buffers()]
570      for name in buffers_to_remove:
571          delattr(ep.graph_module, name)
572  
573      # TODO(zhxhchen17) Return the new graph_signature directly.
574      fake_mode = detect_fake_mode(fake_args)
575      fake_mode = contextlib.nullcontext() if fake_mode is None else fake_mode  # type: ignore[assignment]
576      custom_triton_ops_decomposition_ctx = (
577          contextlib.nullcontext
578          if decompose_custom_triton_ops
579          else _disable_custom_triton_op_functional_decomposition
580      )
581      with (
582          _ignore_backend_decomps(),
583          fake_mode,
584          _override_composite_implicit_decomp(cia_to_decomp),
585          custom_triton_ops_decomposition_ctx(),
586      ):
587          gm, graph_signature = aot_export_module(
588              ep.graph_module,
589              fake_args,
590              decompositions=python_decomp_table,
591              trace_joint=True if joint_loss_index is not None else False,
592              output_loss_index=(
593                  joint_loss_index if joint_loss_index is not None else None
594              ),
595          )
596          gm.graph.eliminate_dead_code()
597  
598      # Update the signatures with the new placeholder names in case they
599      # changed when calling aot_export
600      def update_arg(old_arg, new_ph):
601          if isinstance(old_arg, ConstantArgument):
602              return old_arg
603          elif isinstance(old_arg, TensorArgument):
604              return TensorArgument(name=new_ph.name)
605          elif isinstance(old_arg, SymIntArgument):
606              return SymIntArgument(name=new_ph.name)
607          elif isinstance(old_arg, SymFloatArgument):
608              return SymFloatArgument(name=new_ph.name)
609          elif isinstance(old_arg, SymBoolArgument):
610              return SymBoolArgument(name=new_ph.name)
611          raise RuntimeError(f"Type of old_arg not supported: {type(old_arg)}")
612  
613      new_placeholders = [node for node in gm.graph.nodes if node.op == "placeholder"]
614      new_outputs = list(gm.graph.nodes)[-1].args[0]
615  
616      # rename the placeholders
617      assert len(new_placeholders) == len(old_placeholders)
618      for old_ph, new_ph in zip(old_placeholders, new_placeholders):
619          new_ph.name = new_ph.target = old_ph.name
620  
621      # handle name collisions with newly decomposed graph nodes
622      name_map = {ph.name: ph.name for ph in new_placeholders}
623      for node in gm.graph.nodes:
624          if node.op == "placeholder":
625              continue
626          node.name = _rename_without_collisions(name_map, node.name, node.name)
627  
628      # propagate names to higher order op subgraphs
629      _name_hoo_subgraph_placeholders(gm)
630  
631      # Run this pass before creating input/output specs, since size-related CSE/DCE might affect output signature.
632      # Overwrite output specs afterwards.
633      from torch._export.passes._node_metadata_hook import (
634          _node_metadata_hook,
635          _set_node_metadata_hook,
636      )
637      from torch._functorch._aot_autograd.input_output_analysis import _graph_output_names
638  
639      if not torch._dynamo.config.do_not_emit_runtime_asserts:
640          stack_trace = (
641              'File "torch/fx/passes/runtime_assert.py", line 24, '
642              "in insert_deferred_runtime_asserts"
643          )
644          shape_env = _get_shape_env(gm)
645          if shape_env is not None:
646              with _set_node_metadata_hook(
647                  gm, functools.partial(_node_metadata_hook, stack_trace=stack_trace)
648              ):
649                  insert_deferred_runtime_asserts(
650                      gm,
651                      shape_env,
652                      f"exported program: {first_call_function_nn_module_stack(gm.graph)}",
653                      export=True,
654                  )
655  
656      # update output specs
657      gm.recompile()
658      for i, name in enumerate(_graph_output_names(gm)):
659          if isinstance(new_outputs[i], torch.fx.Node):
660              new_outputs[i].name = name
661  
662      # To match the output target with correct input for input mutations
663      # need to find the old to new placeholder map
664      old_new_placeholder_map = {
665          spec.arg.name: new_placeholders[i].name
666          for i, spec in enumerate(ep.graph_signature.input_specs)
667          if not isinstance(spec.arg, ConstantArgument)
668      }
669  
670      input_specs = [
671          InputSpec(
672              spec.kind,
673              update_arg(spec.arg, new_placeholders[i]),
674              spec.target,
675              spec.persistent,
676          )
677          for i, spec in enumerate(ep.graph_signature.input_specs)
678      ]
679  
680      output_specs = []
681  
682      # handle buffer & input mutations; these appear before loss output & gradients
683      # (1) ep.graph_signature.input_specs tells us types of inputs
684      # (2) graph_signature.user_inputs tells us node input names in order
685      # (3) graph_signature.user_inputs_to_mutate tells us buffer & input mutations
686      # map (3) -> (2) for input order, -> (1) for input type
687      user_inputs_index = {name: i for i, name in enumerate(graph_signature.user_inputs)}
688      mutation_names = list(graph_signature.user_inputs_to_mutate.keys())
689      assert mutation_names == [node.name for node in new_outputs[: len(mutation_names)]]
690      for output_name, input_name in graph_signature.user_inputs_to_mutate.items():
691          i = user_inputs_index[input_name]
692          input_spec = ep.graph_signature.input_specs[i]
693          assert input_spec.kind in (InputKind.USER_INPUT, InputKind.BUFFER)
694          output_kind = (
695              OutputKind.BUFFER_MUTATION
696              if input_spec.kind == InputKind.BUFFER
697              else OutputKind.USER_INPUT_MUTATION
698          )
699          target = (
700              input_spec.target
701              if input_spec.kind == InputKind.BUFFER
702              else input_spec.arg.name
703          )
704          output_specs.append(
705              OutputSpec(
706                  kind=output_kind,
707                  arg=TensorArgument(name=output_name),
708                  target=target,
709              )
710          )
711  
712      # handle actual user outputs
713      for i, spec in enumerate(ep.graph_signature.output_specs):
714          output_specs.append(
715              OutputSpec(
716                  OutputKind.LOSS_OUTPUT if i == joint_loss_index else spec.kind,
717                  update_arg(spec.arg, new_outputs[len(mutation_names) + i]),
718                  old_new_placeholder_map.get(spec.target, spec.target),
719              )
720          )
721  
722      if joint_loss_index is not None:
723          assert graph_signature.backward_signature is not None
724          gradients = graph_signature.backward_signature.gradients_to_user_inputs
725          assert len(graph_signature.user_inputs) == len(ep.graph_signature.input_specs)
726          specs = {
727              graph_signature.user_inputs[i]: spec
728              for i, spec in enumerate(ep.graph_signature.input_specs)
729              if isinstance(spec.arg, TensorArgument)
730          }
731          for i, node in enumerate(new_outputs[len(output_specs) :]):
732              source = gradients[node.name]
733              spec = specs[source]  # type: ignore[index]
734              if spec.kind == InputKind.PARAMETER:
735                  kind = OutputKind.GRADIENT_TO_PARAMETER
736                  target = spec.target
737              elif spec.kind == InputKind.USER_INPUT:
738                  kind = OutputKind.GRADIENT_TO_USER_INPUT
739                  target = source
740              else:
741                  raise AssertionError(f"Unknown input kind: {spec.kind}")
742              output_specs.append(
743                  OutputSpec(
744                      kind,
745                      TensorArgument(name=node.name),
746                      target,
747                  )
748              )
749  
750      assert len(new_placeholders) == len(old_placeholders)
751  
752      new_graph_signature = ExportGraphSignature(
753          input_specs=input_specs, output_specs=output_specs
754      )
755      # NOTE: aot_export adds symint metadata for placeholders with int
756      # values; since these become specialized, we replace such metadata with
757      # the original values.
758      # Also, set the param/buffer metadata back to the placeholders.
759      for old_node, new_node in zip(old_placeholders, new_placeholders):
760          if not isinstance(old_node.meta["val"], torch.Tensor):
761              new_node.meta["val"] = old_node.meta["val"]
762  
763          if (
764              new_node.target in new_graph_signature.inputs_to_parameters
765              or new_node.target in new_graph_signature.inputs_to_buffers
766          ):
767              for k, v in old_node.meta.items():
768                  new_node.meta[k] = v
769      return gm, new_graph_signature, ep.state_dict

772  def _remove_unneccessary_copy_op_pass(
773      gm: torch.fx.GraphModule, new_graph_signature: ExportGraphSignature
774  ) -> tuple[torch.fx.GraphModule, ExportGraphSignature]:
775      """ 
776      Removes redundant copy_ node that was introduced due to mutated buffer. 
777      """
778      with gm._set_replace_hook(new_graph_signature.get_replace_hook()):
779          for node in gm.graph.nodes:
780              if node.op == "output":
781                  args, _ = pytree.tree_flatten(node.args)
782                  for out in args:
783                      if (
784                          isinstance(out, torch.fx.Node)
785                          and out.name in new_graph_signature.buffers_to_mutate
786                      ):
787                          if (
788                              out.op == "call_function"
789                              and out.target == torch.ops.aten.copy.default
790                          ):
791                              out.replace_all_uses_with(out.args[1])  # type: ignore[arg-type]
792                              gm.graph.erase_node(out)
793          gm.recompile()
794      return gm, new_graph_signature

797  def _common_getitem_elimination_pass(
798      gm: torch.fx.GraphModule, graph_signature, module_call_graph
799  ):
800      with gm._set_replace_hook(graph_signature.get_replace_hook()):
801          for module in gm.modules():
802              if not isinstance(module, torch.fx.GraphModule):
803                  continue
804  
805              node_id: dict[torch.fx.Node, str] = {}
806              getitems: dict[str, torch.fx.Node] = {}
807              for node in list(module.graph.nodes):
808                  if node.op == "call_function" and node.target == operator.getitem:
809                      source, idx = node.args
810                      new_id = f"{node_id[source]}.{idx}"
811                      if new_id in getitems:
812                          node.replace_all_uses_with(getitems[new_id])
813                          for entry in module_call_graph:
814                              if entry.signature is not None:
815                                  entry.signature.replace_all_uses_with(
816                                      node, getitems[new_id]
817                                  )
818                          module.graph.erase_node(node)
819                      else:
820                          getitems[new_id] = node
821                          node_id[node] = new_id
822                  else:
823                      node_id[node] = node.name

826  def _get_updated_module_call_graph(
827      old_gm: torch.fx.GraphModule,
828      old_graph_signature: ExportGraphSignature,
829      gm: torch.fx.GraphModule,
830      graph_signature: ExportGraphSignature,
831      old_module_call_graph: list[ModuleCallEntry],
832  ):
833      new_module_call_graph = copy.deepcopy(old_module_call_graph)
834  
835      old_nodes = {node.name: node for node in old_gm.graph.nodes}
836  
837      old_graph_params_buffers = {
838          **old_graph_signature.inputs_to_parameters,
839          **old_graph_signature.inputs_to_buffers,
840      }
841      new_graph_params_buffers = {
842          **graph_signature.inputs_to_parameters,
843          **graph_signature.inputs_to_buffers,
844      }
845  
846      # use node-level provenance metadata to create a map
847      # from old node names to new node names
848      provenance: dict[str, str] = {}
849  
850      user_input_counter = 0
851      old_user_input_names = [
852          node.target for node in old_gm.graph.nodes if node.op == "placeholder"
853      ]
854      old_user_input_names = list(
855          filter(
856              lambda x: x not in old_graph_params_buffers
857              and x not in old_graph_signature.input_tokens,
858              old_user_input_names,
859          )
860      )
861      new_user_input_names = [
862          node.target for node in gm.graph.nodes if node.op == "placeholder"
863      ]
864  
865      for node in gm.graph.nodes:
866          if history := node.meta.get("from_node", []):
867              provenance[history[-1].name] = node.name
868  
869          # For params and buffers, we might have applied parameterizaiton rule
870          # so that the names might have changed. But for user inputs, we know we
871          # must preserve the old name.
872          elif node.op == "placeholder":
873              if not (
874                  node.target in new_graph_params_buffers
875                  or node.target in graph_signature.input_tokens
876              ):
877                  if node.target in new_user_input_names:
878                      assert isinstance(node.name, str)
879                      old_name = old_user_input_names[user_input_counter]
880                      assert isinstance(old_name, str)
881                      provenance[old_name] = node.name
882                      user_input_counter += 1
883  
884      # For all the parameters and buffers, we first see
885      # if they are result of paramerizaitons and if they
886      # are, we log them and error later
887      old_param_to_desugared = defaultdict(list)
888      for name, target in new_graph_params_buffers.items():
889          # if the parameters are not parametrized, the naming won't change.
890          if not target.startswith("parametrizations."):
891              # If we are in strict mode, we can't just reuse the param names
892              if name in old_graph_params_buffers:
893                  provenance[name] = name
894          else:
895              old_target = ".".join(target.split(".")[1:-1])
896              old_param_to_desugared[old_target].append(name)
897  
898      # map old names to new names in module call signatures
899      for entry in new_module_call_graph:
900          signature = entry.signature
901          if signature is None:
902              continue
903          for x in [*signature.inputs, *signature.outputs]:
904              # We noticed that submodule is taking subclass as input. we can't
905              # preserve signature here.
906              if x.name in old_param_to_desugared:
907                  raise ValueError(
908                      f"It looks like {x.name} is a tensor subclass. "
909                      f"Preserving submodule that takes subclass parameter is not supported"
910                      f" in inference IR because we desugar them, resulting in more tensors"
911                  )
912  
913              if x.name in provenance:
914                  x.name = provenance[x.name]
915  
916              # This can happen when aten.to is called at graph boundaries.
917              # Basically aten.to at post-dispatch level can either be copy
918              # or alias. In the alias case, we will no-op it so it will
919              # disappear from the graph. If we detect such case, we should
920              # reuse the input to aten.to as the new input to the submodule.
921              # Technically this can happen for other maybe aliasing ops,
922              # but aten.to is probably the most common one.
923              elif x.name in old_nodes:
924                  old_node = old_nodes[x.name]
925                  if old_node.op == "call_function" and old_node.target in [
926                      torch.ops.aten.to.dtype_layout,
927                      torch.ops.aten.to.device,
928                      torch.ops.aten.to.dtype,
929                  ]:
930                      old_target = old_node.args[0].name
931                      if old_target not in provenance:
932                          raise ValueError(
933                              f"It looks like {old_target} is a tensor subclass. "
934                              f"Preserving submodule that takes subclass parameter is not supported"
935                              f" in inference IR because we desugar them, resulting in more tensors"
936                          )
937  
938                      x.name = provenance[old_target]
939  
940      return new_module_call_graph

998  class ExportedProgram:
999      """ 
1000     Package of a program from :func:`export`. It contains 
1001     an :class:`torch.fx.Graph` that represents Tensor computation, a state_dict containing 
1002     tensor values of all lifted parameters and buffers, and various metadata. 
1003  
1004     You can call an ExportedProgram like the original callable traced by 
1005     :func:`export` with the same calling convention. 
1006  
1007     To perform transformations on the graph, use ``.module`` property to access 
1008     an :class:`torch.fx.GraphModule`. You can then use 
1009     `FX transformation <https://pytorch.org/docs/stable/fx.html#writing-transformations>`_ 
1010     to rewrite the graph. Afterwards, you can simply use :func:`export` 
1011     again to construct a correct ExportedProgram. 
1012     """

1017         graph: torch.fx.Graph,

1256     def _get_flat_args_with_check(self, args, kwargs):
1257         """Flatten args, kwargs using pytree, then, check specs. 
1258

1271             kwargs = reorder_kwargs(kwargs, in_spec)

1279     def _graph_module_flat_inputs(self, args: Any, kwargs: Any) -> Any:
1280         """Transform args, kwargs of __call__ to args for graph_module. 
1281  
1282         self.graph_module takes stuff from state dict as inputs. 
1283         The invariant is for ep: ExportedProgram is 
1284         ep(args, kwargs) == 
1285           ep.postprocess(ep.graph_module(ep.graph_module_flat_inputs(args, kwargs))) 
1286         """
1287 
1288         in_spec = self.call_spec.in_spec
1289         flat_args, received_spec = self._get_flat_args_with_check(args, kwargs)
1290         if in_spec is not None and not is_equivalent(
1291             received_spec, in_spec, _fx_collection_equivalence_fn
1292         ):
1293             raise ValueError(
1294                 "Trying to flatten user inputs with exported input tree spec: \n"
1295                 f"{in_spec}\n"
1296                 "but actually got inputs with tree spec of: \n"
1297                 f"{received_spec}"
1298             )
1299 
1300         additional_inputs = []
1301         for input_ in self.graph_signature.input_specs:
1302             if input_.kind == InputKind.USER_INPUT:
1303                 continue
1304             elif input_.kind in (
1305                 InputKind.PARAMETER,
1306                 InputKind.BUFFER,
1307             ):
1308                 if input_.persistent is False:
1309                     # This is a non-persistent buffer, grab it from our
1310                     # constants instead of the state dict.
1311                     additional_inputs.append(self.constants[input_.target])
1312                 else:
1313                     additional_inputs.append(self.state_dict[input_.target])
1314             elif input_.kind in (
1315                 InputKind.CONSTANT_TENSOR,
1316                 InputKind.CUSTOM_OBJ,
1317             ):
1318                 additional_inputs.append(self.constants[input_.target])
1319         additional_inputs = tuple(additional_inputs)
1320 
1321         # NOTE: calling convention is first params, then buffers, then args as user supplied them.
1322         # See: torch/_functorch/aot_autograd.py#L1034
1323         return additional_inputs + flat_args

1331     def _postprocess_graph_module_outputs(self, res, orig_args, orig_kwargs):
1332         """Process potential mutations to the input. 
1333  
1334         Because self.graph_module is functional, so mutations has to be written 
1335         back after execution of graph_module. 
1336         """
1337         import torch._export.error as error
1338 
1339         flat_args, _ = self._get_flat_args_with_check(orig_args, orig_kwargs)
1340         if self.call_spec.out_spec is not None:
1341             buffer_mutation = self.graph_signature.buffers_to_mutate
1342             user_input_mutation = self.graph_signature.user_inputs_to_mutate
1343             num_mutated = len(buffer_mutation) + len(user_input_mutation)
1344             mutated_values = res[:num_mutated]
1345 
1346             # Exclude dependency token from final result.
1347             assertion_dep_token = self.graph_signature.assertion_dep_token
1348             if assertion_dep_token is not None:
1349                 assertion_dep_token_index = next(iter(assertion_dep_token.keys()))
1350                 res = res[:assertion_dep_token_index]
1351 
1352             res = res[num_mutated:]
1353             try:
1354                 res = pytree.tree_unflatten(res, self.call_spec.out_spec)
1355             except Exception:
1356                 _, received_spec = pytree.tree_flatten(res)
1357                 raise error.InternalError(  # noqa: B904
1358                     "Trying to flatten user outputs with exported output tree spec: \n"
1359                     f"{self.call_spec.out_spec}\n"
1360                     "but actually got outputs with tree spec of: \n"
1361                     f"{received_spec}"
1362                 )
1363             finally:
1364                 user_inputs = [
1365                     spec
1366                     for spec in self.graph_signature.input_specs
1367                     if spec.kind == InputKind.USER_INPUT
1368                 ]
1369                 for i, value in enumerate(mutated_values):
1370                     output_spec = self.graph_signature.output_specs[i]
1371                     if output_spec.kind == OutputKind.BUFFER_MUTATION:
1372                         assert output_spec.target is not None
1373                         self.state_dict[output_spec.target] = value
1374                     elif output_spec.kind == OutputKind.USER_INPUT_MUTATION:
1375                         assert output_spec.target is not None
1376                         index = next(
1377                             i
1378                             for i, spec in enumerate(user_inputs)
1379                             if spec.arg.name == output_spec.target
1380                         )
1381                         flat_args[index].copy_(value)
1382                     else:
1383                         raise AssertionError(f"Unexpected kind: {output_spec.kind}")
1384         return res

1428     def run_decompositions(
1429         self,
1430         decomp_table: Optional[dict[torch._ops.OperatorBase, Callable]] = None,
1431         decompose_custom_triton_ops: bool = False,
1432     ) -> "ExportedProgram":
1433         """ 
1434         Run a set of decompositions on the exported program and returns a new 
1435         exported program. By default we will run the Core ATen decompositions to 
1436         get operators in the 
1437         `Core ATen Operator Set <https://pytorch.org/docs/stable/torch.compiler_ir.html>`_. 
1438  
1439         For now, we do not decompose joint graphs. 
1440  
1441         Args: 
1442             decomp_table: 
1443              An optional argument that specifies decomp behaviour for Aten ops 
1444              (1) If None, we decompose to core aten decompositions 
1445              (2) If empty, we don't decompose any operator 
1446  
1447  
1448         Some examples: 
1449  
1450         If you don't want to decompose anything 
1451  
1452         .. code-block:: python 
1453  
1454             ep = torch.export.export(model, ...) 
1455             ep = ep.run_decompositions(decomp_table={}) 
1456  
1457         If you want to get a core aten operator set except for certain operator, you can do following: 
1458  
1459         .. code-block:: python 
1460  
1461             ep = torch.export.export(model, ...) 
1462             decomp_table = torch.export.default_decompositions() 
1463             decomp_table[your_op] = your_custom_decomp 
1464             ep = ep.run_decompositions(decomp_table=decomp_table) 
1465         """
1466         _decomp_table = (
1467             default_decompositions() if decomp_table is None else dict(decomp_table)
1468         )
1469 
1470         if isinstance(_decomp_table, CustomDecompTable):
1471             _decomp_table = _decomp_table.materialize()
1472 
1473         # Note [Seperating decomp_table into CIA decomps and non-CIA decomps]
1474         # At this point, we have a decomp_table that contains decomp behaviour for
1475         # both CIA and post-autograd ops.
1476         # We need to separate the op into two categories:
1477         # 1. CIA op: These are the ops that we want to override
1478         #    CompositeImplicitAutograd decomp for. For them, we need to use _override_composite_implicit_decomp
1479         #    context manager to plumb it through AOTDispatcher
1480         # 2. Non-CIA op: These ops are only relevant after AOTDIspatcher runs, so just
1481         #    checking if they are statically functional is enough.
1482         # For joint IR case tho, we need to use the old path because we can't register
1483         # custom decomps this way because we can't use context manager as it installs
1484         # autograd_error node.
1485         (
1486             cia_to_decomp,
1487             python_decomp_table,
1488         ) = _split_decomp_table_to_cia_and_python_decomp(_decomp_table)
1489 
1490         return _decompose_exported_program(
1491             self,
1492             cia_to_decomp=cia_to_decomp,
1493             python_decomp_table=python_decomp_table,
1494             joint_loss_index=None,
1495             decompose_custom_triton_ops=decompose_custom_triton_ops,
1496         )
