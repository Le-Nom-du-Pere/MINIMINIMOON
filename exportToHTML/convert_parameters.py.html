7    def parameters_to_vector(parameters: Iterable[torch.Tensor]) -> torch.Tensor:
8        r"""Flatten an iterable of parameters into a single vector. 
9     
10       Args: 
11           parameters (Iterable[Tensor]): an iterable of Tensors that are the 
12               parameters of a model. 
13    
14       Returns: 
15           The parameters represented by a single vector 
16       """
17       # Flag for the device where the parameter is located
18       param_device = None
19   
20       vec = []
21       for param in parameters:
22           # Ensure the parameters are located in the same device
23           param_device = _check_param_device(param, param_device)
24   
25           vec.append(param.view(-1))
26       return torch.cat(vec)

29   def vector_to_parameters(vec: torch.Tensor, parameters: Iterable[torch.Tensor]) -> None:
30       r"""Copy slices of a vector into an iterable of parameters. 
31    
32       Args: 
33           vec (Tensor): a single vector representing the parameters of a model. 
34           parameters (Iterable[Tensor]): an iterable of Tensors that are the 
35               parameters of a model. 
36       """
37       # Ensure vec of type Tensor
38       if not isinstance(vec, torch.Tensor):
39           raise TypeError(f"expected torch.Tensor, but got: {torch.typename(vec)}")
40       # Flag for the device where the parameter is located
41       param_device = None
42   
43       # Pointer for slicing the vector for each parameter
44       pointer = 0
45       for param in parameters:
46           # Ensure the parameters are located in the same device
47           param_device = _check_param_device(param, param_device)
48   
49           # The length of the parameter
50           num_param = param.numel()
51           # Slice the vector, reshape it, and replace the old data of the parameter
52           param.data = vec[pointer : pointer + num_param].view_as(param).data
53   
54           # Increment the pointer
55           pointer += num_param

58   def _check_param_device(param: torch.Tensor, old_param_device: Optional[int]) -> int:
59       r"""Check if the parameters are located on the same device. 
60    
61       Currently, the conversion between model parameters and single vector form is not supported 
62       for multiple allocations, e.g. parameters in different GPUs/PrivateUse1s, or mixture of CPU/GPU/PrivateUse1. 
63    
64       Args: 
65           param ([Tensor]): a Tensor of a parameter of a model 
66           old_param_device (int): the device where the first parameter of a 
67                                   model is allocated. 
68    
69       Returns: 
70           old_param_device (int): report device for the first time 
71       """
72       # Meet the first parameter
73       support_device_types = ["cuda", torch._C._get_privateuse1_backend_name()]
74       if old_param_device is None:
75           old_param_device = (
76               param.get_device() if param.device.type in support_device_types else -1
77           )
78       else:
79           warn = False
80           if (
81               param.device.type in support_device_types
82           ):  # Check if in same GPU/PrivateUse1
83               warn = param.get_device() != old_param_device
84           else:  # Check if in CPU
85               warn = old_param_device != -1
86           if warn:
87               raise TypeError(
88                   "Found two parameters on different devices, "
89                   "this is currently not supported."
90               )
91       return old_param_device
