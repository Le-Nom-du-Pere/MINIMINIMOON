26   def _convert_to_int(val):
27       # Convert simple sympy Integers into concrete int
28       if val in (sympy.oo, int_oo):
29           return math.inf
30       if val in (-sympy.oo, -int_oo):
31           return -math.inf
32       if isinstance(val, sympy.Integer):
33           return int(val)
34       raise RuntimeError("Export constraints cannot be non-integer expressions")

37   def _convert_range_to_int(range: ValueRanges):
38       assert isinstance(range, ValueRanges)
39       min_val = _convert_to_int(range.lower)
40       max_val = _convert_to_int(range.upper)
41       return min_val, max_val

44   class _AddRuntimeAssertionsForInlineConstraintsPass(PassBase):
45       def __init__(
46           self,
47           range_constraints: dict[sympy.Symbol, ValueRanges],
48       ):
49           super().__init__()
50           self.range_constraints: dict[sympy.Symbol, ValueRanges] = range_constraints
51           self._asserts_generated_unbacked_symbols: set[sympy.Symbol] = set()
52           self.counter = 0
53   
54       def _assert_range_constraint(self, node, lower, upper, assert_msg):
55           last_node = node
56           if lower > -math.inf:
57               last_node = self._insert_assert_async(
58                   last_node, operator.ge, node, lower, assert_msg
59               )
60   
61           if upper < math.inf:
62               last_node = self._insert_assert_async(
63                   last_node, operator.le, node, upper, assert_msg
64               )
65   
66       def _insert_assert_async(self, last_node, op, lower, upper, assert_msg):
67           """ 
68           Inserts assert_async call_function nodes in the graph. This function is 
69           called **during** the interpreter-based pass. 
70           """
71           self.counter += 1
72           graph = last_node.graph
73           with graph.inserting_after(last_node):
74               cmp = graph.call_function(op, (lower, upper), {})
75           with graph.inserting_after(cmp):
76               cmp_tensor = graph.call_function(
77                   torch.ops.aten.scalar_tensor.default, (cmp,), {}
78               )
79           with graph.inserting_after(cmp_tensor):
80               assert_async = graph.call_function(
81                   torch.ops.aten._assert_async.msg,
82                   (cmp_tensor, assert_msg),
83                   {},
84               )
85           return assert_async
86   
87       def call(self, graph_module) -> PassResult:
88           self.existing_inline_assertions = _get_existing_inline_assertions(
89               graph_module, self.range_constraints
90           )
91   
92           for module in graph_module.modules():
93               if not isinstance(module, torch.fx.GraphModule):
94                   continue
95               for node in module.graph.nodes:
96                   if node.op != "call_function":
97                       continue
98                   if "val" not in node.meta:
99                       continue
100  
101                  val = node.meta["val"]
102                  # In general, we may have to deal the case such as: ret[1].shape[0].
103                  # We need first find out what symbols require assertion, then we need to follow the path
104                  # from ret to the symbol, construct the proxies along the way and construct the messages
105                  # piece-wise at the same time.
106                  #
107                  # We use post-order traversal to collect all the proxies callbacks needed, construct
108                  # the error message callbacks, and at the top-level traversal tree we execute all the callbacks.
109                  # We need the callbacks because, in order to call the function to create a proxy for shape[0], we
110                  # need the proxy for shape, which further requires the proxy for ret[1], etc.
111  
112                  def add_assertions(val):
113                      call_backs: list[Callable] = []
114                      messages: list[str] = []
115                      if isinstance(val, (torch.SymInt, torch.SymFloat, torch.SymBool)):
116                          symbol = val.node.expr
117                          if symbol in self.existing_inline_assertions:
118                              return call_backs, messages
119                          if isinstance(symbol, sympy.Symbol) and free_unbacked_symbols(
120                              symbol
121                          ):
122                              if symbol in self._asserts_generated_unbacked_symbols:
123                                  return call_backs, messages
124                              # We only care about unbacked symints for these inline
125                              # constraints, which are prefixed with 'u'
126                              constraint = self.range_constraints[symbol]
127                              min_val, max_val = _convert_range_to_int(constraint)
128                              assert_msg = f" is outside of inline constraint [{min_val}, {max_val}]."
129                              call_backs.append(
130                                  partial(
131                                      self._assert_range_constraint,
132                                      lower=min_val,
133                                      upper=max_val,
134                                  )
135                              )
136                              messages.append(assert_msg)
137                              self._asserts_generated_unbacked_symbols.add(symbol)
138  
139                      elif isinstance(val, torch.Tensor):
140                          for i, sym in enumerate(val.shape):
141                              cbs, msgs = add_assertions(sym)
142                              for cb, msg in zip(cbs, msgs):
143  
144                                  def sym_size_cb(node, assert_msg, dim):
145                                      with node.graph.inserting_after(node):
146                                          dim_node = module.graph.call_function(
147                                              torch.ops.aten.sym_size.int,
148                                              (node, dim),
149                                              {},
150                                          )
151                                      cb(node=dim_node, assert_msg=assert_msg)
152  
153                                  call_backs.append(partial(sym_size_cb, dim=i))
154                                  messages.append(f".shape[{i}]" + msg)
155                      return call_backs, messages
156  
157                  callbacks, messages = add_assertions(val)
158                  for cb, msg in zip(callbacks, messages):
159                      cb(node=node, assert_msg=f"{node}" + msg)
160  
161              module.recompile()
162  
163          # Sometimes this pass would return a wrong graph where we have mismatched
164          # node names in signature. Before we fix it, let's just skip it.
165          if (
166              self.counter == 0
167              and type(self) is _AddRuntimeAssertionsForInlineConstraintsPass
168          ):
169              return PassResult(graph_module, False)
170  
171          # Populate the stack trace with dummy vals to respect IR
172          for node in graph_module.graph.nodes:
173              if not node.meta.get("stack_trace", None) and node.op not in [
174                  "placeholder",
175                  "output",
176              ]:
177                  node.meta["stack_trace"] = "".join(traceback.format_stack(limit=1))
178          return PassResult(graph_module, True)

181  def _get_existing_inline_assertions(
182      graph_module: torch.fx.GraphModule,
183      range_constraints: dict[sympy.Symbol, ValueRanges],
184  ) -> dict[sympy.Symbol, ValueRanges]:
185      existing_inline_assertions: dict[sympy.Symbol, ValueRanges] = {}
186  
187      for module in graph_module.modules():
188          if not isinstance(module, torch.fx.GraphModule):
189              continue
190  
191          # Find all the existing inline assertions. They will look something like:
192          # %_local_scalar_dense = call_function[target=torch.ops.aten._local_scalar_dense.default](args = (%arg1_1,), kwargs = {})
193          # %ge = call_function[target=operator.ge](args = (%_local_scalar_dense, 0), kwargs = {})
194          # %_assert_scalar = call_function[target=torch.ops.aten._assert_scalar.default](args = (%scalar_tensor, "..."), kwargs = {})
195          for node in module.graph.nodes:
196              if node.target != torch.ops.aten._assert_scalar.default:
197                  continue
198  
199              compare_arg = node.args[0]
200              if not (
201                  isinstance(compare_arg, torch.fx.Node)
202                  and compare_arg.op == "call_function"
203                  and compare_arg.target in (operator.le, operator.ge)
204                  and len(compare_arg.args) == 2
205              ):
206                  continue
207  
208              compare_op = compare_arg.target
209              lhs, rhs = compare_arg.args
210  
211              def maybe_get_symint(x):
212                  if (
213                      isinstance(x, torch.fx.Node)
214                      and "val" in x.meta
215                      and isinstance(x.meta["val"], torch.SymInt)
216                  ):
217                      return x.meta["val"].node.expr
218                  return x
219  
220              lhs = maybe_get_symint(lhs)
221              rhs = maybe_get_symint(rhs)
222  
223              if compare_op == operator.ge:
224                  lhs, rhs = rhs, lhs
225  
226              if isinstance(lhs, sympy.Symbol) and isinstance(rhs, int):
227                  symint = lhs
228                  scalar = rhs
229              elif isinstance(rhs, sympy.Symbol) and isinstance(lhs, int):
230                  symint = rhs
231                  scalar = lhs
232              else:
233                  continue
234  
235              if symint not in range_constraints:
236                  raise RuntimeError(
237                      f"Unable to find symint {symint} in {range_constraints}"
238                  )
239  
240              previous_range = existing_inline_assertions.get(
241                  symint, ValueRanges(-math.inf, math.inf)
242              )
243  
244              if symint is lhs:
245                  bounds = ValueRanges(-math.inf, scalar)
246              else:
247                  bounds = ValueRanges(scalar, math.inf)
248              existing_inline_assertions[symint] = previous_range & bounds
249  
250      return existing_inline_assertions
