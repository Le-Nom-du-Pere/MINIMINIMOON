7    template<typename T>
8    inline void FunctionSchema::checkArg(
9        const IValue& value,
10       const Argument& argument,
11       std::optional<size_t> pos) const {
12     if (value.isTensor() && argument.type() == TensorType::get()) {
13       // Fast-path for the common case
14       return;
15     }
16     if (value.isGenericDict() && value.toGenericDict().empty()) {
17       return;
18     }
19     if (!value.type<T>()->isSubtypeOf(*argument.type())) {
20       TORCH_CHECK(
21           false,
22           formatTypeMismatchMsg(
23               argument, value.type<T>()->repr_str(), pos));
24     }
25   }

27   template <typename T>
28   inline void FunctionSchema::checkAndNormalizeInputs(
29       std::vector<IValue>& inputs,
30       const std::unordered_map<std::string, IValue>& kwargs) const {
31     // Do we have more inputs than the schema accepts?
32     TORCH_CHECK(
33         inputs.size() <= arguments().size(),
34         "Expected at most ",
35         arguments().size(),
36         " argument(s) for operator '",
37         name(),
38         "', but received ",
39         inputs.size(),
40         " argument(s). Declaration: ",
41         *this);
42   
43     size_t consumed_kwargs = 0;
44     for (const auto pos : c10::irange(arguments().size())) {
45       const auto& argument = arguments()[pos];
46       if (pos < inputs.size()) {
47         checkArg<T>(inputs[pos], argument, pos);
48         continue;
49       }
50       auto it = kwargs.find(argument.name());
51       if (it != kwargs.end()) {
52         checkArg<T>(it->second, argument, std::nullopt);
53         inputs.push_back(it->second);
54         consumed_kwargs++;
55         continue;
56       }
57       if (argument.default_value()) {
58         inputs.push_back(*argument.default_value());
59         continue;
60       }
61       TORCH_CHECK(false,
62           name(),
63           "() is missing value for argument '",
64           argument.name(),
65           "'. Declaration: ",
66           *this);
67     }
68     if (consumed_kwargs != kwargs.size()) {
69       std::vector<std::string> names;
70       names.reserve(kwargs.size());
71       for(const auto& k : kwargs) {
72         names.emplace_back(k.first);
73       }
74       TORCH_CHECK(false, findErrorInKwargs(names));
75     }
76   }
