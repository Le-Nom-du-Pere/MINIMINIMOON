129  @public
130  def sring(exprs, *symbols, **options):
131      """Construct a ring deriving generators and domain from options and input expressions. 
132   
133      Parameters 
134      ========== 
135   
136      exprs : :class:`~.Expr` or sequence of :class:`~.Expr` (sympifiable) 
137      symbols : sequence of :class:`~.Symbol`/:class:`~.Expr` 
138      options : keyword arguments understood by :class:`~.Options` 
139   
140      Examples 
141      ======== 
142   
143      >>> from sympy import sring, symbols 
144   
145      >>> x, y, z = symbols("x,y,z") 
146      >>> R, f = sring(x + 2*y + 3*z) 
147      >>> R 
148      Polynomial ring in x, y, z over ZZ with lex order 
149      >>> f 
150      x + 2*y + 3*z 
151      >>> type(_) 
152      <class 'sympy.polys.rings.PolyElement'> 
153   
154      """
155      single = False
156  
157      if not is_sequence(exprs):
158          exprs, single = [exprs], True
159  
160      exprs = list(map(sympify, exprs))
161      opt = build_options(symbols, options)
162  
163      # TODO: rewrite this so that it doesn't use expand() (see poly()).
164      reps, opt = _parallel_dict_from_expr(exprs, opt)
165  
166      if opt.domain is None:
167          coeffs = sum([ list(rep.values()) for rep in reps ], [])
168  
169          opt.domain, coeffs_dom = construct_domain(coeffs, opt=opt)
170  
171          coeff_map = dict(zip(coeffs, coeffs_dom))
172          reps = [{m: coeff_map[c] for m, c in rep.items()} for rep in reps]
173  
174      _ring = PolyRing(opt.gens, opt.domain, opt.order)
175      polys = list(map(_ring.from_dict, reps))
176  
177      if single:
178          return (_ring, polys[0])
179      else:
180          return (_ring, polys)

182  def _parse_symbols(symbols):
183      if isinstance(symbols, str):
184          return _symbols(symbols, seq=True) if symbols else ()
185      elif isinstance(symbols, Expr):
186          return (symbols,)
187      elif is_sequence(symbols):
188          if all(isinstance(s, str) for s in symbols):
189              return _symbols(symbols)
190          elif all(isinstance(s, Expr) for s in symbols):
191              return symbols
192  
193      raise GeneratorsError("expected a string, Symbol or expression or a non-empty sequence of strings, Symbols or expressions")

196  class PolyRing(DefaultPrinting, IPolys):
197      """Multivariate distributed polynomial ring. """
198  
199      gens: tuple[PolyElement, ...]
200      symbols: tuple[Expr, ...]
201      ngens: int
202      domain: Domain
203      order: MonomialOrder
204  
205      def __new__(cls, symbols, domain, order=lex):
206          symbols = tuple(_parse_symbols(symbols))
207          ngens = len(symbols)
208          domain = DomainOpt.preprocess(domain)
209          order = OrderOpt.preprocess(order)
210  
211          _hash_tuple = (cls.__name__, symbols, ngens, domain, order)
212  
213          if domain.is_Composite and set(symbols) & set(domain.symbols):
214              raise GeneratorsError("polynomial ring and it's ground domain share generators")
215  
216          obj = object.__new__(cls)
217          obj._hash_tuple = _hash_tuple
218          obj._hash = hash(_hash_tuple)
219          obj.symbols = symbols
220          obj.ngens = ngens
221          obj.domain = domain
222          obj.order = order
223  
224          obj.dtype = PolyElement(obj, ()).new
225  
226          obj.zero_monom = (0,)*ngens
227          obj.gens = obj._gens()
228          obj._gens_set = set(obj.gens)
229  
230          obj._one = [(obj.zero_monom, domain.one)]
231  
232          if ngens:
233              # These expect monomials in at least one variable
234              codegen = MonomialOps(ngens)
235              obj.monomial_mul = codegen.mul()
236              obj.monomial_pow = codegen.pow()
237              obj.monomial_mulpow = codegen.mulpow()
238              obj.monomial_ldiv = codegen.ldiv()
239              obj.monomial_div = codegen.div()
240              obj.monomial_lcm = codegen.lcm()
241              obj.monomial_gcd = codegen.gcd()
242          else:
243              monunit = lambda a, b: ()
244              obj.monomial_mul = monunit
245              obj.monomial_pow = monunit
246              obj.monomial_mulpow = lambda a, b, c: ()
247              obj.monomial_ldiv = monunit
248              obj.monomial_div = monunit
249              obj.monomial_lcm = monunit
250              obj.monomial_gcd = monunit
251  
252  
253          if order is lex:
254              obj.leading_expv = max
255          else:
256              obj.leading_expv = lambda f: max(f, key=order)
257  
258          for symbol, generator in zip(obj.symbols, obj.gens):
259              if isinstance(symbol, Symbol):
260                  name = symbol.name
261  
262                  if not hasattr(obj, name):
263                      setattr(obj, name, generator)
264  
265          return obj

267      def _gens(self):
268          """Return a list of polynomial generators. """
269          one = self.domain.one
270          _gens = []
271          for i in range(self.ngens):
272              expv = self.monomial_basis(i)
273              poly = self.zero
274              poly[expv] = one
275              _gens.append(poly)
276          return tuple(_gens)

343      def ring_new(self, element):
344          if isinstance(element, PolyElement):
345              if self == element.ring:
346                  return element
347              elif isinstance(self.domain, PolynomialRing) and self.domain.ring == element.ring:
348                  return self.ground_new(element)
349              else:
350                  raise NotImplementedError("conversion")
351          elif isinstance(element, str):
352              raise NotImplementedError("parsing")
353          elif isinstance(element, dict):
354              return self.from_dict(element)
355          elif isinstance(element, list):
356              try:
357                  return self.from_terms(element)
358              except ValueError:
359                  return self.from_list(element)
360          elif isinstance(element, Expr):
361              return self.from_expr(element)
362          else:
363              return self.ground_new(element)

384      def _rebuild_expr(self, expr, mapping):
385          domain = self.domain
386  
387          def _rebuild(expr):
388              generator = mapping.get(expr)
389  
390              if generator is not None:
391                  return generator
392              elif expr.is_Add:
393                  return reduce(add, list(map(_rebuild, expr.args)))
394              elif expr.is_Mul:
395                  return reduce(mul, list(map(_rebuild, expr.args)))
396              else:
397                  # XXX: Use as_base_exp() to handle Pow(x, n) and also exp(n)
398                  # XXX: E can be a generator e.g. sring([exp(2)]) -> ZZ[E]
399                  base, exp = expr.as_base_exp()
400                  if exp.is_Integer and exp > 1:
401                      return _rebuild(base)**int(exp)
402                  else:
403                      return self.ground_new(domain.convert(expr))
404  
405          return _rebuild(sympify(expr))

407      def from_expr(self, expr):
408          mapping = dict(list(zip(self.symbols, self.gens)))
409  
410          try:
411              poly = self._rebuild_expr(expr, mapping)
412          except CoercionFailed:
413              raise ValueError("expected an expression convertible to a polynomial in %s, got %s" % (self, expr))
414          else:
415              return self.ring_new(poly)

417      def index(self, gen):
418          """Compute index of ``gen`` in ``self.gens``. """
419          if gen is None:
420              if self.ngens:
421                  i = 0
422              else:
423                  i = -1  # indicate impossible choice
424          elif isinstance(gen, int):
425              i = gen
426  
427              if 0 <= i and i < self.ngens:
428                  pass
429              elif -self.ngens <= i and i <= -1:
430                  i = -i - 1
431              else:
432                  raise ValueError("invalid generator index: %s" % gen)
433          elif self.is_element(gen):
434              try:
435                  i = self.gens.index(gen)
436              except ValueError:
437                  raise ValueError("invalid generator: %s" % gen)
438          elif isinstance(gen, str):
439              try:
440                  i = self.symbols.index(gen)
441              except ValueError:
442                  raise ValueError("invalid generator: %s" % gen)
443          else:
444              raise ValueError("expected a polynomial generator, an integer, a string or None, got %s" % gen)
445  
446          return i

488      def add(self, *objs):
489          """ 
490          Add a sequence of polynomials or containers of polynomials. 
491   
492          Examples 
493          ======== 
494   
495          >>> from sympy.polys.rings import ring 
496          >>> from sympy.polys.domains import ZZ 
497   
498          >>> R, x = ring("x", ZZ) 
499          >>> R.add([ x**2 + 2*i + 3 for i in range(4) ]) 
500          4*x**2 + 24 
501          >>> _.factor_list() 
502          (4, [(x**2 + 6, 1)]) 
503   
504          """
505          p = self.zero
506  
507          for obj in objs:
508              if is_sequence(obj, include=GeneratorType):
509                  p += self.add(*obj)
510              else:
511                  p += obj
512  
513          return p

515      def mul(self, *objs):
516          """ 
517          Multiply a sequence of polynomials or containers of polynomials. 
518   
519          Examples 
520          ======== 
521   
522          >>> from sympy.polys.rings import ring 
523          >>> from sympy.polys.domains import ZZ 
524   
525          >>> R, x = ring("x", ZZ) 
526          >>> R.mul([ x**2 + 2*i + 3 for i in range(4) ]) 
527          x**8 + 24*x**6 + 206*x**4 + 744*x**2 + 945 
528          >>> _.factor_list() 
529          (1, [(x**2 + 3, 1), (x**2 + 5, 1), (x**2 + 7, 1), (x**2 + 9, 1)]) 
530   
531          """
532          p = self.one
533  
534          for obj in objs:
535              if is_sequence(obj, include=GeneratorType):
536                  p *= self.mul(*obj)
537              else:

556      def compose(self, other):
557          """Add the generators of ``other`` to ``self``"""
558          if self != other:
559              syms = set(self.symbols).union(set(other.symbols))
560              return self.clone(symbols=list(syms))
561          else:
562              return self

569      def symmetric_poly(self, n):
570          """ 
571          Return the elementary symmetric polynomial of degree *n* over 
572          this ring's generators. 
573          """
574          if n < 0 or n > self.ngens:
575              raise ValueError("Cannot generate symmetric polynomial of order %s for %s" % (n, self.gens))
576          elif not n:
577              return self.one
578          else:
579              poly = self.zero
580              for s in subsets(range(self.ngens), int(n)):
581                  monom = tuple(int(i in s) for i in range(self.ngens))
582                  poly += self.term_new(monom, self.domain.one)
583              return poly

586  class PolyElement(DomainElement, DefaultPrinting, CantSympify, dict):
587      """Element of multivariate distributed polynomial ring. """
588  
589      def __init__(self, ring, init):
590          super().__init__(init)
591          self.ring = ring

595      def _check(self):
596          assert isinstance(self, PolyElement)
597          assert isinstance(self.ring, PolyRing)
598          dom = self.ring.domain
599          assert isinstance(dom, Domain)
600          for monom, coeff in self.items():
601              assert dom.of_type(coeff)
602              assert len(monom) == self.ring.ngens
603              assert all(isinstance(exp, int) and exp >= 0 for exp in monom)

618          # protect dictionary from being changed so any use site modifications
619          # will make hashing go wrong. Use this feature with caution until we
620          # figure out how to make a safe API without compromising speed of this
621          # low-level class.

627      def copy(self):
628          """Return a copy of polynomial self. 
629   
630          Polynomials are mutable; if one is interested in preserving 
631          a polynomial, and one plans to use inplace operations, one 
632          can copy the polynomial. This method makes a shallow copy. 
633   
634          Examples 
635          ======== 
636   
637          >>> from sympy.polys.domains import ZZ 
638          >>> from sympy.polys.rings import ring 
639   
640          >>> R, x, y = ring('x, y', ZZ) 
641          >>> p = (x + y)**2 
642          >>> p1 = p.copy() 
643          >>> p2 = p 
644          >>> p[R.zero_monom] = 3 
645          >>> p 
646          x**2 + 2*x*y + y**2 + 3 
647          >>> p1 
648          x**2 + 2*x*y + y**2 
649          >>> p2 
650          x**2 + 2*x*y + y**2 + 3 
651   
652          """
653          return self.new(self)

655      def set_ring(self, new_ring):
656          if self.ring == new_ring:
657              return self
658          elif self.ring.symbols != new_ring.symbols:
659              terms = list(zip(*_dict_reorder(self, self.ring.symbols, new_ring.symbols)))
660              return new_ring.from_terms(terms, self.ring.domain)
661          else:
662              return new_ring.from_dict(self, self.ring.domain)

664      def as_expr(self, *symbols):
665          if not symbols:
666              symbols = self.ring.symbols
667          elif len(symbols) != self.ring.ngens:
668              raise ValueError(
669                  "Wrong number of symbols, expected %s got %s" %
670                  (self.ring.ngens, len(symbols))
671              )
672  
673          return expr_from_dict(self.as_expr_dict(), *symbols)

675      def as_expr_dict(self):
676          to_sympy = self.ring.domain.to_sympy
677          return {monom: to_sympy(coeff) for monom, coeff in self.iterterms()}

679      def clear_denoms(self):
680          domain = self.ring.domain
681  
682          if not domain.is_Field or not domain.has_assoc_Ring:
683              return domain.one, self
684  
685          ground_ring = domain.get_ring()
686          common = ground_ring.one
687          lcm = ground_ring.lcm
688          denom = domain.denom
689  
690          for coeff in self.values():
691              common = lcm(common, denom(coeff))
692  
693          poly = self.new([ (k, v*common) for k, v in self.items() ])
694          return common, poly

696      def strip_zero(self):
697          """Eliminate monomials with zero coefficient. """
698          for k, v in list(self.items()):

836      def str(self, printer, precedence, exp_pattern, mul_symbol):
837          if not self:
838              return printer._print(self.ring.domain.zero)
839          prec_mul = precedence["Mul"]
840          prec_atom = precedence["Atom"]
841          ring = self.ring
842          symbols = ring.symbols
843          ngens = ring.ngens
844          zm = ring.zero_monom
845          sexpvs = []
846          for expv, coeff in self.terms():
847              negative = ring.domain.is_negative(coeff)
848              sign = " - " if negative else " + "
849              sexpvs.append(sign)
850              if expv == zm:
851                  scoeff = printer._print(coeff)
852                  if negative and scoeff.startswith("-"):
853                      scoeff = scoeff[1:]
854              else:
855                  if negative:
856                      coeff = -coeff
857                  if coeff != self.ring.domain.one:
858                      scoeff = printer.parenthesize(coeff, prec_mul, strict=True)
859                  else:
860                      scoeff = ''
861              sexpv = []
862              for i in range(ngens):
863                  exp = expv[i]
864                  if not exp:
865                      continue
866                  symbol = printer.parenthesize(symbols[i], prec_atom, strict=True)
867                  if exp != 1:
868                      if exp != int(exp) or exp < 0:
869                          sexp = printer.parenthesize(exp, prec_atom, strict=False)
870                      else:
871                          sexp = exp
872                      sexpv.append(exp_pattern % (symbol, sexp))
873                  else:
874                      sexpv.append('%s' % symbol)
875              if scoeff:
876                  sexpv = [scoeff] + sexpv
877              sexpvs.append(mul_symbol.join(sexpv))
878          if sexpvs[0] in [" + ", " - "]:
879              head = sexpvs.pop(0)
880              if head == " - ":
881                  sexpvs.insert(0, "-")
882          return "".join(sexpvs)

1495     def div(self, fv):
1496         """Division algorithm, see [CLO] p64. 
1497  
1498         fv array of polynomials 
1499            return qv, r such that 
1500            self = sum(fv[i]*qv[i]) + r 
1501  
1502         All polynomials are required not to be Laurent polynomials. 
1503  
1504         Examples 
1505         ======== 
1506  
1507         >>> from sympy.polys.rings import ring 
1508         >>> from sympy.polys.domains import ZZ 
1509  
1510         >>> _, x, y = ring('x, y', ZZ) 
1511         >>> f = x**3 
1512         >>> f0 = x - y**2 
1513         >>> f1 = x - y 
1514         >>> qv, r = f.div((f0, f1)) 
1515         >>> qv[0] 
1516         x**2 + x*y**2 + y**4 
1517         >>> qv[1] 
1518         0 
1519         >>> r 
1520         y**6 
1521  
1522         """
1523         ring = self.ring
1524         ret_single = False
1525         if isinstance(fv, PolyElement):
1526             ret_single = True
1527             fv = [fv]
1528         if not all(fv):
1529             raise ZeroDivisionError("polynomial division")
1530         if not self:
1531             if ret_single:
1532                 return ring.zero, ring.zero
1533             else:
1534                 return [], ring.zero
1535         for f in fv:
1536             if f.ring != ring:
1537                 raise ValueError('self and f must have the same ring')
1538         s = len(fv)
1539         qv = [ring.zero for i in range(s)]
1540         p = self.copy()
1541         r = ring.zero
1542         term_div = self._term_div()
1543         expvs = [fx.leading_expv() for fx in fv]
1544         while p:
1545             i = 0
1546             divoccurred = 0
1547             while i < s and divoccurred == 0:
1548                 expv = p.leading_expv()
1549                 term = term_div((expv, p[expv]), (expvs[i], fv[i][expvs[i]]))
1550                 if term is not None:
1551                     expv1, c = term
1552                     qv[i] = qv[i]._iadd_monom((expv1, c))
1553                     p = p._iadd_poly_monom(fv[i], (expv1, -c))
1554                     divoccurred = 1
1555                 else:
1556                     i += 1
1557             if not divoccurred:
1558                 expv =  p.leading_expv()
1559                 r = r._iadd_monom((expv, p[expv]))
1560                 del p[expv]
1561         if expv == ring.zero_monom:
1562             r += p
1563         if ret_single:
1564             if not qv:
1565                 return ring.zero, r
1566             else:
1567                 return qv[0], r
1568         else:
1569             return qv, r

1571     def rem(self, G):
1572         f = self
1573         if isinstance(G, PolyElement):
1574             G = [G]
1575         if not all(G):
1576             raise ZeroDivisionError("polynomial division")
1577         ring = f.ring

1579         zero = domain.zero
1580         monomial_mul = ring.monomial_mul
1581         r = ring.zero
1582         term_div = f._term_div()

1627     def _iadd_monom(self, mc):
1628         """add to self the monomial coeff*x0**i0*x1**i1*... 
1629         unless self is a generator -- then just return the sum of the two. 
1630  
1631         mc is a tuple, (monom, coeff), where monomial is (i0, i1, ...) 
1632  
1633         Examples 
1634         ======== 
1635  
1636         >>> from sympy.polys.rings import ring 
1637         >>> from sympy.polys.domains import ZZ 
1638  
1639         >>> _, x, y = ring('x, y', ZZ) 
1640         >>> p = x**4 + 2*y 
1641         >>> m = (1, 2) 
1642         >>> p1 = p._iadd_monom((m, 5)) 
1643         >>> p1 
1644         x**4 + 5*x*y**2 + 2*y 
1645         >>> p1 is p 
1646         True 
1647         >>> p = x 
1648         >>> p1 = p._iadd_monom((m, 5)) 
1649         >>> p1 
1650         5*x*y**2 + x 
1651         >>> p1 is p 
1652         False 
1653  
1654         """
1655         if self in self.ring._gens_set:
1656             cpself = self.copy()
1657         else:
1658             cpself = self
1659         expv, coeff = mc
1660         c = cpself.get(expv)
1661         if c is None:
1662             cpself[expv] = coeff
1663         else:
1664             c += coeff
1665             if c:
1666                 cpself[expv] = c
1667             else:
1668                 del cpself[expv]
1669         return cpself

1671     def _iadd_poly_monom(self, p2, mc):
1672         """add to self the product of (p)*(coeff*x0**i0*x1**i1*...) 
1673         unless self is a generator -- then just return the sum of the two. 
1674  
1675         mc is a tuple, (monom, coeff), where monomial is (i0, i1, ...) 
1676  
1677         Examples 
1678         ======== 
1679  
1680         >>> from sympy.polys.rings import ring 
1681         >>> from sympy.polys.domains import ZZ 
1682  
1683         >>> _, x, y, z = ring('x, y, z', ZZ) 
1684         >>> p1 = x**4 + 2*y 
1685         >>> p2 = y + z 
1686         >>> m = (1, 2, 3) 
1687         >>> p1 = p1._iadd_poly_monom(p2, (m, 3)) 
1688         >>> p1 
1689         x**4 + 3*x*y**3*z**3 + 3*x*y**2*z**4 + 2*y 
1690  
1691         """
1692         p1 = self
1693         if p1 in p1.ring._gens_set:
1694             p1 = p1.copy()
1695         (m, c) = mc
1696         get = p1.get
1697         zero = p1.ring.domain.zero
1698         monomial_mul = p1.ring.monomial_mul
1699         for k, v in p2.items():
1700             ka = monomial_mul(k, m)
1701             coeff = get(ka, zero) + v*c
1702             if coeff:
1703                 p1[ka] = coeff
1704             else:
1705                 del p1[ka]
1706         return p1

2476     def symmetrize(self):
2477         r""" 
2478         Rewrite *self* in terms of elementary symmetric polynomials. 
2479  
2480         Explanation 
2481         =========== 
2482  
2483         If this :py:class:`~.PolyElement` belongs to a ring of $n$ variables, 
2484         we can try to write it as a function of the elementary symmetric 
2485         polynomials on $n$ variables. We compute a symmetric part, and a 
2486         remainder for any part we were not able to symmetrize. 
2487  
2488         Examples 
2489         ======== 
2490  
2491         >>> from sympy.polys.rings import ring 
2492         >>> from sympy.polys.domains import ZZ 
2493         >>> R, x, y = ring("x,y", ZZ) 
2494  
2495         >>> f = x**2 + y**2 
2496         >>> f.symmetrize() 
2497         (x**2 - 2*y, 0, [(x, x + y), (y, x*y)]) 
2498  
2499         >>> f = x**2 - y**2 
2500         >>> f.symmetrize() 
2501         (x**2 - 2*y, -2*y**2, [(x, x + y), (y, x*y)]) 
2502  
2503         Returns 
2504         ======= 
2505  
2506         Triple ``(p, r, m)`` 
2507             ``p`` is a :py:class:`~.PolyElement` that represents our attempt 
2508             to express *self* as a function of elementary symmetric 
2509             polynomials. Each variable in ``p`` stands for one of the 
2510             elementary symmetric polynomials. The correspondence is given 
2511             by ``m``. 
2512  
2513             ``r`` is the remainder. 
2514  
2515             ``m`` is a list of pairs, giving the mapping from variables in 
2516             ``p`` to elementary symmetric polynomials. 
2517  
2518             The triple satisfies the equation ``p.compose(m) + r == self``. 
2519             If the remainder ``r`` is zero, *self* is symmetric. If it is 
2520             nonzero, we were not able to represent *self* as symmetric. 
2521  
2522         See Also 
2523         ======== 
2524  
2525         sympy.polys.polyfuncs.symmetrize 
2526  
2527         References 
2528         ========== 
2529  
2530         .. [1] Lauer, E. Algorithms for symmetrical polynomials, Proc. 1976 
2531             ACM Symp. on Symbolic and Algebraic Computing, NY 242-247. 
2532             https://dl.acm.org/doi/pdf/10.1145/800205.806342 
2533  
2534         """
2535         f = self.copy()
2536         ring = f.ring
2537         n = ring.ngens
2538 
2539         if not n:
2540             return f, ring.zero, []
2541 
2542         polys = [ring.symmetric_poly(i+1) for i in range(n)]
2543 
2544         poly_powers = {}
2545         def get_poly_power(i, n):
2546             if (i, n) not in poly_powers:
2547                 poly_powers[(i, n)] = polys[i]**n
2548             return poly_powers[(i, n)]
2549 
2550         indices = list(range(n - 1))
2551         weights = list(range(n, 0, -1))
2552 
2553         symmetric = ring.zero
2554 
2555         while f:
2556             _height, _monom, _coeff = -1, None, None
2557 
2558             for i, (monom, coeff) in enumerate(f.terms()):
2559                 if all(monom[i] >= monom[i + 1] for i in indices):
2560                     height = max(n*m for n, m in zip(weights, monom))
2561 
2562                     if height > _height:
2563                         _height, _monom, _coeff = height, monom, coeff
2564 
2565             if _height != -1:
2566                 monom, coeff = _monom, _coeff
2567             else:
2568                 break
2569 
2570             exponents = []
2571             for m1, m2 in zip(monom, monom[1:] + (0,)):
2572                 exponents.append(m1 - m2)
2573 
2574             symmetric += ring.term_new(tuple(exponents), coeff)
2575 
2576             product = coeff
2577             for i, n in enumerate(exponents):
2578                 product *= get_poly_power(i, n)
2579             f -= product
2580 
2581         mapping = list(zip(ring.gens, polys))
2582 
2583         return symmetric, f, mapping

2670     def prem(self, g, x=None):
2671         """ 
2672         Pseudo-remainder of the polynomial ``self`` with respect to ``g``. 
2673  
2674         The pseudo-quotient ``q`` and pseudo-remainder ``r`` with respect to 
2675         ``z`` when dividing ``f`` by ``g`` satisfy ``m*f = g*q + r``, 
2676         where ``deg(r,z) < deg(g,z)`` and 
2677         ``m = LC(g,z)**(deg(f,z) - deg(g,z)+1)``. 
2678  
2679         See :meth:`pdiv` for explanation of pseudo-division. 
2680  
2681  
2682         Parameters 
2683         ========== 
2684  
2685         g : :py:class:`~.PolyElement` 
2686             The polynomial to divide ``self`` by. 
2687         x : generator or generator index, optional 
2688             The main variable of the polynomials and default is first generator. 
2689  
2690         Returns 
2691         ======= 
2692  
2693         :py:class:`~.PolyElement` 
2694             The pseudo-remainder polynomial. 
2695  
2696         Raises 
2697         ====== 
2698  
2699         ZeroDivisionError : If ``g`` is the zero polynomial. 
2700  
2701         Examples 
2702         ======== 
2703  
2704         >>> from sympy.polys import ring, ZZ 
2705         >>> R, x, y = ring("x, y", ZZ) 
2706  
2707         >>> f = x**2 + x*y 
2708         >>> g = 2*x + 2 
2709         >>> f.prem(g) # first generator is chosen by default if it is not given 
2710         -4*y + 4 
2711         >>> f.rem(g) # shows the difference between prem and rem 
2712         x**2 + x*y 
2713         >>> f.prem(g, y) # generator is given 
2714         0 
2715         >>> f.prem(g, 1) # generator index is given 
2716         0 
2717  
2718         See Also 
2719         ======== 
2720  
2721         pdiv, pquo, pexquo, sympy.polys.domains.ring.Ring.rem 
2722  
2723         """
2724         f = self
2725         x = f.ring.index(x)
2726         df = f.degree(x)
2727         dg = g.degree(x)
2728 
2729         if dg < 0:
2730             raise ZeroDivisionError('polynomial division')
2731 
2732         r, dr = f, df
2733 
2734         if df < dg:
2735             return r
2736 
2737         N = df - dg + 1
2738 
2739         lc_g = g.coeff_wrt(x, dg)
2740 
2741         xp = f.ring.gens[x]
2742 
2743         while True:
2744 
2745             lc_r = r.coeff_wrt(x, dr)
2746             j, N = dr - dg, N - 1
2747 
2748             R = r * lc_g
2749             G = g * lc_r * xp**j
2750             r = R - G
2751 
2752             dr = r.degree(x)
2753 
2754             if dr < dg:
2755                 break
2756 
2757         c = lc_g ** N
2758 
2759         return r * c

2761     def pdiv(self, g, x=None):
2762         """ 
2763         Computes the pseudo-division of the polynomial ``self`` with respect to ``g``. 
2764  
2765         The pseudo-division algorithm is used to find the pseudo-quotient ``q`` 
2766         and pseudo-remainder ``r`` such that ``m*f = g*q + r``, where ``m`` 
2767         represents the multiplier and ``f`` is the dividend polynomial. 
2768  
2769         The pseudo-quotient ``q`` and pseudo-remainder ``r`` are polynomials in 
2770         the variable ``x``, with the degree of ``r`` with respect to ``x`` 
2771         being strictly less than the degree of ``g`` with respect to ``x``. 
2772  
2773         The multiplier ``m`` is defined as 
2774         ``LC(g, x) ^ (deg(f, x) - deg(g, x) + 1)``, 
2775         where ``LC(g, x)`` represents the leading coefficient of ``g``. 
2776  
2777         It is important to note that in the context of the ``prem`` method, 
2778         multivariate polynomials in a ring, such as ``R[x,y,z]``, are treated 
2779         as univariate polynomials with coefficients that are polynomials, 
2780         such as ``R[x,y][z]``. When dividing ``f`` by ``g`` with respect to the 
2781         variable ``z``, the pseudo-quotient ``q`` and pseudo-remainder ``r`` 
2782         satisfy ``m*f = g*q + r``, where ``deg(r, z) < deg(g, z)`` 
2783         and ``m = LC(g, z)^(deg(f, z) - deg(g, z) + 1)``. 
2784  
2785         In this function, the pseudo-remainder ``r`` can be obtained using the 
2786         ``prem`` method, the pseudo-quotient ``q`` can 
2787         be obtained using the ``pquo`` method, and 
2788         the function ``pdiv`` itself returns a tuple ``(q, r)``. 
2789  
2790  
2791         Parameters 
2792         ========== 
2793  
2794         g : :py:class:`~.PolyElement` 
2795             The polynomial to divide ``self`` by. 
2796         x : generator or generator index, optional 
2797             The main variable of the polynomials and default is first generator. 
2798  
2799         Returns 
2800         ======= 
2801  
2802         :py:class:`~.PolyElement` 
2803             The pseudo-division polynomial (tuple of ``q`` and ``r``). 
2804  
2805         Raises 
2806         ====== 
2807  
2808         ZeroDivisionError : If ``g`` is the zero polynomial. 
2809  
2810         Examples 
2811         ========
