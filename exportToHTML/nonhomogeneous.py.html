29   def _test_term(coeff, func, order):
30       r""" 
31       Linear Euler ODEs have the form  K*x**order*diff(y(x), x, order) = F(x), 
32       where K is independent of x and y(x), order>= 0. 
33       So we need to check that for each term, coeff == K*x**order from 
34       some K.  We have a few cases, since coeff may have several 
35       different types. 
36       """
37       x = func.args[0]
38       f = func.func
39       if order < 0:
40           raise ValueError("order should be greater than 0")
41       if coeff == 0:
42           return True
43       if order == 0:
44           if x in coeff.free_symbols:
45               return False
46           return True
47       if coeff.is_Mul:
48           if coeff.has(f(x)):
49               return False
50           return x**order in coeff.args
51       elif coeff.is_Pow:
52           return coeff.as_base_exp() == (x, order)
53       elif order == 1:
54           return x == coeff
55       return False

58   def _get_euler_characteristic_eq_sols(eq, func, match_obj):
59       r""" 
60       Returns the solution of homogeneous part of the linear euler ODE and 
61       the list of roots of characteristic equation. 
62    
63       The parameter ``match_obj`` is a dict of order:coeff terms, where order is the order 
64       of the derivative on each term, and coeff is the coefficient of that derivative. 
65    
66       """
67       x = func.args[0]
68       f = func.func
69   
70       # First, set up characteristic equation.
71       chareq, symbol = S.Zero, Dummy('x')
72   
73       for i in match_obj:
74           if i >= 0:
75               chareq += (match_obj[i]*diff(x**symbol, x, i)*x**-symbol).expand()
76   
77       chareq = Poly(chareq, symbol)
78       chareqroots = [rootof(chareq, k) for k in range(chareq.degree())]
79       collectterms = []
80   
81       # A generator of constants
82       constants = list(get_numbered_constants(eq, num=chareq.degree()*2))
83       constants.reverse()
84   
85       # Create a dict root: multiplicity or charroots
86       charroots = Counter(chareqroots)
87       gsol = S.Zero
88       ln = log
89       for root, multiplicity in charroots.items():
90           for i in range(multiplicity):
91               if isinstance(root, RootOf):
92                   gsol += (x**root) * constants.pop()
93                   if multiplicity != 1:
94                       raise ValueError("Value should be 1")
95                   collectterms = [(0, root, 0)] + collectterms
96               elif root.is_real:
97                   gsol += ln(x)**i*(x**root) * constants.pop()
98                   collectterms = [(i, root, 0)] + collectterms
99               else:
100                  reroot = re(root)
101                  imroot = im(root)
102                  gsol += ln(x)**i * (x**reroot) * (
103                      constants.pop() * sin(abs(imroot)*ln(x))
104                      + constants.pop() * cos(imroot*ln(x)))
105                  collectterms = [(i, reroot, imroot)] + collectterms
106  
107      gsol = Eq(f(x), gsol)
108  
109      gensols = []
110      # Keep track of when to use sin or cos for nonzero imroot
111      for i, reroot, imroot in collectterms:
112          if imroot == 0:
113              gensols.append(ln(x)**i*x**reroot)
114          else:
115              sin_form = ln(x)**i*x**reroot*sin(abs(imroot)*ln(x))
116              if sin_form in gensols:
117                  cos_form = ln(x)**i*x**reroot*cos(imroot*ln(x))
118                  gensols.append(cos_form)
119              else:
120                  gensols.append(sin_form)
121      return gsol, gensols

124  def _solve_variation_of_parameters(eq, func, roots, homogen_sol, order, match_obj, simplify_flag=True):
125      r""" 
126      Helper function for the method of variation of parameters and nonhomogeneous euler eq. 
127   
128      See the 
129      :py:meth:`~sympy.solvers.ode.single.NthLinearConstantCoeffVariationOfParameters` 
130      docstring for more information on this method. 
131   
132      The parameter are ``match_obj`` should be a dictionary that has the following 
133      keys: 
134   
135      ``list`` 
136      A list of solutions to the homogeneous equation. 
137   
138      ``sol`` 
139      The general solution. 
140   
141      """
142      f = func.func
143      x = func.args[0]
144      r = match_obj
145      psol = 0
146      wr = wronskian(roots, x)
147  
148      if simplify_flag:
149          wr = simplify(wr)  # We need much better simplification for
150                          # some ODEs. See issue 4662, for example.
151          # To reduce commonly occurring sin(x)**2 + cos(x)**2 to 1
152          wr = trigsimp(wr, deep=True, recursive=True)
153      if not wr:
154          # The wronskian will be 0 iff the solutions are not linearly
155          # independent.
156          raise NotImplementedError("Cannot find " + str(order) +
157          " solutions to the homogeneous equation necessary to apply " +
158          "variation of parameters to " + str(eq) + " (Wronskian == 0)")
159      if len(roots) != order:
160          raise NotImplementedError("Cannot find " + str(order) +
161          " solutions to the homogeneous equation necessary to apply " +
162          "variation of parameters to " +
163          str(eq) + " (number of terms != order)")
164      negoneterm = S.NegativeOne**(order)
165      for i in roots:
166          psol += negoneterm*Integral(wronskian([sol for sol in roots if sol != i], x)*r[-1]/wr, x)*i/r[order]
167          negoneterm *= -1
168  
169      if simplify_flag:
170          psol = simplify(psol)
171          psol = trigsimp(psol, deep=True)
172      return Eq(f(x), homogen_sol.rhs + psol)

175  def _get_const_characteristic_eq_sols(r, func, order):
176      r""" 
177      Returns the roots of characteristic equation of constant coefficient 
178      linear ODE and list of collectterms which is later on used by simplification 
179      to use collect on solution. 
180   
181      The parameter `r` is a dict of order:coeff terms, where order is the order of the 
182      derivative on each term, and coeff is the coefficient of that derivative. 
183   
184      """
185      x = func.args[0]
186      # First, set up characteristic equation.
187      chareq, symbol = S.Zero, Dummy('x')
188  
189      for i in r.keys():
190          if isinstance(i, str) or i < 0:
191              pass
192          else:
193              chareq += r[i]*symbol**i
194  
195      chareq = Poly(chareq, symbol)
196      # Can't just call roots because it doesn't return rootof for unsolveable
197      # polynomials.
198      chareqroots = roots(chareq, multiple=True)
199      if len(chareqroots) != order:
200          chareqroots = [rootof(chareq, k) for k in range(chareq.degree())]
201  
202      chareq_is_complex = not all(i.is_real for i in chareq.all_coeffs())
203  
204      # Create a dict root: multiplicity or charroots
205      charroots = Counter(chareqroots)
206      # We need to keep track of terms so we can run collect() at the end.
207      # This is necessary for constantsimp to work properly.
208      collectterms = []
209      gensols = []
210      conjugate_roots = [] # used to prevent double-use of conjugate roots
211      # Loop over roots in theorder provided by roots/rootof...
212      for root in chareqroots:
213          # but don't repoeat multiple roots.
214          if root not in charroots:
215              continue
216          multiplicity = charroots.pop(root)
217          for i in range(multiplicity):
218              if chareq_is_complex:
219                  gensols.append(x**i*exp(root*x))
220                  collectterms = [(i, root, 0)] + collectterms
221                  continue
222              reroot = re(root)
223              imroot = im(root)
224              if imroot.has(atan2) and reroot.has(atan2):
225                  # Remove this condition when re and im stop returning
226                  # circular atan2 usages.
227                  gensols.append(x**i*exp(root*x))
228                  collectterms = [(i, root, 0)] + collectterms
229              else:
230                  if root in conjugate_roots:
231                      collectterms = [(i, reroot, imroot)] + collectterms
232                      continue
233                  if imroot == 0:
234                      gensols.append(x**i*exp(reroot*x))
235                      collectterms = [(i, reroot, 0)] + collectterms
236                      continue
237                  conjugate_roots.append(conjugate(root))
238                  gensols.append(x**i*exp(reroot*x) * sin(abs(imroot) * x))
239                  gensols.append(x**i*exp(reroot*x) * cos(    imroot  * x))
240  
241                  # This ordering is important
242                  collectterms = [(i, reroot, imroot)] + collectterms
243      return gensols, collectterms

248  def _get_simplified_sol(sol, func, collectterms):
249      r""" 
250      Helper function which collects the solution on 
251      collectterms. Ideally this should be handled by odesimp.It is used 
252      only when the simplify is set to True in dsolve. 
253   
254      The parameter ``collectterms`` is a list of tuple (i, reroot, imroot) where `i` is 
255      the multiplicity of the root, reroot is real part and imroot being the imaginary part. 
256   
257      """
258      f = func.func
259      x = func.args[0]
260      collectterms.sort(key=default_sort_key)
261      collectterms.reverse()
262      assert len(sol) == 1 and sol[0].lhs == f(x)
263      sol = sol[0].rhs
264      sol = expand_mul(sol)
265      for i, reroot, imroot in collectterms:
266          sol = collect(sol, x**i*exp(reroot*x)*sin(abs(imroot)*x))
267          sol = collect(sol, x**i*exp(reroot*x)*cos(imroot*x))
268      for i, reroot, imroot in collectterms:
269          sol = collect(sol, x**i*exp(reroot*x))
270      sol = powsimp(sol)
271      return Eq(f(x), sol)

274  def _undetermined_coefficients_match(expr, x, func=None, eq_homogeneous=S.Zero):
275      r""" 
276      Returns a trial function match if undetermined coefficients can be applied 
277      to ``expr``, and ``None`` otherwise. 
278   
279      A trial expression can be found for an expression for use with the method 
280      of undetermined coefficients if the expression is an 
281      additive/multiplicative combination of constants, polynomials in `x` (the 
282      independent variable of expr), `\sin(a x + b)`, `\cos(a x + b)`, and 
283      `e^{a x}` terms (in other words, it has a finite number of linearly 
284      independent derivatives). 
285   
286      Note that you may still need to multiply each term returned here by 
287      sufficient `x` to make it linearly independent with the solutions to the 
288      homogeneous equation. 
289   
290      This is intended for internal use by ``undetermined_coefficients`` hints. 
291   
292      SymPy currently has no way to convert `\sin^n(x) \cos^m(y)` into a sum of 
293      only `\sin(a x)` and `\cos(b x)` terms, so these are not implemented.  So, 
294      for example, you will need to manually convert `\sin^2(x)` into `[1 + 
295      \cos(2 x)]/2` to properly apply the method of undetermined coefficients on 
296      it. 
297   
298      Examples 
299      ======== 
300   
301      >>> from sympy import log, exp 
302      >>> from sympy.solvers.ode.nonhomogeneous import _undetermined_coefficients_match 
303      >>> from sympy.abc import x 
304      >>> _undetermined_coefficients_match(9*x*exp(x) + exp(-x), x) 
305      {'test': True, 'trialset': {x*exp(x), exp(-x), exp(x)}} 
306      >>> _undetermined_coefficients_match(log(x), x) 
307      {'test': False} 
308   
309      """
310      a = Wild('a', exclude=[x])
311      b = Wild('b', exclude=[x])
312      expr = powsimp(expr, combine='exp')  # exp(x)*exp(2*x + 1) => exp(3*x + 1)
313      retdict = {}
314  
315      def _test_term(expr, x) -> bool:
316          r""" 
317          Test if ``expr`` fits the proper form for undetermined coefficients. 
318          """
319          if not expr.has(x):
320              return True
321          if expr.is_Add:
322              return all(_test_term(i, x) for i in expr.args)
323          if expr.is_Mul:
324              if expr.has(sin, cos):
325                  foundtrig = False
326                  # Make sure that there is only one trig function in the args.
327                  # See the docstring.
328                  for i in expr.args:
329                      if i.has(sin, cos):
330                          if foundtrig:
331                              return False
332                          else:
333                              foundtrig = True
334              return all(_test_term(i, x) for i in expr.args)
335          if expr.is_Function:
336              return expr.func in (sin, cos, exp, sinh, cosh) and \
337                     bool(expr.args[0].match(a*x + b))
338          if expr.is_Pow and expr.base.is_Symbol and expr.exp.is_Integer and \
339                  expr.exp >= 0:
340              return True
341          if expr.is_Pow and expr.base.is_number:
342              return bool(expr.exp.match(a*x + b))
343          return expr.is_Symbol or bool(expr.is_number)
344  
345      def _get_trial_set(expr, x, exprs=set()):
346          r""" 
347          Returns a set of trial terms for undetermined coefficients. 
348   
349          The idea behind undetermined coefficients is that the terms expression 
350          repeat themselves after a finite number of derivatives, except for the 
351          coefficients (they are linearly dependent).  So if we collect these, 
352          we should have the terms of our trial function. 
353          """
354          def _remove_coefficient(expr, x):
355              r""" 
356              Returns the expression without a coefficient. 
357   
358              Similar to expr.as_independent(x)[1], except it only works 
359              multiplicatively. 
360              """
361              term = S.One
362              if expr.is_Mul:
363                  for i in expr.args:
364                      if i.has(x):
365                          term *= i
366              elif expr.has(x):
367                  term = expr
368              return term
369  
370          expr = expand_mul(expr)
371          if expr.is_Add:
372              for term in expr.args:
373                  if _remove_coefficient(term, x) in exprs:
374                      pass
375                  else:
376                      exprs.add(_remove_coefficient(term, x))
377                      exprs = exprs.union(_get_trial_set(term, x, exprs))
378          else:
379              term = _remove_coefficient(expr, x)
380              tmpset = exprs.union({term})
381              oldset = set()
382              while tmpset != oldset:
383                  # If you get stuck in this loop, then _test_term is probably
384                  # broken
385                  oldset = tmpset.copy()
386                  expr = expr.diff(x)
387                  term = _remove_coefficient(expr, x)
388                  if term.is_Add:
389                      tmpset = tmpset.union(_get_trial_set(term, x, tmpset))
390                  else:
391                      tmpset.add(term)
392              exprs = tmpset
393          return exprs
394  
395      def is_homogeneous_solution(term):
396          r""" This function checks whether the given trialset contains any root 
397              of homogeneous equation"""
398          return expand(sub_func_doit(eq_homogeneous, func, term)).is_zero
399  
400      retdict['test'] = _test_term(expr, x)
401      if retdict['test']:
402          # Try to generate a list of trial solutions that will have the
403          # undetermined coefficients. Note that if any of these are not linearly
404          # independent with any of the solutions to the homogeneous equation,
405          # then they will need to be multiplied by sufficient x to make them so.
406          # This function DOES NOT do that (it doesn't even look at the
407          # homogeneous equation).
408          temp_set = set()
409          for i in Add.make_args(expr):
410              act = _get_trial_set(i, x)
411              if eq_homogeneous is not S.Zero:
412                  while any(is_homogeneous_solution(ts) for ts in act):
413                      act = {x*ts for ts in act}
414              temp_set = temp_set.union(act)
415  
416          retdict['trialset'] = temp_set
417      return retdict

420  def _solve_undetermined_coefficients(eq, func, order, match, trialset):
421      r""" 
422      Helper function for the method of undetermined coefficients. 
423   
424      See the 
425      :py:meth:`~sympy.solvers.ode.single.NthLinearConstantCoeffUndeterminedCoefficients` 
426      docstring for more information on this method. 
427   
428      The parameter ``trialset`` is the set of trial functions as returned by 
429      ``_undetermined_coefficients_match()['trialset']``. 
430   
431      The parameter ``match`` should be a dictionary that has the following 
432      keys: 
433   
434      ``list`` 
435      A list of solutions to the homogeneous equation. 
436   
437      ``sol`` 
438      The general solution. 
439   
440      """
441      r = match
442      coeffs = numbered_symbols('a', cls=Dummy)
443      coefflist = []
444      gensols = r['list']
445      gsol = r['sol']
446      f = func.func
447      x = func.args[0]
448  
449      if len(gensols) != order:
450          raise NotImplementedError("Cannot find " + str(order) +
451          " solutions to the homogeneous equation necessary to apply" +
452          " undetermined coefficients to " + str(eq) +
453          " (number of terms != order)")
454  
455      trialfunc = 0
456      for i in trialset:
457          c = next(coeffs)
458          coefflist.append(c)
459          trialfunc += c*i
460  
461      eqs = sub_func_doit(eq, f(x), trialfunc)
462  
463      coeffsdict = dict(list(zip(trialset, [0]*(len(trialset) + 1))))
464  
465      eqs = _mexpand(eqs)
466  
467      for i in Add.make_args(eqs):
468          s = separatevars(i, dict=True, symbols=[x])
469          if coeffsdict.get(s[x]):
470              coeffsdict[s[x]] += s['coeff']
471          else:
472              coeffsdict[s[x]] = s['coeff']
473  
474      coeffvals = solve(list(coeffsdict.values()), coefflist)
475  
476      if not coeffvals:
477          raise NotImplementedError(
478              "Could not solve `%s` using the "
479              "method of undetermined coefficients "
480              "(unable to solve for coefficients)." % eq)
481  
482      psol = trialfunc.subs(coeffvals)
483  
484      return Eq(f(x), gsol.rhs + psol)
