19   static inline void cvtfp8e4m3_fp32(const __m128i& a, __m512& o) {
20     // Zero Extend
21     __m512i x = _mm512_cvtepu8_epi32(a);
22     __m512i val = _mm512_and_epi32(
23         _mm512_slli_epi32(x, 24), _mm512_set1_epi32(0x7FFFFFFF)); // nonsign_val
24     __m512i mant =
25         _mm512_and_si512(x, _mm512_set1_epi32(0x07)); // mantissa = x & 0x07
26     __m512i exp = _mm512_and_si512(
27         _mm512_srli_epi32(x, 3),
28         _mm512_set1_epi32(0x0F)); // exp = (x >> 3) & 0x0F
29     __m512i sign =
30         _mm512_and_si512(x, _mm512_set1_epi32(0x80)); // sign = x & 0x80
31     __m512i _zeros = _mm512_setzero_si512();
32   
33     // --- Step 1: Calculate the renorm_shift
34     __m512i renorm_shift = _zeros;
35     // Denorm case (exp == 0 && mant != 0) ---
36     __mmask16 denormal_mask = _mm512_cmpeq_epi32_mask(exp, _zeros) &
37         _mm512_cmpneq_epi32_mask(mant, _zeros);
38     if (denormal_mask) {
39       // An alternative solution is as what scalar did in
40       // pytorch/c10/util/Float8_e4m3fn.h To count the num of leading zeros, since
41       // here we know the unsigned denorm value has zero sign and exp which is 5
42       // leading zeros, we need to count the leading zero of mant (3bit) which may
43       // done through table lookup for example: const uint8_t lz_table[8] = {3, 2,
44       // 1, 1, 0, 0, 0, 0}; num_leading_zero = lz_table[mant] + 5;
45   
46       __m512i _ones = _mm512_set1_epi32(1);
47       __m512i _twos = _mm512_set1_epi32(2);
48       __m512i _threes = _mm512_set1_epi32(3);
49   
50       // Default leading zero number for denorm value is 1 = 5 - 4
51       __m512i denorm_renorm_shift = _ones;
52       // For mant 001, leading zero number is 3 = 7 -4
53       __mmask16 leading_Zero_mask = _mm512_cmpeq_epi32_mask(mant, _ones);
54       denorm_renorm_shift =
55           _mm512_mask_mov_epi32(denorm_renorm_shift, leading_Zero_mask, _threes);
56       // For mant 010 and 011, leading zero number is 2 = 6 -4
57       leading_Zero_mask = _mm512_cmpeq_epi32_mask(mant, _twos);
58       denorm_renorm_shift =
59           _mm512_mask_mov_epi32(denorm_renorm_shift, leading_Zero_mask, _twos);
60       leading_Zero_mask = _mm512_cmpeq_epi32_mask(mant, _threes);
61       denorm_renorm_shift =
62           _mm512_mask_mov_epi32(denorm_renorm_shift, leading_Zero_mask, _twos);
63   
64       renorm_shift =
65           _mm512_mask_mov_epi32(renorm_shift, denormal_mask, denorm_renorm_shift);
66     }
67   
68     // --- Step 2: calculate norm and denorm ---
69     __m512i norm_shifted =
70         _mm512_srli_epi32(_mm512_sllv_epi32(val, renorm_shift), 4);
71     // exponent bias adjustment: (0x78 - renorm_shift) << 23
72     __m512i exp_bias = _mm512_slli_epi32(
73         _mm512_sub_epi32(_mm512_set1_epi32(0x78), renorm_shift), 23);
74     val = _mm512_add_epi32(norm_shifted, exp_bias);
75   
76     // --- Step 3: Nan case (exp == 0xF && mant == 0x07) ---
77     __mmask16 nan_mask = _mm512_cmpeq_epi32_mask(exp, _mm512_set1_epi32(0xF)) &
78         _mm512_cmpeq_epi32_mask(mant, _mm512_set1_epi32(0x07));
79     if (nan_mask) {
80       const __m512i nan_values = _mm512_set1_epi32(0x7FC00000);
81       val = _mm512_mask_mov_epi32(val, nan_mask, nan_values);
82     }
83   
84     // --- Step 4: Zero case (exp == 0x00 && mant == 0x00) ---
85     __mmask16 zero_mask = _mm512_cmpeq_epi32_mask(exp, _zeros) &
86         _mm512_cmpeq_epi32_mask(mant, _zeros);
87     if (zero_mask) {
88       val = _mm512_mask_mov_epi32(val, zero_mask, _zeros);
89     }
90   
91     // --- Step 5: OR with sign (sign bit << 24 to get to bit 31) ---
92     val = _mm512_or_si512(val, _mm512_slli_epi32(sign, 24));
93   
94     o = _mm512_castsi512_ps(val);
95   }

97   static inline __m128i cvtfp32_fp8e4m3(const __m512& src) {
98     // cvt 16x32 from fp32 to fp8 e4m3
99     const __m512i sign_mask = _mm512_set1_epi32(0x80000000);
100    const __m512i fp8_max = _mm512_set1_epi32(UINT32_C(1087) << 20);
101    const __m512i denorm_thresh = _mm512_set1_epi32(UINT32_C(121) << 23);
102    const __m512i denorm_mask = _mm512_set1_epi32(UINT32_C(141) << 23);
103    const __m512i bias_part1 = _mm512_set1_epi32((uint32_t)(7 - 127) << 23);
104    const __m512i rounding_bias = _mm512_set1_epi32(0x7FFFF);
105    __m512i f_bits = _mm512_castps_si512(src);
106    // Extract and save sign
107    __m512i sign = _mm512_and_epi32(f_bits, sign_mask);
108    f_bits = _mm512_xor_epi32(f_bits, sign);
109  
110    // Prepare result containers
111    __m512i result = _mm512_setzero_si512();
112  
113    // Step 1: Handle case of overflow
114    // (f_bits >= fp8_max): set result = 0x7f
115    __mmask16 overflow_mask = _mm512_cmpge_epu32_mask(f_bits, fp8_max);
116    if (overflow_mask) {
117      result = _mm512_mask_set1_epi32(result, overflow_mask, 0x7f);
118    }
119  
120    // Step 2: Handle small numbers (denormals)
121    // Small numbers (f_bits < denorm_thresh)
122    __mmask16 denorm_thresh_mask = _mm512_cmplt_epu32_mask(f_bits, denorm_thresh);
123  
124    if (denorm_thresh_mask) {
125      __m512 small_input = _mm512_castsi512_ps(f_bits);
126      __m512 small_denorm =
127          _mm512_add_ps(small_input, _mm512_castsi512_ps(denorm_mask));
128      __m512i small_denorm_bits = _mm512_castps_si512(small_denorm);
129      __m512i small_result = _mm512_sub_epi32(small_denorm_bits, denorm_mask);
130      result = _mm512_mask_mov_epi32(result, denorm_thresh_mask, small_result);
131    }
132  
133    // Step 3: Handle normal numbers
134    __mmask16 normal_mask = ~(overflow_mask | denorm_thresh_mask);
135  
136    if (normal_mask) {
137      // mant_odd = (f_bits >> 20) & 1
138      __m512i mant_odd =
139          _mm512_and_epi32(_mm512_srli_epi32(f_bits, 20), _mm512_set1_epi32(1));
140      // f_bits += bias_part1 + rounding_bias
141      __m512i rounded = _mm512_add_epi32(f_bits, bias_part1);
142      rounded = _mm512_add_epi32(rounded, rounding_bias);
143      // Add mant_odd
144      rounded = _mm512_add_epi32(rounded, mant_odd);
145      // Shift right by 20 bits
146      __m512i normal_result = _mm512_srli_epi32(rounded, 20);
147      result = _mm512_mask_mov_epi32(result, normal_mask, normal_result);
148    }
149  
150    // Merge back the sign
151    __m512i sign_shifted = _mm512_srli_epi32(sign, 24);
152    result = _mm512_or_epi32(result, sign_shifted);
153  
154    // Now result is 16 x 32-bit integers, but we only need 8-bit for each
155    __m512i packed = _mm512_and_si512(result, _mm512_set1_epi32(0xFF));
156  
157    // Narrow 32-bit integers to 8-bit
158    return _mm512_cvtepi32_epi8(packed);
159  }

162    __m512i v = _mm512_set1_epi8(val);
163    __m128i v_128 = _mm512_castsi512_si128(v);
164    __m512 o;
165    cvtfp8e4m3_fp32(v_128, o);
166    return _mm512_cvtss_f32(o);

175  static inline void cvtfp8e5m2_fp32(const __m128i& a, __m512& o) {
176    __m256i a_256 = _mm256_castsi128_si256(a);
177    __m512i a_512 = _mm512_cvtepu8_epi16(a_256);
178    a_512 = _mm512_slli_epi16(a_512, 8);
179    a_256 = _mm512_castsi512_si256(a_512);
180    cvtfp16_fp32(a_256, o);
181  }

183  static inline __m128i cvtfp32_fp8e5m2(const __m512& src) {
184    constexpr uint32_t fp32_inf = UINT32_C(255) << 23;
185    constexpr uint32_t fp8_max = UINT32_C(143) << 23;
186    constexpr uint32_t denorm_mask = UINT32_C(134) << 23;
187  
188    // Cvt to bits
189    __m512i input_bits = _mm512_castps_si512(src);
190    __m512i result = _mm512_setzero_si512();
191  
192    // Get the sign
193    __m512i sign = _mm512_and_si512(input_bits, _mm512_set1_epi32(0x80000000));
194  
195    // Get the unsigned input
196    input_bits = _mm512_xor_si512(input_bits, sign);
197  
198    // Calculate the mask for inf, nan and denorm
199    __mmask16 greater_than_fp8_max =
200        _mm512_cmpge_epi32_mask(input_bits, _mm512_set1_epi32(fp8_max));
201    __mmask16 greater_than_fp32_inf =
202        _mm512_cmpgt_epi32_mask(input_bits, _mm512_set1_epi32(fp32_inf));
203    __mmask16 less_than_normal = _mm512_cmpgt_epi32_mask(
204        _mm512_set1_epi32((UINT32_C(113) << 23)), input_bits);
205    __m512i temp_bits_for_denorm = _mm512_setzero_si512();
206    if (less_than_normal) {
207      __m512i denorm_mask_512i = _mm512_set1_epi32(denorm_mask);
208      temp_bits_for_denorm = _mm512_castps_si512(_mm512_add_ps(
209          _mm512_castsi512_ps(input_bits),
210          _mm512_castsi512_ps(denorm_mask_512i)));
211      temp_bits_for_denorm =
212          _mm512_sub_epi32(temp_bits_for_denorm, denorm_mask_512i);
213    }
214  
215    // Step 1: Norm Val
216    __m512i mant_odd_mask =
217        _mm512_and_epi32(_mm512_srli_epi32(input_bits, 21), _mm512_set1_epi32(1));
218    input_bits = _mm512_add_epi32(
219        input_bits, _mm512_set1_epi32(((uint32_t)(15 - 127) << 23) + 0xFFFFF));
220    input_bits = _mm512_add_epi32(input_bits, mant_odd_mask);
221    result = _mm512_srli_epi32(input_bits, 21);
222  
223    // Step 2: INF and NAN
224    if (greater_than_fp8_max) {
225      result = _mm512_mask_mov_epi32(
226          result, greater_than_fp8_max, _mm512_set1_epi8(0x7C));
227      if (greater_than_fp32_inf) {
228        result = _mm512_mask_mov_epi32(
229            result, greater_than_fp32_inf, _mm512_set1_epi8(0x7F));
230      }
231    }
232  
233    // Step 3: Denorm val
234    if (less_than_normal) {
235      result =
236          _mm512_mask_mov_epi32(result, less_than_normal, temp_bits_for_denorm);
237    }
238  
239    // Step 4: restore sign
240    result = _mm512_or_si512(result, _mm512_srli_epi32(sign, 24));
241  
242    return _mm512_cvtepi32_epi8(result);
243  }

259  template <typename T>
260  class Vectorizedf8 {
261    static_assert(
262        std::integral_constant < bool,
263        std::is_same_v<T, at::Float8_e4m3fn> || std::is_same_v < T,
264        at::Float8_e5m2 >> ::value,
265        "Support only float8 e4m3.");
266  
267   private:
268    __m512i values;
269    template <typename Op, typename VectorizedType>
270    Vectorized<T> inline binary_compare(const VectorizedType& b, Op op) const {
271      __m512 a0, a1, a2, a3;
272      __m512 b0, b1, b2, b3;
273      __m512 o0, o1, o2, o3;
274      if constexpr (std::is_same_v<T, c10::Float8_e4m3fn>) {
275        cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(values, 0), a0);
276        cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(b.values, 0), b0);
277        cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(values, 1), a1);
278        cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(b.values, 1), b1);
279        cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(values, 2), a2);
280        cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(b.values, 2), b2);
281        cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(values, 3), a3);
282        cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(b.values, 3), b3);
283      } else {
284        cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(values, 0), a0);
285        cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(b.values, 0), b0);
286        cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(values, 1), a1);
287        cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(b.values, 1), b1);
288        cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(values, 2), a2);
289        cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(b.values, 2), b2);
290        cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(values, 3), a3);
291        cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(b.values, 3), b3);
292      }
293  
294      o0 = op(a0, b0);
295      o1 = op(a1, b1);
296      o2 = op(a2, b2);
297      o3 = op(a3, b3);
298      __m128i o128_0, o128_1, o128_2, o128_3;
299      if constexpr (std::is_same_v<T, c10::Float8_e4m3fn>) {
300        o128_0 = cvtfp32_fp8e4m3(o0);
301        o128_1 = cvtfp32_fp8e4m3(o1);
302        o128_2 = cvtfp32_fp8e4m3(o2);
303        o128_3 = cvtfp32_fp8e4m3(o3);
304      } else {
305        o128_0 = cvtfp32_fp8e5m2(o0);
306        o128_1 = cvtfp32_fp8e5m2(o1);
307        o128_2 = cvtfp32_fp8e5m2(o2);
308        o128_3 = cvtfp32_fp8e5m2(o3);
309      }
310  
311      __m512i result = _mm512_setzero_si512();
312      result = _mm512_inserti32x4(result, o128_0, 0);
313      result = _mm512_inserti32x4(result, o128_1, 1);
314      result = _mm512_inserti32x4(result, o128_2, 2);
315      result = _mm512_inserti32x4(result, o128_3, 3);
316  
317      return result;
318    }
319  
320   public:
321    using value_type = uint8_t;
322    using size_type = int;
323    static constexpr size_type size() {
324      return 64;
325    }
326    Vectorizedf8() {}
327    Vectorizedf8(__m512i v) : values(v) {}
328    Vectorizedf8(T val) {
329      value_type uw = val.x;
330      values = _mm512_set1_epi8(uw);
331    }
332    operator __m512i() const {
333      return values;
334    }
335    T& operator[](int idx) = delete;
336    const T& operator[](int idx) const = delete;
337    static Vectorized<T> loadu(const void* ptr, int16_t count = size()) {
338      if (count == size()) {
339        return _mm512_loadu_si512(reinterpret_cast<const __m512i*>(ptr));
340      } else if (count == 16) {
341        // Fast path if only load element number of 16
342        __m128i input_128 =
343            _mm_loadu_si128(reinterpret_cast<const __m128i*>(ptr));
344        return _mm512_castsi128_si512(input_128);
345      } else {
346        __mmask64 mask = (1ULL << count) - 1;
347        return _mm512_maskz_loadu_epi8(mask, ptr);
348      }
349    }
350    void store(void* ptr, int count = size()) const {
351      if (count == size()) {
352        _mm512_storeu_si512(reinterpret_cast<__m512i*>(ptr), values);
353      } else if (count > 0) {
354        if (count == 16) {
355          // Fast path if only store element number of 16
356          _mm_storeu_si128(
357              reinterpret_cast<__m128i*>(ptr), _mm512_castsi512_si128(values));
358        } else {
359          __mmask64 mask = (1ULL << count) - 1;
360          _mm512_mask_storeu_epi8(ptr, mask, values);
361        }
362      }
363    }
364  
365    Vectorized<T> abs() const {
366      return _mm512_andnot_si512(_mm512_set1_epi8(0x80), values);
367    }
368  
369    Vectorized<T> inline operator==(const Vectorizedf8<T>& other) const {
370      return binary_compare(other, [](__m512 x, __m512 y) {
371        auto zero_vec = _mm512_set1_epi32(0);
372        auto cmp = _mm512_cmp_ps_mask(x, y, _CMP_EQ_OQ);
373        return _mm512_castsi512_ps(
374            _mm512_mask_set1_epi32(zero_vec, cmp, 0xFFFFFFFF));
375      });
376    }
377  
378    Vectorized<T> inline operator!=(const Vectorizedf8<T>& other) const {
379      return binary_compare(other, [](__m512 x, __m512 y) {
380        auto zero_vec = _mm512_set1_epi32(0);
381        auto cmp = _mm512_cmp_ps_mask(x, y, _CMP_NEQ_UQ);
382        return _mm512_castsi512_ps(
383            _mm512_mask_set1_epi32(zero_vec, cmp, 0xFFFFFFFF));
384      });
385    }
386  
387    Vectorized<T> inline operator>(const Vectorizedf8<T>& other) const {
388      return binary_compare(other, [](__m512 x, __m512 y) {
389        auto zero_vec = _mm512_set1_epi32(0);
390        auto cmp = _mm512_cmp_ps_mask(x, y, _CMP_GT_OQ);
391        return _mm512_castsi512_ps(
392            _mm512_mask_set1_epi32(zero_vec, cmp, 0xFFFFFFFF));
393      });
394    }
395  
396    Vectorized<T> inline operator>=(const Vectorizedf8<T>& other) const {
397      return binary_compare(other, [](__m512 x, __m512 y) {
398        auto zero_vec = _mm512_set1_epi32(0);
399        auto cmp = _mm512_cmp_ps_mask(x, y, _CMP_GE_OQ);
400        return _mm512_castsi512_ps(
401            _mm512_mask_set1_epi32(zero_vec, cmp, 0xFFFFFFFF));
402      });
403    }
404  
405    Vectorized<T> inline operator<(const Vectorizedf8<T>& other) const {
406      return binary_compare(other, [](__m512 x, __m512 y) {
407        auto zero_vec = _mm512_set1_epi32(0);
408        auto cmp = _mm512_cmp_ps_mask(x, y, _CMP_LT_OQ);
409        return _mm512_castsi512_ps(
410            _mm512_mask_set1_epi32(zero_vec, cmp, 0xFFFFFFFF));
411      });
412    }
413  
414    Vectorized<T> inline operator<=(const Vectorizedf8<T>& other) const {
415      return binary_compare(other, [](__m512 x, __m512 y) {
416        auto zero_vec = _mm512_set1_epi32(0);
417        auto cmp = _mm512_cmp_ps_mask(x, y, _CMP_LE_OQ);
418        return _mm512_castsi512_ps(
419            _mm512_mask_set1_epi32(zero_vec, cmp, 0xFFFFFFFF));
420      });
421    }

439  template <
440      typename T,
441      typename Op,
442      std::enable_if_t<
443          std::is_same_v<T, c10::Float8_e4m3fn> ||
444              std::is_same_v<T, c10::Float8_e5m2>,
445          int> = 0>
446  static inline Vectorized<T> binary_fp8_op_as_fp32(
447      const Vectorized<T>& a,
448      const Vectorized<T>& b,
449      Op op) {
450    __m512 a0, a1, a2, a3;
451    __m512 b0, b1, b2, b3;
452    __m512 o0, o1, o2, o3;
453    if constexpr (std::is_same_v<T, c10::Float8_e4m3fn>) {
454      cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(a, 0), a0);
455      cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(b, 0), b0);
456      cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(a, 1), a1);
457      cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(b, 1), b1);
458      cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(a, 2), a2);
459      cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(b, 2), b2);
460      cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(a, 3), a3);
461      cvtfp8e4m3_fp32(_mm512_extracti32x4_epi32(b, 3), b3);
462    } else {
463      cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(a, 0), a0);
464      cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(b, 0), b0);
465      cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(a, 1), a1);
466      cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(b, 1), b1);
467      cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(a, 2), a2);
468      cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(b, 2), b2);
469      cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(a, 3), a3);
470      cvtfp8e5m2_fp32(_mm512_extracti32x4_epi32(b, 3), b3);
471    }
472    o0 = op(a0, b0);
473    o1 = op(a1, b1);
474    o2 = op(a2, b2);
475    o3 = op(a3, b3);
476  
477    __m128i o128_0, o128_1, o128_2, o128_3;
478    if constexpr (std::is_same_v<T, c10::Float8_e4m3fn>) {
479      o128_0 = cvtfp32_fp8e4m3(o0);
480      o128_1 = cvtfp32_fp8e4m3(o1);
481      o128_2 = cvtfp32_fp8e4m3(o2);
482      o128_3 = cvtfp32_fp8e4m3(o3);
483    } else {
484      o128_0 = cvtfp32_fp8e5m2(o0);
485      o128_1 = cvtfp32_fp8e5m2(o1);
486      o128_2 = cvtfp32_fp8e5m2(o2);
487      o128_3 = cvtfp32_fp8e5m2(o3);
488    }
489  
490    __m512i result = _mm512_setzero_si512();
491    result = _mm512_inserti32x4(result, o128_0, 0);
492    result = _mm512_inserti32x4(result, o128_1, 1);
493    result = _mm512_inserti32x4(result, o128_2, 2);
494    result = _mm512_inserti32x4(result, o128_3, 3);
495  
496    return result;
497  }
