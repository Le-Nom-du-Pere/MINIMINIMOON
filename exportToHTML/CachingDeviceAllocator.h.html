9    // Struct containing memory allocator summary statistics for a device.
10   struct DeviceStats {
11     // COUNT: allocations requested by client code
12     StatArray allocation;
13     // COUNT: number of allocated segments from device memory allocation.
14     StatArray segment;
15     // COUNT: number of active memory blocks (allocated or used by stream)
16     StatArray active;
17     // COUNT: number of inactive, split memory blocks (unallocated but can't be
18     // released via device memory deallocation)
19     StatArray inactive_split;
20   
21     // SUM: bytes allocated by this memory allocator
22     StatArray allocated_bytes;
23     // SUM: bytes reserved by this memory allocator (both free and used)
24     StatArray reserved_bytes;
25     // SUM: bytes within active memory blocks
26     StatArray active_bytes;
27     // SUM: bytes within inactive, split memory blocks
28     StatArray inactive_split_bytes;
29     // SUM: bytes requested by client code
30     StatArray requested_bytes;
31   
32     // COUNT: total number of failed calls to device malloc necessitating cache
33     // flushes.
34     int64_t num_alloc_retries = 0;
35   
36     // COUNT: total number of OOMs (i.e. failed calls to device memory allocation
37     // after cache flush)
38     int64_t num_ooms = 0;
39   
40     // COUNT: total number of oversize blocks allocated from pool
41     Stat oversize_allocations;
42   
43     // COUNT: total number of oversize blocks requiring malloc
44     Stat oversize_segments;
45   
46     // COUNT: total number of synchronize_and_free_events() calls
47     int64_t num_sync_all_streams = 0;
48   
49     // COUNT: total number of device memory allocation calls. This includes both
50     // mapped and malloced memory.
51     int64_t num_device_alloc = 0;
52   
53     // COUNT: total number of device memory deallocation calls. This includes both
54     // un-mapped and free memory.
55     int64_t num_device_free = 0;
56   
57     // SIZE: maximum block size that is allowed to be split.
58     int64_t max_split_size = 0;
59   };
60   
61   } // namespace c10::CachingDeviceAllocator
