1    """ 
2    JSON Utilities with NaN/Inf Handling 
3    ==================================== 
4     
5    Comprehensive utilities for safe JSON serialization with automatic cleaning 
6    of special float values (NaN, Infinity, -Infinity) that are not JSON-compliant. 
7     
8    Features: 
9    - Recursive data structure cleaning 
10   - Configurable replacement strategies for special float values   
11   - Thread-safe operations 
12   - Consistent json.dump() parameters across all JSON writes 
13   - Support for nested data structures (lists, dicts, tuples) 
14   """
15   
16   import json
17   import math
18   from typing import Any, Dict, List, Union, Optional, TextIO
19   import numpy as np
20   
21   
22   def clean_data_for_json(data: Any, 
23                          nan_replacement: Union[str, None] = None,
24                          inf_replacement: Union[str, None] = "Infinity",
25                          neg_inf_replacement: Union[str, None] = "-Infinity") -> Any:
26       """ 
27       Recursively clean data structure by replacing special float values. 
28        
29       Args: 
30           data: Input data structure to clean 
31           nan_replacement: Replacement for NaN values (default: None) 
32           inf_replacement: Replacement for positive infinity (default: "Infinity")  
33           neg_inf_replacement: Replacement for negative infinity (default: "-Infinity") 
34        
35       Returns: 
36           Cleaned data structure safe for JSON serialization 
37       """
38       if isinstance(data, dict):
39           return {key: clean_data_for_json(value, nan_replacement, inf_replacement, neg_inf_replacement) 
40                   for key, value in data.items()}
41       elif isinstance(data, (list, tuple)):
42           cleaned_list = [clean_data_for_json(item, nan_replacement, inf_replacement, neg_inf_replacement) 
43                          for item in data]
44           return cleaned_list if isinstance(data, list) else type(data)(cleaned_list)
45       elif isinstance(data, (float, np.floating)):
46           if math.isnan(data):
47               return nan_replacement
48           elif math.isinf(data):
49               return inf_replacement if data > 0 else neg_inf_replacement
50           else:
51               return float(data)  # Convert numpy floats to Python floats
52       elif isinstance(data, np.integer):
53           return int(data)  # Convert numpy integers to Python integers
54       elif isinstance(data, np.ndarray):
55           return clean_data_for_json(data.tolist(), nan_replacement, inf_replacement, neg_inf_replacement)
56       else:
57           return data

60   def safe_json_dump(data: Any, 
61                      file_handle: TextIO,
62                      nan_replacement: Union[str, None] = None,
63                      inf_replacement: Union[str, None] = "Infinity", 
64                      neg_inf_replacement: Union[str, None] = "-Infinity",
65                      **kwargs) -> None:
66       """ 
67       Safe JSON dump with automatic data cleaning and consistent parameters. 
68        
69       Args: 
70           data: Data to serialize 
71           file_handle: File handle to write to 
72           nan_replacement: Replacement for NaN values 
73           inf_replacement: Replacement for positive infinity 
74           neg_inf_replacement: Replacement for negative infinity 
75           **kwargs: Additional arguments passed to json.dump (will override defaults) 
76       """
77       # Default parameters for consistent JSON formatting
78       default_params = {
79           'ensure_ascii': False,
80           'indent': 2,
81           'allow_nan': False
82       }
83       
84       # Update with any user-provided parameters
85       default_params.update(kwargs)
86       
87       # Clean the data
88       cleaned_data = clean_data_for_json(data, nan_replacement, inf_replacement, neg_inf_replacement)
89       
90       # Write to file
91       json.dump(cleaned_data, file_handle, **default_params)

94   def safe_json_dumps(data: Any,
95                       nan_replacement: Union[str, None] = None,
96                       inf_replacement: Union[str, None] = "Infinity",
97                       neg_inf_replacement: Union[str, None] = "-Infinity", 
98                       **kwargs) -> str:
99       """ 
100      Safe JSON dumps with automatic data cleaning and consistent parameters. 
101       
102      Args: 
103          data: Data to serialize 
104          nan_replacement: Replacement for NaN values 
105          inf_replacement: Replacement for positive infinity 
106          neg_inf_replacement: Replacement for negative infinity 
107          **kwargs: Additional arguments passed to json.dumps (will override defaults) 
108           
109      Returns: 
110          JSON string representation 
111      """
112      # Default parameters for consistent JSON formatting
113      default_params = {
114          'ensure_ascii': False,
115          'indent': 2,
116          'allow_nan': False
117      }
118      
119      # Update with any user-provided parameters
120      default_params.update(kwargs)
121      
122      # Clean the data
123      cleaned_data = clean_data_for_json(data, nan_replacement, inf_replacement, neg_inf_replacement)
124      
125      # Return JSON string
126      return json.dumps(cleaned_data, **default_params)

129  def safe_json_write(data: Any, 
130                      filepath: str,
131                      nan_replacement: Union[str, None] = None,
132                      inf_replacement: Union[str, None] = "Infinity",
133                      neg_inf_replacement: Union[str, None] = "-Infinity",
134                      encoding: str = "utf-8",
135                      **kwargs) -> None:
136      """ 
137      Convenience function to write JSON data to a file with safety checks. 
138       
139      Args: 
140          data: Data to serialize 
141          filepath: Path to the output file 
142          nan_replacement: Replacement for NaN values 
143          inf_replacement: Replacement for positive infinity 
144          neg_inf_replacement: Replacement for negative infinity 
145          encoding: File encoding (default: utf-8) 
146          **kwargs: Additional arguments passed to json.dump 
147      """
148      with open(filepath, 'w', encoding=encoding) as f:
149          safe_json_dump(data, f, nan_replacement, inf_replacement, neg_inf_replacement, **kwargs)
