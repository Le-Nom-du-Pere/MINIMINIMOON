34   class OnnxFunctionDispatcher:
35       """A dispatcher that finds the best ONNX Function for ATen/Custom operators. 
36    
37       It uses the `torch.ops` name to find the function. If not found, it falls back to default. 
38       Otherwise, the best match is found among all function overloads. An exact match has 
39       higher precedence over the closest ones. 
40    
41       Below is a breakdown on how the dispatch mechanism works: 
42    
43       1. Use the torch.ops name to find the function: 
44           a. Check if the ATen overload exists in the registry. 
45           b. If not, check if the default overload exists in the registry. 
46    
47       2. Find the nearest match among all overloaded functions: 
48           a. If the types match perfectly, select the function. 
49           b. Otherwise, find the nearest one with the highest matching score. Because of 
50               the potential wrongly annotated dtypes and attributes matching, we use 
51               nearest match to find the best function once the aten name is targeted. 
52    
53       3. Tie-breaker: If there are multiple nearest matches, we will select the one with 
54           the highest matching score. 
55    
56       NOTE: The nearest match `doesn't guarantee` a correct match, and a warning message is logged. 
57       """
58

60           self,
61           onnx_registry: OnnxRegistry,
62       ):
63           """Initialize the ONNX Function dispatcher. 
64    
65           Args: 
66               onnx_registry: The ONNX registry. 
67           """
68           self.onnx_registry = onnx_registry

70       def dispatch(
71           self,
72           node: torch.fx.Node,
73           onnx_args: Sequence[
74               fx_type_utils.TensorLike | str | int | float | bool | list | complex | None
75           ],
76           onnx_kwargs: dict[str, fx_type_utils.Argument],
77       ) -> onnxscript.OnnxFunction | onnxscript.TracedOnnxFunction:
78           """Dispatches an ONNX function based on the given FX node, arguments, and keyword arguments. 
79           Args: 
80               node: The TorchFX node to dispatch the function for. 
81               onnx_args: The arguments of the ONNX function. 
82               onnx_kwargs: The keyword arguments of the ONNX function. 
83    
84           Returns: 
85               Either an `onnxscript.OnnxFunction` or `onnxscript.TracedOnnxFunction` instance based on the dispatch algorithm. 
86           Raises: 
87               RuntimeError: If there are no overloaded functions available for the given FX node. 
88           """
89           # If there are no overloaded functions available for the given FX node, raise an
90           # unsupported error
91           default_and_custom_functions = self.get_function_overloads(node)
92   
93           # If there are overloaded functions available, we will find one that perfect or
94           # nearest matches the given arguments and keyword arguments
95           return self._find_the_perfect_or_nearest_match_onnxfunction(
96               node,
97               default_and_custom_functions,
98               onnx_args,
99               onnx_kwargs,
100          )

102      def _filter_or_keep_complex(
103          self,
104          node,
105          default_and_custom_functions: list[registration.ONNXFunction],
106      ) -> list[registration.ONNXFunction]:
107          """Filter the complex functions if the input has complex dtype."""
108  
109          args_with_complex_dtype = [_is_arg_with_complex_dtype(arg) for arg in node.args]
110          if any(args_with_complex_dtype):
111              default_and_custom_functions = [
112                  func for func in default_and_custom_functions if func.is_complex
113              ]
114              # If we can't find the complex function group, raise error.
115              if not default_and_custom_functions:
116                  op_full_name = self._get_aten_name(node).qualified_name()
117                  raise RuntimeError(
118                      f"Cannot find any COMPLEX symbolic function for {op_full_name}, "
119                      f"which should be registered under {node.target}.",
120                  )
121          else:
122              default_and_custom_functions = [
123                  func for func in default_and_custom_functions if not func.is_complex
124              ]
125              # If we can't find the complex function group, raise error.
126              if not default_and_custom_functions:
127                  op_full_name = self._get_aten_name(node).qualified_name()
128                  raise RuntimeError(
129                      f"Can ONLY find COMPLEX symbolic function for {op_full_name}, "
130                      f"which should be registered under {node.target}.",
131                  )
132          return default_and_custom_functions

134      def _find_the_perfect_or_nearest_match_onnxfunction(
135          self,
136          node: torch.fx.Node,
137          default_and_custom_functions: list[registration.ONNXFunction],
138          onnx_args: Sequence[
139              fx_type_utils.TensorLike | str | int | float | bool | list | complex | None
140          ],
141          onnx_kwargs: dict[str, fx_type_utils.Argument],
142      ):
143          """Find the perfect/nearest matched OnnxFunction for the given FX node, arguments, and keyword arguments. 
144   
145          Args: 
146              default_and_custom_functions: The list includes overloaded functions, with 
147                  custom ones appearing after the default ones. 
148              onnx_args: Arguments organized in PyTorch inputs way. 
149              onnx_kwargs: Keyword arguments organized in PyTorch inputs way. 
150   
151              Returns: 
152                  Either an `onnxscript.OnnxFunction` or `onnxscript.TracedOnnxFunction` instance based on the dispatch algorithm. 
153              Raises: 
154                  RuntimeError: If there are no overloaded functions available for the given FX node. 
155          """
156          overload_match_ranking: dict[registration.ONNXFunction, int | None] = {}
157  
158          # Iterate the overloaded functions in reverse order to prioritize the custom ones
159          # over the default ones, and find the perfect match.
160          for symbolic_function in reversed(default_and_custom_functions):
161              function_opschema = _OnnxSchemaChecker(symbolic_function.onnx_function)
162  
163              # NOTE: 1. If the perfect match is found, return the function
164              if function_opschema.perfect_match_inputs(onnx_args, onnx_kwargs):
165                  return symbolic_function.onnx_function
166              # Record the match score for the nearest match if it's not the perfect match
167              overload_match_ranking[symbolic_function] = function_opschema.match_score
168  
169          # NOTE: 2. If there is no perfect match, find the nearest match among the nearest matche candidates
170          # If there is no nearest match, raise an error
171          overload_match_ranking = {
172              k: v for k, v in overload_match_ranking.items() if v is not None
173          }
174          if not overload_match_ranking:
175              # If there are no overloaded functions available for the given FX node, raise an
176              # unsupported error
177              op_full_name = self._get_aten_name(node).qualified_name()
178              raise RuntimeError(
179                  f"Cannot find any perfect/nearest match of symbolic function for {op_full_name},"
180                  f"which should be registered under {node.target}.",
181              )
182  
183          # NOTE: 3. Tie breaker: if there are multiple nearest matches, we will choose the one
184          # that is custom first. If there are multiple custom ones, we will choose the one
185          # that is added lastly in the list.
186          symbolic_function_list: list[registration.ONNXFunction] = sorted(
187              overload_match_ranking,
188              key=lambda k: (
189                  overload_match_ranking[k],
190                  k.is_custom,
191                  default_and_custom_functions.index(k),
192              ),
193              reverse=True,
194          )
195          return symbolic_function_list[0].onnx_function

197      def _get_aten_name(self, node: torch.fx.Node) -> registration.OpName:
198          """Get the OpName from the target. 
199   
200          Args: 
201              node: The TorchFX node to get the aten name for. 
202   
203          Returns: 
204              The internal op name within dataclass: registration.OpName. 
205          """
206          if node.target == operator.getitem:
207              return registration.OpName.from_name_parts(
208                  namespace="aten", op_name="getitem"
209              )
210          if isinstance(node.target, torch._ops.OpOverloadPacket):
211              # aten::sym_size is the only OverloadPacket that we support.
212              # schema: aten::sym_size(Tensor self, int dim) -> Tensor
213              if node.target != torch.ops.aten.sym_size:
214                  raise RuntimeError(
215                      f"Unsupported OverloadPacket: {node.target}, aten.sym_size is the only allowed OverloadPacket!",
216                  )
217              # TODO(titaiwang): aten::sym_size has overload, but fx graph is using
218              # overloadpacket for some reasons.
219              # https://github.com/pytorch/pytorch/issues/97201
220              aten_op_default = node.target.default
221              return registration.OpName.from_op_overload(op_overload=aten_op_default)  # type: ignore[no-any-return]
222  
223          if isinstance(node.target, types.BuiltinFunctionType):
224              # Make sure it's symint/symfloat consuming builtin ops.
225              for node_arg in node.args:
226                  if (not isinstance(node_arg, (torch.fx.Node, int, float))) or (
227                      isinstance(node_arg, torch.fx.Node)
228                      and not fx_type_utils.is_torch_symbolic_type(node_arg.meta["val"])
229                  ):
230                      raise RuntimeError(
231                          f"Unsupported node arg: {node_arg} (type {type(node_arg)}) with builtin function: {node.target},"
232                          " only int/float/SymInt/SymFloat is supported with built-in ops!",
233                      )
234              return registration.OpName.from_builtin_function(node.target)
235  
236          if isinstance(node.target, torch._ops.OpOverload):
237              return registration.OpName.from_op_overload(op_overload=node.target)
238  
239          # Unexpected target, raise error.
240          raise RuntimeError(f"Unknown call_function target: {node.target}")

242      def get_function_overloads(
243          self,
244          node: torch.fx.Node,
245      ) -> list[registration.ONNXFunction]:
246          """Get the function overloads from the registry. 
247   
248          Args: 
249              node: The node to get the function overloads for. 
250   
251          Returns: 
252              The list contains ONNXFunctions, starting with the default ones and 
253              followed by any custom ones. 
254          """
255  
256          internal_opname: registration.OpName = self._get_aten_name(node=node)
257  
258          # If the ATen/Custom operators are not registered, the group will be None.
259          # And non-registered ATen/Custom operators will trigger error in the next step.
260          function_group: list[registration.ONNXFunction] | None = None
261  
262          function_group = self.onnx_registry.get_op_functions(
263              namespace=internal_opname.namespace,
264              op_name=internal_opname.op_name,
265              overload=internal_opname.overload,
266          )
267  
268          # NOTE: Fall back to default overload if the ONNX registry doesn't have the overload.
269          if function_group is None:
270              function_group = self.onnx_registry.get_op_functions(
271                  namespace=internal_opname.namespace,
272                  op_name=internal_opname.op_name,
273                  overload=None,
274              )
275              if function_group is not None:
276                  op_full_name = internal_opname.qualified_name()
277  
278          if function_group is not None:
279              # NOTE: If the input has complex dtype, we will only dispatch to the complex functions.
280              function_group = self._filter_or_keep_complex(node, function_group)
281              return function_group  # type: ignore[return-value]
282  
283          op_full_name = internal_opname.qualified_name()
284          raise RuntimeError(
285              f"Cannot find symbolic function for {op_full_name}, "
286              f"which should be registered under {node.target}.",
287          )

290  class _OnnxSchemaChecker:
291      """ 
292      The OnnxSchemaChecker class is a checker for ONNX OpSchema and param schema. 
293   
294      It provides methods to check for input compatibility based on the OpSchema. It also 
295      provides a matching score to indicate how well the OpSchema matches the input and 
296      kwargs types. A function will be evaluated as perfect match, nearest match eligible, 
297      or no match. 
298   
299      Here are some common examples in categories: 
300   
301      1. [NOTE: Perfect match]: The number of inputs and attributes are exactly the same as 
302          the OpSchema. The types of inputs and attributes are exactly the same as the 
303          OpSchema. 
304   
305          ```python 
306          inputs = (Tensor[2, 3], Tensor[2, 3]) 
307          attributes = {"alpha": 1.0} 
308   
309   
310          @torch_op("aten::op") 
311          def aten_op(self: TReal, other: TReal, alpha: float = 1) -> TReal: ... 
312          ``` 
313          Result: Perfect match. 
314   
315      2. [NOTE: Optional input]: The dispatcher recognizes optional inputs. However, 
316          the input can't be ignored. None must be provided. 
317   
318          ```python 
319          inputs = (Tensor([2, 3]), None) 
320          attributes = {} 
321   
322          aten_op(X: TTensor, Y: Optional[INT64]): 
323              ... 
324          ``` 
325          Result: Perfect match. 
326          Real example: `aten::convolution`. 
327   
328      3. [NOTE: Different attributes]: If an attribute is provided with value, it's 
329          a must to match the attribute in function signature. 
330          ```python 
331          inputs = (Tensor([2, 3]),) 
332          attributes = {"a":1, "b":2} 
333   
334          aten_op(X: TTensor, a: int): 
335              ... 
336          ``` 
337          Result: No match. 
338          Real example: `aten::div` vs `aten::div.Tensor_mode`. 
339   
340      4. [NOTE: Default attributes]: Default attribute will fill in the value into 
341          inputs/attributes. 
342          ```python 
343          inputs = (Tensor([2, 3]),) 
344          attributes = {} 
345   
346          aten_op(X: TTensor, a: int = 3): 
347              ... 
348          ``` 
349          Result: Perfect match. 
350          Real example: `aten::clone` 
351   
352      5. [NOTE: Ignore attribute with None value]: The attributes with None value 
353          will be ignored in matching. 
354          ```python 
355          inputs = (Tensor([2, 3]),) 
356          attributes = {"a": None} 
357   
358          aten_op(X: TTensor): 
359              ... 
360          ``` 
361          Result: Perfect match. 
362   
363          ```python 
364          inputs = (Tensor([2, 3]),) 
365          attributes = {"a": None} 
366   
367          aten_op(X: TTensor, a: int = 3): 
368              ... 
369          ``` 
370          Result: Nearest match eligible. 
371   
372          Real example: `aten::div` vs `aten::div.Tensor_mode`. 
373   
374      Attributes: 
375          onnxfunction: The OnnxFunction. 
376          param_schema: The parameter schema defined in the OnnxFunction. 
377          op_schema: The ONNX OpSchema. 
378          type_constraints: The type constraints defined in the OpSchema. 
379          attributes: The attributes defined in the OpSchema. 
380          _matching_score: The matching score of the OnnxSchemaChecker . 
381   
382      """
383  
384      def __init__(
385          self,
386          onnxfunction: onnxscript.OnnxFunction | onnxscript.TracedOnnxFunction,
387      ):
388          """Initialize the OnnxSchemaChecker . 
389   
390          Args: 
391              onnxfunction: The OnnxFunction. 
392          """
393          self.onnxfunction = onnxfunction
394          self.param_schema = self.onnxfunction.param_schemas()
395          op_schema = self.onnxfunction.op_schema
396          # Both `OnnxFunction` and `TracedOnnxFunction` never return None for `op_schema`.
397          # However their base class would. Hence return type is annotated as Optional[OpSchema].
398          assert op_schema is not None
399          self.op_schema = op_schema
400          self.type_constraints = {
401              # "T": {"tensor(int64)"}
402              constraint.type_param_str: set(constraint.allowed_type_strs)
403              for constraint in self.op_schema.type_constraints
404          }
405          self.attributes = self.op_schema.attributes
406          self._matching_score: int | None = None
407  
408      @property
409      def match_score(self) -> int | None:
410          """The matching score of the OnnxSchemaChecker . 
411   
412          If this remains None, it means the matching score has not been calculated, 
413          and it's not a nearest match candidate. 
414   
415          Returns: 
416              The matching score of the OnnxSchemaChecker . 
417          """
418          return self._matching_score

420      def perfect_match_inputs(
421          self,
422          args: Sequence[
423              fx_type_utils.TensorLike | str | int | float | bool | list | complex | None
424          ],
425          kwargs: dict[str, fx_type_utils.Argument],
426      ) -> bool:
427          """Check if the inputs perfectly match the OpSchema requirements. 
428   
429          The definition of perfect match is that the input types are all in the type 
430          constraints and the number of inputs matches the number of inputs in the 
431          OpSchema. 
432   
433          Checking steps: 
434          1. The function signature matches the inputs number, and attribute names. 
435          2. The input/attribute types are all in the type constraints. 
436   
437          A function should at least pass the first step to be eligible for the 
438          nearest matching. 
439   
440          Args: 
441              args: The input arguments organized in PyTorch inputs way. 
442              kwargs: The input keyword arguments organized in PyTorch inputs way. 
443   
444          Returns: 
445              True if the inputs match the requirements, False otherwise. 
446          """
447  
448          # NOTE: OnnxFunction does not have the same function signature as the original
449          # PyTorch operator. We need to separate the input/attributes from the arguments.
450          (
451              function_inputs,
452              function_attributes,
453          ) = self._separate_input_attributes_from_arguments(
454              self.param_schema,
455              args,
456              kwargs,
457              fill_defaults=True,  # fill defaults for optional arguments to match
458          )
459          # NOTE: 1. Check if the input number and attribute names match the
460          # OpSchema. If it's not, we know the function is not eligible to be a perfect
461          # match, nor a nearest match.
462          # We use is_perfect_match to postpone the return value to the end
463          # of the function, as we want to log all the mismatch info.
464          is_perfect_match = True
465          if len(function_inputs) != len(self.op_schema.inputs):
466              logger.info(
467                  "Actual %d vs expected %d",
468                  len(function_inputs),
469                  len(self.op_schema.inputs),
470              )
471              logger.info("The function is not a nearest match candidate.")
472              is_perfect_match = False
473  
474          if set(function_attributes) != set(self.attributes):
475              logger.info("The function is not a nearest match candidate.")
476              is_perfect_match = False
477  
478          # If it's already not a perfect match, we can return False directly. Further
479          # checking is only for the functions that are eligible for nearest match.
480          if not is_perfect_match:
481              return False
482  
483          # NOTE: 2. The dtypes of inputs and attributes should be in the
484          # type constraints of the OpSchema. If they are not, we know the function is not
485          # eligible to be a perfect match, but can be a nearest match candidate.
486          for schema_input, torch_input in zip(self.op_schema.inputs, function_inputs):
487              torch_input_compatible_types = _find_onnx_data_type(torch_input)
488              allowed_types = self.type_constraints[schema_input.type_str]
489              if not allowed_types.intersection(torch_input_compatible_types) and not any(
490                  fx_type_utils.is_optional_onnx_dtype_str(onnx_type_str)
491                  for onnx_type_str in allowed_types
492              ):
493                  # If torch_input_compatible_types isn't in allowed_types
494                  # of this input defined in the OpSchema, we know the function
495                  # and the input are not compatible
496                  logger.info(
497                      "Actual %s vs\nExpected %s",
498                      torch_input_compatible_types,
499                      allowed_types,
500                  )
501                  is_perfect_match = False
502  
503          for attribute_name, attribute in function_attributes.items():
504              if not self._match_onnx_attribute_type(attribute_name, attribute):
505                  # If the attribute type of the OpSchema and the attribute type don't match,
506                  # we know the function and the input are not compatible
507                  logger.info(
508                      "Actual %s vs\nExpected %s",
509                      type(attribute),
510                      self.attributes[attribute_name].type,
511                  )
512                  is_perfect_match = False
513  
514          # NOTE: This is still a candidate for nearest match, as it only mismatches attributes on dtype.
515          self._record_matching_score(function_inputs, function_attributes)
516          logger.info("match score: %d", self.match_score)
517          return is_perfect_match
518  
519      def _match_onnx_attribute_type(
520          self,
521          attribute_name: str,
522          attribute: fx_type_utils.Argument | onnxscript_graph_building.TorchScriptTensor,
523          is_sequence: bool = False,
524      ) -> bool:
525          if isinstance(attribute, (int, float, bool, str)):
526              attribute_onnx_type = fx_type_utils.from_python_type_to_onnx_attribute_type(
527                  type(attribute), is_sequence=is_sequence
528              )
529              if attribute_onnx_type != self.attributes[attribute_name].type:
530                  return False
531          # If the attribute is an empty list, we don't know the type of the list
532          # so it's a mismatch
533          elif isinstance(attribute, (list, tuple)) and attribute:
534              return self._match_onnx_attribute_type(
535                  attribute_name, attribute[0], is_sequence=True
536              )
537          else:
538              # NOTE: Unrecognized attribute type
539              return False
540          return True

542      def _record_matching_score(
543          self,
544          inputs: Sequence[
545              fx_type_utils.TensorLike | str | int | float | bool | list | complex | None
546          ],
547          attributes: dict[str, fx_type_utils.Argument],
548      ):
549          """Calculate the inputs matching score of the OpSchema requirements to find the nearest match. 
550   
551          Only the functions which have the same number of inputs and attributes as the 
552          OpSchema are eligible to be a nearest match candidate. Thus, we don't need to 
553          check the length of inputs and attributes here, and only check the types of 
554          inputs and attributes. 
555   
556          How the matchsing score is calculated: 
557              score += 1 if one input/attribute type is in the type constraints. 
558   
559          Limitations: 
560              None/NoeType/[] could result in zero matches, and the same score of overloads. 
561   
562          Args: 
563              inputs: The input arguments. 
564              attributes: The input keyword arguments. 
565   
566          Returns: 
567              True if the inputs match the requirements, False otherwise. 
568          """
569          self._matching_score = 0
570          # If they have different length of arguments, the score would be lower to those
571          # functions which have the same length of arguments.
572          for schema_input, torch_input in zip(self.op_schema.inputs, inputs):
573              torch_input_compatible_types = _find_onnx_data_type(torch_input)
574              allowed_types = self.type_constraints[schema_input.type_str]
575              if allowed_types.intersection(torch_input_compatible_types):
576                  # If torch_input_compatible_types is in allowed_types
577                  # of this input defined in the OpSchema, we know the function
578                  # and the input are compatible
579                  self._matching_score += 1
580          # NOTE: The penalty is applied to those functions which have different attributes.
581          for attribute_name, attribute_proto in self.attributes.items():
582              attribute = attributes[attribute_name]
583              attribute_onnx_type = fx_type_utils.from_python_type_to_onnx_attribute_type(
584                  type(attribute)
585              )
586              if attribute_onnx_type != attribute_proto.type:
587                  # If the attribute type of the OpSchema and the attribute type don't match,
588                  # we know the function and the input are not compatible
589                  self._matching_score -= 1

591      # NOTE: Referenced from onnxscript internal function.
592      # Importing this function makes the code less robust, as it is not a public API.

594      def _separate_input_attributes_from_arguments(
595          self,
596          param_schemas: Sequence[onnxscript.values.ParamSchema],
597          args: Sequence[
598              fx_type_utils.TensorLike | str | int | float | bool | list | complex | None
599          ],
600          kwargs: dict[str, fx_type_utils.Argument],
601          fill_defaults: bool = True,
602      ) -> tuple[list[Any], dict[str, Any]]:
603          """Separate Python args and kwargs into ONNX inputs and attributes. 
604   
605          Extra_kwargs are ignored if their values are None. For example, if the 
606          OpSchema has an attribute "rounding_mode" and the caller provides 
607          "rounding_mode=None", the attribute "rounding_mode" will not be included 
608          in the returned attributes when the OnnxFunction signature doesn't have 
609          "rounding_mode" as an attribute. 
610   
611          Args: 
612              param_schemas: The parameter schemas of an Op or a OnnxFunction. 
613              args: The Python positional arguments supplied by the caller. 
614              kwargs: The Python keyword arguments supplied by the caller. 
615              fill_defaults: Whether to fill the default values for attributes. 
616   
617          Returns: 
618              A tuple of two elements: 
619              - A list of ONNX inputs. 
620              - An dictionary of ONNX attribute names and values. 
621   
622          Raises: 
623              TypeError: When allow_extra_kwargs is False and there are unknown kwargs. 
624              TypeError: When a required input is not provided. 
625          """
626          # args, kwargs and param_schemas should be all in order
627          # user may not specify all inputs or attributes
628  
629          import onnx
630  
631          onnx_inputs: list[Any] = []
632          onnx_attributes: dict[str, Any] = {}
633          # NOTE: We need to copy kwargs because we will mutate it
634          copy_kwargs = kwargs.copy()
635          for i, param in enumerate(param_schemas):
636              if param.is_variadic_input:
637                  # Exhaust all remaining args
638                  onnx_inputs.extend(args[i:])
639                  args = []
640                  continue
641              if i < len(args):
642                  if param.is_input:
643                      onnx_inputs.append(args[i])
644                  else:
645                      onnx_attributes[param.name] = args[i]
646              elif param.name in copy_kwargs:
647                  if param.is_input:
648                      # Move the input from kwargs to inputs
649                      onnx_inputs.append(copy_kwargs[param.name])
650                      copy_kwargs.pop(param.name)
651                  else:
652                      onnx_attributes[param.name] = copy_kwargs[param.name]
653              elif (
654                  param.is_attribute
655                  and self.attributes[param.name].default_value.type
656                  != onnx.AttributeProto.UNDEFINED  # type: ignore[attr-defined]
657              ):
658                  # User did not provide the attribute
659                  if fill_defaults:
660                      onnx_attributes[param.name] = param.default
661              # optional input
662              elif param.is_input:
663                  if fill_defaults:
664                      onnx_inputs.append(None)
665  
666          # NOTE: Pick up extra kwargs if it's not None. None is not expected
667          # as an attribute value in torchlib.
668          for k, v in copy_kwargs.items():
669              if k not in onnx_attributes and v is not None:
670                  onnx_attributes[k] = v
671          return onnx_inputs, onnx_attributes

674  def _is_arg_with_complex_dtype(arg: fx_type_utils.Argument) -> bool:
675      """Check if the node has complex dtype recursively."""
676      if (
677          isinstance(arg, torch.fx.Node)
678          and "val" in arg.meta
679          and isinstance(arg.meta["val"], torch.Tensor)
680          and torch.is_complex(arg.meta["val"])
681      ):
682          return True
683      elif isinstance(arg, list):
684          for item in arg:
685              return _is_arg_with_complex_dtype(item)
686      return False

689  def _find_onnx_data_type(
690      torch_input: fx_type_utils.TensorLike
691      | str
692      | int
693      | float
694      | bool
695      | list
696      | tuple
697      | complex
698      | None,
699  ) -> set[str]:
700      """Convert inputs data type from torch acceptable dtype to the compatible onnx dtype string."""
701      if (
702          isinstance(torch_input, fx_type_utils.TensorLike)
703          and torch_input.dtype is not None
704      ):
705          return fx_type_utils.from_torch_dtype_to_onnx_dtype_str(torch_input.dtype)
706      if isinstance(torch_input, (int, float, bool, str, complex)):
707          return fx_type_utils.from_torch_dtype_to_onnx_dtype_str(type(torch_input))
708      if isinstance(torch_input, (list, tuple)) and torch_input:  # [Tensor, Tensor]
709          the_first_non_none_item = next(
710              (item for item in torch_input if item is not None), None
711          )
712          set_dtype = _find_onnx_data_type(the_first_non_none_item)
713          if any(isinstance(input, fx_type_utils.TensorLike) for input in torch_input):
714              # NOTE: Any Tensor involved in a list would make it a seq(tensor(onnx_type))
715              return {f"seq({dtype})" for dtype in set_dtype}
716          else:
717              # constant list of non-tensor type
718              return set_dtype
719      if (
720          torch_input is None
721          or (
722              isinstance(torch_input, fx_type_utils.TensorLike)
723              and torch_input.dtype is None
724          )
725          or (isinstance(torch_input, (list, tuple)) and not torch_input)
726      ):
727          # NOTE: None, No dtype, and empty list are edge cases, we allow it to be any type to relax the type check
728          # seq(tensor) also goes to here, as it is not supported in torchscript, and it would be None in this case.
729          return set()
730  
731      raise RuntimeError(f"Unknown input type from input: {torch_input}")
