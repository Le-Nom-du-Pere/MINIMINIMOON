17   @implements_per_sample_grads(F.layer_norm)
18   class LayerNormPerSampleGrad(torch.autograd.Function):
19       @staticmethod
20       def forward(ctx, kwarg_names, _, *expanded_args_and_kwargs):
21           expanded_args, expanded_kwargs = standard_kwargs(
22               kwarg_names, expanded_args_and_kwargs
23           )
24           input = expanded_args[0]
25           normalized_shape = expanded_args[1]
26           if len(input.shape) <= len(normalized_shape):
27               raise RuntimeError(
28                   "Expanded Weights: Layer norm should not normalize over batch dimension for per sample gradient"
29                   f"computations but got that normalized shape, {normalized_shape}, matched input shape."
30               )
31           output, mean, rstd = forward_helper(
32               torch.native_layer_norm, expanded_args, expanded_kwargs
33           )
34           ctx.args = expanded_args
35   
36           if input.requires_grad or isinstance(expanded_kwargs["weight"], ExpandedWeight):
37               ctx.weight = expanded_kwargs["weight"]
38           if input.requires_grad or isinstance(expanded_kwargs["bias"], ExpandedWeight):
39               ctx.bias = expanded_kwargs["bias"]
40           ctx.eps = expanded_kwargs["eps"]
41           ctx.mean, ctx.rstd = mean, rstd
42           return output
43   
44       @staticmethod
45       def backward(ctx, grad_output):
46           def weight_per_sample_grad(weight):
47               return sum_over_all_but_batch_and_last_n(
48                   F.layer_norm(input, normalized_shape, eps=ctx.eps) * grad_output,
49                   weight.dim(),
50               )
51   
52           input, normalized_shape = ctx.args
53           mean, rstd = ctx.mean, ctx.rstd
54   
55           results: list[Optional[torch.Tensor]] = []
56           results.append(None)  # for kwarg names
57           results.append(None)  # for op reference
58           if input.requires_grad:
59               weight_ = unpack_expanded_weight_or_tensor(ctx.weight)
60               bias_ = unpack_expanded_weight_or_tensor(ctx.bias)
61               results.append(
62                   torch.ops.aten.native_layer_norm_backward(
63                       grad_output,
64                       input,
65                       normalized_shape,
66                       mean,
67                       rstd,
68                       weight_,
69                       bias_,
70                       (True, False, False),
71                   )[0]
72               )
73           else:
74               results.append(None)
75   
76           # weight and bias don't compute batched gradients; no other arguments are differentiable
77           results = results + [None] * 4
78   
79           # set grad_sample field for weight and bias with per sample gradients
80           if hasattr(ctx, "weight"):
81               set_grad_sample_if_exists(ctx.weight, weight_per_sample_grad)
82           if hasattr(ctx, "bias"):
83               set_grad_sample_if_exists(
84                   ctx.bias,
85                   lambda bias: sum_over_all_but_batch_and_last_n(grad_output, bias.dim()),
86               )
87           return tuple(results)
