14   
15   /// Gets n step value.
16   ///
17   /// The starting row index must be divisible by `n_step`.
18   ///
19   /// @return The n step value.
20   size_t kai_get_n_step_rhs_pack_nxk_x16p2vlx2b_x16_x16_sme(void);
21   
22   /// Gets the offset in bytes to the data element in the RHS matrix buffer.
23   ///
24   /// @param[in] n_idx Column index.
25   /// @param[in] rhs_stride Row stride in bytes of the RHS matrix.
26   ///
27   /// @return The offset in bytes to the data element.
28   size_t kai_get_rhs_offset_rhs_pack_nxk_x16p2vlx2b_x16_x16_sme(size_t n_idx, size_t rhs_stride);
29   
30   /// Gets the offset in bytes to the data element in the bias buffer.
31   ///
32   /// @param[in] n_idx Column index.
33   ///
34   /// @return The offset in bytes to the data element.
35   size_t kai_get_bias_offset_rhs_pack_nxk_x16p2vlx2b_x16_x16_sme(size_t n_idx);
36   
37   /// Gets the row stride in bytes of the packed RHS matrix.
38   ///
39   /// @param[in] k The number of columns in the transposed RHS matrix.
40   ///
41   /// @return The row stride in bytes.
42   size_t kai_get_rhs_packed_stride_rhs_pack_nxk_x16p2vlx2b_x16_x16_sme(size_t k);
43   
44   /// Gets the offset in bytes to the data element in the packed RHS buffer.
45   ///
46   /// @param[in] n_idx Row index.
47   /// @param[in] k Number of columns.
48   ///
49   /// @return The offset in bytes to the data element.
50   size_t kai_get_rhs_packed_offset_rhs_pack_nxk_x16p2vlx2b_x16_x16_sme(size_t n_idx, size_t k);
51   
52   /// Gets the size in bytes of the packed RHS buffer.
53   ///
54   /// @param[in] n Number of rows.
55   /// @param[in] k Number of columns.
56   ///
57   /// @return The size in bytes of the packed RHS buffer.
58   size_t kai_get_rhs_packed_size_rhs_pack_nxk_x16p2vlx2b_x16_x16_sme(size_t n, size_t k);
59   
60   /// Runs the RHS packing function for matrix multiplication.
61   ///
62   /// The pointer of each buffers (RHS, bias and packed RHS) needs to be added with offset
63   /// calculated using the following functions:
64   ///
65   ///   * RHS: @ref kai_get_rhs_offset_rhs_pack_nxk_x16p2vlx2b_x16_x16_sme.
66   ///   * Bias: @ref kai_get_bias_offset_rhs_pack_nxk_x16p2vlx2b_x16_x16_sme.
67   ///   * Output: @ref kai_get_rhs_packed_offset_rhs_pack_nxk_x16p2vlx2b_x16_x16_sme.
68   ///
69   /// @param[in] num_groups Number of groups. It must be 1.
70   /// @param[in] n Number of columns of the output matrix.
71   /// @param[in] k Common dimension between the LHS and RHS matrix.
72   /// @param[in] nr Block size in N dimension. It must match kai_get_n_step_rhs_pack_nxk_x16p2vlx2b_x16_x16_sme().
73   /// @param[in] kr Block size in K dimension. It must be 2.
74   /// @param[in] sr Number of kr splits. It must be 1.
75   /// @param[in] rhs_stride Row stride in bytes of the RHS matrix.
76   /// @param[in] rhs RHS matrix data buffer.
77   /// @param[in] bias Bias matrix data buffer.
78   /// @param[in] scale Scale data buffer. It must be NULL.
79   /// @param[out] rhs_packed Packed RHS matrix.
80   /// @param[in] extra_bytes Extra bytes to append to the end of each row of the packed RHS matrix. It must be 0.
81   /// @param[in] params Extra packing parameters. It must be NULL.
82   void kai_run_rhs_pack_nxk_x16p2vlx2b_x16_x16_sme(
83       size_t num_groups, size_t n, size_t k, size_t nr, size_t kr, size_t sr, size_t rhs_stride, const void* rhs,
84       const void* bias, const void* scale, void* rhs_packed, size_t extra_bytes, const void* params);
85   
86   #ifdef __cplusplus
87   }  // extern "C"
88   #endif  // __cplusplus
