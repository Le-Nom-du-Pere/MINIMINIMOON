27   struct BaseInfo {
28     // profiling info types
29     enum class Type {
30       GRAPH,
31       KERNEL,
32       COPY,
33       CPU_FALLBACK,
34     };
35   
36     BaseInfo(Type infoType, uint64_t Id, const uintptr_t Handle)
37         : type(infoType), profileId(Id), handle(Handle) {}
38     virtual ~BaseInfo() = default;
39   
40     // type of profiling info
41     Type type;
42     // unique profile ID for execution instances of operations or copies
43     uint64_t profileId;
44     // ID generated by os_signpost
45     // since it's possible to use event and interval-based signposts at the
46     // same time, we need separate IDs for each.
47     os_signpost_id_t eventSignpostId = 0, intervalSignpostId = 0;
48     // accumulated GPU time in ms (obtained from CompletionHandler's "GPUEndTime -
49     // GPUStartTime")
50     std::atomic<double> totalGpuTime{0.0};
51     // accumulated Scheduling time in ms (obtained from CompletionHandler's
52     // "KernelEndTime - KernelStartTime")
53     std::atomic<double> totalSchedulingTime{0.0};
54     // indicates if the operation or copy execution has completed
55     std::atomic_bool completed{false};
56     // handle used to identify the profile info's instance (usually the pointer)
57     const uintptr_t handle;
58   
59     virtual const std::string toString(
60         double gpuTime = 0,
61         double schedulingTime = 0) const;
62     // builds a string for a tensor (format: Device:ScalarType[tensor.sizes()])
63     static std::string buildTensorString(
64         const Tensor& tensor,
65         bool includeBufferId = false);
66     static uint64_t getTime() {
67       return clock_gettime_nsec_np(CLOCK_MONOTONIC_RAW);
68     }
69   };
70   
71   struct OperationInfo : BaseInfo {
72     OperationInfo(
73         const void* Handle,
74         bool IsGraph,
75         uint64_t Id,
76         const std::string& StrKey)
77         : BaseInfo(IsGraph ? Type::GRAPH : Type::KERNEL, Id, uintptr_t(Handle)),
78           strKey(StrKey) {}
79   
80     uint64_t runCount = 0;
81     std::string strKey;
82   
83     const std::string toString(double gpuTime = 0, double schedulingTime = 0)
84         const override;
85   
86     // builds a string for a kernel
87     static std::string buildKernelString(
88         const std::string& kernelName,
89         const TensorList& tensors,
90         bool includeBufferId = false) {
91       std::stringstream kernelStr;
92       kernelStr << kernelName;
93       for (const Tensor& tensor : tensors) {
94         kernelStr << ":" << BaseInfo::buildTensorString(tensor, includeBufferId);
95       }
96       return kernelStr.str();
97     }
98   };

100  struct CpuFbInfo : BaseInfo {
101    CpuFbInfo(uint64_t Id, const std::string& OpName)
102        : BaseInfo(Type::CPU_FALLBACK, Id, 0), opName(OpName) {}
103  
104    uint64_t runCount = 0;
105    // the current and total overhead of copies in bytes required to convert the
106    // Op's input tensors from MPS to CPU and then output from CPU back to MPS
107    size_t currentCopyOverhead = 0;
108    size_t totalCopyOverhead = 0;
109    std::string opName;
110    std::string strKey;
111    uint64_t startTime = 0;
112  
113    const std::string toString(double gpuTime = 0, double schedulingTime = 0)
114        const override;
115  
116    void updateCopyOverhead(const TensorList& tensors) {
117      currentCopyOverhead = 0;
118      for (const Tensor& tensor : tensors) {
119        if (tensor.defined()) {
120          currentCopyOverhead += tensor.nbytes();
121        }
122      }
123      totalCopyOverhead += currentCopyOverhead;
124    }
125  };

127  struct CopyInfo : BaseInfo {
128    enum class Kind {
129      MPS_TO_MPS,
130      MPS_TO_CPU,
131      CPU_TO_MPS,
132    };
133  
134    CopyInfo(
135        const void* Handle,
136        size_t Length,
137        uint64_t Id,
138        bool IsNonBlocking,
139        bool UsesBlitter)
140        : BaseInfo(Type::COPY, Id, uintptr_t(Handle)),
141          kind(Kind::MPS_TO_MPS),
142          length(Length),
143          isNonBlocking(IsNonBlocking),
144          usesBlitter(UsesBlitter) {}
145  
146    Kind kind;
147    size_t length;
148    bool isNonBlocking;
149    bool usesBlitter;
150    std::string srcStrKey;
151    std::string dstStrKey;
152    // for copies that don't use blitters, we measure CPU time
153    uint64_t startTime = 0;
154  
155    const std::string toString(double gpuTime = 0, double schedulingTime = 0)
156        const override;
157  
158    static std::string buildTensorString(
159        const void* buffer,
160        const OptionalTensorRef tensor,
161        bool includeBufferId = false);
162  
163    static bool isStorageOnMPS(
164        const void* buffer,
165        const OptionalTensorRef tensor) {
166      if (tensor.has_value()) {
167        return tensor->device().type() == at::kMPS;
168      }
169      TORCH_INTERNAL_ASSERT_DEBUG_ONLY(buffer);
170      // getUnalignedBufferSize() returns -1 if input buffer is not on MPS device
171      return getIMPSAllocator()->getUnalignedBufferSize(buffer) >= 0;
172    }
173  
174    static Kind getCopyKind(
175        const void* srcBuffer,
176        const void* dstBuffer,
177        const OptionalTensorRef srcTensor,
178        const OptionalTensorRef dstTensor) {
179      const bool isSrcOnMPS = isStorageOnMPS(srcBuffer, srcTensor);
180      const bool isDstOnMPS = isStorageOnMPS(dstBuffer, dstTensor);
181      TORCH_INTERNAL_ASSERT_DEBUG_ONLY(isSrcOnMPS || isDstOnMPS);
182      if (isSrcOnMPS && !isDstOnMPS) {
183        return Kind::MPS_TO_CPU;
184      } else if (!isSrcOnMPS && isDstOnMPS) {
185        return Kind::CPU_TO_MPS;
186      }
187      return Kind::MPS_TO_MPS;
188    }
189  };

191  struct CopyStat : CopyInfo {
192    explicit CopyStat(std::string CopyKindStr)
193        : CopyInfo(nullptr, 0, 0, false, false),
194          kindStr(std::move(CopyKindStr)) {}
195    // total number of copies
196    size_t totalCount = 0;
197    // number of Scalar copies (i.e., less than sizeof(int64))
198    size_t scalarsCount = 0;
199    // number of blocking copies (i.e., require syncing to GPU)
200    size_t blockingCount = 0;
201    // number of copies that used memcpy(), instead of Metal Blit Encoder
202    size_t memcpyCount = 0;
203    // accumulated GPU time in ms for the scalar copies
204    std::atomic<double> scalarsGpuTime{0.0};
205    // copy kind in string type
206    std::string kindStr;
207  };

209  class MPSProfiler {
210   public:
211    // lower 16 bits used for profiler options
212    enum ProfileOptions : uint32_t {
213      OPTIONS_NONE = 0,
214      // ALL_* means, all signpost types (RUN_OPERATION|BLIT_COPY|CPU_FALLBACK,
215      // etc.) (used for convenience to not compute bit flags by OR-ing manually)
216      // trace all signpost types using events
217      ALL_SIGNPOST_EVENTS = (1 << 0),
218      // trace all signpost types using intervals
219      ALL_SIGNPOST_INTERVALS = (1 << 1),
220      // always wait for command buffer to finish executing after each commit
221      WAIT_UNTIL_COMPLETED = (1 << 2),
222      // for interval-based signposts, include the scheduling portion of
223      // Graph/Kernel/Copy executions as well.
224      // if flag is disable, only "GPU run time" is included in interval,
225      // and not schedule time.
226      INCLUDE_SCHEDULE_INTERVAL = (1 << 3),
227  
228      // use these if you need to trace signposts types individually (rarely
229      // required) trace signpost using intervals
230      USE_INTERVALS = (1 << 4),
231      // trace signpost by emitting events
232      USE_EVENTS = (1 << 5),
233      // used for sanity check (Change this when new option added)
234      OPTIONS_COUNT = (USE_EVENTS << 1) - 1,
235    };
236  
237    // when adding new types, #define the type string in MPSProfiler.mm as well.
238    // upper 16 bits used for event types
239    enum SignpostTypes : uint32_t {
240      SIGNPOST_NONE = 0,
241      // trace signposts for PyTorch operation executions
242      RUN_OPERATION = (1 << 16),
243      // trace signposts for blitter copies
244      BLIT_COPY = (1 << 17),
245      // trace signposts for ops that fall back on CPU
246      CPU_FALLBACK = (1 << 18),
247      // used for sanity check (Change this when new type added)
248      SIGNPOST_COUNT = (CPU_FALLBACK << 1) - 1,
249    };
250  
251    enum LogOptions : uint32_t {
252      LOG_NONE = 0,
253  
254      // Info logging options during execution
255      // -------------------------------------
256      // prints operation info (id/key/run_count) during execution
257      OPERATION_INFO = (1 << 0),
258      // prints copy info (src/dst tensors/buffers, size, etc.) during execution
259      COPY_INFO = (1 << 1),
260      // prints CPU Fallback info (id/runCount/opName/copyOverhead) during
261      // execution
262      CPU_FALLBACK_INFO = (1 << 2),
263  
264      // Profiling Statistics logging options when process terminates
265      // ------------------------------------------------------------
266      // prints all stats (OPERATION_STATS, COPY_STATS, CPU_FALLBACK_STATS) before
267      // process terminates this is convenient to not combine following stats bit
268      // flags manually
269      ALL_STATS = (1 << 3),
270      // prints operation stats (GPU times, run count, etc.) before process
271      // terminates
272      OPERATION_STATS = (1 << 4),
273      // prints copies stats (GPU times, copy kinds, sizes, etc.) before process
274      // terminates
275      COPY_STATS = (1 << 5),
276      // prints CPU Fallback stats (CPU times, run times, size of MPS<->CPU copies
277      // for tensors, etc.) before process terminates
278      CPU_FALLBACK_STATS = (1 << 6),
279  
280      // Metadata format options when logging the info
281      // ---------------------------------------------
282      // if enabled, includes GPU run time in metadata (i.e.,
283      // GPUEndTime-GPUStartTime from Metal Command Buffers) (e.g., [GPU=0.324
284      // ms])
285      INCLUDE_GPU_TIME = (1 << 7),
286      // if enabled, includes GPU scheduling time in metadata separately
287      // (i.e., KernelEndTime-KernelStartTime from Metal Command Buffers)
288      // e.g., [GPU=0.324 ms, KRNL=0.036 ms]
289      INCLUDE_KERNEL_TIME = (1 << 8),
290      // if enabled, includes the unique buffer ID in metadata for the storage
291      // of a tensor that was allocated on MPSAllocator. This is useful (along
292      // with the EV "PYTORCH_DEBUG_MPS_ALLOCATOR") to identify buffers that are
293      // involved with various operations.
294      INCLUDE_BUFFER_ID = (1 << 9),
295  
296      // used for sanity check (Change this when new option added)
297      LOG_COUNT = (INCLUDE_BUFFER_ID << 1) - 1,
298    };

345    // convenience functions to indicate whether signpost tracing or
346    // logging are enabled for the SignpostTypes
347    bool isOperationProfilingEnabled() const {
348      return (m_signpost_types & SignpostTypes::RUN_OPERATION) ||
349          (m_log_options &
350           (LogOptions::OPERATION_INFO | LogOptions::OPERATION_STATS));
351    }
352    bool isCopyProfilingEnabled() const {
353      return (m_signpost_types & SignpostTypes::BLIT_COPY) ||
354          (m_log_options & (LogOptions::COPY_INFO | LogOptions::COPY_STATS));
355    }
356    bool isCPUFallbackProfilingEnabled() const {
357      return (m_signpost_types & SignpostTypes::CPU_FALLBACK) ||
358          (m_log_options &
359           (LogOptions::CPU_FALLBACK_INFO | LogOptions::CPU_FALLBACK_STATS));
360    }
361    bool isSignpostTracingEnabled() const {
362      return (m_signpost_types != SignpostTypes::SIGNPOST_NONE);
363    }

388    // We use the following lists for two reasons:
389    // 1- for interval-based signposts the "begin" point won't be in same function
390    // as the "end" point where we need to be able to retrieve signpost's info
391    // 2- if Operations info need to be logged when process ends using
392    // LogOptions::OPERATION_INFO.
393  
394    // the pointer key for this map is either "MPSGraph*" or
395    // "id<MTLComputePipelineState>" for Metal Kernels this list is retained and
396    // could be logged along with aggregate profiling numbers when the process
397    // ends.
398    std::unordered_map<uintptr_t, std::unique_ptr<OperationInfo>>
399        m_op_info_list{};
400    // the string key for this map is the op name that we fall back to execute on
401    // CPU this list is retained and could be logged along with aggregate
402    // profiling numbers when the process ends.
403    std::unordered_map<std::string, std::unique_ptr<CpuFbInfo>>
404        m_cpu_fb_info_list{};
405    // this list contains the info for copies, and its key is the unique profileId
406    // which is generated from m_copy_counter
407    // The copyInfo list is not retained.
408    std::unordered_map<uint64_t, std::unique_ptr<CopyInfo>> m_copy_info_list{};
409    // a short list that contains copy stats
410    std::unordered_map<CopyInfo::Kind, std::unique_ptr<CopyStat>>
411        m_copy_stat_list{};
