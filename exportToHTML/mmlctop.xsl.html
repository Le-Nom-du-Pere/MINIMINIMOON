202  <!-- ***************** SEMANTICS HANDLING ***************** -->
203  
204  <!-- This template is called recursively.  At each level   -->
205  <!-- in the source tree it decides whether to strip off,   -->
206  <!-- pass or add semantics at that level (depending on the -->
207  <!-- value of SEM_SW parameter).  Then the actual template -->
208  <!-- is applied to the node.                               -->
209  
210  <xsl:template match = "*" mode = "semantics">
211    <xsl:param name="IN_PREC" select="$NO_PREC"/>
212    <xsl:param name="PARAM" select="$NO_PARAM"/>
213    <xsl:param name="PAREN" select="$PAR_NO"/>
214    <xsl:param name="PAR_NO_IGNORE" select="$YES"/>
215    <xsl:choose>
216      <xsl:when test="$SEM_SW=$SEM_STRIP and self::semantics">
217        <xsl:apply-templates select="annotation-xml[@encoding='MathML']">
218          <xsl:with-param name="IN_PREC" select="$IN_PREC"/>
219          <xsl:with-param name="PARAM" select="$PARAM"/>
220          <xsl:with-param name="PAREN" select="$PAREN"/>
221          <xsl:with-param name="PAR_NO_IGNORE" select="$PAR_NO_IGNORE"/>
222        </xsl:apply-templates>
223      </xsl:when>
224      <xsl:when test="($SEM_SW=$SEM_PASS or $SEM_SW=$SEM_TOP) and self::semantics">
225        <semantics>
226          <xsl:apply-templates select="*[1]">
227          <xsl:with-param name="IN_PREC" select="$IN_PREC"/>
228          <xsl:with-param name="PARAM" select="$PARAM"/>
229          <xsl:with-param name="PAREN" select="$PAREN"/>
230          <xsl:with-param name="PAR_NO_IGNORE" select="$PAR_NO_IGNORE"/>
231        </xsl:apply-templates>
232          <xsl:copy-of select="annotation-xml"/>
233        </semantics>
234      </xsl:when>
235      <xsl:when test="$SEM_SW=$SEM_ALL">
236        <semantics>
237          <xsl:choose>
238            <xsl:when test="self::semantics">
239              <xsl:apply-templates select="*[1]">
240                <xsl:with-param name="IN_PREC" select="$IN_PREC"/>
241                <xsl:with-param name="PARAM" select="$PARAM"/>
242                <xsl:with-param name="PAREN" select="$PAREN"/>
243                <xsl:with-param name="PAR_NO_IGNORE" select="$PAR_NO_IGNORE"/>
244              </xsl:apply-templates>
245              <xsl:copy-of select="annotation-xml"/>
246            </xsl:when>
247            <xsl:otherwise>
248              <xsl:apply-templates select=".">
249                <xsl:with-param name="IN_PREC" select="$IN_PREC"/>
250                <xsl:with-param name="PARAM" select="$PARAM"/>
251                <xsl:with-param name="PAREN" select="$PAREN"/>
252                <xsl:with-param name="PAR_NO_IGNORE" select="$PAR_NO_IGNORE"/>
253              </xsl:apply-templates>
254              <annotation-xml encoding="MathML">
255                <xsl:copy-of select="."/>
256              </annotation-xml>
257            </xsl:otherwise>
258          </xsl:choose>
259        </semantics>
260      </xsl:when>
261      <xsl:when test="($SEM_SW=$SEM_XREF or $SEM_SW=$SEM_XREF_EXT) and @id">
262        <xsl:choose>
263          <xsl:when test="self::semantics">
264            <xsl:copy>
265              <xsl:copy-of select="@*"/>
266              <xsl:attribute name="xref">
267                <xsl:value-of select="@id"/>
268              </xsl:attribute>
269              <xsl:copy-of select="*[1]"/>
270              <xsl:copy-of select="annotation-xml"/>
271            </xsl:copy>
272          </xsl:when>
273          <xsl:otherwise>
274            <xsl:apply-templates select=".">
275              <xsl:with-param name="IN_PREC" select="$IN_PREC"/>
276              <xsl:with-param name="PARAM" select="$PARAM"/>
277              <xsl:with-param name="PAREN" select="$PAREN"/>
278              <xsl:with-param name="PAR_NO_IGNORE" select="$PAR_NO_IGNORE"/>
279            </xsl:apply-templates>
280          </xsl:otherwise>
281        </xsl:choose>
282      </xsl:when>
283      <xsl:otherwise>
284        <xsl:apply-templates select=".">
285          <xsl:with-param name="IN_PREC" select="$IN_PREC"/>
286          <xsl:with-param name="PARAM" select="$PARAM"/>
287          <xsl:with-param name="PAREN" select="$PAREN"/>
288          <xsl:with-param name="PAR_NO_IGNORE" select="$PAR_NO_IGNORE"/>
289        </xsl:apply-templates>
290      </xsl:otherwise>
291    </xsl:choose>
292  </xsl:template>
