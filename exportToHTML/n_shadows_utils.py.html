45   class OutputProp:
46       """ 
47       Output propagation (modeled from shape propagation). 
48    
49       Given a GraphModule and an example input, saves the output flowing 
50       through each node on `node.traced_result`. 
51    
52       Code based on the example from 
53       https://pytorch.org/docs/stable/fx.html#the-interpreter-pattern 
54       """
55   
56       def __init__(self, mod):
57           self.mod = mod
58           self.graph = mod.graph
59           self.modules = dict(self.mod.named_modules())
60   
61       def propagate(self, *args):
62           args_iter = iter(args)
63           env: dict[str, Node] = {}
64   
65           def load_arg(a):
66               return torch.fx.graph.map_arg(a, lambda n: env[n.name])
67   
68           def fetch_attr(target: str):
69               target_atoms = target.split(".")
70               attr_itr = self.mod
71               for i, atom in enumerate(target_atoms):
72                   if not hasattr(attr_itr, atom):
73                       raise RuntimeError(
74                           f"Node referenced nonexistent target {'.'.join(target_atoms[:i])}"
75                       )
76                   attr_itr = getattr(attr_itr, atom)
77               return attr_itr
78   
79           for node in self.graph.nodes:
80               if node.op == "placeholder":
81                   result = next(args_iter)
82               elif node.op == "get_attr":
83                   result = fetch_attr(node.target)
84               elif node.op == "call_function":
85                   result = node.target(*load_arg(node.args), **load_arg(node.kwargs))
86               elif node.op == "call_method":
87                   self_obj, *args = load_arg(node.args)
88                   kwargs = load_arg(node.kwargs)
89                   result = getattr(self_obj, node.target)(*args, **kwargs)
90               elif node.op == "call_module":
91                   result = self.modules[node.target](
92                       *load_arg(node.args), **load_arg(node.kwargs)
93                   )
94   
95               if isinstance(result, torch.Tensor):  # type: ignore[possibly-undefined]
96                   node.traced_result = result
97   
98               env[node.name] = result
99   
100          return None

103  def _get_dedup_subgraphs(matches: dict[str, _MatchResult]) -> dict[str, list[Node]]:
104      # the original matches variable is unique by node, make it unique by subgraph
105      # instead
106      seen_nodes = set()
107      subgraphs_dedup = {}
108  
109      # Dict items are not reversible until Python 3.8, so we hack it
110      # to be compatible with previous Python versions
111      # TODO(future PR): try reversed(list(matches.items()))
112      matches_items_reversed: list[tuple[str, _MatchResult]] = list(
113          reversed(matches.items())
114      )
115  
116      # Note: the order is important.  `matches` currently provides the matches
117      # in reverse order.  We would like to process the matches in non-reverse
118      # order, so that we can create an intuitive naming scheme, such as
119      # naming the first op's submodules `shadow_0_0` through `shadow_0_(n-1)`
120      for name, cur_match in matches_items_reversed:  # type: ignore[call-overload]
121          was_seen = False
122          for node_or_tuple in cur_match[1]:
123              # Cur_match[1] has an unusual type. It says that it's a `List[Node]`,
124              # but it is really not. Furthermore, the contents of this field
125              # can change from match results of multiple nodes of the same pattern
126              #
127              # For example, for conv -> bn -> relu, we see
128              # match_results = {
129              #   'conv': (relu, [(bn, conv), relu], ...),
130              #   'bn': (relu, [(bn, conv), relu], ...),
131              #   'relu': (relu, [(bn, conv), relu], ...),
132              # }
133              #
134              # Ideally we should clean up the `find_matches` function to make
135              # this more intuitive. For the purposes of this prototype, we hack
136              # around it.
137  
138              if isinstance(node_or_tuple, Node):
139                  if node_or_tuple in seen_nodes:
140                      was_seen = True
141                  seen_nodes.add(node_or_tuple)
142  
143              else:
144                  assert isinstance(node_or_tuple, tuple)
145                  for node in node_or_tuple:
146                      assert isinstance(node, Node)
147                      if node in seen_nodes:
148                          was_seen = True
149                      seen_nodes.add(node)
150  
151          if was_seen:
152              continue
153  
154          # Start with the unusual type, convert it to [op_0, ..., op_n]
155          list_of_nodes = []
156  
157          if len(cur_match[1]) == 1:
158              list_of_nodes = cur_match[1]
159          else:
160              assert len(cur_match[1]) == 2
161              # either (a, b), or ((a, b), c) or (c, (a, b))
162              # cannot make any assumptions on order, not clear what the
163              # _find_matches function is doing to populate this
164              # TODO(future PR): make this code less confusing,  see discussion
165              # in https://github.com/pytorch/pytorch/pull/80521/files#r975918836
166  
167              def _order_nodes(node_a, node_b, node_c) -> list[Node]:
168                  nodes = [node_a, node_b, node_c]
169                  first_node = None
170                  mid_node = None
171                  last_node = None
172                  for n in nodes:
173                      prev_n = n.args[0]
174                      next_n = next(iter(n.users))
175                      if prev_n not in nodes:
176                          first_node = n
177                      elif next_n not in nodes:
178                          last_node = n
179                      else:
180                          mid_node = n
181                  assert (
182                      first_node is not None
183                      and mid_node is not None
184                      and last_node is not None
185                  )
186                  assert mid_node.args[0] is first_node
187                  assert last_node.args[0] is mid_node
188                  return [last_node, mid_node, first_node]
189  
190              if isinstance(cur_match[1][0], Node) and isinstance(cur_match[1][1], Node):
191                  # (a, b)
192                  list_of_nodes = cur_match[1]
193              elif isinstance(cur_match[1][0], tuple):
194                  # ((a, b), c)
195                  node_a, node_b = cur_match[1][0]
196                  node_c = cur_match[1][1]
197                  list_of_nodes = _order_nodes(node_a, node_b, node_c)
198              elif isinstance(cur_match[1][1], tuple):
199                  # (a, (b, c))
200                  node_a, node_b = cur_match[1][1]
201                  node_c = cur_match[1][0]
202                  list_of_nodes = _order_nodes(node_a, node_b, node_c)
203  
204          # [node_n, ..., node_0], note that the order is reversed
205          # to make it chronological for simple subgraphs
206          list_of_nodes.reverse()
207          subgraphs_dedup[name] = list_of_nodes
208  
209      return subgraphs_dedup

212  def _get_logger_for_subgraph(
213      model: GraphModule,
214      first_node: Node,
215      last_node: Node,
216      subgraph_idx: int,
217      subgraph_candidate_idx: int,
218      qconfig_str: str,
219      logger_cls: Callable,
220      fqn: Optional[str],
221  ) -> torch.nn.Module:
222      """ 
223      Given a model and a linear subgraph starting from `first_node` and 
224      ending with `last_node`, creates a logger for the end of this 
225      subgraph. 
226      """
227      if fqn is None:
228          fqn = ""
229      logger_mod_orig = logger_cls(
230          first_node.name,  # ref_node_name
231          last_node.name,  # prev_node_name
232          f"subgraph_{subgraph_idx}_{subgraph_candidate_idx}",  # model_name
233          "model",  # ref_name
234          get_target_type_str(last_node, model),  # prev_node_target_type
235          get_target_type_str(first_node, model),  # ref_node_target_type
236          NSSingleResultValuesType.NODE_OUTPUT.value,  # results_type
237          0,  # index_within_arg
238          0,  # index_of_arg
239          fqn,  # fqn
240          qconfig_str,
241      )
242      # Usually we expect the user to add loggers, then calibrate, then convert,
243      # and then populate loggers.  This is why the loggers start disabled.
244      # TODO(future PR): reconsider the design to make this more intuitive.
245      logger_mod_orig.enabled = False
246      return logger_mod_orig

249  def create_submodule_from_subgraph(
250      model: torch.nn.Module,
251      first_node: Node,
252      last_node: Node,
253  ) -> GraphModule:
254      """ 
255      Input: a model, and a linear subgraph within the model from first_node to 
256        last_node. 
257   
258      Output: a new submodule containing a copy of the subgraph, with the inputs 
259        to the first node becoming the inputs to the submodule, and all other 
260        nodes in the subgraph being copied. 
261   
262      Example inputs: 
263   
264      `model`: a module with graph 
265   
266        x0 -> op1 -> x1 -> op2 -> x2 
267               | 
268              arg1 
269   
270      `first_node`: op1 
271      `last_node`: op2 
272   
273      Example output: a new module with graph 
274   
275        input1 -> op1_copy -> x1 -> op2_copy -> output1 
276                     | 
277                    arg1 
278      """
279  
280      #
281      # create a blank GraphModule with an empty graph
282      #
283  
284      class M(torch.nn.Module):
285          def forward(self, x):
286              pass
287  
288      m = M()
289      gm = torch.fx.symbolic_trace(m)
290      g = gm.graph
291      for node in reversed(gm.graph.nodes):
292          g.erase_node(node)
293  
294      #
295      # modify the graph to have a copy of our subgraph
296      #
297  
298      cur_node_orig = first_node
299  
300      cur_name_idx = 0
301  
302      iteration_limit = 100
303      cur_iteration = 0
304  
305      while True:
306          if cur_node_orig is first_node:
307              # we are at the first node, we need to set up graph inputs
308              # TODO(future): some graphs could have placeholders which are unrelated
309              # to the first node, need to handle this
310              cur_args_copy = []
311              cur_kwargs_copy = {}
312              seen_names: set[str] = set()
313              old_name_to_new_node: dict[str, Node] = {}
314  
315              def _add_placeholder(
316                  g: Graph, node: Node, seen_names, old_name_to_new_node
317              ):
318                  # note: for graphs starting with patterns such as `y = x + x`, we
319                  # need to ensure we do not add multiple placeholders with the
320                  # same name
321                  counter = 0
322                  while node.name + "_" + str(counter) in seen_names:
323                      counter += 1
324                  cur_name = node.name + "_" + str(counter)
325                  seen_names.add(cur_name)
326                  placeholder = g.placeholder(cur_name)
327                  old_name_to_new_node[node.name] = placeholder
328                  return placeholder
329  
330              for arg in cur_node_orig.args:
331                  if isinstance(arg, Node):
332                      p = _add_placeholder(g, arg, seen_names, old_name_to_new_node)
333                      cur_args_copy.append(p)
334                  elif isinstance(arg, (list, tuple)):
335                      new_arg = []
336                      for inner_arg in arg:
337                          if isinstance(inner_arg, Node):
338                              new_arg.append(
339                                  _add_placeholder(
340                                      g, inner_arg, seen_names, old_name_to_new_node
341                                  )
342                              )
343                          else:
344                              new_arg.append(inner_arg)
345                      cur_args_copy.append(new_arg)
346                  else:
347                      cur_args_copy.append(arg)
348  
349              # TODO(future PR): handle non-normalized kwargs
350              for kwarg_name, kwarg in cur_node_orig.kwargs.items():
351                  if isinstance(kwarg, Node):
352                      cur_kwargs_copy[kwarg_name] = _add_placeholder(
353                          g, kwarg, seen_names, old_name_to_new_node
354                      )
355                  elif isinstance(kwarg, (list, tuple)):
356                      new_kwarg = []
357                      for inner_kwarg in kwarg:
358                          p = _add_placeholder(
359                              g,
360                              inner_kwarg,  # type: ignore[arg-type]
361                              seen_names,
362                              old_name_to_new_node,
363                          )
364                          new_kwarg.append(p)
365                      cur_kwargs_copy[kwarg_name] = new_kwarg
366                  else:
367                      cur_kwargs_copy[kwarg_name] = kwarg
368  
369              cur_args_copy = tuple(cur_args_copy)  # type: ignore[assignment]
370          else:
371              # we are not at first node, first arg is from the previous node,
372              # and all other args are copied
373  
374              # the current implementation is simplistic and cannot handle
375              # ops with two or more arguments which need to be passed from
376              # the previous op, so we assert them out
377              assert cur_node_orig.target not in BINARY_FUNCTIONS
378  
379              # at this point in the code, cur_node_copy is pointing to the copy
380              # of the previous node
381              # TODO(future PR): this is not handling complicated graphs correctly, need to
382              # look at actual relationships instead of assuming sequential graph
383              # TODO(future PR): this is ignoring kwargs, will need to support kwargs
384              # for any fusion pattern which has them for a node that is not the
385              # first node.
386              cur_args_copy = [cur_node_copy]  # type: ignore[has-type, possibly-undefined]  # noqa: F821
387  
388              if len(cur_node_orig.args) > 1:
389                  for arg in cur_node_orig.args[1:]:
390                      if isinstance(arg, torch.nn.Parameter):
391                          new_arg = arg.detach().clone()  # type: ignore[assignment]
392                          mod_name = f"mod_{cur_name_idx}"
393                          cur_name_idx += 1
394                          setattr(gm, mod_name, new_arg)
395                          new_arg_placeholder = gm.placeholder(mod_name)  # type: ignore[operator]
396                          cur_args_copy.append(new_arg_placeholder)
397                      elif isinstance(arg, (float, int, torch.dtype)):
398                          cur_args_copy.append(arg)
399                      else:
400                          raise AssertionError(f"arg of type {type(arg)} not handled yet")
401              cur_args_copy = tuple(cur_args_copy)  # type: ignore[assignment]
402  
403          # copy the node
404          if cur_node_orig.op == "call_module":
405              orig_mod = getattr_from_fqn(model, cur_node_orig.target)  # type: ignore[arg-type]
406              orig_mod_copy = copy.deepcopy(orig_mod)
407              mod_name = f"mod_{cur_name_idx}"
408              setattr(gm, mod_name, orig_mod_copy)
409              cur_name_idx += 1
410              cur_node_copy = g.call_module(mod_name, cur_args_copy, cur_kwargs_copy)  # type: ignore[possibly-undefined,arg-type]
411  
412          elif cur_node_orig.op == "call_function":
413              cur_node_copy = g.call_function(
414                  cur_node_orig.target,  # type: ignore[arg-type]
415                  cur_args_copy,  # type: ignore[arg-type]
416                  cur_kwargs_copy,  # type: ignore[possibly-undefined]
417              )
418  
419          elif cur_node_orig.op == "call_method":
420              cur_node_copy = g.call_method(
421                  cur_node_orig.target,  # type: ignore[arg-type]
422                  cur_args_copy,  # type: ignore[arg-type]
423                  cur_kwargs_copy,  # type: ignore[possibly-undefined]
424              )
425  
426          else:
427              raise AssertionError(f"{cur_node_orig.op} not supported yet")
428  
429          if cur_node_orig is last_node:
430              break
431  
432          # go to next node
433          assert len(cur_node_orig.users.keys()) == 1, (
434              f"{cur_node_orig} has more than 1 users, not supported yet"
435          )
436          cur_node_orig = next(iter(cur_node_orig.users.keys()))
437          cur_iteration += 1
438          if cur_iteration > iteration_limit:
439              raise AssertionError("iteration limit exceeded")
440  
441      # set up outputs
442      g.output(cur_node_copy)
443  
444      gm.recompile()
445      return gm

448  def create_one_transformed_and_logged_copy_of_subgraph(
449      mt: GraphModule,
450      subgraph_idx: int,
451      subgraph_candidate_idx: int,
452      first_node: Node,
453      last_node: Node,
454      fqn: Optional[str],
455      list_of_node_name_to_qconfig: list[dict[str, QConfigAny]],
456      example_inputs: Any,
457      last_added_shadow_node_list: list[Optional[Node]],
458      custom_prepare_fn: Optional[Callable] = None,
459      custom_prepare_kwargs: Optional[dict[str, Any]] = None,
460  ) -> None:
461      """ 
462      Given a subgraph in `mt` and a subgraph candidate idx, inserts the 
463      subgraph candidate copy and instruments it with loggers. 
464   
465      If subgraph_candidate_idx is 0, this is the baseline fp32 subgraph and we just 
466      add a logger to the end. 
467   
468      If subgraph_candidate_idx is not 0, we create a copy of the subgraph and 
469      prepare it with `prepare_fx`. 
470      """
471  
472      # TODO(future PR): move logger classes to utils to remove circular dependency
473      from torch.ao.ns._numeric_suite_fx import OutputComparisonLogger, OutputLogger
474  
475      if subgraph_candidate_idx == 0:
476          # idx = 0 is the floating point (original) version of the subgraph
477          # We keep the subgraph as is, and add a logger at the end
478  
479          qconfig_str = ""
480          logger_mod_orig = _get_logger_for_subgraph(
481              mt,
482              first_node,
483              last_node,
484              subgraph_idx,
485              subgraph_candidate_idx,
486              qconfig_str,
487              OutputLogger,
488              fqn,
489          )
490  
491          attr_name = _get_attr_name(subgraph_idx, subgraph_candidate_idx)
492          assert not hasattr(mt, attr_name)
493          setattr(mt, attr_name, logger_mod_orig)
494          with mt.graph.inserting_after(last_node):
495              new_node = mt.graph.call_module(attr_name, args=(last_node,), kwargs={})
496              last_added_shadow_node_list[0] = new_node
497  
498      else:
499          # idx > 0 means we have a candidate qconfig to try, so we need
500          # to make a copy of the subgraph, feed it with the right inputs,
501          # and add a logger at the end
502  
503          # get the qconfig
504          # subtract one because the first candidate is the floating point
505          # version of the subgraph
506          node_name_to_qconfig = list_of_node_name_to_qconfig[subgraph_candidate_idx - 1]
507          qconfig = node_name_to_qconfig[first_node.name]
508  
509          # if no quantization is requested, skip
510          # TODO(future PR): deduplicate equivalent qconfigs that come from
511          #   different qconfig mapping objects
512          if qconfig is None:
513              return
514  
515          qconfig_mapping = QConfigMapping().set_global(qconfig)
516  
517          # create a copy of the submodule, wrapped in a separate module
518          orig_mod_copy_wrapped = create_submodule_from_subgraph(
519              mt, first_node, last_node
520          )
521  
522          # add a call to prepare_fx on the wrapper module
523          if custom_prepare_fn is None:
524              orig_mod_copy_wrapped = torch.ao.quantization.quantize_fx.prepare_fx(
525                  orig_mod_copy_wrapped, qconfig_mapping, example_inputs=example_inputs
526              )
527          else:
528              if custom_prepare_kwargs is None:
529                  custom_prepare_kwargs = {}
530              for kwarg_name in [
531                  "example_inputs",
532                  "prepare_custom_config",
533                  "qconfig_mapping",
534              ]:
535                  assert kwarg_name not in custom_prepare_kwargs, (
536                      f"cannot specify {kwarg_name} in custom_prepare_kwargs"
537                  )
538              prepare_kwargs: dict[str, Any] = {
539                  "example_inputs": example_inputs,
540                  "qconfig_mapping": qconfig_mapping,
541              }
542              prepare_kwargs.update(custom_prepare_kwargs)
543              orig_mod_copy_wrapped = custom_prepare_fn(
544                  orig_mod_copy_wrapped, **prepare_kwargs
545              )
546  
547          # attach the wrapper to the model
548          attr_name = _get_attr_wrapper_name(subgraph_idx, subgraph_candidate_idx)
549          assert not hasattr(mt, attr_name)
550          setattr(mt, attr_name, orig_mod_copy_wrapped)
551  
552          # add a call to the wrapper module from the parent graph
553          insert_after_node = last_added_shadow_node_list[0]
554          with mt.graph.inserting_after(insert_after_node):
555              # TODO(future PR): handle fusion patterns where non-first nodes
556              # need inputs
557  
558              # pass in all node args and kwargs
559  
560              new_args = []
561              for arg in first_node.args:
562                  if isinstance(arg, Node):
563                      new_args.append(arg)
564                  elif (
565                      isinstance(arg, (list, tuple))
566                      and len(arg)
567                      and isinstance(arg[0], Node)
568                  ):
569                      new_args.extend(
570                          inner_arg for inner_arg in arg if isinstance(inner_arg, Node)
571                      )
572  
573              new_kwargs = {}
574              for name, old_kwarg in first_node.kwargs.items():
575                  if isinstance(old_kwarg, Node):
576                      new_kwargs[name] = old_kwarg
577                  elif isinstance(old_kwarg, (list, tuple)) and len(old_kwarg):
578                      # TODO(future PR): clarify why we are adding kwargs to args
579                      new_args.extend(old_kwarg)  # type: ignore[arg-type]
580  
581              new_args = tuple(new_args)  # type: ignore[assignment]
582  
583              new_node = mt.graph.call_module(attr_name, args=new_args, kwargs=new_kwargs)  # type: ignore[arg-type]
584  
585          # add a logger to parent graph to observe the shadow wrapper
586          logger_mod_orig = _get_logger_for_subgraph(
587              mt,
588              first_node,
589              last_node,
590              subgraph_idx,
591              subgraph_candidate_idx,
592              str(qconfig),
593              OutputComparisonLogger,
594              fqn,
595          )
596  
597          attr_name = _get_attr_name(subgraph_idx, subgraph_candidate_idx)
598          assert not hasattr(mt, attr_name)
599          setattr(mt, attr_name, logger_mod_orig)
600          with mt.graph.inserting_after(new_node):
601              logger = mt.graph.call_module(
602                  attr_name, args=(new_node, last_node), kwargs={}
603              )
604              last_added_shadow_node_list[0] = logger
605  
606      mt.recompile()

609  def create_n_transformed_and_logged_copies_of_subgraph(
610      mt: GraphModule,
611      subgraph_idx: int,
612      match_name: str,
613      nodes_in_this_subgraph: list[Any],
614      qconfig_mappings: list[QConfigMapping],
615      list_of_node_name_to_qconfig: list[dict[str, QConfigAny]],
616      custom_prepare_fn: Optional[Callable] = None,
617      custom_prepare_kwargs: Optional[dict[str, Any]] = None,
618  ) -> None:
619      """ 
620      Given a model `mt` and a subgraph_idx, creates the needed copies 
621      of the subgraph for all qconfigs, and instruments them with loggers. 
622      """
623      # for now, assume that
624      # 1. the first node has one input
625      # 2. the last node has one output
626  
627      # for now, ignore all subgraphs that contain non-nodes (tuples, etc)
628      # TODO(future PR): implement this
629      if any(not isinstance(node, Node) for node in nodes_in_this_subgraph):
630          return
631  
632      first_node = nodes_in_this_subgraph[0]
633      last_node = nodes_in_this_subgraph[-1]
634      # We used output propagation to populate example values on each
635      # node. Use the example values from the previous node as the input
636      # to the current node.
637      prev_node = get_normalized_nth_input(first_node, mt, 0)
638      if isinstance(prev_node, list):
639          example_inputs = [x.traced_result for x in prev_node]
640      elif isinstance(prev_node, tuple):
641          example_inputs = (x.traced_result for x in prev_node)  # type: ignore[assignment]
642      else:
643          # currently some customer models do not have a traced_result in
644          # every node, so we have to guard for this case since we cannot
645          # quantize without an example input
646          # TODO(future PR): add a test case for this once we have an easy
647          # repro, see https://github.com/pytorch/pytorch/pull/80521/files#r975940489
648          # for additional context
649          if hasattr(prev_node, "traced_result"):
650              example_inputs = (prev_node.traced_result,)  # type: ignore[attr-defined, assignment]
651          else:
652              print(
653                  "unable to get example input for node "
654                  + f"{first_node.format_node()}, skipping"
655              )
656              return
657  
658      # If there are no quantization configs for this subgraph, skip adding
659      # loggers. This reduces memory usage for models where not all layers are
660      # quantized.
661      # TODO(future): consider making this configurable
662      found_at_least_one_qconfig = False
663      for subgraph_candidate_idx in range(len(qconfig_mappings) + 1):
664          if subgraph_candidate_idx == 0:
665              # fp32 baseline does not need a qconfig
666              continue
667  
668          # a. we have N shadows, so len(qconfig_mappings) is N
669          # b. we will have the fp32 layer + N shadows, so overall number of
670          #    (original_op) + (*shadows) will be N+1
671          # c. since `subgraph_candidate_idx` represents (b), we need
672          #    to subtract 1 to query from (a)
673          node_name_to_qconfig = list_of_node_name_to_qconfig[subgraph_candidate_idx - 1]
674          qconfig = node_name_to_qconfig[first_node.name]
675          if qconfig is not None:
676              found_at_least_one_qconfig = True
677              break
678      if not found_at_least_one_qconfig:
679          print(
680              "unable to find at least one qconfig for node "
681              + f"{first_node.format_node()}, skipping"
682          )
683          return
684  
685      fqn = _maybe_get_fqn(first_node, mt)
686  
687      # We want the results to contain the subgraphs in natural order,
688      # and the graph to also contain shadow wrappers and shadow loggers
689      # in natural order.
690      # If we just iterate in reverse, the graph will be in natural
691      # order but the eventual results will be in reverse order.
692      # So, we keep track of the last shadow logger we added and
693      # always insert after it.
694      last_added_shadow_node_list: list[Optional[Node]] = [None]
695      for subgraph_candidate_idx in range(len(qconfig_mappings) + 1):
696          create_one_transformed_and_logged_copy_of_subgraph(
697              mt,
698              subgraph_idx,
699              subgraph_candidate_idx,
700              first_node,
701              last_node,
702              fqn,
703              list_of_node_name_to_qconfig,
704              example_inputs,
705              last_added_shadow_node_list,
706              custom_prepare_fn,
707              custom_prepare_kwargs,
708          )

711  def create_add_loggers_graph(
712      model: GraphModule,
713      subgraphs_dedup: dict[str, list[Node]],
714      qconfig_mapping: QConfigMapping,
715      node_name_to_qconfig: dict[str, QConfigAny],
716  ) -> None:
717      r""" 
718      Given a model, a model graph partition (currently a set of matched 
719      subgraphs) and instructions how to transform each subgraph 
720      (currently quantizing it according to qconfig_mapping), modifies 
721      the model graph to create an alternate path through the original graph, 
722      with each of the subgraphs quantized.  This is useful to compare 
723      propagation error of a transformation such as quantization. 
724   
725      For example, given layer op0 and op1, there are four cases when handling op1: 
726      1. op0 and op1 quantized 
727      2. op0 and op1 unquantized 
728      3. op0 quantized, op1 unquantized 
729      4. op0 unquantized, op1 quantized 
730   
731      Example input, case 1: 
732   
733      .. code:: 
734   
735        x0_0 -> op0_0 -> x1_0 -> log -----> op1_0 -> x2_0 -> log 
736         \                        \          \                 \       # noqa: W605 
737           ---> op0_1 -> x1_1 ----> clog    op1_1 -> x2_1 ----> clog 
738   
739      Example output, case 1: 
740   
741      .. code:: 
742   
743        x0_0 -> op0_0 -> x1_0 -> log -----> op1_0 -> x2_0 -> log 
744         \                        \                           \        # noqa: W605 
745           ---> op0_1 -> x1_1 ----> clog -> op1_1 -> x2_1 ----> clog 
746   
747      """
748      # TODO(future PR): move logger classes to utils to remove circular dependency
749      from torch.ao.ns._numeric_suite_fx import OutputComparisonLogger, OutputLogger
750  
751      def _get_subgraph_containing_node(node, subgraphs_dedup):
752          for subgraph in subgraphs_dedup.values():
753              if node in subgraph:
754                  return subgraph
755          return None
756  
757      # First, we need to create shadow branches, going from
758      #
759      #   x0 -> op0 -> x1 -> ...
760      #
761      #
762      # to
763      #
764      #   x0 -> op0_0 -> x1_0 -> log -> ...
765      #    \                     \
766      #      -> op0_1 -> x1_1 -> clog
767      #
768      # Later, the outputs of each shadow will be rerouted to calculate
769      # propagation error.
770  
771      # Note: we cannot iterate over matched subgraphs because some nodes
772      # may not be matched. So, we iterate over nodes in the graph, and
773      # associate them to matched subgraphs if possible.
774  
775      nodes_to_skip = set()
776      # for each subgraph, save a mapping from first node of subgraph
777      # to first and last node of the shadow of this subgraph
778      orig_first_node_to_shadow_in_node = {}
779      orig_first_node_to_shadow_out_node = {}
780      # need to record original list because we will mutate the graph as we go
781      orig_nodes = list(model.graph.nodes)  # type: ignore[union-attr, arg-type]
782      cur_subgraph_idx = 0
783      for n in orig_nodes:
784          if n.op in ("placeholder", "get_attr", "output") or n in nodes_to_skip:
785              continue
786  
787          maybe_subgraph = _get_subgraph_containing_node(n, subgraphs_dedup)
788          insert_submodule_copy = False
789          if maybe_subgraph is not None:
790              first_node, last_node = maybe_subgraph[0], maybe_subgraph[-1]
791              nodes_to_skip.update(maybe_subgraph)
792              qconfig = node_name_to_qconfig[first_node.name]
793              if qconfig is not None:
794                  insert_submodule_copy = True
795          else:
796              first_node, last_node = n, n
797  
798          if insert_submodule_copy:
799              match_name = first_node.name
800              create_n_transformed_and_logged_copies_of_subgraph(
801                  model,
802                  cur_subgraph_idx,
803                  match_name,
804                  maybe_subgraph,
805                  [qconfig_mapping],
806                  [node_name_to_qconfig],
807                  None,
808                  None,  # type: ignore[arg-type]
809              )
810              # find the created shadow module and record it so we
811              # can find it easily in step 2
812              expected_shadow_target = f"shadow_wrapper_{cur_subgraph_idx}_1"
813              new_shadow_mod = None
814              for maybe_shadow_mod in model.graph.nodes:
815                  if (
816                      maybe_shadow_mod.op == "call_module"
817                      and maybe_shadow_mod.target == expected_shadow_target
818                  ):
819                      new_shadow_mod = maybe_shadow_mod
820                      break
821              assert new_shadow_mod is not None
822              orig_first_node_to_shadow_in_node[first_node] = new_shadow_mod
823              orig_first_node_to_shadow_out_node[first_node] = new_shadow_mod
824  
825          else:
826              # create a copy of the subgraph by only copying FX nodes
827              # but not copying any parameters, to minimize memory usage
828              subgraph_to_use = (
829                  maybe_subgraph if maybe_subgraph is not None else [first_node]
830              )
831  
832              # add a regular logger after last_node
833              qconfig_str = ""
834              subgraph_candidate_idx = 0
835              fqn = _maybe_get_fqn(first_node, model)
836              logger_mod_orig = _get_logger_for_subgraph(
837                  model,
838                  first_node,
839                  last_node,
840                  cur_subgraph_idx,
841                  subgraph_candidate_idx,
842                  qconfig_str,
843                  OutputLogger,
844                  fqn,
845              )
846              attr_name = _get_attr_name(cur_subgraph_idx, subgraph_candidate_idx)
847              assert not hasattr(model, attr_name)
848              setattr(model, attr_name, logger_mod_orig)
849              insertion_point = last_node
850              with model.graph.inserting_after(insertion_point):
851                  logger = model.graph.call_module(
852                      attr_name, args=(last_node,), kwargs={}
853                  )
854                  insertion_point = logger
855  
856              # create a copy of the subgraph
857              cur_node_orig = first_node
858              cur_node_copy = None
859              first_node_copy = None
860              while cur_node_orig in subgraph_to_use:
861                  # TODO(future PR): make this support all possible args/kwargs
862                  if cur_node_orig is first_node:
863                      new_args = cur_node_orig.args
864                      new_kwargs = cur_node_orig.kwargs
865                  else:
866                      first_arg_for_copy: Optional[Node] = cur_node_copy
867                      new_args = (first_arg_for_copy, *cur_node_orig.args[1:])
868                      new_kwargs = cur_node_orig.kwargs
869                  # make a copy of cur_node_orig
870                  with model.graph.inserting_after(insertion_point):
871                      cur_node_copy = model.graph.create_node(
872                          cur_node_orig.op,
873                          cur_node_orig.target,
874                          new_args,
875                          new_kwargs,
876                          # cur_node_orig.name,  # TODO(future PR): set name explicitly
877                      )
878                      if first_node_copy is None:
879                          first_node_copy = cur_node_copy
880                  # since now only linear subgraphs are supported, all nodes
881                  # except the last one must have only one user
882                  if cur_node_orig != last_node:
883                      assert len(cur_node_orig.users.keys()) == 1
884                  cur_node_orig = next(iter(cur_node_orig.users.keys()))
885                  assert not cur_node_orig.name.startswith(SHADOW_NODE_NAME_PREFIX)
886                  insertion_point = cur_node_copy
887  
888              # add a comparison logger after last_node's copy
889              subgraph_candidate_idx = 1
890              logger_mod_orig = _get_logger_for_subgraph(
891                  model,
892                  first_node,
893                  last_node,
894                  cur_subgraph_idx,
895                  subgraph_candidate_idx,
896                  qconfig_str,
897                  OutputComparisonLogger,
898                  fqn,
899              )
900              attr_name = _get_attr_name(cur_subgraph_idx, subgraph_candidate_idx)
901              assert not hasattr(model, attr_name)
902              setattr(model, attr_name, logger_mod_orig)
903              with model.graph.inserting_after(insertion_point):
904                  logger = model.graph.call_module(
905                      attr_name, args=(cur_node_copy, last_node), kwargs={}
906                  )
907  
908              # save the final node so we can use it in step 2
909              orig_first_node_to_shadow_in_node[first_node] = first_node_copy
910              orig_first_node_to_shadow_out_node[first_node] = cur_node_copy
911  
912          cur_subgraph_idx += 1
913  
914      model.recompile()
915  
916      # Now, we go from
917      #
918      #   x0 -> op0_0 -> x1_0 -> log -> x1 -> op1_0 -> ...
919      #    \                     \       \
920      #      -> op0_1 -> x1_1 -> clog      -> op1_1 -> ...
921      #
922      # to
923      #
924      #   x0 -> op0_0 -> x1_0 -> log --> x1_0 -> op1_0 -> ...
925      #    \                     \
926      #      -> op0_1 -> x1_1 -> clog -> x1_1 -> op1_1 -> ...
927      #
928      # sample values of key internal variables for the example above:
929      #
930      #   orig_first_node_to_shadow_in_node = {op0_0: op0_1, op1_0: op1_1}
931      #   orig_first_node_to_shadow_out_node = {op0_0: op0_1, op1_0: op1_1}
932      #
933      # note: for subgraphs with more than one node, in_node will be different
934      # compared to out_node
935  
936      nodes_to_skip = set()
937      for n in orig_nodes:
938          if n.op in ("placeholder", "get_attr", "output") or n in nodes_to_skip:
939              continue
940  
941          maybe_subgraph = _get_subgraph_containing_node(n, subgraphs_dedup)
942          if maybe_subgraph is not None:
943              first_node, last_node = maybe_subgraph[0], maybe_subgraph[-1]
944              nodes_to_skip.update(maybe_subgraph)
945          else:
946              first_node, last_node = n, n
947  
948          def maybe_remap_node_to_shadow(node):
949              """ 
950              If unshadowed `node` has a shadow version, return that. If not, 
951              return `node`. 
952              """
953              if not isinstance(node, Node):
954                  # handle scalars
955                  return node
956  
957              if node.op in ("placeholder", "get_attr"):
958                  return node
959  
960              # Find the shadowed version of this arg from the previous
961              # subgraph. For this, we need to:
962              # 1. navigate to the first node of the previous subgraph
963              # 2. get the output of the shadow wrapper which has (1) as an input
964  
965              # For now, assume the arg is in matched subgraphs. In the
966              # future we may have to handle the case where this is not true.
967              prev_subgraph = _get_subgraph_containing_node(node, subgraphs_dedup)
968              if prev_subgraph is None:
969                  prev_subgraph = [node]
970              prev_first_node = prev_subgraph[0]
971              prev_shadow_output = orig_first_node_to_shadow_out_node[prev_first_node]
972              return prev_shadow_output
973  
974          cur_shadow_input = orig_first_node_to_shadow_in_node[first_node]
975          assert cur_shadow_input is not None
976          cur_shadow_input.args = tree_map(
977              maybe_remap_node_to_shadow, cur_shadow_input.args
978          )
979          cur_shadow_input.kwargs = tree_map(
980              maybe_remap_node_to_shadow, cur_shadow_input.kwargs
981          )
982  
983          model.recompile()

986  def _get_weight_info_from_shadow_wrapper(shadow_wrapper: torch.nn.Module):
987      # input: shadow wrapper module
988      # output if shadow wrapper module has a weighted op:
989      #   (quantize_fn, (quantize_fn_args))
990      # output if shadow wrapper module doesn't have a weighted op:
991      #   None
992  
993      # For now, assume that the weight is the second input
994      # to the shadow module. If that changes, we can fix it later.
995      placeholders_seen = 0
996      for shadow_n in shadow_wrapper.graph.nodes:  # type: ignore[union-attr]
997          if shadow_n.op != "placeholder":
998              continue
999  
1000         placeholders_seen += 1
1001         if placeholders_seen != 2:
1002             continue
1003 
1004         # the subgraph looks like
1005         #
1006         #   _input_scale_1 = self._input_scale_1
1007         #   _input_zero_point_1 = self._input_zero_point_1
1008         #   quantize_per_channel = torch.quantize_per_channel(
1009         #       w2_0, _input_scale_1, _input_zero_point_1,
1010         #       0, torch.qint8)
1011         #
1012         #  we have `w2_0`, and are navigating this subgraph
1013         #  to get `_input_scale_1` and `_input_zero_point_1`
1014 
1015         assert len(shadow_n.users) == 1
1016         quant_node = next(iter(shadow_n.users.keys()))
1017         new_args: Any = None
1018         if quant_node.target == torch.quantize_per_channel:
1019             _weight, scale_node, zp_node, axis, dtype = quant_node.args
1020             scale_val = getattr_from_fqn(shadow_wrapper, scale_node.target)
1021             zp_val = getattr_from_fqn(shadow_wrapper, zp_node.target)
1022             new_args = (scale_val, zp_val, axis, dtype)
1023         else:
1024             assert quant_node.target == torch.quantize_per_tensor
1025             _weight, scale_node, zp_node, dtype = quant_node.args
1026             scale_val = getattr_from_fqn(shadow_wrapper, scale_node.target)
1027             zp_val = getattr_from_fqn(shadow_wrapper, zp_node.target)
1028             new_args = (scale_val, zp_val, dtype)
1029         return (quant_node.target, new_args)
1030 
1031     return None

1034 def extract_weight_comparison(m: GraphModule) -> NSResultsType:
1035     # example graph:
1036     #
1037     #   w1 = self.w1
1038     #   b1 = self.b1
1039     #   linear = torch._C._nn.linear(x, w1, b1)
1040     #   shadow_0_0 = self.shadow_0_0(linear)
1041     #   shadow_wrapper_0_1 = self.shadow_wrapper_0_1(x, w1, b1)
1042     #   shadow_0_1 = self.shadow_0_1(shadow_wrapper_0_1, linear)
1043     #
1044     # algorithm:
1045     # 1. for each call_function node matching our allowlist:
1046     # 2.   if corresponding shadow wrapper exists, extract the weight pair
1047     #
1048     # Note: this is not super robust, but that's ok because this is
1049     # just for legacy customers who depend on the previous two-model version
1050     # of this API. TBD if we need to make this robust.
1051     # Note: modules are not supported, since existing customers only
1052     # use functions.
1053 
1054     # TODO(future PR): move this to config
1055     weighted_ops = {
1056         torch.nn.functional.linear,
1057     }
1058 
1059     results: NSResultsType = {"model": {NSSingleResultValuesType.WEIGHT.value: {}}}
1060 
1061     for n in m.graph.nodes:  # type: ignore[union-attr]
1062         if not (n.op == "call_function" and n.target in weighted_ops):
1063             continue
1064 
1065         # Check if we have a corresponding shadow wrapper
1066         # TODO(future PR, if needed): support kwargs
1067         # TODO(future PR, if needed): support multiple shadow users
1068         first_arg = n.args[0]
1069         shadow_wrapper_node = None
1070         for user in first_arg.users:
1071             # TODO(before land): fix string match
1072             if user.op == "call_module" and user.target.startswith("shadow_wrapper"):
1073                 shadow_wrapper_node = user
1074                 break
1075 
1076         if shadow_wrapper_node is None:
1077             continue
1078 
1079         shadow_wrapper = getattr_from_fqn(m, shadow_wrapper_node.target)  # type: ignore[arg-type]
1080         weight_info = _get_weight_info_from_shadow_wrapper(shadow_wrapper)
1081         if weight_info is None:
1082             continue
1083 
1084         # get weight
1085         w_node = n.args[1]
1086         w_obj = getattr_from_fqn(m, w_node.target).detach()
1087 
1088         # get a quantized version of weight
1089         quant_fn, quant_fn_args_except_first = weight_info
1090         new_args = (w_obj, *quant_fn_args_except_first)
1091         w_obj_q = quant_fn(*new_args)
1092 
1093         # add a comparison
1094         ref_node_name = n.name
1095         prev_node_name = n.name
1096         ref_node_type = get_target_type_str(n, m)
1097         prev_node_type = ref_node_type
1098         fqn = None
1099         if hasattr(m, "_node_name_to_scope"):
1100             fqn = m._node_name_to_scope[n.name][0]  # type: ignore[index]
1101         comparison = torch.ao.ns.fx.utils.compute_sqnr(w_obj, w_obj_q)
1102         result_fp32 = {
1103             "res_type": NSSingleResultValuesType.WEIGHT.value,
1104             "values": [w_obj],
1105             "prev_node_name": prev_node_name,
1106             "prev_node_target_type": prev_node_type,
1107             "ref_node_name": ref_node_name,
1108             "ref_node_target_type": ref_node_type,
1109             "index_within_arg": 0,
1110             "index_of_arg": 0,
1111             "fqn": fqn,
1112             "qconfig_str": "",
1113             "comparisons": [comparison],
1114             "comparison_fn_name": "sqnr",
1115         }
1116         result_q = {
1117             "res_type": NSSingleResultValuesType.WEIGHT.value,
1118             "values": [w_obj_q],
1119             "prev_node_name": prev_node_name,
1120             "prev_node_target_type": prev_node_type,
1121             "ref_node_name": ref_node_name,
1122             "ref_node_target_type": ref_node_type,
1123             "index_within_arg": 0,
1124             "index_of_arg": 0,
1125             "fqn": fqn,
1126             "qconfig_str": "",
1127             "comparisons": [comparison],
1128             "comparison_fn_name": "sqnr",
1129         }
1130 
1131         # go from subgraph_n_1 to subgraph_n_0
1132         _1, _2, node_idx, _3 = shadow_wrapper_node.target.split("_")
1133         name_fp32 = f"subgraph_{node_idx}_0"
1134         name_q = f"subgraph_{node_idx}_1"
1135 
1136         results["model"][NSSingleResultValuesType.WEIGHT.value][name_fp32] = [
1137             result_fp32
1138         ]
1139         results["model"][NSSingleResultValuesType.WEIGHT.value][name_q] = [result_q]
1140 
1141     return results

1145 def group_results_by_subgraph(results: NSResultsType) -> Any:
1146     """ 
1147     Creates a comparison of results 
1148  
1149     Input: 
1150  
1151     { 
1152       'model': { 
1153         'node_output': { 
1154           'subgraph_0_0': [ 
1155             'values': [torch.tensor(...), ...], ... 
1156             'ref_node_name': ..., 
1157             'ref_node_target_type': ..., 
1158             'qconfig_str': ..., 
1159             'comparisons': [], ... 
1160             'comparison_fn_name': '', 
1161             'fqn': '...', 
1162           ], 
1163           'subgraph_0_1': [ 
1164             'values': [torch.tensor(...), ...], ... 
1165             'ref_node_name': ..., 
1166             'ref_node_target_type': ..., 
1167             'qconfig_str': ..., 
1168             'comparisons': [torch.tensor(...), ...], ... 
1169             'comparison_fn_name': '...', 
1170             'fqn': '...', 
1171           ], 
1172           ... 
1173         }, 
1174       }, 
1175     } 
1176  
1177     Output: 
1178     { 
1179       'subgraph_0': { 
1180         '0': { 
1181           'ref_node_name': '...', 
1182           'ref_node_target_type': ..., 
1183           'values': [torch.tensor(...), ...], 
1184           'qconfig_str': None, 
1185           'comparisons': [torch.tensor(...), ...], ... 
1186           'comparison_fn_name': '...', 
1187           'fqn': '...', 
1188         }, 
1189         '1': { 
1190           'ref_node_name': '...', 
1191           'ref_node_target_type': ..., 
1192           'values': [torch.tensor(...), ...], 
1193           'qconfig_str': '...', 
1194           'comparisons': [torch.tensor(...), ...], ... 
1195           'comparison_fn_name': '...', 
1196           'fqn': '...', 
1197         }, 
1198       }, 
1199     } 
1200  
1201     """
1202     subgraph_name_to_subgraph_results: Any = collections.defaultdict(dict)

1238 def create_results_comparison(
1239     results_grouped,
1240 ) -> Any:
1241     """ 
1242     Input: 
1243  
1244     { 
1245       'subgraph_0': { 
1246         '0': { 
1247           'ref_node_name': '...', 
1248           'ref_node_target_type': ..., 
1249           'values': [torch.tensor(...), ...], 
1250           'qconfig_str': '', 
1251           'comparisons': [], 
1252           'comparison_fn_name': '', 
1253           'fqn': '...', 
1254         }, 
1255         '1': { 
1256           'ref_node_name': '...', 
1257           'ref_node_target_type': ..., 
1258           'values': [torch.tensor(...), ...], 
1259           'qconfig_str': '...', 
1260           'comparisons': [torch.tensor(...), ...], 
1261           'comparison_fn_name': 'sqnr', 
1262           'fqn': '...', 
1263         }, 
1264       }, 
1265     } 
1266  
1267     Output: 
1268     { 
1269       'subgraph_0': { 
1270         'ref_node_name': '...', 
1271         'ref_node_target_type': '...', 
1272         'fqn': '...', 
1273         'candidates': { 
1274           '1': { 
1275             'qconfig_str': ..., 
1276             'comparison_fn_name': 'sqnr', 
1277             'cmp_raw': [..., ...], 
1278             'cmp_mean': ..., 
1279           }, 
1280           ..., 
1281         }, 
1282       }, 
1283     } 
1284     """
1285 
1286     results_comparison = {}
1287 
1288     for subgraph_name, subgraph_results in results_grouped.items():
1289         candidates = {}
1290         for subgraph_inner_name, subgraph_inner_result in subgraph_results.items():
1291             # skip comparing baseline to baseline
1292             if subgraph_inner_name == "0":
1293                 continue
1294 
1295             # we expect the comparisons to be precalculated from
1296             # calibration, so we just fetch them here
1297             cmp_raw = subgraph_inner_result["comparisons"]
1298             cmp_raw_tensor = torch.stack(cmp_raw)
1299 
1300             candidates[subgraph_inner_name] = {
1301                 "qconfig_str": subgraph_inner_result["qconfig_str"],
1302                 "comparison_fn_name": subgraph_inner_result["comparison_fn_name"],
1303                 "cmp_raw": cmp_raw_tensor,
1304                 "cmp_mean": torch.mean(cmp_raw_tensor),
1305             }
1306 
1307         results_comparison[subgraph_name] = {
1308             "ref_node_name": subgraph_results["0"]["ref_node_name"],
1309             "ref_node_target_type": subgraph_results["0"]["ref_node_target_type"],
1310             "fqn": subgraph_results["0"]["fqn"],
1311             "candidates": candidates,
1312         }
1313 
1314     return results_comparison
