10   // Shared implementation between Vectorized<c10::Half> and
11   // Vectorized<c10::BFloat16>. Uses CRTP to allow derived class
12   // customization.
13   template <
14       typename VecT,
15       typename ValueT,
16       template <int, bool> typename BlendRegs,
17       typename Derived>
18   struct Vectorized16 {
19    protected:
20     VecT values;
21   
22    public:
23     using value_type = ValueT;
24     using size_type = int;
25     static constexpr size_type size() {
26       static_assert(sizeof(VecT) == 8 * sizeof(value_type));
27       return 8;
28     }
29   
30    protected:
31     Derived map2(
32         const Derived& second,
33         value_type (*const f)(value_type, value_type)) const {
34       __at_align__ value_type tmp_first[size()];
35       __at_align__ value_type tmp_second[size()];
36       static_cast<const Derived*>(this)->store(
37           tmp_first); // store this to tmp_first
38       second.store(tmp_second);
39       for (const auto i : c10::irange(size())) {
40         tmp_first[i] = f(tmp_first[i], tmp_second[i]);
41       }
42       return Derived::loadu(tmp_first);
43     }
44   
45    public:
46     Vectorized16() = default;
47     Vectorized16(VecT v) : values(v) {}
48   
49     operator VecT() const {
50       return values;
51     }
52   
53     template <int64_t mask>
54     static Derived blend(const Derived& a, const Derived& b) {
55       Derived vec;
56       vec.values = BlendRegs < 0,
57       (mask & 0x01) != 0 > ::impl(a.values, b.values, vec.values);
58       vec.values = BlendRegs < 1,
59       (mask & 0x02) != 0 > ::impl(a.values, b.values, vec.values);
60       vec.values = BlendRegs < 2,
61       (mask & 0x04) != 0 > ::impl(a.values, b.values, vec.values);
62       vec.values = BlendRegs < 3,
63       (mask & 0x08) != 0 > ::impl(a.values, b.values, vec.values);
64   
65       vec.values = BlendRegs < 4,
66       (mask & 0x10) != 0 > ::impl(a.values, b.values, vec.values);
67       vec.values = BlendRegs < 5,
68       (mask & 0x20) != 0 > ::impl(a.values, b.values, vec.values);
69       vec.values = BlendRegs < 6,
70       (mask & 0x40) != 0 > ::impl(a.values, b.values, vec.values);
71       vec.values = BlendRegs < 7,
72       (mask & 0x80) != 0 > ::impl(a.values, b.values, vec.values);
73   
74       return vec;
75     }
76   
77     template <typename step_t>
78     static Derived arange(
79         value_type base = 0,
80         step_t step = static_cast<step_t>(1)) {
81       const Derived base_vec(base);
82       const Derived step_vec(step);
83       const Derived step_sizes(
84           value_type(0),
85           value_type(1),
86           value_type(2),
87           value_type(3),
88           value_type(4),
89           value_type(5),
90           value_type(6),
91           value_type(7));
92       return fmadd(step_sizes, step_vec, base_vec);
93     }
94   
95     // Very slow implementation of indexing.
96     // Only required because vec256_qint refers to this.
97     // Once we specialize that implementation for ARM
98     // this should be removed. TODO (kimishpatel)
99     value_type operator[](int idx) const {
100      __at_align__ value_type tmp[size()];
101      static_cast<const Derived*>(this)->store(tmp);
102      return tmp[idx];
103    }
104  
105    int zero_mask() const {
106      __at_align__ value_type tmp[size()];
107      static_cast<const Derived*>(this)->store(tmp);
108      int mask = 0;
109      for (int i = 0; i < size(); ++i) {
110        if (tmp[i] == 0) {
111          mask |= (1 << i);
112        }
113      }
114      return mask;
115    }
116  
117    Derived map(value_type (*const f)(value_type)) const {
118      __at_align__ value_type tmp[size()];
119      static_cast<const Derived*>(this)->store(tmp);
120      for (const auto i : c10::irange(size())) {
121        tmp[i] = f(tmp[i]);
122      }
123      return Derived::loadu(tmp);
124    }
125  
126    Derived angle() const {
127      auto zero = Derived(0);
128      auto pi = Derived(c10::pi<value_type>);
129      auto tmp =
130          Derived::blendv(zero, pi, *static_cast<const Derived*>(this) < zero);
131      return Derived::blendv(
132          tmp,
133          *static_cast<const Derived*>(this),
134          static_cast<const Derived*>(this)->isnan());
135    }
136    Derived real() const {
137      return *this;
138    }
139    Derived imag() const {
140      return Derived(0);
141    }
142    Derived conj() const {
143      return *this;
144    }
