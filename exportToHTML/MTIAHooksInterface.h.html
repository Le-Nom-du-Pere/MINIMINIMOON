21       "The MTIA backend requires MTIA extension for PyTorch;"

25   struct TORCH_API MTIAHooksInterface : AcceleratorHooksInterface {
26   // this fails the implementation if MTIAHooks functions are called, but
27   // MTIA backend is not present.
28   #define FAIL_MTIAHOOKS_FUNC(func) \
29     TORCH_CHECK(false, "Cannot execute ", func, "() without MTIA backend.");
30   
31     ~MTIAHooksInterface() override = default;
32   
33     void init() const override {
34       // Avoid logging here, since MTIA needs init devices first then it will know
35       // how many devices are available. Make it as no-op if mtia extension is not
36       // dynamically loaded.
37       return;
38     }
39   
40     virtual bool hasMTIA() const {
41       return false;
42     }
43   
44     DeviceIndex deviceCount() const override {
45       return 0;
46     }
47   
48     virtual void deviceSynchronize(c10::DeviceIndex /*device_index*/) const {
49       FAIL_MTIAHOOKS_FUNC(__func__);
50     }
51   
52     virtual std::string showConfig() const {
53       FAIL_MTIAHOOKS_FUNC(__func__);
54     }
55   
56     bool hasPrimaryContext(DeviceIndex /*device_index*/) const override {
57       return false;
58     }
59   
60     void setCurrentDevice(DeviceIndex /*device*/) const override {
61       FAIL_MTIAHOOKS_FUNC(__func__);
62     }
63   
64     DeviceIndex getCurrentDevice() const override {
65       FAIL_MTIAHOOKS_FUNC(__func__);
66       return -1;
67     }
68   
69     DeviceIndex exchangeDevice(DeviceIndex /*device*/) const override {
70       FAIL_MTIAHOOKS_FUNC(__func__);
71       return -1;
72     }
73   
74     DeviceIndex maybeExchangeDevice(DeviceIndex /*device*/) const override {
75       FAIL_MTIAHOOKS_FUNC(__func__);
76       return -1;
77     }
78   
79     virtual c10::Stream getCurrentStream(DeviceIndex /*device*/) const {
80       FAIL_MTIAHOOKS_FUNC(__func__);
81       return c10::Stream::unpack3(-1, 0, c10::DeviceType::MTIA);
82     }
83   
84     virtual int64_t getCurrentRawStream(DeviceIndex /*device*/) const {
85       FAIL_MTIAHOOKS_FUNC(__func__);
86       return -1;
87     }
88   
89     virtual c10::Stream getDefaultStream(DeviceIndex /*device*/) const {
90       FAIL_MTIAHOOKS_FUNC(__func__);
91       return c10::Stream::unpack3(-1, 0, c10::DeviceType::MTIA);
92     }
93   
94     virtual void setCurrentStream(const c10::Stream& /*stream*/ ) const {
95       FAIL_MTIAHOOKS_FUNC(__func__);
96     }
97   
98     bool isPinnedPtr(const void* /*data*/) const override {
99       return false;
100    }
101  
102    Allocator* getPinnedMemoryAllocator() const override {
103      FAIL_MTIAHOOKS_FUNC(__func__);
104      return nullptr;
105    }
106  
107    virtual PyObject* memoryStats(DeviceIndex /*device*/) const {
108      FAIL_MTIAHOOKS_FUNC(__func__);
109      return nullptr;
110    }
111  
112    virtual PyObject* getDeviceCapability(DeviceIndex /*device*/) const {
113      FAIL_MTIAHOOKS_FUNC(__func__);
114      return nullptr;
115    }
116  
117    virtual PyObject* getDeviceProperties(DeviceIndex device) const {
118      FAIL_MTIAHOOKS_FUNC(__func__);
119      return nullptr;
120    }
121  
122    virtual void emptyCache() const {
123      FAIL_MTIAHOOKS_FUNC(__func__);
124    }
125  
126  
127    virtual void recordMemoryHistory(
128      const std::optional<std::string>& /*enabled*/,
129      const std::string& /*stacks*/,
130      size_t /*max_entries*/) const {
131      FAIL_MTIAHOOKS_FUNC(__func__);
132    }
133  
134    virtual PyObject* memorySnapshot(const std::optional<std::string>& local_path) const {
135      FAIL_MTIAHOOKS_FUNC(__func__);
136      return nullptr;
137    }
138  
139    virtual DeviceIndex getDeviceCount() const {
140      FAIL_MTIAHOOKS_FUNC(__func__);
141      return 0;
142    }
143  
144    virtual void resetPeakMemoryStats(DeviceIndex /*device*/) const {
145      FAIL_MTIAHOOKS_FUNC(__func__);
146    }
147  
148    virtual void attachOutOfMemoryObserver(PyObject* observer) const {
149      FAIL_MTIAHOOKS_FUNC(__func__);
150      return;
151    }
152  };
