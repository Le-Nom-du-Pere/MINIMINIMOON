3    """ 
4    Side effect tracking and management for TorchDynamo's compilation system. 
5     
6    This module provides infrastructure for tracking and managing side effects that occur 
7    during symbolic execution, including: 
8     
9    - Tracking mutations to objects, attributes, and variables 
10   - Managing context changes (cell variables, global namespace modifications) 
11   - Handling aliasing and object identity preservation 
12   - Managing stack frame state and local variable changes 
13   - Tracking function calls with side effects 
14    
15   Key classes: 
16   - SideEffects: Main container for tracking all side effects during execution 
17   - MutableSideEffects: Specialization for mutable object tracking 
18   - AttributeMutation/ValueMutation: Track specific types of mutations 
19   - Various specialized side effect classes for different scenarios 
20    
21   The side effect system ensures that mutations performed during symbolic execution 
22   are properly replayed during runtime, maintaining the correctness of compiled code 
23   while enabling optimizations where safe. 
24   """

64   def _manual_dict_setitem(dict_from, dict_to, mro_index):
65       # Carefully calls the dict or OrderedDict `clear` or `__setitem__`. We have
66       # to be careful because we don't want to trigger the user defined object
67       # setitem or clear. The mro_index is used to find the dict/OrderedDict from
68       # the class mro.
69       dict_class = type(dict_to).__mro__[mro_index]
70       dict_class.clear(dict_to)
71       for k, v in dict_from.items():
72           dict_class.__setitem__(dict_to, k, v)
73

75   def _manual_list_update(list_from, list_to):
76       list.clear(list_to)
77       list.extend(list_to, list_from)

80   class SideEffects:
81       """ 
82       Maintain records of mutations and provide methods to apply them during code generation. 
83    
84       Handles tracking and applying side effects during PyTorch Dynamo compilation, 
85       maintaining Python semantics by managing mutations, attribute modifications, 
86       and other side effects that occur during program execution. 
87    
88       Key responsibilities: 
89       - Tracks mutations to Python objects, lists, and dictionaries that need to be 
90       applied after an FX graph is run. 
91       - Manages attribute modifications and deletions 
92       - Handles tensor hooks and backward pass state 
93       - Tracks cell variable mutations and global variable changes 
94       - Ensures correct ordering and application of side effects after graph execution 
95    
96       This ensures that optimized code behaves identically to the original Python code with 
97       respect to object mutations and other side effects. 
98       """

100      id_to_variable: dict[int, VariableTracker]
101      store_attr_mutations: dict[VariableTracker, dict[str, VariableTracker]]
102      keepalive: list[Any]

104      def __init__(
105          self,
106          output_graph,
107          id_to_variable=None,
108          store_attr_mutations=None,
109          keepalive=None,
110          save_for_backward=None,
111          tensor_hooks=None,
112      ):
113          super().__init__()
114          self.output_graph_weakref = weakref.ref(output_graph)
115          self.id_to_variable = id_to_variable or {}
116          self.store_attr_mutations = store_attr_mutations or {}
117          self.keepalive = keepalive or []
118          self.save_for_backward = save_for_backward or []
119          self.tensor_hooks = tensor_hooks or {}
120          # Used by MappingProxyVariable to graph break in case of any mutated
121          # dict
122          self._has_existing_dict_mutation = False
123          # Track Compiled Autograd final callbacks that must be called at the end of Compiled Autograd backward graph.
124          # Only applicable if this graph is created from Dynamo tracing in Compiled Autograd.
125          self.ca_final_callbacks_var = None

128          assert isinstance(other, SideEffects)
129          # NB: do NOT test keepalive
130          return (
131              self.id_to_variable == other.id_to_variable
132              and self.store_attr_mutations == other.store_attr_mutations
133              and self.save_for_backward == other.save_for_backward
134              and self.tensor_hooks == other.tensor_hooks

137      def diff(self, other: "SideEffects") -> Optional[str]:
138          if self.id_to_variable != other.id_to_variable:
139              sk_itv = self.id_to_variable.keys()
140              ok_itv = other.id_to_variable.keys()
141              if sk_itv != ok_itv:
142                  return f"id_to_variable keys: {sk_itv} != {ok_itv}"
143              # Feel free to augment this with more fancy diffing logic
144              # if needed for debugging
145              return "id_to_variable: unknown diff"
146          elif self.store_attr_mutations != other.store_attr_mutations:
147              sk_sam = self.store_attr_mutations.keys()
148              ok_sam = other.store_attr_mutations.keys()
149              if sk_sam != ok_sam:
150                  return f"store_attr_mutations keys: {sk_sam} != {ok_sam}"
151              return "store_attr_mutations: unknown diff"
152          elif self.save_for_backward != other.save_for_backward:
153              return "save_for_backward"
154          elif self.tensor_hooks != other.tensor_hooks:
155              return "tensor_hooks"
156          else:
157              return None

201      def check_allowed_side_effect(self, item):
202          from torch._dynamo.variables.misc import AutogradFunctionContextVariable
203  
204          # People do things like self.dim = dim inside autograd.Function.
205          # These are benign.
206          if isinstance(item, AutogradFunctionContextVariable):
207              return True
208          if self.should_allow_externally_visible_side_effects_in_subtracer():
209              return True
210          if self.should_allow_side_effects_under_checkpoint():
211              return True
212          if self.is_reconstructing_generator():
213              # This is missing the case where one mutates a tensor. See
214              # test_generator.py::test_reconstruct_generator_tensor_mutation
215              raise SideEffectsError(
216                  "Cannot reconstruct a generator with variable mutations. "
217                  "Dynamo needs to fully exhaust the generator, which may cause "
218                  "unintended variable modifications."
219              )
220          if not is_side_effect_safe(item.mutation_type):
221              # TODO plumb HOP information here
222              unimplemented_v2(
223                  gb_type="HigherOrderOperator: Mutating a variable not in the current scope (SideEffects)",
224                  context="",
225                  explanation="This is not supported.",
226                  hints=[],
227              )

229      def store_attr(self, item: VariableTracker, name: str, value: VariableTracker):
230          assert self.is_attribute_mutation(item)
231          self.check_allowed_side_effect(item)
232          if item not in self.store_attr_mutations:
233              self.store_attr_mutations[item] = {}
234          self.store_attr_mutations[item][name] = value
235  
236      def load_attr(self, item, name, deleted_ok=False, check=False):
237          if check:
238              assert self.is_attribute_mutation(item)
239          result = self.store_attr_mutations[item][name]
240          if not deleted_ok and isinstance(result, variables.DeletedVariable):
241              unimplemented_v2(
242                  gb_type="Attempted to read a deleted variable",
243                  context=f"item: {item}, name: {name}",
244                  explanation="",
245                  hints=[*graph_break_hints.USER_ERROR],
246              )
247          return result

249      def store_cell(self, cellvar, value):
250          if cellvar.is_immutable():
251              unimplemented_v2(
252                  gb_type="Write to immutable cell",
253                  context=f"cellvar: {cellvar}, value: {value}",
254                  explanation="Dynamo doesn't support writing to immutable/sourceless cell variables.",
255                  hints=[*graph_break_hints.DIFFICULT],
256              )
257          assert isinstance(cellvar, variables.CellVariable)
258          assert isinstance(value, variables.VariableTracker)
259          self.store_attr(cellvar, "cell_contents", value)

261      def load_cell(self, cellvar):
262          assert isinstance(cellvar, variables.CellVariable)
263          if self.has_pending_mutation_of_attr(cellvar, "cell_contents"):
264              return self.load_attr(cellvar, "cell_contents", check=False)
265          if cellvar.pre_existing_contents:
266              return cellvar.pre_existing_contents
267          unimplemented_v2(
268              gb_type="Read uninitialized cell",
269              context=str(cellvar),
270              explanation="Attempted to read a cell variable that has not been populated yet.",
271              hints=[*graph_break_hints.USER_ERROR],
272          )

308      def is_modified(self, item):
309          if item.is_immutable():
310              return False
311          if isinstance(item.mutation_type, (AttributeMutationNew, ValueMutationNew)):
312              return True
313  
314          if isinstance(item, variables.UserDefinedObjectVariable):
315              # Checks if the underlying dict or tuple vt has been modified
316              return item in self.store_attr_mutations or item.is_underlying_vt_modified(
317                  self
318              )
319  
320          if self.is_attribute_mutation(item):
321              return item in self.store_attr_mutations
322  
323          return item.mutation_type.is_modified

325      def _track_obj(
326          self,
327          item: Any,
328          variable: VariableTracker,
329          mutation_type_cls=ValueMutationExisting,
330      ):
331          """Start tracking an existing or new variable for mutation"""
332          if id(item) in self.id_to_variable:
333              raise AssertionError(
334                  f"{variable} is already tracked for mutation. This could be "
335                  "because you are not using VariableBuilder to construct "
336                  "the variable tracker. "
337                  f"Source of new object: {variable.source}. "
338                  f"Source of previously tracked object: {self.id_to_variable[id(item)].source}."
339              )
340  
341          variable.mutation_type = mutation_type_cls()
342          self.id_to_variable[id(item)] = variable
343          self.keepalive.append(item)
344          return variable

360          self,
361          cls_source: Source,
362          user_cls: Any,
363          variable_cls: Any,
364          options,
365      ):
366          if user_cls is torch.autograd.function.FunctionCtx:

380      def get_variable_cls(self, user_cls):
381          from torch.overrides import TorchFunctionMode
382  
383          from .variables.ctx_manager import GenericContextWrappingVariable
384          from .variables.torch_function import TorchFunctionModeVariable
385          from .variables.user_defined import is_forbidden_context_manager
386  
387          variable_cls: type[variables.UserDefinedObjectVariable] = (
388              variables.UserDefinedObjectVariable
389          )
390          if issubclass(
391              user_cls, TorchFunctionMode
392          ) and TorchFunctionModeVariable.is_supported_torch_function_mode(user_cls):
393              variable_cls = TorchFunctionModeVariable
394          elif (
395              hasattr(user_cls, "__enter__")
396              and hasattr(user_cls, "__exit__")
397              and not is_forbidden_context_manager(user_cls)
398          ):
399              variable_cls = GenericContextWrappingVariable
400          elif issubclass(user_cls, torch.nn.Module):
401              variable_cls = variables.UnspecializedNNModuleVariable
402          elif issubclass(user_cls, (dict, collections.OrderedDict)):
403              variable_cls = variables.UserDefinedDictVariable
404          elif issubclass(user_cls, tuple):
405              variable_cls = variables.UserDefinedTupleVariable
406          elif issubclass(user_cls, list):
407              variable_cls = variables.UserDefinedListVariable
408          elif issubclass(user_cls, MutableMapping):
409              variable_cls = variables.MutableMappingVariable
410          elif is_frozen_dataclass(user_cls):
411              variable_cls = FrozenDataClassVariable
412          elif issubclass(user_cls, BaseException):
413              variable_cls = variables.UserDefinedExceptionObjectVariable
414          assert issubclass(variable_cls, variables.UserDefinedObjectVariable)
415          return variable_cls

417      def get_example_value(
418          self,
419          base_cls_vt,
420          cls_vt,
421          init_args,
422      ):
423          user_cls = cls_vt.value
424          if issubclass(user_cls, torch.nn.Module):
425              # TODO(anijain2305) - Is it possible to remove this specialization?
426              obj = nn_module_new(user_cls)
427          else:
428              if isinstance(base_cls_vt, variables.BuiltinVariable):
429                  base_cls = base_cls_vt.fn
430              elif isinstance(base_cls_vt, variables.UserDefinedClassVariable):
431                  base_cls = base_cls_vt.value
432              else:
433                  raise RuntimeError(f"Unexpected base_cls_vt {base_cls_vt}")
434  
435              assert variables.UserDefinedClassVariable.is_supported_new_method(
436                  base_cls.__new__
437              )
438              # TODO(anijain2305) - Consider adding get_example_value method to
439              # each VT to get an example value for all args. As we expand the
440              # scope to other __new__ methods, we might need to call __new__ with
441              # init_args (like functools.partial)
442              # init_args = [arg.get_example_value() for arg in init_args]
443              # obj = base_cls.__new__(user_cls, *init_args)
444  
445              obj = base_cls.__new__(user_cls)
446          return obj

448      def track_new_user_defined_object(
449          self,
450          base_cls_vt,
451          cls_vt,
452          init_args,
453      ):
454          """ 
455          Creates a UserDefinedObjectVariable (or its subclass) variable tracker 
456          and mark it for attribute mutation tracking. 
457   
458          Also records the variable trackers to call __new__ method on 
459          reconstruction. Roughly, the reconstruction looks like this 
460              base_cls_vt.__new__(user_cls, *init_args) 
461          """
462          cls_source = cls_vt.source
463          user_cls = cls_vt.value
464          variable_cls = self.get_variable_cls(user_cls)
465          obj = self.get_example_value(base_cls_vt, cls_vt, init_args)
466  
467          variable = variable_cls(
468              obj,
469              cls_source=cls_vt.source,
470              base_cls_vt=base_cls_vt,
471              init_args=init_args,
472              mutation_type=AttributeMutationNew(cls_source),
473          )
474          self.id_to_variable[id(obj)] = variable
475          self.keepalive.append(obj)
476          return variable

516      def track_tensor_variables_from_runahead_side_effects(self, other):
517          # In higher order ops we want to keep track of tensors seen in the
518          # speculate_subgraph so that we don't lift them again as a new input in
519          # other speculate_subgraph or in the root tracer.
520          for other_item in other.keepalive:
521              other_id = id(other_item)
522              other_variable = other.id_to_variable[other_id]
523              if other_id not in self.id_to_variable and isinstance(
524                  other_variable, variables.TensorVariable
525              ):
526                  self.track_object_existing(other_item, other_variable)

528      def prune_dead_object_new(self, tx):
529          # Avoid VT cycles from e.g., recursive function.
530          visited: set[VariableTracker] = set()
531          live_new_objects: set[VariableTracker] = set()
532  
533          def visit(var: VariableTracker):
534              if var in visited:
535                  return
536              visited.add(var)
537              # Object may have been mutated, store this mutation.
538              if isinstance(var.mutation_type, AttributeMutationNew):
539                  live_new_objects.add(var)
540              # It's possible that we have mutated the value of this variable
541              # to be another one. The new value is in store_attr_mutations.
542              # Also recurse through the new value to detect alive AttributeMutationNew.
543              if var in self.store_attr_mutations:
544                  VariableTracker.visit(
545                      visit,  # noqa: F821
546                      self.store_attr_mutations[var],
547                  )
548  
549          def is_live(var: VariableTracker):
550              if isinstance(var.mutation_type, AttributeMutationNew):
551                  return var in live_new_objects
552              return True
553  
554          pre_existing_vars = [
555              var
556              for var in self.id_to_variable.values()
557              if not isinstance(var.mutation_type, AttributeMutationNew)
558          ]
559  
560          # The only live side effects come from returns (tx.stack), any intermediates
561          # during a graph break (tx.symbolic_locals), and mutation on pre-existing variables.
562          # Recursively visit Variables and see if any of them have been mutated.
563          VariableTracker.visit(
564              visit,
565              # TODO track from all possible sources.
566              (
567                  tx.stack,
568                  tx.symbolic_locals,
569                  pre_existing_vars,
570                  tx.output.backward_state,
571                  self.tensor_hooks,
572              ),
573          )
574          # Manually release the self-referential function, which indirectly
575          # captures certain `VariableTracker` and affects parts of PT test/logic
576          # that are sensitive to when certain objects get released.
577          del visit
578  
579          # NB: cell variable handling.is tricky.
580          # cell variables must stay alive if any NestedUserFunctionVariable
581          # are live. "visit"-ing the NestedUserFunctionVariable visits
582          # the .closures field, from which we will see if we need to keep
583          # any mutations to cell variables alive.
584  
585          self.id_to_variable = {
586              k: v for k, v in self.id_to_variable.items() if is_live(v)
587          }
588          self.store_attr_mutations = {
589              k: v for k, v in self.store_attr_mutations.items() if is_live(k)
590          }

592      def mutation(self, var):
593          self.check_allowed_side_effect(var)
594          if isinstance(var.mutation_type, ValueMutationExisting):
595              var.mutation_type.is_modified = True
596          if (
597              var.source
598              and isinstance(var, variables.ConstDictVariable)
599              and not isinstance(var, variables.SetVariable)
600          ):
601              self._has_existing_dict_mutation = True

603      def has_existing_dict_mutation(self):
604          return self._has_existing_dict_mutation

606      def _get_modified_vars(self):
607          return [var for var in self.id_to_variable.values() if self.is_modified(var)]

609      def codegen_save_tempvars(self, cg: PyCodegen):
610          # We must codegen modified VT to their source by default, so that
611          # mutation and aliasing are properly accounted for.
612          #
613          # Since newly constructed objects don't have a source, we manually
614          # codegen their construction and store them to a newly assigned local
615          # source. Note that `ValueMutationNew` isn't tracked by SideEffects.
616          for var in self._get_modified_vars():
617              if not isinstance(var.mutation_type, AttributeMutationNew):
618                  assert var.source is not None
619                  continue
620  
621              if isinstance(var, variables.CellVariable):
622                  # Cells created in the root frame are created either by
623                  # `MAKE_CELL` or by them being in `co_cellvars`, so we only emit
624                  # `make_cell` for the non-root-frame cells here.
625                  # TODO generalize this so we never need to call `make_cell`.
626                  if var.local_name is None:
627                      cg.add_push_null(
628                          lambda: cg.load_import_from(utils.__name__, "make_cell")
629                      )
630                      cg.extend_output(create_call_function(0, False))
631                      cg.add_cache(var)
632                      var.source = LocalSource(cg.tempvars[var])  # type: ignore[attr-defined]
633                  elif var.source is None:
634                      var.source = LocalCellSource(var.local_name)
635              elif isinstance(var, variables.TensorVariable):
636                  # NOTE: for historical reasons we never assigned local sources
637                  # to newly constructed tensor object, so we keep it that way.
638                  # They are always loaded from output of the fx graph, so one can
639                  # think of it as having a "OutputGraphSource" for codegen
640                  # purposes.
641                  #
642                  # However, tensor subclass objects are different, because the
643                  # reconstruction logic in `PyCodegen` loads the data tensor from
644                  # graph output and then calls `as_subclass`, meaning we must
645                  # assign a source to it to ensure we only reconstruct one
646                  # subclass instance.
647                  if isinstance(
648                      var, variables.torch_function.TensorWithTFOverrideVariable
649                  ):
650                      # Don't codegen from temp source assigned from the 1st pass.
651                      cg(var, allow_cache=False)
652                      cg.add_cache(var)
653                      # `add_cache` generates STORE and consumes TOS, but we never
654                      # cleared it. TODO move this call into `add_cache`
655                      cg.clear_tos()
656                      var.source = LocalSource(cg.tempvars[var])
657              elif isinstance(var, variables.AutogradFunctionContextVariable):
658                  unimplemented_v2(
659                      gb_type="AutogradFunctionContextVariable escaped Dynamo-traced region",
660                      context="",
661                      explanation="We cannot reconstruct a torch.autograd.Function's context object.",
662                      hints=[],
663                  )
664              else:
665                  # Reconstruct the bytecode for
666                  # base_cls.__new__(user_cls, *args)
667                  if isinstance(var, variables.UserDefinedObjectVariable):
668  
669                      def load_new_method():
670                          assert var.base_cls_vt is not None
671                          cg(var.base_cls_vt)  # type: ignore[attr-defined]
672                          cg.extend_output([cg.create_load_attr("__new__")])
673  
674                      cg.add_push_null(load_new_method)
675                  else:
676                      cg.add_push_null(
677                          lambda: cg.load_import_from(utils.__name__, "object_new")
678                      )
679                  cg(var.mutation_type.cls_source)
680  
681                  # Generate the args to the __new__ method
682                  for arg in var.init_args:
683                      cg(arg)
684  
685                  # Call the __new__ method
686                  cg.extend_output(create_call_function(1 + len(var.init_args), False))
687  
688                  cg.add_cache(var)
689                  var.source = LocalSource(cg.tempvars[var])
690  
691          for ctx, args in self.save_for_backward:
692              cg(ctx.source)
693              cg.load_method("save_for_backward")
694              for arg in args:
695                  cg(arg)
696              cg.extend_output(
697                  [
698                      *create_call_method(len(args)),
699                      create_instruction("POP_TOP"),
700                  ]
701              )

703      def register_hook(self, tensor, hook, handle, name):
704          assert isinstance(tensor, variables.TensorVariable)
705          assert isinstance(hook, variables.VariableTracker)
706          assert (
707              isinstance(handle, variables.RemovableHandleVariable)
708              and handle.is_mutable()
709          )
710          assert hasattr(torch.Tensor, name)
711          idx = len(self.tensor_hooks.keys())
712          # duplicate index possible because of self.remove_hook()
713          while idx in self.tensor_hooks:
714              idx += 1
715          self.tensor_hooks[idx] = (tensor, hook, handle, name)
716          assert not handle.idx
717          handle.idx = idx
718  
719      def remove_hook(self, idx):
720          del self.tensor_hooks[idx]

722      def codegen_hooks(self, cg):
723          for (
724              tensor,
725              hook,
726              handle,
727              name,
728          ) in self.tensor_hooks.values():
729              # Note: [On tensor.register_hook]
730              #
731              # register_hook on a tensor, AKA backward hooks, have slightly nuanced differences in how they are implemented
732              # when it comes to hooks on objects with sources (inputs, params) vs objects without sources (intermediaries).
733              #
734              # For tensors with a source, we bypass direct inclusion of register_hook calls in the graph.
735              # Instead, these are tracked and stashed as a global variable, enabling their association with tensors in
736              # the residuals. During dynamo's frame creation, these hooks are invoked seamlessly on known reconstructible/fetch-able
737              # tensors. Because a source indicates knowledge of this object outside the torch compile region, and
738              # because we are running residuals firmly before .backward() can be run, it is sound to invoke
739              # `register_hook` on a known tensor.
740              #
741              # For tensors without a source, we support a limited subset of hooks. Global functions only, and
742              # compiled_autograd must be enabled or we will graph break.
743              #
744              # Handling the Handle: When a user retains the register_hook result in a handle, we intercept the
745              # STORE_FAST operation to record the user-designated local variable name. This ensures the reconstructed
746              # bytecode retains this name. If no handle is defined, we simply pop the generated value to keep the
747              # stack intact.
748              #
749              # Dynamo Tensor Hooks Workflow:
750              # - Functions passed to register_hook are lifted globally.
751              # - For tensors with sources:
752              #   - In the "side_effects" phase of codegen, we iterate over tensors with hooks to:
753              #     - Generate the tensor.
754              #     - Issue a register_hook call on the tensor, linking to the globally stored function.
755              #     - Incorporate a handle if one was established in the eager phase.
756              #  - For tensors without sources:
757              #    - We don't generate any instructions for registering a hook.
758              #    - Handles from intermediary hooks are NYI.
759              #    - We produce a call function that utilizes the trace_wrapped higher order op, closing over it.
760              #    - We then manually insert the call function above into the graph.
761              # - The handle's exact user-specified name, "user_code_variable_name", is discerned and associated during STORE_FAST.
762              assert tensor.source, "Hooks on non input tensors NYI - should not get here"
763  
764              def gen_fn():
765                  cg(tensor)
766                  cg.extend_output([cg.create_load_attr(name)])
767  
768              cg.add_push_null(gen_fn)
769              cg(hook)
770              cg.extend_output(create_call_function(1, False))
771  
772              # Adding the handle to the cache means RemovableHandleVariable().reconstruct() will
773              # be associated with the return value of register_hook().  This consumes the top of stack.
774              cg.add_cache(handle)
775

785      def codegen_update_mutated(self, cg: PyCodegen):
786          suffixes = []
787          for var in self._get_modified_vars():
788              if isinstance(var, variables.ListVariable):
789                  # old[:] = new
790                  cg(var, allow_cache=False)  # Don't codegen via source
791                  cg(var.source)  # type: ignore[attr-defined]
792                  cg.extend_output(
793                      [
794                          cg.create_load_const(None),
795                          cg.create_load_const(None),
796                          create_instruction("BUILD_SLICE", arg=2),
797                      ]
798                  )
799                  suffixes.append([create_instruction("STORE_SUBSCR")])
800              elif isinstance(var, variables.lists.DequeVariable):
801                  # For limited maxlen, the order of operations matter for side
802                  # effect, but we currently don't track the order, so no support.
803                  if not (
804                      isinstance(var.maxlen, variables.ConstantVariable)
805                      and var.maxlen.value is None
806                  ):
807                      unimplemented_v2(
808                          gb_type="Side effect on existing deque with limited maxlen",
809                          context="",
810                          explanation="This is not supported.",
811                          hints=[
812                              "Don't use a deque with `maxlen` specified.",
813                          ],
814                      )
815  
816                  # old.extend(new), this runs last
817                  cg(var.source)
818                  cg.load_method("extend")
819                  cg(var, allow_cache=False)  # Don't codegen via source
820                  suffixes.append(
821                      [
822                          *create_call_method(1),
823                          create_instruction("POP_TOP"),
824                      ]
825                  )
826  
827                  # old.clear(), this runs first
828                  cg(var.source)
829                  cg.load_method("clear")
830                  suffixes.append(
831                      [
832                          *create_call_method(0),
833                          create_instruction("POP_TOP"),
834                      ]
835                  )
836  
837              elif isinstance(var, variables.ConstDictVariable):
838                  # Reconstruct works as follow:
839                  # (1) Skip codegen if there are no new items
840                  # (2) codegen(...) each pair of key/value
841                  # (3) create a new dictionary with the pairs of key/values above
842                  # (4) clear the original dictionary
843                  #   + only if a key was removed from the input dict
844                  # (5) update the original dictionary with the dict created in (2)
845  
846                  if var.has_new_items():
847                      cg(var.source)  # type: ignore[attr-defined]
848                      cg.load_method("update")
849                      cg(var, allow_cache=False)  # Don't codegen via source
850  
851                      if var.should_reconstruct_all:
852                          cg(var.source)  # type: ignore[attr-defined]
853                          cg.load_method("clear")
854  
855                      suffixes.append(
856                          [
857                              *create_call_method(1),  # update
858                              create_instruction("POP_TOP"),
859                          ]
860                      )
861  
862                      if var.should_reconstruct_all:
863                          # clear will appear before "update" as the suffixes are
864                          # applied in reverse order.
865                          suffixes.append(
866                              [
867                                  *create_call_method(0),  # clear
868                                  create_instruction("POP_TOP"),
869                              ]
870                          )
871  
872              elif isinstance(
873                  var, variables.torch_function.TorchFunctionModeStackVariable
874              ):
875                  # Needed in the finally block for stack restoration
876                  cg.add_push_null(
877                      lambda: cg.load_import_from(
878                          utils.__name__, "get_torch_function_mode_stack"
879                      )
880                  )
881                  cg.call_function(0, False)
882                  name = variables.torch_function.get_prev_stack_var_name()
883                  cg.code_options["co_varnames"] += (name,)
884                  cg.append_output(create_instruction("STORE_FAST", argval=name))
885                  cg.add_push_null(
886                      lambda: cg.load_import_from(
887                          utils.__name__, "set_torch_function_mode_stack"
888                      )
889                  )
890  
891                  cg.foreach(var.symbolic_stack)
892                  cg.append_output(
893                      create_instruction("BUILD_LIST", arg=len(var.symbolic_stack))
894                  )
895                  cg.call_function(1, False)
896                  cg.append_output(create_instruction("POP_TOP"))
897  
898              elif isinstance(var, variables.CellVariable) and var.local_name is not None:
899                  # Emit more readable and performant bytecode.
900                  # TODO generalize this for cells created during inlining.
901                  if var in self.store_attr_mutations:
902                      contents_var = self.load_cell(var)
903                      cg(contents_var)
904                      suffixes.append([cg.create_store_deref(var.local_name)])
905  
906              elif self.is_attribute_mutation(var):
907                  if isinstance(
908                      var, variables.UserDefinedDictVariable
909                  ) and self.is_modified(var._dict_vt):
910                      # Do dict related update manually here. The store_attr
911                      # mutations will be applied later.
912                      varname_map = {}
913                      for name in _manual_dict_setitem.__code__.co_varnames:
914                          varname_map[name] = cg.tx.output.new_var()
915  
916                      try:
917                          mro_index = type(var.value).__mro__.index(
918                              collections.OrderedDict
919                          )
920                      except ValueError:
921                          mro_index = type(var.value).__mro__.index(dict)
922  
923                      cg.extend_output(
924                          [
925                              create_instruction("LOAD_CONST", argval=mro_index),
926                              create_instruction(
927                                  "STORE_FAST", argval=varname_map["mro_index"]
928                              ),
929                          ]
930                      )
931  
932                      cg(var.source)  # type: ignore[attr-defined]
933                      cg.extend_output(
934                          [
935                              create_instruction(
936                                  "STORE_FAST", argval=varname_map["dict_to"]
937                              )
938                          ]
939                      )
940  
941                      cg(var._dict_vt, allow_cache=False)  # Don't codegen via source
942                      cg.extend_output(
943                          [
944                              create_instruction(
945                                  "STORE_FAST", argval=varname_map["dict_from"]
946                              )
947                          ]
948                      )
949  
950                      dict_update_insts = bytecode_from_template(
951                          _manual_dict_setitem, varname_map=varname_map
952                      )
953  
954                      suffixes.append(
955                          [
956                              *dict_update_insts,
957                              create_instruction("POP_TOP"),
958                          ]
959                      )
960                  elif isinstance(
961                      var, variables.UserDefinedListVariable
962                  ) and self.is_modified(var._list_vt):
963                      # Update the list to the updated items. Be careful in
964                      # calling the list methods and not the overridden methods.
965                      varname_map = {}
966                      for name in _manual_list_update.__code__.co_varnames:
967                          varname_map[name] = cg.tx.output.new_var()
968  
969                      cg(var.source)  # type: ignore[attr-defined]
970                      cg.extend_output(
971                          [
972                              create_instruction(
973                                  "STORE_FAST", argval=varname_map["list_to"]
974                              )
975                          ]
976                      )
977  
978                      cg(var._list_vt, allow_cache=False)  # Don't codegen via source
979                      cg.extend_output(
980                          [
981                              create_instruction(
982                                  "STORE_FAST", argval=varname_map["list_from"]
983                              )
984                          ]
985                      )
986  
987                      list_update_insts = bytecode_from_template(
988                          _manual_list_update, varname_map=varname_map
989                      )
990  
991                      suffixes.append(
992                          [
993                              *list_update_insts,
994                              create_instruction("POP_TOP"),
995                          ]
996                      )
997  
998                  # Applying mutations involves two steps: 1) Push all
999                  # reconstructed objects onto the stack.  2) Call STORE_ATTR to
1000                 # apply the mutations.
1001                 #
1002                 # Dynamo must ensure that mutations are applied in the same
1003                 # order as in the original program. Therefore, two reverse
1004                 # operations occur below.
1005                 #
1006                 # The first reverse operation concerns `suffixes`. We apply
1007                 # suffixes in reverse order due to the way Python handles the
1008                 # stack. In Step 1, we push all reconstructed objects onto the
1009                 # stack, but the item at the top of the stack refers to the last
1010                 # attribute in the mutation order. If not fixed, this will apply
1011                 # the mutations of attributes in the reverse order.  To account
1012                 # for this reversal, we iterate through the mutable attributes
1013                 # in reverse order.
1014                 for name, value in reversed(
1015                     self.store_attr_mutations.get(var, {}).items()
1016                 ):
1017                     if isinstance(var, variables.NewGlobalVariable):
1018                         cg.tx.output.update_co_names(name)
1019                         cg(value)
1020                         assert isinstance(var.source, GlobalSource)  # type: ignore[attr-defined]
1021                         suffixes.append(
1022                             [create_instruction("STORE_GLOBAL", argval=name)]
1023                         )
1024                     elif isinstance(value, variables.DeletedVariable):
1025                         if isinstance(
1026                             var.mutation_type, AttributeMutationExisting
1027                         ) and hasattr(getattr(var, "value", None), name):
1028                             cg.tx.output.update_co_names(name)
1029                             cg(var.source)
1030                             suffixes.append(
1031                                 [create_instruction("DELETE_ATTR", argval=name)]
1032                             )
1033                     elif isinstance(
1034                         var, variables.UserDefinedObjectVariable
1035                     ) and var.should_skip_descriptor_setter(name):
1036                         cg.add_push_null(
1037                             lambda: cg.load_import_from(
1038                                 utils.__name__, "object_setattr_ignore_descriptor"
1039                             )
1040                         )
1041                         cg(var.source)  # type: ignore[attr-defined]
1042                         cg(variables.ConstantVariable(name))
1043                         cg(value)
1044                         suffixes.append(
1045                             [
1046                                 *create_call_function(3, False),
1047                                 create_instruction("POP_TOP"),
1048                             ]
1049                         )
1050                     elif (
1051                         isinstance(var, variables.UserDefinedObjectVariable)
1052                         and var.needs_slow_setattr()
1053                     ):
1054                         # __setattr__ is defined on this object, so call object.__setattr__ directly
1055                         cg.load_import_from("builtins", "object")
1056                         cg.load_method("__setattr__")
1057                         cg(var.source)  # type: ignore[attr-defined]
1058                         cg(variables.ConstantVariable(name))
1059                         cg(value)
1060                         suffixes.append(
1061                             [*create_call_method(3), create_instruction("POP_TOP")]
1062                         )
1063                     else:
1064                         cg.tx.output.update_co_names(name)
1065                         cg(value)
1066                         cg(var)
1067                         suffixes.append([create_instruction("STORE_ATTR", argval=name)])
1068             elif isinstance(var, variables.ListIteratorVariable):
1069                 for _ in range(var.index):
1070                     cg.add_push_null(
1071                         lambda: cg.load_import_from(utils.__name__, "iter_next")
1072                     )
1073                     cg(var.source)  # type: ignore[attr-defined]
1074                     cg.call_function(1, False)
1075                     cg.pop_top()
1076             elif isinstance(var, variables.RandomVariable):
1077                 # set correct random seed state
1078                 def gen_fn():
1079                     cg(var.source)  # type: ignore[attr-defined]
1080                     cg.load_attr("setstate")
1081 
1082                 cg.add_push_null(gen_fn)
1083                 cg(var.wrap_state(var.random.getstate()))
1084 
1085                 suffixes.append(
1086                     [
1087                         *create_call_function(1, False),  # setstate
1088                         create_instruction("POP_TOP"),
1089                     ]
1090                 )
1091             else:
1092                 raise AssertionError(type(var))
1093 
1094         # do all the actual mutations at the very end to handle dependencies
1095         for suffix in reversed(suffixes):
1096             cg.extend_output(suffix)
