123  template<typename key_t, typename value_t, typename OffsetIteratorT> 
124  inline void segmented_sort_pairs(
125      const key_t *keys_in, key_t *keys_out,
126      const value_t *values_in, value_t *values_out,
127      int64_t num_elements, int64_t num_segments,
128      OffsetIteratorT begin_offsets, OffsetIteratorT end_offsets,
129      bool descending=false, int64_t begin_bit=0, int64_t end_bit=sizeof(key_t)*8
130  ) {
131    TORCH_CHECK(num_elements <= std::numeric_limits<int>::max(),
132      "cub sort does not support sorting more than INT_MAX elements");
133    TORCH_CHECK(num_segments <= std::numeric_limits<int>::max(),
134      "cub sort does not support sorting more than INT_MAX elements");
135    using key_t_ = typename detail::cuda_type<key_t>::type;
136  
137    auto allocator = c10::cuda::CUDACachingAllocator::get();
138    c10::DataPtr keys_out_owner;
139  
140    if (keys_out == nullptr) {
141      keys_out_owner = allocator->allocate(num_elements * sizeof(key_t));
142      keys_out = reinterpret_cast<key_t *>(keys_out_owner.get());
143    }
144  
145    const key_t_ *keys_in_ = reinterpret_cast<const key_t_*>(keys_in);
146    key_t_ *keys_out_ = reinterpret_cast<key_t_*>(keys_out);
147  
148    if (descending) {
149      CUB_WRAPPER(NO_ROCM(at_cuda_detail)::cub::DeviceSegmentedRadixSort::SortPairsDescending,
150        keys_in_, keys_out_, values_in, values_out,
151        num_elements, num_segments, begin_offsets, end_offsets,
152        begin_bit, end_bit, c10::cuda::getCurrentCUDAStream());
153    } else {
154      CUB_WRAPPER(NO_ROCM(at_cuda_detail)::cub::DeviceSegmentedRadixSort::SortPairs,
155        keys_in_, keys_out_, values_in, values_out,
156        num_elements, num_segments, begin_offsets, end_offsets,
157        begin_bit, end_bit, c10::cuda::getCurrentCUDAStream());
158    }

190  template<typename ValueT, typename InputIteratorT> 
191  struct chained_iterator {
192    using iterator_category = std::random_access_iterator_tag;
193    using difference_type   = std::ptrdiff_t;
194    using value_type        = ValueT;
195    using pointer           = ValueT*;
196    using reference         = ValueT&;
197  
198    InputIteratorT iter;
199    ValueT *first;
200    difference_type offset = 0;
201  
202    __device__ ValueT operator[](difference_type i) {
203      i +=  offset;
204      if (i == 0) {
205        return *first;
206      } else {
207        return ValueT(iter[i - 1]);
208      }
209    }
210    __device__ chained_iterator operator+(difference_type i) {
211      return chained_iterator{iter, first, i};
212    }
213    __device__ ValueT operator*() {
214      return (*this)[0];
215    }
216  };

224  // non synchronizing cub call 
225  // even though cub is supposed to support tensors with int_max elements, in reality it doesn't, 
226  // so split at int_max/2 
227  template<typename InputIteratorT, typename OutputIteratorT, typename ScanOpT, int max_cub_size=impl::max_cub_size> 
228  inline void inclusive_scan(InputIteratorT input, OutputIteratorT output, ScanOpT scan_op, int64_t num_items) {
229  #if defined(USE_ROCM)
230    //For ROCm, use hipCUB chained iterators 
231    CUB_WRAPPER(NO_ROCM(detail)::hipcub::DeviceScan::InclusiveScan,
232        input,
233        output,
234        scan_op,
235        num_items,
236        at::cuda::getCurrentCUDAStream());
237    C10_HIP_KERNEL_LAUNCH_CHECK();
238  #else
239    // non synchronizing cub call 
240    // even though cub is supposed to support tensors with int_max elements, in reality it doesn't, 
241    // so split at int_max/2 
242    int size_cub = std::min<int64_t>(num_items, max_cub_size);
243    CUB_WRAPPER(NO_ROCM(at_cuda_detail)::cub::DeviceScan::InclusiveScan,
244        input,
245        output,
246        scan_op,
247        size_cub,
248        at::cuda::getCurrentCUDAStream());
249    C10_CUDA_KERNEL_LAUNCH_CHECK();
250    using input_t = typename std::iterator_traits<InputIteratorT>::value_type;
251    for (int64_t i = max_cub_size; i < num_items; i += max_cub_size) {
252      auto allocator = c10::cuda::CUDACachingAllocator::get();
253      c10::DataPtr first_elem = allocator->allocate(sizeof(input_t));
254      auto first_elem_ptr = reinterpret_cast<input_t *>(first_elem.get());
255  
256      size_cub = std::min<int64_t>(num_items - i, max_cub_size);
257      impl::transform_vals<<<1, 1, 0, at::cuda::getCurrentCUDAStream()>>>(
258          output + i - 1,
259          input + i,
260          first_elem_ptr,
261          scan_op);
262      C10_CUDA_KERNEL_LAUNCH_CHECK();
263  #if !CUB_SUPPORTS_FUTURE_VALUE()
264      using ArgIndexInputIterator = NO_ROCM(at_cuda_detail)::cub::ArgIndexInputIterator<InputIteratorT>;
265      using tuple = typename ArgIndexInputIterator::value_type;
266      auto input_iter_transform = [=] __device__ (const tuple &x)->input_t  {
267        if (x.key == 0) {
268          return *first_elem_ptr;
269        } else {
270          return x.value;
271        }
272      };
273      auto input_ = NO_ROCM(at_cuda_detail)::cub::TransformInputIterator<input_t, decltype(input_iter_transform), ArgIndexInputIterator>(
274        ArgIndexInputIterator(input + i), input_iter_transform);
275      CUB_WRAPPER(NO_ROCM(at_cuda_detail)::cub::DeviceScan::InclusiveScan,
276          input_,
277          output + i,
278          scan_op,
279          size_cub,
280          at::cuda::getCurrentCUDAStream());
281  #else
282      CUB_WRAPPER(NO_ROCM(at_cuda_detail)::cub::DeviceScan::ExclusiveScan,
283          input + i + 1,
284          output + i,
285          scan_op,
286          ::at_cuda_detail::cub::FutureValue<input_t>(first_elem_ptr),
287          size_cub,
288          at::cuda::getCurrentCUDAStream());
289  #endif
290    }
291  #endif
292  }

296  template<typename T> 
297  struct BlockPrefixCallbackOp
298  {
299      public:
300      T running_total;
301  
302      __host__ __device__ BlockPrefixCallbackOp(T running_total) : running_total(running_total) {}
303  
304      // Callback operator to be entered by the first warp of threads in the block. 
305      // Thread-0 is responsible for returning a value for seeding the block-wide scan. 
306      __host__ __device__ T operator()(T block_aggregate)
307      {
308          T old_prefix = running_total;
309          running_total += block_aggregate;
310          return old_prefix;
311      }
312  };

314  template<int BLOCK_THREADS, int ITEMS_PER_THREAD, typename T> 
315  __global__ void final_scan_kernel(const T* d_in, T* d_out, T* agg, int64_t nelem, int iters_per_cta) {
316    int64_t offset = BLOCK_THREADS * ITEMS_PER_THREAD * iters_per_cta * (int64_t)blockIdx.x;
317    int64_t remaining =  nelem - offset;
318    if (remaining <= 0) {
319      return;
320    }
321  
322    d_in += offset;
323    d_out += offset;
324  
325    using BlockLoadT = ROCM_HIPCUB(at_cuda_detail::cub)::BlockLoad<T, BLOCK_THREADS, ITEMS_PER_THREAD, ROCM_HIPCUB(at_cuda_detail::cub)::BLOCK_LOAD_WARP_TRANSPOSE>;
326  
327    // Specialize BlockStore type for our thread block (uses warp-striped loads for coalescing, then transposes in shared 
328    // memory to a blocked arrangement) 
329    using BlockStoreT = ROCM_HIPCUB(at_cuda_detail::cub)::BlockStore<T, BLOCK_THREADS, ITEMS_PER_THREAD, ROCM_HIPCUB(at_cuda_detail::cub)::BLOCK_STORE_WARP_TRANSPOSE>;
330  
331    // Specialize BlockScan type for our thread block 
332    using BlockScanT = ROCM_HIPCUB(at_cuda_detail::cub)::BlockScan<T, BLOCK_THREADS, ROCM_HIPCUB(at_cuda_detail::cub)::BLOCK_SCAN_WARP_SCANS>;
333    using BlockReduceT = ROCM_HIPCUB(at_cuda_detail::cub)::BlockReduce<T, BLOCK_THREADS>;
334  
335  
336    // Shared memory 
337    __shared__ union TempStorage
338    {
339      typename BlockLoadT::TempStorage load;
340      typename BlockStoreT::TempStorage store;
341      typename BlockScanT::TempStorage scan;
342      typename BlockReduceT::TempStorage reduce;
343    } temp_storage;
344  
345    // load agg and reduce my starting value 
346    T agg_data;
347    agg_data = threadIdx.x >= blockIdx.x ? T(0) : agg[threadIdx.x];
348    // if there are fewer threads than previous values to be read, 
349    // read another value 
350    if (threadIdx.x + blockDim.x < blockIdx.x) {
351      agg_data += agg[threadIdx.x + blockDim.x];
352    }
353    T aggregate = BlockReduceT(temp_storage.reduce).Sum(agg_data);
354    __syncthreads();
355    BlockPrefixCallbackOp prefix_op(aggregate);
356  
357  
358    // Per-thread tile data 
359    T data[ITEMS_PER_THREAD];
360  
361    for (int i=0; i<iters_per_cta; i++){
362    // Load items into a blocked arrangement 
363      if (remaining >= BLOCK_THREADS * ITEMS_PER_THREAD) {
364        BlockLoadT(temp_storage.load).Load(d_in, data);
365      } else {
366         #pragma unroll
367         for (int j=0; j<ITEMS_PER_THREAD; j++) {
368           data[j] = 0;
369         }
370         BlockLoadT(temp_storage.load).Load(d_in, data, remaining);
371      }
372  
373      // Barrier for smem reuse 
374      __syncthreads();
375  
376      // Compute inclusive prefix sum 
377      BlockScanT(temp_storage.scan).InclusiveSum(data, data, prefix_op);
378  
379      // Barrier for smem reuse 
380      __syncthreads();
381  
382      // Store items from a blocked arrangement 
383      if (remaining >= BLOCK_THREADS * ITEMS_PER_THREAD) {
384        BlockStoreT(temp_storage.store).Store(d_out, data);
385      } else {
386        BlockStoreT(temp_storage.store).Store(d_out, data, remaining);
387      }
388      d_in += BLOCK_THREADS * ITEMS_PER_THREAD;
389      d_out += BLOCK_THREADS * ITEMS_PER_THREAD;
390      remaining -= BLOCK_THREADS * ITEMS_PER_THREAD;
391      if (remaining <= 0) return;
392      __syncthreads();
393    }
394  
395  }

397  template <typename T, typename aggT, bool nonzero> 
398  struct TransformFunctor {
399    __device__ aggT operator()(T value) const {
400      if constexpr (!nonzero) {
401        return value;
402      } else {
403        return (value != T(0)) ? 1 : 0;
404      }
405    }
406  };

408  template<int BLOCK_THREADS, int ITEMS_PER_THREAD, bool nonzero, typename T, typename aggT> 
409  __global__ void calc_block_sums(const T * d_in, aggT * agg, int64_t nelem, int iters_per_cta){
410      int64_t offset = BLOCK_THREADS * ITEMS_PER_THREAD * iters_per_cta * (int64_t)blockIdx.x;
411      int64_t remaining = nelem - offset;
412      if (remaining <= 0) {
413        return;
414      }
415      d_in += offset;
416  
417      using BlockLoadT = ROCM_HIPCUB(at_cuda_detail::cub)::BlockLoad<aggT, BLOCK_THREADS, ITEMS_PER_THREAD, ROCM_HIPCUB(at_cuda_detail::cub)::BLOCK_LOAD_STRIPED>;
418      using BlockReduceT = ROCM_HIPCUB(at_cuda_detail::cub)::BlockReduce<aggT, BLOCK_THREADS>;
419      // Shared memory 
420      __shared__ union TempStorage
421      {
422        typename BlockLoadT::TempStorage load;
423        typename BlockReduceT::TempStorage reduce;
424      } temp_storage;
425      aggT data[ITEMS_PER_THREAD];
426      aggT agg_val = 0;
427      TransformFunctor<T, aggT, nonzero> transform_functor;
428      auto iter_in = ROCM_HIPCUB(at_cuda_detail::cub)::TransformInputIterator<aggT, TransformFunctor<T, aggT, nonzero>, const T*>(d_in, transform_functor);
429      for (int i=0; i<iters_per_cta; i++){
430        if (remaining >= BLOCK_THREADS * ITEMS_PER_THREAD) {
431          BlockLoadT(temp_storage.load).Load(iter_in, data);
432          __syncthreads();
433          agg_val += BlockReduceT(temp_storage.reduce).Sum(data);
434  
435        } else {
436          BlockLoadT(temp_storage.load).Load(iter_in, data, remaining, aggT(0));
437          __syncthreads();
438          agg_val += BlockReduceT(temp_storage.reduce).Sum(data);
439        }
440        iter_in += BLOCK_THREADS * ITEMS_PER_THREAD;
441        remaining -= BLOCK_THREADS * ITEMS_PER_THREAD;
442        if (remaining <= 0) {
443          // for nonzeros we need to write out last blocks 
444          // accumulated value to be able to compute 
445          // total number of nonzeros 
446          if (nonzero && threadIdx.x == 0) {
447            agg[blockIdx.x] = agg_val;
448          }
449          return;
450        }
451        __syncthreads();
452  
453      }
454      if (threadIdx.x == 0) {
455        agg[blockIdx.x] = agg_val;
456      }
457  
458  }

478  template<typename scalar_t, typename ScanOpT> 
479  inline void inclusive_deterministic_scan(const scalar_t *  input, scalar_t * output, ScanOpT scan_op, int64_t num_items) {
480    static_assert(std::is_same_v<ScanOpT, std::plus<scalar_t>>, "");
481    constexpr int BLOCK_THREADS = block_threads<sizeof(scalar_t)>();
482    constexpr int ITEMS_PER_THREAD = 16;
483    auto grid_size = (num_items + BLOCK_THREADS * ITEMS_PER_THREAD - 1) / (BLOCK_THREADS * ITEMS_PER_THREAD);
484    const int64_t num_sms = at::cuda::getCurrentDeviceProperties()->multiProcessorCount;
485  
486    const int iters_per_cta = (grid_size + num_sms - 1)/num_sms;
487    grid_size = std::min(num_sms, grid_size);
488    // simple reduction in scan kernel handles at most 2 items per thread 
489    TORCH_INTERNAL_ASSERT(2 * BLOCK_THREADS >= grid_size);
490    auto& allocator = *c10::cuda::CUDACachingAllocator::get();
491    auto agg = allocator.allocate(grid_size * sizeof(scalar_t));
492    calc_block_sums<BLOCK_THREADS, ITEMS_PER_THREAD, false>
493    <<<grid_size, BLOCK_THREADS, 0, at::cuda::getCurrentCUDAStream()>>>(
494      input, (scalar_t*)agg.get(), num_items, iters_per_cta);
495    C10_CUDA_KERNEL_LAUNCH_CHECK();
496    final_scan_kernel<BLOCK_THREADS, ITEMS_PER_THREAD>
497    <<<grid_size, BLOCK_THREADS, 0, at::cuda::getCurrentCUDAStream()>>>(
498    input, output, (scalar_t*)agg.get(), num_items, iters_per_cta);
499    C10_CUDA_KERNEL_LAUNCH_CHECK();
500  }
