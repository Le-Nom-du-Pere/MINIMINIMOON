11   template <typename T>
12   inline ::metal::enable_if_t<!::metal::is_same_v<T, long>, T> simd_sum(T val) {
13     return ::metal::simd_sum(val);

21   // Metal does not support SIMD reductions over 64-bit types, but it could be
22   // implement using simd_shuffle_down, that yields result in log2(simdgroup_size)
23   // iterations Use fill variant, as shuffle down returns garbage if inactive
24   // thread is referenced (on M1/M2, works fine on M4) and broadcast result to all
25   // threads in the end. Implementation heavily borrows from
26   // https://github.com/ml-explore/mlx/blob/86389bf9707f46101af45d90510e8e97c8a90b93/mlx/backend/metal/kernels/reduction/ops.h#L16
27   template <typename T>
28   inline ::metal::enable_if_t<::metal::is_same_v<T, long>, T> simd_sum(T val) {
29     for (ushort i = simdgroup_size / 2; i > 0; i /= 2) {
30       val += as_type<T>(
31           ::metal::simd_shuffle_and_fill_down(as_type<int2>(val), int2(0), i));
32     }
33     return as_type<T>(::metal::simd_broadcast(as_type<int2>(val), 0));
34   }
35   
36   template <typename T>
37   inline ::metal::enable_if_t<::metal::is_same_v<T, long>, T> simd_prod(T val) {
38     for (ushort i = simdgroup_size / 2; i > 0; i /= 2) {
39       val *= as_type<T>(
40           ::metal::simd_shuffle_and_fill_down(as_type<int2>(val), int2(0), i));
41     }
42     return as_type<T>(::metal::simd_broadcast(as_type<int2>(val), 0));
43   }

45   // Below algorithms are  written with hardcoded assumption that simdgroup is 32
46   // and threadgroup_max is 1024, i.e. reduction can be done in two stages max
47   template <typename T>
48   opmath_t<T> threadgroup_sum(
49       threadgroup opmath_t<T>* data,
50       T val,
51       unsigned idx,
52       unsigned size) {
53     auto rc = simd_sum(static_cast<opmath_t<T>>(val));
54     if (idx % simdgroup_size == 0) {
55       data[idx / simdgroup_size] = rc;
56     }
57     if (size > simdgroup_size) {
58       ::metal::threadgroup_barrier(::metal::mem_flags::mem_threadgroup);
59       if (idx < ((size + simdgroup_size - 1) / simdgroup_size)) {
60         auto rc1 = simd_sum(data[idx]);
61         if (idx == 0) {
62           data[0] = rc1;
63         }
64       }
65     }
66     ::metal::threadgroup_barrier(::metal::mem_flags::mem_threadgroup);
67     return data[0];
68   }

93   template <typename T>
94   float3 threadgroup_welford_reduce(threadgroup T* data, unsigned size) {
95     ::metal::threadgroup_barrier(::metal::mem_flags::mem_threadgroup);
96     float m = data[0];
97     float m2 = 0;
98     for (unsigned idx = 1; idx < size; ++idx) {
99       float delta = data[idx] - m;
100      m += delta / (idx + 1);
101      m2 += delta * (data[idx] - m);
102    }
103    return float3(m, m2, size);
104  }

106  // Each vec3type is tuple of mean, m2 and weight
107  template <typename T>
108  float3 welford_combine(T a, T b) {
109    float delta = b.x - a.x;
110    float new_weight = a.z + b.z;
111    auto w2_over_w = new_weight != 0 ? b.z / new_weight : 0.0;
112    return float3(
113        a.x + delta * w2_over_w,
114        a.y + b.y + delta * delta * a.z * w2_over_w,
115        new_weight);
116  }

118  template <typename T>
119  float3 threadgroup_welford_combine(threadgroup T* data, unsigned size) {
120    ::metal::threadgroup_barrier(::metal::mem_flags::mem_threadgroup);
121    float3 rc = data[0];
122    for (unsigned idx = 1; idx < size; ++idx) {
123      rc = welford_combine(rc, data[idx]);
124    }
125    return rc;
126  }

152    // TODO: This should be moved to the callee
153    ::metal::threadgroup_barrier(::metal::mem_flags::mem_threadgroup);
154    int rc = 0;
155    for (unsigned idx = 1; idx < size; ++idx) {
156      if (data[idx] > data[rc]) {
157        rc = idx;
