157    static Vectorized<ComplexFlt> blendv(
158        const Vectorized<ComplexFlt>& a,
159        const Vectorized<ComplexFlt>& b,
160        const Vectorized<ComplexFlt>& mask) {
161      // convert std::complex<V> index mask to V index mask: xy -> xxyy
162      auto mask_complex = Vectorized<ComplexFlt>(
163          vec_mergeh(mask._vec0, mask._vec0), vec_mergeh(mask._vec1, mask._vec1));
164      return {
165          vec_sel(
166              a._vec0, b._vec0, reinterpret_cast<vbool32>(mask_complex._vec0)),
167          vec_sel(
168              a._vec1, b._vec1, reinterpret_cast<vbool32>(mask_complex._vec1)),
169      };
170    }

209    static Vectorized<value_type> C10_ALWAYS_INLINE
210    loadu(const void* ptr, int count = size()) {
211      if (count == size()) {
212        return {
213            vec_vsx_ld(offset0, reinterpret_cast<const float*>(ptr)),
214            vec_vsx_ld(offset16, reinterpret_cast<const float*>(ptr))};
215      }
216  
217      __at_align__ value_type tmp_values[size()] = {};
218      std::memcpy(tmp_values, ptr, std::min(count, size()) * sizeof(value_type));
219  
220      return {
221          vec_vsx_ld(offset0, reinterpret_cast<const float*>(tmp_values)),
222          vec_vsx_ld(offset16, reinterpret_cast<const float*>(tmp_values))};
223    }
224  
225    void C10_ALWAYS_INLINE store(void* ptr, int count = size()) const {
226      if (count == size()) {
227        vec_vsx_st(_vec0, offset0, reinterpret_cast<float*>(ptr));
228        vec_vsx_st(_vec1, offset16, reinterpret_cast<float*>(ptr));
229      } else if (count > 0) {
230        __at_align__ value_type tmp_values[size()];
231        vec_vsx_st(_vec0, offset0, reinterpret_cast<float*>(tmp_values));
232        vec_vsx_st(_vec1, offset16, reinterpret_cast<float*>(tmp_values));
233        std::memcpy(
234            ptr, tmp_values, std::min(count, size()) * sizeof(value_type));
235      }
236    }

259    static Vectorized<ComplexFlt> horizontal_add(
260        Vectorized<ComplexFlt>& first,
261        Vectorized<ComplexFlt>& second) {
262      // Operates on individual floats, see _mm_hadd_ps
263      // {f0+f1, s0+s1, f2+f3, s2+s3, ...}
264      // i.e. it sums the re and im of each value and interleaves first and
265      // second: {f_re0 + f_im0, s_re0 + s_im0, f_re1 + f_im1, s_re1 + s_im1, ...}
266      return el_mergee(first, second) + el_mergeo(first, second);
267    }

269    static Vectorized<ComplexFlt> horizontal_sub_permD8(
270        Vectorized<ComplexFlt>& first,
271        Vectorized<ComplexFlt>& second) {
272      // we will simulate it differently with 6 instructions total
273      // lets permute second so that we can add it getting horizontal sums
274      auto first_perm = first.el_swapped(); // 2perm
275      auto second_perm = second.el_swapped(); // 2perm
276      // sum
277      auto first_ret = first - first_perm; // 2sub
278      auto second_ret = second - second_perm; // 2 sub
279      // now lets choose evens
280      return el_mergee(first_ret, second_ret); // 2 mergee's
281    }

283    Vectorized<ComplexFlt> abs_2_() const {
284      auto a = (*this).elwise_mult(*this);
285      auto permuted = a.el_swapped();
286      a = a + permuted;
287      return a.el_mergee();
288    }

444    Vectorized<ComplexFlt> reciprocal() const {
445      // re + im*i = (a + bi)  / (c + di)
446      // re = (ac + bd)/abs_2() = c/abs_2()
447      // im = (bc - ad)/abs_2() = d/abs_2()
448      auto c_d = *this ^ isign_mask; // c       -d
449      auto abs = abs_2_();
450      return c_d.elwise_div(abs);
451    }

468    Vectorized<ComplexFlt> atan() const {
469      // atan(x) = i/2 * ln((i + z)/(i - z))
470      auto ione = Vectorized(imag_one);
471      auto sum = ione + *this;
472      auto sub = ione - *this;
473      auto ln = (sum / sub).log(); // ln((i + z)/(i - z))
474      return ln * imag_half; // i/2*ln()

485    Vectorized<ComplexFlt> inline operator*(
486        const Vectorized<ComplexFlt>& b) const {
487      //(a + bi)  * (c + di) = (ac - bd) + (ad + bc)i
488  
489  #if 1
490      // this is more vsx friendly than simulating horizontal from x86
491  
492      auto vi = b.el_mergeo();
493      auto vr = b.el_mergee();
494      vi = vi ^ rsign_mask;
495      auto ret = elwise_mult(vr);
496      auto vx_swapped = el_swapped();
497      ret = vx_swapped.elwise_mult(vi) + ret;
498      return ret;
499  
500  #else
501  
502      auto ac_bd = elwise_mult(b);
503      auto d_c = b.el_swapped();
504      d_c = d_c ^ isign_mask;
505      auto ad_bc = elwise_mult(d_c);
506      auto ret = horizontal_sub_permD8(ac_bd, ad_bc);
507      return ret;
508  #endif
509    }

511    Vectorized<ComplexFlt> inline operator/(
512        const Vectorized<ComplexFlt>& b) const {
513  #if 1
514      __at_align__ c10::complex<float>
515          tmp1[Vectorized<c10::complex<float>>::size()];
516      __at_align__ c10::complex<float>
517          tmp2[Vectorized<c10::complex<float>>::size()];
518      __at_align__ c10::complex<float>
519          out[Vectorized<c10::complex<float>>::size()];
520      this->store(tmp1);
521      b.store(tmp2);
522  
523      for (const auto i : c10::irange(Vectorized<c10::complex<float>>::size())) {
524        out[i] = tmp1[i] / tmp2[i];
525      }
526      return loadu(out);
527  #else
528      auto fabs_cd = Vectorized{
529          vec_andc(b._vec0, sign_mask), vec_andc(b._vec1, sign_mask)}; // |c| |d|
530      auto fabs_dc = fabs_cd.el_swapped(); // |d|            |c|
531      auto scale = fabs_cd.elwise_max(fabs_dc); // sc = max(|c|, |d|)
532      auto a2 = elwise_div(scale); // a/sc           b/sc
533      auto b2 = b.elwise_div(scale); // c/sc           d/sc
534      auto acbd2 = a2.elwise_mult(b2); // ac/sc^2        bd/s
535      auto dc2 = b2.el_swapped(); // d/sc           c/sc
536      dc2 = dc2 ^ rsign_mask; // -d/sc          c/sc
537      auto adbc2 = a2.elwise_mult(dc2); // -ad/sc^2       bc/sc^2
538      auto ret = horizontal_add(acbd2, adbc2); // (ac+bd)/sc^2   (bc-ad)/sc^2
539      auto denom2 = b2.abs_2_(); // (c^2+d^2)/sc^2 (c^2+d^2)/sc^2
540      ret = ret.elwise_div(denom2);
541      return ret;
542  #endif
543    }

545    Vectorized<ComplexFlt> asin() const {
546      // asin(x)
547      // = -i*ln(iz + sqrt(1 -z^2))
548      // = -i*ln((ai - b) + sqrt(1 - (a + bi)*(a + bi)))
549      // = -i*ln((-b + ai) + sqrt(1 - (a**2 - b**2) - 2*abi))
550  
551  #if 1
552      auto conj = conj_();
553      auto b_a = conj.el_swapped();
554      auto ab = conj.elwise_mult(b_a);
555      auto im = ab + ab;
556      auto val_2 = (*this).elwise_mult(*this);
557      auto val_2_swapped = val_2.el_swapped();
558      auto re = horizontal_sub_permD8(val_2, val_2_swapped);
559      re = Vectorized<ComplexFlt>(one) - re;
560      auto root = el_blend<0xAA>(re, im).sqrt();
561      auto ln = (b_a + root).log();
562      return ln.el_swapped().conj();
563  #else
564      return map(std::asin);
565  #endif
566    }

578    Vectorized<ComplexFlt> eq(const Vectorized<ComplexFlt>& other) const {
579      auto eq = (*this == other); // compares real and imag individually
580      // If both real numbers and imag numbers are equal, then the complex numbers
581      // are equal
582      return (eq.real() & eq.imag()) & one;
583    }
584    Vectorized<ComplexFlt> ne(const Vectorized<ComplexFlt>& other) const {
585      auto ne = (*this != other); // compares real and imag individually
586      // If either real numbers or imag numbers are not equal, then the complex
587      // numbers are not equal
588      return (ne.real() | ne.imag()) & one;
589    }

697  template <>
698  Vectorized<ComplexFlt> C10_ALWAYS_INLINE
699  operator*(const Vectorized<ComplexFlt>& a, const Vectorized<ComplexFlt>& b) {
700    // (a + ib) * (c + id) = (ac - bd) + i(ad + bc)
701    // Split into real and imaginary parts
702    auto a_real = a.el_mergee(); // real part of a
703    auto a_imag = a.el_mergeo(); // imag part of a
704    auto b_real = b.el_mergee(); // real part of b
705    auto b_imag = b.el_mergeo(); // imag part of b
706  
707    auto b_imag_neg = b_imag ^ rsign_mask;
708    // Compute components
709    auto ac = a_real.elwise_mult(b_real); // real * real
710    auto bd = a_imag.elwise_mult(b_imag_neg); // imag * imag
711    auto ad = a_real.elwise_mult(b_imag); // real * imag
712    auto bc = a_imag.elwise_mult(b_real); // imag * real
713  
714    // Real = ac - bd (fix the negative bd part)
715    auto real = ac + bd; // Real part calculation
716    auto imag = ad + bc; // Imaginary part calculation
717  
718    // Step 1: Extract from real and imag
719    __vector float r0 = real.vec0(); // {r0, r1, r2, r3}
720    __vector float i0 = imag.vec0(); // {i0, i1, i2, i3}
721  
722    __vector float r1 = real.vec1(); // imag[0..3]
723    __vector float i1 = imag.vec1(); // imag[4..7]
724  
725    __vector unsigned char perm_lo = {
726        0,
727        1,
728        2,
729        3, // r0
730        16,
731        17,
732        18,
733        19, //
734        8,
735        9,
736        10,
737        11, // r1
738        24,
739        25,
740        26,
741        27};
742    __vector float v0 =
743        vec_perm(r0, i0, perm_lo); // Interleave r0 and i0, r1 and i1
744    __vector float v1 = vec_perm(r1, i1, perm_lo);
745    Vectorized<ComplexFlt> result(v0, v1);
746    return result;
747  }

751  operator/(const Vectorized<ComplexFlt>& a, const Vectorized<ComplexFlt>& b) {
752    // Take absolute values of real and imaginary parts of b
753    __at_align__ c10::complex<float>
754        tmp1[Vectorized<c10::complex<float>>::size()];
755    __at_align__ c10::complex<float>
756        tmp2[Vectorized<c10::complex<float>>::size()];
757    __at_align__ c10::complex<float> out[Vectorized<c10::complex<float>>::size()];
758    a.store(tmp1);
759    b.store(tmp2);
760    for (const auto i :
761         c10::irange(Vectorized<c10::complex<float>>::
762                         size())) { //{Vectorized<c10::complex<float>>::size()))
763                                    //{
764      out[i] = tmp1[i] / tmp2[i];
765    }
766    return Vectorized<ComplexFlt>::loadu(out);
767  }
