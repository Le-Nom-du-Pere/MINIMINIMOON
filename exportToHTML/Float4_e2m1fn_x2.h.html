6    /// Defines the Float4_e2m1fn_x2 type (4-bit floating-point, two elements packed

9    /// Section 5.3.3)
10   ///
11   /// Given two high precision values val0 and val1, here is the
12   /// binary configuration of their packed representation, from MSB to LSB:
13   ///
14   ///   original value             | val1 : val0
15   ///   ========================================
16   ///   bit index (MSB==7, LSB==0) | 7654 : 3210
17   ///   sign/exponent/mantissa     | seem : seem
18   ///
19   
20   namespace c10 {
21

23     uint8_t val_;
24     Float4_e2m1fn_x2() = default;
