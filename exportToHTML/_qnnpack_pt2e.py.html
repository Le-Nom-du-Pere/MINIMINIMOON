21   def get_linear_configs():
22       linear_configs = []
23       observation_type = ObservationType.OUTPUT_USE_DIFFERENT_OBSERVER_AS_INPUT
24       dtype_configs = [weighted_op_quint8_dtype_config]
25   
26       # TODO: need to fix the way we insert observers for this pattern
27       # should be solved in the new fusion API
28       # reason that this doesn't work: the pattern is a bit complicated and we don't
29       # have a way to specify which input of the pattern we would like to observe
30       # pattern:
31       # bias input weight
32       # \     |    /
33       #  \    |   t
34       #   \   |  /
35       #    addmm
36       # we want to observe "weight" as weight, but there is not way to convey this
37       # information with current pattern language
38       #
39       # right now:
40       # original:
41       #         weight - t \
42       #         input  - addmm
43       # observed (no hack):
44       #      weight - t - observer \
45       #       input - observer - addmm
46       # target:
47       #      weight - observer - t \
48       #        input - observer - addmm
49   
50       # def root_node_getter(node_pattern):
51       #     addmm, bias, act, weight = node_pattern
52       #     return addmm
53   
54       # linear_configs.append(
55       #     BackendPatternConfig((torch.ops.aten.addmm.default, MatchAllNode, MatchAllNode, torch.ops.aten.t.default))
56       #     .set_observation_type(observation_type)  # noqa: E131
57       #     .set_dtype_configs(dtype_configs)
58       #     ._set_root_node_getter(root_node_getter))
59   
60       linear_configs.append(
61           BackendPatternConfig(torch.ops.aten.addmm.default)
62           .set_observation_type(observation_type)  # noqa: E131
63           .set_dtype_configs(dtype_configs)
64           ._set_input_type_to_index({"weight": 2, "bias": 0})
65       )
66       # linear is decomposed to `t - mm` if bias is not present
67       linear_configs.append(
68           BackendPatternConfig(torch.ops.aten.mm.default)
69           .set_observation_type(observation_type)  # noqa: E131
70           .set_dtype_configs(dtype_configs)
71           ._set_input_type_to_index({"weight": 1})
72       )
73       return linear_configs

76   def get_conv_configs():
77       conv_configs = []
78       observation_type = ObservationType.OUTPUT_USE_DIFFERENT_OBSERVER_AS_INPUT
79       dtype_configs = [weighted_op_quint8_dtype_config]
80       conv_configs.append(
81           BackendPatternConfig(torch.ops.aten.convolution.default)
82           .set_observation_type(observation_type)  # noqa: E131
83           .set_dtype_configs(dtype_configs)
84           ._set_input_type_to_index({"weight": 1, "bias": 2})
85       )
86       conv_configs.append(
87           BackendPatternConfig(
88               (torch.ops.aten.convolution.default, torch.ops.aten.relu.default)
89           )
90           .set_observation_type(observation_type)  # noqa: E131
91           .set_dtype_configs(dtype_configs)
92           ._set_input_type_to_index({"weight": 1, "bias": 2})
93       )
94       # TODO: remove when functionalization is supported in PT2 mode
95       conv_configs.append(
96           BackendPatternConfig(
97               (torch.ops.aten.convolution.default, torch.ops.aten.relu_.default)
98           )
99           .set_observation_type(observation_type)  # noqa: E131
100          .set_dtype_configs(dtype_configs)
101          ._set_input_type_to_index({"weight": 1, "bias": 2})
102      )

106  def get_pooling_configs():
107      backend_pattern_configs = []
108      observation_type = ObservationType.OUTPUT_SHARE_OBSERVER_WITH_INPUT
109      dtype_configs = [weighted_op_quint8_dtype_config]
110  
111      def root_node_getter(node_pattern):
112          _getitem, maxpool, _index = node_pattern
113          return maxpool
114  
115      backend_pattern_configs.append(
116          BackendPatternConfig()
117          ._set_pattern_complex_format(
118              (operator.getitem, torch.ops.aten.max_pool2d_with_indices.default, 0)
119          )
120          .set_observation_type(observation_type)  # noqa: E131
121          .set_dtype_configs(dtype_configs)
122          ._set_root_node_getter(root_node_getter)
123      )
124  
125      return backend_pattern_configs

140  def get_binary_op_configs():
141      binary_op_configs: list[BackendPatternConfig] = []
142      dtype_configs = [weighted_op_quint8_dtype_config]
143      num_tensor_args_to_observation_type_mapping = {
144          # TODO: this is not used right now since we have extra check in prepare
145          # will need to change this to NO_OBSERVER later after we implemented
146          # Tensor dtype inference properly
147          0: ObservationType.OUTPUT_USE_DIFFERENT_OBSERVER_AS_INPUT,
148          1: ObservationType.OUTPUT_SHARE_OBSERVER_WITH_INPUT,
149          2: ObservationType.OUTPUT_USE_DIFFERENT_OBSERVER_AS_INPUT,
150      }
151      for op_with_quantized_bop_scalar_variant in [
152          torch.ops.aten.add.Tensor,
153          torch.ops.aten.add_.Tensor,
154      ]:
155          bop_patterns = [
156              (op_with_quantized_bop_scalar_variant, torch.ops.aten.relu.default),
157              op_with_quantized_bop_scalar_variant,
158              # TODO: remove when functionalization is supported in pt2_mode
159              (op_with_quantized_bop_scalar_variant, torch.ops.aten.relu_.default),
160          ]
161          binary_op_configs.extend(
162              BackendPatternConfig(bop_pattern)
163              .set_dtype_configs(dtype_configs)  # noqa: E131
164              ._set_num_tensor_args_to_observation_type(
165                  num_tensor_args_to_observation_type_mapping
166              )
167              for bop_pattern in bop_patterns
168          )
169  
170      return binary_op_configs
