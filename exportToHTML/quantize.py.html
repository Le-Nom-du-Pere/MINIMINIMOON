76   def _propagate_qconfig_helper(
77       module,
78       qconfig_dict,
79       qconfig_parent=None,
80       prefix="",
81       prepare_custom_config_dict=None,
82   ):
83       r"""This is a helper function for `propagate_qconfig_` 
84    
85       Args: 
86           module: input module 
87           qconfig_dict: dictionary that maps from name of submodule to quantization 
88                        configuration 
89           qconfig_parent: quantization config of parent module, we will fallback to 
90                          this config when there is no specified config for current 
91                          module 
92           prefix: corresponding prefix of the current module, used as key in 
93                   qconfig_dict 
94           prepare_custom_config_dict: dictionary for custom handling of modules 
95                                       see docs for :func:`~torch.ao.quantization.prepare_fx` 
96    
97       Return: 
98           None, module is modified inplace with qconfig attached 
99       """
100  
101      module_qconfig = qconfig_dict.get(
102          type_before_parametrizations(module), qconfig_parent
103      )
104      module_qconfig = qconfig_dict.get(prefix, module_qconfig)
105      module_qconfig = getattr(module, "qconfig", module_qconfig)
106  
107      torch.ao.quantization.qconfig._assert_valid_qconfig(module_qconfig, module)
108  
109      qconfig_with_device_check = _add_module_to_qconfig_obs_ctr(module_qconfig, module)
110      module.qconfig = qconfig_with_device_check
111  
112      for name, child in module.named_children():
113          module_prefix = prefix + "." + name if prefix else name
114          #  do no not propagate qconfig to child if child is non traceable
115          if prepare_custom_config_dict is None or not (
116              name in prepare_custom_config_dict.get("non_traceable_module_name", [])
117              or type(child)
118              in prepare_custom_config_dict.get("non_traceable_module_class", [])
119          ):
120              _propagate_qconfig_helper(
121                  child, qconfig_dict, qconfig_with_device_check, module_prefix
122              )

125  def propagate_qconfig_(module, qconfig_dict=None, prepare_custom_config_dict=None):
126      r"""Propagate qconfig through the module hierarchy and assign `qconfig` 
127      attribute on each leaf module 
128   
129      Args: 
130          module: input module 
131          qconfig_dict: dictionary that maps from name or type of submodule to 
132              quantization configuration, qconfig applies to all submodules of a 
133              given module unless qconfig for the submodules are specified (when 
134              the submodule already has qconfig attribute) 
135          prepare_custom_config_dict: dictionary for custom handling of modules 
136              see docs for :func:`~torch.ao.quantization.prepare_fx` 
137   
138      Return: 
139          None, module is modified inplace with qconfig attached 
140      """
141      if qconfig_dict is None:
142          qconfig_dict = {}
143      if prepare_custom_config_dict is None:
144          prepare_custom_config_dict = {}
145      _propagate_qconfig_helper(
146          module, qconfig_dict, prepare_custom_config_dict=prepare_custom_config_dict
147      )

160  def _register_activation_post_process_hook(module, pre_hook=False):
161      assert hasattr(module, "activation_post_process"), (
162          "Expect activation_post_process attribute already attached to the module"
163      )
164      if pre_hook:
165          module.register_forward_pre_hook(_observer_forward_pre_hook, prepend=True)

170  def _add_observer_(
171      module,
172      qconfig_propagation_list=None,
173      non_leaf_module_list=None,
174      device=None,
175      custom_module_class_mapping=None,
176  ):
177      r"""Add observer for the leaf child of the module. 
178   
179      This function insert observer module to all leaf child module that 
180      has a valid qconfig attribute. 
181   
182      Args: 
183          module: input module with qconfig attributes for all the leaf modules that we want to quantize 
184          qconfig_propagation_list: a list of quantizable modules that will have observers added to them 
185              if they are leaf nodes 
186          device: parent device, if any 
187          non_leaf_module_list: list of non-leaf modules we want to add observer 
188   
189      Return: 
190          None, module is modified inplace with added observer modules and forward_hooks 
191      """
192      if qconfig_propagation_list is None:
193          qconfig_propagation_list = get_default_qconfig_propagation_list()
194  
195      if custom_module_class_mapping is None:
196          custom_module_class_mapping = {}
197  
198      # respect device affinity when adding observers
199      if device is None:
200          devices = _get_unique_devices_(module)
201          assert len(devices) <= 1, (
202              f"_add_observer_ only works with cpu or single-device CUDA modules, but got devices {devices}"
203          )
204          device = next(iter(devices)) if len(devices) > 0 else None
205  
206      def get_activation_post_process(qconfig, device, special_act_post_process=None):
207          activation = (
208              qconfig.activation()
209              if special_act_post_process is None
210              else special_act_post_process()
211          )
212          if device is not None:
213              activation.to(device)
214          return activation
215  
216      def needs_observation(m):
217          return hasattr(m, "qconfig") and m.qconfig is not None
218  
219      def insert_activation_post_process(m, special_act_post_process=None):
220          """Adds an activation post process module and register 
221          a pre or post hook that calls the module 
222          """
223          # We don't insert observer/fake_quantize for DeQuantStub
224          if needs_observation(m) and not isinstance(m, DeQuantStub):
225              # observer and hook will be gone after we swap the module
226              m.add_module(
227                  "activation_post_process",
228                  get_activation_post_process(
229                      m.qconfig, device, special_act_post_process
230                  ),
231              )
232              # Register observer as the first entry in the hook list
233              # All post forward hooks are preserved and will be executed after the observer before convert
234              _register_activation_post_process_hook(
235                  m, pre_hook=_activation_is_memoryless(m.qconfig)
236              )
237  
238      for name, child in module.named_children():
239          # TODO remove Dropout special after codebase stable
240          if type_before_parametrizations(child) in [nn.Dropout]:
241              continue
242          elif issubclass(
243              type_before_parametrizations(child), (nnq.FloatFunctional, nnq.QFunctional)
244          ):
245              if needs_observation(child):
246                  assert hasattr(child, "activation_post_process"), (
247                      f"functional class {type_before_parametrizations(child)} has no pre-defined `activation_post_process`"
248                  )
249                  child.activation_post_process = get_activation_post_process(
250                      child.qconfig, device
251                  )
252          elif isinstance(child, _FusedModule):
253              # activation_post_process are now added directly to nn.Sequential/_FusedModule
254              if needs_observation(child):
255                  insert_activation_post_process(child)
256          elif (
257              non_leaf_module_list is not None
258              and type_before_parametrizations(child) in non_leaf_module_list
259          ):
260              if needs_observation(child):
261                  insert_activation_post_process(child)
262          elif _has_special_act_post_process(child):
263              special_act_post_process = _get_special_act_post_process(child)
264              insert_activation_post_process(child, special_act_post_process)
265          elif (
266              needs_observation(child)
267              and type_before_parametrizations(child) in custom_module_class_mapping
268          ):
269              observed_class = custom_module_class_mapping[
270                  type_before_parametrizations(child)
271              ]
272              observed_child = observed_class.from_float(child)
273              setattr(module, name, observed_child)
274              # TODO: These are the modules that cannot be observed
275              #       Once there are more, we should move them to a separate list
276              if not issubclass(observed_class, tuple(no_observer_set())):
277                  insert_activation_post_process(observed_child)
278          else:
279              _add_observer_(
280                  child,
281                  qconfig_propagation_list,
282                  non_leaf_module_list,
283                  device,
284                  custom_module_class_mapping,
285              )
286  
287      # Insert observers only for leaf nodes, note that this observer is for
288      # the output of the module, for input QuantStub will observe them
289      if (
290          has_no_children_ignoring_parametrizations(module)
291          and not isinstance(module, torch.nn.Sequential)
292          and type_before_parametrizations(module) in qconfig_propagation_list
293      ):
294          insert_activation_post_process(module)
295      # This is a special case for AdaRound eager mode
296      # AdaRound contains weight_fake_quant to be propagated from API to convert
297      # leaf node check with a number of children looks naive assumption that blocks
298      # Adding an exception case for AdaRound
299      if (
300          hasattr(module, "weight_fake_quant")
301          and not isinstance(module, torch.nn.Sequential)
302          and type_before_parametrizations(module) in qconfig_propagation_list
303      ):
304          insert_activation_post_process(module)
305

307  def _get_unique_devices_(module):
308      return {p.device for p in module.parameters() if p.device.type != "meta"} | {
309          p.device for p in module.buffers() if p.device.type != "meta"
310      }

313  def add_quant_dequant(module):
314      r"""Wrap the leaf child module in QuantWrapper if it has a valid qconfig 
315      Note that this function will modify the children of module inplace and it 
316      can return a new module which wraps the input module as well. 
317   
318      Args: 
319          module: input module with qconfig attributes for all the leaf modules 
320          that we want to quantize 
321   
322      Return: 
323          Either the inplace modified module with submodules wrapped in 
324          `QuantWrapper` based on qconfig or a new `QuantWrapper` module which 
325          wraps the input module, the latter case only happens when the input 
326          module is a leaf module and we want to quantize it. 
327      """
328      if (
329          has_no_children_ignoring_parametrizations(module)
330          and hasattr(module, "qconfig")
331          and module.qconfig
332      ):
333          return QuantWrapper(module)
334  
335      for name, child in module.named_children():
336          module._modules[name] = add_quant_dequant(child)
337      return module

340  @typing_extensions.deprecated(DEPRECATION_WARNING)
341  def prepare(
342      model,
343      inplace=False,
344      allow_list=None,
345      observer_non_leaf_module_list=None,
346      prepare_custom_config_dict=None,
347  ):
348      r"""Prepares a copy of the model for quantization calibration or quantization-aware training. 
349   
350      Quantization configuration should be assigned preemptively 
351      to individual submodules in `.qconfig` attribute. 
352   
353      The model will be attached with observer or fake quant modules, and qconfig 
354      will be propagated. 
355   
356      Args: 
357          `model`: input model to be modified in-place 
358          `inplace`: carry out model transformations in-place, the original module is mutated 
359          `allow_list`: list of quantizable modules 
360          `observer_non_leaf_module_list`: list of non-leaf modules we want to add observer 
361          `prepare_custom_config_dict`: customization configuration dictionary for prepare function 
362   
363      .. code-block:: python 
364   
365         # Example of prepare_custom_config_dict: 
366         prepare_custom_config_dict = { 
367             # user will manually define the corresponding observed 
368             # module class which has a from_float class method that converts 
369             # float custom module to observed custom module 
370             "float_to_observed_custom_module_class": {CustomModule: ObservedCustomModule} 
371         } 
372   
373      """
374      torch._C._log_api_usage_once("quantization_api.quantize.prepare")
375      if prepare_custom_config_dict is None:
376          prepare_custom_config_dict = get_default_custom_config_dict()
377      custom_module_class_mapping = prepare_custom_config_dict.get(
378          "float_to_observed_custom_module_class", {}
379      )
380  
381      if not inplace:
382          model = copy.deepcopy(model)
383  
384      # TODO: remove allow_list
385      qconfig_propagation_list = allow_list
386      if allow_list is None:
387          qconfig_propagation_list = get_default_qconfig_propagation_list()
388      propagate_qconfig_(model, qconfig_dict=None)
389  
390      # sanity check common API misusage
391      if not any(hasattr(m, "qconfig") and m.qconfig for m in model.modules()):
392          warnings.warn(
393              "None of the submodule got qconfig applied. Make sure you "
394              "passed correct configuration through `qconfig_dict` or "
395              "by assigning the `.qconfig` attribute directly on submodules"
396          )
397  
398      _add_observer_(
399          model,
400          qconfig_propagation_list,
401          observer_non_leaf_module_list,
402          custom_module_class_mapping=custom_module_class_mapping,
403      )
404      return model

407  def _remove_activation_post_process(module):
408      # TODO: maybe we should change activation_post_process to _activation_post_process
409      # to prevent it from being used by user
410      if hasattr(module, "activation_post_process") and _is_activation_post_process(
411          module.activation_post_process
412      ):
413          delattr(module, "activation_post_process")
414  
415      # remove activation_post_process pre and post hooks
416      def remove_hooks(pre_hook=False):
417          hook_map = module._forward_pre_hooks if pre_hook else module._forward_hooks
418          observer_hook = (
419              _observer_forward_pre_hook if pre_hook else _observer_forward_hook
420          )
421          handle_ids_to_remove = set()
422          for handle_id, hook_fn in hook_map.items():
423              if hook_fn is observer_hook:
424                  handle_ids_to_remove.add(handle_id)
425          for handle_id in handle_ids_to_remove:
426              hook_map.pop(handle_id)
427  
428      remove_hooks(pre_hook=True)
429      remove_hooks(pre_hook=False)

433  def _remove_qconfig(module):
434      r"""Clean up the qconfig left in the module so that new qconfig can be 
435      propagated. 
436   
437      Args: 
438          module: module to be cleaned up 
439      """
440      for child in module.children():
441          _remove_qconfig(child)
442  
443      if hasattr(module, "qconfig"):
444          del module.qconfig
445  
446      _remove_activation_post_process(module)

449  @typing_extensions.deprecated(DEPRECATION_WARNING)
450  def quantize(model, run_fn, run_args, mapping=None, inplace=False):
451      r"""Quantize the input float model with post training static quantization. 
452   
453      First it will prepare the model for calibration, then it calls 
454      `run_fn` which will run the calibration step, after that we will 
455      convert the model to a quantized model. 
456   
457      Args: 
458          model: input float model 
459          run_fn: a calibration function for calibrating the prepared model 
460          run_args: positional arguments for `run_fn` 
461          inplace: carry out model transformations in-place, the original module is mutated 
462          mapping: correspondence between original module types and quantized counterparts 
463   
464      Return: 
465          Quantized model. 
466      """
467      torch._C._log_api_usage_once("quantization_api.quantize.quantize")
468      if mapping is None:
469          mapping = get_default_static_quant_module_mappings()
470      if not inplace:
471          model = copy.deepcopy(model)
472      model.eval()
473      prepare(model, inplace=True)
474      run_fn(model, *run_args)
475      convert(model, mapping, inplace=True)
476      return model

479  @typing_extensions.deprecated(DEPRECATION_WARNING)
480  def quantize_dynamic(
481      model, qconfig_spec=None, dtype=torch.qint8, mapping=None, inplace=False
482  ):
483      r"""Converts a float model to dynamic (i.e. weights-only) quantized model. 
484   
485      Replaces specified modules with dynamic weight-only quantized versions and output the quantized model. 
486   
487      For simplest usage provide `dtype` argument that can be float16 or qint8. Weight-only quantization 
488      by default is performed for layers with large weights size - i.e. Linear and RNN variants. 
489   
490      Fine grained control is possible with `qconfig` and `mapping` that act similarly to `quantize()`. 
491      If `qconfig` is provided, the `dtype` argument is ignored. 
492   
493      Args: 
494          model: input model 
495          qconfig_spec: Either: 
496   
497              - A dictionary that maps from name or type of submodule to quantization 
498                configuration, qconfig applies to all submodules of a given 
499                module unless qconfig for the submodules are specified (when the 
500                submodule already has qconfig attribute). Entries in the dictionary 
501                need to be QConfig instances. 
502   
503              - A set of types and/or submodule names to apply dynamic quantization to, 
504                in which case the `dtype` argument is used to specify the bit-width 
505   
506          inplace: carry out model transformations in-place, the original module is mutated 
507          mapping: maps type of a submodule to a type of corresponding dynamically quantized version 
508              with which the submodule needs to be replaced 
509   
510      """
511      torch._C._log_api_usage_once("quantization_api.quantize.quantize_dynamic")
512      if qconfig_spec is None:
513          if dtype == torch.qint8:
514              qconfig_spec = {
515                  nn.Linear: default_dynamic_qconfig,
516                  nn.LSTM: default_dynamic_qconfig,
517                  nn.GRU: default_dynamic_qconfig,
518                  nn.LSTMCell: default_dynamic_qconfig,
519                  nn.RNNCell: default_dynamic_qconfig,
520                  nn.GRUCell: default_dynamic_qconfig,
521              }
522          elif dtype == torch.float16:
523              qconfig_spec = {
524                  nn.Linear: float16_dynamic_qconfig,
525                  nn.LSTM: float16_dynamic_qconfig,
526                  nn.GRU: float16_dynamic_qconfig,
527                  nn.LSTMCell: float16_dynamic_qconfig,
528                  nn.RNNCell: float16_dynamic_qconfig,
529                  nn.GRUCell: float16_dynamic_qconfig,
530              }
531          elif dtype == torch.quint8:
532              qconfig_spec = {
533                  nn.EmbeddingBag: float_qparams_weight_only_qconfig,
534                  nn.Embedding: float_qparams_weight_only_qconfig,
535              }
536          elif dtype == torch.quint4x2:
537              qconfig_spec = {
538                  nn.EmbeddingBag: float_qparams_weight_only_qconfig_4bit,
539              }
540          else:
541              raise ValueError(
542                  f"Don't know how to quantize with default settings for {dtype}. Provide full qconfig please"
543              )
544      elif isinstance(qconfig_spec, set):
545          if dtype is torch.qint8:
546              default_qconfig = default_dynamic_qconfig
547          elif dtype is torch.float16:
548              default_qconfig = float16_dynamic_qconfig
549          elif dtype is torch.quint8:
550              default_qconfig = float_qparams_weight_only_qconfig
551          elif dtype is torch.quint4x2:
552              default_qconfig = float_qparams_weight_only_qconfig_4bit
553          else:
554              raise RuntimeError(
555                  "Unknown dtype specified for quantize_dynamic: ", str(dtype)
556              )
557          qconfig_spec = dict(zip(qconfig_spec, itertools.repeat(default_qconfig)))
558  
559      if mapping is None:
560          mapping = get_default_dynamic_quant_module_mappings()
561  
562      if not inplace:
563          model = copy.deepcopy(model)
564      model.eval()
565      propagate_qconfig_(model, qconfig_spec)
566      convert(model, mapping, inplace=True)
567      return model

572      r""" 
573      Prepares a copy of the model for quantization calibration or

596      prepare(model, observer_non_leaf_module_list=set(mapping.values()), inplace=True)

624  @typing_extensions.deprecated(DEPRECATION_WARNING)
625  def convert(
626      module,
627      mapping=None,
628      inplace=False,
629      remove_qconfig=True,
630      is_reference=False,
631      convert_custom_config_dict=None,
632      use_precomputed_fake_quant=False,
633  ):
634      r"""Converts submodules in input module to a different module according to `mapping` 
635      by calling `from_float` method on the target module class. And remove qconfig at the 
636      end if remove_qconfig is set to True. 
637   
638      Args: 
639          `module`: prepared and calibrated module 
640          `mapping`: a dictionary that maps from source module type to target 
641                     module type, can be overwritten to allow swapping user defined 
642                     Modules 
643          `inplace`: carry out model transformations in-place, the original module 
644                     is mutated 
645          `convert_custom_config_dict`: custom configuration dictionary for convert function 
646          `use_precomputed_fake_quant`: a flag to enable use of precomputed fake quant 
647   
648      .. code-block:: python 
649   
650         # Example of convert_custom_config_dict: 
651         convert_custom_config_dict = { 
652             # user will manually define the corresponding quantized 
653             # module class which has a from_observed class method that converts 
654             # observed custom module to quantized custom module 
655             "observed_to_quantized_custom_module_class": { 
656                 ObservedCustomModule: QuantizedCustomModule 
657             } 
658         } 
659   
660      """
661      torch._C._log_api_usage_once("quantization_api.quantize.convert")
662      if not inplace:
663          module = copy.deepcopy(module)
664      _convert(
665          module,
666          mapping,
667          inplace=True,
668          is_reference=is_reference,
669          convert_custom_config_dict=convert_custom_config_dict,
670          use_precomputed_fake_quant=use_precomputed_fake_quant,
671      )
672      if remove_qconfig:
673          _remove_qconfig(module)
674      return module

677  def _convert(
678      module,
679      mapping=None,
680      inplace=False,
681      is_reference=False,
682      convert_custom_config_dict=None,
683      use_precomputed_fake_quant=False,
684  ):
685      r"""Converts submodules in input module to a different module according to `mapping` 
686      by calling `from_float` method on the target module class 
687   
688      Args: 
689          module: input module 
690          mapping: a dictionary that maps from source module type to target 
691                   module type, can be overwritten to allow swapping user defined 
692                   Modules 
693          inplace: carry out model transformations in-place, the original module 
694                   is mutated 
695          is_reference: a flag to enable quantized reference module 
696          use_precomputed_fake_quant: a flag to enable use of precomputed fake quant 
697   
698      """
699      if mapping is None:
700          mapping = (
701              get_default_static_quant_reference_module_mappings()
702              if is_reference
703              else get_default_static_quant_module_mappings()
704          )
705      if convert_custom_config_dict is None:
706          convert_custom_config_dict = get_default_custom_config_dict()
707      custom_module_class_mapping = convert_custom_config_dict.get(
708          "observed_to_quantized_custom_module_class", {}
709      )
710  
711      if not inplace:
712          module = copy.deepcopy(module)
713      reassign = {}
714      for name, mod in module.named_children():
715          # both fused modules and observed custom modules are
716          # swapped as one unit
717          if (
718              not isinstance(mod, _FusedModule)
719              and type_before_parametrizations(mod) not in custom_module_class_mapping
720          ):
721              _convert(
722                  mod,
723                  mapping,
724                  True,  # inplace
725                  is_reference,
726                  convert_custom_config_dict,
727                  use_precomputed_fake_quant=use_precomputed_fake_quant,
728              )
729          reassign[name] = swap_module(
730              mod, mapping, custom_module_class_mapping, use_precomputed_fake_quant
731          )
732  
733      for key, value in reassign.items():
734          module._modules[key] = value
735  
736      return module

739  def swap_module(
740      mod, mapping, custom_module_class_mapping, use_precomputed_fake_quant=False
741  ):
742      r"""Swaps the module if it has a quantized counterpart and it has an 
743      `observer` attached. 
744   
745      Args: 
746          mod: input module 
747          mapping: a dictionary that maps from nn module to nnq module 
748   
749      Return: 
750          The corresponding quantized module of `mod` 
751      """
752      new_mod = mod
753      if hasattr(mod, "qconfig") and mod.qconfig is not None:
754          swapped = False
755          if type_before_parametrizations(mod) in custom_module_class_mapping:
756              new_mod = custom_module_class_mapping[
757                  type_before_parametrizations(mod)
758              ].from_observed(mod)
759              swapped = True
760          elif type_before_parametrizations(mod) in mapping:
761              qmod = mapping[type_before_parametrizations(mod)]
762              if hasattr(qmod, "_IS_REFERENCE") and qmod._IS_REFERENCE:
763                  assert mod.qconfig is not None
764                  weight_post_process = mod.qconfig.weight()
765                  weight_post_process(mod.weight)
766                  weight_qparams = get_qparam_dict(weight_post_process)
767                  new_mod = qmod.from_float(mod, weight_qparams)
768              else:
769                  sig = inspect.signature(qmod.from_float)
770                  if "use_precomputed_fake_quant" in sig.parameters:
771                      new_mod = qmod.from_float(
772                          mod, use_precomputed_fake_quant=use_precomputed_fake_quant
773                      )
774                  else:
775                      new_mod = qmod.from_float(mod)
776              swapped = True
777  
778          if swapped:
779              # Preserve module's pre forward hooks. They'll be called on quantized input
780              for pre_hook_fn in mod._forward_pre_hooks.values():
781                  new_mod.register_forward_pre_hook(pre_hook_fn)
782              # Preserve module's post forward hooks except _observer_forward_hook
783              # After convert they'll work with quantized output
784              for hook_fn in mod._forward_hooks.values():
785                  if hook_fn is not _observer_forward_hook:
786                      new_mod.register_forward_hook(hook_fn)
787  
788              # respect device affinity when swapping modules
789              devices = _get_unique_devices_(mod)
790              assert len(devices) <= 1 or (
791                  len(devices) == 2 and torch.device("meta") in devices
792              ), (
793                  f"swap_module only works with cpu or single-device CUDA modules, but got devices {devices}"
794              )
795              device = next(iter(devices)) if len(devices) > 0 else None
796              if device:
797                  new_mod.to(device)
798      return new_mod

801  def _get_observer_dict(mod, target_dict, prefix=""):
802      r"""Traverse the modules and save all observers into dict. 
803      This is mainly used for quantization accuracy debug 
804      Args: 
805          mod: the top module we want to save all observers 
806          prefix: the prefix for the current module 
807          target_dict: the dictionary used to save all the observers 
808      """
809  
810      def get_prefix(prefix):
811          return prefix if prefix == "" else prefix + "."
812  
813      if hasattr(mod, "activation_post_process"):
814          target_dict[get_prefix(prefix) + "activation_post_process"] = (
815              mod.activation_post_process
816          )
817      for name, child in mod.named_children():
818          module_prefix = get_prefix(prefix) + name if prefix else name
819          _get_observer_dict(child, target_dict, module_prefix)
