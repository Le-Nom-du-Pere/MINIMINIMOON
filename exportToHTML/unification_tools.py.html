25   def _get_factory(f, kwargs):
26       factory = kwargs.pop("factory", dict)
27       if kwargs:
28           raise TypeError(
29               f"{f.__name__}() got an unexpected keyword argument '{kwargs.popitem()[0]}'"
30           )
31       return factory

34   def merge(*dicts, **kwargs):
35       """Merge a collection of dictionaries 
36    
37       >>> merge({1: "one"}, {2: "two"}) 
38       {1: 'one', 2: 'two'} 
39    
40       Later dictionaries have precedence 
41    
42       >>> merge({1: 2, 3: 4}, {3: 3, 4: 4}) 
43       {1: 2, 3: 3, 4: 4} 
44    
45       See Also: 
46           merge_with 
47       """
48       if len(dicts) == 1 and not isinstance(dicts[0], Mapping):
49           dicts = dicts[0]
50       factory = _get_factory(merge, kwargs)
51   
52       rv = factory()
53       for d in dicts:
54           rv.update(d)
55       return rv

58   def merge_with(func, *dicts, **kwargs):
59       """Merge dictionaries and apply function to combined values 
60    
61       A key may occur in more than one dict, and all values mapped from the key 
62       will be passed to the function as a list, such as func([val1, val2, ...]). 
63    
64       >>> merge_with(sum, {1: 1, 2: 2}, {1: 10, 2: 20}) 
65       {1: 11, 2: 22} 
66    
67       >>> merge_with(first, {1: 1, 2: 2}, {2: 20, 3: 30})  # doctest: +SKIP 
68       {1: 1, 2: 2, 3: 30} 
69    
70       See Also: 
71           merge 
72       """
73       if len(dicts) == 1 and not isinstance(dicts[0], Mapping):
74           dicts = dicts[0]
75       factory = _get_factory(merge_with, kwargs)
76   
77       result = factory()
78       for d in dicts:
79           for k, v in d.items():
80               if k not in result:
81                   result[k] = [v]
82               else:
83                   result[k].append(v)
84       return valmap(func, result, factory)

87   def valmap(func, d, factory=dict):
88       """Apply function to values of dictionary 
89    
90       >>> bills = {"Alice": [20, 15, 30], "Bob": [10, 35]} 
91       >>> valmap(sum, bills)  # doctest: +SKIP 
92       {'Alice': 65, 'Bob': 45} 
93    
94       See Also: 
95           keymap 
96           itemmap 
97       """
98       rv = factory()
99       rv.update(zip(d.keys(), map(func, d.values())))
100      return rv

135  def valfilter(predicate, d, factory=dict):
136      """Filter items in dictionary by value 
137   
138      >>> iseven = lambda x: x % 2 == 0 
139      >>> d = {1: 2, 2: 3, 3: 4, 4: 5} 
140      >>> valfilter(iseven, d) 
141      {1: 2, 3: 4} 
142   
143      See Also: 
144          keyfilter 
145          itemfilter 
146          valmap 
147      """
148      rv = factory()
149      for k, v in d.items():
150          if predicate(v):
151              rv[k] = v
152      return rv

175  def itemfilter(predicate, d, factory=dict):
176      """Filter items in dictionary by item 
177   
178      >>> def isvalid(item): 
179      ...     k, v = item 
180      ...     return k % 2 == 0 and v < 4 
181   
182      >>> d = {1: 2, 2: 3, 3: 4, 4: 5} 
183      >>> itemfilter(isvalid, d) 
184      {2: 3} 
185   
186      See Also: 
187          keyfilter 
188          valfilter 
189          itemmap 
190      """
191      rv = factory()
192      for item in d.items():
193          if predicate(item):
194              k, v = item
195              rv[k] = v
196      return rv

199  def assoc(d, key, value, factory=dict):
200      """Return a new dict with new key value pair 
201   
202      New dict has d[key] set to value. Does not modify the initial dictionary. 
203   
204      >>> assoc({"x": 1}, "x", 2) 
205      {'x': 2} 
206      >>> assoc({"x": 1}, "y", 3)  # doctest: +SKIP 
207      {'x': 1, 'y': 3} 
208      """
209      d2 = factory()
210      d2.update(d)
211      d2[key] = value
212      return d2

215  def dissoc(d, *keys, **kwargs):
216      """Return a new dict with the given key(s) removed. 
217   
218      New dict has d[key] deleted for each supplied key. 
219      Does not modify the initial dictionary. 
220   
221      >>> dissoc({"x": 1, "y": 2}, "y") 
222      {'x': 1} 
223      >>> dissoc({"x": 1, "y": 2}, "y", "x") 
224      {} 
225      >>> dissoc({"x": 1}, "y")  # Ignores missing keys 
226      {'x': 1} 
227      """
228      factory = _get_factory(dissoc, kwargs)
229      d2 = factory()
230  
231      if len(keys) < len(d) * 0.6:
232          d2.update(d)
233          for key in keys:
234              if key in d2:
235                  del d2[key]
236      else:
237          remaining = set(d)
238          remaining.difference_update(keys)
239          for k in remaining:
240              d2[k] = d[k]
241      return d2

244  def assoc_in(d, keys, value, factory=dict):
245      """Return a new dict with new, potentially nested, key value pair 
246   
247      >>> purchase = { 
248      ...     "name": "Alice", 
249      ...     "order": {"items": ["Apple", "Orange"], "costs": [0.50, 1.25]}, 
250      ...     "credit card": "5555-1234-1234-1234", 
251      ... } 
252      >>> assoc_in(purchase, ["order", "costs"], [0.25, 1.00])  # doctest: +SKIP 
253      {'credit card': '5555-1234-1234-1234', 
254       'name': 'Alice', 
255       'order': {'costs': [0.25, 1.00], 'items': ['Apple', 'Orange']}} 
256      """
257      return update_in(d, keys, lambda x: value, value, factory)

260  def update_in(d, keys, func, default=None, factory=dict):
261      """Update value in a (potentially) nested dictionary 
262   
263      inputs: 
264      d - dictionary on which to operate 
265      keys - list or tuple giving the location of the value to be changed in d 
266      func - function to operate on that value 
267   
268      If keys == [k0,..,kX] and d[k0]..[kX] == v, update_in returns a copy of the 
269      original dictionary with v replaced by func(v), but does not mutate the 
270      original dictionary. 
271   
272      If k0 is not a key in d, update_in creates nested dictionaries to the depth 
273      specified by the keys, with the innermost value set to func(default). 
274   
275      >>> inc = lambda x: x + 1 
276      >>> update_in({"a": 0}, ["a"], inc) 
277      {'a': 1} 
278   
279      >>> transaction = { 
280      ...     "name": "Alice", 
281      ...     "purchase": {"items": ["Apple", "Orange"], "costs": [0.50, 1.25]}, 
282      ...     "credit card": "5555-1234-1234-1234", 
283      ... } 
284      >>> update_in(transaction, ["purchase", "costs"], sum)  # doctest: +SKIP 
285      {'credit card': '5555-1234-1234-1234', 
286       'name': 'Alice', 
287       'purchase': {'costs': 1.75, 'items': ['Apple', 'Orange']}} 
288   
289      >>> # updating a value when k0 is not in d 
290      >>> update_in({}, [1, 2, 3], str, default="bar") 
291      {1: {2: {3: 'bar'}}} 
292      >>> update_in({1: "foo"}, [2, 3, 4], inc, 0) 
293      {1: 'foo', 2: {3: {4: 1}}} 
294      """
295      ks = iter(keys)
296      k = next(ks)
297  
298      rv = inner = factory()
299      rv.update(d)
300  
301      for key in ks:
302          if k in d:
303              d = d[k]
304              dtemp = factory()
305              dtemp.update(d)
306          else:
307              d = dtemp = factory()
308  
309          inner[k] = inner = dtemp
310          k = key
311  
312      if k in d:
313          inner[k] = func(d[k])
314      else:
315          inner[k] = func(default)
316      return rv

319  def get_in(keys, coll, default=None, no_default=False):
320      """Returns coll[i0][i1]...[iX] where [i0, i1, ..., iX]==keys. 
321   
322      If coll[i0][i1]...[iX] cannot be found, returns ``default``, unless 
323      ``no_default`` is specified, then it raises KeyError or IndexError. 
324   
325      ``get_in`` is a generalization of ``operator.getitem`` for nested data 
326      structures such as dictionaries and lists. 
327   
328      >>> transaction = { 
329      ...     "name": "Alice", 
330      ...     "purchase": {"items": ["Apple", "Orange"], "costs": [0.50, 1.25]}, 
331      ...     "credit card": "5555-1234-1234-1234", 
332      ... } 
333      >>> get_in(["purchase", "items", 0], transaction) 
334      'Apple' 
335      >>> get_in(["name"], transaction) 
336      'Alice' 
337      >>> get_in(["purchase", "total"], transaction) 
338      >>> get_in(["purchase", "items", "apple"], transaction) 
339      >>> get_in(["purchase", "items", 10], transaction) 
340      >>> get_in(["purchase", "total"], transaction, 0) 
341      0 
342      >>> get_in(["y"], {}, no_default=True) 
343      Traceback (most recent call last): 
344          ... 
345      KeyError: 'y' 
346   
347      See Also: 
348          itertoolz.get 
349          operator.getitem 
350      """
351      try:
352          return reduce(operator.getitem, keys, coll)
353      except (KeyError, IndexError, TypeError):
354          if no_default:
355              raise
356          return default

359  def getter(index):
360      if isinstance(index, list):
361          if len(index) == 1:
362              index = index[0]
363              return lambda x: (x[index],)
364          elif index:
365              return operator.itemgetter(*index)
366          else:
367              return lambda x: ()
368      else:
369          return operator.itemgetter(index)

372  def groupby(key, seq):
373      """Group a collection by a key function 
374   
375      >>> names = ["Alice", "Bob", "Charlie", "Dan", "Edith", "Frank"] 
376      >>> groupby(len, names)  # doctest: +SKIP 
377      {3: ['Bob', 'Dan'], 5: ['Alice', 'Edith', 'Frank'], 7: ['Charlie']} 
378   
379      >>> iseven = lambda x: x % 2 == 0 
380      >>> groupby(iseven, [1, 2, 3, 4, 5, 6, 7, 8])  # doctest: +SKIP 
381      {False: [1, 3, 5, 7], True: [2, 4, 6, 8]} 
382   
383      Non-callable keys imply grouping on a member. 
384   
385      >>> groupby( 
386      ...     "gender", 
387      ...     [ 
388      ...         {"name": "Alice", "gender": "F"}, 
389      ...         {"name": "Bob", "gender": "M"}, 
390      ...         {"name": "Charlie", "gender": "M"}, 
391      ...     ], 
392      ... )  # doctest:+SKIP 
393      {'F': [{'gender': 'F', 'name': 'Alice'}], 
394       'M': [{'gender': 'M', 'name': 'Bob'}, 
395             {'gender': 'M', 'name': 'Charlie'}]} 
396   
397      Not to be confused with ``itertools.groupby`` 
398   
399      See Also: 
400          countby 
401      """
402      if not callable(key):
403          key = getter(key)
404      d = collections.defaultdict(lambda: [].append)  # type: ignore[var-annotated]
405      for item in seq:
406          d[key(item)](item)
407      rv = {}
408      for k, v in d.items():
409          rv[k] = v.__self__  # type: ignore[var-annotated, attr-defined]
410      return rv
