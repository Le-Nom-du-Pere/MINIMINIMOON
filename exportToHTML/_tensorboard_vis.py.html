23   def dump_tensorboard_summary(graph_executor, logdir):
24       with FileWriter(logdir) as w:
25           pb_graph = visualize(graph_executor)
26           evt = event_pb2.Event(
27               wall_time=time.time(), graph_def=pb_graph.SerializeToString()
28           )
29           w.add_event(evt)

32   def visualize(graph, name_prefix="", pb_graph=None, executors_it=None):
33       """Visualizes an independent graph, or a graph executor."""
34       value_map = {}
35       pb_graph = pb_graph or graph_pb2.GraphDef()
36   
37       if isinstance(graph, torch._C.GraphExecutorState):
38           visualize_graph_executor(
39               graph, name_prefix, pb_graph, partial(visualize, pb_graph=pb_graph)
40           )
41           return pb_graph
42   
43       # Set up an input node
44       pb_graph.node.add(op="input", name=name_prefix + "input")
45       for i, value in enumerate(graph.param_node().outputs()):
46           value_map[value.unique()] = name_prefix + "input:" + str(i)
47   
48       visualize_rec(graph, value_map, name_prefix, pb_graph, executors_it)
49   
50       # Gather all outputs
51       return_node = pb_graph.node.add(op="output", name=name_prefix + "output")
52       for value in graph.return_node().inputs():
53           return_node.input.append(value_map[value.unique()])
54   
55       return pb_graph

58   def visualize_graph_executor(state, name_prefix, pb_graph, inline_graph):
59       """Append the state of a given GraphExecutor to the graph protobuf. 
60    
61       Args: 
62           state (GraphExecutor or GraphExecutorState): GraphExecutor to display. 
63           name_prefix (str): Name prefix of the containing subgraph. 
64           pb_graph (GraphDef): graph to append to. 
65           inline_graph (Callable): a function that handles setting up a value_map, 
66               so that some graphs in here can be inlined. This is necessary, because 
67               this will simply be `visualize` for the top-level GraphExecutor, 
68               or `inline_graph` for all nested ones. 
69    
70               The signature should look like (Graph, name_prefix) -> (). 
71               It will be called exactly once. 
72    
73       The strategy is to embed all different configurations as independent subgraphs, 
74       while inlining the original graph as the one that actually produces the values. 
75       """
76       if state.autograd_fallback_graph is not None:
77           visualize(
78               graph=state.autograd_fallback_graph,
79               name_prefix=name_prefix + "autograd_fallback/",
80               pb_graph=pb_graph,
81               executors_it=iter(state.autograd_fallback.executors()),
82           )
83   
84       for i, (arg_spec, plan) in enumerate(state.execution_plans.items()):
85           subgraph_name = name_prefix + f"plan{i}/"
86   
87           # Create a disconnected node that will keep information regarding the input
88           # types of this trace. This is unfortunately a bit too verbose to be included
89           # in the subgraph name.
90           input_kinds = pb_graph.node.add(op="INPUT_KIND", name=subgraph_name)
91           input_kinds.attr["inputs"].s = repr(arg_spec).encode("ascii")
92   
93           visualize(plan.graph, subgraph_name, pb_graph, iter(plan.code.executors()))
94   
95           # Show gradient as an independent subgraph of this plan
96           if plan.grad_executor is not None:
97               grad_subgraph_name = subgraph_name + "grad/"
98               visualize(plan.grad_executor, grad_subgraph_name, pb_graph)
99   
100      return inline_graph(state.graph, name_prefix + "original/")

103  def visualize_rec(graph, value_map, name_prefix, pb_graph, executors_it=None):
104      """Recursive part of visualize (basically skips setting up the input and output nodes)."""
105  
106      def inline_graph(subgraph, name, node):
107          rec_value_map = {
108              inp.unique(): value_map[val.unique()]
109              for inp, val in zip(subgraph.inputs(), node.inputs())
110          }
111          visualize_rec(
112              graph=subgraph, value_map=rec_value_map, name_prefix=name, pb_graph=pb_graph
113          )
114          for out, val in zip(subgraph.outputs(), node.outputs()):
115              value_map[val.unique()] = rec_value_map[out.unique()]
116  
117      op_id_counter: defaultdict[str, int] = defaultdict(int)
118  
119      def name_for(node):
120          kind = node.kind()[node.kind().index("::") + 2 :]
121          op_id_counter[kind] += 1
122          return kind, name_prefix + kind + "_" + str(op_id_counter[kind])
123  
124      def add_fusion_group(node):
125          op, name = name_for(node)
126          inline_graph(node.g("Subgraph"), name + "/", node)
127  
128      def add_graph_executor(node):
129          op, name = name_for(node)
130          if executors_it is None:
131              add_node(node)
132          else:
133              ge = next(executors_it)
134              visualize_graph_executor(
135                  ge, name + "/", pb_graph, partial(inline_graph, node=node)
136              )
137  
138      def add_node(node):
139          if node.kind() == "prim::FusionGroup":
140              return add_fusion_group(node)
141          elif node.kind() == "prim::GraphExecutor":
142              return add_graph_executor(node)
143          op, name = name_for(node)
144          pb_node = pb_graph.node.add(op=op, name=name)
145          for value in node.inputs():
146              pb_node.input.append(value_map[value.unique()])
147          # TODO: handle attrs
148          for i, value in enumerate(node.outputs()):
149              value_map[value.unique()] = name + ":" + str(i)
150  
151      for node in graph.nodes():
152          add_node(node)
