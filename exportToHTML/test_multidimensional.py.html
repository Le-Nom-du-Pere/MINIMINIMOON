9    def test_vectorize():
10       @vectorize(0)
11       def vsin(x):
12           return sin(x)

19   
20       assert vdiff([f(x, y, z), g(x, y, z), h(x, y, z)], [x, y, z]) == \
21            [[Derivative(f(x, y, z), x), Derivative(f(x, y, z), y),
22              Derivative(f(x, y, z), z)], [Derivative(g(x, y, z), x),
23                        Derivative(g(x, y, z), y), Derivative(g(x, y, z), z)],
24            [Derivative(h(x, y, z), x), Derivative(h(x, y, z), y), Derivative(h(x, y, z), z)]]
