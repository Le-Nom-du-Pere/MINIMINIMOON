9    // This file defines Vectorized<> for the quantized types.
10   //
11   //
12   // Currently, we simply use these classes as efficient converters between
13   // the quantized types and Vectorized<float>, usually in bandwidth-bound cases
14   // where doing the arithmetic in full-precision is acceptable (e.g.
15   // elementwise operators).
16   //
17   //
18   // Conversions are as follows:
19   //  Vectorized<qint32> -> 1x Vectorized<float>
20   //
21   // The size of the returned float vector is specified by the special
22   // constexpr function float_num_vecs. The type of the value returned
23   // from dequantize (and expected as an argument to quantize) is
24   // specified by float_vec_return_type.
25   //
26   // When writing kernels with these vectors, it is expected that floating-
27   // point operations will be carried out in a loop over
28   // Vectorized<T>::float_num_vecs iterations.
29

34   template <>
35   struct is_vec_specialized_for<c10::qint32> : std::bool_constant<true> {};
36   template <>
37   struct Vectorized<c10::qint32> {
38    private:
39     union {
40       struct {
41         vint32 _vec0;
42         vint32 _vec1;
43       };
44       struct {
45         vbool32 _vecb0;
46         vbool32 _vecb1;
47       };
48   
49     } __attribute__((__may_alias__));
50   
51    public:
52     Vectorized() {}
53   
54     using size_type = int;
55     static constexpr size_type size() {
56       return 8;
57     }
58   
59     static constexpr size_t float_num_vecs() {
60       return 1;
61     }
62     static constexpr int int_num_vecs() {
63       return 1;
64     }
65     using float_vec_return_type = std::array<Vectorized<float>, 1>;
66     using int_vec_return_type = std::array<Vectorized<c10::qint32>, 1>;
67     using value_type = c10::qint32::underlying;
68     using vec_internal_type = vint32;
69     using vec_internal_mask_type = vbool32;
70     C10_ALWAYS_INLINE Vectorized(vint32 v) : _vec0{v}, _vec1{v} {}
71     C10_ALWAYS_INLINE Vectorized(vbool32 vmask) : _vecb0{vmask}, _vecb1{vmask} {}
72     C10_ALWAYS_INLINE Vectorized(vint32 v1, vint32 v2) : _vec0{v1}, _vec1{v2} {}
73     C10_ALWAYS_INLINE Vectorized(vbool32 v1, vbool32 v2)
74         : _vecb0{v1}, _vecb1{v2} {}
75   
76     Vectorized(const c10::qint32& val)
77         : _vec0(vec_splats(val.val_)), _vec1(vec_splats(val.val_)) {}
78   
79     static Vectorized<c10::qint32> C10_ALWAYS_INLINE
80     loadu(const void* ptr, int count = size()) {
81       if (count == size()) {
82         return {
83             vec_vsx_ld(offset0, reinterpret_cast<const value_type*>(ptr)),
84             vec_vsx_ld(offset16, reinterpret_cast<const value_type*>(ptr))};
85       }
86   
87       __at_align__ value_type tmp_values[size()] = {};
88       std::memcpy(tmp_values, ptr, std::min(count, size()) * sizeof(value_type));
89   
90       return {vec_vsx_ld(offset0, tmp_values), vec_vsx_ld(offset16, tmp_values)};
91     }
92     void C10_ALWAYS_INLINE store(void* ptr, int count = size()) const {
93       if (count == size()) {
94         vec_vsx_st(_vec0, offset0, reinterpret_cast<value_type*>(ptr));
95         vec_vsx_st(_vec1, offset16, reinterpret_cast<value_type*>(ptr));
96       } else if (count > 0) {
97         __at_align__ value_type tmp_values[size()];
98         vec_vsx_st(_vec0, offset0, tmp_values);
99         vec_vsx_st(_vec1, offset16, tmp_values);
100        std::memcpy(
101            ptr, tmp_values, std::min(count, size()) * sizeof(value_type));
102      }
103    }
104  
105    C10_ALWAYS_INLINE const vec_internal_type& vec0() const {
106      return _vec0;
107    }
108    C10_ALWAYS_INLINE const vec_internal_type& vec1() const {
109      return _vec1;
110    }

112    float_vec_return_type dequantize(
113        Vectorized<float> scale,
114        Vectorized<float> zero_point,
115        Vectorized<float> scale_zp_premul) const {
116      vfloat32 float_vals0 = vec_float(_vec0);
117      vfloat32 float_vals1 = vec_float(_vec1);
118      vfloat32 scale_vec0 = scale.vec0();
119      vfloat32 scale_vec1 = scale.vec1();
120      vfloat32 zero_point_vec0 = zero_point.vec0();
121      vfloat32 zero_point_vec1 = zero_point.vec1();
122  
123      vfloat32 vec_sub_zero_point_0 = vec_sub(float_vals0, zero_point_vec0);
124      vfloat32 vec_sub_zero_point_1 = vec_sub(float_vals1, zero_point_vec1);
125      Vectorized<float> vf0 = {
126          vec_mul(scale_vec0, vec_sub_zero_point_0),
127          vec_mul(scale_vec1, vec_sub_zero_point_1)};
128      return {vf0};
129    }
130  
131    float_vec_return_type dequantize(
132        Vectorized<float> scale,
133        Vectorized<float> zero_point) const {
134      vfloat32 float_vals0 = vec_float(_vec0);
135      vfloat32 float_vals1 = vec_float(_vec1);
136      vfloat32 scale_vec0 = scale.vec0();
137      vfloat32 scale_vec1 = scale.vec1();
138      vfloat32 zero_point0 = zero_point.vec0();
139      vfloat32 zero_point1 = zero_point.vec1();
140      return {Vectorized<float>{
141          (float_vals0 - zero_point0) * scale_vec0,
142          (float_vals1 - zero_point1) * scale_vec1}};
143    }

145    static Vectorized<c10::qint32> quantize(
146        const float_vec_return_type& rhs,
147        float scale,
148        int32_t zero_point,
149        float inverse_scale) {
150      Vectorized<c10::qint32> retval;
151  
152      const vint32 vmin = vec_splats(std::numeric_limits<value_type>::min());
153      const vint32 vmax = vec_splats(std::numeric_limits<value_type>::max());
154      vfloat32 inverse_scale_v = vec_splats(inverse_scale);
155      vfloat32 vec_zero_point = vec_splats((float)(zero_point));
156      Vectorized<float> vf0 = rhs[0];
157  
158      vfloat32 vecf0 = vf0.vec0();
159      vfloat32 vecf1 = vf0.vec1();
160      vecf0 = vec_mul(vecf0, inverse_scale_v);
161      vecf1 = vec_mul(vecf1, inverse_scale_v);
162      vecf0 = vec_add(vec_rint(vecf0), vec_zero_point);
163      vecf1 = vec_add(vec_rint(vecf1), vec_zero_point);
164      vint32 veci0 = vec_signed(vecf0);
165      vint32 veci1 = vec_signed(vecf1);
166  
167      veci0 = vec_max(veci0, vmin);
168      veci1 = vec_max(veci1, vmin);
169      veci0 = vec_min(veci0, vmax);
170      veci1 = vec_min(veci1, vmax);
171  
172      return {veci0, veci1};
173    }
174  
175    Vectorized<c10::qint32> relu(Vectorized<c10::qint32> zero_point) const {
176      return {vec_max(_vec0, zero_point._vec0), vec_max(_vec1, zero_point._vec1)};
177    }

179    Vectorized<c10::qint32> relu6(
180        Vectorized<c10::qint32> zero_point,
181        Vectorized<c10::qint32> q_six) const {
182      vint32 max0 = vec_max(_vec0, zero_point._vec0);
183      vint32 max1 = vec_max(_vec1, zero_point._vec1);
184      return {vec_min(max0, q_six._vec0), vec_min(max1, q_six._vec1)};
185    }

187    int_vec_return_type widening_subtract(Vectorized<c10::qint32> b) const {
188      return {*this - b};
189    }

191    static Vectorized<c10::qint32> requantize_from_int(
192        const int_vec_return_type& inp,
193        float multiplier,
194        int32_t zero_point) {
195      const vint32 vmin = vec_splats(std::numeric_limits<value_type>::min());
196      const vint32 vmax = vec_splats(std::numeric_limits<value_type>::max());
197      vfloat32 vec_mult = vec_splats(multiplier);
198      vint32 vec_zero_point = vec_splats(zero_point);
199      Vectorized<c10::qint32> vi = inp[0];
200      vfloat32 vecf0 = vec_float(vi.vec0());
201      vfloat32 vecf1 = vec_float(vi.vec1());
202  
203      vecf0 = vec_mul(vecf0, vec_mult);
204      vecf1 = vec_mul(vecf1, vec_mult);
205  
206      vecf0 = vec_rint(vecf0);
207      vecf1 = vec_rint(vecf1);
208  
209      vint32 veci0 = vec_add(vec_signed(vecf0), vec_zero_point);
210      vint32 veci1 = vec_add(vec_signed(vecf1), vec_zero_point);
211  
212      veci0 = vec_max(veci0, vmin);
213      veci1 = vec_max(veci1, vmin);
214      veci0 = vec_min(veci0, vmax);
215      veci1 = vec_min(veci1, vmax);
216  
217      return {veci0, veci1};
218    }
