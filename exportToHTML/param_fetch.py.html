16   # Matching method matches the attribute name of current version to the attribute name of `target_version`
17   @compatibility(is_backward_compatible=False)
18   def default_matching(name: str, target_version: int) -> str:
19       """Default matching method"""
20       return name

23   # This dict maps the nn.Module class name to the attribute name list that we want to fetch for lowering.
24   # The first integer in the tuple is the version number of the nn.Module class when we create the parameter list.
25   # If there's a version mismatch then it means the parameter names in the book might be mismatched with nn.Module.

27       torch.nn.modules.linear.Linear: (1, ["weight", "bias"], default_matching),

57   @compatibility(is_backward_compatible=False)
58   def extract_attrs_for_lowering(mod: nn.Module) -> dict[str, Any]:
59       """If `mod` is in `module_fetch_book`, fetch the mod's attributes that in the `module_fetch_book` 
60       after checking module's version is compatible with the `module_fetch_book`. 
61       """
62       attrs_for_lowering: dict[str, Any] = {}
63       attrs_for_lowering["name"] = torch.typename(mod)
64   
65       if type(mod) in module_fetch_book:
66           version, param_to_fetch, matching_method = module_fetch_book[type(mod)]
67           if version < mod._version:
68               raise RuntimeError(
69                   f"Fetcher version {version} try to fetch {torch.typename(mod)} version {mod._version}, "
70                   "please upgrade the module_fetch_book, open an issue and @842974287 "
71                   "or report a bug to AIACC team directly."
72               )
73           for attr in param_to_fetch:
74               attrs_for_lowering[attr] = getattr(mod, matching_method(attr, mod._version))
75       else:
76           raise RuntimeError(
77               f"{torch.typename(mod)} is not in the module_fetch_book yet, "
78               "please add it to the module_fetch_book, open an issue and @842974287 "
79               "or report a bug to AIACC team directly."
80           )
81       return attrs_for_lowering

84   @compatibility(is_backward_compatible=False)
85   def lift_lowering_attrs_to_nodes(fx_module: GraphModule) -> None:
86       """Recursively traverse all `fx_module` nodes and fetch the module's attributes if the node is a leaf module."""
87       submodules = dict(fx_module.named_modules())
88   
89       for node in fx_module.graph.nodes:
90           if node.op == "call_module":
91               if isinstance(submodules[node.target], GraphModule):
92                   lift_lowering_attrs_to_nodes(submodules[node.target])
93               else:
94                   node.attrs_for_lowering = extract_attrs_for_lowering(
95                       submodules[node.target]
96                   )
