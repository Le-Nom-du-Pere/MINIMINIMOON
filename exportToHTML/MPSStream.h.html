46   enum class SyncType {
47     NONE, // no commit to command buffer
48     COMMIT, // commit and flush the command buffer
49     COMMIT_AND_WAIT, // flush and wait for command buffer execution to finish
50     COMMIT_AND_CONTINUE, // commit and continue with a new underlying command buffer
51     COMMIT_ADAPTIVE, // commit adaptively based on available memory
52   };
53   
54   class TORCH_API MPSStream {
55    public:
56     enum Unchecked { UNCHECKED };
57   
58     /// Construct a MPSStream from a Stream.  This construction is checked,
59     /// and will raise an error if the Stream is not, in fact, a MPS stream.
60     explicit MPSStream(Stream stream);
61   
62     ~MPSStream();
63   
64     MTLCommandQueue_t commandQueue() const {
65       return _commandQueue;
66     }
67   
68     dispatch_queue_t queue() const {
69       return _serialQueue;
70     }
71   
72     MPSCommandBuffer_t commandBuffer();
73     MTLComputeCommandEncoder_t commandEncoder();
74     void endKernelCoalescing();
75     void synchronize(SyncType syncType);
76     void fill(MTLBuffer_t buffer, uint8_t value, size_t length, size_t offset, SyncType syncType = SyncType::NONE);
77     void copy(MTLBuffer_t srcBuffer,
78               MTLBuffer_t dstBuffer,
79               size_t length,
80               size_t srcOffset,
81               size_t dstOffset,
82               uint64_t profileId,
83               SyncType syncType = SyncType::NONE);
84     void copy_and_sync(MTLBuffer_t srcBuffer,
85                        MTLBuffer_t dstBuffer,
86                        size_t length,
87                        size_t srcOffset,
88                        size_t dstOffset,
89                        bool non_blocking,
90                        uint64_t profileId);
91     void executeMPSGraph(MPSGraph* mpsGraph,
92                          NSDictionary* feeds,
93                          NSDictionary* results,
94                          SyncType syncType = SyncType::NONE);
95     void addCompletedHandler(MTLCommandBufferHandler block);
96   
97     /// Get the MPS device index that this stream is associated with.
98     c10::DeviceIndex device_index() const {
99       return _stream.device_index();
100    }
101  
102    MTLCommandQueue_t stream() const {
103      return _commandQueue;
104    }
105  
106    MTLDevice_t device() const;
107  
108    /// Explicit conversion to Stream.
109    Stream unwrap() const {
110      return _stream;
111    }
112  
113   private:
114    Stream _stream;
115    MTLCommandQueue_t _commandQueue = nil;
116    MPSCommandBuffer_t _commandBuffer = nil;
117    MPSCommandBuffer_t _prevCommandBuffer = nil;
118    MTLComputeCommandEncoder_t _commandEncoder = nil;
119    MPSGraphExecutionDescriptor* _executionDescriptor = nil;
120    MPSGraphCompilationDescriptor* _compilationDescriptor = nil;
121    dispatch_queue_t _serialQueue = nullptr;
122    // CommitAndContinue is enabled by default
123    bool _enableCommitAndContinue = true;
124  
125    // use synchronize() to access any of these commit functions outside MPSStream
126    void commit();
127    void commitAndWait();
128    void commitAndContinue();
129    void flush();
130  };
