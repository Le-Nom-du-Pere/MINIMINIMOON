18   class RestoreParameterAndBufferNames(_pass.Transform):
19       """Restore parameter and buffer names from original nn.module. 
20    
21       This pass is useful for readability of the exported ONNX graph. It restores the 
22       parameter and buffer names from the original nn.module. For example, if the original 
23       nn.module has a parameter named `root.linear.0.weight`, and the parameter is renamed to 
24       `_param_constant9` by FX, this pass will rename it back. 
25    
26       This pass must be run after `Decompose` pass. Because this pass is expected to be called on 
27       `fx.GraphModule` produced by `proxy_tensor.make_fx`, where all parameters and buffers 
28       are registered at root level. 
29       """
30   
31       def __init__(
32           self,
33           fx_module: torch.fx.GraphModule,
34           original_nn_module: torch.nn.Module,
35       ):
36           super().__init__(fx_module)
37           self.original_nn_module = original_nn_module
38   
39       def _rename_param_and_buffer(
40           self,
41           nodes: Sequence[torch.fx.Node],
42           new_name: str,
43       ) -> None:
44           """Rename the parameter/buffer and replace corresponding nodes with new nodes of updated target."""
45           assert len(nodes) > 0, "`nodes` cannot be empty"
46           assert len({node.target for node in nodes}) == 1, (
47               "`nodes` must all have same `target`"
48           )
49           old_name = nodes[0].target
50           assert isinstance(old_name, str), f"Expected str, got type({old_name})"
51           # Parameter/buffer name cannot contain "."
52           normalized_name = new_name.replace(".", "/")
53           attr_value = getattr(self.module, old_name)
54           setattr(self.module, normalized_name, attr_value)
55           delattr(self.module, old_name)
56           for node in nodes:
57               with self.module.graph.inserting_before(node):
58                   new_node = self.module.graph.get_attr(normalized_name)
59                   new_node.meta = node.meta
60                   node.replace_all_uses_with(new_node)
61                   self.module.graph.erase_node(node)
62           logger.info(
63               "Renamed 'self.%s' to 'self.%s', "
64               "normalized from original parameter name '%s'.",
65               old_name,
66               normalized_name,
67               new_name,
68           )
69   
70       def _run(self, *args, **kwargs) -> torch.fx.GraphModule:
71           """Restore parameter and buffer names from original module. 
72    
73           For each `get_attr` node, if the target is a str representing a parameter or buffer 
74           under `self.module`, we rename the parameter or buffer to its original name. 
75           The parameters and buffers between `self.module` and `self.original_nn_module` refer 
76           to the same objects, allowing us to use it as key to retrieve the original name. 
77           """
78           assert len(args) == 0, "RestoreParameterAndBufferNames does not take any args"
79           assert len(kwargs) == 0, (
80               "RestoreParameterAndBufferNames does not take any kwargs"
81           )
82           # state_to_readable_name[parameter/buffer] returns the original readable name of
83           # the parameter/buffer. E.g., "self.linear.weight".
84           state_to_readable_name: dict[torch.nn.Parameter | torch.Tensor, str] = {}
85           state_to_readable_name.update(
86               {v: k for k, v in self.original_nn_module.named_parameters()}
87           )
88           state_to_readable_name.update(
89               {v: k for k, v in self.original_nn_module.named_buffers()}
90           )
91   
92           # old_name_to_nodes[old_name] returns a tuple of (nodes, new_name)
93           # where `nodes` is a list of `get_attr` nodes with `old_name` as `target` and
94           # `new_name` is the new readable name.
95           old_name_to_nodes: dict[str, tuple[list[torch.fx.Node], str]] = {}
96   
97           for node in self.module.graph.nodes:
98               if node.op == "get_attr":
99                   assert isinstance(node.target, str), (
100                      f"Expected str, got type({node.target})"
101                  )
102                  if node.target.find(".") != -1:
103                      raise RuntimeError(
104                          f"Unexpected target {node.target} in get_attr, found '.' in target. "
105                          f"All parameters and buffers are expected to be registered at root level, "
106                          f"i.e., self.module. "
107                      )
108                  if node.target in old_name_to_nodes:
109                      # We have already processed this parameter/buffer.
110                      old_name_to_nodes[node.target][0].append(node)
111                      continue
112                  attr_value = getattr(self.module, node.target)
113                  if (
114                      isinstance(attr_value, (torch.nn.Parameter, torch.Tensor))
115                      and attr_value in state_to_readable_name
116                  ):
117                      readable_name = state_to_readable_name[attr_value]
118                      old_name_to_nodes[node.target] = ([node], readable_name)
119                      continue
120  
121                  logger.info(
122                      "Cannot find readable name for self.%s: %s. The name is unchanged.",
123                      node.target,
124                      type(attr_value),
125                  )
126  
127          for nodes, new_name in old_name_to_nodes.values():
128              self._rename_param_and_buffer(nodes, new_name)
129  
130          return self.module
