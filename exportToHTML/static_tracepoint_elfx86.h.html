114  // Structure of note section for the probe.
115  #define TORCH_SDT_NOTE_CONTENT(provider, name, has_semaphore, arg_template)    \
116    TORCH_SDT_ASM_1(990: TORCH_SDT_NOP)                                          \
117    TORCH_SDT_ASM_3(     .pushsection .note.stapsdt,"","note")                   \
118    TORCH_SDT_ASM_1(     .balign 4)                                              \
119    TORCH_SDT_ASM_3(     .4byte 992f-991f, 994f-993f, TORCH_SDT_NOTE_TYPE)       \
120    TORCH_SDT_ASM_1(991: .asciz TORCH_SDT_NOTE_NAME)                             \
121    TORCH_SDT_ASM_1(992: .balign 4)                                              \
122    TORCH_SDT_ASM_1(993: TORCH_SDT_ASM_ADDR 990b)                                \
123    TORCH_SDT_ASM_1(     TORCH_SDT_ASM_ADDR 0) /*Reserved for Base Address*/     \
124    TORCH_SDT_SEMAPHORE_NOTE_##has_semaphore(provider, name)                     \
125    TORCH_SDT_ASM_STRING(provider)                                               \
126    TORCH_SDT_ASM_STRING(name)                                                   \
127    TORCH_SDT_ASM_STRING(arg_template)                                           \
128    TORCH_SDT_ASM_1(994: .balign 4)                                              \
129    TORCH_SDT_ASM_1(     .popsection)
