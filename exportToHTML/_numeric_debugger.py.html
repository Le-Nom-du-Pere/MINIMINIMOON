21   def generate_numeric_debug_handle(ep: ExportedProgram) -> None:
22       """ 
23       Attach numeric_debug_handle_id for all nodes in the graph module of the given 
24       ExportedProgram, like conv2d, squeeze, conv1d, etc, except for placeholder. 
25       Notice that nodes like getattr are out of scope since they are not in the graph. 
26    
27       The graph nodes of input exported program are modified inplace. 
28    
29       Here's an example of using debug handle quantize flow:: 
30    
31           ep = export_for_training(eager_model, example_inputs) 
32           generate_numeric_debug_handle(ep) 
33    
34           m = ep.module() 
35           quantizer = XNNPACKQuantizer() 
36           m = prepare_pt2e(m, quantizer) 
37           m = convert_pt2e(m) 
38       """
39   
40       # Sanity check the input data type
41       if not isinstance(ep, ExportedProgram):
42           raise ValueError(
43               f"Expected ep to be ExportedProgram, got {type(ExportedProgram)}"
44           )
45   
46       unique_id = 0
47   
48       def _find_max_id(node: torch.fx.Node) -> None:
49           nonlocal unique_id
50           unique_id = max(
51               unique_id, node.meta.get(CUSTOM_KEY, {}).get(NUMERIC_DEBUG_HANDLE_KEY, 0)
52           )
53   
54       def _assign_debug_handle(node: torch.fx.Node) -> None:
55           nonlocal unique_id
56           if CUSTOM_KEY not in node.meta:
57               node.meta[CUSTOM_KEY] = {}
58   
59           if NUMERIC_DEBUG_HANDLE_KEY not in node.meta[CUSTOM_KEY]:
60               node.meta[CUSTOM_KEY][NUMERIC_DEBUG_HANDLE_KEY] = unique_id
61               unique_id += 1
62   
63       # Find the max ID that exists in the graph first, in case part of the graph
64       # has already been annotated. This way we guarantee there are no duplicate
65       # handle IDs.
66       bfs_trace_with_node_process(ep, _find_max_id)
67   
68       unique_id += 1
69   
70       # Assign debug handles to all nodes in the graph that don't have one based on the
71       # max ID found in the previous step.
72       bfs_trace_with_node_process(ep, _assign_debug_handle)

75   def _detach(x: object) -> object:
76       detached: object = None
77       if isinstance(x, torch.Tensor):
78           detached = x.detach()
79       elif isinstance(x, (list, tuple)):
80           detached = type(x)([_detach(e) for e in x])
81       elif isinstance(x, dict):
82           detached = {k: _detach(e) for k, e in x.items()}
83       else:
84           detached = x
85       return detached

88   def _tensor_shape_equals(x: object, y: object) -> bool:
89       if isinstance(x, torch.Tensor) and isinstance(y, torch.Tensor):
90           return x.shape == y.shape
91       elif isinstance(x, (list, tuple)) and isinstance(y, (list, tuple)):
92           return all(_tensor_shape_equals(e1, e2) for e1, e2 in zip(x, y))
93       elif isinstance(x, dict) and isinstance(y, dict):
94           all_equal = True
95           for k in x:
96               all_equal = all_equal and k in y and (_tensor_shape_equals(x[k], y[k]))
97           return all_equal
98       else:
99           log.debug("Comparing non Tensors: %s and %s, they must be equal", x, y)
100          return type(x) == type(y) and x == y

103  def _loss_fn(
104      loss: Callable[[torch.Tensor, torch.Tensor], torch.Tensor], x: object, y: object
105  ) -> object:
106      """The returned loss will have the same structure as `x` and `y`, e.g. 
107      if both are Tensor, we'll return a Tensor 
108      if both are list, we'll return a list of Tensors 
109      if both are dict, we'll return a dict with the same key, and value being the loss between the 
110      two Tensors 
111      """
112      if isinstance(x, torch.Tensor) and isinstance(y, torch.Tensor):
113          return loss(x.to(torch.float32), y.to(torch.float32))
114      elif isinstance(x, (list, tuple)) and isinstance(y, (list, tuple)):
115          return type(x)([_loss_fn(loss, e1, e2) for e1, e2 in zip(x, y)])
116      elif isinstance(x, dict) and isinstance(y, dict):
117          return {k: _loss_fn(loss, e, y[k]) for k, e in x.items()}
118      else:
119          return None

122  class OutputLogger(torch.nn.Module):
123      """ 
124      Base class for capturing output values for nodes in a GraphModule, it only captures 
125      Tensor output currently, but we can extend it to work for other types of inputs later if needed 
126      """
127  
128      # Mark as impure so that calls to it will not be removed during DCE.
129      _is_impure = True
130  
131      def __init__(
132          self,
133          debug_handle: int,
134          node_name: Optional[str] = None,
135          nn_module_stack: Optional[object] = None,
136      ) -> None:
137          super().__init__()
138          self.node_name = node_name
139          self.nn_module_stack = nn_module_stack
140          self.debug_handle = debug_handle
141          self.stats: list[object] = []
142  
143      def forward(self, x: object) -> object:
144          self.stats.append(_detach(x))
145          return x
146  
147      def __extra_repr__(self) -> str:
148          return (
149              f"debug_handle={self.debug_handle}, node_name={self.node_name}, "
150              "nn_module_stack={self.nn_module_stack}, num_stats={len(self.stats)})"
151          )

154  def _insert_logger(model: GraphModule, node: Node, debug_handle: int) -> Node:
155      """For a given node, adds an OutputLogger that observes the output of that node, 
156      and all its users use the OutputLogger output instead. 
157      The OutputLogger will contain the debug_handle which can be used to compare 
158      graphs after transforms"""
159  
160      # to avoid circular dep
161      from torch.ao.quantization.fx.utils import get_new_attr_name_with_prefix
162  
163      # add a logger after the node
164      with model.graph.inserting_after(node):
165          get_new_attr_name = get_new_attr_name_with_prefix(f"{node.name}_logger")
166          logger_name = get_new_attr_name(model)
167          setattr(
168              model,
169              logger_name,
170              OutputLogger(debug_handle, node.name, node.meta.get("nn_module_stack")),
171          )
172          logger_node = model.graph.call_module(logger_name, (node,), {})
173  
174      orig_users = list(node.users.keys())
175      for user_node in orig_users:
176          if user_node is logger_node:
177              continue
178          user_node.replace_input_with(node, logger_node)
179  
180      return logger_node

183  def prepare_for_propagation_comparison(model: GraphModule) -> GraphModule:
184      """Add output loggers to node that has numeric_debug_handle 
185   
186      Args: 
187          model (GraphModule): original model 
188      Returns: 
189          a model with output loggers for all nodes that has numeric_debug_handle_id 
190      """
191      # don't change the original model
192      model = copy.deepcopy(model)
193      for n in model.graph.nodes:
194          if (
195              CUSTOM_KEY not in n.meta
196              or NUMERIC_DEBUG_HANDLE_KEY not in n.meta[CUSTOM_KEY]
197          ):
198              continue
199          numeric_debug_handle = n.meta[CUSTOM_KEY][NUMERIC_DEBUG_HANDLE_KEY]
200          _insert_logger(model, n, numeric_debug_handle)
201  
202      model.recompile()
203      return model

206  @dataclass(frozen=True)
207  class QuantizationComparisonResult:
208      actual: torch.Tensor
209      ref: torch.Tensor
210  
211      @property
212      def mse_loss(self) -> object:
213          return self.loss(F.mse_loss)
214  
215      @property
216      def sqnr(self) -> object:
217          return self.loss(compute_sqnr)
218  
219      def loss(
220          self, loss_function: Callable[[torch.Tensor, torch.Tensor], torch.Tensor]
221      ) -> object:
222          return _loss_fn(loss_function, self.actual, self.ref)
223  
224      def __repr__(self) -> str:
225          # Don't include the tensors themselves as they are quite large to print
226          # out.
227          return (
228              f"QuantizationComparisonResult(mse_loss={self.mse_loss}, sqnr={self.sqnr})"
229          )
230  
231      def __post_init__(self) -> None:
232          if not isinstance(self.actual, (torch.Tensor, list, tuple, dict)):
233              raise ValueError(
234                  f"`self.actual` value must be a Tensor, list, tuple or dict, got: {self.actual}"
235              )
236  
237          if not isinstance(self.ref, (torch.Tensor, list, tuple, dict)):
238              raise ValueError(
239                  f"`self.ref` value must be a Tensor, list, tuple or dict, got: {self.ref}"
240              )
241  
242          if not _tensor_shape_equals(self.ref, self.actual):
243              raise ValueError(
244                  f"Cannot compare tensors with different shapes: ref={self.ref} vs actual={self.actual}"
245              )

258  def _module_stack_to_str(module_stack: object) -> str:
259      """Simplifies the stack from ("mod", "mod.foo", "mod.foo.0", "mod.foo.0.linear") 
260      to "mod.foo.0.linear" 
261      """
262      if not isinstance(module_stack, dict):
263          return str(module_stack)
264      module_values_list = list(module_stack.values())
265      if len(module_values_list) > 0:
266          owning_module = module_values_list[-1][0]
267          return str(owning_module)
268      else:
269          return str(module_stack)

272  def extract_results_from_loggers(
273      model: GraphModule,
274  ) -> dict[int, tuple[Optional[str], object, list[object]]]:
275      """For a given model, extract the tensors stats and related information for each debug handle. 
276      The reason we have a list of object, instead of Tensor is because the output of node may not be 
277      a Tensor, it could be (nested) list, tuple or dict as well. 
278   
279      Returns: 
280          A dict is keyed by the debug_handle id and the values are a list of object recorded 
281          in loggers 
282   
283      """
284      # Results maps debug handle to a tensor list for each model being compared.
285      handles: dict[int, tuple[Optional[str], object, list[object]]] = {}
286      for _name, module in model.named_children():
287          if isinstance(module, OutputLogger) and len(module.stats) > 0:
288              handles[module.debug_handle] = (
289                  module.node_name,
290                  module.nn_module_stack,
291                  module.stats,
292              )
293  
294      return handles

297  def compare_results(
298      ref_results: dict[int, tuple[Optional[str], object, list[torch.Tensor]]],
299      actual_results: dict[int, tuple[Optional[str], object, list[torch.Tensor]]],
300  ) -> dict[int, NodeAccuracySummary]:
301      """Given two dict mapping from `debug_handle_id` (int) to list of tensors 
302      return a map from `debug_handle_id` to `NodeAccuracySummary` that contains 
303      comparison information like SQNR, MSE etc. 
304   
305      Args: 
306          ref_results (Dict[int, Tuple[str, object, List[torch.Tensor]]]): reference results for each debug_handle_id 
307          actual_results (Dict[int, Tuple[str, object, List[torch.Tensor]]]): actual results for each debug_handle_id 
308   
309      Returns: 
310          Dict[int, NodeAccuracySummary] 
311      """
312      comparisons = {}
313      for debug_handle, (ref_name, ref_stack, ref_stats) in ref_results.items():
314          if debug_handle not in actual_results:
315              log.debug(
316                  "Cannot compare for handle %s because it wasn't found in the transformed model",
317                  debug_handle,
318              )
319              continue
320          actual_name, actual_stack, actual_stats = actual_results[debug_handle]
321          try:
322              results = [
323                  QuantizationComparisonResult(actual=a, ref=b)
324                  for a, b in zip(actual_stats, ref_stats)
325              ]
326          except Exception as e:
327              # Add extra information for an exception from QuantizationComparisonResult
328              # if the shapes didn't match, to include the handle and the node names.
329              raise ValueError(
330                  f"For numeric_debug_handle={debug_handle} from ref node {ref_name} and actual node {actual_name}"
331              ) from e
332  
333          comparisons[debug_handle] = NodeAccuracySummary(
334              handle=debug_handle,
335              actual_node_name=actual_name or "",
336              actual_module_stack=_module_stack_to_str(actual_stack),
337              ref_node_name=ref_name or "",
338              ref_module_stack=_module_stack_to_str(ref_stack),
339              results=results,
340          )
341  
342      return comparisons
