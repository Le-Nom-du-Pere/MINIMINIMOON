13   // Semantically, each value of BackendComponent identifies a "backend" for our
14   // dispatch. Some functionalities that we may dispatch to are allowed to
15   // register different handlers for each backend. The BackendComponent is then
16   // used to figure out which backend implementation to dispatch to.
17   
18   // In implementation terms, the backend component identifies a specific "bit" in
19   // a DispatchKeySet. The bits in the DispatchKeySet are split between the bottom
20   // ~12 "BackendComponent" bits, while the remaining upper bits are assigned to
21   // functionalities. When we encounter a functionality bit that is known to be
22   // customizable per-backend, then we also look at the lower BackendComponent
23   // bits and take the highest bit to determine which backend's implementation to
24   // use.
25   
26   // WARNING!  If you add a new backend component to the end of this list,
27   // make sure you register it before Meta.
28   // Meta must be at the end so that meta key in tls triggers meta kernels.
29   // (But you shouldn't: private use keys should have higher precedence than all
30   // built-in keys)
31   
32   // If you add a new (non-privateuse) backend here,
33   // make sure to add an Autograd<Backend> fallthrough kernel
34   // in aten/src/ATen/core/VariableFallbackKernel.cpp

41     _(MPS, extra)                                 \
42     _(IPU, extra)                                 \
43     _(XPU, extra)                                 \
44     _(HPU, extra)                                 \
45     _(VE, extra)                                  \
46     _(Lazy, extra)                                \
47     _(MTIA, extra)                                \
48     _(MAIA, extra)                                \
49     _(PrivateUse1, extra)                         \
50     _(PrivateUse2, extra)                         \
51     _(PrivateUse3, extra)                         \

65   enum class BackendComponent : uint8_t {
66   
67     // A "backend" is colloquially used to refer to handlers for dispatch
68     // which actually implement the numerics of an operation in question.
69     //
70     // Due to the nature of the enum, these backends are specified in
71     // an ordered way, but for most backends this order is not semantically
72     // meaningful (e.g., it's valid to reorder these backends without changing
73     // semantics).  The only situation when backend ordering is meaningful
74     // is when the backend participates in multiple dispatch with another
75     // backend; e.g., CPU and CUDA (cuda must have higher priority).
76   
77     // These keys don't correspond to individual kernels.
78     // Instead, they represent the backends that are allowed to override specific
79     // pieces of functionality:
80     // - dense kernels (e.g. DispatchKey::CPU)
81     // - sparse kernels (e.g. DispatchKey::SparseCPU)
82     // - quantized kernels (e.g. DispatchKey::QuantizedCPU)
83     // - autograd kernels (e.g. DispatchKey::AutogradCPU)
84     // We reserve space in the runtime operator table for this full cross product
85     // of
86     // [backends in this enum] x [keys below that are explicitly marked as having
87     // per-backend functionality]
88     //
89     // A meta tensor is a tensor without any data associated with it.  (They
90     // have also colloquially been referred to as tensors on the "null" device).
91     // A meta tensor can be used to dry run operators without actually doing any
92     // computation, e.g., add on two meta tensors would give you another meta
93     // tensor with the output shape and dtype, but wouldn't actually add anything.
94   
95     InvalidBit = 0,
96   #define DEFINE_BACKEND_COMPONENT(n, _) n##Bit,
97     C10_FORALL_BACKEND_COMPONENTS(DEFINE_BACKEND_COMPONENT, unused)
98   #undef DEFINE_BACKEND_COMPONENT
99   
100    // Define an alias to represent end of backend dispatch keys.
101    // If you add new backend keys after PrivateUse3, please also update it here.
102    EndOfBackendKeys = MetaBit,
103  };

105  // Semantically, a dispatch key identifies a possible "level" in our
106  // dispatch, for which a handler may be registered. Each handler corresponds
107  // to a type of functionality.
108  //
109  // In implementation terms, the dispatch key identifies a specific "bit" in a
110  // DispatchKeySet.  Higher bit indexes get handled by dispatching first (because
111  // we "count leading zeros" when we extract the highest priority dispatch
112  // key.)
113  //
114  // Note [DispatchKey Classification]
115  // This enum actually contains several types of keys, which are explained
116  // in more detail further down:
117  // (1) non-customizable backends (e.g. FPGA)
118  // (2) non-customizable functionalities (e.g. Functionalize)
119  // (3) functionalized that are customizable per backend (e.g. Dense, Sparse,
120  // AutogradFunctionality) (4) per-backend instances of customizable
121  // functionalities (e.g. CPU, SparseCPU, AutogradCPU) (5) alias keys (e.g.
122  // CompositeImplicitAutograd)
123  //
124  // Of the categories above, it's important to note:
125  // (a) which keys are assigned individual bits in a DispatchKeySet
126  // (b) which keys are assigned individual slots in the runtime operator table
127  // ("Runtime keys")
128  //
129  // (1), (2) and (3) all get their own dedicated bits in the DispatchKeySet.
130  // (1), (2) and (4) all get their own dedicated slots in the runtime operator
131  // table.
132  
133  // See Note [DispatchKeySet Internal Representation] for more details.
134  //
135  // NOTE: Keep the list in sync with `DispatchKey` in torchgen/model.py

495  // Note [Private use DispatchKey]
496  // ~~~~~~~~~~~~~~~~~~~~~~~~~~~
497  // Private use tensor IDs are preallocated tensor type IDs for use in user
498  // applications.  Similar to private use fields in HTTP, they can be used
499  // by end users for experimental or private applications, without needing
500  // to "standardize" the tensor ID (which would be done by submitting a PR
501  // to PyTorch to add your type ID).
502  //
503  // Private use tensor IDs are appropriate to use if you want to experiment
504  // with adding a new tensor type (without having to patch PyTorch first) or
505  // have a private, non-distributed application that needs to make use of a
506  // new tensor type.  Private use tensor IDs are NOT appropriate to use for
507  // libraries intended to be distributed to further users: please contact
508  // the PyTorch developers to get a type ID registered in this case.
509  //
510  // We provide two classes of private user tensor id: regular DispatchKeys
511  // and Autograd DispatchKeys.  DispatchKeys serve the role of ordinary "backend"
512  // DispatchKeys; if you were adding support for a new type of accelerator, you
513  // would use a backend DispatchKey, and ideally automatically reuse
514  // AutogradOther definitions already defined in PyTorch.  AutogradPrivateUse
515  // DispatchKeys serve as "wrapper" DispatchKeys: they are only necessary for
516  // tensors that compose multiple internal tensors, and for cases when the
517  // built-in autograd formulas for operators are not appropriate.
518  
519  static_assert(
520      (static_cast<uint8_t>(BackendComponent::EndOfBackendKeys) +
521       static_cast<uint8_t>(DispatchKey::EndOfFunctionalityKeys)) <= 64,
522      "The BackendComponent and DispatchKey enums (below EndOfFunctionalityKeys)"
523      " both map to backend and functionality bits"
524      " into a 64-bit bitmask; you must have less than 64 total entries between them");
525  
526  // Check if a DispatchKey is an alias mapping to other runtime keys.
527  constexpr bool isAliasDispatchKey(DispatchKey k) {
528    return k >= DispatchKey::StartOfAliasKeys && k <= DispatchKey::EndOfAliasKeys;
529  }

531  // [Note: Per-Backend Functionality Dispatch Keys]
532  // Check if a DispatchKey is a per-backend functionality key
533  // Any functionalities that can be customized per-backend should be added here.
534  // These keys correspond to functionalities that can be customized individually
535  // per backend. While they only take up one bit in the `DispatchKeySet` bitset,
536  // they map to (# backends) slots in the operator table.
537  // Each of these keys also has a separate set of "runtime keys" in the dispatch
538  // key enum, per backend, which *do* map to the individual operator table slots.
539  // For example, the "Sparse" key maps to an individual bit in the
540  // DispatchKeySet, while `SparseCPU`, `SparseCUDA`, etc all map to individual
541  // slots in the runtime operator table.
542  
543  constexpr bool isPerBackendFunctionalityKey(DispatchKey k) {
544    if (k == DispatchKey::Dense || k == DispatchKey::Quantized ||
545        k == DispatchKey::Sparse || k == DispatchKey::SparseCsr ||
546        k == DispatchKey::AutogradFunctionality ||
547        k == DispatchKey::NestedTensor) {
548      return true;
549    } else {
550      return false;
551    }
552  }

554  // Note that this includes Undefined in the total count.
555  // BUT EndOfFunctionalityKeys is its own (placeholder) key.
556  // e.g. Undefined=0, Dense=1, Sparse=2, EndOfFunctionalityKeys=3.
557  // In the above example, there are 3 total functionality keys.
558  constexpr uint8_t num_functionality_keys =
559      static_cast<uint8_t>(DispatchKey::EndOfFunctionalityKeys);
560  
561  constexpr uint8_t num_backends =
562      static_cast<uint8_t>(BackendComponent::EndOfBackendKeys);
563  
564  // Note [No More Than 16 Backends]
565  // Search for this note to find places in the code where the "no more than 16
566  // backends" invariant is baked in.
567  static_assert(
568      static_cast<uint8_t>(BackendComponent::EndOfBackendKeys) <= 16,
569      "BackendComponent currently only supports <= 16 backends. If we really need to extend this, \
570  there are a few places where this invariant is baked in");
571  
572  constexpr uint8_t numPerBackendFunctionalityKeys() {
573    uint8_t count = 0;
574    for (uint8_t k = 0; k <= num_functionality_keys; ++k) {
575      if (isPerBackendFunctionalityKey(static_cast<DispatchKey>(k)))
576        ++count;
577    }
578    return count;
579  }

581  #if defined(C10_MOBILE_TRIM_DISPATCH_KEYS)
582  // See [Note: Trimmed Mobile Dispatch Keys]
583  constexpr uint16_t num_runtime_entries = 8;
584  #else
585  constexpr uint16_t num_runtime_entries = num_functionality_keys +
586      (numPerBackendFunctionalityKeys() * (num_backends - 1));
587  #endif

591      (static_cast<uint16_t>(1) << num_backends) - 1;

604  // These are some convenience identifiers for dispatch keys which are
605  // shorter to type than their long counterparts.  Note that some of these
606  // dispatch keys directly correspond to DeviceType; and most APIs that
607  // accept DispatchKey also accept DeviceType; e.g.,
608  // torch::dispatch(torch::kCPU, ...) is also valid.
609  constexpr DispatchKey kAutograd = DispatchKey::Autograd;
610  
611  // See Note [The Ordering of Per-Backend Dispatch Keys Matters!]
612  // This function relies on the invariant that the dispatch keys between
613  // StartOfDenseBackends and EndOfRuntimeBackendKeys are ordered by backend
614  // in the same order as `BackendComponent`.
615  constexpr BackendComponent toBackendComponent(DispatchKey k) {
616    if (k >= DispatchKey::StartOfDenseBackends &&
617        k <= DispatchKey::EndOfDenseBackends) {
618      return static_cast<BackendComponent>(
619          static_cast<uint8_t>(k) -
620          static_cast<uint8_t>(DispatchKey::StartOfDenseBackends));
621    } else if (
622        k >= DispatchKey::StartOfQuantizedBackends &&
623        k <= DispatchKey::EndOfQuantizedBackends) {
624      return static_cast<BackendComponent>(
625          static_cast<uint8_t>(k) -
626          static_cast<uint8_t>(DispatchKey::StartOfQuantizedBackends));
627    } else if (
628        k >= DispatchKey::StartOfSparseBackends &&
629        k <= DispatchKey::EndOfSparseBackends) {
630      return static_cast<BackendComponent>(
631          static_cast<uint8_t>(k) -
632          static_cast<uint8_t>(DispatchKey::StartOfSparseBackends));
633    } else if (
634        k >= DispatchKey::StartOfSparseCsrBackends &&
635        k <= DispatchKey::EndOfSparseCsrBackends) {
636      return static_cast<BackendComponent>(
637          static_cast<uint8_t>(k) -
638          static_cast<uint8_t>(DispatchKey::StartOfSparseCsrBackends));
639    } else if (
640        k >= DispatchKey::StartOfNestedTensorBackends &&
641        k <= DispatchKey::EndOfNestedTensorBackends) {
642      return static_cast<BackendComponent>(
643          static_cast<uint8_t>(k) -
644          static_cast<uint8_t>(DispatchKey::StartOfNestedTensorBackends));
645    } else if (
646        k >= DispatchKey::StartOfAutogradFunctionalityBackends &&
647        k <= DispatchKey::EndOfAutogradFunctionalityBackends) {
648      return static_cast<BackendComponent>(
649          static_cast<uint8_t>(k) -
650          static_cast<uint8_t>(
651              DispatchKey::StartOfAutogradFunctionalityBackends));
652    } else {
653      return BackendComponent::InvalidBit;
654    }
655  }
656  
657  constexpr DispatchKey toFunctionalityKey(DispatchKey k) {
658    if (k <= DispatchKey::EndOfFunctionalityKeys) {
659      return k;
660    } else if (k <= DispatchKey::EndOfDenseBackends) {
661      return DispatchKey::Dense;
662    } else if (k <= DispatchKey::EndOfQuantizedBackends) {
663      return DispatchKey::Quantized;
664    } else if (k <= DispatchKey::EndOfSparseBackends) {
665      return DispatchKey::Sparse;
666    } else if (k <= DispatchKey::EndOfSparseCsrBackends) {
667      return DispatchKey::SparseCsr;
668    } else if (k <= DispatchKey::EndOfNestedTensorBackends) {
669      return DispatchKey::NestedTensor;
670    } else if (k <= DispatchKey::EndOfAutogradFunctionalityBackends) {
671      return DispatchKey::AutogradFunctionality;
672    } else {
673      return DispatchKey::Undefined;
674    }
675  }

679  // Given (DispatchKey::Dense, BackendComponent::CUDABit), returns
680  // DispatchKey::CUDA.
681  // See Note [The Ordering of Per-Backend Dispatch Keys Matters!]
682  // This function relies on the invariant that the dispatch keys between
683  // StartOfDenseBackends and EndOfRuntimeBackendKeys are ordered by backend
684  // in the same order as `BackendComponent`.
685  constexpr DispatchKey toRuntimePerBackendFunctionalityKey(
686      DispatchKey functionality_k,
687      BackendComponent backend_k) {
688    if (functionality_k == DispatchKey::Dense) {
689      return static_cast<DispatchKey>(
690          static_cast<uint8_t>(DispatchKey::StartOfDenseBackends) +
691          static_cast<uint8_t>(backend_k));
692    }
693    if (functionality_k == DispatchKey::Sparse) {
694      return static_cast<DispatchKey>(
695          static_cast<uint8_t>(DispatchKey::StartOfSparseBackends) +
696          static_cast<uint8_t>(backend_k));
697    }
698    if (functionality_k == DispatchKey::SparseCsr) {
699      return static_cast<DispatchKey>(
700          static_cast<uint8_t>(DispatchKey::StartOfSparseCsrBackends) +
701          static_cast<uint8_t>(backend_k));
702    }
703    if (functionality_k == DispatchKey::Quantized) {
704      return static_cast<DispatchKey>(
705          static_cast<uint8_t>(DispatchKey::StartOfQuantizedBackends) +
706          static_cast<uint8_t>(backend_k));
707    }
708    if (functionality_k == DispatchKey::NestedTensor) {
709      return static_cast<DispatchKey>(
710          static_cast<uint8_t>(DispatchKey::StartOfNestedTensorBackends) +
711          static_cast<uint8_t>(backend_k));
712    }
713    if (functionality_k == DispatchKey::AutogradFunctionality) {
714      return static_cast<DispatchKey>(
715          static_cast<uint8_t>(
716              DispatchKey::StartOfAutogradFunctionalityBackends) +
717          static_cast<uint8_t>(backend_k));
718    }
719    return DispatchKey::Undefined;
720  }
