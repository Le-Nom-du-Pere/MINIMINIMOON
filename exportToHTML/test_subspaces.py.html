49   def test_nullspace_second():
50       # first test reduced row-ech form
51       R = Rational
52   
53       M = Matrix([[5, 7, 2,  1],
54                   [1, 6, 2, -1]])
55       out, tmp = M.rref()
56       assert out == Matrix([[1, 0, -R(2)/23, R(13)/23],
57                             [0, 1,  R(8)/23, R(-6)/23]])
58   
59       M = Matrix([[-5, -1,  4, -3, -1],
60                   [ 1, -1, -1,  1,  0],
61                   [-1,  0,  0,  0,  0],
62                   [ 4,  1, -4,  3,  1],
63                   [-2,  0,  2, -2, -1]])
64       assert M*M.nullspace()[0] == Matrix(5, 1, [0]*5)
65   
66       M = Matrix([[ 1,  3, 0,  2,  6, 3, 1],
67                   [-2, -6, 0, -2, -8, 3, 1],
68                   [ 3,  9, 0,  0,  6, 6, 2],
69                   [-1, -3, 0,  1,  0, 9, 3]])
70       out, tmp = M.rref()
71       assert out == Matrix([[1, 3, 0, 0, 2, 0, 0],
72                             [0, 0, 0, 1, 2, 0, 0],
73                             [0, 0, 0, 0, 0, 1, R(1)/3],
74                             [0, 0, 0, 0, 0, 0, 0]])
75   
76       # now check the vectors
77       basis = M.nullspace()
78       assert basis[0] == Matrix([-3, 1, 0, 0, 0, 0, 0])
79       assert basis[1] == Matrix([0, 0, 1, 0, 0, 0, 0])
80       assert basis[2] == Matrix([-2, 0, 0, -2, 1, 0, 0])
81       assert basis[3] == Matrix([0, 0, 0, 0, 0, R(-1)/3, 1])
82   
83       # issue 4797; just see that we can do it when rows > cols
84       M = Matrix([[1, 2], [2, 4], [3, 6]])
85       assert M.nullspace()

89       M = Matrix([[ 1,  2,  0,  2,  5],
90                   [-2, -5,  1, -1, -8],
91                   [ 0, -3,  3,  4,  1],
92                   [ 3,  6,  0, -7,  2]])
93   
94       # now check the vectors
95       basis = M.columnspace()
96       assert basis[0] == Matrix([1, -2, 0, 3])
97       assert basis[1] == Matrix([2, -5, -3, 6])
98       assert basis[2] == Matrix([2, -1, 4, -7])
99   
100      #check by columnspace definition
101      a, b, c, d, e = symbols('a b c d e')
102      X = Matrix([a, b, c, d, e])
103      for i in range(len(basis)):
104          eq=M*X-basis[i]
105          assert len(solve(eq, X)) != 0
106  
107      #check if rank-nullity theorem holds
108      assert M.rank() == len(basis)
