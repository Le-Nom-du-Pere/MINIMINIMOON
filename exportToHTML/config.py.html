287  # Specify how to optimize a compiled DDP module. The flag accepts a boolean
288  # value or a string. There are 3 modes.
289  # 1. "ddp_optimizer" (or True): with "ddp_optimizer", Dynamo will automatically
290  # split model graph into pieces to match DDP bucket sizes to allow DDP
291  # comm/compute overlap.
292  # 2. "python_reducer" (experimental): this optimization requires the usage
293  # of compiled_autograd. With "python_reducer", DDP will disable the C++ reducer
294  # and use the Python reducer to allow compiled_autograd to trace the
295  # communication and allow comm/compute overlap without graph-breaks.
296  # 3. "no_optimization" (or False): Dynamo won't split the model graph, nor
297  # will Python reducer be used. With this mode, there will be no graph-breaks
298  # and the original DDP C++ reducer will be used. There will no comm/compute
299  # overlap. This mode CANNOT be used with compiled_autograd.
300  # Note that to avoid breaking the existing usage, mode 1 and mode 4 can be
301  # specified with a boolean value. True is using ddp_optimizer and False is
302  # no optimization.
303  optimize_ddp: Union[
304      bool,
305      Literal[
306          "ddp_optimizer",
307          "python_reducer",
308          "python_reducer_without_compiled_forward",
309          "no_optimization",
310      ],
311  ] = True

570  # Enables a local, filesystem "profile" which can be used for automatic
571  # dynamic decisions, analogous to profile-guided optimization.  This config
572  # ONLY has an effect if torch.compiler.config.workflow_id is specified,
573  # which specifies the name of the profile we will save/load.
574  #
575  # The idea is that if we observe that a particular input is dynamic over
576  # multiple iterations on one run, we can save a profile with this information
577  # so the next time we run we can just make it dynamic the first time around,
578  # skipping an unnecessary static compilation.  The profile can be soundly
579  # stale, if it is wrong, it just means we may make more things dynamic than
580  # was actually necessary (NB: this /can/ cause a failure if making something
581  # dynamic causes the compiler to stop working because you tickled a latent
582  # bug.)
583  #
584  # The profile is ONLY guaranteed to work if the user source code is 100%
585  # unchanged.  Applying the profile if there are user code changes is only
586  # best effort otherwise.  In particular, we identify particular code objects
587  # by filename, line number and name of their function, so adding/removing newlines
588  # will typically cause cache misses.  We continuously update the profile,
589  # so if we only discover something is dynamic on the second run, we will update
590  # the profile for subsequent runs.
