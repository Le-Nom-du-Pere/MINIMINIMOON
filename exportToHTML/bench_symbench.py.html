69   def bench_R8():
70       "right(x^2,0,5,10^4)"
71       def right(f, a, b, n):
72           a = sympify(a)
73           b = sympify(b)
74           n = sympify(n)
75           x = f.atoms(Symbol).pop()
76           Deltax = (b - a)/n
77           c = a
78           est = 0
79           for i in range(n):
80               c += Deltax
81               est += f.subs(x, c)
82           return est*Deltax
83   
84       right(x**2, 0, 5, 10**4)
