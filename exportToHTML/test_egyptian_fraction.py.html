8    def test_egyptian_fraction():

37   def test_input():
38       r = (2,3), Rational(2, 3), (Rational(2), Rational(3))
39       for m in ["Greedy", "Graham Jewett", "Takenouchi", "Golomb"]:
40           for i in r:
41               d = egyptian_fraction(i, m)
42               assert all(i.is_Integer for i in d)
43               if m == "Graham Jewett":
44                   assert d == [3, 4, 12]

46                   assert d == [2, 6]
