28   def _generate_square_subsequent_mask(
29       sz: int,
30       device: Optional[torch.device] = None,
31       dtype: Optional[torch.dtype] = None,
32   ) -> Tensor:
33       r"""Generate a square causal mask for the sequence. 
34    
35       The masked positions are filled with float('-inf'). Unmasked positions are filled with float(0.0). 
36       """
37       return torch.triu(
38           torch.full((sz, sz), float("-inf"), dtype=dtype, device=device),
39           diagonal=1,
40       )

43   def _get_seq_len(src: Tensor, batch_first: bool) -> Optional[int]:
44       if src.is_nested:
45           return None
46       else:
47           src_size = src.size()
48           if len(src_size) == 2:
49               # unbatched: S, E
50               return src_size[0]
51           else:
52               # batched: B, S, E if batch_first else S, B, E
53               seq_len_pos = 1 if batch_first else 0
54               return src_size[seq_len_pos]

57   class Transformer(Module):
58       r"""A basic transformer layer. 
59    
60    
61       This Transformer layer implements the original Transformer architecture described 
62       in the `Attention Is All You Need <https://arxiv.org/abs/1706.03762>`_ paper. The 
63       intent of this layer is as a reference implementation for foundational understanding 
64       and thus it contains only limited features relative to newer Transformer architectures. 
65       Given the fast pace of innovation in transformer-like architectures, we recommend 
66       exploring this `tutorial <https://pytorch.org/tutorials/intermediate/transformer_building_blocks.html>`_ 
67       to build an efficient transformer layer from building blocks in core or using higher 
68       level libraries from the `PyTorch Ecosystem <https://landscape.pytorch.org/>`_. 
69    
70       Args: 
71           d_model: the number of expected features in the encoder/decoder inputs (default=512). 
72           nhead: the number of heads in the multiheadattention models (default=8). 
73           num_encoder_layers: the number of sub-encoder-layers in the encoder (default=6). 
74           num_decoder_layers: the number of sub-decoder-layers in the decoder (default=6). 
75           dim_feedforward: the dimension of the feedforward network model (default=2048). 
76           dropout: the dropout value (default=0.1). 
77           activation: the activation function of encoder/decoder intermediate layer, can be a string 
78               ("relu" or "gelu") or a unary callable. Default: relu 
79           custom_encoder: custom encoder (default=None). 
80           custom_decoder: custom decoder (default=None). 
81           layer_norm_eps: the eps value in layer normalization components (default=1e-5). 
82           batch_first: If ``True``, then the input and output tensors are provided 
83               as (batch, seq, feature). Default: ``False`` (seq, batch, feature). 
84           norm_first: if ``True``, encoder and decoder layers will perform LayerNorms before 
85               other attention and feedforward operations, otherwise after. Default: ``False`` (after). 
86           bias: If set to ``False``, ``Linear`` and ``LayerNorm`` layers will not learn an additive 
87               bias. Default: ``True``. 
88    
89       Examples: 
90           >>> transformer_model = nn.Transformer(nhead=16, num_encoder_layers=12) 
91           >>> src = torch.rand((10, 32, 512)) 
92           >>> tgt = torch.rand((20, 32, 512)) 
93           >>> out = transformer_model(src, tgt) 
94    
95       Note: A full example to apply nn.Transformer module for the word language model is available in 
96       https://github.com/pytorch/examples/tree/master/word_language_model 
97       """
98   
99       def __init__(
100          self,
101          d_model: int = 512,
102          nhead: int = 8,
103          num_encoder_layers: int = 6,
104          num_decoder_layers: int = 6,
105          dim_feedforward: int = 2048,
106          dropout: float = 0.1,
107          activation: Union[str, Callable[[Tensor], Tensor]] = F.relu,
108          custom_encoder: Optional[Any] = None,
109          custom_decoder: Optional[Any] = None,
110          layer_norm_eps: float = 1e-5,
111          batch_first: bool = False,
112          norm_first: bool = False,
113          bias: bool = True,
114          device=None,
115          dtype=None,
116      ) -> None:
117          factory_kwargs = {"device": device, "dtype": dtype}
118          super().__init__()
119          torch._C._log_api_usage_once(f"torch.nn.modules.{self.__class__.__name__}")
120  
121          if custom_encoder is not None:
122              self.encoder = custom_encoder
123          else:
124              encoder_layer = TransformerEncoderLayer(
125                  d_model,
126                  nhead,
127                  dim_feedforward,
128                  dropout,

135              )

137                  d_model, eps=layer_norm_eps, bias=bias, **factory_kwargs
138              )
139              self.encoder = TransformerEncoder(
140                  encoder_layer, num_encoder_layers, encoder_norm

143          if custom_decoder is not None:
144              self.decoder = custom_decoder
145          else:
146              decoder_layer = TransformerDecoderLayer(
147                  d_model,
148                  nhead,
149                  dim_feedforward,
150                  dropout,
151                  activation,
152                  layer_norm_eps,
153                  batch_first,
154                  norm_first,
155                  bias,
156                  **factory_kwargs,
157              )
158              decoder_norm = LayerNorm(
159                  d_model, eps=layer_norm_eps, bias=bias, **factory_kwargs
160              )
161              self.decoder = TransformerDecoder(
162                  decoder_layer, num_decoder_layers, decoder_norm
163              )
164  
165          self._reset_parameters()
166  
167          self.d_model = d_model
168          self.nhead = nhead
169  
170          self.batch_first = batch_first

172      def forward(
173          self,
174          src: Tensor,
175          tgt: Tensor,
176          src_mask: Optional[Tensor] = None,
177          tgt_mask: Optional[Tensor] = None,
178          memory_mask: Optional[Tensor] = None,
179          src_key_padding_mask: Optional[Tensor] = None,
180          tgt_key_padding_mask: Optional[Tensor] = None,
181          memory_key_padding_mask: Optional[Tensor] = None,
182          src_is_causal: Optional[bool] = None,
183          tgt_is_causal: Optional[bool] = None,
184          memory_is_causal: bool = False,
185      ) -> Tensor:
186          r"""Take in and process masked source/target sequences. 
187   
188          .. note:: 
189   
190              If a boolean tensor is provided for any of the [src/tgt/memory]_mask arguments, positions with a ``True`` value are 
191              not allowed to participate in the attention, 
192              which is the opposite of the definition for :attr:`attn_mask` 
193              in :func:`torch.nn.functional.scaled_dot_product_attention`. 
194   
195          Args: 
196              src: the sequence to the encoder (required). 
197              tgt: the sequence to the decoder (required). 
198              src_mask: the additive mask for the src sequence (optional). 
199              tgt_mask: the additive mask for the tgt sequence (optional). 
200              memory_mask: the additive mask for the encoder output (optional). 
201              src_key_padding_mask: the Tensor mask for src keys per batch (optional). 
202              tgt_key_padding_mask: the Tensor mask for tgt keys per batch (optional). 
203              memory_key_padding_mask: the Tensor mask for memory keys per batch (optional). 
204              src_is_causal: If specified, applies a causal mask as ``src_mask``. 
205                  Default: ``None``; try to detect a causal mask. 
206                  Warning: 
207                  ``src_is_causal`` provides a hint that ``src_mask`` is 
208                  the causal mask. Providing incorrect hints can result in 
209                  incorrect execution, including forward and backward 
210                  compatibility. 
211              tgt_is_causal: If specified, applies a causal mask as ``tgt_mask``. 
212                  Default: ``None``; try to detect a causal mask. 
213                  Warning: 
214                  ``tgt_is_causal`` provides a hint that ``tgt_mask`` is 
215                  the causal mask. Providing incorrect hints can result in 
216                  incorrect execution, including forward and backward 
217                  compatibility. 
218              memory_is_causal: If specified, applies a causal mask as 
219                  ``memory_mask``. 
220                  Default: ``False``. 
221                  Warning: 
222                  ``memory_is_causal`` provides a hint that 
223                  ``memory_mask`` is the causal mask. Providing incorrect 
224                  hints can result in incorrect execution, including 
225                  forward and backward compatibility. 
226   
227          Shape: 
228              - src: :math:`(S, E)` for unbatched input, :math:`(S, N, E)` if `batch_first=False` or 
229                `(N, S, E)` if `batch_first=True`. 
230              - tgt: :math:`(T, E)` for unbatched input, :math:`(T, N, E)` if `batch_first=False` or 
231                `(N, T, E)` if `batch_first=True`. 
232              - src_mask: :math:`(S, S)` or :math:`(N\cdot\text{num\_heads}, S, S)`. 
233              - tgt_mask: :math:`(T, T)` or :math:`(N\cdot\text{num\_heads}, T, T)`. 
234              - memory_mask: :math:`(T, S)`. 
235              - src_key_padding_mask: :math:`(S)` for unbatched input otherwise :math:`(N, S)`. 
236              - tgt_key_padding_mask: :math:`(T)` for unbatched input otherwise :math:`(N, T)`. 
237              - memory_key_padding_mask: :math:`(S)` for unbatched input otherwise :math:`(N, S)`. 
238   
239              Note: [src/tgt/memory]_mask ensures that position :math:`i` is allowed to attend the unmasked 
240              positions. If a BoolTensor is provided, positions with ``True`` 
241              are not allowed to attend while ``False`` values will be unchanged. If a FloatTensor 
242              is provided, it will be added to the attention weight. 
243              [src/tgt/memory]_key_padding_mask provides specified elements in the key to be ignored by 
244              the attention. If a BoolTensor is provided, the positions with the 
245              value of ``True`` will be ignored while the position with the value of ``False`` will be unchanged. 
246   
247              - output: :math:`(T, E)` for unbatched input, :math:`(T, N, E)` if `batch_first=False` or 
248                `(N, T, E)` if `batch_first=True`. 
249   
250              Note: Due to the multi-head attention architecture in the transformer model, 
251              the output sequence length of a transformer is same as the input sequence 
252              (i.e. target) length of the decoder. 
253   
254              where :math:`S` is the source sequence length, :math:`T` is the target sequence length, :math:`N` is the 
255              batch size, :math:`E` is the feature number 
256   
257          Examples: 
258              >>> # xdoctest: +SKIP 
259              >>> output = transformer_model( 
260              ...     src, tgt, src_mask=src_mask, tgt_mask=tgt_mask 
261              ... ) 
262          """
263          is_batched = src.dim() == 3
264          if not self.batch_first and src.size(1) != tgt.size(1) and is_batched:
265              raise RuntimeError("the batch number of src and tgt must be equal")
266          elif self.batch_first and src.size(0) != tgt.size(0) and is_batched:
267              raise RuntimeError("the batch number of src and tgt must be equal")
268  
269          if src.size(-1) != self.d_model or tgt.size(-1) != self.d_model:
270              raise RuntimeError(
271                  "the feature number of src and tgt must be equal to d_model"
272              )
273  
274          memory = self.encoder(
275              src,
276              mask=src_mask,
277              src_key_padding_mask=src_key_padding_mask,
278              is_causal=src_is_causal,
279          )
280          output = self.decoder(
281              tgt,
282              memory,
283              tgt_mask=tgt_mask,
284              memory_mask=memory_mask,
285              tgt_key_padding_mask=tgt_key_padding_mask,
286              memory_key_padding_mask=memory_key_padding_mask,
287              tgt_is_causal=tgt_is_causal,
288              memory_is_causal=memory_is_causal,
289          )
290          return output

311  class TransformerEncoder(Module):
312      r"""TransformerEncoder is a stack of N encoder layers. 
313   
314      This TransformerEncoder layer implements the original architecture described 
315      in the `Attention Is All You Need <https://arxiv.org/abs/1706.03762>`_ paper. The 
316      intent of this layer is as a reference implementation for foundational understanding 
317      and thus it contains only limited features relative to newer Transformer architectures. 
318      Given the fast pace of innovation in transformer-like architectures, we recommend 
319      exploring this `tutorial <https://pytorch.org/tutorials/intermediate/transformer_building_blocks.html>`_ 
320      to build efficient layers from building blocks in core or using higher 
321      level libraries from the `PyTorch Ecosystem <https://landscape.pytorch.org/>`_. 
322   
323      .. warning:: 
324          All layers in the TransformerEncoder are initialized with the same parameters. 
325          It is recommended to manually initialize the layers after creating the TransformerEncoder instance. 
326   
327      Args: 
328          encoder_layer: an instance of the TransformerEncoderLayer() class (required). 
329          num_layers: the number of sub-encoder-layers in the encoder (required). 
330          norm: the layer normalization component (optional). 
331          enable_nested_tensor: if True, input will automatically convert to nested tensor 
332              (and convert back on output). This will improve the overall performance of 
333              TransformerEncoder when padding rate is high. Default: ``True`` (enabled). 
334   
335      Examples: 
336          >>> encoder_layer = nn.TransformerEncoderLayer(d_model=512, nhead=8) 
337          >>> transformer_encoder = nn.TransformerEncoder(encoder_layer, num_layers=6) 
338          >>> src = torch.rand(10, 32, 512) 
339          >>> out = transformer_encoder(src) 
340      """
341  
342      __constants__ = ["norm"]
343  
344      def __init__(
345          self,
346          encoder_layer: "TransformerEncoderLayer",
347          num_layers: int,
348          norm: Optional[Module] = None,
349          enable_nested_tensor: bool = True,
350          mask_check: bool = True,
351      ) -> None:
352          super().__init__()
353          torch._C._log_api_usage_once(f"torch.nn.modules.{self.__class__.__name__}")
354          self.layers = _get_clones(encoder_layer, num_layers)
355          self.num_layers = num_layers
356          self.norm = norm
357          # this attribute saves the value providedat object construction
358          self.enable_nested_tensor = enable_nested_tensor
359          # this attribute controls whether nested tensors are used
360          self.use_nested_tensor = enable_nested_tensor
361          self.mask_check = mask_check
362  
363          enc_layer = "encoder_layer"
364          why_not_sparsity_fast_path = ""
365          if not isinstance(encoder_layer, torch.nn.TransformerEncoderLayer):
366              why_not_sparsity_fast_path = f"{enc_layer} was not TransformerEncoderLayer"
367          elif encoder_layer.norm_first:
368              why_not_sparsity_fast_path = f"{enc_layer}.norm_first was True"
369          elif not encoder_layer.self_attn.batch_first:
370              why_not_sparsity_fast_path = (
371                  f"{enc_layer}.self_attn.batch_first was not True"
372                  + "(use batch_first for better inference performance)"
373              )
374          elif not encoder_layer.self_attn._qkv_same_embed_dim:
375              why_not_sparsity_fast_path = (
376                  f"{enc_layer}.self_attn._qkv_same_embed_dim was not True"
377              )
378          elif encoder_layer.self_attn.in_proj_bias is None:
379              why_not_sparsity_fast_path = f"{enc_layer}.self_attn was passed bias=False"
380          elif not encoder_layer.activation_relu_or_gelu:
381              why_not_sparsity_fast_path = (
382                  f"{enc_layer}.activation_relu_or_gelu was not True"
383              )
384          elif not (encoder_layer.norm1.eps == encoder_layer.norm2.eps):
385              why_not_sparsity_fast_path = (
386                  f"{enc_layer}.norm1.eps was not equal to {enc_layer}.norm2.eps"
387              )
388          elif encoder_layer.self_attn.num_heads % 2 == 1:
389              why_not_sparsity_fast_path = f"{enc_layer}.self_attn.num_heads is odd"
390  
391          if enable_nested_tensor and why_not_sparsity_fast_path:
392              warnings.warn(
393                  f"enable_nested_tensor is True, but self.use_nested_tensor is False because {why_not_sparsity_fast_path}"
394              )
395              self.use_nested_tensor = False
396  
397      def forward(
398          self,
399          src: Tensor,
400          mask: Optional[Tensor] = None,
401          src_key_padding_mask: Optional[Tensor] = None,
402          is_causal: Optional[bool] = None,
403      ) -> Tensor:
404          r"""Pass the input through the encoder layers in turn. 
405   
406          Args: 
407              src: the sequence to the encoder (required). 
408              mask: the mask for the src sequence (optional). 
409              src_key_padding_mask: the mask for the src keys per batch (optional). 
410              is_causal: If specified, applies a causal mask as ``mask``. 
411                  Default: ``None``; try to detect a causal mask. 
412                  Warning: 
413                  ``is_causal`` provides a hint that ``mask`` is the 
414                  causal mask. Providing incorrect hints can result in 
415                  incorrect execution, including forward and backward 
416                  compatibility. 
417   
418          Shape: 
419              see the docs in :class:`~torch.nn.Transformer`. 
420          """
421          src_key_padding_mask = F._canonical_mask(
422              mask=src_key_padding_mask,
423              mask_name="src_key_padding_mask",
424              other_type=F._none_or_dtype(mask),
425              other_name="mask",
426              target_type=src.dtype,
427          )
428  
429          mask = F._canonical_mask(
430              mask=mask,
431              mask_name="mask",
432              other_type=None,
433              other_name="",
434              target_type=src.dtype,
435              check_other=False,
436          )
437  
438          output = src
439          convert_to_nested = False
440          first_layer = self.layers[0]
441          src_key_padding_mask_for_layers = src_key_padding_mask
442          why_not_sparsity_fast_path = ""
443          str_first_layer = "self.layers[0]"
444          batch_first = first_layer.self_attn.batch_first
445          is_fastpath_enabled = torch.backends.mha.get_fastpath_enabled()
446  
447          if not is_fastpath_enabled:
448              why_not_sparsity_fast_path = (
449                  "torch.backends.mha.get_fastpath_enabled() was not True"
450              )
451          elif not hasattr(self, "use_nested_tensor"):
452              why_not_sparsity_fast_path = "use_nested_tensor attribute not present"
453          elif not self.use_nested_tensor:
454              why_not_sparsity_fast_path = (
455                  "self.use_nested_tensor (set in init) was not True"
456              )
457          elif first_layer.training:
458              why_not_sparsity_fast_path = f"{str_first_layer} was in training mode"
459          elif not src.dim() == 3:
460              why_not_sparsity_fast_path = (
461                  f"input not batched; expected src.dim() of 3 but got {src.dim()}"
462              )
463          elif src_key_padding_mask is None:
464              why_not_sparsity_fast_path = "src_key_padding_mask was None"
465          elif (
466              (not hasattr(self, "mask_check")) or self.mask_check
467          ) and not torch._nested_tensor_from_mask_left_aligned(
468              src, src_key_padding_mask.logical_not()
469          ):
470              why_not_sparsity_fast_path = "mask_check enabled, and src and src_key_padding_mask was not left aligned"
471          elif output.is_nested:
472              why_not_sparsity_fast_path = "NestedTensor input is not supported"
473          elif mask is not None:
474              why_not_sparsity_fast_path = (
475                  "src_key_padding_mask and mask were both supplied"
476              )
477          elif torch.is_autocast_enabled():
478              why_not_sparsity_fast_path = "autocast is enabled"
479  
480          if not why_not_sparsity_fast_path:
481              tensor_args = (
482                  src,
483                  first_layer.self_attn.in_proj_weight,
484                  first_layer.self_attn.in_proj_bias,
485                  first_layer.self_attn.out_proj.weight,
486                  first_layer.self_attn.out_proj.bias,
487                  first_layer.norm1.weight,
488                  first_layer.norm1.bias,
489                  first_layer.norm2.weight,
490                  first_layer.norm2.bias,
491                  first_layer.linear1.weight,
492                  first_layer.linear1.bias,
493                  first_layer.linear2.weight,
494                  first_layer.linear2.bias,
495              )
496              _supported_device_type = [
497                  "cpu",
498                  "cuda",
499                  torch.utils.backend_registration._privateuse1_backend_name,
500              ]
501              if torch.overrides.has_torch_function(tensor_args):
502                  why_not_sparsity_fast_path = "some Tensor argument has_torch_function"
503              elif src.device.type not in _supported_device_type:
504                  why_not_sparsity_fast_path = (
505                      f"src device is neither one of {_supported_device_type}"
506                  )
507              elif torch.is_grad_enabled() and any(x.requires_grad for x in tensor_args):
508                  why_not_sparsity_fast_path = (
509                      "grad is enabled and at least one of query or the "
510                      "input/output projection weights or biases requires_grad"
511                  )
512  
513              if (not why_not_sparsity_fast_path) and (src_key_padding_mask is not None):
514                  convert_to_nested = True
515                  output = torch._nested_tensor_from_mask(
516                      output, src_key_padding_mask.logical_not(), mask_check=False
517                  )
518                  src_key_padding_mask_for_layers = None
519  
520          seq_len = _get_seq_len(src, batch_first)
521          is_causal = _detect_is_causal_mask(mask, is_causal, seq_len)
522  
523          for mod in self.layers:
524              output = mod(
525                  output,
526                  src_mask=mask,
527                  is_causal=is_causal,
528                  src_key_padding_mask=src_key_padding_mask_for_layers,
529              )
530  
531          if convert_to_nested:
532              output = output.to_padded_tensor(0.0, src.size())
533  
534          if self.norm is not None:
535              output = self.norm(output)
536  
537          return output

540  class TransformerDecoder(Module):
541      r"""TransformerDecoder is a stack of N decoder layers. 
542   
543      This TransformerDecoder layer implements the original architecture described 
544      in the `Attention Is All You Need <https://arxiv.org/abs/1706.03762>`_ paper. The 
545      intent of this layer is as a reference implementation for foundational understanding 
546      and thus it contains only limited features relative to newer Transformer architectures. 
547      Given the fast pace of innovation in transformer-like architectures, we recommend 
548      exploring this `tutorial <https://pytorch.org/tutorials/intermediate/transformer_building_blocks.html>`_ 
549      to build efficient layers from building blocks in core or using higher 
550      level libraries from the `PyTorch Ecosystem <https://landscape.pytorch.org/>`_. 
551   
552      .. warning:: 
553          All layers in the TransformerDecoder are initialized with the same parameters. 
554          It is recommended to manually initialize the layers after creating the TransformerDecoder instance. 
555   
556      Args: 
557          decoder_layer: an instance of the TransformerDecoderLayer() class (required). 
558          num_layers: the number of sub-decoder-layers in the decoder (required). 
559          norm: the layer normalization component (optional). 
560   
561      Examples: 
562          >>> decoder_layer = nn.TransformerDecoderLayer(d_model=512, nhead=8) 
563          >>> transformer_decoder = nn.TransformerDecoder(decoder_layer, num_layers=6) 
564          >>> memory = torch.rand(10, 32, 512) 
565          >>> tgt = torch.rand(20, 32, 512) 
566          >>> out = transformer_decoder(tgt, memory) 
567      """
568  
569      __constants__ = ["norm"]
570  
571      def __init__(
572          self,
573          decoder_layer: "TransformerDecoderLayer",
574          num_layers: int,
575          norm: Optional[Module] = None,
576      ) -> None:
577          super().__init__()
578          torch._C._log_api_usage_once(f"torch.nn.modules.{self.__class__.__name__}")
579          self.layers = _get_clones(decoder_layer, num_layers)
580          self.num_layers = num_layers
581          self.norm = norm

583      def forward(
584          self,
585          tgt: Tensor,
586          memory: Tensor,
587          tgt_mask: Optional[Tensor] = None,
588          memory_mask: Optional[Tensor] = None,
589          tgt_key_padding_mask: Optional[Tensor] = None,
590          memory_key_padding_mask: Optional[Tensor] = None,
591          tgt_is_causal: Optional[bool] = None,
592          memory_is_causal: bool = False,
593      ) -> Tensor:
594          r"""Pass the inputs (and mask) through the decoder layer in turn. 
595   
596          Args: 
597              tgt: the sequence to the decoder (required). 
598              memory: the sequence from the last layer of the encoder (required). 
599              tgt_mask: the mask for the tgt sequence (optional). 
600              memory_mask: the mask for the memory sequence (optional). 
601              tgt_key_padding_mask: the mask for the tgt keys per batch (optional). 
602              memory_key_padding_mask: the mask for the memory keys per batch (optional). 
603              tgt_is_causal: If specified, applies a causal mask as ``tgt mask``. 
604                  Default: ``None``; try to detect a causal mask. 
605                  Warning: 
606                  ``tgt_is_causal`` provides a hint that ``tgt_mask`` is 
607                  the causal mask. Providing incorrect hints can result in 
608                  incorrect execution, including forward and backward 
609                  compatibility. 
610              memory_is_causal: If specified, applies a causal mask as 
611                  ``memory mask``. 
612                  Default: ``False``. 
613                  Warning: 
614                  ``memory_is_causal`` provides a hint that 
615                  ``memory_mask`` is the causal mask. Providing incorrect 
616                  hints can result in incorrect execution, including 
617                  forward and backward compatibility. 
618   
619          Shape: 
620              see the docs in :class:`~torch.nn.Transformer`. 
621          """
622          output = tgt
623  
624          seq_len = _get_seq_len(tgt, self.layers[0].self_attn.batch_first)
625          tgt_is_causal = _detect_is_causal_mask(tgt_mask, tgt_is_causal, seq_len)
626  
627          for mod in self.layers:
628              output = mod(
629                  output,
630                  memory,
631                  tgt_mask=tgt_mask,
632                  memory_mask=memory_mask,
633                  tgt_key_padding_mask=tgt_key_padding_mask,
634                  memory_key_padding_mask=memory_key_padding_mask,
635                  tgt_is_causal=tgt_is_causal,
636                  memory_is_causal=memory_is_causal,
637              )
638  
639          if self.norm is not None:
640              output = self.norm(output)
641  
642          return output

645  class TransformerEncoderLayer(Module):
646      r"""TransformerEncoderLayer is made up of self-attn and feedforward network. 
647   
648      This TransformerEncoderLayer implements the original architecture described 
649      in the `Attention Is All You Need <https://arxiv.org/abs/1706.03762>`_ paper. The 
650      intent of this layer is as a reference implementation for foundational understanding 
651      and thus it contains only limited features relative to newer Transformer architectures. 
652      Given the fast pace of innovation in transformer-like architectures, we recommend 
653      exploring this `tutorial <https://pytorch.org/tutorials/intermediate/transformer_building_blocks.html>`_

655      level libraries from the `PyTorch Ecosystem <https://landscape.pytorch.org/>`_. 
656   
657      TransformerEncoderLayer can handle either traditional torch.tensor inputs, 
658      or Nested Tensor inputs.  Derived classes are expected to similarly accept 
659      both input formats.  (Not all combinations of inputs are currently 
660      supported by TransformerEncoderLayer while Nested Tensor is in prototype 
661      state.) 
662

664      the Module or TransformerEncoderLayer class.  If your custom layer 
665      supports both torch.Tensors and Nested Tensors inputs, make its 
666      implementation a derived class of TransformerEncoderLayer. If your custom 
667      Layer supports only torch.Tensor inputs, derive its implementation from 
668      Module.

673          dim_feedforward: the dimension of the feedforward network model (default=2048). 
674          dropout: the dropout value (default=0.1). 
675          activation: the activation function of the intermediate layer, can be a string 
676              ("relu" or "gelu") or a unary callable. Default: relu 
677          layer_norm_eps: the eps value in layer normalization components (default=1e-5). 
678          batch_first: If ``True``, then the input and output tensors are provided 
679              as (batch, seq, feature). Default: ``False`` (seq, batch, feature). 
680          norm_first: if ``True``, layer norm is done prior to attention and feedforward 
681              operations, respectively. Otherwise it's done after. Default: ``False`` (after). 
682          bias: If set to ``False``, ``Linear`` and ``LayerNorm`` layers will not learn an additive 
683              bias. Default: ``True``. 
684

686          >>> encoder_layer = nn.TransformerEncoderLayer(d_model=512, nhead=8) 
687          >>> src = torch.rand(10, 32, 512) 
688          >>> out = encoder_layer(src)

691          >>> encoder_layer = nn.TransformerEncoderLayer( 
692          ...     d_model=512, nhead=8, batch_first=True 
693          ... ) 
694          >>> src = torch.rand(32, 10, 512) 
695          >>> out = encoder_layer(src) 
696   
697      Fast path: 
698          forward() will use a special optimized implementation described in 
699          `FlashAttention: Fast and Memory-Efficient Exact Attention with IO-Awareness`_ if all of the following 
700          conditions are met: 
701   
702          - Either autograd is disabled (using ``torch.inference_mode`` or ``torch.no_grad``) or no tensor 
703            argument ``requires_grad`` 
704          - training is disabled (using ``.eval()``) 
705          - batch_first is ``True`` and the input is batched (i.e., ``src.dim() == 3``) 
706          - activation is one of: ``"relu"``, ``"gelu"``, ``torch.functional.relu``, or ``torch.functional.gelu`` 
707          - at most one of ``src_mask`` and ``src_key_padding_mask`` is passed 
708          - if src is a `NestedTensor <https://pytorch.org/docs/stable/nested.html>`_, neither ``src_mask`` 
709            nor ``src_key_padding_mask`` is passed 
710          - the two ``LayerNorm`` instances have a consistent ``eps`` value (this will naturally be the case 
711            unless the caller has manually modified one without modifying the other) 
712   
713          If the optimized implementation is in use, a 
714          `NestedTensor <https://pytorch.org/docs/stable/nested.html>`_ can be 
715          passed for ``src`` to represent padding more efficiently than using a padding 
716          mask. In this case, a `NestedTensor <https://pytorch.org/docs/stable/nested.html>`_ will be 
717          returned, and an additional speedup proportional to the fraction of the input that 
718          is padding can be expected. 
719   
720          .. _`FlashAttention: Fast and Memory-Efficient Exact Attention with IO-Awareness`: 
721           https://arxiv.org/abs/2205.14135 
722

781      def forward(
782          self,
783          src: Tensor,
784          src_mask: Optional[Tensor] = None,
785          src_key_padding_mask: Optional[Tensor] = None,
786          is_causal: bool = False,
787      ) -> Tensor:
788          r"""Pass the input through the encoder layer. 
789   
790          Args: 
791              src: the sequence to the encoder layer (required). 
792              src_mask: the mask for the src sequence (optional). 
793              src_key_padding_mask: the mask for the src keys per batch (optional). 
794              is_causal: If specified, applies a causal mask as ``src mask``. 
795                  Default: ``False``. 
796                  Warning: 
797                  ``is_causal`` provides a hint that ``src_mask`` is the 
798                  causal mask. Providing incorrect hints can result in 
799                  incorrect execution, including forward and backward 
800                  compatibility. 
801   
802          Shape: 
803              see the docs in :class:`~torch.nn.Transformer`. 
804          """
805          src_key_padding_mask = F._canonical_mask(
806              mask=src_key_padding_mask,
807              mask_name="src_key_padding_mask",
808              other_type=F._none_or_dtype(src_mask),
809              other_name="src_mask",
810              target_type=src.dtype,
811          )
812  
813          src_mask = F._canonical_mask(
814              mask=src_mask,
815              mask_name="src_mask",
816              other_type=None,
817              other_name="",
818              target_type=src.dtype,
819              check_other=False,
820          )
821  
822          is_fastpath_enabled = torch.backends.mha.get_fastpath_enabled()
823  
824          why_not_sparsity_fast_path = ""
825          if not is_fastpath_enabled:
826              why_not_sparsity_fast_path = (
827                  "torch.backends.mha.get_fastpath_enabled() was not True"
828              )
829          elif not src.dim() == 3:
830              why_not_sparsity_fast_path = (
831                  f"input not batched; expected src.dim() of 3 but got {src.dim()}"
832              )
833          elif self.training:
834              why_not_sparsity_fast_path = "training is enabled"
835          elif not self.self_attn.batch_first:
836              why_not_sparsity_fast_path = "self_attn.batch_first was not True"
837          elif self.self_attn.in_proj_bias is None:
838              why_not_sparsity_fast_path = "self_attn was passed bias=False"
839          elif not self.self_attn._qkv_same_embed_dim:
840              why_not_sparsity_fast_path = "self_attn._qkv_same_embed_dim was not True"
841          elif not self.activation_relu_or_gelu:
842              why_not_sparsity_fast_path = "activation_relu_or_gelu was not True"
843          elif not (self.norm1.eps == self.norm2.eps):
844              why_not_sparsity_fast_path = "norm1.eps is not equal to norm2.eps"
845          elif src.is_nested and (
846              src_key_padding_mask is not None or src_mask is not None
847          ):
848              why_not_sparsity_fast_path = "neither src_key_padding_mask nor src_mask are not supported with NestedTensor input"
849          elif self.self_attn.num_heads % 2 == 1:
850              why_not_sparsity_fast_path = "num_head is odd"
851          elif torch.is_autocast_enabled():
852              why_not_sparsity_fast_path = "autocast is enabled"
853          elif any(
854              len(getattr(m, "_forward_hooks", {}))
855              + len(getattr(m, "_forward_pre_hooks", {}))
856              for m in self.modules()
857          ):
858              why_not_sparsity_fast_path = "forward pre-/hooks are attached to the module"
859          if not why_not_sparsity_fast_path:
860              tensor_args = (
861                  src,
862                  self.self_attn.in_proj_weight,
863                  self.self_attn.in_proj_bias,
864                  self.self_attn.out_proj.weight,
865                  self.self_attn.out_proj.bias,
866                  self.norm1.weight,
867                  self.norm1.bias,
868                  self.norm2.weight,
869                  self.norm2.bias,
870                  self.linear1.weight,
871                  self.linear1.bias,
872                  self.linear2.weight,
873                  self.linear2.bias,
874              )
875  
876              # We have to use list comprehensions below because TorchScript does not support
877              # generator expressions.
878              _supported_device_type = [
879                  "cpu",
880                  "cuda",
881                  torch.utils.backend_registration._privateuse1_backend_name,
882              ]
883              if torch.overrides.has_torch_function(tensor_args):
884                  why_not_sparsity_fast_path = "some Tensor argument has_torch_function"
885              elif not all(
886                  (x.device.type in _supported_device_type) for x in tensor_args
887              ):
888                  why_not_sparsity_fast_path = (
889                      "some Tensor argument's device is neither one of "
890                      f"{_supported_device_type}"
891                  )
892              elif torch.is_grad_enabled() and any(x.requires_grad for x in tensor_args):
893                  why_not_sparsity_fast_path = (
894                      "grad is enabled and at least one of query or the "
895                      "input/output projection weights or biases requires_grad"
896                  )
897  
898              if not why_not_sparsity_fast_path:
899                  merged_mask, mask_type = self.self_attn.merge_masks(
900                      src_mask, src_key_padding_mask, src
901                  )
902                  return torch._transformer_encoder_layer_fwd(
903                      src,
904                      self.self_attn.embed_dim,
905                      self.self_attn.num_heads,
906                      self.self_attn.in_proj_weight,
907                      self.self_attn.in_proj_bias,
908                      self.self_attn.out_proj.weight,
909                      self.self_attn.out_proj.bias,
910                      self.activation_relu_or_gelu == 2,
911                      self.norm_first,
912                      self.norm1.eps,
913                      self.norm1.weight,
914                      self.norm1.bias,
915                      self.norm2.weight,
916                      self.norm2.bias,
917                      self.linear1.weight,
918                      self.linear1.bias,
919                      self.linear2.weight,
920                      self.linear2.bias,
921                      merged_mask,
922                      mask_type,
923                  )
924  
925          # see Fig. 1 of https://arxiv.org/pdf/2002.04745v1.pdf
926          x = src
927          if self.norm_first:
928              x = x + self._sa_block(
929                  self.norm1(x), src_mask, src_key_padding_mask, is_causal=is_causal
930              )
931              x = x + self._ff_block(self.norm2(x))
932          else:
933              x = self.norm1(
934                  x
935                  + self._sa_block(x, src_mask, src_key_padding_mask, is_causal=is_causal)
936              )
937              x = self.norm2(x + self._ff_block(x))
938  
939          return x
940  
941      # self-attention block
942      def _sa_block(
943          self,
944          x: Tensor,
945          attn_mask: Optional[Tensor],
946          key_padding_mask: Optional[Tensor],
947          is_causal: bool = False,
948      ) -> Tensor:
949          x = self.self_attn(
950              x,
951              x,
952              x,
953              attn_mask=attn_mask,
954              key_padding_mask=key_padding_mask,
955              need_weights=False,
956              is_causal=is_causal,
957          )[0]
958          return self.dropout1(x)
959  
960      # feed forward block
961      def _ff_block(self, x: Tensor) -> Tensor:
962          x = self.linear2(self.dropout(self.activation(self.linear1(x))))
963          return self.dropout2(x)

1194 def _detect_is_causal_mask(
1195     mask: Optional[Tensor],
1196     is_causal: Optional[bool] = None,
1197     size: Optional[int] = None,
1198 ) -> bool:
1199     """Return whether the given attention mask is causal. 
1200  
1201     Warning: 
1202     If ``is_causal`` is not ``None``, its value will be returned as is.  If a 
1203     user supplies an incorrect ``is_causal`` hint, 
1204  
1205     ``is_causal=False`` when the mask is in fact a causal attention.mask 
1206        may lead to reduced performance relative to what would be achievable 
1207        with ``is_causal=True``; 
1208     ``is_causal=True`` when the mask is in fact not a causal attention.mask 
1209        may lead to incorrect and unpredictable execution - in some scenarios, 
1210        a causal mask may be applied based on the hint, in other execution 
1211        scenarios the specified mask may be used.  The choice may not appear 
1212        to be deterministic, in that a number of factors like alignment, 
1213        hardware SKU, etc influence the decision whether to use a mask or 
1214        rely on the hint. 
1215     ``size`` if not None, check whether the mask is a causal mask of the provided size 
1216        Otherwise, checks for any causal mask. 
1217     """
1218     # Prevent type refinement
1219     make_causal = is_causal is True
1220 
1221     if is_causal is None and mask is not None:
1222         sz = size if size is not None else mask.size(-2)
1223         causal_comparison = _generate_square_subsequent_mask(
1224             sz, device=mask.device, dtype=mask.dtype
1225         )
1226 
1227         # Do not use `torch.equal` so we handle batched masks by
1228         # broadcasting the comparison.
1229         if mask.size() == causal_comparison.size():
1230             make_causal = bool((mask == causal_comparison).all())
1231         else:
1232             make_causal = False
1233 
1234     return make_causal
