8    _scales = [1e0, 1e3, 1e6, 1e9]
9    _units = ['s', 'ms', '\N{GREEK SMALL LETTER MU}s', 'ns']

12   def timed(func, setup="pass", limit=None):
13       """Adaptively measure execution time of a function. """
14       timer = timeit.Timer(func, setup=setup)
15       repeat, number = 3, 1
16   
17       for i in range(1, 10):
18           if timer.timeit(number) >= 0.2:
19               break
20           elif limit is not None and number >= limit:
21               break
22           else:
23               number *= 10
24   
25       time = min(timer.repeat(repeat, number)) / number
26   
27       if time > 0.0:
28           order = min(-int(math.floor(math.log10(time)) // 3), 3)
29       else:
30           order = 3
31   
32       return (number, time, time*_scales[order], _units[order])

37   def __do_timings():
38       import os
39       res = os.getenv('SYMPY_TIMINGS', '')
40       res = [x.strip() for x in res.split(',')]
41       return set(res)
42   
43   _do_timings = __do_timings()
44   _timestack = None
45   
46   
47   def _print_timestack(stack, level=1):
48       print('-'*level, '%.2f %s%s' % (stack[2], stack[0], stack[3]))
49       for s in stack[1]:
50           _print_timestack(s, level + 1)

53   def timethis(name):
54       def decorator(func):
55           if name not in _do_timings:
56               return func
57   
58           def wrapper(*args, **kwargs):
59               from time import time
60               global _timestack
61               oldtimestack = _timestack
62               _timestack = [func.func_name, [], 0, args]
63               t1 = time()
64               r = func(*args, **kwargs)
65               t2 = time()
66               _timestack[2] = t2 - t1
67               if oldtimestack is not None:
68                   oldtimestack[1].append(_timestack)
69                   _timestack = oldtimestack
70               else:
71                   _print_timestack(_timestack)
72                   _timestack = None
73               return r
74           return wrapper
75       return decorator
