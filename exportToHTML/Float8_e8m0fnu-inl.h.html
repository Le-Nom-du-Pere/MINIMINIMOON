23   inline C10_HOST_DEVICE Float8_e8m0fnu::operator float() const {
24     // TODO(#146647): maybe rewrite without control flow
25   
26     // if exponent is zero, need to special case to return 2^-127 instead of zero
27     if (x == 0) {
28       return c10::detail::fp32_from_bits(0x00400000);
29     }
30   
31     // if exponent is NaN, need to special case to return properly encoded NaN
32     if (isnan()) {
33       return c10::detail::fp32_from_bits(0x7f800001);
34     }
35   
36     // leave sign at 0, set the exponent bits, leave stored mantissa at 0
37     uint32_t res = x << 23;
38   
39     return c10::detail::fp32_from_bits(res);
40   }
