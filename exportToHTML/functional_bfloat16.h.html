95   // Note that we already have specialized member of Vectorized<scalar_t> for
96   // BFloat16 so the following functions would run smoothly:
97   //   using Vec = Vectorized<BFloat16>;
98   //   Vec one = Vec(BFloat16(1));
99   //   vec::map([](Vec x) { return one / (one + x.exp()); }, y_ptr, x_ptr, N);
100  //
101  // Then why we still need to specialize "functional"?
102  //   If we do specialization at Vectorized<> level, the above example would need
103  //   3 pairs of conversion of bf16->fp32/fp32->bf16, each for ".exp()", "+" and
104  //   "/". If we do specialization at vec::map<>() level, we have only 1 pair of
105  //   conversion of bf16->fp32/fp32->bf16, for the input and output BFloat16
106  //   vector only.
107  //
108  // The following BFloat16 functionality will only do data type conversion for
109  // input and output vector (reduce functionality will only convert the final
110  // scalar back to bf16). Compared to Vectorized<> specialization,
111  //   1. better performance since we have less data type conversion;
112  //   2. less rounding error since immediate results are kept in fp32;
113  //   3. accumulation done on data type of fp32.
114  //
115  //  If you plan to extend this file, please ensure adding unit tests at
116  //    aten/src/ATen/test/vec_test_all_types.cpp
117  //
118  template <
119      typename scalar_t,
120      typename Op,
121      typename std::enable_if_t<is_reduced_floating_point_v<scalar_t>, int> = 0>
122  inline float reduce_all(const Op& vec_fun, const scalar_t* data, int64_t size) {
123    using bVec = vec::Vectorized<scalar_t>;
124    using fVec = vec::Vectorized<float>;
125    if (size < bVec::size()) {
126      bVec data_bvec = bVec::loadu(data, size);
127      auto [data_fvec0, data_fvec1] = convert_to_float<scalar_t>(data_bvec);
128      if (size > fVec::size()) {
129        data_fvec0 = fVec::set(
130            data_fvec0, vec_fun(data_fvec0, data_fvec1), size - fVec::size());
131        return vec_reduce_all<float>(vec_fun, data_fvec0, fVec::size());
132      } else {
133        return vec_reduce_all<float>(vec_fun, data_fvec0, size);
134      }
135    }
136    int64_t d = bVec::size();
137    bVec acc_bvec = bVec::loadu(data);
138    auto [acc_fvec0, acc_fvec1] = convert_to_float<scalar_t>(acc_bvec);
139    for (; d < size - (size % bVec::size()); d += bVec::size()) {
140      bVec data_bvec = bVec::loadu(data + d);
141      auto [data_fvec0, data_fvec1] = convert_to_float<scalar_t>(data_bvec);
142      acc_fvec0 = vec_fun(acc_fvec0, data_fvec0);
143      acc_fvec1 = vec_fun(acc_fvec1, data_fvec1);
144    }
145    if (size - d > 0) {
146      bVec data_bvec = bVec::loadu(data + d, size - d);
147      auto [data_fvec0, data_fvec1] = convert_to_float<scalar_t>(data_bvec);
148      if (size - d > fVec::size()) {
149        acc_fvec0 = vec_fun(acc_fvec0, data_fvec0);
150        acc_fvec1 = fVec::set(
151            acc_fvec1, vec_fun(acc_fvec1, data_fvec1), size - d - fVec::size());
152      } else {
153        acc_fvec0 =
154            fVec::set(acc_fvec0, vec_fun(acc_fvec0, data_fvec0), size - d);
155      }
156    }
157    acc_fvec0 = vec_fun(acc_fvec0, acc_fvec1);
158    return vec_reduce_all<float>(vec_fun, acc_fvec0);
159  }

161  template <
162      typename scalar_t,
163      typename Op1,
164      typename Op2,
165      typename std::enable_if_t<is_reduced_floating_point_v<scalar_t>, int> = 0>
166  inline std::pair<float, float> reduce2_all(
167      const Op1& vec_fun1,
168      const Op2& vec_fun2,
169      const scalar_t* data,
170      int64_t size) {
171    using bVec = vec::Vectorized<scalar_t>;
172    using fVec = vec::Vectorized<float>;
173    if (size < bVec::size()) {
174      bVec data_bvec = bVec::loadu(data, size);
175      auto [data_fvec0, data_fvec1] = convert_to_float<scalar_t>(data_bvec);
176      if (size > fVec::size()) {
177        fVec acc1_fvec = fVec::set(
178            data_fvec0, vec_fun1(data_fvec0, data_fvec1), size - fVec::size());
179        fVec acc2_fvec = fVec::set(
180            data_fvec0, vec_fun2(data_fvec0, data_fvec1), size - fVec::size());
181        return std::pair<scalar_t, scalar_t>(
182            vec_reduce_all<float>(vec_fun1, acc1_fvec, fVec::size()),
183            vec_reduce_all<float>(vec_fun2, acc2_fvec, fVec::size()));
184      } else {
185        return std::pair<scalar_t, scalar_t>(
186            vec_reduce_all<float>(vec_fun1, data_fvec0, size),
187            vec_reduce_all<float>(vec_fun2, data_fvec0, size));
188      }
189    }
190    int64_t d = bVec::size();
191    bVec acc_bvec = bVec::loadu(data);
192    auto [acc1_fvec0, acc1_fvec1] = convert_to_float<scalar_t>(acc_bvec);
193    auto [acc2_fvec0, acc2_fvec1] = convert_to_float<scalar_t>(acc_bvec);
194    for (; d < size - (size % bVec::size()); d += bVec::size()) {
195      bVec data_bvec = bVec::loadu(data + d);
196      auto [data_fvec0, data_fvec1] = convert_to_float<scalar_t>(data_bvec);
197      acc1_fvec0 = vec_fun1(acc1_fvec0, data_fvec0);
198      acc1_fvec1 = vec_fun1(acc1_fvec1, data_fvec1);
199      acc2_fvec0 = vec_fun2(acc2_fvec0, data_fvec0);
200      acc2_fvec1 = vec_fun2(acc2_fvec1, data_fvec1);
201    }
202    if (size - d > 0) {
203      bVec data_bvec = bVec::loadu(data + d, size - d);
204      auto [data_fvec0, data_fvec1] = convert_to_float<scalar_t>(data_bvec);
205      if (size - d > fVec::size()) {
206        acc1_fvec0 = vec_fun1(acc1_fvec0, data_fvec0);
207        acc1_fvec1 = fVec::set(
208            acc1_fvec1,
209            vec_fun1(acc1_fvec1, data_fvec1),
210            size - d - fVec::size());
211        acc2_fvec0 = vec_fun2(acc2_fvec0, data_fvec0);
212        acc2_fvec1 = fVec::set(
213            acc2_fvec1,
214            vec_fun2(acc2_fvec1, data_fvec1),
215            size - d - fVec::size());
216      } else {
217        acc1_fvec0 =
218            fVec::set(acc1_fvec0, vec_fun1(acc1_fvec0, data_fvec0), size - d);
219        acc2_fvec0 =
220            fVec::set(acc2_fvec0, vec_fun2(acc2_fvec0, data_fvec0), size - d);
221      }
222    }
223    acc1_fvec0 = vec_fun1(acc1_fvec0, acc1_fvec1);
224    acc2_fvec0 = vec_fun2(acc2_fvec0, acc2_fvec1);
225    return std::pair<scalar_t, scalar_t>(
226        vec_reduce_all<float>(vec_fun1, acc1_fvec0),
227        vec_reduce_all<float>(vec_fun2, acc2_fvec0));
228  }

233      typename ReduceOp,
234      typename std::enable_if_t<is_reduced_floating_point_v<scalar_t>, int> = 0>
235  inline float map_reduce_all(
236      const MapOp& map_fun,
237      const ReduceOp& red_fun,
238      const scalar_t* data,
239      int64_t size) {

245      if (size > fVec::size()) {
246        data_fvec0 = map_fun(data_fvec0);
247        data_fvec1 = map_fun(data_fvec1);
248        data_fvec0 = fVec::set(
249            data_fvec0, red_fun(data_fvec0, data_fvec1), size - fVec::size());
250        return vec_reduce_all<float>(red_fun, data_fvec0, fVec::size());
251      } else {
252        data_fvec0 = map_fun(data_fvec0);
253        return vec_reduce_all<float>(red_fun, data_fvec0, size);
254      }
255    }
256    int64_t d = bVec::size();
257    bVec acc_bvec = bVec::loadu(data);
258    auto [acc_fvec0, acc_fvec1] = convert_to_float<scalar_t>(acc_bvec);
259    acc_fvec0 = map_fun(acc_fvec0);
260    acc_fvec1 = map_fun(acc_fvec1);
261    for (; d < size - (size % bVec::size()); d += bVec::size()) {
262      bVec data_bvec = bVec::loadu(data + d);
263      auto [data_fvec0, data_fvec1] = convert_to_float<scalar_t>(data_bvec);
264      data_fvec0 = map_fun(data_fvec0);

266      acc_fvec0 = red_fun(acc_fvec0, data_fvec0);

268    }
269    if (size - d > 0) {
270      bVec data_bvec = bVec::loadu(data + d, size - d);
271      auto [data_fvec0, data_fvec1] = convert_to_float<scalar_t>(data_bvec);
272      if (size - d > fVec::size()) {
273        data_fvec0 = map_fun(data_fvec0);
274        data_fvec1 = map_fun(data_fvec1);
275        acc_fvec0 = red_fun(acc_fvec0, data_fvec0);
276        acc_fvec1 = fVec::set(
277            acc_fvec1, red_fun(acc_fvec1, data_fvec1), size - d - fVec::size());

431  template <
432      typename scalar_t,
433      typename Op,
434      typename std::enable_if_t<
435          !(!detail::should_prefer_converting_through_float_v<scalar_t> &&
436            std::is_invocable_v<Op, vec::Vectorized<scalar_t>>),
437          int> = 0>
438  inline void map(
439      const Op& vec_fun,
440      scalar_t* output_data,
441      const scalar_t* input_data,
442      int64_t size) {
443    using bVec = vec::Vectorized<scalar_t>;
444    using fVec = vec::Vectorized<float>;
445    int64_t d = 0;
446    for (; d < size - (size % bVec::size()); d += bVec::size()) {
447      bVec data_bvec = bVec::loadu(input_data + d);
448      auto [data_fvec0, data_fvec1] = convert_to_float<scalar_t>(data_bvec);
449      fVec output_fvec0 = vec_fun(data_fvec0);
450      fVec output_fvec1 = vec_fun(data_fvec1);
451      bVec output_bvec = convert_from_float<scalar_t>(output_fvec0, output_fvec1);
452      output_bvec.store(output_data + d);
453    }
454    if (size - d > 0) {
455      bVec data_bvec = bVec::loadu(input_data + d, size - d);
456      auto [data_fvec0, data_fvec1] = convert_to_float<scalar_t>(data_bvec);
457      fVec output_fvec0 = vec_fun(data_fvec0);
458      fVec output_fvec1 = vec_fun(data_fvec1);
459      bVec output_bvec = convert_from_float<scalar_t>(output_fvec0, output_fvec1);
460      output_bvec.store(output_data + d, size - d);
461    }

464  template <
465      typename scalar_t,
466      typename Op,
467      typename std::enable_if_t<is_reduced_floating_point_v<scalar_t>, int> = 0>
468  inline void map(
469      const Op& vec_fun,
470      scalar_t* output_data,
471      const float* input_data,
472      int64_t size) {
473    using bVec = vec::Vectorized<scalar_t>;
474    using fVec = vec::Vectorized<float>;
475    int64_t d = 0;
476    for (; d < size - (size % bVec::size()); d += bVec::size()) {
477      fVec data_fvec0 = fVec::loadu(input_data + d);
478      fVec data_fvec1 = fVec::loadu(input_data + d + fVec::size());
479      fVec output_fvec0 = vec_fun(data_fvec0);
480      fVec output_fvec1 = vec_fun(data_fvec1);
481      bVec output_bvec = convert_from_float<scalar_t>(output_fvec0, output_fvec1);
482      output_bvec.store(output_data + d);
483    }
484    if (size - d > 0) {
485      fVec data_fvec0, data_fvec1;
486      if (size - d > fVec::size()) {
487        data_fvec0 = fVec::loadu(input_data + d);
488        data_fvec1 =
489            fVec::loadu(input_data + d + fVec::size(), size - d - fVec::size());
490      } else {
491        // choose to align with behaviour of bVec::loadu(ptr, size),
492        // which leaves data_fvec1 uninitialized
493        data_fvec0 = fVec::loadu(input_data + d, size - d);
494      }
495      fVec output_fvec0 = vec_fun(data_fvec0);
496      fVec output_fvec1 = vec_fun(data_fvec1);
497      bVec output_bvec = convert_from_float<scalar_t>(output_fvec0, output_fvec1);
498      output_bvec.store(output_data + d, size - d);
499    }
500  }

533      auto [data_fvec0, data_fvec1] = convert_to_float<scalar_t>(data_bvec);
