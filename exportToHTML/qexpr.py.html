29   def _qsympify_sequence(seq):
30       """Convert elements of a sequence to standard form. 
31    
32       This is like sympify, but it performs special logic for arguments passed 
33       to QExpr. The following conversions are done: 
34    
35       * (list, tuple, Tuple) => _qsympify_sequence each element and convert 
36         sequence to a Tuple. 
37       * basestring => Symbol 
38       * Matrix => Matrix 
39       * other => sympify 
40    
41       Strings are passed to Symbol, not sympify to make sure that variables like 
42       'pi' are kept as Symbols, not the SymPy built-in number subclasses. 
43    
44       Examples 
45       ======== 
46    
47       >>> from sympy.physics.quantum.qexpr import _qsympify_sequence 
48       >>> _qsympify_sequence((1,2,[3,4,[1,]])) 
49       (1, 2, (3, 4, (1,))) 
50    
51       """
52   
53       return tuple(__qsympify_sequence_helper(seq))
54   
55   
56   def __qsympify_sequence_helper(seq):
57       """ 
58          Helper function for _qsympify_sequence 
59          This function does the actual work. 
60       """
61       #base case. If not a list, do Sympification
62       if not is_sequence(seq):
63           if isinstance(seq, Matrix):
64               return seq
65           elif isinstance(seq, str):
66               return Symbol(seq)
67           else:
68               return sympify(seq)
69   
70       # base condition, when seq is QExpr and also
71       # is iterable.
72       if isinstance(seq, QExpr):
73           return seq
74   
75       #if list, recurse on each item in the list
76       result = [__qsympify_sequence_helper(item) for item in seq]
77   
78       return Tuple(*result)

85   class QExpr(Expr):
86       """A base class for all quantum object like operators and states."""
87   
88       # In sympy, slots are for instance attributes that are computed
89       # dynamically by the __new__ method. They are not part of args, but they
90       # derive from args.
91   
92       # The Hilbert space a quantum Object belongs to.
93       __slots__ = ('hilbert_space', )
94   
95       is_commutative = False
96   
97       # The separator used in printing the label.
98       _label_separator = ''

100      def __new__(cls, *args, **kwargs):
101          """Construct a new quantum object. 
102   
103          Parameters 
104          ========== 
105   
106          args : tuple 
107              The list of numbers or parameters that uniquely specify the 
108              quantum object. For a state, this will be its symbol or its 
109              set of quantum numbers. 
110   
111          Examples 
112          ======== 
113   
114          >>> from sympy.physics.quantum.qexpr import QExpr 
115          >>> q = QExpr(0) 
116          >>> q 
117          0 
118          >>> q.label 
119          (0,) 
120          >>> q.hilbert_space 
121          H 
122          >>> q.args 
123          (0,) 
124          >>> q.is_commutative 
125          False 
126          """
127  
128          # First compute args and call Expr.__new__ to create the instance
129          args = cls._eval_args(args, **kwargs)
130          if len(args) == 0:
131              args = cls._eval_args(tuple(cls.default_args()), **kwargs)
132          inst = Expr.__new__(cls, *args)
133          # Now set the slots on the instance
134          inst.hilbert_space = cls._eval_hilbert_space(args)
135          return inst
136  
137      @classmethod
138      def _new_rawargs(cls, hilbert_space, *args, **old_assumptions):
139          """Create new instance of this class with hilbert_space and args. 
140   
141          This is used to bypass the more complex logic in the ``__new__`` 
142          method in cases where you already have the exact ``hilbert_space`` 
143          and ``args``. This should be used when you are positive these 
144          arguments are valid, in their final, proper form and want to optimize 
145          the creation of the object. 
146          """
147  
148          obj = Expr.__new__(cls, *args, **old_assumptions)
149          obj.hilbert_space = hilbert_space
150          return obj

156      @property
157      def label(self):
158          """The label is the unique set of identifiers for the object. 
159   
160          Usually, this will include all of the information about the state 
161          *except* the time (in the case of time-dependent objects). 
162   
163          This must be a tuple, rather than a Tuple. 
164          """
165          if len(self.args) == 0:  # If there is no label specified, return the default
166              return self._eval_args(list(self.default_args()))
167          else:
168              return self.args

172          return True

188      def _eval_adjoint(self):
189          obj = Expr._eval_adjoint(self)
190          if obj is None:
191              obj = Expr.__new__(Dagger, self)
192          if isinstance(obj, QExpr):
193              obj.hilbert_space = self.hilbert_space
194          return obj

196      @classmethod
197      def _eval_args(cls, args):
198          """Process the args passed to the __new__ method. 
199   
200          This simply runs args through _qsympify_sequence. 
201          """
202          return _qsympify_sequence(args)

204      @classmethod
205      def _eval_hilbert_space(cls, args):
206          """Compute the Hilbert space instance from the args. 
207          """
208          from sympy.physics.quantum.hilbert import HilbertSpace
209          return HilbertSpace()

217      def _print_sequence(self, seq, sep, printer, *args):
218          result = []
219          for item in seq:
220              result.append(printer._print(item, *args))
221          return sep.join(result)

223      def _print_sequence_pretty(self, seq, sep, printer, *args):
224          pform = printer._print(seq[0], *args)
225          for item in seq[1:]:
226              pform = prettyForm(*pform.right(sep))
227              pform = prettyForm(*pform.right(printer._print(item, *args)))
228          return pform

232      def _print_subscript_pretty(self, a, b):
233          top = prettyForm(*b.left(' '*a.width()))
234          bot = prettyForm(*a.right(' '*b.width()))
235          return prettyForm(binding=prettyForm.POW, *bot.below(top))

292          """Default printing behavior of QExpr objects

320          raise NotImplementedError('This object does not have a default basis')

322      def _represent(self, *, basis=None, **options):
323          """Represent this object in a given basis. 
324   
325          This method dispatches to the actual methods that perform the 
326          representation. Subclases of QExpr should define various methods to 
327          determine how the object will be represented in various bases. The 
328          format of these methods is:: 
329   
330              def _represent_BasisName(self, basis, **options): 
331   
332          Thus to define how a quantum object is represented in the basis of 
333          the operator Position, you would define:: 
334   
335              def _represent_Position(self, basis, **options): 
336   
337          Usually, basis object will be instances of Operator subclasses, but 
338          there is a chance we will relax this in the future to accommodate other 
339          types of basis sets that are not associated with an operator. 
340   
341          If the ``format`` option is given it can be ("sympy", "numpy", 
342          "scipy.sparse"). This will ensure that any matrices that result from 
343          representing the object are returned in the appropriate matrix format. 
344   
345          Parameters 
346          ========== 
347   
348          basis : Operator 
349              The Operator whose basis functions will be used as the basis for 
350              representation. 
351          options : dict 
352              A dictionary of key/value pairs that give options and hints for 
353              the representation, such as the number of basis functions to 
354              be used. 
355          """
356          if basis is None:
357              result = self._represent_default_basis(**options)
358          else:
359              result = dispatch_method(self, '_represent', basis, **options)
360  
361          # If we get a matrix representation, convert it to the right format.
362          format = options.get('format', 'sympy')
363          result = self._format_represent(result, format)
364          return result
365  
366      def _format_represent(self, result, format):
367          if format == 'sympy' and not isinstance(result, Matrix):
368              return to_sympy(result)
369          elif format == 'numpy' and not isinstance(result, numpy_ndarray):
370              return to_numpy(result)
371          elif format == 'scipy.sparse' and \
372                  not isinstance(result, scipy_sparse_matrix):
373              return to_scipy_sparse(result)
374  
375          return result

378  def split_commutative_parts(e):
379      """Split into commutative and non-commutative parts."""
380      c_part, nc_part = e.args_cnc()
381      c_part = list(c_part)
382      return c_part, nc_part

385  def split_qexpr_parts(e):
386      """Split an expression into Expr and noncommutative QExpr parts."""
387      expr_part = []
388      qexpr_part = []
389      for arg in e.args:
390          if not isinstance(arg, QExpr):
391              expr_part.append(arg)
392          else:
393              qexpr_part.append(arg)
394      return expr_part, qexpr_part

397  def dispatch_method(self, basename, arg, **options):
398      """Dispatch a method to the proper handlers."""
399      method_name = '%s_%s' % (basename, arg.__class__.__name__)
400      if hasattr(self, method_name):
401          f = getattr(self, method_name)
402          # This can raise and we will allow it to propagate.
403          result = f(arg, **options)
404          if result is not None:
405              return result
406      raise NotImplementedError(
407          "%s.%s cannot handle: %r" %
408          (self.__class__.__name__, basename, arg)
409      )
