51   def bench_discrete_log(data_set, algo=None):
52       if algo is None:
53           f = discrete_log
54       elif algo == 'trial':
55           f = _discrete_log_trial_mul
56       elif algo == 'shanks':
57           f = _discrete_log_shanks_steps
58       elif algo == 'rho':
59           f = _discrete_log_pollard_rho
60       elif algo == 'ph':
61           f = _discrete_log_pohlig_hellman
62       else:
63           raise ValueError("Argument 'algo' should be one"
64                   " of ('trial', 'shanks', 'rho' or 'ph')")
65   
66       for i, data in enumerate(data_set):
67           for j, (n, p, g) in enumerate(data):
68               t = time()
69               l = f(n, pow(g, p - 1, n), g, p)
70               t = time() - t
71               print('[%02d-%03d] %15.10f' % (i, j, t))
72               assert l == p - 1
