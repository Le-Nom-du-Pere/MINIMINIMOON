3    """ 
4    Demo simplificado para mostrar el funcionamiento del manejo de señales 
5    sin dependencias pesadas como spaCy, sentence-transformers, etc.

8    import json
9    import signal
10   import sys
11   import threading
12   import time
13   from datetime import datetime
14   from pathlib import Path
15   import atexit
16   
17   # ==================== SISTEMA DE MONITOREO SIMPLIFICADO ====================
18   class SistemaMonitoreoDemo:
19       """Versión demo del sistema de monitoreo para testing"""
20       
21       def __init__(self):
22           self.ejecuciones = []
23           self.tiempo_inicio = None
24           self.trabajadores_activos = set()
25           self.lock = threading.RLock()
26           self.interrumpido = False
27   
28       def iniciar_monitoreo(self):
29           self.tiempo_inicio = datetime.now()
30           print(f"🚀 Sistema de monitoreo iniciado: {self.tiempo_inicio}")
31   
32       def registrar_trabajador(self, trabajador_id: str):
33           with self.lock:
34               self.trabajadores_activos.add(trabajador_id)
35               print(f"➕ Trabajador registrado: {trabajador_id}")
36   
37       def desregistrar_trabajador(self, trabajador_id: str):
38           with self.lock:
39               self.trabajadores_activos.discard(trabajador_id)
40               print(f"➖ Trabajador desregistrado: {trabajador_id}")
41   
42       def terminar_trabajadores(self):
43           with self.lock:
44               self.interrumpido = True
45               trabajadores_copia = self.trabajadores_activos.copy()
46           print(f"🛑 Terminando {len(trabajadores_copia)} trabajadores...")
47   
48       def registrar_ejecucion(self, nombre: str, resultado: dict):
49           with self.lock:
50               ejecucion = {
51                   "nombre": nombre,
52                   "resultado": resultado,
53                   "timestamp": datetime.now().isoformat()
54               }
55               self.ejecuciones.append(ejecucion)
56               print(f"📝 Ejecución registrada: {nombre} - {resultado.get('status', 'unknown')}")
57   
58       def generar_dump_emergencia(self, output_dir: Path) -> Path:
59           with self.lock:
60               timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
61               dump_path = output_dir / f"dump_emergencia_demo_{timestamp}.json"
62               
63               estado = {
64                   "timestamp_dump": datetime.now().isoformat(),
65                   "sistema_interrumpido": self.interrumpido,
66                   "trabajadores_activos": list(self.trabajadores_activos),
67                   "tiempo_inicio": self.tiempo_inicio.isoformat() if self.tiempo_inicio else None,
68                   "total_ejecuciones": len(self.ejecuciones),
69                   "ejecuciones": self.ejecuciones
70               }
71               
72               with open(dump_path, 'w', encoding='utf-8') as f:
73                   json.dump(estado, f, indent=2, ensure_ascii=False)
74               
75               print(f"💾 Dump de emergencia guardado: {dump_path}")
76               return dump_path
77   
78   # ==================== MANEJO DE SEÑALES ====================
79   _sistema_monitoreo_global = None
80   _output_dir_global = None
81   _signal_handler_lock = threading.RLock()
82   
83   def signal_handler(signum, frame):
84       """Manejador de señales demo"""
85       global _sistema_monitoreo_global, _output_dir_global
86       
87       with _signal_handler_lock:
88           print(f"\n🚨 SEÑAL {signum} RECIBIDA - Iniciando terminación graciosa...")
89           
90           if _sistema_monitoreo_global:
91               try:
92                   _sistema_monitoreo_global.terminar_trabajadores()
93                   
94                   if _output_dir_global:
95                       dump_path = _sistema_monitoreo_global.generar_dump_emergencia(_output_dir_global)
96                       print(f"📊 Estado guardado en: {dump_path}")
97                       
98               except Exception as e:
99                   print(f"❌ Error durante terminación: {e}")
100          
101          print("🛑 Terminación graciosa completada")
102          sys.exit(1)
103  
104  def atexit_handler():
105      """Handler para terminación inesperada"""
106      global _sistema_monitoreo_global, _output_dir_global
107      
108      with _signal_handler_lock:
109          print("\n🚨 Terminación inesperada detectada...")
110          
111          if _sistema_monitoreo_global and _output_dir_global:
112              try:
113                  dump_path = _sistema_monitoreo_global.generar_dump_emergencia(_output_dir_global)
114                  print(f"💾 Estado de emergencia guardado: {dump_path}")
115              except Exception as e:
116                  print(f"❌ Error en atexit handler: {e}")
117  
118  def simular_trabajo(trabajador_id: str, duracion: int):
119      """Simula trabajo de procesamiento"""
120      global _sistema_monitoreo_global
121      
122      if _sistema_monitoreo_global:
123          _sistema_monitoreo_global.registrar_trabajador(trabajador_id)
124      
125      try:
126          print(f"🔄 {trabajador_id} iniciando trabajo por {duracion}s...")
127          
128          for i in range(duracion):
129              time.sleep(1)
130              if _sistema_monitoreo_global and _sistema_monitoreo_global.interrumpido:
131                  print(f"⚠️  {trabajador_id} detectó interrupción, terminando...")
132                  break
133              print(f"🔄 {trabajador_id} trabajando... ({i+1}/{duracion})")
134          
135          # Simular resultado
136          resultado = {
137              "status": "completed" if not (_sistema_monitoreo_global and _sistema_monitoreo_global.interrumpido) else "interrupted",
138              "puntaje": 85.5 if not (_sistema_monitoreo_global and _sistema_monitoreo_global.interrumpido) else 0
139          }
140          
141          if _sistema_monitoreo_global:
142              _sistema_monitoreo_global.registrar_ejecucion(f"plan_{trabajador_id}", resultado)
143              
144          print(f"✅ {trabajador_id} completado")
145          
146      except Exception as e:
147          print(f"❌ Error en {trabajador_id}: {e}")
148          if _sistema_monitoreo_global:
149              _sistema_monitoreo_global.registrar_ejecucion(f"plan_{trabajador_id}", {"status": "failed", "error": str(e)})
150      finally:
151          if _sistema_monitoreo_global:
152              _sistema_monitoreo_global.desregistrar_trabajador(trabajador_id)
153  
154  def main():
155      """Función principal demo"""
156      global _sistema_monitoreo_global, _output_dir_global
157      
158      print("🏭 DEMO: Sistema de Manejo de Señales")
159      print("=" * 50)
160      print("Presiona Ctrl+C para probar la terminación graciosa")
161      print("=" * 50)
162      
163      # Configurar output
164      output_dir = Path("demo_resultados")
165      output_dir.mkdir(exist_ok=True)
166      _output_dir_global = output_dir
167      
168      # Inicializar monitoreo
169      sistema = SistemaMonitoreoDemo()
170      _sistema_monitoreo_global = sistema
171      sistema.iniciar_monitoreo()
172      
173      # Configurar handlers
174      signal.signal(signal.SIGINT, signal_handler)
175      signal.signal(signal.SIGTERM, signal_handler)
176      atexit.register(atexit_handler)
177      
178      # Simular procesamiento con threads
179      threads = []
180      for i in range(3):
181          thread = threading.Thread(
182              target=simular_trabajo, 
183              args=(f"worker_{i+1}", 10),
184              daemon=True
185          )
186          threads.append(thread)
187          thread.start()
188      
189      try:
190          # Esperar a que terminen los threads
191          for thread in threads:
192              thread.join()
193          
194          print("\n✅ Procesamiento completado normalmente")
195          
196      except KeyboardInterrupt:
197          # El signal handler se encarga de esto
198          pass
199      
200      print(f"\n📊 Resultados disponibles en: {output_dir}")
201  
202  if __name__ == "__main__":
203      main()
