2    from typing import cast

4    import torch

6    from .base_structured_sparsifier import BaseStructuredSparsifier, FakeStructuredSparsity
7

9    class LSTMSaliencyPruner(BaseStructuredSparsifier):
10       """ 
11       Prune packed LSTM weights based on saliency. 
12       For each layer {k} inside a LSTM, we have two packed weight matrices 
13       - weight_ih_l{k} 
14       - weight_hh_l{k} 
15    
16       These tensors pack the weights for the 4 linear layers together for efficiency. 
17    
18       [W_ii | W_if | W_ig | W_io] 
19    
20       Pruning this tensor directly will lead to weights being misassigned when unpacked. 
21       To ensure that each packed linear layer is pruned the same amount: 
22           1. We split the packed weight into the 4 constituent linear parts 
23           2. Update the mask for each individual piece using saliency individually 
24    
25       This applies to both weight_ih_l{k} and weight_hh_l{k}. 
26       """
27   
28       def update_mask(self, module, tensor_name, **kwargs):
29           weights = getattr(module, tensor_name)
30   
31           for p in getattr(module.parametrizations, tensor_name):
32               if isinstance(p, FakeStructuredSparsity):
33                   mask = cast(torch.Tensor, p.mask)
34   
35                   # select weights based on magnitude
36                   if weights.dim() <= 1:
37                       raise Exception(  # noqa: TRY002
38                           "Structured pruning can only be applied to a 2+dim weight tensor!"
39                       )
40                   # take norm over all but first dim
41                   dims = tuple(range(1, weights.dim()))
42                   saliency = weights.norm(dim=dims, p=1)
43   
44                   # handle weights in 4 groups
45                   split_size = len(mask) // 4
46                   masks = torch.split(mask, split_size)
47                   saliencies = torch.split(saliency, split_size)
48   
49                   for keep_mask, sal in zip(masks, saliencies):
50                       # mask smallest k values to be removed
51                       k = int(len(keep_mask) * kwargs["sparsity_level"])
52                       prune = sal.topk(k, largest=False, sorted=False).indices
53                       keep_mask.data[prune] = False  # modifies underlying p.mask directly
