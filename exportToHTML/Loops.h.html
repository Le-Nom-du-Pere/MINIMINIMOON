3    // This file provides two functions to help write elementwise kernels:
4    //
5    //   cpu_kernel(TensorIterator iter, <lambda>)
6    //   cpu_kernel_vec(TensorIterator iter, <lambda>, <vec_lambda>)
7    //
8    // Both functions may generate vectorized code. The cpu_kernel implementation
9    // relies on the compiler's auto-vectorization. The cpu_kernel_vec
10   // implementation uses x86 SIMD intrinsics when available. These functions
11   // are only intended to be used in the ATen/native/cpu subdirectory, since files
12   // in other directories are not compiled with AVX/AVX2 enabled. See README.md
13   // for more details.
14   //
15   // For example, to write a multiplication kernel for float:
16   //
17   //   cpu_kernel(iter, [](float a, float b) { return a * b; });
18   //
19   // Or you may write:
20   //
21   //   cpu_kernel_vec(iter,
22   //     [](float a, float b) { return a * b; },
23   //     [](Vectorized<float> a, Vectorized<float> b) { return a * b; });
24   //
25   // See BinaryOpsKernel.cpp for the complete implementation
26   //
27   //

46   template <typename traits, std::size_t... INDEX>
47   typename traits::ArgsTuple
48   dereference_impl(char* C10_RESTRICT data[], const int64_t* strides, int64_t i,
49                    std::index_sequence<INDEX...>) {
50     return std::make_tuple(
51         c10::load<typename traits::template arg<INDEX>::type>(
52             data[INDEX] + i * strides[INDEX])...);
53   }
54   
55   template <typename traits>
56   typename traits::ArgsTuple
57   dereference(char* C10_RESTRICT data[], const int64_t* strides, int64_t i) {
58     using Indices = std::make_index_sequence<traits::arity>;
59     return dereference_impl<traits>(data, strides, i, Indices{});
60   }
61   
62   template <typename traits, std::size_t... INDEX>
63   typename traits::ArgsTuple
64   dereference_vec_impl(char* C10_RESTRICT data[],
65                        const typename traits::result_type& opt_scalar,
66                        size_t S,
67                        int64_t i,
68                        std::index_sequence<INDEX...>) {
69     using Vec = typename traits::result_type;
70     using scalar_t = typename Vec::value_type;
71     return std::make_tuple(
72         S == INDEX + 1 ?
73         opt_scalar :
74         Vec::loadu(data[INDEX] + i * sizeof(scalar_t))...);
75   }
76   
77   template <typename traits>
78   typename traits::ArgsTuple
79   dereference_vec(char* C10_RESTRICT data[], const typename traits::result_type& opt_scalar, size_t S, int64_t i) {
80     using Indices = std::make_index_sequence<traits::arity>;
81     return dereference_vec_impl<traits>(data, opt_scalar, S, i, Indices{});
82   }

84   template <typename func_t,
85       std::enable_if_t<!std::is_void_v<typename function_traits<func_t>::result_type>>* = nullptr>
86   inline void
87   execute_op(char* C10_RESTRICT data[], const int64_t* strides, int64_t i, int64_t n, func_t&& op) {
88     using traits = function_traits<func_t>;
89     using result_type = typename traits::result_type;
90     for (; i < n; i++) {
91       result_type* out_ptr = (result_type*)(data[0] + i * strides[0]);
92       *out_ptr = c10::guts::apply(op, dereference<traits>(
93           &data[1],
94           &strides[1],
95           i));
96     }
97   }
98   
99   template <typename func_t,
100      std::enable_if_t<std::is_void_v<typename function_traits<func_t>::result_type>>* = nullptr>
101  inline void
102  execute_op(char* C10_RESTRICT data[], const int64_t* strides, int64_t i, int64_t n, func_t&& op) {
103    using traits = function_traits<func_t>;
104    for (; i < n; i++) {
105      c10::guts::apply(op, dereference<traits>(
106          &data[0],
107          &strides[0],
108          i));
109    }
110  }

112  // Basic loop operation (one output, N inputs). May be auto-vectorized
113  // by the compiler. Supports inputs and outputs of different types.
114  template <typename func_t>
115  inline void
116  basic_loop(char* C10_RESTRICT data[], const int64_t* strides_, int64_t i, int64_t n, func_t&& op) {
117    using traits = function_traits<func_t>;
118    constexpr int ntensors = traits::arity + 1;
119  
120    // Copying strides to temporary array helps auto vectorization in older GCC
121    // versions.
122    int64_t strides[ntensors];
123    for (const auto arg : c10::irange(ntensors)) {
124      strides[arg] = strides_[arg];
125    }
126  
127    execute_op(data, strides, i, n, std::forward<func_t>(op));
128  }

130  // the recursive variadic template for iterating over the returned tuple
131  template<class T, size_t N>
132  struct TupleOutput {
133    static void handle(char *C10_RESTRICT data[], const int64_t *strides, int64_t i,
134                       const T &tuple) {
135      TupleOutput<T, N - 1>::handle(data, strides, i, tuple);
136  
137      auto output = std::get<N - 1>(tuple);
138      using output_type = decltype(output);
139      output_type * out_ptr = (output_type *)(data[N - 1] + i * strides[N - 1]);
140      *out_ptr = output;
141    }
142  };
143  
144  // Base case for the above recursive template
145  template<class T>
146  struct TupleOutput<T, 1> {
147    static void handle(char *C10_RESTRICT data[], const int64_t *strides, int64_t i,
148                       const T &tuple) {
149      auto output = std::get<0>(tuple);
150      using output_type = decltype(output);
151      output_type* out_ptr = (output_type *)(data[0] + i * strides[0]);
152      *out_ptr = output;
153    }
154  };

156  template<class... Args>
157  void handle_tuple_outputs(char* C10_RESTRICT data[],
158                            const int64_t* strides,
159                            int64_t i,
160                            const std::tuple<Args...> &tuple) {
161    TupleOutput<decltype(tuple), sizeof...(Args)>::handle(data, strides, i, tuple);
162  }

164  // Loop operation for `cpu_kernel_multiple_outputs`.
165  // 1. Use `c10::guts::apply` to make dynamic method invocation
166  //    for the lambda passed in `cpu_kernel_multiple_outputs`.
167  // 2. Iterate over the members of the returned tuple, set the corresponding
168  //    output tensor by the tuple member in `handle_tuple_outputs` function.
169  template <typename func_t>
170  inline void
171  multiple_outputs_loop(char* C10_RESTRICT data[], const int64_t* strides_, int64_t i, int64_t n, func_t&& op) {
172    using traits = function_traits<func_t>;
173  
174    using result_type = typename traits::result_type;
175    constexpr int num_outputs = std::tuple_size_v<result_type>;
176    constexpr int ntensors = traits::arity + num_outputs;
177  
178    // Copying strides to temporary array helps auto vectorization in older GCC
179    // versions.
180    int64_t strides[ntensors];
181    for (const auto arg : c10::irange(ntensors)) {
182      strides[arg] = strides_[arg];
183    }
184  
185    for (; i < n; i++) {
186      auto output = c10::guts::apply(op, dereference<traits>(
187        &data[num_outputs],
188        &strides[num_outputs],
189        i));
190      handle_tuple_outputs(data, strides, i, output);
191    }
192  }

194  // Explicitly vectorized loop implementation. All inputs and outputs must be
195  // the same type and contiguous with one exception: a single input may be
196  // a scalar (stride 0). It's position is indicated by the argument `S`. If `S`
197  // is 0, then there are no scalar inputs.
198  template <typename func_t, typename vec_func_t>
199  inline void
200  vectorized_loop(char** C10_RESTRICT data_, int64_t n, int64_t S, func_t&& op, vec_func_t&& vop) {
201    using traits = function_traits<vec_func_t>;
202    using scalar_t = typename function_traits<func_t>::result_type;
203    using Vec = Vectorized<scalar_t>;
204    constexpr int ntensors = traits::arity + 1;
205  
206    char* C10_RESTRICT data[ntensors];
207    for (const auto arg : c10::irange(ntensors)) {
208      data[arg] = data_[arg];
209    }
210  
211    Vec opt_scalar = Vec(S > 0 ? c10::load((scalar_t*)data[S]) : scalar_t(0));
212    int64_t i = 0;
213    for (; i <= n - 2 * Vec::size(); i += 2 * Vec::size()) {
214      auto args1 = dereference_vec<traits>(&data[1], opt_scalar, S, i);
215      auto args2 = dereference_vec<traits>(&data[1], opt_scalar, S, i + Vec::size());
216      auto out1 = c10::guts::apply(vop, std::move(args1));
217      auto out2 = c10::guts::apply(vop, std::move(args2));
218      out1.store(data[0] + i * sizeof(scalar_t));
219      out2.store(data[0] + (i + Vec::size()) * sizeof(scalar_t));
220    }
221    if (i < n) {
222      int64_t strides[ntensors];
223      for (const auto arg : c10::irange(ntensors)) {
224        strides[arg] = (S > 0 && arg == S) ? 0 : sizeof(scalar_t);
225      }
226      basic_loop(data, strides, i, n, std::forward<func_t>(op));
227    }
228  }

231  template <typename traits, typename cb_t>
232  inline void unroll_contiguous_scalar_checks(
233      const int64_t* /*strides*/,
234      std::index_sequence<>,
235      cb_t&& cb) {
236    cb(0);
237  }
238  
239  template <typename traits, typename cb_t, size_t INDEX0, size_t ...INDEX>
240  inline void unroll_contiguous_scalar_checks(
241      const int64_t* strides,
242      std::index_sequence<INDEX0, INDEX...>,
243      cb_t&& cb) {
244    if (is_contiguous_scalar<traits, INDEX0 + 1>(strides)) {
245      cb(INDEX0 + 1);
246    } else {
247      unroll_contiguous_scalar_checks<traits>(strides, std::index_sequence<INDEX...>{}, std::forward<cb_t>(cb));
248    }
249  }

251  template <typename op_t, typename vop_t>
252  struct VectorizedLoop2d {
253    op_t op;
254    vop_t vop;
255  
256    using traits = function_traits<op_t>;
257    static constexpr int ntensors = traits::arity + 1;
258    using data_t = std::array<char*, ntensors>;
259  
260    VectorizedLoop2d(op_t op, vop_t vop):
261      op(std::move(op)), vop(std::move(vop)) {}
262  
263    static void advance(data_t &data, const int64_t *outer_strides) {
264      for (const auto arg : c10::irange(data.size())) {
265        data[arg] += outer_strides[arg];
266      }
267    }
268  
269    void operator()(char** base, const int64_t *strides, int64_t size0, int64_t size1) {
270      data_t data;
271      std::copy_n(base, ntensors, data.data());
272      const int64_t *outer_strides = &strides[ntensors];
273  
274      if (is_contiguous<traits>(strides)) {
275        for ([[maybe_unused]] const auto i : c10::irange(size1)) {
276          vectorized_loop(data.data(), size0, 0, op, vop);
277          advance(data, outer_strides);
278        }
279      } else {
280        using Indices = std::make_index_sequence<traits::arity>;
281        unroll_contiguous_scalar_checks<traits>(strides, Indices{}, [&](size_t idx) {
282          if (idx) {
283            for ([[maybe_unused]] const auto i : c10::irange(size1)) {
284              vectorized_loop(data.data(), size0, idx, op, vop);
285              advance(data, outer_strides);
286            }
287          } else {
288            for ([[maybe_unused]] const auto i : c10::irange(size1)) {
289              basic_loop(data.data(), strides, 0, size0, op);
290              advance(data, outer_strides);
291            }
292          }
293        });
294      }
295    }
296  };

298  template <typename op_t, typename vop_t>
299  VectorizedLoop2d<op_t, vop_t> make_vectorized_loop2d(
300      op_t &&op, vop_t &&vop) {
301    return VectorizedLoop2d<op_t, vop_t>(std::forward<op_t>(op), std::forward<vop_t>(vop));
302  }

304  template <typename func_t>
305  void cpu_kernel(TensorIteratorBase& iter, func_t&& op, int64_t grain_size = at::internal::GRAIN_SIZE) {
306    using traits = function_traits<func_t>;
307    // this could be extended to work with void return types
308    TORCH_INTERNAL_ASSERT(iter.ninputs() == traits::arity);
309    TORCH_INTERNAL_ASSERT(iter.noutputs() == 1);
310    // dynamic casting not currently supported on CPU
311    TORCH_INTERNAL_ASSERT(!needs_dynamic_casting<func_t>::check(iter));
312  
313    iter.for_each([&](char** data, const int64_t* strides, int64_t n) {
314      // basic loop can handle 1d slices with arbitrary strides, and 1d slices is all that
315      // iter.for_each is ever sending to the loop lambda
316        basic_loop(data, strides, 0, n, op);
317    }, grain_size);
318    iter.cast_outputs();
319  }

321  // This function helps write elementwise kernels that requires multiple outputs.
322  // It follows the similar structure of cpu_kernel.
323  // Instead of `basic_loop` function, a new `multiple_outputs_loop` function is
324  // manipulated to handle multiple return values.
325  // For now `needs_dynamic_casting` check is not added as the passed lambda (`func_t`)
326  // of `multiple_outputs_loop` returns `std::tuple` instead of `scalar_t`.
327  // The `gpu_kernel_multiple_outputs` is also implemented without this check,
328  // We could extend `needs_dynamic_casting` to support both `std::tuple` and
329  // `thrust::tuple` in the future.
330  template <typename func_t>
331  void cpu_kernel_multiple_outputs(TensorIteratorBase& iter, func_t&& op, int64_t grain_size = at::internal::GRAIN_SIZE) {
332    using traits = function_traits<func_t>;
333    TORCH_INTERNAL_ASSERT(iter.ninputs() == traits::arity);
334  
335    iter.for_each([&](char** data, const int64_t* strides, int64_t n) {
336      multiple_outputs_loop(data, strides, 0, n, op);
337    }, grain_size);
338    iter.cast_outputs();
339  }

341  template <bool check_dynamic_cast=true, typename func_t, typename vec_func_t>
342  void cpu_kernel_vec(TensorIteratorBase& iter, func_t&& op, vec_func_t&& vop, int64_t grain_size = at::internal::GRAIN_SIZE) {
343    using traits = function_traits<func_t>;
344    // this could be extended to work with void return types
345    TORCH_INTERNAL_ASSERT(iter.ninputs() == traits::arity);
346    TORCH_INTERNAL_ASSERT(iter.noutputs() == 1);
347    // dynamic casting not currently supported on CPU, but some kernels (like Fill)
348    // explicitly dynamic_cast, so we give the opt-out of checking.
349    if constexpr (check_dynamic_cast) {
350      TORCH_INTERNAL_ASSERT(!needs_dynamic_casting<func_t>::check(iter));
351    }
352  
353    iter.for_each(make_vectorized_loop2d(std::forward<func_t>(op), std::forward<vec_func_t>(vop)), grain_size);
354    iter.cast_outputs();
355  }

357  template <typename func_t>
358  void cpu_serial_kernel(TensorIteratorBase& iter, func_t&& op, const Range& range) {
359    using traits = function_traits<func_t>;
360    constexpr bool result_void = std::is_void_v<typename traits::result_type>;
361    TORCH_INTERNAL_ASSERT(iter.ninputs() == traits::arity &&
362                          ((result_void && iter.noutputs() == 0) || (!result_void && iter.noutputs() == 1)));
363    // dynamic casting not currently supported on CPU
364    TORCH_INTERNAL_ASSERT(!needs_dynamic_casting<func_t>::check(iter));
365  
366    iter.serial_for_each([&](char** data, const int64_t* strides, int64_t n) {
367      basic_loop(data, strides, 0, n, op);
368    }, range);
369    iter.cast_outputs();
370  }

373  void cpu_serial_kernel(TensorIteratorBase& iter, func_t&& op) {
374    cpu_serial_kernel(iter, std::forward<func_t>(op), {0, iter.numel()});
