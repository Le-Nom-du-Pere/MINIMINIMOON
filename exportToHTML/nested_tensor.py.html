15   def get_tensor_symint(tensor, *, coeff=1):
16       from torch._subclasses.fake_tensor import FakeTensor
17       from torch._subclasses.functional_tensor import mb_unwrap_functional_tensor
18   
19       # NB: Only FakeTensor is associated with a memo
20       tensor = mb_unwrap_functional_tensor(tensor)
21       if isinstance(tensor, FakeTensor):
22           return tensor.get_nested_int(coeff=coeff)
23   
24       global _tensor_id_counter
25   
26       tensor_symint = _tensor_symint_registry.get(tensor)
27       if tensor_symint is None:
28           tensor_symint = torch.SymInt(NestedIntNode(_tensor_id_counter, coeff))
29           _tensor_id_counter += 1
30           _tensor_symint_registry[tensor] = tensor_symint
31       return tensor_symint

53   class NestedTensor(torch.Tensor):
54       _values: torch.Tensor  # type: ignore[assignment]
55       _offsets: torch.Tensor
56       _lengths: Optional[torch.Tensor]
57       # NOTE [ Nested ints for ragged sizes and strides ]
58       #
59       # Jagged layout tensors are tensors that represent a n-dim tensor with a
60       # ragged dimension, but are backed by an (n-1)-dim tensor underneath, e.g.,
61       # a jagged tensor with outer shape [B, x, D] is represented internally by a
62       # tensor with shape [sum(x), D] where we introduce what we call a nested int
63       # denoted as "x" here (but sometimes denoted with "*" to
64       # represent the ragged dimension, and sum(x) represents the dim of the inner
65       # tensor or equivalently the sum of all the sizes of the constituent
66       # tensors' varying lengths.
67       #
68       # We also use nested ints to represent the strides of this tensor.
69       # For example, a jagged tensor with shape [B, x, D] can be strided in two
70       # ways: [xD, D, 1] and [x, 1, sum(x)], where xD represents x multiplied by D
71       _size: tuple[int, ...]
72       _strides: tuple[int, ...]
73       # Indicates that the nth dimension is ragged
74       _ragged_idx: int
75       _metadata_cache: Dict[str, Any]
76   
77       @staticmethod
78       def __new__(
79           cls,
80           values,
81           offsets,
82           *,
83           lengths=None,
84           **kwargs,
85       ):
86           ks = DispatchKeySet(DispatchKey.NestedTensor)
87           ks = ks.add(DispatchKey.AutogradNestedTensor)
88   
89           # Only support jagged for now.
90           assert offsets is not None
91           assert offsets.ndim == 1
92           assert not isinstance(values, NestedTensor)
93           assert values.device == offsets.device
94   
95           # Query cache for the symint associated with offsets or lengths
96           # (create a new one if needed).
97           ragged_source = offsets if lengths is None else lengths
98           ragged_size = get_tensor_symint(ragged_source, coeff=1)
99           _ragged_idx = kwargs.get("_ragged_idx", 1)
100          B = offsets.shape[0] - 1
101          if lengths is not None:
102              assert B == lengths.shape[0]
103  
104          # subtract 1 to convert to values dim space
105          r = _ragged_idx - 1
106          _size = (B, *values.shape[:r], ragged_size, *values.shape[r + 1 :])
107          stride = values.stride()
108          _strides = (ragged_size * stride[r], *stride)
109  
110          r = torch.Tensor._make_wrapper_subclass(
111              cls,
112              _size,
113              _strides,
114              0,
115              torch.contiguous_format,
116              values.dtype,
117              torch.jagged,
118              values.device,
119              False,
120              kwargs.get("requires_grad", False),
121              "sizes",
122              False,
123              True,  # dispatch_layout
124              ks,
125              # don't try to calculate storage based on non-zero size
126              storage_size=values.untyped_storage().size(),
127          )
128          r._ragged_idx = _ragged_idx
129          r._size = _size
130          r._strides = _strides
131  
132          return r

135          super().__init__()
136  
137          self._values = values
138          self._offsets = offsets

172      def _get_max_seqlen(self):
173          max_seqlen_tensor = self._max_seqlen_tensor
174          if max_seqlen_tensor is None:
175              # compute & cache
176              max_val = _get_sdpa_extreme_seqlen(
177                  torch.max,
178                  self._offsets.diff() if self._lengths is None else self._lengths,
179              )
180              max_seqlen_tensor = _store_val_in_tensor(max_val)
181              self._metadata_cache["max_seqlen"] = max_seqlen_tensor
182          return _load_val_from_tensor(max_seqlen_tensor)

269          args = (constructor_kwargs,)
270          return (torch._tensor._rebuild_from_type_v2, (func, type(self), args, state))

287      def __tensor_unflatten__(inner_tensors: Dict, meta, outer_size, outer_stride):
288          from torch._subclasses.fake_tensor import FakeTensor
289  
290          # inner tensors: _values, _offsets, [_lengths], [_min_seqlen], [_max_seqlen]
291          assert len(inner_tensors) >= 2 and len(inner_tensors) <= 5
292          values = inner_tensors["_values"]
293          offsets = inner_tensors["_offsets"]
294          lengths = inner_tensors.get("_lengths", None)
295          min_seqlen_tensor = inner_tensors.get("_min_seqlen_tensor", None)
296          max_seqlen_tensor = inner_tensors.get("_max_seqlen_tensor", None)
297  
298          metadata_cache = {}
299          if min_seqlen_tensor is not None:
300              metadata_cache["min_seqlen"] = min_seqlen_tensor
301          if max_seqlen_tensor is not None:
302              metadata_cache["max_seqlen"] = max_seqlen_tensor
303          ragged_idx = meta["ragged_idx"]
304  
305          # Alternatively, we could make it the caller's responsibility to
306          # cache it. But this heuristic seems simple enough.
307          ragged_source = offsets if lengths is None else lengths
308          if isinstance(ragged_source, FakeTensor):
309              ragged_size = outer_size[ragged_idx]
310              ragged_source.nested_int_memo = ragged_size
311  
312          return NestedTensor(
313              values,
314              offsets=offsets,
315              lengths=lengths,
316              requires_grad=meta["requires_grad"],
317              _ragged_idx=ragged_idx,
318              _metadata_cache=metadata_cache,
319          )

321      @classmethod
322      def __torch_dispatch__(cls, func, types, args=(), kwargs=None):  # type: ignore[override]
323          # If you're wondering why there's a nested tensor with one of its
324          # size = -1, see note: [NJT outer_size in AOTDispatcher]
325          kwargs = {} if kwargs is None else kwargs
326  
327          # Lazy import to avoid circular dependency
328          from .ops import lookup_jagged
329  
330          fn = lookup_jagged(func, *args, **kwargs)
331          if fn is not None:
332              return fn(*args, **kwargs)
333  
334          # Poor man's redispatch for composite ops. This becomes relevant under inference
335          # mode, where disabling autograd key dispatch prevents decomposition.
336          all_dks = (
337              # We want to handle both the cases where NestedTensor overrides the
338              # composite implicit autograd kernel, and the case where it doesn't.
339              # Prioritize calling into NestedTensor's kernel if it exists.
340              torch._C.DispatchKey.CompositeImplicitAutogradNestedTensor,
341              torch._C.DispatchKey.CompositeImplicitAutograd,
342          )
343          for dk in all_dks:
344              if torch._C._dispatch_has_kernel_for_dispatch_key(func.name(), dk):
345                  with torch.overrides.enable_reentrant_dispatch():
346                      return func._op_dk(dk, *args, **kwargs)
347  
348          raise NotImplementedError(func)

352          if kwargs is None:

428  # Need to make it obvious that users should be passing in offsets
429  def jagged_from_list(
430      tensors: List[torch.Tensor],
431      offsets: Optional[torch.Tensor],
432      dtype=None,
433      device=None,
434  ) -> tuple[NestedTensor, torch.Tensor]:
435      """Constructs a NestedTensor backed by jagged layout from a list of tensors"""
436  
437      if len(tensors) == 0:
438          raise RuntimeError("Cannot construct a nested tensor from an empty tensor list")
439      if not len(set(t.dtype for t in tensors)) == 1:  # noqa: C401
440          raise RuntimeError(
441              "When constructing a nested tensor, all tensors in list must have the same dtype"
442          )
443      if not len(set(t.device for t in tensors)) == 1:  # noqa: C401
444          raise RuntimeError(
445              "When constructing a nested tensor, all tensors in list must be on the same device"
446          )
447      if not len(set(t.dim() for t in tensors)) == 1:  # noqa: C401
448          raise RuntimeError(
449              "When constructing a nested tensor, all tensors in list must have the same dim"
450          )
451      component_dim = tensors[0].dim()
452      if component_dim == 0:
453          raise RuntimeError(
454              "Cannot construct a nested tensor from a list of zero-dim tensors"
455          )
456  
457      # Check that the NT is representable by the jagged layout, which
458      # allows for a single ragged dimension after the batch dim.
459      # e.g. (B, *, D_0, ..., D_N), (B, D_0, *, ..., D_N), etc.
460      sizes = [t.shape for t in tensors]
461      ragged_idx = None
462      for d in range(component_dim):
463          dim_is_ragged = any(size[d] != sizes[0][d] for size in sizes)
464          if dim_is_ragged:
465              if ragged_idx is None:
466                  # add 1 to convert to outer NJT dim space
467                  ragged_idx = d + 1
468              else:
469                  raise RuntimeError(
470                      "Cannot represent given tensor list as a nested tensor with the jagged layout. "
471                      "Note that the jagged layout only allows for a single ragged dimension. "
472                      "For example: (B, *, D_0, D_1, ..., D_N), with ragged * dim."
473                  )
474  
475      # allow for a rectangular NJT and default the ragged dim next to the batch dim
476      if ragged_idx is None:
477          ragged_idx = 1
478  
479      # Set properties appropriately.
480      values = torch.cat(tensors, dim=(ragged_idx - 1))
481      to_kwargs = {}
482      if device is not None:
483          to_kwargs["device"] = device
484      if dtype is not None:
485          to_kwargs["dtype"] = dtype
486      values = values.to(**to_kwargs)
487  
488      # Calculate jagged offsets if not provided.
489      if offsets is None:
490          # Jagged layout specifies that offsets are stored as int64 on the same device as values.
491          # TODO: An alternative way to construct offsets is to use F.pad. This avoids creating
492          # an extra leaf tensor during the forward, potentially resolving compatibility issues.
493          offsets = torch.cat(
494              [
495                  torch.zeros(1, dtype=torch.int64, device=values.device),
496                  torch.tensor(
497                      [s[ragged_idx - 1] for s in sizes], device=values.device
498                  ).cumsum(dim=0),
499              ]
500          )
501  
502      # compute this now since it's easy
503      min_seqlen = min(t.shape[ragged_idx - 1] for t in tensors)
504      max_seqlen = max(t.shape[ragged_idx - 1] for t in tensors)
505      ret_nt = nested_view_from_values_offsets(
506          values,
507          offsets,
508          min_seqlen=min_seqlen,
509          max_seqlen=max_seqlen,
510          ragged_idx=ragged_idx,
511      )
512      return (ret_nt, offsets)  # type: ignore[return-value]

515  def jagged_from_tensor_and_lengths(
516      tensor: torch.Tensor, starts: torch.Tensor, lengths: torch.Tensor
517  ) -> tuple[NestedTensor, torch.Tensor, Optional[torch.Tensor]]:
518      """Constructs a NestedTensor backed by jagged layout from a tensor, starts of sequences, and sequence lengths"""
519      batch_size = tensor.shape[0]
520      if is_expandable_to(starts.shape, (batch_size,)) and is_expandable_to(
521          lengths.shape, (batch_size,)
522      ):
523          start_list = starts.expand(batch_size)
524          length_list = lengths.expand(batch_size)
525      else:
526          raise RuntimeError(
527              "When constructing a jagged nested tensor using narrow(), "
528              "your start and length must be Tensors that broadcast to input.shape[0]"
529          )
530  
531      # Calculate jagged offsets
532      assert len(tensor.shape) >= 2, (
533          "tensor must at least be 2D for the nested narrow op to work"
534      )
535      max_seq_len = tensor.shape[1]
536      offset_lengths = max_seq_len * torch.arange(
537          0, batch_size, dtype=torch.int64, device=tensor.device
538      )
539      # Jagged layout specifies that offsets are stored as int64 on the same device as values.
540      offsets = torch.cat(
541          [
542              start_list + offset_lengths,
543              (start_list[-1] + offset_lengths[-1] + length_list[-1]).unsqueeze(0),
544          ]
545      )
546  
547      # Reshape buffer to flatten the 1st and 2nd dimension (view used to enforce non-copy)
548      if len(tensor.shape) > 2:
549          values = tensor.view(-1, *tensor.shape[2:])
550      else:
551          values = tensor.view(-1)
552  
553      # Check if offsets and lengths make it possibly contiguous and return a regular NT
554      is_contiguous = True
555      orig_dim = tensor.shape[1]
556      if torch.any(length_list[1:-1].ne(orig_dim)):
557          is_contiguous = False
558      if torch.any(offsets[1:-2].diff().ne(orig_dim)):
559          is_contiguous = False
560      if offsets[0] + length_list[0] != orig_dim:
561          is_contiguous = False
562  
563      actual_max_seqlen = int(torch.max(lengths).item())
564      min_seqlen = int(torch.min(lengths).item())
565  
566      if is_contiguous:
567          ret_nt = nested_view_from_values_offsets(
568              values[offsets[0] : offsets[-1]],
569              offsets - offsets[0],
570              min_seqlen=min_seqlen,
571              max_seqlen=actual_max_seqlen,
572          )
573      else:
574          ret_nt = nested_view_from_values_offsets_lengths(
575              values,
576              offsets,
577              length_list,
578              min_seqlen=min_seqlen,
579              max_seqlen=actual_max_seqlen,
580          )
581  
582      return (ret_nt, offsets, None if is_contiguous else length_list)
