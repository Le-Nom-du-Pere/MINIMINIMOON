6    # NOTE: Why do we need a different registry than the one in torchlib?
7    # The registry in torchlib is used to register functions that are already implemented in
8    # torchlib, and is designed to be a static singleton. It does not take into account custom ops or different
9    # opsets etc. The registry implemented for the exporter is designed to be modifiable at
10   # export time by users, and is designed with dispatching in mind.
11

36   @dataclasses.dataclass
37   class OnnxDecompMeta:
38       """A wrapper of onnx-script function with additional metadata. 
39    
40       onnx_function: The onnx-script function from torchlib. 
41       fx_target: The PyTorch node callable target. 
42       signature: The ONNX signature of the function. When None, the signature is inferred. 
43       is_custom: Whether the function is a custom function. 
44       is_complex: Whether the function is a function that handles complex valued inputs. 
45       opset_introduced: 
46           The ONNX opset version in which the function was introduced. 
47           Its specifies the minimum ONNX opset version required to use the function. 
48       device: The device the function is registered to. If None, it is registered to all devices. 
49       skip_signature_inference: Whether to skip signature inference for the function. 
50       """
51   
52       onnx_function: Callable
53       fx_target: TorchOp
54       signature: _schemas.OpSignature | None
55       is_custom: bool = False
56       is_complex: bool = False
57       opset_introduced: int = 18
58       device: Literal["cuda", "cpu"] | str | None = None  # noqa: PYI051
59       skip_signature_inference: bool = False
60   
61       def __post_init__(self) -> None:
62           if self.signature is None and not self.skip_signature_inference:
63               try:
64                   if isinstance(self.onnx_function, onnxscript.OnnxFunction):
65                       signature = _schemas.OpSignature.from_function(  # type: ignore[attr-defined]
66                           self.onnx_function,
67                           self.onnx_function.function_ir.domain,
68                           self.onnx_function.name,
69                           opset_version=self.onnx_function.opset.version,
70                       )
71                   else:
72                       signature = _schemas.OpSignature.from_function(
73                           self.onnx_function, "__traced", self.onnx_function.__name__
74                       )
75               except Exception as e:
76                   # Log an warning if the op is custom. Raise exception for builtin ops.
77                   if not self.is_custom:
78                       raise
79                   else:
80                       # When the function is targeting an HOP, for example, it will accept
81                       # functions as arguments and fail to generate an ONNX signature.
82                       # In this case we set signature to None and dispatch to this function always.
83                       logger.warning(
84                           "Failed to infer the signature for function '%s' because '%s'"
85                           "All nodes targeting `%s` will be dispatched to this function",
86                           self.onnx_function,
87                           e,
88                           self.fx_target,
89                       )
90               else:
91                   self.signature = signature
92                   self.onnx_function._pt_onnx_signature = signature  # type: ignore[attr-defined]

95   def _get_overload(qualified_name: str) -> torch._ops.OpOverload | None:
96       """Obtain the torch op from <namespace>::<op_name>[.<overload>]"""
97       # TODO(justinchuby): Handle arbitrary custom ops
98       namespace, opname_overload = qualified_name.split("::")
99       op_name, *maybe_overload = opname_overload.split(".", 1)
100      if namespace == "_operator":
101          # Builtin functions
102          return getattr(operator, op_name)
103      if namespace == "math":
104          return getattr(math, op_name)
105      if namespace == "torchvision":
106          if importlib.util.find_spec("torchvision") is None:
107              logger.warning("torchvision is not installed. Skipping %s", qualified_name)
108              return None
109      try:
110          op_packet = getattr(getattr(torch.ops, namespace), op_name)
111          if maybe_overload:
112              overload = maybe_overload[0]
113          elif "default" in op_packet._overload_names or "" in op_packet._overload_names:
114              # Has a default overload
115              overload = "default"
116          else:
117              logger.warning(
118                  "'%s' does not have a 'default' overload. This could be an error in specifying the op name. Ignoring.",
119                  qualified_name,
120                  stacklevel=1,
121              )
122              return None
123  
124          return getattr(op_packet, overload)  # type: ignore[call-overload]
125      except AttributeError:
126          if qualified_name.endswith("getitem"):
127              # This is a special case where we registered the function incorrectly,
128              # but for BC reasons (pt<=2.4) we need to keep it.
129              return None
130          logger.info("'%s' is not found in this version of PyTorch.", qualified_name)
131          return None
132      except Exception:
133          logger.exception("Failed to find torch op '%s'", qualified_name)
134          return None

137  class ONNXRegistry:
138      """Registry for ONNX functions. 
139   
140      The registry maintains a mapping from qualified names to symbolic functions under a 
141      fixed opset version. It supports registering custom onnx-script functions and for 
142      dispatcher to dispatch calls to the appropriate function. 
143   
144      """
145  
146      def __init__(self) -> None:
147          """Initializes the registry"""
148          self._opset_version = _constants.TORCHLIB_OPSET
149          self.functions: dict[TorchOp | str, list[OnnxDecompMeta]] = {}
150  
151      @property
152      def opset_version(self) -> int:
153          """The ONNX opset version the exporter should target."""
154          return self._opset_version
155  
156      @classmethod
157      def from_torchlib(cls, opset_version=_constants.TORCHLIB_OPSET) -> ONNXRegistry:
158          """Populates the registry with ATen functions from torchlib. 
159   
160          Args: 
161              torchlib_registry: The torchlib registry to use for populating the registry. 
162          """
163          registry = cls()
164          registry._opset_version = opset_version
165          for meta in _torchlib_registry.get_torchlib_ops():
166              registry._register(meta.fx_target, meta)
167  
168          # TODO(justinchuby): Remove this once torchlib is migrated to PyTorch
169          torchlib_ops = onnxscript_apis.get_torchlib_ops()
170  
171          for torchlib_meta in torchlib_ops:
172              qualified_name = torchlib_meta.qualified_name
173              overload_func = torchlib_meta.function
174              try:
175                  # NOTE: This is heavily guarded with try-except because we don't want
176                  # to fail the entire registry population if one function fails.
177                  target = _get_overload(qualified_name)
178                  if target is None:
179                      continue
180  
181                  meta = OnnxDecompMeta(
182                      onnx_function=overload_func,
183                      fx_target=target,
184                      signature=None,
185                      is_custom=False,
186                      is_complex=torchlib_meta.is_complex,
187                  )
188                  registry._register(target, meta)
189              except Exception:
190                  logger.exception("Failed to register '%s'. Skipped", qualified_name)
191                  continue
192  
193          registry._cleanup_registry_based_on_opset_version()
194          return registry
195  
196      def _register(
197          self,
198          target: TorchOp,
199          onnx_decomposition: OnnxDecompMeta,
200      ) -> None:
201          """Registers a OnnxDecompMeta to an operator. 
202   
203          Args: 
204              target: The PyTorch node callable target. 
205              onnx_decomposition: The OnnxDecompMeta to register. 
206          """
207          target_or_name: str | TorchOp
208          if isinstance(target, torch._ops.OpOverload):
209              # Get the qualified name of the aten op because torch._ops.OpOverload lookup in
210              # a dictionary is unreliable for some reason.
211              target_or_name = target.name()
212          else:
213              target_or_name = target
214          if onnx_decomposition.is_custom:
215              self.functions.setdefault(target_or_name, []).insert(0, onnx_decomposition)
216          else:
217              self.functions.setdefault(target_or_name, []).append(onnx_decomposition)
218  
219      def register_op(
220          self,
221          target: TorchOp,
222          function: Callable,
223          is_complex: bool = False,
224      ) -> None:
225          """Registers a custom operator: torch.ops.<namespace>.<op_name>.<overload>. 
226   
227          Args: 
228              target: The PyTorch node callable target. 
229              function: The onnx-script function to register. 
230              is_complex: Whether the function is a function that handles complex valued inputs. 
231          """
232          if isinstance(target, torch._ops.OpOverloadPacket):
233              raise TypeError(
234                  f"Target '{target}' should be provided as an OpOverload instead of an "
235                  "OpOverloadPacket. You can get the default overload with "
236                  "<op>.default"
237              )
238  
239          self._register(
240              target,
241              OnnxDecompMeta(
242                  onnx_function=function,
243                  fx_target=target,
244                  signature=None,
245                  is_custom=True,
246                  is_complex=is_complex,
247              ),
248          )
249  
250      def get_decomps(self, target: TorchOp) -> list[OnnxDecompMeta]:
251          """Returns a list of OnnxDecompMeta for the given op: torch.ops.<namespace>.<op_name>.<overload>. 
252   
253          The list is ordered by the time of registration. The custom operators should come 
254          first in the list. 
255   
256          Args: 
257              target: The PyTorch node callable target. 
258          Returns: 
259              A list of OnnxDecompMeta corresponding to the given name, or None if 
260              the name is not in the registry. 
261          """
262          target_or_name: str | TorchOp
263          if isinstance(target, torch._ops.OpOverload):
264              # Get the qualified name of the aten op because torch._ops.OpOverload lookup in
265              # a dictionary is unreliable for some reason.
266              target_or_name = target.name()
267          else:
268              target_or_name = target
269          decomps = self.functions.get(target_or_name, [])
270          return sorted(decomps, key=lambda x: x.is_custom, reverse=True)
271  
272      def is_registered(self, target: TorchOp) -> bool:
273          """Returns whether the given op is registered: torch.ops.<namespace>.<op_name>.<overload>. 
274   
275          Args: 
276              target: The PyTorch node callable target. 
277   
278          Returns: 
279              True if the given op is registered, otherwise False. 
280          """
281          return bool(self.get_decomps(target))
282  
283      def _cleanup_registry_based_on_opset_version(self) -> None:
284          """Pick the implementation with the highest opset version valid until the current opset version."""
285          cleaned_functions = {}
286          for target_or_name, decomps in self.functions.items():
287              # Filter decompositions to only include those with opset_introduced <= opset_version
288              decomps = [d for d in decomps if d.opset_introduced <= self.opset_version]
289  
290              # Keep only the decomposition with the highest opset_introduced
291              if decomps:
292                  # Find the maximum opset_introduced
293                  max_opset = max(d.opset_introduced for d in decomps)
294  
295                  # Keep all decompositions with the maximum opset_introduced
296                  cleaned_functions[target_or_name] = [
297                      d for d in decomps if d.opset_introduced == max_opset
298                  ]
299  
300          self.functions = cleaned_functions
301  
302      def __repr__(self) -> str:
303          return f"{self.__class__.__name__}(functions={self.functions})"
