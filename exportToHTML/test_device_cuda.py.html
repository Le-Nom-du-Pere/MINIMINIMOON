9    def test_cuda_config():
10       """Test CUDA configuration and thread settings."""
11       print("=== CUDA Configuration Test ===")
12       
13       print(f"CUDA available: {torch.cuda.is_available()}")
14       if torch.cuda.is_available():
15           print(f"CUDA devices: {torch.cuda.device_count()}")
16           for i in range(torch.cuda.device_count()):
17               print(f"  Device {i}: {torch.cuda.get_device_name(i)}")
18       
19       # Test CUDA configuration
20       config = initialize_device_config('cuda')
21       device = config.get_device()
22       threads = torch.get_num_threads()
23       
24       print(f"Configured device: {device}")
25       print(f"PyTorch threads: {threads}")
26       
27       if device.type == 'cuda':
28           print("✓ Successfully configured CUDA device")
29           print(f"✓ PyTorch threads set to {threads} (should be 1 for CUDA)")
30       else:
31           print("✓ Fallback to CPU (CUDA not available)")
32           print(f"✓ PyTorch threads: {threads}")
33       
34       # Test CPU configuration
35       config_cpu = initialize_device_config('cpu')
36       device_cpu = config_cpu.get_device()
37       threads_cpu = torch.get_num_threads()
38       
39       print(f"\nCPU device: {device_cpu}")
40       print(f"CPU threads: {threads_cpu}")
41   
42   
43   if __name__ == "__main__":
44       test_cuda_config()
