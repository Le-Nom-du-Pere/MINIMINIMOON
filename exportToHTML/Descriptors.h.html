12   inline int dataSize(miopenDataType_t dataType)
13   {
14     switch (dataType) {
15       case miopenHalf: return 2;
16       case miopenFloat: return 4;
17       case miopenBFloat16: return 2;
18       default: return 8;
19     }
20   }

22   template <typename T, miopenStatus_t (*dtor)(T*)>
23   struct DescriptorDeleter {
24     void operator()(T* x) {
25       if (x != nullptr) {
26         MIOPEN_CHECK(dtor(x));
27       }
28     }
29   };

31   // A generic class for wrapping MIOpen descriptor types.  All you need
32   // is to give the underlying type the Descriptor_t points to (usually,
33   // if it's miopenTensorDescriptor_t it points to miopenTensorStruct),
34   // the constructor and the destructor.  Subclasses are responsible
35   // for defining a set() function to actually set the descriptor.
36   //
37   // Descriptors default construct to a nullptr, and have a descriptor
38   // initialized the first time you call set() or any other initializing
39   // function.
40   template <typename T, miopenStatus_t (*ctor)(T**), miopenStatus_t (*dtor)(T*)>
41   // NOLINTNEXTLINE(bugprone-exception-escape)
42   class TORCH_CUDA_CPP_API Descriptor {
43    public:
44     // Use desc() to access the underlying descriptor pointer in
45     // a read-only fashion.  Most client code should use this.
46     // If the descriptor was never initialized, this will return
47     // nullptr.
48     T* desc() const { return desc_.get(); }
49     T* desc() { return desc_.get(); }
50   
51     // Use mut_desc() to access the underlying descriptor pointer
52     // if you intend to modify what it points to (e.g., using
53     // miopenSetFooDescriptor).  This will ensure that the descriptor
54     // is initialized.  Code in this file will use this function.
55     T* mut_desc() { init(); return desc_.get(); }
56   protected:
57     void init() {
58       if (desc_ == nullptr) {
59         T* raw_desc = nullptr;
60         MIOPEN_CHECK(ctor(&raw_desc));
61         desc_.reset(raw_desc);
62       }
63     }
64   private:
65     std::unique_ptr<T, DescriptorDeleter<T, dtor>> desc_;
66   };

108  struct TORCH_CUDA_CPP_API ConvolutionDescriptor
109      : public Descriptor<
110            miopenConvolutionDescriptor,
111            &miopenCreateConvolutionDescriptor,
112            &miopenDestroyConvolutionDescriptor> {
113    void set(miopenDataType_t dataType, miopenConvolutionMode_t c_mode,  int dim, int* pad, int* stride, int * upscale /* aka dilation */, int groups, bool benchmark, bool deterministic) {
114      MIOPEN_CHECK(miopenInitConvolutionNdDescriptor(mut_desc(), dim, pad, stride, upscale, c_mode));
115      MIOPEN_CHECK(miopenSetConvolutionGroupCount(mut_desc(), groups));
116      MIOPEN_CHECK(miopenSetConvolutionAttribute(mut_desc(), MIOPEN_CONVOLUTION_ATTRIB_DETERMINISTIC, deterministic ? 1 : 0));
117      if (benchmark) {
118        MIOPEN_CHECK(miopenSetConvolutionFindMode(mut_desc(), miopenConvolutionFindModeNormal));
119      }
120    }
121  };

156  union Constant
157  {
158    float f;
159    double d;
160    Constant(miopenDataType_t dataType, double value) {
161      if (dataType == miopenHalf || dataType == miopenFloat || dataType == miopenBFloat16) {
162        f = static_cast<float>(value);
163      } else {
164        d = value;
165      }
166    }
167  };
