42   def _sympy_to_scalar(e):
43       """Convert from a SymPy scalar to a Python scalar."""
44       if isinstance(e, Expr):
45           if e.is_Integer:
46               return int(e)
47           elif e.is_Float:
48               return float(e)
49           elif e.is_Rational:
50               return float(e)
51           elif e.is_Number or e.is_NumberSymbol or e == I:
52               return complex(e)
53       raise TypeError('Expected number, got: %r' % e)

56   def represent(expr, **options):
57       """Represent the quantum expression in the given basis. 
58    
59       In quantum mechanics abstract states and operators can be represented in 
60       various basis sets. Under this operation the follow transforms happen: 
61    
62       * Ket -> column vector or function 
63       * Bra -> row vector of function 
64       * Operator -> matrix or differential operator 
65    
66       This function is the top-level interface for this action. 
67    
68       This function walks the SymPy expression tree looking for ``QExpr`` 
69       instances that have a ``_represent`` method. This method is then called 
70       and the object is replaced by the representation returned by this method. 
71       By default, the ``_represent`` method will dispatch to other methods 
72       that handle the representation logic for a particular basis set. The 
73       naming convention for these methods is the following:: 
74    
75           def _represent_FooBasis(self, e, basis, **options) 
76    
77       This function will have the logic for representing instances of its class 
78       in the basis set having a class named ``FooBasis``. 
79    
80       Parameters 
81       ========== 
82    
83       expr  : Expr 
84           The expression to represent. 
85       basis : Operator, basis set 
86           An object that contains the information about the basis set. If an 
87           operator is used, the basis is assumed to be the orthonormal 
88           eigenvectors of that operator. In general though, the basis argument 
89           can be any object that contains the basis set information. 
90       options : dict 
91           Key/value pairs of options that are passed to the underlying method 
92           that finds the representation. These options can be used to 
93           control how the representation is done. For example, this is where 
94           the size of the basis set would be set. 
95    
96       Returns 
97       ======= 
98    
99       e : Expr 
100          The SymPy expression of the represented quantum expression. 
101   
102      Examples 
103      ======== 
104   
105      Here we subclass ``Operator`` and ``Ket`` to create the z-spin operator 
106      and its spin 1/2 up eigenstate. By defining the ``_represent_SzOp`` 
107      method, the ket can be represented in the z-spin basis. 
108   
109      >>> from sympy.physics.quantum import Operator, represent, Ket 
110      >>> from sympy import Matrix 
111   
112      >>> class SzUpKet(Ket): 
113      ...     def _represent_SzOp(self, basis, **options): 
114      ...         return Matrix([1,0]) 
115      ... 
116      >>> class SzOp(Operator): 
117      ...     pass 
118      ... 
119      >>> sz = SzOp('Sz') 
120      >>> up = SzUpKet('up') 
121      >>> represent(up, basis=sz) 
122      Matrix([ 
123      [1], 
124      [0]]) 
125   
126      Here we see an example of representations in a continuous 
127      basis. We see that the result of representing various combinations 
128      of cartesian position operators and kets give us continuous 
129      expressions involving DiracDelta functions. 
130   
131      >>> from sympy.physics.quantum.cartesian import XOp, XKet, XBra 
132      >>> X = XOp() 
133      >>> x = XKet() 
134      >>> y = XBra('y') 
135      >>> represent(X*x) 
136      x*DiracDelta(x - x_2) 
137      """
138  
139      format = options.get('format', 'sympy')
140      if format == 'numpy':
141          import numpy as np
142      if isinstance(expr, QExpr) and not isinstance(expr, OuterProduct):
143          options['replace_none'] = False
144          temp_basis = get_basis(expr, **options)
145          if temp_basis is not None:
146              options['basis'] = temp_basis
147          try:
148              return expr._represent(**options)
149          except NotImplementedError as strerr:
150              #If no _represent_FOO method exists, map to the
151              #appropriate basis state and try
152              #the other methods of representation
153              options['replace_none'] = True
154  
155              if isinstance(expr, (KetBase, BraBase)):
156                  try:
157                      return rep_innerproduct(expr, **options)
158                  except NotImplementedError:
159                      raise NotImplementedError(strerr)
160              elif isinstance(expr, Operator):
161                  try:
162                      return rep_expectation(expr, **options)
163                  except NotImplementedError:
164                      raise NotImplementedError(strerr)
165              else:
166                  raise NotImplementedError(strerr)
167      elif isinstance(expr, Add):
168          result = represent(expr.args[0], **options)
169          for args in expr.args[1:]:
170              # scipy.sparse doesn't support += so we use plain = here.
171              result = result + represent(args, **options)
172          return result
173      elif isinstance(expr, Pow):
174          base, exp = expr.as_base_exp()
175          if format in ('numpy', 'scipy.sparse'):
176              exp = _sympy_to_scalar(exp)
177          base = represent(base, **options)
178          # scipy.sparse doesn't support negative exponents
179          # and warns when inverting a matrix in csr format.
180          if format == 'scipy.sparse' and exp < 0:
181              from scipy.sparse.linalg import inv
182              exp = - exp
183              base = inv(base.tocsc()).tocsr()
184          if format == 'numpy':
185              return np.linalg.matrix_power(base, exp)
186          return base ** exp
187      elif isinstance(expr, TensorProduct):
188          new_args = [represent(arg, **options) for arg in expr.args]
189          return TensorProduct(*new_args)
190      elif isinstance(expr, Dagger):
191          return Dagger(represent(expr.args[0], **options))
192      elif isinstance(expr, Commutator):
193          A = expr.args[0]
194          B = expr.args[1]
195          return represent(Mul(A, B) - Mul(B, A), **options)
196      elif isinstance(expr, AntiCommutator):
197          A = expr.args[0]
198          B = expr.args[1]
199          return represent(Mul(A, B) + Mul(B, A), **options)
200      elif not isinstance(expr, (Mul, OuterProduct, InnerProduct)):
201          # We have removed special handling of inner products that used to be
202          # required (before automatic transforms).
203          # For numpy and scipy.sparse, we can only handle numerical prefactors.
204          if format in ('numpy', 'scipy.sparse'):
205              return _sympy_to_scalar(expr)
206          return expr
207  
208      if not isinstance(expr, (Mul, OuterProduct, InnerProduct)):
209          raise TypeError('Mul expected, got: %r' % expr)
210  
211      if "index" in options:
212          options["index"] += 1
213      else:
214          options["index"] = 1
215  
216      if "unities" not in options:
217          options["unities"] = []
218  
219      result = represent(expr.args[-1], **options)
220      last_arg = expr.args[-1]
221  
222      for arg in reversed(expr.args[:-1]):
223          if isinstance(last_arg, Operator):
224              options["index"] += 1
225              options["unities"].append(options["index"])
226          elif isinstance(last_arg, BraBase) and isinstance(arg, KetBase):
227              options["index"] += 1
228          elif isinstance(last_arg, KetBase) and isinstance(arg, Operator):
229              options["unities"].append(options["index"])
230          elif isinstance(last_arg, KetBase) and isinstance(arg, BraBase):
231              options["unities"].append(options["index"])
232  
233          next_arg = represent(arg, **options)
234          if format == 'numpy' and isinstance(next_arg, np.ndarray):
235              # Must use np.matmult to "matrix multiply" two np.ndarray
236              result = np.matmul(next_arg, result)
237          else:
238              result = next_arg*result
239          last_arg = arg
240  
241      # All three matrix formats create 1 by 1 matrices when inner products of
242      # vectors are taken. In these cases, we simply return a scalar.
243      result = flatten_scalar(result)
244  
245      result = integrate_result(expr, result, **options)
246  
247      return result

250  def rep_innerproduct(expr, **options):
251      """ 
252      Returns an innerproduct like representation (e.g. ``<x'|x>``) for the 
253      given state. 
254   
255      Attempts to calculate inner product with a bra from the specified 
256      basis. Should only be passed an instance of KetBase or BraBase 
257   
258      Parameters 
259      ========== 
260   
261      expr : KetBase or BraBase 
262          The expression to be represented 
263   
264      Examples 
265      ======== 
266   
267      >>> from sympy.physics.quantum.represent import rep_innerproduct 
268      >>> from sympy.physics.quantum.cartesian import XOp, XKet, PxOp, PxKet 
269      >>> rep_innerproduct(XKet()) 
270      DiracDelta(x - x_1) 
271      >>> rep_innerproduct(XKet(), basis=PxOp()) 
272      sqrt(2)*exp(-I*px_1*x/hbar)/(2*sqrt(hbar)*sqrt(pi)) 
273      >>> rep_innerproduct(PxKet(), basis=XOp()) 
274      sqrt(2)*exp(I*px*x_1/hbar)/(2*sqrt(hbar)*sqrt(pi)) 
275   
276      """
277  
278      if not isinstance(expr, (KetBase, BraBase)):
279          raise TypeError("expr passed is not a Bra or Ket")
280  
281      basis = get_basis(expr, **options)
282  
283      if not isinstance(basis, StateBase):
284          raise NotImplementedError("Can't form this representation!")
285  
286      if "index" not in options:
287          options["index"] = 1
288  
289      basis_kets = enumerate_states(basis, options["index"], 2)
290  
291      if isinstance(expr, BraBase):
292          bra = expr
293          ket = (basis_kets[1] if basis_kets[0].dual == expr else basis_kets[0])
294      else:
295          bra = (basis_kets[1].dual if basis_kets[0]
296                 == expr else basis_kets[0].dual)
297          ket = expr
298  
299      prod = InnerProduct(bra, ket)
300      result = prod.doit()
301  
302      format = options.get('format', 'sympy')
303      result = expr._format_represent(result, format)
304      return result

307  def rep_expectation(expr, **options):
308      """ 
309      Returns an ``<x'|A|x>`` type representation for the given operator. 
310   
311      Parameters 
312      ========== 
313   
314      expr : Operator 
315          Operator to be represented in the specified basis 
316   
317      Examples 
318      ======== 
319   
320      >>> from sympy.physics.quantum.cartesian import XOp, PxOp, PxKet 
321      >>> from sympy.physics.quantum.represent import rep_expectation 
322      >>> rep_expectation(XOp()) 
323      x_1*DiracDelta(x_1 - x_2) 
324      >>> rep_expectation(XOp(), basis=PxOp()) 
325      <px_2|*X*|px_1> 
326      >>> rep_expectation(XOp(), basis=PxKet()) 
327      <px_2|*X*|px_1> 
328   
329      """
330  
331      if "index" not in options:
332          options["index"] = 1
333  
334      if not isinstance(expr, Operator):
335          raise TypeError("The passed expression is not an operator")
336  
337      basis_state = get_basis(expr, **options)
338  
339      if basis_state is None or not isinstance(basis_state, StateBase):
340          raise NotImplementedError("Could not get basis kets for this operator")
341  
342      basis_kets = enumerate_states(basis_state, options["index"], 2)
343  
344      bra = basis_kets[1].dual
345      ket = basis_kets[0]
346  
347      result = qapply(bra*expr*ket)
348      return result

351  def integrate_result(orig_expr, result, **options):
352      """ 
353      Returns the result of integrating over any unities ``(|x><x|)`` in 
354      the given expression. Intended for integrating over the result of 
355      representations in continuous bases. 
356   
357      This function integrates over any unities that may have been 
358      inserted into the quantum expression and returns the result. 
359      It uses the interval of the Hilbert space of the basis state 
360      passed to it in order to figure out the limits of integration. 
361      The unities option must be 
362      specified for this to work. 
363   
364      Note: This is mostly used internally by represent(). Examples are 
365      given merely to show the use cases. 
366   
367      Parameters 
368      ========== 
369   
370      orig_expr : quantum expression 
371          The original expression which was to be represented 
372   
373      result: Expr 
374          The resulting representation that we wish to integrate over 
375   
376      Examples 
377      ======== 
378   
379      >>> from sympy import symbols, DiracDelta 
380      >>> from sympy.physics.quantum.represent import integrate_result 
381      >>> from sympy.physics.quantum.cartesian import XOp, XKet 
382      >>> x_ket = XKet() 
383      >>> X_op = XOp() 
384      >>> x, x_1, x_2 = symbols('x, x_1, x_2') 
385      >>> integrate_result(X_op*x_ket, x*DiracDelta(x-x_1)*DiracDelta(x_1-x_2)) 
386      x*DiracDelta(x - x_1)*DiracDelta(x_1 - x_2) 
387      >>> integrate_result(X_op*x_ket, x*DiracDelta(x-x_1)*DiracDelta(x_1-x_2), 
388      ...     unities=[1]) 
389      x*DiracDelta(x - x_2) 
390   
391      """
392      if not isinstance(result, Expr):
393          return result
394  
395      options['replace_none'] = True
396      if "basis" not in options:
397          arg = orig_expr.args[-1]
398          options["basis"] = get_basis(arg, **options)
399      elif not isinstance(options["basis"], StateBase):
400          options["basis"] = get_basis(orig_expr, **options)
401  
402      basis = options.pop("basis", None)
403  
404      if basis is None:
405          return result
406  
407      unities = options.pop("unities", [])
408  
409      if len(unities) == 0:
410          return result
411  
412      kets = enumerate_states(basis, unities)
413      coords = [k.label[0] for k in kets]
414  
415      for coord in coords:
416          if coord in result.free_symbols:
417              #TODO: Add support for sets of operators
418              basis_op = state_to_operators(basis)
419              start = basis_op.hilbert_space.interval.start
420              end = basis_op.hilbert_space.interval.end
421              result = integrate(result, (coord, start, end))
422  
423      return result

426  def get_basis(expr, *, basis=None, replace_none=True, **options):
427      """ 
428      Returns a basis state instance corresponding to the basis specified in 
429      options=s. If no basis is specified, the function tries to form a default 
430      basis state of the given expression. 
431   
432      There are three behaviors: 
433   
434      1. The basis specified in options is already an instance of StateBase. If 
435         this is the case, it is simply returned. If the class is specified but 
436         not an instance, a default instance is returned. 
437   
438      2. The basis specified is an operator or set of operators. If this 
439         is the case, the operator_to_state mapping method is used. 
440   
441      3. No basis is specified. If expr is a state, then a default instance of 
442         its class is returned.  If expr is an operator, then it is mapped to the 
443         corresponding state.  If it is neither, then we cannot obtain the basis 
444         state. 
445   
446      If the basis cannot be mapped, then it is not changed. 
447   
448      This will be called from within represent, and represent will 
449      only pass QExpr's. 
450   
451      TODO (?): Support for Muls and other types of expressions? 
452   
453      Parameters 
454      ========== 
455   
456      expr : Operator or StateBase 
457          Expression whose basis is sought 
458   
459      Examples 
460      ======== 
461   
462      >>> from sympy.physics.quantum.represent import get_basis 
463      >>> from sympy.physics.quantum.cartesian import XOp, XKet, PxOp, PxKet 
464      >>> x = XKet() 
465      >>> X = XOp() 
466      >>> get_basis(x) 
467      |x> 
468      >>> get_basis(X) 
469      |x> 
470      >>> get_basis(x, basis=PxOp()) 
471      |px> 
472      >>> get_basis(x, basis=PxKet) 
473      |px> 
474   
475      """
476  
477      if basis is None and not replace_none:
478          return None
479  
480      if basis is None:
481          if isinstance(expr, KetBase):
482              return _make_default(expr.__class__)
483          elif isinstance(expr, BraBase):
484              return _make_default(expr.dual_class())
485          elif isinstance(expr, Operator):
486              state_inst = operators_to_state(expr)
487              return (state_inst if state_inst is not None else None)
488          else:
489              return None
490      elif (isinstance(basis, Operator) or
491            (not isinstance(basis, StateBase) and issubclass(basis, Operator))):
492          state = operators_to_state(basis)
493          if state is None:
494              return None
495          elif isinstance(state, StateBase):
496              return state
497          else:
498              return _make_default(state)
499      elif isinstance(basis, StateBase):
500          return basis
501      elif issubclass(basis, StateBase):
502          return _make_default(basis)
503      else:
504          return None

507  def _make_default(expr):
508      # XXX: Catching TypeError like this is a bad way of distinguishing
509      # instances from classes. The logic using this function should be
510      # rewritten somehow.
511      try:
512          expr = expr()
513      except TypeError:
514          return expr
515  
516      return expr

519  def enumerate_states(*args, **options):
520      """ 
521      Returns instances of the given state with dummy indices appended 
522   
523      Operates in two different modes: 
524   
525      1. Two arguments are passed to it. The first is the base state which is to 
526         be indexed, and the second argument is a list of indices to append. 
527   
528      2. Three arguments are passed. The first is again the base state to be 
529         indexed. The second is the start index for counting.  The final argument 
530         is the number of kets you wish to receive. 
531   
532      Tries to call state._enumerate_state. If this fails, returns an empty list 
533   
534      Parameters 
535      ========== 
536   
537      args : list 
538          See list of operation modes above for explanation 
539   
540      Examples 
541      ======== 
542   
543      >>> from sympy.physics.quantum.cartesian import XBra, XKet 
544      >>> from sympy.physics.quantum.represent import enumerate_states 
545      >>> test = XKet('foo') 
546      >>> enumerate_states(test, 1, 3) 
547      [|foo_1>, |foo_2>, |foo_3>] 
548      >>> test2 = XBra('bar') 
549      >>> enumerate_states(test2, [4, 5, 10]) 
550      [<bar_4|, <bar_5|, <bar_10|] 
551   
552      """
553  
554      state = args[0]
555  
556      if len(args) not in (2, 3):
557          raise NotImplementedError("Wrong number of arguments!")
558  
559      if not isinstance(state, StateBase):
560          raise TypeError("First argument is not a state!")
561  
562      if len(args) == 3:
563          num_states = args[2]
564          options['start_index'] = args[1]
565      else:
566          num_states = len(args[1])
567          options['index_list'] = args[1]
568  
569      try:
570          ret = state._enumerate_state(num_states, **options)
571      except NotImplementedError:
572          ret = []
573  
574      return ret
