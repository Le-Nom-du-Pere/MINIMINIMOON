51   def rv(symbol, cls, *args, **kwargs):
52       args = list(map(sympify, args))
53       dist = cls(*args)
54       if kwargs.pop('check', True):
55           dist.check(*args)
56       pspace = SingleDiscretePSpace(symbol, dist)
57       if any(is_random(arg) for arg in args):
58           from sympy.stats.compound_rv import CompoundPSpace, CompoundDistribution
59           pspace = CompoundPSpace(symbol, CompoundDistribution(dist))
60       return pspace.value

63   class DiscreteDistributionHandmade(SingleDiscreteDistribution):
64       _argnames = ('pdf',)
65   
66       def __new__(cls, pdf, set=S.Integers):
67           return Basic.__new__(cls, pdf, set)
68   
69       @property
70       def set(self):
71           return self.args[1]
72   
73       @staticmethod
74       def check(pdf, set):
75           x = Dummy('x')
76           val = Sum(pdf(x), (x, set._inf, set._sup)).doit()
77           _value_check(Eq(val, 1) != S.false, "The pdf is incorrect on the given set.")

81   def DiscreteRV(symbol, density, set=S.Integers, **kwargs):
82       """ 
83       Create a Discrete Random Variable given the following: 
84    
85       Parameters 
86       ========== 
87    
88       symbol : Symbol 
89           Represents name of the random variable. 
90       density : Expression containing symbol 
91           Represents probability density function. 
92       set : set 
93           Represents the region where the pdf is valid, by default is real line. 
94       check : bool 
95           If True, it will check whether the given density 
96           integrates to 1 over the given set. If False, it 
97           will not perform this check. Default is False. 
98    
99       Examples 
100      ======== 
101   
102      >>> from sympy.stats import DiscreteRV, P, E 
103      >>> from sympy import Rational, Symbol 
104      >>> x = Symbol('x') 
105      >>> n = 10 
106      >>> density = Rational(1, 10) 
107      >>> X = DiscreteRV(x, density, set=set(range(n))) 
108      >>> E(X) 
109      9/2 
110      >>> P(X>3) 
111      3/5 
112   
113      Returns 
114      ======= 
115   
116      RandomSymbol 
117   
118      """
119      set = sympify(set)
120      pdf = Piecewise((density, set.as_relational(symbol)), (0, True))
121      pdf = Lambda(symbol, pdf)
122      # have a default of False while `rv` should have a default of True
123      kwargs['check'] = kwargs.pop('check', False)
124      return rv(symbol.name, DiscreteDistributionHandmade, pdf, set, **kwargs)

130  class FlorySchulzDistribution(SingleDiscreteDistribution):
131      _argnames = ('a',)
132      set = S.Naturals
133  
134      @staticmethod
135      def check(a):
136          _value_check((0 < a, a < 1), "a must be between 0 and 1")
137  
138      def pdf(self, k):
139          a = self.a
140          return (a**2 * k * (1 - a)**(k - 1))
141  
142      def _characteristic_function(self, t):
143          a = self.a
144          return a**2*exp(I*t)/((1 + (a - 1)*exp(I*t))**2)
145  
146      def _moment_generating_function(self, t):
147          a = self.a
148          return a**2*exp(t)/((1 + (a - 1)*exp(t))**2)

201  class GeometricDistribution(SingleDiscreteDistribution):
202      _argnames = ('p',)
203      set = S.Naturals
204  
205      @staticmethod
206      def check(p):
207          _value_check((0 < p, p <= 1), "p must be between 0 and 1")
208  
209      def pdf(self, k):
210          return (1 - self.p)**(k - 1) * self.p
211  
212      def _characteristic_function(self, t):
213          p = self.p
214          return p * exp(I*t) / (1 - (1 - p)*exp(I*t))
215  
216      def _moment_generating_function(self, t):
217          p = self.p
218          return p * exp(t) / (1 - (1 - p) * exp(t))

277  class HermiteDistribution(SingleDiscreteDistribution):
278      _argnames = ('a1', 'a2')
279      set = S.Naturals0
280  
281      @staticmethod
282      def check(a1, a2):
283          _value_check(a1.is_nonnegative, 'Parameter a1 must be >= 0.')
284          _value_check(a2.is_nonnegative, 'Parameter a2 must be >= 0.')
285  
286      def pdf(self, k):
287          a1, a2 = self.a1, self.a2
288          term1 = exp(-(a1 + a2))
289          j = Dummy("j", integer=True)
290          num = a1**(k - 2*j) * a2**j
291          den = factorial(k - 2*j) * factorial(j)
292          return term1 * Sum(num/den, (j, 0, k//2)).doit()
293  
294      def _moment_generating_function(self, t):
295          a1, a2 = self.a1, self.a2
296          term1 = a1 * (exp(t) - 1)
297          term2 = a2 * (exp(2*t) - 1)
298          return exp(term1 + term2)
299  
300      def _characteristic_function(self, t):
301          a1, a2 = self.a1, self.a2
302          term1 = a1 * (exp(I*t) - 1)
303          term2 = a2 * (exp(2*I*t) - 1)
304          return exp(term1 + term2)

364  class LogarithmicDistribution(SingleDiscreteDistribution):
365      _argnames = ('p',)
366  
367      set = S.Naturals
368  
369      @staticmethod
370      def check(p):
371          _value_check((p > 0, p < 1), "p should be between 0 and 1")
372  
373      def pdf(self, k):
374          p = self.p
375          return (-1) * p**k / (k * log(1 - p))
376  
377      def _characteristic_function(self, t):
378          p = self.p
379          return log(1 - p * exp(I*t)) / log(1 - p)
380  
381      def _moment_generating_function(self, t):
382          p = self.p
383          return log(1 - p * exp(t)) / log(1 - p)

441  class NegativeBinomialDistribution(SingleDiscreteDistribution):
442      _argnames = ('r', 'p')
443      set = S.Naturals0
444  
445      @staticmethod
446      def check(r, p):
447          _value_check(r > 0, 'r should be positive')
448          _value_check((p > 0, p < 1), 'p should be between 0 and 1')
449  
450      def pdf(self, k):
451          r = self.r
452          p = self.p
453  
454          return binomial(k + r - 1, k) * (1 - p)**k * p**r
455  
456      def _characteristic_function(self, t):
457          r = self.r
458          p = self.p
459  
460          return (p / (1 - (1 - p) * exp(I*t)))**r
461  
462      def _moment_generating_function(self, t):
463          r = self.r
464          p = self.p
465  
466          return (p / (1 - (1 - p) * exp(t)))**r

527  class PoissonDistribution(SingleDiscreteDistribution):
528      _argnames = ('lamda',)
529  
530      set = S.Naturals0
531  
532      @staticmethod
533      def check(lamda):
534          _value_check(lamda > 0, "Lambda must be positive")
535  
536      def pdf(self, k):
537          return self.lamda**k / factorial(k) * exp(-self.lamda)
538  
539      def _characteristic_function(self, t):
540          return exp(self.lamda * (exp(I*t) - 1))
541  
542      def _moment_generating_function(self, t):
543          return exp(self.lamda * (exp(t) - 1))
544  
545      def expectation(self, expr, var, evaluate=True, **kwargs):
546          if evaluate:
547              if expr == var:
548                  return self.lamda
549              if (
550                  isinstance(expr, FallingFactorial)
551                  and expr.args[1].is_integer
552                  and expr.args[1].is_positive
553                  and expr.args[0] == var
554              ):
555                  return self.lamda ** expr.args[1]
556          return super().expectation(expr, var, evaluate, **kwargs)

614  class SkellamDistribution(SingleDiscreteDistribution):
615      _argnames = ('mu1', 'mu2')
616      set = S.Integers
617  
618      @staticmethod
619      def check(mu1, mu2):
620          _value_check(mu1 >= 0, 'Parameter mu1 must be >= 0')
621          _value_check(mu2 >= 0, 'Parameter mu2 must be >= 0')
622  
623      def pdf(self, k):
624          (mu1, mu2) = (self.mu1, self.mu2)
625          term1 = exp(-(mu1 + mu2)) * (mu1 / mu2) ** (k / 2)
626          term2 = besseli(k, 2 * sqrt(mu1 * mu2))
627          return term1 * term2
628  
629      def _cdf(self, x):
630          raise NotImplementedError(
631              "Skellam doesn't have closed form for the CDF.")
632  
633      def _characteristic_function(self, t):
634          (mu1, mu2) = (self.mu1, self.mu2)
635          return exp(-(mu1 + mu2) + mu1 * exp(I * t) + mu2 * exp(-I * t))
636  
637      def _moment_generating_function(self, t):
638          (mu1, mu2) = (self.mu1, self.mu2)
639          return exp(-(mu1 + mu2) + mu1 * exp(t) + mu2 * exp(-t))

642  def Skellam(name, mu1, mu2):
643      r""" 
644      Create a discrete random variable with a Skellam distribution. 
645   
646      Explanation 
647      =========== 
648   
649      The Skellam is the distribution of the difference N1 - N2 
650      of two statistically independent random variables N1 and N2 
651      each Poisson-distributed with respective expected values mu1 and mu2. 
652   
653      The density of the Skellam distribution is given by 
654   
655      .. math:: 
656          f(k) := e^{-(\mu_1+\mu_2)}(\frac{\mu_1}{\mu_2})^{k/2}I_k(2\sqrt{\mu_1\mu_2}) 
657   
658      Parameters 
659      ========== 
660   
661      mu1 : A non-negative value 
662      mu2 : A non-negative value 
663   
664      Returns 
665      ======= 
666   
667      RandomSymbol 
668   
669      Examples 
670      ======== 
671   
672      >>> from sympy.stats import Skellam, density, E, variance 
673      >>> from sympy import Symbol, pprint 
674   
675      >>> z = Symbol("z", integer=True) 
676      >>> mu1 = Symbol("mu1", positive=True) 
677      >>> mu2 = Symbol("mu2", positive=True) 
678      >>> X = Skellam("x", mu1, mu2) 
679   
680      >>> pprint(density(X)(z), use_unicode=False) 
681           z 
682           - 
683           2 
684      /mu1\   -mu1 - mu2        /       _____   _____\ 
685      |---| *e          *besseli\z, 2*\/ mu1 *\/ mu2 / 
686      \mu2/ 
687      >>> E(X) 
688      mu1 - mu2 
689      >>> variance(X).expand() 
690      mu1 + mu2 
691   
692      References 
693      ========== 
694   
695      .. [1] https://en.wikipedia.org/wiki/Skellam_distribution 
696   
697      """
698      return rv(name, SkellamDistribution, mu1, mu2)

704  class YuleSimonDistribution(SingleDiscreteDistribution):
705      _argnames = ('rho',)
706      set = S.Naturals
707  
708      @staticmethod
709      def check(rho):
710          _value_check(rho > 0, 'rho should be positive')
711  
712      def pdf(self, k):
713          rho = self.rho
714          return rho * beta(k, rho + 1)
715  
716      def _cdf(self, x):
717          return Piecewise((1 - floor(x) * beta(floor(x), self.rho + 1), x >= 1), (0, True))
718  
719      def _characteristic_function(self, t):
720          rho = self.rho
721          return rho * hyper((1, 1), (rho + 2,), exp(I*t)) * exp(I*t) / (rho + 1)
722  
723      def _moment_generating_function(self, t):
724          rho = self.rho
725          return rho * hyper((1, 1), (rho + 2,), exp(t)) * exp(t) / (rho + 1)

782  class ZetaDistribution(SingleDiscreteDistribution):
783      _argnames = ('s',)
784      set = S.Naturals
785  
786      @staticmethod
787      def check(s):
788          _value_check(s > 1, 's should be greater than 1')
789  
790      def pdf(self, k):
791          s = self.s
792          return 1 / (k**s * zeta(s))
793  
794      def _characteristic_function(self, t):
795          return polylog(self.s, exp(I*t)) / zeta(self.s)
796  
797      def _moment_generating_function(self, t):
798          return polylog(self.s, exp(t)) / zeta(self.s)
