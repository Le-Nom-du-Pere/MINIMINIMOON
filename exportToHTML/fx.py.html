15   @substitute_in_graph(_fx_map_aggregate, can_constant_fold_through=True)
16   def map_aggregate(a: Any, fn: Callable[[Any], Any]) -> Any:
17       result: Any
18       if isinstance(a, tuple):
19           it = (map_aggregate(elem, fn) for elem in a)
20           # Support NamedTuple (if it has `_fields`) by repacking into original type.
21           result = type(a)(*it) if hasattr(a, "_fields") else tuple(it)
22       elif isinstance(a, list):
23           result = immutable_list([map_aggregate(elem, fn) for elem in a])
24       elif isinstance(a, dict):
25           result = immutable_dict([(k, map_aggregate(v, fn)) for k, v in a.items()])
26       elif isinstance(a, slice):
27           result = slice(
28               map_aggregate(a.start, fn),
29               map_aggregate(a.stop, fn),
30               map_aggregate(a.step, fn),
31           )
32       else:
33           result = fn(a)
34       return result
