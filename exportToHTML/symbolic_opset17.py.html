38   @_onnx_symbolic("aten::layer_norm")
39   @symbolic_helper.parse_args("v", "is", "v", "v", "f", "none")
40   def layer_norm(
41       g: jit_utils.GraphContext,
42       input: _C.Value,
43       normalized_shape: Sequence[int],
44       weight: _C.Value,
45       bias: _C.Value,
46       eps: float,
47       cudnn_enable: bool,
48   ):
49       # normalized_shape: input shape from an expected input of size
50       # axis: The first normalization dimension.
51       # layer_norm normalizes on the last D dimensions,
52       # where D is the size of normalized_shape
53       axis = -len(normalized_shape)
54       scalar_type = _type_utils.JitScalarType.from_value(
55           input, _type_utils.JitScalarType.FLOAT
56       )
57       dtype = scalar_type.dtype()
58       if symbolic_helper._is_none(weight):
59           weight_value = torch.ones(normalized_shape, dtype=dtype)
60           weight = g.op("Constant", value_t=weight_value)
61       if symbolic_helper._is_none(bias):
62           bias_value = torch.zeros(normalized_shape, dtype=dtype)
63           bias = g.op("Constant", value_t=bias_value)
64       return g.op(
65           "LayerNormalization",
66           input,
67           weight,
68           bias,
69           epsilon_f=eps,
70           axis_i=axis,
71       )

74   @_onnx_symbolic("quantized::layer_norm")
75   def quantized_layer_norm(
76       g: jit_utils.GraphContext,
77       x,
78       normalized_shape,
79       weight,
80       bias,
81       eps,
82       op_scale,
83       op_zero_point,
84   ):
85       x, _, _, _ = symbolic_helper.dequantize_helper(g, x)
86   
87       output = layer_norm(g, x, normalized_shape, weight, bias, eps, False)
88   
89       return symbolic_helper.quantize_helper(g, output, op_scale, op_zero_point)

92   def _compute_edge_sizes(n_fft, window_size):
93       """Helper function to compute the sizes of the edges (left and right) 
94       of a given window centered within an FFT size."""
95       left = (n_fft - window_size) // 2
96       right = n_fft - left - window_size
97       return left, right

100  @_onnx_symbolic("aten::stft")
101  @symbolic_helper.parse_args("v", "i", "i", "i", "v", "b", "b", "b", "b")
102  def stft(
103      g: jit_utils.GraphContext,
104      input: _C.Value,
105      n_fft: int,
106      hop_length: Optional[int] = None,
107      win_length: Optional[int] = None,
108      window: Optional[_C.Value] = None,
109      normalized: bool = False,
110      onesided: Optional[bool] = True,
111      return_complex: Optional[bool] = False,
112      align_to_window: Optional[bool] = None,
113  ) -> _C.Value:
114      """Associates `torch.stft` with the `STFT` ONNX operator. 
115      Note that torch.stft calls _VF.stft, without centering or padding options. 
116      Hence, this function does not contain these two arguments. 
117      See torch.stft source code for more info. 
118   
119      Args: 
120          g: Graph to write the ONNX representation into 
121          input: Input tensor for the transformation 
122          n_fft: FFT size 
123          hop_length: Size of the hop. Defaults to `floot(n_fft // 4)` 
124          win_length: Size of the analysis window. Defaults to `n_fft` 
125          window: Analysis window. Defaults to a window of all ones 
126          normalized: Whether to return a normalized STFT 
127          onesided: Whether to return only half (+1) of the results, given the 
128              symmetry of the STFT 
129          return_complex: Whether to return the complex value (Note: Must be 
130              `False` or `None`) 
131   
132      Returns: 
133          op: Operator for torch.stft associated with STFT (ONNX) 
134      """
135      # Checks
136      if return_complex:
137          raise errors.SymbolicValueError(
138              msg="STFT does not currently support complex types", value=input
139          )
140  
141      if align_to_window is not None:
142          raise errors.SymbolicValueError(
143              msg="STFT does not currently support the align_to_window option",
144              value=input,
145          )  # TODO(#145944): add compatibility with align_to_window option.
146  
147      # Get STFT sizes
148      frame_step_value = hop_length if hop_length is not None else n_fft // 4
149      frame_step_const = g.op(
150          "Constant", value_t=torch.tensor(frame_step_value, dtype=torch.int64)
151      )
152      frame_length_const = g.op(
153          "Constant", value_t=torch.tensor(n_fft, dtype=torch.int64)
154      )
155  
156      # Pre-process input if needed
157      signal = input
158      signal_rank = symbolic_helper._get_tensor_rank(signal)
159      if signal_rank == 1:
160          # Add batch dimension
161          signal = g.op(
162              "Unsqueeze",
163              signal,
164              g.op("Constant", value_t=torch.tensor([0], dtype=torch.int64)),
165          )
166      elif signal_rank is None or signal_rank > 2:
167          raise errors.SymbolicValueError(
168              msg="STFT can only take inputs of 1 [signal] or 2 [batch, signal] dimensions. "
169              f"Current rank of signal is {signal_rank}, please reduce it.",
170              value=input,
171          )
172  
173      # Get window and make sure it's the same size as `win_length` or `n_fft`
174      n_win = symbolic_helper._get_tensor_dim_size(window, dim=0)
175      if n_win is not None:
176          win_length_default = win_length if win_length else n_fft
177          assert n_win == win_length_default, (
178              "Analysis window size must equal `win_length` or `n_fft`. "
179              f"Please, set `win_length` or `n_fft` to match `window` size ({n_win})",
180          )
181  
182          # Center window around zeros if needed (required by ONNX's STFT)
183          if n_win < n_fft:
184              left, right = _compute_edge_sizes(n_fft, n_win)
185              left_win = g.op("Constant", value_t=torch.zeros(left))
186              right_win = g.op("Constant", value_t=torch.zeros(right))
187              window = g.op("Concat", left_win, window, right_win, axis_i=0)
188  
189      # Create window, if needed
190      if symbolic_helper._is_none(window):
191          if win_length:
192              if win_length > n_fft:
193                  raise errors.SymbolicValueError(
194                      msg="The analysis window can't be longer than the size of the FFT. "
195                      f"Please set `win_length` ({win_length}) to `n_fft` ({n_fft}) or less.",
196                      value=input,
197                  )
198  
199              # Center window, if needed
200              left, right = _compute_edge_sizes(n_fft, win_length)
201              torch_window = torch.hstack(
202                  (torch.zeros(left), torch.ones(win_length), torch.zeros(right))
203              )
204          else:
205              # Rectangle window
206              torch_window = torch.ones(n_fft)
207          assert torch_window.shape[0] == n_fft
208          window = g.op("Constant", value_t=torch_window)
209      window = g.op(
210          "Cast", window, to_i=_type_utils.JitScalarType.from_value(signal).onnx_type()
211      )
212  
213      # Run STFT
214      result = g.op(
215          "STFT",
216          signal,
217          frame_step_const,
218          window,
219          frame_length_const,
220          onesided_i=1 if onesided is None or onesided else 0,
221      )
222  
223      # Transpose to mimic torch.stft's behavior
224      result = g.op("Transpose", result, perm_i=[0, 2, 1, 3])
225  
226      # Remove batch dimension, if needed
227      if signal_rank == 1:
228          result = g.op(
229              "Squeeze",
230              result,
231              g.op("Constant", value_t=torch.tensor([0], dtype=torch.int64)),
232          )
233  
234      # Normalize, if needed
235      if normalized:
236          sqrt_nfft = torch.sqrt(torch.tensor(n_fft, dtype=signal.type().dtype()))
237          result = g.op("Div", result, g.op("Constant", value_t=sqrt_nfft))
238  
239      return result
