40   c10::SmallVector<std::string> get_extra_args_typenames(const c10::SmallVector<at::Scalar>& extra_args) {
41     c10::SmallVector<std::string> args_typenames(extra_args.size());
42     for (const auto i : c10::irange(extra_args.size())) {
43       args_typenames[i] = at::cuda::jit::typeName(extra_args[i].type());
44     }
45     return args_typenames;

48   int can_vectorize_up_to(at::ScalarType type, char* pointer) {
49     switch(type) {
50   #define DEFINE_CASE(ctype, scalartype)                                   \
51       case ScalarType::scalartype : return memory::can_vectorize_up_to<ctype>(pointer);
52   
53       AT_FORALL_SCALAR_TYPES_WITH_COMPLEX(DEFINE_CASE)
54   #undef DEFINE_CASE
55   
56       default: TORCH_INTERNAL_ASSERT(false, "Unrecognized ScalarType: ", type);
57     }
58   }

60   // jitted version of the above
61   // See Note [Jiterator], this relies on the assumptions enumerated there
62   int jitted_can_vectorize_up_to(const TensorIteratorBase& iter) {
63     const at::ScalarType common_dtype = iter.common_dtype();
64     const at::ScalarType result_dtype = common_dtype;
65   
66     // Deals with output
67     int result = can_vectorize_up_to(result_dtype, static_cast<char*>(iter.data_ptr(0)));
68   
69     // Incorporates input(s)
70     for (auto i = 1; i < iter.ntensors(); ++i) {
71       result = std::min<int>(result, can_vectorize_up_to(common_dtype, static_cast<char*>(iter.data_ptr(i))));
72     }
73   
74     return result;
75   }

77   template<bool IS_INPUT, int N>
78   static std::unique_ptr<OffsetCalculator<N>> make_unique_offset_calculator(
79             const TensorIteratorBase& iter) {
80     // array size can not be 0, this happens when N == 0
81     constexpr int array_size = std::max<int>(N, 1);
82     TORCH_INTERNAL_ASSERT(N == (IS_INPUT ? iter.ninputs() : iter.noutputs()));
83   
84     std::array<const int64_t*, array_size> strides;
85     int64_t element_sizes[array_size];
86     for (int i = 0; i < N; i++) {
87       int index = IS_INPUT ? i + iter.noutputs() : i;
88       strides[i] = iter.strides(index).data();
89       element_sizes[i] = iter.element_size(index);
90     }
91     return std::make_unique<OffsetCalculator<N>>(iter.ndim(), iter.shape().data(), strides.data(), element_sizes);
92   }

94   template <bool IS_INPUT>
95   struct OffsetCalculatorVariant {
96   #define DEFINE_CASE(index) std::unique_ptr<OffsetCalculator<index>>
97     using OffsetCalculatorTypes = std::variant<
98       AT_FOR_8_CASES_WITH_COMMA(DEFINE_CASE)
99     >;
100  #undef DEFINE_CASE
101  
102    OffsetCalculatorVariant(const TensorIteratorBase& iter) {
103      int num = IS_INPUT ? iter.ninputs() : iter.noutputs();
104  
105      switch(num) {
106  #define DEFINE_CASE(index)        \
107        case index : v = make_unique_offset_calculator<IS_INPUT, index>(iter); break;
108  
109        AT_FOR_8_CASES(DEFINE_CASE)
110  #undef DEFINE_CASE
111        default:
112          TORCH_CHECK(false, "OffsetCalculatorVariant is not implemented for num_tensor = ", num);
113      }
114    }
115  
116    void* data_ptr() {
117      return std::visit([](auto & v){ return static_cast<void*>(v.get()); }, v);
118    }
119  
120   private:
121    OffsetCalculatorTypes v{};
122  };

124  struct ArrayVariant {
125  // works for up to 8 input + 8 outputs
126  #define DEFINE_CASE(index) std::array<char*, index>, std::array<char*, index+8>
127    using ArrayTypes = std::variant<
128      AT_FOR_8_CASES_WITH_COMMA(DEFINE_CASE)
129    >;
130  #undef DEFINE_CASE
131  
132    ArrayVariant(const TensorIteratorBase& iter) {
133      int ntensors = iter.ntensors();
134      switch(ntensors) {
135  #define DEFINE_CASE(index)                                            \
136        case index: array = std::array<char*, index>{}; break;   \
137        case index+8: array = std::array<char*, index+8>{}; break;
138  
139        AT_FOR_8_CASES(DEFINE_CASE)
140  #undef DEFINE_CASE
141  
142        default:
143          TORCH_CHECK(false, "ArrayVariant is not implemented for ntensors = ", ntensors);
144      }
145  
146      std::visit([&](auto& a) {
147        for (auto i = 0; i < ntensors; ++i) {
148          a[i] = (char*)iter.data_ptr(i);
149        }
150      }, array);
151    }
152  
153    void* data_ptr() {
154      return std::visit([](auto & a){ return static_cast<void*>(&a); }, array);
155    }
156  
157  private:
158    ArrayTypes array;
159  };
