28   class DenseMatrix(RepMatrix):
29       """Matrix implementation based on DomainMatrix as the internal representation"""
30   
31       #
32       # DenseMatrix is a superclass for both MutableDenseMatrix and
33       # ImmutableDenseMatrix. Methods shared by both classes but not for the
34       # Sparse classes should be implemented here.
35       #
36   
37       is_MatrixExpr: bool = False
38   
39       _op_priority = 10.01
40       _class_priority = 4
41   
42       @property
43       def _mat(self):
44           sympy_deprecation_warning(
45               """ 
46               The private _mat attribute of Matrix is deprecated. Use the 
47               .flat() method instead. 
48               """,
49               deprecated_since_version="1.9",
50               active_deprecations_target="deprecated-private-matrix-attributes"
51           )
52   
53           return self.flat()
54   
55       def _eval_inverse(self, **kwargs):
56           return self.inv(method=kwargs.get('method', 'GE'),
57                           iszerofunc=kwargs.get('iszerofunc', _iszero),
58                           try_block_diag=kwargs.get('try_block_diag', False))
59   
60       def as_immutable(self):
61           """Returns an Immutable version of this Matrix 
62           """
63           from .immutable import ImmutableDenseMatrix as cls
64           return cls._fromrep(self._rep.copy())
65   
66       def as_mutable(self):
67           """Returns a mutable version of this matrix 
68    
69           Examples 
70           ======== 
71    
72           >>> from sympy import ImmutableMatrix 
73           >>> X = ImmutableMatrix([[1, 2], [3, 4]]) 
74           >>> Y = X.as_mutable() 
75           >>> Y[1, 1] = 5 # Can set values in Y 
76           >>> Y 
77           Matrix([ 
78           [1, 2], 
79           [3, 5]]) 
80           """
81           return Matrix(self)

83       def cholesky(self, hermitian=True):
84           return _cholesky(self, hermitian=hermitian)
85   
86       def LDLdecomposition(self, hermitian=True):
87           return _LDLdecomposition(self, hermitian=hermitian)
88   
89       def lower_triangular_solve(self, rhs):
90           return _lower_triangular_solve(self, rhs)
91   
92       def upper_triangular_solve(self, rhs):
93           return _upper_triangular_solve(self, rhs)
94

96       LDLdecomposition.__doc__       = _LDLdecomposition.__doc__
97       lower_triangular_solve.__doc__ = _lower_triangular_solve.__doc__

101  def _force_mutable(x):
102      """Return a matrix as a Matrix, otherwise return x."""
103      if getattr(x, 'is_Matrix', False):
104          return x.as_mutable()
105      elif isinstance(x, Basic):
106          return x
107      elif hasattr(x, '__array__'):
108          a = x.__array__()
109          if len(a.shape) == 0:
110              return sympify(a)
111          return Matrix(x)
112      return x

118          """Applies simplify to the elements of a matrix in place. 
119   
120          This is a shortcut for M.applyfunc(lambda x: simplify(x, ratio, measure)) 
121   
122          See Also 
123          ======== 
124

128          for (i, j), element in self.todok().items():

140  def list2numpy(l, dtype=object):  # pragma: no cover
141      """Converts Python list of SymPy expressions to a NumPy array. 
142   
143      See Also 
144      ======== 
145   
146      matrix2numpy 
147      """
148      from numpy import empty
149      a = empty(len(l), dtype)
150      for i, s in enumerate(l):
151          a[i] = s
152      return a

155  def matrix2numpy(m, dtype=object):  # pragma: no cover
156      """Converts SymPy's matrix to a NumPy array. 
157   
158      See Also 
159      ======== 
160   
161      list2numpy 
162      """
163      from numpy import empty
164      a = empty(m.shape, dtype)
165      for i in range(m.rows):
166          for j in range(m.cols):
167              a[i, j] = m[i, j]
168      return a

177  def rot_givens(i, j, theta, dim=3):
178      r"""Returns a a Givens rotation matrix, a a rotation in the 
179      plane spanned by two coordinates axes. 
180   
181      Explanation 
182      =========== 
183   
184      The Givens rotation corresponds to a generalization of rotation 
185      matrices to any number of dimensions, given by: 
186   
187      .. math:: 
188          G(i, j, \theta) = 
189              \begin{bmatrix} 
190                  1   & \cdots &    0   & \cdots &    0   & \cdots &    0   \\ 
191                  \vdots & \ddots & \vdots &        & \vdots &        & \vdots \\ 
192                  0   & \cdots &    c   & \cdots &   -s   & \cdots &    0   \\ 
193                  \vdots &        & \vdots & \ddots & \vdots &        & \vdots \\ 
194                  0   & \cdots &    s   & \cdots &    c   & \cdots &    0   \\ 
195                  \vdots &        & \vdots &        & \vdots & \ddots & \vdots \\ 
196                  0   & \cdots &    0   & \cdots &    0   & \cdots &    1 
197              \end{bmatrix} 
198   
199      Where $c = \cos(\theta)$ and $s = \sin(\theta)$ appear at the intersections 
200      ``i``\th and ``j``\th rows and columns. 
201   
202      For fixed ``i > j``\, the non-zero elements of a Givens matrix are 
203      given by: 
204   
205      - $g_{kk} = 1$ for $k \ne i,\,j$ 
206      - $g_{kk} = c$ for $k = i,\,j$ 
207      - $g_{ji} = -g_{ij} = -s$ 
208   
209      Parameters 
210      ========== 
211   
212      i : int between ``0`` and ``dim - 1`` 
213          Represents first axis 
214      j : int between ``0`` and ``dim - 1`` 
215          Represents second axis 
216      dim : int bigger than 1 
217          Number of dimensions. Defaults to 3. 
218   
219      Examples 
220      ======== 
221   
222      >>> from sympy import pi, rot_givens 
223   
224      A counterclockwise rotation of pi/3 (60 degrees) around 
225      the third axis (z-axis): 
226   
227      >>> rot_givens(1, 0, pi/3) 
228      Matrix([ 
229      [      1/2, -sqrt(3)/2, 0], 
230      [sqrt(3)/2,        1/2, 0], 
231      [        0,          0, 1]]) 
232   
233      If we rotate by pi/2 (90 degrees): 
234   
235      >>> rot_givens(1, 0, pi/2) 
236      Matrix([ 
237      [0, -1, 0], 
238      [1,  0, 0], 
239      [0,  0, 1]]) 
240   
241      This can be generalized to any number 
242      of dimensions: 
243   
244      >>> rot_givens(1, 0, pi/2, dim=4) 
245      Matrix([ 
246      [0, -1, 0, 0], 
247      [1,  0, 0, 0], 
248      [0,  0, 1, 0], 
249      [0,  0, 0, 1]]) 
250   
251      References 
252      ========== 
253   
254      .. [1] https://en.wikipedia.org/wiki/Givens_rotation 
255   
256      See Also 
257      ======== 
258   
259      rot_axis1: Returns a rotation matrix for a rotation of theta (in radians) 
260          about the 1-axis (clockwise around the x axis) 
261      rot_axis2: Returns a rotation matrix for a rotation of theta (in radians) 
262          about the 2-axis (clockwise around the y axis) 
263      rot_axis3: Returns a rotation matrix for a rotation of theta (in radians) 
264          about the 3-axis (clockwise around the z axis) 
265      rot_ccw_axis1: Returns a rotation matrix for a rotation of theta (in radians) 
266          about the 1-axis (counterclockwise around the x axis) 
267      rot_ccw_axis2: Returns a rotation matrix for a rotation of theta (in radians) 
268          about the 2-axis (counterclockwise around the y axis) 
269      rot_ccw_axis3: Returns a rotation matrix for a rotation of theta (in radians) 
270          about the 3-axis (counterclockwise around the z axis) 
271      """
272      if not isinstance(dim, int) or dim < 2:
273          raise ValueError('dim must be an integer biggen than one, '
274                           'got {}.'.format(dim))
275  
276      if i == j:
277          raise ValueError('i and j must be different, '
278                           'got ({}, {})'.format(i, j))
279  
280      for ij in [i, j]:
281          if not isinstance(ij, int) or ij < 0 or ij > dim - 1:
282              raise ValueError('i and j must be integers between 0 and '
283                               '{}, got i={} and j={}.'.format(dim-1, i, j))
284  
285      theta = sympify(theta)
286      c = cos(theta)
287      s = sin(theta)
288      M = eye(dim)
289      M[i, i] = c
290      M[j, j] = c
291      M[i, j] = s
292      M[j, i] = -s
293      return M

296  def rot_axis3(theta):
297      r"""Returns a rotation matrix for a rotation of theta (in radians) 
298      about the 3-axis. 
299   
300      Explanation 
301      =========== 
302   
303      For a right-handed coordinate system, this corresponds to a 
304      clockwise rotation around the `z`-axis, given by: 
305   
306      .. math:: 
307   
308          R  = \begin{bmatrix} 
309                   \cos(\theta) & \sin(\theta) & 0 \\ 
310                  -\sin(\theta) & \cos(\theta) & 0 \\ 
311                              0 &            0 & 1 
312              \end{bmatrix} 
313   
314      Examples 
315      ======== 
316   
317      >>> from sympy import pi, rot_axis3 
318   
319      A rotation of pi/3 (60 degrees): 
320   
321      >>> theta = pi/3 
322      >>> rot_axis3(theta) 
323      Matrix([ 
324      [       1/2, sqrt(3)/2, 0], 
325      [-sqrt(3)/2,       1/2, 0], 
326      [         0,         0, 1]]) 
327   
328      If we rotate by pi/2 (90 degrees): 
329   
330      >>> rot_axis3(pi/2) 
331      Matrix([ 
332      [ 0, 1, 0], 
333      [-1, 0, 0], 
334      [ 0, 0, 1]]) 
335   
336      See Also 
337      ======== 
338   
339      rot_givens: Returns a Givens rotation matrix (generalized rotation for 
340          any number of dimensions) 
341      rot_ccw_axis3: Returns a rotation matrix for a rotation of theta (in radians) 
342          about the 3-axis (counterclockwise around the z axis) 
343      rot_axis1: Returns a rotation matrix for a rotation of theta (in radians) 
344          about the 1-axis (clockwise around the x axis) 
345      rot_axis2: Returns a rotation matrix for a rotation of theta (in radians) 
346          about the 2-axis (clockwise around the y axis) 
347      """
348      return rot_givens(0, 1, theta, dim=3)

626  @doctest_depends_on(modules=('numpy',))
627  def symarray(prefix, shape, **kwargs):  # pragma: no cover
628      r"""Create a numpy ndarray of symbols (as an object array). 
629   
630      The created symbols are named ``prefix_i1_i2_``...  You should thus provide a 
631      non-empty prefix if you want your symbols to be unique for different output 
632      arrays, as SymPy symbols with identical names are the same object. 
633   
634      Parameters 
635      ---------- 
636   
637      prefix : string 
638        A prefix prepended to the name of every symbol. 
639   
640      shape : int or tuple 
641        Shape of the created array.  If an int, the array is one-dimensional; for 
642        more than one dimension the shape must be a tuple. 
643   
644      \*\*kwargs : dict 
645        keyword arguments passed on to Symbol 
646   
647      Examples 
648      ======== 
649      These doctests require numpy. 
650   
651      >>> from sympy import symarray 
652      >>> symarray('', 3) 
653      [_0 _1 _2] 
654   
655      If you want multiple symarrays to contain distinct symbols, you *must* 
656      provide unique prefixes: 
657   
658      >>> a = symarray('', 3) 
659      >>> b = symarray('', 3) 
660      >>> a[0] == b[0] 
661      True 
662      >>> a = symarray('a', 3) 
663      >>> b = symarray('b', 3) 
664      >>> a[0] == b[0] 
665      False 
666   
667      Creating symarrays with a prefix: 
668   
669      >>> symarray('a', 3) 
670      [a_0 a_1 a_2] 
671   
672      For more than one dimension, the shape must be given as a tuple: 
673   
674      >>> symarray('a', (2, 3)) 
675      [[a_0_0 a_0_1 a_0_2] 
676       [a_1_0 a_1_1 a_1_2]] 
677      >>> symarray('a', (2, 3, 2)) 
678      [[[a_0_0_0 a_0_0_1] 
679        [a_0_1_0 a_0_1_1] 
680        [a_0_2_0 a_0_2_1]] 
681      <BLANKLINE> 
682       [[a_1_0_0 a_1_0_1] 
683        [a_1_1_0 a_1_1_1] 
684        [a_1_2_0 a_1_2_1]]] 
685   
686      For setting assumptions of the underlying Symbols: 
687   
688      >>> [s.is_real for s in symarray('a', 2, real=True)] 
689      [True, True] 
690      """
691      from numpy import empty, ndindex
692      arr = empty(shape, dtype=object)
693      for index in ndindex(shape):
694          arr[index] = Symbol('%s_%s' % (prefix, '_'.join(map(str, index))),
695                              **kwargs)
696      return arr

703  def casoratian(seqs, n, zero=True):
704      """Given linear difference operator L of order 'k' and homogeneous 
705         equation Ly = 0 we want to compute kernel of L, which is a set 
706         of 'k' sequences: a(n), b(n), ... z(n). 
707   
708         Solutions of L are linearly independent iff their Casoratian, 
709         denoted as C(a, b, ..., z), do not vanish for n = 0. 
710   
711         Casoratian is defined by k x k determinant:: 
712   
713                    +  a(n)     b(n)     . . . z(n)     + 
714                    |  a(n+1)   b(n+1)   . . . z(n+1)   | 
715                    |    .         .     .        .     | 
716                    |    .         .       .      .     | 
717                    |    .         .         .    .     | 
718                    +  a(n+k-1) b(n+k-1) . . . z(n+k-1) + 
719   
720         It proves very useful in rsolve_hyper() where it is applied 
721         to a generating set of a recurrence to factor out linearly 
722         dependent solutions and return a basis: 
723   
724         >>> from sympy import Symbol, casoratian, factorial 
725         >>> n = Symbol('n', integer=True) 
726   
727         Exponential and factorial are linearly independent: 
728   
729         >>> casoratian([2**n, factorial(n)], n) != 0 
730         True 
731   
732      """
733  
734      seqs = list(map(sympify, seqs))
735  
736      if not zero:
737          f = lambda i, j: seqs[j].subs(n, n + i)
738      else:
739          f = lambda i, j: seqs[j].subs(n, i)
740  
741      k = len(seqs)
742  
743      return Matrix(k, k, f).det()

838  def hessian(f, varlist, constraints=()):
839      """Compute Hessian matrix for a function f wrt parameters in varlist 
840      which may be given as a sequence or a row/column vector. A list of 
841      constraints may optionally be given. 
842   
843      Examples 
844      ======== 
845   
846      >>> from sympy import Function, hessian, pprint 
847      >>> from sympy.abc import x, y 
848      >>> f = Function('f')(x, y) 
849      >>> g1 = Function('g')(x, y) 
850      >>> g2 = x**2 + 3*y 
851      >>> pprint(hessian(f, (x, y), [g1, g2])) 
852      [                   d               d            ] 
853      [     0        0    --(g(x, y))     --(g(x, y))  ] 
854      [                   dx              dy           ] 
855      [                                                ] 
856      [     0        0        2*x              3       ] 
857      [                                                ] 
858      [                     2               2          ] 
859      [d                   d               d           ] 
860      [--(g(x, y))  2*x   ---(f(x, y))   -----(f(x, y))] 
861      [dx                   2            dy dx         ] 
862      [                   dx                           ] 
863      [                                                ] 
864      [                     2               2          ] 
865      [d                   d               d           ] 
866      [--(g(x, y))   3   -----(f(x, y))   ---(f(x, y)) ] 
867      [dy                dy dx              2          ] 
868      [                                   dy           ] 
869   
870      References 
871      ========== 
872   
873      .. [1] https://en.wikipedia.org/wiki/Hessian_matrix 
874   
875      See Also 
876      ======== 
877   
878      sympy.matrices.matrixbase.MatrixBase.jacobian 
879      wronskian 
880      """
881      # f is the expression representing a function f, return regular matrix
882      if isinstance(varlist, MatrixBase):
883          if 1 not in varlist.shape:
884              raise ShapeError("`varlist` must be a column or row vector.")
885          if varlist.cols == 1:
886              varlist = varlist.T
887          varlist = varlist.tolist()[0]
888      if is_sequence(varlist):
889          n = len(varlist)
890          if not n:
891              raise ShapeError("`len(varlist)` must not be zero.")
892      else:
893          raise ValueError("Improper variable list in hessian function")
894      if not getattr(f, 'diff'):
895          # check differentiability
896          raise ValueError("Function `f` (%s) is not differentiable" % f)
897      m = len(constraints)
898      N = m + n
899      out = zeros(N)
900      for k, g in enumerate(constraints):
901          if not getattr(g, 'diff'):
902              # check differentiability
903              raise ValueError("Function `f` (%s) is not differentiable" % f)
904          for i in range(n):
905              out[k, i + m] = g.diff(varlist[i])
906      for i in range(n):
907          for j in range(i, n):
908              out[i + m, j + m] = f.diff(varlist[i]).diff(varlist[j])
909      for i in range(N):
910          for j in range(i + 1, N):
911              out[j, i] = out[i, j]
912      return out

972  def randMatrix(r, c=None, min=0, max=99, seed=None, symmetric=False,
973                 percent=100, prng=None):
974      """Create random matrix with dimensions ``r`` x ``c``. If ``c`` is omitted 
975      the matrix will be square. If ``symmetric`` is True the matrix must be 
976      square. If ``percent`` is less than 100 then only approximately the given 
977      percentage of elements will be non-zero. 
978   
979      The pseudo-random number generator used to generate matrix is chosen in the 
980      following way. 
981   
982      * If ``prng`` is supplied, it will be used as random number generator. 
983        It should be an instance of ``random.Random``, or at least have 
984        ``randint`` and ``shuffle`` methods with same signatures. 
985      * if ``prng`` is not supplied but ``seed`` is supplied, then new 
986        ``random.Random`` with given ``seed`` will be created; 
987      * otherwise, a new ``random.Random`` with default seed will be used. 
988   
989      Examples 
990      ======== 
991   
992      >>> from sympy import randMatrix 
993      >>> randMatrix(3) # doctest:+SKIP 
994      [25, 45, 27] 
995      [44, 54,  9] 
996      [23, 96, 46] 
997      >>> randMatrix(3, 2) # doctest:+SKIP 
998      [87, 29] 
999      [23, 37] 
1000     [90, 26] 
1001     >>> randMatrix(3, 3, 0, 2) # doctest:+SKIP 
1002     [0, 2, 0] 
1003     [2, 0, 1] 
1004     [0, 0, 1] 
1005     >>> randMatrix(3, symmetric=True) # doctest:+SKIP 
1006     [85, 26, 29] 
1007     [26, 71, 43] 
1008     [29, 43, 57] 
1009     >>> A = randMatrix(3, seed=1) 
1010     >>> B = randMatrix(3, seed=2) 
1011     >>> A == B 
1012     False 
1013     >>> A == randMatrix(3, seed=1) 
1014     True 
1015     >>> randMatrix(3, symmetric=True, percent=50) # doctest:+SKIP 
1016     [77, 70,  0], 
1017     [70,  0,  0], 
1018     [ 0,  0, 88] 
1019     """
1020     # Note that ``Random()`` is equivalent to ``Random(None)``
1021     prng = prng or random.Random(seed)
1022 
1023     if c is None:
1024         c = r
1025 
1026     if symmetric and r != c:
1027         raise ValueError('For symmetric matrices, r must equal c, but %i != %i' % (r, c))
1028 
1029     ij = range(r * c)
1030     if percent != 100:
1031         ij = prng.sample(ij, int(len(ij)*percent // 100))
1032 
1033     m = zeros(r, c)
1034 
1035     if not symmetric:
1036         for ijk in ij:
1037             i, j = divmod(ijk, c)
1038             m[i, j] = prng.randint(min, max)
1039     else:
1040         for ijk in ij:
1041             i, j = divmod(ijk, c)
1042             if i <= j:
1043                 m[i, j] = m[j, i] = prng.randint(min, max)
1044 
1045     return m
1046

1048 def wronskian(functions, var, method='bareiss'):
1049     """ 
1050     Compute Wronskian for [] of functions 
1051  
1052     :: 
1053  
1054                          | f1       f2        ...   fn      | 
1055                          | f1'      f2'       ...   fn'     | 
1056                          |  .        .        .      .      | 
1057         W(f1, ..., fn) = |  .        .         .     .      | 
1058                          |  .        .          .    .      | 
1059                          |  (n)      (n)            (n)     | 
1060                          | D   (f1) D   (f2)  ...  D   (fn) | 
1061  
1062     see: https://en.wikipedia.org/wiki/Wronskian 
1063  
1064     See Also 
1065     ======== 
1066  
1067     sympy.matrices.matrixbase.MatrixBase.jacobian 
1068     hessian 
1069     """
1070 
1071     functions = [sympify(f) for f in functions]
1072     n = len(functions)
1073     if n == 0:
1074         return S.One
1075     W = Matrix(n, n, lambda i, j: functions[i].diff(var, j))
1076     return W.det(method)
