9    _EMPTY_NN_MODULE_STACK_KEY = "_empty_nn_module_stack_from_metadata_hook"

12   def _node_metadata_hook(node: torch.fx.Node, stack_trace: Optional[str] = None) -> None:
13       """ 
14       Hook for adding the appropriate metadata to nodes that are created during a 
15       pass using graph.create_node. An example of how to use it: 
16    
17       ``` 
18       with _set_node_metadata_hook(gm, 
19           functools.partial(_node_metadata_hook, stack_trace="file") 
20       ): 
21           pass(gm) 
22       ``` 
23    
24       This hook should not work for all generic cases -- specifically it assumes 
25       that nodes being added are only call_function nodes, and copies over the 
26       first argument node's nn_module_stack. 
27       """
28       assert node.op == "call_function" and callable(node.target)
29   
30       arg_meta = [arg.meta for arg in node.args if isinstance(arg, torch.fx.Node)]
31       assert len(arg_meta) >= 1
32       arg_meta = arg_meta[0]
33   
34       if (
35           isinstance(node.target, torch._ops.OpOverload)
36           and len(node.target._schema.returns) == 0
37       ):
38           node.meta["val"] = None
39       else:
40           fake_args = [
41               arg.meta["val"] if isinstance(arg, torch.fx.Node) else arg
42               for arg in node.args
43           ]
44           fake_res = node.target(*fake_args)
45           node.meta["val"] = fake_res
46   
47       node.meta["stack_trace"] = stack_trace
48       node.meta["nn_module_stack"] = arg_meta.get(
49           "nn_module_stack",
50           {
51               _EMPTY_NN_MODULE_STACK_KEY: (
52                   _EMPTY_NN_MODULE_STACK_KEY,
53                   _EMPTY_NN_MODULE_STACK_KEY,
54               )
55           },
56       )
57       node.meta["torch_fn"] = (
58           f"{node.target.__name__}_0",
59           f"{node.target.__class__.__name__}.{node.target.__name__}",
60       )
61   
62   
63   @contextlib.contextmanager
64   def _set_node_metadata_hook(gm: torch.fx.GraphModule, f):
65       """ 
66       Takes a callable which will be called after we create a new node. The 
67       callable takes the newly created node as input and returns None. 
68       """
69       assert callable(f), "node_metadata_hook must be a callable."
70   
71       # Add the hook to all submodules
72       for m in gm.modules():
73           if isinstance(m, GraphModule):
74               m._register_create_node_hook(f)
75       try:
76           yield
77       finally:
78           # Restore hook for all submodules
79           for m in gm.modules():
80               if isinstance(m, GraphModule):
81                   m._unregister_create_node_hook(f)
