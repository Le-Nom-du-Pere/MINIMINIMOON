122  // Specializations for float/double taken from Eigen
123  template <>
124  C10_ALWAYS_INLINE vfloat32
125  vec_min_nan<vfloat32>(const vfloat32& a, const vfloat32& b) {
126    // NOTE: about 10% slower than vec_min, but consistent with std::min and SSE
127    // regarding NaN
128    vfloat32 ret;
129    __asm__("xvcmpgesp %x0,%x1,%x2\n\txxsel %x0,%x1,%x2,%x0"
130            : "=&wa"(ret)
131            : "wa"(a), "wa"(b));
132    return ret;
133  }

144    return ret;

171  #define C10_VSX_VEC_NAN_PROPAG(name, type, btype, func)       \
172    C10_ALWAYS_INLINE type name(const type& a, const type& b) { \
173      type tmp = func(a, b);                                    \
174      btype nan_a = vec_cmpne(a, a);                            \
175      btype nan_b = vec_cmpne(b, b);                            \
176      tmp = vec_sel(tmp, a, nan_a);                             \
177      return vec_sel(tmp, b, nan_b);                            \
178    }

214    Vectorized<op_type> C10_ALWAYS_INLINE op(const Vectorized<op_type>& b) \
215        const {                                                            \
216      Vectorized<op_type>::vec_internal_type ret_0;                        \
217      Vectorized<op_type>::vec_internal_type ret_1;                        \
218      for (int i = 0; i < Vectorized<op_type>::size() / 2; i++) {          \
219        ret_0[i] = _vec0[i] binary_op b._vec0[i];                          \
220        ret_1[i] = _vec1[i] binary_op b._vec1[i];                          \
221      }                                                                    \
222      return Vectorized<op_type>{ret_0, ret_1};                            \
223    }

226    Vectorized<op_type> C10_ALWAYS_INLINE op(const Vectorized<op_type>& other) \
227        const {                                                                \
228      using vvtype = Vectorized<op_type>::vec_internal_type;                   \

232      return Vectorized<op_type>{vec_and(ret0, v_one), vec_and(ret1, v_one)};  \

277  // it can be used to emulate blend faster
278  constexpr int blendChoice(
279      uint32_t mask,
280      uint32_t half1 = 0xF,
281      uint32_t half2 = 0xF0) {
282    uint32_t none = 0;
283    uint32_t both = half1 | half2;
284    // clamp it between 0 and both
285    mask = mask & both;
286    // return  (a._vec0, a._vec1)
287    if (mask == none)
288      return 0;
289    // return (b._vec0,b._vec1)
290    else if (mask == both)
291      return 1;
292    // return  (b._vec0,a._vec1)
293    else if (mask == half1)
294      return 2;
295    // return  (a._vec0,b._vec1)
296    else if (mask == half2)
297      return 3;
298    // return  (*_vec0,a._vec1)
299    else if (mask > 0 && mask < half1)
300      return 4;
301    // return  (*_vec0,b._vec1)
302    else if ((mask & half2) == half2)
303      return 5;
304    // return (a._vec0,*_vec1)
305    else if ((mask & half1) == 0 && mask > half1)
306      return 6;
307    // return (b._vec0,*_vec1)
308    else if ((mask & half1) == half1 && mask > half1)
309      return 7;
310    // return (*_vec0,*_vec1)
311    return 8;
312  }

315  constexpr int blendChoiceDbl(uint32_t mask) {

320  constexpr vbool32 VsxMask1(uint32_t mask) {
321    uint32_t g0 = (mask & 1) * 0xffffffff;
322    uint32_t g1 = ((mask & 2) >> 1) * 0xffffffff;
323    uint32_t g2 = ((mask & 4) >> 2) * 0xffffffff;
324    uint32_t g3 = ((mask & 8) >> 3) * 0xffffffff;
325    return (vbool32){g0, g1, g2, g3};
326  }

344  constexpr int maskForComplex(uint32_t mask) {
345    mask = mask & 0xF;
346    int complex_mask = 0;
347    if (mask & 1)
348      complex_mask |= 3;
349    if (mask & 2)
350      complex_mask |= (3 << 2);
351    if (mask & 4)
352      complex_mask |= (3 << 4);
353    if (mask & 8)
354      complex_mask |= (3 << 6);
355    return complex_mask;
356  }

358  constexpr int maskForComplexDbl(uint32_t mask) {
359    mask = mask & 0x3;
360    int complex_mask = 0;
361    if (mask & 1)
362      complex_mask |= 3;
363    if (mask & 2)
364      complex_mask |= (3 << 2);
365    return complex_mask;
366  }
