9    # Create some useful type aliases
10   
11   # Template for arguments which can be supplied as a tuple, or which can be a scalar which PyTorch will internally
12   # broadcast to a tuple.
13   # Comes in several variants: A tuple of unknown size, and a fixed-size tuple for 1d, 2d, or 3d operations.
14   T = TypeVar("T")
15   _scalar_or_tuple_any_t: _TypeAlias = Union[T, tuple[T, ...]]
16   _scalar_or_tuple_1_t: _TypeAlias = Union[T, tuple[T]]
17   _scalar_or_tuple_2_t: _TypeAlias = Union[T, tuple[T, T]]
18   _scalar_or_tuple_3_t: _TypeAlias = Union[T, tuple[T, T, T]]
19   _scalar_or_tuple_4_t: _TypeAlias = Union[T, tuple[T, T, T, T]]
20   _scalar_or_tuple_5_t: _TypeAlias = Union[T, tuple[T, T, T, T, T]]
21   _scalar_or_tuple_6_t: _TypeAlias = Union[T, tuple[T, T, T, T, T, T]]
22   
23   # For arguments which represent size parameters (eg, kernel size, padding)
24   _size_any_t: _TypeAlias = _scalar_or_tuple_any_t[int]
25   _size_1_t: _TypeAlias = _scalar_or_tuple_1_t[int]
26   _size_2_t: _TypeAlias = _scalar_or_tuple_2_t[int]
27   _size_3_t: _TypeAlias = _scalar_or_tuple_3_t[int]
28   _size_4_t: _TypeAlias = _scalar_or_tuple_4_t[int]
29   _size_5_t: _TypeAlias = _scalar_or_tuple_5_t[int]
30   _size_6_t: _TypeAlias = _scalar_or_tuple_6_t[int]
31   
32   # For arguments which represent optional size parameters (eg, adaptive pool parameters)
33   _size_any_opt_t: _TypeAlias = _scalar_or_tuple_any_t[Optional[int]]
34   _size_2_opt_t: _TypeAlias = _scalar_or_tuple_2_t[Optional[int]]
35   _size_3_opt_t: _TypeAlias = _scalar_or_tuple_3_t[Optional[int]]
36   
37   # For arguments that represent a ratio to adjust each dimension of an input with (eg, upsampling parameters)
38   _ratio_2_t: _TypeAlias = _scalar_or_tuple_2_t[float]
39   _ratio_3_t: _TypeAlias = _scalar_or_tuple_3_t[float]
40   _ratio_any_t: _TypeAlias = _scalar_or_tuple_any_t[float]
41   
42   _tensor_list_t: _TypeAlias = _scalar_or_tuple_any_t[Tensor]
43   
44   # For the return value of max pooling operations that may or may not return indices.
45   # With the proposed 'Literal' feature to Python typing, it might be possible to
46   # eventually eliminate this.
47   _maybe_indices_t: _TypeAlias = _scalar_or_tuple_2_t[Tensor]
