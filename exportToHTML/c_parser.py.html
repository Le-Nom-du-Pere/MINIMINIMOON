24   - Compound statements, Return statements 
25    
26   Notes 
27   ===== 
28    
29   The module is dependent on an external dependency which needs to be installed 
30   to use the features of this module.

33   C source code.

120          def parse(self, filename, flags):
121              """Function to parse a file with C source code 
122   
123              It takes the filename as an attribute and creates a Clang AST 
124              Translation Unit parsing the file. 
125              Then the transformation function is called on the translation unit, 
126              whose results are collected into a list which is returned by the 
127              function. 
128   
129              Parameters 
130              ========== 
131   
132              filename : string 
133                  Path to the C file to be parsed 
134   
135              flags: list 
136                  Arguments to be passed to Clang while parsing the C code 
137   
138              Returns 
139              ======= 
140   
141              py_nodes: list 
142                  A list of SymPy AST nodes 
143   
144              """
145              filepath = os.path.abspath(filename)
146              self.tu = self.index.parse(
147                  filepath,
148                  args=flags,
149                  options=cin.TranslationUnit.PARSE_DETAILED_PROCESSING_RECORD
150              )
151              for child in self.tu.cursor.get_children():
152                  if child.kind == cin.CursorKind.VAR_DECL or child.kind == cin.CursorKind.FUNCTION_DECL:
153                      self._py_nodes.append(self.transform(child))
154              return self._py_nodes

156          def parse_str(self, source, flags):
157              """Function to parse a string with C source code 
158   
159              It takes the source code as an attribute, stores it in a temporary 
160              file and creates a Clang AST Translation Unit parsing the file. 
161              Then the transformation function is called on the translation unit, 
162              whose results are collected into a list which is returned by the 
163              function. 
164   
165              Parameters 
166              ========== 
167   
168              source : string 
169                  A string containing the C source code to be parsed 
170   
171              flags: list 
172                  Arguments to be passed to Clang while parsing the C code 
173   
174              Returns 
175              ======= 
176   
177              py_nodes: list 
178                  A list of SymPy AST nodes 
179   
180              """
181              file = tempfile.NamedTemporaryFile(mode = 'w+', suffix = '.cpp')
182              file.write(source)
183              file.flush()
184              file.seek(0)
185              self.tu = self.index.parse(
186                  file.name,
187                  args=flags,
188                  options=cin.TranslationUnit.PARSE_DETAILED_PROCESSING_RECORD
189              )
190              file.close()
191              for child in self.tu.cursor.get_children():
192                  if child.kind == cin.CursorKind.VAR_DECL or child.kind == cin.CursorKind.FUNCTION_DECL:
193                      self._py_nodes.append(self.transform(child))
194              return self._py_nodes

196          def transform(self, node):
197              """Transformation Function for Clang AST nodes 
198   
199              It determines the kind of node and calls the respective 
200              transformation function for that node. 
201   
202              Raises 
203              ====== 
204   
205              NotImplementedError : if the transformation for the provided node 
206              is not implemented 
207   
208              """
209              handler = getattr(self, 'transform_%s' % node.kind.name.lower(), None)
210  
211              if handler is None:
212                  print(
213                      "Ignoring node of type %s (%s)" % (
214                          node.kind,
215                          ' '.join(
216                              t.spelling for t in node.get_tokens())
217                          ),
218                      file=sys.stderr
219                  )
220  
221              return handler(node)

223          def transform_var_decl(self, node):
224              """Transformation Function for Variable Declaration 
225   
226              Used to create nodes for variable declarations and assignments with 
227              values or function call for the respective nodes in the clang AST 
228   
229              Returns 
230              ======= 
231   
232              A variable node as Declaration, with the initial value if given 
233   
234              Raises 
235              ====== 
236   
237              NotImplementedError : if called for data types not currently 
238              implemented 
239   
240              Notes 
241              ===== 
242   
243              The function currently supports following data types: 
244   
245              Boolean: 
246                  bool, _Bool 
247   
248              Integer: 
249                  8-bit: signed char and unsigned char 
250                  16-bit: short, short int, signed short, 
251                      signed short int, unsigned short, unsigned short int 
252                  32-bit: int, signed int, unsigned int 
253                  64-bit: long, long int, signed long, 
254                      signed long int, unsigned long, unsigned long int 
255   
256              Floating point: 
257                  Single Precision: float 
258                  Double Precision: double 
259                  Extended Precision: long double 
260   
261              """
262              if node.type.kind in self._data_types["int"]:
263                  type = self._data_types["int"][node.type.kind]
264              elif node.type.kind in self._data_types["float"]:
265                  type = self._data_types["float"][node.type.kind]
266              elif node.type.kind in self._data_types["bool"]:
267                  type = self._data_types["bool"][node.type.kind]
268              else:
269                  raise NotImplementedError("Only bool, int "
270                      "and float are supported")
271              try:
272                  children = node.get_children()
273                  child = next(children)
274  
275                  #ignoring namespace and type details for the variable
276                  while child.kind == cin.CursorKind.NAMESPACE_REF or child.kind == cin.CursorKind.TYPE_REF:
277                      child = next(children)
278  
279                  val = self.transform(child)
280  
281                  supported_rhs = [
282                      cin.CursorKind.INTEGER_LITERAL,
283                      cin.CursorKind.FLOATING_LITERAL,
284                      cin.CursorKind.UNEXPOSED_EXPR,
285                      cin.CursorKind.BINARY_OPERATOR,
286                      cin.CursorKind.PAREN_EXPR,
287                      cin.CursorKind.UNARY_OPERATOR,
288                      cin.CursorKind.CXX_BOOL_LITERAL_EXPR
289                  ]
290  
291                  if child.kind in supported_rhs:
292                      if isinstance(val, str):
293                          value = Symbol(val)
294                      elif isinstance(val, bool):
295                          if node.type.kind in self._data_types["int"]:
296                              value = Integer(0) if val == False else Integer(1)
297                          elif node.type.kind in self._data_types["float"]:
298                              value = Float(0.0) if val == False else Float(1.0)
299                          elif node.type.kind in self._data_types["bool"]:
300                              value = sympify(val)
301                      elif isinstance(val, (Integer, int, Float, float)):
302                          if node.type.kind in self._data_types["int"]:
303                              value = Integer(val)
304                          elif node.type.kind in self._data_types["float"]:
305                              value = Float(val)
306                          elif node.type.kind in self._data_types["bool"]:
307                              value = sympify(bool(val))
308                      else:
309                          value = val
310  
311                      return Variable(
312                      node.spelling
313                      ).as_Declaration(
314                          type = type,
315                          value = value
316                      )
317  
318                  elif child.kind == cin.CursorKind.CALL_EXPR:
319                      return Variable(
320                          node.spelling
321                          ).as_Declaration(
322                              value = val
323                          )
324  
325                  else:
326                      raise NotImplementedError("Given "
327                          "variable declaration \"{}\" "
328                          "is not possible to parse yet!"
329                          .format(" ".join(
330                              t.spelling for t in node.get_tokens()
331                              )
332                          ))
333  
334              except StopIteration:
335                  return Variable(
336                  node.spelling
337                  ).as_Declaration(
338                      type = type
339                  )

341          def transform_function_decl(self, node):
342              """Transformation Function For Function Declaration 
343   
344              Used to create nodes for function declarations and definitions for 
345              the respective nodes in the clang AST 
346   
347              Returns 
348              ======= 
349   
350              function : Codegen AST node 
351                  - FunctionPrototype node if function body is not present 
352                  - FunctionDefinition node if the function body is present 
353   
354   
355              """
356  
357              if node.result_type.kind in self._data_types["int"]:
358                  ret_type = self._data_types["int"][node.result_type.kind]
359              elif node.result_type.kind in self._data_types["float"]:
360                  ret_type = self._data_types["float"][node.result_type.kind]
361              elif node.result_type.kind in self._data_types["bool"]:
362                  ret_type = self._data_types["bool"][node.result_type.kind]
363              elif node.result_type.kind in self._data_types["void"]:
364                  ret_type = self._data_types["void"][node.result_type.kind]
365              else:
366                  raise NotImplementedError("Only void, bool, int "
367                      "and float are supported")
368              body = []
369              param = []
370  
371              # Subsequent nodes will be the parameters for the function.
372              for child in node.get_children():
373                  decl = self.transform(child)
374                  if child.kind == cin.CursorKind.PARM_DECL:
375                      param.append(decl)
376                  elif child.kind == cin.CursorKind.COMPOUND_STMT:
377                      for val in decl:
378                          body.append(val)
379                  else:
380                      body.append(decl)
381  
382              if body == []:
383                  function = FunctionPrototype(
384                      return_type = ret_type,
385                      name = node.spelling,
386                      parameters = param
387                  )
388              else:
389                  function = FunctionDefinition(
390                      return_type = ret_type,
391                      name = node.spelling,
392                      parameters = param,
393                      body = body
394                  )
395              return function

397          def transform_parm_decl(self, node):
398              """Transformation function for Parameter Declaration 
399   
400              Used to create parameter nodes for the required functions for the 
401              respective nodes in the clang AST 
402   
403              Returns 
404              ======= 
405   
406              param : Codegen AST Node 
407                  Variable node with the value and type of the variable 
408   
409              Raises 
410              ====== 
411   
412              ValueError if multiple children encountered in the parameter node 
413   
414              """
415              if node.type.kind in self._data_types["int"]:
416                  type = self._data_types["int"][node.type.kind]
417              elif node.type.kind in self._data_types["float"]:
418                  type = self._data_types["float"][node.type.kind]
419              elif node.type.kind in self._data_types["bool"]:
420                  type = self._data_types["bool"][node.type.kind]
421              else:
422                  raise NotImplementedError("Only bool, int "
423                      "and float are supported")
424              try:
425                  children = node.get_children()
426                  child = next(children)
427  
428                  # Any namespace nodes can be ignored
429                  while child.kind in [cin.CursorKind.NAMESPACE_REF,
430                                       cin.CursorKind.TYPE_REF,
431                                       cin.CursorKind.TEMPLATE_REF]:
432                      child = next(children)
433  
434                  # If there is a child, it is the default value of the parameter.
435                  lit = self.transform(child)
436                  if node.type.kind in self._data_types["int"]:
437                      val = Integer(lit)
438                  elif node.type.kind in self._data_types["float"]:
439                      val = Float(lit)
440                  elif node.type.kind in self._data_types["bool"]:
441                      val = sympify(bool(lit))
442                  else:
443                      raise NotImplementedError("Only bool, int "
444                          "and float are supported")
445  
446                  param = Variable(
447                      node.spelling
448                  ).as_Declaration(
449                      type = type,
450                      value = val
451                  )
452              except StopIteration:
453                  param = Variable(
454                      node.spelling
455                  ).as_Declaration(
456                      type = type
457                  )
458  
459              try:
460                  self.transform(next(children))
461                  raise ValueError("Can't handle multiple children on parameter")
462              except StopIteration:
463                  pass
464  
465              return param

579              """Transformation function for unexposed expression

619          def transform_call_expr(self, node):
620              """Transformation function for a call expression 
621   
622              Used to create function call nodes for the function calls present 
623              in the C code 
624   
625              Returns 
626              ======= 
627   
628              FunctionCall : Codegen AST Node 
629                  FunctionCall node with parameters if any parameters are present 
630   
631              """
632              param = []
633              children = node.get_children()
634              child = next(children)
635  
636              while child.kind == cin.CursorKind.NAMESPACE_REF:
637                  child = next(children)
638              while child.kind == cin.CursorKind.TYPE_REF:
639                  child = next(children)
640  
641              first_child = self.transform(child)
642              try:
643                  for child in children:
644                      arg = self.transform(child)
645                      if child.kind == cin.CursorKind.INTEGER_LITERAL:
646                          param.append(Integer(arg))
647                      elif child.kind == cin.CursorKind.FLOATING_LITERAL:
648                          param.append(Float(arg))
649                      else:
650                          param.append(arg)
651                  return FunctionCall(first_child, param)
652  
653              except StopIteration:
654                  return FunctionCall(first_child)

742          def transform_unary_operator(self, node):
743              """Transformation function for handling unary operators 
744   
745              Returns 
746              ======= 
747   
748              unary_expression: Codegen AST node 
749                      simplified unary expression represented as Codegen AST 
750   
751              Raises 
752              ====== 
753   
754              NotImplementedError 
755                  If dereferencing operator(*), address operator(&) or 
756                  bitwise NOT operator(~) is encountered 
757   
758              """
759              # supported operators list
760              operators_list = ['+', '-', '++', '--', '!']
761              tokens = list(node.get_tokens())
762  
763              # it can be either pre increment/decrement or any other operator from the list
764              if tokens[0].spelling in operators_list:
765                  child = self.transform(next(node.get_children()))
766                  # (decl_ref) e.g.; int a = ++b; or simply ++b;
767                  if isinstance(child, str):
768                      if tokens[0].spelling == '+':
769                          return Symbol(child)
770                      if tokens[0].spelling == '-':
771                          return Mul(Symbol(child), -1)
772                      if tokens[0].spelling == '++':
773                          return PreIncrement(Symbol(child))
774                      if tokens[0].spelling == '--':
775                          return PreDecrement(Symbol(child))
776                      if tokens[0].spelling == '!':
777                          return Not(Symbol(child))
778                  # e.g.; int a = -1; or int b = -(1 + 2);
779                  else:
780                      if tokens[0].spelling == '+':
781                          return child
782                      if tokens[0].spelling == '-':
783                          return Mul(child, -1)
784                      if tokens[0].spelling == '!':
785                          return Not(sympify(bool(child)))
786  
787              # it can be either post increment/decrement
788              # since variable name is obtained in token[0].spelling
789              elif tokens[1].spelling in ['++', '--']:
790                  child = self.transform(next(node.get_children()))
791                  if tokens[1].spelling == '++':
792                      return PostIncrement(Symbol(child))
793                  if tokens[1].spelling == '--':
794                      return PostDecrement(Symbol(child))
795              else:
796                  raise NotImplementedError("Dereferencing operator, "
797                      "Address operator and bitwise NOT operator "
798                      "have not been implemented yet!")

800          def transform_binary_operator(self, node):
801              """Transformation function for handling binary operators 
802   
803              Returns 
804              ======= 
805   
806              binary_expression: Codegen AST node 
807                      simplified binary expression represented as Codegen AST 
808   
809              Raises 
810              ====== 
811   
812              NotImplementedError 
813                  If a bitwise operator or 
814                  unary operator(which is a child of any binary 
815                  operator in Clang AST) is encountered 
816   
817              """
818              # get all the tokens of assignment
819              # and store it in the tokens list
820              tokens = list(node.get_tokens())
821  
822              # supported operators list
823              operators_list = ['+', '-', '*', '/', '%','=',
824              '>', '>=', '<', '<=', '==', '!=', '&&', '||', '+=', '-=',
825              '*=', '/=', '%=']
826  
827              # this stack will contain variable content
828              # and type of variable in the rhs
829              combined_variables_stack = []
830  
831              # this stack will contain operators
832              # to be processed in the rhs
833              operators_stack = []
834  
835              # iterate through every token
836              for token in tokens:
837                  # token is either '(', ')' or
838                  # any of the supported operators from the operator list
839                  if token.kind == cin.TokenKind.PUNCTUATION:
840  
841                      # push '(' to the operators stack
842                      if token.spelling == '(':
843                          operators_stack.append('(')
844  
845                      elif token.spelling == ')':
846                          # keep adding the expression to the
847                          # combined variables stack unless
848                          # '(' is found
849                          while (operators_stack
850                              and operators_stack[-1] != '('):
851                              if len(combined_variables_stack) < 2:
852                                  raise NotImplementedError(
853                                      "Unary operators as a part of "
854                                      "binary operators is not "
855                                      "supported yet!")
856                              rhs = combined_variables_stack.pop()
857                              lhs = combined_variables_stack.pop()
858                              operator = operators_stack.pop()
859                              combined_variables_stack.append(
860                                  self.perform_operation(
861                                  lhs, rhs, operator))
862  
863                          # pop '('
864                          operators_stack.pop()
865  
866                      # token is an operator (supported)
867                      elif token.spelling in operators_list:
868                          while (operators_stack
869                              and self.priority_of(token.spelling)
870                              <= self.priority_of(
871                              operators_stack[-1])):
872                              if len(combined_variables_stack) < 2:
873                                  raise NotImplementedError(
874                                      "Unary operators as a part of "
875                                      "binary operators is not "
876                                      "supported yet!")
877                              rhs = combined_variables_stack.pop()
878                              lhs = combined_variables_stack.pop()
879                              operator = operators_stack.pop()
880                              combined_variables_stack.append(
881                                  self.perform_operation(
882                                  lhs, rhs, operator))
883  
884                          # push current operator
885                          operators_stack.append(token.spelling)
886  
887                      # token is a bitwise operator
888                      elif token.spelling in ['&', '|', '^', '<<', '>>']:
889                          raise NotImplementedError(
890                              "Bitwise operator has not been "
891                              "implemented yet!")
892  
893                      # token is a shorthand bitwise operator
894                      elif token.spelling in ['&=', '|=', '^=', '<<=',
895                      '>>=']:
896                          raise NotImplementedError(
897                              "Shorthand bitwise operator has not been "
898                              "implemented yet!")
899                      else:
900                          raise NotImplementedError(
901                              "Given token {} is not implemented yet!"
902                              .format(token.spelling))
903  
904                  # token is an identifier(variable)
905                  elif token.kind == cin.TokenKind.IDENTIFIER:
906                      combined_variables_stack.append(
907                          [token.spelling, 'identifier'])
908  
909                  # token is a literal
910                  elif token.kind == cin.TokenKind.LITERAL:
911                      combined_variables_stack.append(
912                          [token.spelling, 'literal'])
913  
914                  # token is a keyword, either true or false
915                  elif (token.kind == cin.TokenKind.KEYWORD
916                      and token.spelling in ['true', 'false']):
917                      combined_variables_stack.append(
918                          [token.spelling, 'boolean'])
919                  else:
920                      raise NotImplementedError(
921                          "Given token {} is not implemented yet!"
922                          .format(token.spelling))
923  
924              # process remaining operators
925              while operators_stack:
926                  if len(combined_variables_stack) < 2:
927                      raise NotImplementedError(
928                          "Unary operators as a part of "
929                          "binary operators is not "
930                          "supported yet!")
931                  rhs = combined_variables_stack.pop()
932                  lhs = combined_variables_stack.pop()
933                  operator = operators_stack.pop()
934                  combined_variables_stack.append(
935                      self.perform_operation(lhs, rhs, operator))
936  
937              return combined_variables_stack[-1][0]

939          def priority_of(self, op):
940              """To get the priority of given operator"""
941              if op in ['=', '+=', '-=', '*=', '/=', '%=']:
942                  return 1
943              if op in ['&&', '||']:
944                  return 2
945              if op in ['<', '<=', '>', '>=', '==', '!=']:
946                  return 3
947              if op in ['+', '-']:
948                  return 4
949              if op in ['*', '/', '%']:
950                  return 5
951              return 0

953          def perform_operation(self, lhs, rhs, op):
954              """Performs operation supported by the SymPy core 
955   
956              Returns 
957              ======= 
958   
959              combined_variable: list 
960                  contains variable content and type of variable 
961   
962              """
963              lhs_value = self.get_expr_for_operand(lhs)
964              rhs_value = self.get_expr_for_operand(rhs)
965              if op == '+':
966                  return [Add(lhs_value, rhs_value), 'expr']
967              if op == '-':
968                  return [Add(lhs_value, -rhs_value), 'expr']
969              if op == '*':
970                  return [Mul(lhs_value, rhs_value), 'expr']
971              if op == '/':
972                  return [Mul(lhs_value, Pow(rhs_value, Integer(-1))), 'expr']
973              if op == '%':
974                  return [Mod(lhs_value, rhs_value), 'expr']
975              if op in ['<', '<=', '>', '>=', '==', '!=']:
976                  return [Rel(lhs_value, rhs_value, op), 'expr']
977              if op == '&&':
978                  return [And(as_Boolean(lhs_value), as_Boolean(rhs_value)), 'expr']
979              if op == '||':
980                  return [Or(as_Boolean(lhs_value), as_Boolean(rhs_value)), 'expr']
981              if op == '=':
982                  return [Assignment(Variable(lhs_value), rhs_value), 'expr']
983              if op in ['+=', '-=', '*=', '/=', '%=']:
984                  return [aug_assign(Variable(lhs_value), op[0], rhs_value), 'expr']

986          def get_expr_for_operand(self, combined_variable):
987              """Gives out SymPy Codegen AST node 
988   
989              AST node returned is corresponding to 
990              combined variable passed.Combined variable contains 
991              variable content and type of variable 
992   
993              """
994              if combined_variable[1] == 'identifier':
995                  return Symbol(combined_variable[0])
996              if combined_variable[1] == 'literal':
997                  if '.' in combined_variable[0]:
998                      return Float(float(combined_variable[0]))
999                  else:
1000                     return Integer(int(combined_variable[0]))
1001             if combined_variable[1] == 'expr':
1002                 return combined_variable[0]
1003             if combined_variable[1] == 'boolean':
1004                     return true if combined_variable[0] == 'true' else false
