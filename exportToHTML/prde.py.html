1    """ 
2    Algorithms for solving Parametric Risch Differential Equations. 
3     
4    The methods used for solving Parametric Risch Differential Equations parallel 
5    those for solving Risch Differential Equations.  See the outline in the 
6    docstring of rde.py for more information. 
7     
8    The Parametric Risch Differential Equation problem is, given f, g1, ..., gm in 
9    K(t), to determine if there exist y in K(t) and c1, ..., cm in Const(K) such 
10   that Dy + f*y == Sum(ci*gi, (i, 1, m)), and to find such y and ci if they exist. 
11    
12   For the algorithms here G is a list of tuples of factions of the terms on the 
13   right hand side of the equation (i.e., gi in k(t)), and Q is a list of terms on 
14   the right hand side of the equation (i.e., qi in k[t]).  See the docstring of 
15   each function for more information. 
16   """

35   def prde_normal_denom(fa, fd, G, DE):
36       """ 
37       Parametric Risch Differential Equation - Normal part of the denominator. 
38    
39       Explanation 
40       =========== 
41    
42       Given a derivation D on k[t] and f, g1, ..., gm in k(t) with f weakly 
43       normalized with respect to t, return the tuple (a, b, G, h) such that 
44       a, h in k[t], b in k<t>, G = [g1, ..., gm] in k(t)^m, and for any solution 
45       c1, ..., cm in Const(k) and y in k(t) of Dy + f*y == Sum(ci*gi, (i, 1, m)), 
46       q == y*h in k<t> satisfies a*Dq + b*q == Sum(ci*Gi, (i, 1, m)). 
47       """
48       dn, ds = splitfactor(fd, DE)
49       Gas, Gds = list(zip(*G))
50       gd = reduce(lambda i, j: i.lcm(j), Gds, Poly(1, DE.t))
51       en, es = splitfactor(gd, DE)
52   
53       p = dn.gcd(en)
54       h = en.gcd(en.diff(DE.t)).quo(p.gcd(p.diff(DE.t)))
55   
56       a = dn*h
57       c = a*h
58   
59       ba = a*fa - dn*derivation(h, DE)*fd
60       ba, bd = ba.cancel(fd, include=True)
61   
62       G = [(c*A).cancel(D, include=True) for A, D in G]
63   
64       return (a, (ba, bd), G, h)

66   def real_imag(ba, bd, gen):
67       """ 
68       Helper function, to get the real and imaginary part of a rational function 
69       evaluated at sqrt(-1) without actually evaluating it at sqrt(-1). 
70    
71       Explanation 
72       =========== 
73    
74       Separates the even and odd power terms by checking the degree of terms wrt 
75       mod 4. Returns a tuple (ba[0], ba[1], bd) where ba[0] is real part 
76       of the numerator ba[1] is the imaginary part and bd is the denominator 
77       of the rational function. 
78       """
79       bd = bd.as_poly(gen).as_dict()
80       ba = ba.as_poly(gen).as_dict()
81       denom_real = [value if key[0] % 4 == 0 else -value if key[0] % 4 == 2 else 0 for key, value in bd.items()]
82       denom_imag = [value if key[0] % 4 == 1 else -value if key[0] % 4 == 3 else 0 for key, value in bd.items()]
83       bd_real = sum(r for r in denom_real)
84       bd_imag = sum(r for r in denom_imag)
85       num_real = [value if key[0] % 4 == 0 else -value if key[0] % 4 == 2 else 0 for key, value in ba.items()]
86       num_imag = [value if key[0] % 4 == 1 else -value if key[0] % 4 == 3 else 0 for key, value in ba.items()]
87       ba_real = sum(r for r in num_real)
88       ba_imag = sum(r for r in num_imag)
89       ba = ((ba_real*bd_real + ba_imag*bd_imag).as_poly(gen), (ba_imag*bd_real - ba_real*bd_imag).as_poly(gen))
90       bd = (bd_real*bd_real + bd_imag*bd_imag).as_poly(gen)
91       return (ba[0], ba[1], bd)

94   def prde_special_denom(a, ba, bd, G, DE, case='auto'):
95       """ 
96       Parametric Risch Differential Equation - Special part of the denominator. 
97    
98       Explanation 
99       =========== 
100   
101      Case is one of {'exp', 'tan', 'primitive'} for the hyperexponential, 
102      hypertangent, and primitive cases, respectively.  For the hyperexponential 
103      (resp. hypertangent) case, given a derivation D on k[t] and a in k[t], 
104      b in k<t>, and g1, ..., gm in k(t) with Dt/t in k (resp. Dt/(t**2 + 1) in 
105      k, sqrt(-1) not in k), a != 0, and gcd(a, t) == 1 (resp. 
106      gcd(a, t**2 + 1) == 1), return the tuple (A, B, GG, h) such that A, B, h in 
107      k[t], GG = [gg1, ..., ggm] in k(t)^m, and for any solution c1, ..., cm in 
108      Const(k) and q in k<t> of a*Dq + b*q == Sum(ci*gi, (i, 1, m)), r == q*h in 
109      k[t] satisfies A*Dr + B*r == Sum(ci*ggi, (i, 1, m)). 
110   
111      For case == 'primitive', k<t> == k[t], so it returns (a, b, G, 1) in this 
112      case. 
113      """
114      # TODO: Merge this with the very similar special_denom() in rde.py
115      if case == 'auto':
116          case = DE.case
117  
118      if case == 'exp':
119          p = Poly(DE.t, DE.t)
120      elif case == 'tan':
121          p = Poly(DE.t**2 + 1, DE.t)
122      elif case in ('primitive', 'base'):
123          B = ba.quo(bd)
124          return (a, B, G, Poly(1, DE.t))
125      else:
126          raise ValueError("case must be one of {'exp', 'tan', 'primitive', "
127              "'base'}, not %s." % case)
128  
129      nb = order_at(ba, p, DE.t) - order_at(bd, p, DE.t)
130      nc = min(order_at(Ga, p, DE.t) - order_at(Gd, p, DE.t) for Ga, Gd in G)
131      n = min(0, nc - min(0, nb))
132      if not nb:
133          # Possible cancellation.
134          if case == 'exp':
135              dcoeff = DE.d.quo(Poly(DE.t, DE.t))
136              with DecrementLevel(DE):  # We are guaranteed to not have problems,
137                                        # because case != 'base'.
138                  alphaa, alphad = frac_in(-ba.eval(0)/bd.eval(0)/a.eval(0), DE.t)
139                  etaa, etad = frac_in(dcoeff, DE.t)
140                  A = parametric_log_deriv(alphaa, alphad, etaa, etad, DE)
141                  if A is not None:
142                      Q, m, z = A
143                      if Q == 1:
144                          n = min(n, m)
145  
146          elif case == 'tan':
147              dcoeff = DE.d.quo(Poly(DE.t**2 + 1, DE.t))
148              with DecrementLevel(DE):  # We are guaranteed to not have problems,
149                                        # because case != 'base'.
150                  betaa, alphaa, alphad =  real_imag(ba, bd*a, DE.t)
151                  betad = alphad
152                  etaa, etad = frac_in(dcoeff, DE.t)
153                  if recognize_log_derivative(Poly(2, DE.t)*betaa, betad, DE):
154                      A = parametric_log_deriv(alphaa, alphad, etaa, etad, DE)
155                      B = parametric_log_deriv(betaa, betad, etaa, etad, DE)
156                      if A is not None and B is not None:
157                          Q, s, z = A
158                          # TODO: Add test
159                          if Q == 1:
160                              n = min(n, s/2)
161  
162      N = max(0, -nb)
163      pN = p**N
164      pn = p**-n  # This is 1/h
165  
166      A = a*pN
167      B = ba*pN.quo(bd) + Poly(n, DE.t)*a*derivation(p, DE).quo(p)*pN
168      G = [(Ga*pN*pn).cancel(Gd, include=True) for Ga, Gd in G]
169      h = pn
170  
171      # (a*p**N, (b + n*a*Dp/p)*p**N, g1*p**(N - n), ..., gm*p**(N - n), p**-n)
172      return (A, B, G, h)

175  def prde_linear_constraints(a, b, G, DE):
176      """ 
177      Parametric Risch Differential Equation - Generate linear constraints on the constants. 
178   
179      Explanation 
180      =========== 
181   
182      Given a derivation D on k[t], a, b, in k[t] with gcd(a, b) == 1, and 
183      G = [g1, ..., gm] in k(t)^m, return Q = [q1, ..., qm] in k[t]^m and a 
184      matrix M with entries in k(t) such that for any solution c1, ..., cm in 
185      Const(k) and p in k[t] of a*Dp + b*p == Sum(ci*gi, (i, 1, m)), 
186      (c1, ..., cm) is a solution of Mx == 0, and p and the ci satisfy 
187      a*Dp + b*p == Sum(ci*qi, (i, 1, m)). 
188   
189      Because M has entries in k(t), and because Matrix does not play well with 
190      Poly, M will be a Matrix of Basic expressions. 
191      """
192      m = len(G)
193  
194      Gns, Gds = list(zip(*G))
195      d = reduce(lambda i, j: i.lcm(j), Gds)
196      d = Poly(d, field=True)
197      Q = [(ga*(d).quo(gd)).div(d) for ga, gd in G]
198  
199      if not all(ri.is_zero for _, ri in Q):
200          N = max(ri.degree(DE.t) for _, ri in Q)
201          M = Matrix(N + 1, m, lambda i, j: Q[j][1].nth(i), DE.t)
202      else:
203          M = Matrix(0, m, [], DE.t)  # No constraints, return the empty matrix.
204  
205      qs, _ = list(zip(*Q))
206      return (qs, M)

208  def poly_linear_constraints(p, d):
209      """ 
210      Given p = [p1, ..., pm] in k[t]^m and d in k[t], return 
211      q = [q1, ..., qm] in k[t]^m and a matrix M with entries in k such 
212      that Sum(ci*pi, (i, 1, m)), for c1, ..., cm in k, is divisible 
213      by d if and only if (c1, ..., cm) is a solution of Mx = 0, in 
214      which case the quotient is Sum(ci*qi, (i, 1, m)). 
215      """
216      m = len(p)
217      q, r = zip(*[pi.div(d) for pi in p])
218  
219      if not all(ri.is_zero for ri in r):
220          n = max(ri.degree() for ri in r)
221          M = Matrix(n + 1, m, lambda i, j: r[j].nth(i), d.gens)
222      else:
223          M = Matrix(0, m, [], d.gens)  # No constraints.
224  
225      return q, M

227  def constant_system(A, u, DE):
228      """ 
229      Generate a system for the constant solutions. 
230   
231      Explanation 
232      =========== 
233   
234      Given a differential field (K, D) with constant field C = Const(K), a Matrix 
235      A, and a vector (Matrix) u with coefficients in K, returns the tuple 
236      (B, v, s), where B is a Matrix with coefficients in C and v is a vector 
237      (Matrix) such that either v has coefficients in C, in which case s is True 
238      and the solutions in C of Ax == u are exactly all the solutions of Bx == v, 
239      or v has a non-constant coefficient, in which case s is False Ax == u has no 
240      constant solution. 
241   
242      This algorithm is used both in solving parametric problems and in 
243      determining if an element a of K is a derivative of an element of K or the 
244      logarithmic derivative of a K-radical using the structure theorem approach. 
245   
246      Because Poly does not play well with Matrix yet, this algorithm assumes that 
247      all matrix entries are Basic expressions. 
248      """
249      if not A:
250          return A, u
251      Au = A.row_join(u)
252      Au, _ = Au.rref()
253      # Warning: This will NOT return correct results if cancel() cannot reduce
254      # an identically zero expression to 0.  The danger is that we might
255      # incorrectly prove that an integral is nonelementary (such as
256      # risch_integrate(exp((sin(x)**2 + cos(x)**2 - 1)*x**2), x).
257      # But this is a limitation in computer algebra in general, and implicit
258      # in the correctness of the Risch Algorithm is the computability of the
259      # constant field (actually, this same correctness problem exists in any
260      # algorithm that uses rref()).
261      #
262      # We therefore limit ourselves to constant fields that are computable
263      # via the cancel() function, in order to prevent a speed bottleneck from
264      # calling some more complex simplification function (rational function
265      # coefficients will fall into this class).  Furthermore, (I believe) this
266      # problem will only crop up if the integral explicitly contains an
267      # expression in the constant field that is identically zero, but cannot
268      # be reduced to such by cancel().  Therefore, a careful user can avoid this
269      # problem entirely by being careful with the sorts of expressions that
270      # appear in his integrand in the variables other than the integration
271      # variable (the structure theorems should be able to completely decide these
272      # problems in the integration variable).
273  
274      A, u = Au[:, :-1], Au[:, -1]
275  
276      D = lambda x: derivation(x, DE, basic=True)
277  
278      for j, i in itertools.product(range(A.cols), range(A.rows)):
279          if A[i, j].expr.has(*DE.T):
280              # This assumes that const(F(t0, ..., tn) == const(K) == F
281              Ri = A[i, :]
282              # Rm+1; m = A.rows
283              DAij = D(A[i, j])
284              Rm1 = Ri.applyfunc(lambda x: D(x) / DAij)
285              um1 = D(u[i]) / DAij
286  
287              Aj = A[:, j]
288              A = A - Aj * Rm1
289              u = u - Aj * um1
290  
291              A = A.col_join(Rm1)
292              u = u.col_join(Matrix([um1], u.gens))
293  
294      return (A, u)

297  def prde_spde(a, b, Q, n, DE):
298      """ 
299      Special Polynomial Differential Equation algorithm: Parametric Version. 
300   
301      Explanation 
302      =========== 
303   
304      Given a derivation D on k[t], an integer n, and a, b, q1, ..., qm in k[t] 
305      with deg(a) > 0 and gcd(a, b) == 1, return (A, B, Q, R, n1), with 
306      Qq = [q1, ..., qm] and R = [r1, ..., rm], such that for any solution 
307      c1, ..., cm in Const(k) and q in k[t] of degree at most n of 
308      a*Dq + b*q == Sum(ci*gi, (i, 1, m)), p = (q - Sum(ci*ri, (i, 1, m)))/a has 
309      degree at most n1 and satisfies A*Dp + B*p == Sum(ci*qi, (i, 1, m)) 
310      """
311      R, Z = list(zip(*[gcdex_diophantine(b, a, qi) for qi in Q]))
312  
313      A = a
314      B = b + derivation(a, DE)
315      Qq = [zi - derivation(ri, DE) for ri, zi in zip(R, Z)]
316      R = list(R)
317      n1 = n - a.degree(DE.t)
318  
319      return (A, B, Qq, R, n1)

322  def prde_no_cancel_b_large(b, Q, n, DE):
323      """ 
324      Parametric Poly Risch Differential Equation - No cancellation: deg(b) large enough. 
325   
326      Explanation 
327      =========== 
328   
329      Given a derivation D on k[t], n in ZZ, and b, q1, ..., qm in k[t] with 
330      b != 0 and either D == d/dt or deg(b) > max(0, deg(D) - 1), returns 
331      h1, ..., hr in k[t] and a matrix A with coefficients in Const(k) such that 
332      if c1, ..., cm in Const(k) and q in k[t] satisfy deg(q) <= n and 
333      Dq + b*q == Sum(ci*qi, (i, 1, m)), then q = Sum(dj*hj, (j, 1, r)), where 
334      d1, ..., dr in Const(k) and A*Matrix([[c1, ..., cm, d1, ..., dr]]).T == 0. 
335      """
336      db = b.degree(DE.t)
337      m = len(Q)
338      H = [Poly(0, DE.t)]*m
339  
340      for N, i in itertools.product(range(n, -1, -1), range(m)):  # [n, ..., 0]
341          si = Q[i].nth(N + db)/b.LC()
342          sitn = Poly(si*DE.t**N, DE.t)
343          H[i] = H[i] + sitn
344          Q[i] = Q[i] - derivation(sitn, DE) - b*sitn
345  
346      if all(qi.is_zero for qi in Q):
347          dc = -1
348      else:
349          dc = max(qi.degree(DE.t) for qi in Q)
350      M = Matrix(dc + 1, m, lambda i, j: Q[j].nth(i), DE.t)
351      A, u = constant_system(M, zeros(dc + 1, 1, DE.t), DE)
352      c = eye(m, DE.t)
353      A = A.row_join(zeros(A.rows, m, DE.t)).col_join(c.row_join(-c))
354  
355      return (H, A)

358  def prde_no_cancel_b_small(b, Q, n, DE):
359      """ 
360      Parametric Poly Risch Differential Equation - No cancellation: deg(b) small enough. 
361   
362      Explanation 
363      =========== 
364   
365      Given a derivation D on k[t], n in ZZ, and b, q1, ..., qm in k[t] with 
366      deg(b) < deg(D) - 1 and either D == d/dt or deg(D) >= 2, returns 
367      h1, ..., hr in k[t] and a matrix A with coefficients in Const(k) such that 
368      if c1, ..., cm in Const(k) and q in k[t] satisfy deg(q) <= n and 
369      Dq + b*q == Sum(ci*qi, (i, 1, m)) then q = Sum(dj*hj, (j, 1, r)) where 
370      d1, ..., dr in Const(k) and A*Matrix([[c1, ..., cm, d1, ..., dr]]).T == 0. 
371      """
372      m = len(Q)
373      H = [Poly(0, DE.t)]*m
374  
375      for N, i in itertools.product(range(n, 0, -1), range(m)):  # [n, ..., 1]
376          si = Q[i].nth(N + DE.d.degree(DE.t) - 1)/(N*DE.d.LC())
377          sitn = Poly(si*DE.t**N, DE.t)
378          H[i] = H[i] + sitn
379          Q[i] = Q[i] - derivation(sitn, DE) - b*sitn
380  
381      if b.degree(DE.t) > 0:
382          for i in range(m):
383              si = Poly(Q[i].nth(b.degree(DE.t))/b.LC(), DE.t)
384              H[i] = H[i] + si
385              Q[i] = Q[i] - derivation(si, DE) - b*si
386          if all(qi.is_zero for qi in Q):
387              dc = -1
388          else:
389              dc = max(qi.degree(DE.t) for qi in Q)
390          M = Matrix(dc + 1, m, lambda i, j: Q[j].nth(i), DE.t)
391          A, u = constant_system(M, zeros(dc + 1, 1, DE.t), DE)
392          c = eye(m, DE.t)
393          A = A.row_join(zeros(A.rows, m, DE.t)).col_join(c.row_join(-c))
394          return (H, A)
395  
396      # else: b is in k, deg(qi) < deg(Dt)
397  
398      t = DE.t
399      if DE.case != 'base':
400          with DecrementLevel(DE):
401              t0 = DE.t  # k = k0(t0)
402              ba, bd = frac_in(b, t0, field=True)
403              Q0 = [frac_in(qi.TC(), t0, field=True) for qi in Q]
404              f, B = param_rischDE(ba, bd, Q0, DE)
405  
406              # f = [f1, ..., fr] in k^r and B is a matrix with
407              # m + r columns and entries in Const(k) = Const(k0)
408              # such that Dy0 + b*y0 = Sum(ci*qi, (i, 1, m)) has
409              # a solution y0 in k with c1, ..., cm in Const(k)
410              # if and only y0 = Sum(dj*fj, (j, 1, r)) where
411              # d1, ..., dr ar in Const(k) and
412              # B*Matrix([c1, ..., cm, d1, ..., dr]) == 0.
413  
414          # Transform fractions (fa, fd) in f into constant
415          # polynomials fa/fd in k[t].
416          # (Is there a better way?)
417          f = [Poly(fa.as_expr()/fd.as_expr(), t, field=True)
418               for fa, fd in f]
419          B = Matrix.from_Matrix(B.to_Matrix(), t)
420      else:
421          # Base case. Dy == 0 for all y in k and b == 0.
422          # Dy + b*y = Sum(ci*qi) is solvable if and only if
423          # Sum(ci*qi) == 0 in which case the solutions are
424          # y = d1*f1 for f1 = 1 and any d1 in Const(k) = k.
425  
426          f = [Poly(1, t, field=True)]  # r = 1
427          B = Matrix([[qi.TC() for qi in Q] + [S.Zero]], DE.t)
428          # The condition for solvability is
429          # B*Matrix([c1, ..., cm, d1]) == 0
430          # There are no constraints on d1.
431  
432      # Coefficients of t^j (j > 0) in Sum(ci*qi) must be zero.
433      d = max(qi.degree(DE.t) for qi in Q)
434      if d > 0:
435          M = Matrix(d, m, lambda i, j: Q[j].nth(i + 1), DE.t)
436          A, _ = constant_system(M, zeros(d, 1, DE.t), DE)
437      else:
438          # No constraints on the hj.
439          A = Matrix(0, m, [], DE.t)
440  
441      # Solutions of the original equation are
442      #    y = Sum(dj*fj, (j, 1, r) + Sum(ei*hi, (i, 1, m)),
443      # where  ei == ci  (i = 1, ..., m),  when
444      # A*Matrix([c1, ..., cm]) == 0 and
445      # B*Matrix([c1, ..., cm, d1, ..., dr]) == 0
446  
447      # Build combined constraint matrix with m + r + m columns.
448  
449      r = len(f)
450      I = eye(m, DE.t)
451      A = A.row_join(zeros(A.rows, r + m, DE.t))
452      B = B.row_join(zeros(B.rows, m, DE.t))
453      C = I.row_join(zeros(m, r, DE.t)).row_join(-I)
454  
455      return f + H, A.col_join(B).col_join(C)

495              # building up Sum(djn*(D(fjn*t^n) - b*fjnt^n))
496              Fi[j] = -(derivation(hji, DE) - b*hji)
497  
498          H += hi
499          # in the next loop instead of Q it has
500          # to be Q + Fi taking its place
501          Q = Q + Fi
502  
503      return (H, M)

506  def param_poly_rischDE(a, b, q, n, DE):
507      """Polynomial solutions of a parametric Risch differential equation. 
508   
509      Explanation 
510      =========== 
511   
512      Given a derivation D in k[t], a, b in k[t] relatively prime, and q 
513      = [q1, ..., qm] in k[t]^m, return h = [h1, ..., hr] in k[t]^r and 
514      a matrix A with m + r columns and entries in Const(k) such that 
515      a*Dp + b*p = Sum(ci*qi, (i, 1, m)) has a solution p of degree <= n 
516      in k[t] with c1, ..., cm in Const(k) if and only if p = Sum(dj*hj, 
517      (j, 1, r)) where d1, ..., dr are in Const(k) and (c1, ..., cm, 
518      d1, ..., dr) is a solution of Ax == 0. 
519      """
520      m = len(q)
521      if n < 0:
522          # Only the trivial zero solution is possible.
523          # Find relations between the qi.
524          if all(qi.is_zero for qi in q):
525              return [], zeros(1, m, DE.t)  # No constraints.
526  
527          N = max(qi.degree(DE.t) for qi in q)
528          M = Matrix(N + 1, m, lambda i, j: q[j].nth(i), DE.t)
529          A, _ = constant_system(M, zeros(M.rows, 1, DE.t), DE)
530  
531          return [], A
532  
533      if a.is_ground:
534          # Normalization: a = 1.
535          a = a.LC()
536          b, q = b.to_field().exquo_ground(a), [qi.to_field().exquo_ground(a) for qi in q]
537  
538          if not b.is_zero and (DE.case == 'base' or
539                  b.degree() > max(0, DE.d.degree() - 1)):
540              return prde_no_cancel_b_large(b, q, n, DE)
541  
542          elif ((b.is_zero or b.degree() < DE.d.degree() - 1)
543                  and (DE.case == 'base' or DE.d.degree() >= 2)):
544              return prde_no_cancel_b_small(b, q, n, DE)
545  
546          elif (DE.d.degree() >= 2 and
547                b.degree() == DE.d.degree() - 1 and
548                n > -b.as_poly().LC()/DE.d.as_poly().LC()):
549              raise NotImplementedError("prde_no_cancel_b_equal() is "
550                  "not yet implemented.")
551  
552          else:
553              # Liouvillian cases
554              if DE.case in ('primitive', 'exp'):
555                  return prde_cancel_liouvillian(b, q, n, DE)
556              else:
557                  raise NotImplementedError("non-linear and hypertangent "
558                          "cases have not yet been implemented")
559  
560      # else: deg(a) > 0
561  
562      # Iterate SPDE as long as possible cumulating coefficient
563      # and terms for the recovery of original solutions.
564      alpha, beta = a.one, [a.zero]*m
565      while n >= 0:  # and a, b relatively prime
566          a, b, q, r, n = prde_spde(a, b, q, n, DE)
567          beta = [betai + alpha*ri for betai, ri in zip(beta, r)]
568          alpha *= a
569          # Solutions p of a*Dp + b*p = Sum(ci*qi) correspond to
570          # solutions alpha*p + Sum(ci*betai) of the initial equation.
571          d = a.gcd(b)
572          if not d.is_ground:
573              break
574  
575      # a*Dp + b*p = Sum(ci*qi) may have a polynomial solution
576      # only if the sum is divisible by d.
577  
578      qq, M = poly_linear_constraints(q, d)
579      # qq = [qq1, ..., qqm] where qqi = qi.quo(d).
580      # M is a matrix with m columns an entries in k.
581      # Sum(fi*qi, (i, 1, m)), where f1, ..., fm are elements of k, is
582      # divisible by d if and only if M*Matrix([f1, ..., fm]) == 0,
583      # in which case the quotient is Sum(fi*qqi).
584  
585      A, _ = constant_system(M, zeros(M.rows, 1, DE.t), DE)
586      # A is a matrix with m columns and entries in Const(k).
587      # Sum(ci*qqi) is Sum(ci*qi).quo(d), and the remainder is zero
588      # for c1, ..., cm in Const(k) if and only if
589      # A*Matrix([c1, ...,cm]) == 0.
590  
591      V = A.nullspace()
592      # V = [v1, ..., vu] where each vj is a column matrix with
593      # entries aj1, ..., ajm in Const(k).
594      # Sum(aji*qi) is divisible by d with exact quotient Sum(aji*qqi).
595      # Sum(ci*qi) is divisible by d if and only if ci = Sum(dj*aji)
596      # (i = 1, ..., m) for some d1, ..., du in Const(k).
597      # In that case, solutions of
598      #     a*Dp + b*p = Sum(ci*qi) = Sum(dj*Sum(aji*qi))
599      # are the same as those of
600      #     (a/d)*Dp + (b/d)*p = Sum(dj*rj)
601      # where rj = Sum(aji*qqi).
602  
603      if not V:  # No non-trivial solution.
604          return [], eye(m, DE.t)  # Could return A, but this has
605                                   # the minimum number of rows.
606  
607      Mqq = Matrix([qq])  # A single row.
608      r = [(Mqq*vj)[0] for vj in V]  # [r1, ..., ru]
609  
610      # Solutions of (a/d)*Dp + (b/d)*p = Sum(dj*rj) correspond to
611      # solutions alpha*p + Sum(Sum(dj*aji)*betai) of the initial
612      # equation. These are equal to alpha*p + Sum(dj*fj) where
613      # fj = Sum(aji*betai).
614      Mbeta = Matrix([beta])
615      f = [(Mbeta*vj)[0] for vj in V]  # [f1, ..., fu]
616  
617      #
618      # Solve the reduced equation recursively.
619      #
620      g, B = param_poly_rischDE(a.quo(d), b.quo(d), r, n, DE)
621  
622      # g = [g1, ..., gv] in k[t]^v and and B is a matrix with u + v
623      # columns and entries in Const(k) such that
624      # (a/d)*Dp + (b/d)*p = Sum(dj*rj) has a solution p of degree <= n
625      # in k[t] if and only if p = Sum(ek*gk) where e1, ..., ev are in
626      # Const(k) and B*Matrix([d1, ..., du, e1, ..., ev]) == 0.
627      # The solutions of the original equation are then
628      # Sum(dj*fj, (j, 1, u)) + alpha*Sum(ek*gk, (k, 1, v)).
629  
630      # Collect solution components.
631      h = f + [alpha*gk for gk in g]
632  
633      # Build combined relation matrix.
634      A = -eye(m, DE.t)
635      for vj in V:
636          A = A.row_join(vj)
637      A = A.row_join(zeros(m, len(g), DE.t))
638      A = A.col_join(zeros(B.rows, m, DE.t).row_join(B))
639  
640      return h, A

643  def param_rischDE(fa, fd, G, DE):
644      """ 
645      Solve a Parametric Risch Differential Equation: Dy + f*y == Sum(ci*Gi, (i, 1, m)). 
646   
647      Explanation 
648      =========== 
649   
650      Given a derivation D in k(t), f in k(t), and G 
651      = [G1, ..., Gm] in k(t)^m, return h = [h1, ..., hr] in k(t)^r and 
652      a matrix A with m + r columns and entries in Const(k) such that 
653      Dy + f*y = Sum(ci*Gi, (i, 1, m)) has a solution y 
654      in k(t) with c1, ..., cm in Const(k) if and only if y = Sum(dj*hj, 
655      (j, 1, r)) where d1, ..., dr are in Const(k) and (c1, ..., cm, 
656      d1, ..., dr) is a solution of Ax == 0. 
657   
658      Elements of k(t) are tuples (a, d) with a and d in k[t]. 
659      """
660      m = len(G)
661      q, (fa, fd) = weak_normalizer(fa, fd, DE)
662      # Solutions of the weakly normalized equation Dz + f*z = q*Sum(ci*Gi)
663      # correspond to solutions y = z/q of the original equation.
664      gamma = q
665      G = [(q*ga).cancel(gd, include=True) for ga, gd in G]
666  
667      a, (ba, bd), G, hn = prde_normal_denom(fa, fd, G, DE)
668      # Solutions q in k<t> of  a*Dq + b*q = Sum(ci*Gi) correspond
669      # to solutions z = q/hn of the weakly normalized equation.
670      gamma *= hn
671  
672      A, B, G, hs = prde_special_denom(a, ba, bd, G, DE)
673      # Solutions p in k[t] of  A*Dp + B*p = Sum(ci*Gi) correspond
674      # to solutions q = p/hs of the previous equation.
675      gamma *= hs
676  
677      g = A.gcd(B)
678      a, b, g = A.quo(g), B.quo(g), [gia.cancel(gid*g, include=True) for
679          gia, gid in G]
680  
681      # a*Dp + b*p = Sum(ci*gi)  may have a polynomial solution
682      # only if the sum is in k[t].
683  
684      q, M = prde_linear_constraints(a, b, g, DE)
685  
686      # q = [q1, ..., qm] where qi in k[t] is the polynomial component
687      # of the partial fraction expansion of gi.
688      # M is a matrix with m columns and entries in k.
689      # Sum(fi*gi, (i, 1, m)), where f1, ..., fm are elements of k,
690      # is a polynomial if and only if M*Matrix([f1, ..., fm]) == 0,
691      # in which case the sum is equal to Sum(fi*qi).
692  
693      M, _ = constant_system(M, zeros(M.rows, 1, DE.t), DE)
694      # M is a matrix with m columns and entries in Const(k).
695      # Sum(ci*gi) is in k[t] for c1, ..., cm in Const(k)
696      # if and only if M*Matrix([c1, ..., cm]) == 0,
697      # in which case the sum is Sum(ci*qi).
698  
699      ## Reduce number of constants at this point
700  
701      V = M.nullspace()
702      # V = [v1, ..., vu] where each vj is a column matrix with
703      # entries aj1, ..., ajm in Const(k).
704      # Sum(aji*gi) is in k[t] and equal to Sum(aji*qi) (j = 1, ..., u).
705      # Sum(ci*gi) is in k[t] if and only is ci = Sum(dj*aji)
706      # (i = 1, ..., m) for some d1, ..., du in Const(k).
707      # In that case,
708      #     Sum(ci*gi) = Sum(ci*qi) = Sum(dj*Sum(aji*qi)) = Sum(dj*rj)
709      # where rj = Sum(aji*qi) (j = 1, ..., u) in k[t].
710  
711      if not V:  # No non-trivial solution
712          return [], eye(m, DE.t)
713  
714      Mq = Matrix([q])  # A single row.
715      r = [(Mq*vj)[0] for vj in V]  # [r1, ..., ru]
716  
717      # Solutions of a*Dp + b*p = Sum(dj*rj) correspond to solutions
718      # y = p/gamma of the initial equation with ci = Sum(dj*aji).
719  
720      try:
721          # We try n=5. At least for prde_spde, it will always
722          # terminate no matter what n is.
723          n = bound_degree(a, b, r, DE, parametric=True)
724      except NotImplementedError:
725          # A temporary bound is set. Eventually, it will be removed.
726          # the currently added test case takes large time
727          # even with n=5, and much longer with large n's.
728          n = 5
729  
730      h, B = param_poly_rischDE(a, b, r, n, DE)
731  
732      # h = [h1, ..., hv] in k[t]^v and and B is a matrix with u + v
733      # columns and entries in Const(k) such that
734      # a*Dp + b*p = Sum(dj*rj) has a solution p of degree <= n
735      # in k[t] if and only if p = Sum(ek*hk) where e1, ..., ev are in
736      # Const(k) and B*Matrix([d1, ..., du, e1, ..., ev]) == 0.
737      # The solutions of the original equation for ci = Sum(dj*aji)
738      # (i = 1, ..., m) are then y = Sum(ek*hk, (k, 1, v))/gamma.
739  
740      ## Build combined relation matrix with m + u + v columns.
741  
742      A = -eye(m, DE.t)
743      for vj in V:
744          A = A.row_join(vj)
745      A = A.row_join(zeros(m, len(h), DE.t))
746      A = A.col_join(zeros(B.rows, m, DE.t).row_join(B))
747  
748      ## Eliminate d1, ..., du.
749  
750      W = A.nullspace()
751  
752      # W = [w1, ..., wt] where each wl is a column matrix with
753      # entries blk (k = 1, ..., m + u + v) in Const(k).
754      # The vectors (bl1, ..., blm) generate the space of those
755      # constant families (c1, ..., cm) for which a solution of
756      # the equation Dy + f*y == Sum(ci*Gi) exists. They generate
757      # the space and form a basis except possibly when Dy + f*y == 0
758      # is solvable in k(t}. The corresponding solutions are
759      # y = Sum(blk'*hk, (k, 1, v))/gamma, where k' = k + m + u.
760  
761      v = len(h)
762      shape = (len(W), m+v)
763      elements = [wl[:m] + wl[-v:] for wl in W] # excise dj's.
764      items = [e for row in elements for e in row]
765  
766      # Need to set the shape in case W is empty
767      M = Matrix(*shape, items, DE.t)
768      N = M.nullspace()
769  
770      # N = [n1, ..., ns] where the ni in Const(k)^(m + v) are column
771      # vectors generating the space of linear relations between
772      # c1, ..., cm, e1, ..., ev.
773  
774      C = Matrix([ni[:] for ni in N], DE.t)  # rows n1, ..., ns.
775  
776      return [hk.cancel(gamma, include=True) for hk in h], C

780      """ 
781      Simpler version of step 1 & 2 for the limited integration problem. 
782   
783      Explanation 
784      =========== 
785   
786      Given a derivation D on k(t) and f, g1, ..., gn in k(t), return 
787      (a, b, h, N, g, V) such that a, b, h in k[t], N is a non-negative integer, 
788      g in k(t), V == [v1, ..., vm] in k(t)^m, and for any solution v in k(t), 
789      c1, ..., cm in C of f == Dv + Sum(ci*wi, (i, 1, m)), p = v*h is in k<t>, and 
790      p and the ci satisfy a*Dp + b*p == g + Sum(ci*vi, (i, 1, m)).  Furthermore, 
791      if S1irr == Sirr, then p is in k[t], and if t is nonlinear or Liouvillian 
792      over k, then deg(p) <= N. 
793   
794      So that the special part is always computed, this function calls the more 
795      general prde_special_denom() automatically if it cannot determine that 
796      S1irr == Sirr.  Furthermore, it will automatically call bound_degree() when 
797      t is linear and non-Liouvillian, which for the transcendental case, implies 
798      that Dt == a*t + b with for some a, b in k*. 
799      """
800      dn, ds = splitfactor(fd, DE)
801      E = [splitfactor(gd, DE) for _, gd in G]
802      En, Es = list(zip(*E))
803      c = reduce(lambda i, j: i.lcm(j), (dn,) + En)  # lcm(dn, en1, ..., enm)
804      hn = c.gcd(c.diff(DE.t))
805      a = hn
806      b = -derivation(hn, DE)
807      N = 0
808  
809      # These are the cases where we know that S1irr = Sirr, but there could be
810      # others, and this algorithm will need to be extended to handle them.
811      if DE.case in ('base', 'primitive', 'exp', 'tan'):
812          hs = reduce(lambda i, j: i.lcm(j), (ds,) + Es)  # lcm(ds, es1, ..., esm)
813          a = hn*hs
814          b -= (hn*derivation(hs, DE)).quo(hs)
815          mu = min(order_at_oo(fa, fd, DE.t), min(order_at_oo(ga, gd, DE.t) for
816              ga, gd in G))
817          # So far, all the above are also nonlinear or Liouvillian, but if this
818          # changes, then this will need to be updated to call bound_degree()
819          # as per the docstring of this function (DE.case == 'other_linear').
820          N = hn.degree(DE.t) + hs.degree(DE.t) + max(0, 1 - DE.d.degree(DE.t) - mu)
821      else:
822          # TODO: implement this
823          raise NotImplementedError
824  
825      V = [(-a*hn*ga).cancel(gd, include=True) for ga, gd in G]
826      return (a, b, a, N, (a*hn*fa).cancel(fd, include=True), V)

830      """ 
831      Solves the limited integration problem:  f = Dv + Sum(ci*wi, (i, 1, n)) 
832      """
833      fa, fd = fa*Poly(1/fd.LC(), DE.t), fd.monic()
834      # interpreting limited integration problem as a
835      # parametric Risch DE problem
836      Fa = Poly(0, DE.t)
837      Fd = Poly(1, DE.t)
838      G = [(fa, fd)] + G
839      h, A = param_rischDE(Fa, Fd, G, DE)
840      V = A.nullspace()
841      V = [v for v in V if v[0] != 0]
842      if not V:
843          return None
844      else:
845          # we can take any vector from V, we take V[0]

847          # v = [-1, c1, ..., cm, d1, ..., dr]
848          v = V[0]/(-c0)

855          Ya, Yd = Poly(y_num, DE.t), Poly(y_den, DE.t)
856          Y = Ya*Poly(1/Yd.LC(), DE.t), Yd.monic()

963  def is_deriv_k(fa, fd, DE):
964      r""" 
965      Checks if Df/f is the derivative of an element of k(t). 
966   
967      Explanation 
968      =========== 
969   
970      a in k(t) is the derivative of an element of k(t) if there exists b in k(t) 
971      such that a = Db.  Either returns (ans, u), such that Df/f == Du, or None, 
972      which means that Df/f is not the derivative of an element of k(t).  ans is 
973      a list of tuples such that Add(*[i*j for i, j in ans]) == u.  This is useful 
974      for seeing exactly which elements of k(t) produce u. 
975   
976      This function uses the structure theorem approach, which says that for any 
977      f in K, Df/f is the derivative of a element of K if and only if there are ri 
978      in QQ such that:: 
979   
980              ---               ---       Dt 
981              \    r  * Dt   +  \    r  *   i      Df 
982              /     i     i     /     i   ---   =  --. 
983              ---               ---        t        f 
984           i in L            i in E         i 
985                 K/C(x)            K/C(x) 
986   
987   
988      Where C = Const(K), L_K/C(x) = { i in {1, ..., n} such that t_i is 
989      transcendental over C(x)(t_1, ..., t_i-1) and Dt_i = Da_i/a_i, for some a_i 
990      in C(x)(t_1, ..., t_i-1)* } (i.e., the set of all indices of logarithmic 
991      monomials of K over C(x)), and E_K/C(x) = { i in {1, ..., n} such that t_i 
992      is transcendental over C(x)(t_1, ..., t_i-1) and Dt_i/t_i = Da_i, for some 
993      a_i in C(x)(t_1, ..., t_i-1) } (i.e., the set of all indices of 
994      hyperexponential monomials of K over C(x)).  If K is an elementary extension 
995      over C(x), then the cardinality of L_K/C(x) U E_K/C(x) is exactly the 
996      transcendence degree of K over C(x).  Furthermore, because Const_D(K) == 
997      Const_D(C(x)) == C, deg(Dt_i) == 1 when t_i is in E_K/C(x) and 
998      deg(Dt_i) == 0 when t_i is in L_K/C(x), implying in particular that E_K/C(x) 
999      and L_K/C(x) are disjoint. 
1000  
1001     The sets L_K/C(x) and E_K/C(x) must, by their nature, be computed 
1002     recursively using this same function.  Therefore, it is required to pass 
1003     them as indices to D (or T).  E_args are the arguments of the 
1004     hyperexponentials indexed by E_K (i.e., if i is in E_K, then T[i] == 
1005     exp(E_args[i])).  This is needed to compute the final answer u such that 
1006     Df/f == Du. 
1007  
1008     log(f) will be the same as u up to a additive constant.  This is because 
1009     they will both behave the same as monomials. For example, both log(x) and 
1010     log(2*x) == log(x) + log(2) satisfy Dt == 1/x, because log(2) is constant. 
1011     Therefore, the term const is returned.  const is such that 
1012     log(const) + f == u.  This is calculated by dividing the arguments of one 
1013     logarithm from the other.  Therefore, it is necessary to pass the arguments 
1014     of the logarithmic terms in L_args. 
1015  
1016     To handle the case where we are given Df/f, not f, use is_deriv_k_in_field(). 
1017  
1018     See also 
1019     ======== 
1020     is_log_deriv_k_t_radical_in_field, is_log_deriv_k_t_radical 
1021  
1022     """
1023     # Compute Df/f
1024     dfa, dfd = (fd*derivation(fa, DE) - fa*derivation(fd, DE)), fd*fa
1025     dfa, dfd = dfa.cancel(dfd, include=True)
1026 
1027     # Our assumption here is that each monomial is recursively transcendental
1028     if len(DE.exts) != len(DE.D):
1029         if [i for i in DE.cases if i == 'tan'] or \
1030                 ({i for i in DE.cases if i == 'primitive'} -
1031                         set(DE.indices('log'))):
1032             raise NotImplementedError("Real version of the structure "
1033                 "theorems with hypertangent support is not yet implemented.")
1034 
1035         # TODO: What should really be done in this case?
1036         raise NotImplementedError("Nonelementary extensions not supported "
1037             "in the structure theorems.")
1038 
1039     E_part = [DE.D[i].quo(Poly(DE.T[i], DE.T[i])).as_expr() for i in DE.indices('exp')]
1040     L_part = [DE.D[i].as_expr() for i in DE.indices('log')]
1041 
1042     # The expression dfa/dfd might not be polynomial in any of its symbols so we
1043     # use a Dummy as the generator for PolyMatrix.
1044     dum = Dummy()
1045     lhs = Matrix([E_part + L_part], dum)
1046     rhs = Matrix([dfa.as_expr()/dfd.as_expr()], dum)
1047 
1048     A, u = constant_system(lhs, rhs, DE)
1049 
1050     u = u.to_Matrix()  # Poly to Expr
1051 
1052     if not A or not all(derivation(i, DE, basic=True).is_zero for i in u):
1053         # If the elements of u are not all constant
1054         # Note: See comment in constant_system
1055 
1056         # Also note: derivation(basic=True) calls cancel()
1057         return None
1058     else:
1059         if not all(i.is_Rational for i in u):
1060             raise NotImplementedError("Cannot work with non-rational "
1061                 "coefficients in this case.")
1062         else:
1063             terms = ([DE.extargs[i] for i in DE.indices('exp')] +
1064                     [DE.T[i] for i in DE.indices('log')])
1065             ans = list(zip(terms, u))
1066             result = Add(*[Mul(i, j) for i, j in ans])
1067             argterms = ([DE.T[i] for i in DE.indices('exp')] +
1068                     [DE.extargs[i] for i in DE.indices('log')])
1069             l = []
1070             ld = []
1071             for i, j in zip(argterms, u):
1072                 # We need to get around things like sqrt(x**2) != x
1073                 # and also sqrt(x**2 + 2*x + 1) != x + 1
1074                 # Issue 10798: i need not be a polynomial
1075                 i, d = i.as_numer_denom()
1076                 icoeff, iterms = sqf_list(i)
1077                 l.append(Mul(*([Pow(icoeff, j)] + [Pow(b, e*j) for b, e in iterms])))
1078                 dcoeff, dterms = sqf_list(d)
1079                 ld.append(Mul(*([Pow(dcoeff, j)] + [Pow(b, e*j) for b, e in dterms])))
1080             const = cancel(fa.as_expr()/fd.as_expr()/Mul(*l)*Mul(*ld))
1081 
1082             return (ans, result, const)

1085 def is_log_deriv_k_t_radical(fa, fd, DE, Df=True):
1086     r""" 
1087     Checks if Df is the logarithmic derivative of a k(t)-radical. 
1088  
1089     Explanation 
1090     =========== 
1091  
1092     b in k(t) can be written as the logarithmic derivative of a k(t) radical if 
1093     there exist n in ZZ and u in k(t) with n, u != 0 such that n*b == Du/u. 
1094     Either returns (ans, u, n, const) or None, which means that Df cannot be 
1095     written as the logarithmic derivative of a k(t)-radical.  ans is a list of 
1096     tuples such that Mul(*[i**j for i, j in ans]) == u.  This is useful for 
1097     seeing exactly what elements of k(t) produce u.

1100     f in K, Df is the logarithmic derivative of a K-radical if and only if there

1112     transcendental over C(x)(t_1, ..., t_i-1) and Dt_i = Da_i/a_i, for some a_i

1116     a_i in C(x)(t_1, ..., t_i-1) } (i.e., the set of all indices of 
1117     hyperexponential monomials of K over C(x)).  If K is an elementary extension

1120     Const_D(C(x)) == C, deg(Dt_i) == 1 when t_i is in E_K/C(x) and 
1121     deg(Dt_i) == 0 when t_i is in L_K/C(x), implying in particular that E_K/C(x) 
1122     and L_K/C(x) are disjoint.

1124     The sets L_K/C(x) and E_K/C(x) must, by their nature, be computed 
1125     recursively using this same function.  Therefore, it is required to pass 
1126     them as indices to D (or T).  L_args are the arguments of the logarithms 
1127     indexed by L_K (i.e., if i is in L_K, then T[i] == log(L_args[i])).  This is 
1128     needed to compute the final answer u such that n*f == Du/u. 
1129  
1130     exp(f) will be the same as u up to a multiplicative constant.  This is 
1131     because they will both behave the same as monomials.  For example, both 
1132     exp(x) and exp(x + 1) == E*exp(x) satisfy Dt == t. Therefore, the term const 
1133     is returned.  const is such that exp(const)*f == u.  This is calculated by 
1134     subtracting the arguments of one exponential from the other.  Therefore, it 
1135     is necessary to pass the arguments of the exponential terms in E_args.

1138     is_log_deriv_k_t_radical_in_field().

1161         raise NotImplementedError("Nonelementary extensions not supported "
1162             "in the structure theorems.")

1164     E_part = [DE.D[i].quo(Poly(DE.T[i], DE.T[i])).as_expr() for i in DE.indices('exp')]
1165     L_part = [DE.D[i].as_expr() for i in DE.indices('log')]

1208 def is_log_deriv_k_t_radical_in_field(fa, fd, DE, case='auto', z=None):
1209     """ 
1210     Checks if f can be written as the logarithmic derivative of a k(t)-radical. 
1211  
1212     Explanation 
1213     =========== 
1214  
1215     It differs from is_log_deriv_k_t_radical(fa, fd, DE, Df=False) 
1216     for any given fa, fd, DE in that it finds the solution in the 
1217     given field not in some (possibly unspecified extension) and 
1218     "in_field" with the function name is used to indicate that. 
1219  
1220     f in k(t) can be written as the logarithmic derivative of a k(t) radical if 
1221     there exist n in ZZ and u in k(t) with n, u != 0 such that n*f == Du/u. 
1222     Either returns (n, u) or None, which means that f cannot be written as the 
1223     logarithmic derivative of a k(t)-radical.

1226     hyperexponential, and hypertangent cases, respectively.  If case is 'auto', 
1227     it will attempt to determine the type of the derivation automatically.

1245         # None in this case. This should never happen for the
1246         # functions given when solving the parametric logarithmic

1276         return None
