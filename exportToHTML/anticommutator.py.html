22   class AntiCommutator(Expr):
23       """The standard anticommutator, in an unevaluated state. 
24    
25       Explanation 
26       =========== 
27    
28       Evaluating an anticommutator is defined [1]_ as: ``{A, B} = A*B + B*A``. 
29       This class returns the anticommutator in an unevaluated form.  To evaluate 
30       the anticommutator, use the ``.doit()`` method. 
31    
32       Canonical ordering of an anticommutator is ``{A, B}`` for ``A < B``. The 
33       arguments of the anticommutator are put into canonical order using 
34       ``__cmp__``. If ``B < A``, then ``{A, B}`` is returned as ``{B, A}``. 
35    
36       Parameters 
37       ========== 
38    
39       A : Expr 
40           The first argument of the anticommutator {A,B}. 
41       B : Expr 
42           The second argument of the anticommutator {A,B}. 
43    
44       Examples 
45       ======== 
46    
47       >>> from sympy import symbols 
48       >>> from sympy.physics.quantum import AntiCommutator 
49       >>> from sympy.physics.quantum import Operator, Dagger 
50       >>> x, y = symbols('x,y') 
51       >>> A = Operator('A') 
52       >>> B = Operator('B') 
53    
54       Create an anticommutator and use ``doit()`` to multiply them out. 
55    
56       >>> ac = AntiCommutator(A,B); ac 
57       {A,B} 
58       >>> ac.doit() 
59       A*B + B*A 
60    
61       The commutator orders it arguments in canonical order: 
62    
63       >>> ac = AntiCommutator(B,A); ac 
64       {A,B} 
65    
66       Commutative constants are factored out: 
67    
68       >>> AntiCommutator(3*x*A,x*y*B) 
69       3*x**2*y*{A,B} 
70    
71       Adjoint operations applied to the anticommutator are properly applied to 
72       the arguments: 
73    
74       >>> Dagger(AntiCommutator(A,B)) 
75       {Dagger(A),Dagger(B)} 
76    
77       References 
78       ========== 
79    
80       .. [1] https://en.wikipedia.org/wiki/Commutator 
81       """
82       is_commutative = False
83   
84       _kind_dispatcher = KindDispatcher("AntiCommutator_kind_dispatcher", commutative=True)
85   
86       @property
87       def kind(self):
88           arg_kinds = (a.kind for a in self.args)
89           return self._kind_dispatcher(*arg_kinds)
90   
91       def __new__(cls, A, B):
92           r = cls.eval(A, B)
93           if r is not None:
94               return r
95           obj = Expr.__new__(cls, A, B)
96           return obj
97   
98       @classmethod
99       def eval(cls, a, b):
100          if not (a and b):
101              return S.Zero
102          if a == b:
103              return Integer(2)*a**2
104          if a.is_commutative or b.is_commutative:
105              return Integer(2)*a*b
106  
107          # [xA,yB]  ->  xy*[A,B]
108          ca, nca = a.args_cnc()
109          cb, ncb = b.args_cnc()
110          c_part = ca + cb
111          if c_part:
112              return Mul(Mul(*c_part), cls(Mul._from_args(nca), Mul._from_args(ncb)))
113  
114          # Canonical ordering of arguments
115          #The Commutator [A,B] is on canonical form if A < B.
116          if a.compare(b) == 1:
117              return cls(b, a)
118  
119      def doit(self, **hints):
120          """ Evaluate anticommutator """
121          # Keep the import of Operator here to avoid problems with
122          # circular imports.
123          from sympy.physics.quantum.operator import Operator
124          A = self.args[0]
125          B = self.args[1]
126          if isinstance(A, Operator) and isinstance(B, Operator):
127              try:
128                  comm = A._eval_anticommutator(B, **hints)
129              except NotImplementedError:
130                  try:
131                      comm = B._eval_anticommutator(A, **hints)
132                  except NotImplementedError:
133                      comm = None
134              if comm is not None:
135                  return comm.doit(**hints)
136          return (A*B + B*A).doit(**hints)
137  
138      def _eval_adjoint(self):
139          return AntiCommutator(Dagger(self.args[0]), Dagger(self.args[1]))
140  
141      def _sympyrepr(self, printer, *args):
142          return "%s(%s,%s)" % (
143              self.__class__.__name__, printer._print(
144                  self.args[0]), printer._print(self.args[1])
145          )
146  
147      def _sympystr(self, printer, *args):
148          return "{%s,%s}" % (
149              printer._print(self.args[0]), printer._print(self.args[1]))
150  
151      def _pretty(self, printer, *args):
152          pform = printer._print(self.args[0], *args)
153          pform = prettyForm(*pform.right(prettyForm(',')))
154          pform = prettyForm(*pform.right(printer._print(self.args[1], *args)))
155          pform = prettyForm(*pform.parens(left='{', right='}'))
156          return pform
157  
158      def _latex(self, printer, *args):
159          return "\\left\\{%s,%s\\right\\}" % tuple([
160              printer._print(arg, *args) for arg in self.args])
