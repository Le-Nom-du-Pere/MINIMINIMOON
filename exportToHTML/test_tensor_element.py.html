1    from sympy.tensor.tensor import (Tensor, TensorIndexType, TensorSymmetry,

7    def test_tensor_element():
8        L = TensorIndexType("L")
9        i, j, k, l, m, n = tensor_indices("i j k l m n", L)
10   
11       A = TensorHead("A", [L, L], TensorSymmetry.no_symmetry(2))
12   
13       a = A(i, j)
14   
15       assert isinstance(TensorElement(a, {}), Tensor)
16       assert isinstance(TensorElement(a, {k: 1}), Tensor)
17   
18       te1 = TensorElement(a, {Symbol("i"): 1})
19       assert te1.free == [(j, 0)]
20       assert te1.get_free_indices() == [j]
21       assert te1.dum == []
22   
23       te2 = TensorElement(a, {i: 1})
24       assert te2.free == [(j, 0)]
25       assert te2.get_free_indices() == [j]
26       assert te2.dum == []
27   
28       assert te1 == te2
29   
30       array = Array([[1, 2], [3, 4]])
31       assert te1.replace_with_arrays({A(i, j): array}, [j]) == array[1, :]
