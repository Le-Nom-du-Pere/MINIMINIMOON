125          parts = fqn.split('.')

143      def _expand_fold_binary_op(self, op, args):
144          """ 
145          This method expands a fold on binary operations. 
146   
147          ``functools.reduce`` is an example of a folded operation. 
148   
149          For example, the expression 
150   
151          `A + B + C + D` 
152   
153          is folded into 
154   
155          `((A + B) + C) + D` 
156          """
157          if len(args) == 1:
158              return self._print(args[0])
159          else:
160              return "%s(%s, %s)" % (
161                  self._module_format(op),
162                  self._expand_fold_binary_op(op, args[:-1]),
163                  self._print(args[-1]),
164              )
165  
166      def _expand_reduce_binary_op(self, op, args):
167          """ 
168          This method expands a reduction on binary operations. 
169   
170          Notice: this is NOT the same as ``functools.reduce``. 
171   
172          For example, the expression 
173   
174          `A + B + C + D` 
175   
176          is reduced into: 
177   
178          `(A + B) + (C + D)` 
179          """
180          if len(args) == 1:
181              return self._print(args[0])
182          else:
183              N = len(args)
184              Nhalf = N // 2
185              return "%s(%s, %s)" % (
186                  self._module_format(op),
187                  self._expand_reduce_binary_op(args[:Nhalf]),
188                  self._expand_reduce_binary_op(args[Nhalf:]),
189              )

207      def _print_Piecewise(self, expr):
208          result = []
209          i = 0
210          for arg in expr.args:
211              e = arg.expr
212              c = arg.cond
213              if i == 0:
214                  result.append('(')
215              result.append('(')
216              result.append(self._print(e))
217              result.append(')')
218              result.append(' if ')
219              result.append(self._print(c))
220              result.append(' else ')
221              i += 1
222          result = result[:-1]
223          if result[-1] == 'True':
224              result = result[:-2]
225              result.append(')')
226          else:
227              result.append(' else None)')
228          return ''.join(result)

230      def _print_Relational(self, expr):
231          "Relational printer for Equality and Unequality"
232          op = {
233              '==' :'equal',
234              '!=' :'not_equal',
235              '<'  :'less',
236              '<=' :'less_equal',
237              '>'  :'greater',
238              '>=' :'greater_equal',
239          }
240          if expr.rel_op in op:
241              lhs = self._print(expr.lhs)
242              rhs = self._print(expr.rhs)
243              return '({lhs} {op} {rhs})'.format(op=expr.rel_op, lhs=lhs, rhs=rhs)
244          return super()._print_Relational(expr)

349      def _hprint_Pow(self, expr, rational=False, sqrt='math.sqrt'):
350          """Printing helper function for ``Pow`` 
351   
352          Notes 
353          ===== 
354   
355          This preprocesses the ``sqrt`` as math formatter and prints division 
356   
357          Examples 
358          ======== 
359   
360          >>> from sympy import sqrt 
361          >>> from sympy.printing.pycode import PythonCodePrinter 
362          >>> from sympy.abc import x 
363   
364          Python code printer automatically looks up ``math.sqrt``. 
365   
366          >>> printer = PythonCodePrinter() 
367          >>> printer._hprint_Pow(sqrt(x), rational=True) 
368          'x**(1/2)' 
369          >>> printer._hprint_Pow(sqrt(x), rational=False) 
370          'math.sqrt(x)' 
371          >>> printer._hprint_Pow(1/sqrt(x), rational=True) 
372          'x**(-1/2)' 
373          >>> printer._hprint_Pow(1/sqrt(x), rational=False) 
374          '1/math.sqrt(x)' 
375          >>> printer._hprint_Pow(1/x, rational=False) 
376          '1/x' 
377          >>> printer._hprint_Pow(1/x, rational=True) 
378          'x**(-1)' 
379   
380          Using sqrt from numpy or mpmath 
381   
382          >>> printer._hprint_Pow(sqrt(x), sqrt='numpy.sqrt') 
383          'numpy.sqrt(x)' 
384          >>> printer._hprint_Pow(sqrt(x), sqrt='mpmath.sqrt') 
385          'mpmath.sqrt(x)' 
386   
387          See Also 
388          ======== 
389   
390          sympy.printing.str.StrPrinter._print_Pow 
391          """
392          PREC = precedence(expr)
393  
394          if expr.exp == S.Half and not rational:
395              func = self._module_format(sqrt)
396              arg = self._print(expr.base)
397              return '{func}({arg})'.format(func=func, arg=arg)
398  
399          if expr.is_commutative and not rational:
400              if -expr.exp is S.Half:
401                  func = self._module_format(sqrt)
402                  num = self._print(S.One)
403                  arg = self._print(expr.base)
404                  return f"{num}/{func}({arg})"
405              if expr.exp is S.NegativeOne:
406                  num = self._print(S.One)
407                  arg = self.parenthesize(expr.base, PREC, strict=False)
408                  return f"{num}/{arg}"
409  
410  
411          base_str = self.parenthesize(expr.base, PREC, strict=False)
412          exp_str = self.parenthesize(expr.exp, PREC, strict=False)
413          return "{}**{}".format(base_str, exp_str)

425      def _get_einsum_string(self, subranks, contraction_indices):
426          letters = self._get_letter_generator_for_einsum()
427          contraction_string = ""
428          counter = 0
429          d = {j: min(i) for i in contraction_indices for j in i}
430          indices = []
431          for rank_arg in subranks:
432              lindices = []
433              for i in range(rank_arg):
434                  if counter in d:
435                      lindices.append(d[counter])
436                  else:
437                      lindices.append(counter)
438                  counter += 1
439              indices.append(lindices)
440          mapping = {}
441          letters_free = []
442          letters_dum = []
443          for i in indices:
444              for j in i:
445                  if j not in mapping:
446                      l = next(letters)
447                      mapping[j] = l
448                  else:
449                      l = mapping[j]
450                  contraction_string += l
451                  if j in d:
452                      if l not in letters_dum:
453                          letters_dum.append(l)
454                  else:
455                      letters_free.append(l)
456              contraction_string += ","
457          contraction_string = contraction_string[:-1]
458          return contraction_string, letters_free, letters_dum

460      def _get_letter_generator_for_einsum(self):
461          for i in range(97, 123):
462              yield chr(i)
463          for i in range(65, 91):
464              yield chr(i)
465          raise ValueError("out of letters")

467      def _print_ArrayTensorProduct(self, expr):
468          letters = self._get_letter_generator_for_einsum()
469          contraction_string = ",".join(["".join([next(letters) for j in range(i)]) for i in expr.subranks])
470          return '%s("%s", %s)' % (
471                  self._module_format(self._module + "." + self._einsum),
472                  contraction_string,
473                  ", ".join([self._print(arg) for arg in expr.args])
474          )

476      def _print_ArrayContraction(self, expr):
477          from sympy.tensor.array.expressions.array_expressions import ArrayTensorProduct
478          base = expr.expr
479          contraction_indices = expr.contraction_indices
480  
481          if isinstance(base, ArrayTensorProduct):
482              elems = ",".join(["%s" % (self._print(arg)) for arg in base.args])
483              ranks = base.subranks
484          else:
485              elems = self._print(base)
486              ranks = [len(base.shape)]
487  
488          contraction_string, letters_free, letters_dum = self._get_einsum_string(ranks, contraction_indices)
489  
490          if not contraction_indices:
491              return self._print(base)
492          if isinstance(base, ArrayTensorProduct):
493              elems = ",".join(["%s" % (self._print(arg)) for arg in base.args])
494          else:
495              elems = self._print(base)
496          return "%s(\"%s\", %s)" % (
497              self._module_format(self._module + "." + self._einsum),
498              "{}->{}".format(contraction_string, "".join(sorted(letters_free))),
499              elems,
500          )

502      def _print_ArrayDiagonal(self, expr):
503          from sympy.tensor.array.expressions.array_expressions import ArrayTensorProduct
504          diagonal_indices = list(expr.diagonal_indices)
505          if isinstance(expr.expr, ArrayTensorProduct):
506              subranks = expr.expr.subranks
507              elems = expr.expr.args
508          else:
509              subranks = expr.subranks
510              elems = [expr.expr]
511          diagonal_string, letters_free, letters_dum = self._get_einsum_string(subranks, diagonal_indices)
512          elems = [self._print(i) for i in elems]
513          return '%s("%s", %s)' % (
514              self._module_format(self._module + "." + self._einsum),
515              "{}->{}".format(diagonal_string, "".join(letters_free+letters_dum)),
516              ", ".join(elems)
517          )

610  def pycode(expr, **settings):
611      """ Converts an expr to a string of Python code 
612   
613      Parameters 
614      ========== 
615   
616      expr : Expr 
617          A SymPy expression. 
618      fully_qualified_modules : bool 
619          Whether or not to write out full module names of functions 
620          (``math.sin`` vs. ``sin``). default: ``True``. 
621      standard : str or None, optional 
622          Only 'python3' (default) is supported. 
623          This parameter may be removed in the future. 
624   
625      Examples 
626      ======== 
627   
628      >>> from sympy import pycode, tan, Symbol 
629      >>> pycode(tan(Symbol('x')) + 1) 
630      'math.tan(x) + 1' 
631   
632      """
633      return PythonCodePrinter(settings).doprint(expr)

734  def _unpack_integral_limits(integral_expr):
735      """ helper function for _print_Integral that 
736          - accepts an Integral expression 
737          - returns a tuple of 
738             - a list variables of integration 
739             - a list of tuples of the upper and lower limits of integration 
740      """
741      integration_vars = []
742      limits = []
743      for integration_range in integral_expr.limits:
744          if len(integration_range) == 3:
745              integration_var, lower_limit, upper_limit = integration_range
746          else:
747              raise NotImplementedError("Only definite integrals are supported")
748          integration_vars.append(integration_var)
749          limits.append((lower_limit, upper_limit))
750      return integration_vars, limits
