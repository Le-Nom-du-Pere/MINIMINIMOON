16   PARAM_NAMES = "query key value attn_mask dropout is_causal enable_gqa".split()
17   
18   
19   class SDPAParamsVariable(VariableTracker):
20       """Represents the c++ params struct for scaled dot product attention. 
21       This is a read-only container."""
22   
23       @staticmethod
24       def create(tx: "InstructionTranslator", value, source):
25           from torch.backends.cuda import SDPAParams
26   
27           from .torch import TorchInGraphFunctionVariable
28   
29           params = [
30               VariableTracker.build(tx, getattr(value, p), AttrSource(source, p))
31               for p in PARAM_NAMES
32           ]
33           return TorchInGraphFunctionVariable(SDPAParams).call_function(tx, params, {})
34   
35       def __init__(self, proxy, param_vars, **kwargs) -> None:
36           self.proxy = proxy
37           self.param_vars = param_vars
38           super().__init__(**kwargs)
39   
40       def reconstruct(self, codegen: "PyCodegen"):
41           assert self.source is None
42           assert self.param_vars is not None
43           codegen.add_push_null(
44               lambda: codegen.load_import_from("torch._C", "_SDPAParams")
45           )
46           codegen.foreach(self.param_vars)
47           codegen.extend_output(create_call_function(len(self.param_vars), False))
48   
49       def as_proxy(self):
50           return self.proxy
51   
52       def var_getattr(self, tx: "InstructionTranslator", name: str) -> VariableTracker:
53           import torch._C
54   
55           from .builder import wrap_fx_proxy
56           from .misc import GetAttrVariable
57   
58           try:
59               getattr_static(torch._C._SDPAParams, name)
60           except AttributeError:
61               # Using raise from is too verbose here
62               raise Unsupported(
63                   f"Unsupported torch._C._SDPAParams attribute {name}"
64               ) from None
65   
66           proxy = GetAttrVariable.create_getattr_proxy(self.as_proxy(), name)
67           if self.source is not None:
68               return wrap_fx_proxy(
69                   tx=tx, proxy=proxy, source=AttrSource(self.source, name)
70               )
71           else:
72               return wrap_fx_proxy(tx=tx, proxy=proxy)
73   
74       @staticmethod
75       def is_sdpa_params(value):
76           from torch.backends.cuda import SDPAParams
77   
78           return value is SDPAParams
