2    from typing import Optional

4    import torch.fx
5    from torch._subclasses.fake_tensor import FakeTensor, FakeTensorMode
6    from torch.fx import Node
7    from torch.fx._compatibility import compatibility
8    from torch.fx.experimental.proxy_tensor import py_sym_types, snapshot_fake
9    from torch.fx.node import map_aggregate
10   from torch.utils._ordered_set import OrderedSet
11   
12   
13   __all__ = ["FakeTensorProp"]
14   
15   
16   @compatibility(is_backward_compatible=False)
17   class FakeTensorProp(torch.fx.Interpreter):
18       """ 
19       Execute an FX graph Node-by-Node and record a fake tensor representing 
20       the metadata for the node.  Unlike ShapeProp, (1) this propagation 
21       is cheap--it does the propagation with meta tensors which do not actually 
22       store data, and (2) the fake tensors have much more fine grained information, 
23       e.g., they have accurate alias information that can be consulted by looking 
24       at the storages. 
25    
26       Args: 
27            module (GraphModule): The module to be executed 
28            mode (Optional[FakeTensorMode]): The dispatch mode used to execute computation indicated by each FX Node. 
29       """
30   
31       def __init__(
32           self, module: torch.fx.GraphModule, mode: Optional[FakeTensorMode] = None
33       ):
34           super().__init__(module)
35           if mode is None:
36               mode = FakeTensorMode()
37           self._mode = mode
38           mode.epoch += 1
39           mode.reset_nt_tensor_id_counter()
40           self.seen_subgraphs: OrderedSet[str] = OrderedSet()
41   
42       def run_node(self, n: Node):
43           from torch.fx.experimental.symbolic_shapes import (
44               compute_unbacked_bindings,
45               rebind_unbacked,
46           )
47   
48           if (
49               n.op == "call_function"
50               and n.target is torch.ops.higher_order.invoke_subgraph
51               and n.args[1] not in self.seen_subgraphs
52           ):
53               # Prevent redundant fake tensor prop for invoke_subgraphs. Note that
54               # there is also fake tensor caching for the entire subgraph. This
55               # happens the next time we call `run_node` for the same subgraph,
56               # which goes through super.run_node and caches the fake tensor prop.
57               # Therefore, we are propagating fake tensor through the subgraphs
58               # twice.
59               assert isinstance(n.args[1], str)
60               assert (
61                   isinstance(n.args[0], torch.fx.Node)
62                   and n.args[0].op == "get_attr"
63                   and isinstance(n.args[0].target, str)
64               )
65               self.seen_subgraphs.add(n.args[1])
66               operands = n.args[2:]
67               example_inputs = []
68               for operand in operands:
69                   assert isinstance(operand, torch.fx.Node) and "val" in operand.meta
70                   example_inputs.append(operand.meta["val"])
71               return FakeTensorProp(
72                   getattr(self.module, n.args[0].target), mode=self._mode
73               ).propagate(*example_inputs)
74   
75           result = super().run_node(n)
76           rebind_unbacked(self._mode.shape_env, n, result)
77   
78           def extract_val(obj):
79               if isinstance(obj, FakeTensor):
80                   return snapshot_fake(obj)
81               elif isinstance(obj, torch.Tensor):
82                   # TODO: How is it possible that we get a non fake tensor?  We
83                   # should be running under the mode...
84                   return snapshot_fake(self._mode.from_tensor(obj, static_shapes=True))
85               elif isinstance(obj, py_sym_types):
86                   return obj
87               else:
88                   return None
89   
90           meta = map_aggregate(result, extract_val)
91           if meta is not None:
92               n.meta["val"] = meta
93               if (shape_env := self._mode.shape_env) and (
94                   symbol_to_path := compute_unbacked_bindings(shape_env, result)
95               ):
96                   n.meta["unbacked_bindings"] = symbol_to_path
97   
98           return result
99   
100      def propagate(self, *args):
101          fake_args = [
102              self._mode.from_tensor(a) if isinstance(a, torch.Tensor) else a
103              for a in args
104          ]
105          return self.propagate_dont_convert_inputs(*fake_args)
106  
107      def propagate_dont_convert_inputs(self, *args):
108          with self._mode:
109              return super().run(*args)
