67   /// KleidiAI data types

74       kai_dt_int32 = 0x0421,
75       kai_dt_int16 = 0x0222,
76       kai_dt_int8 = 0x0124,
77       kai_dt_uint32 = 0x0431,
78       kai_dt_uint16 = 0x0232,
79       kai_dt_uint8 = 0x0134,
80       kai_dt_bool = 0x0441

83   /// Gets number of bytes for a given data type
84   /// @param[in] dt KleidiAI data type
85   ///
86   /// @return the numbers of bytes for the data type
87   inline static size_t kai_get_datatype_size_in_bytes(enum kai_datatype dt) {
88       return (size_t)(dt >> 8);
89   }

91   /// Converts a scalar f16 value to f32
92   /// @param[in] f16 The f16 value
93   ///
94   /// @return the f32 value
95   #if defined(__ARM_NEON)
96   inline static float kai_cast_f32_f16(uint16_t f16) {
97       float16_t f32 = 0;
98       memcpy(&f32, &f16, sizeof(uint16_t));
99       return (float)f32;
100  }
101  #endif

103  /// Converts a scalar bf16 value to f32
104  /// @param[in] bf16 The f16 value
105  ///
106  /// @return the f32 value
107  inline static float kai_cast_f32_bf16(uint16_t bf16) {
108      const uint32_t i32 = (bf16 << 16);
109      float f32 = 0;
110      memcpy(&f32, &i32, sizeof(i32));
111      return f32;
112  }
113  
114  /// Converts a f32 value to bf16
115  /// @param[in] f32 The f32 value
116  ///
117  /// @return the bf16 value
118  inline static uint16_t kai_cast_bf16_f32(float f32) {
119      uint16_t bf16 = 0;
120  #ifdef __ARM_FEATURE_BF16
121      __asm__ __volatile__("bfcvt %h[output], %s[input]" : [output] "=w"(bf16) : [input] "w"(f32));
122  #else
123      const uint32_t* i32 = (uint32_t*)(&f32);
124      bf16 = (*i32 >> 16);
125  #endif
126      return bf16;
127  }
128  
129  /// Converts a scalar f32 value to f16
130  /// @param[in] f32 The f32 value
131  ///
132  /// @return the f16 value
133  #if defined(__ARM_NEON)
134  inline static uint16_t kai_cast_f16_f32(float f32) {
135      uint16_t f16 = 0;
136      float16_t tmp = (float16_t)f32;
137      memcpy(&f16, &tmp, sizeof(uint16_t));
138      return f16;
139  }

142  inline static size_t kai_roundup(size_t a, size_t b) {
143      return ((a + b - 1) / b) * b;
144  }

146  #ifdef __ARM_FEATURE_SVE2
147  /// Gets the SME vector length for 8-bit elements.
148  inline static uint64_t kai_get_sme_vector_length_u8(void) {
149      uint64_t res = 0;
150      __asm__ __volatile__(
151          ".inst 0x04bf5827 // rdsvl x7, #1\n"
152          "mov %0, x7\n"
153          : "=r"(res)
154          : /* no inputs */
155          : "x7");
156      return res;
157  }
158  
159  /// Gets the SME vector length for 16-bit elements.
160  inline static uint64_t kai_get_sme_vector_length_u16(void) {
161      return kai_get_sme_vector_length_u8() / 2;
162  }
163  
164  /// Gets the SME vector length for 32-bit elements.
165  inline static uint64_t kai_get_sme_vector_length_u32(void) {
166      return kai_get_sme_vector_length_u8() / 4;
167  }
168  #endif  // __ARM_FEATURE_SVE2

170  /// Extends the sign bit of int 4-bit value (stored in int8_t variable)
171  /// @param[in] value The 4-bit int value
172  ///
173  /// @return the int8_t value with sign extended
174  inline static int8_t kai_ext_sign_i8_i4(int8_t value) {
175      // Make sure value holds correct int4 value
176      KAI_ASSERT(value <= 0xF);
177  
178      return (value ^ 0x8) - 8;  // NOLINT(bugprone-narrowing-conversions,cppcoreguidelines-narrowing-conversions)
179  }
