36   def sT(expr, string, import_stmt=None, **kwargs):
37       """ 
38       sT := sreprTest 
39    
40       Tests that srepr delivers the expected string and that 
41       the condition eval(srepr(expr))==expr holds. 
42       """
43       if import_stmt is None:
44           ENV2 = ENV
45       else:
46           ENV2 = ENV.copy()
47           exec(import_stmt, ENV2)
48   
49       assert srepr(expr, **kwargs) == string
50       assert eval(string, ENV2) == expr
