55     static Vectorized<BFloat16> blendv(
56         const Vectorized<BFloat16>& a,
57         const Vectorized<BFloat16>& b,
58         const Vectorized<BFloat16>& mask_) {
59       svbool_t mask =
60           svcmpeq_s16(ptrue, svreinterpret_s16_bf16(mask_), ALL_S16_TRUE_MASK);
61       return svsel_bf16(mask, b, a);
62     }
63     template <typename step_t>
64     static Vectorized<BFloat16> arange(
65         BFloat16 base = 0.f,
66         step_t step = static_cast<step_t>(1)) {
67       __at_align__ BFloat16 buffer[size()];
68       for (int64_t i = 0; i < size(); i++) {
69         buffer[i] = base + i * step;
70       }
71       return svld1_bf16(ptrue, reinterpret_cast<bfloat16_t*>(buffer));
72     }
73     static Vectorized<BFloat16> set(
74         const Vectorized<BFloat16>& a,
75         const Vectorized<BFloat16>& b,
76         int64_t count = size()) {
77       if (count == 0) {
78         return a;
79       } else if (count < size()) {
80         return svsel_bf16(svwhilelt_b16(0ull, count), b, a);
81       }
82       return b;
83     }
84     static Vectorized<BFloat16> loadu(const void* ptr, int64_t count = size()) {
85       if (count == size())
86         return svld1_bf16(ptrue, reinterpret_cast<const bfloat16_t*>(ptr));
87       svbool_t pg = svwhilelt_b16(0ull, count);
88       return svld1_bf16(pg, reinterpret_cast<const bfloat16_t*>(ptr));
89     }
90     void store(void* ptr, int64_t count = size()) const {
91       __at_align__ bfloat16_t tmp[size()];
92       std::memset(tmp, 0, sizeof(tmp));
93       if (count == size()) {
94         svst1_bf16(ptrue, reinterpret_cast<bfloat16_t*>(tmp), values);
95       } else {
96         svbool_t pg = svwhilelt_b16(0ull, count);
97         svst1_bf16(pg, reinterpret_cast<bfloat16_t*>(tmp), values);
98       }
99       std::memcpy(
100          reinterpret_cast<bfloat16_t*>(ptr),
101          reinterpret_cast<const bfloat16_t*>(tmp),
102          count * sizeof(bfloat16_t));
103    }
104    const BFloat16& operator[](int idx) const = delete;
105    BFloat16& operator[](int idx) = delete;
106    int64_t zero_mask() const {
107      int64_t mask = 0;
108      // returns an integer mask where all zero elements are translated to
109      // 1-bit and others are translated to 0-bit int64_t mask = 0;
110      __at_align__ int16_t mask_array[size()];
111  
112      svbool_t svbool_mask =
113          svcmpeq_f16(ptrue, svreinterpret_f16_bf16(values), ZERO_F16);
114      svst1_s16(
115          ptrue,
116          mask_array,
117          svsel_s16(svbool_mask, ALL_S16_TRUE_MASK, ALL_S16_FALSE_MASK));
118      for (int64_t i = 0; i < size(); ++i) {
119        if (mask_array[i])
120          mask |= (1ull << i);
121      }
122      return mask;
123    }

223  inline std::tuple<Vectorized<float>, Vectorized<float>> convert_bfloat16_float(
224      const Vectorized<c10::BFloat16>& a) {
225    static_assert(
226        Vectorized<c10::BFloat16>::size() == 2 * Vectorized<float>::size());
227    auto zero = svreinterpret_bf16_f32(svdup_n_f32(0.0f));
228    auto bf16_vec1 = svzip1_bf16(zero, a);
229    auto bf16_vec2 = svzip2_bf16(zero, a);
230    auto x1 = svreinterpret_f32_bf16(bf16_vec1);
231    auto x2 = svreinterpret_f32_bf16(bf16_vec2);
232    return {Vectorized<float>(x1), Vectorized<float>(x2)};
233  }

235  inline Vectorized<c10::BFloat16> convert_float_bfloat16(
236      const Vectorized<float>& a,
237      const Vectorized<float>& b) {
238    static_assert(
239        Vectorized<c10::BFloat16>::size() == 2 * Vectorized<float>::size());
240    svbfloat16_t x1 = svcvt_bf16_f32_z(ptrue, a);
241    svbfloat16_t x2 = svcvt_bf16_f32_z(ptrue, b);
242    return Vectorized<c10::BFloat16>(svuzp1_bf16(x1, x2));
243  }

245  inline void load_fp32_from_bf16(const BFloat16* data, Vectorized<float>& out) {
246    __at_align__ float values[Vectorized<float>::size()];
247    for (const auto k : c10::irange(Vectorized<float>::size())) {
248      values[k] = data[k];
249    }
250    out = Vectorized<float>::loadu(values);
251  }

253  inline void load_fp32_from_bf16(
254      const BFloat16* data,
255      Vectorized<float>& out1,
256      Vectorized<float>& out2) {
257    Vectorized<BFloat16> bf16_vec = Vectorized<BFloat16>::loadu(data);
258    auto floats = convert_bfloat16_float(bf16_vec);
259    out1 = std::get<0>(floats);
260    out2 = std::get<1>(floats);
261  }

263  template <typename Op>
264  Vectorized<c10::BFloat16> binary_operator_via_float(
265      Op op,
266      const Vectorized<c10::BFloat16>& a,
267      const Vectorized<c10::BFloat16>& b) {
268    const auto [a_float_low, a_float_high] = convert_bfloat16_float(a);
269    const auto [b_float_low, b_float_high] = convert_bfloat16_float(b);
270    return convert_float_bfloat16(
271        op(a_float_low, b_float_low), op(a_float_high, b_float_high));
272  }

381  Vectorized<BFloat16> inline Vectorized<BFloat16>::operator==(
382      const Vectorized<BFloat16>& other) const {
383    auto [f1, f2] = convert_bfloat16_float(values);
384    auto [f3, f4] = convert_bfloat16_float(other);
385    svbool_t mask1 = svcmpeq_f32(ptrue, f1, f3);
386    svbool_t mask2 = svcmpeq_f32(ptrue, f2, f4);
387    auto res1 = svsel_f32(mask1, ALL_F32_TRUE_MASK, ALL_F32_FALSE_MASK);
388    auto res2 = svsel_f32(mask2, ALL_F32_TRUE_MASK, ALL_F32_FALSE_MASK);
389  
390    auto bf16_1 = svreinterpret_bf16_f32(res1);
391    auto bf16_2 = svreinterpret_bf16_f32(res2);
392    return svuzp1_bf16(bf16_1, bf16_2);
393  }
394  Vectorized<BFloat16> inline Vectorized<BFloat16>::operator!=(
395      const Vectorized<BFloat16>& other) const {
396    auto [f1, f2] = convert_bfloat16_float(values);
397    auto [f3, f4] = convert_bfloat16_float(other);
398    svbool_t mask1 = svcmpne_f32(ptrue, f1, f3);
399    svbool_t mask2 = svcmpne_f32(ptrue, f2, f4);
400    auto res1 = svsel_f32(mask1, ALL_F32_TRUE_MASK, ALL_F32_FALSE_MASK);
401    auto res2 = svsel_f32(mask2, ALL_F32_TRUE_MASK, ALL_F32_FALSE_MASK);
402  
403    auto bf16_1 = svreinterpret_bf16_f32(res1);
404    auto bf16_2 = svreinterpret_bf16_f32(res2);
405    return svuzp1_bf16(bf16_1, bf16_2);
406  }
