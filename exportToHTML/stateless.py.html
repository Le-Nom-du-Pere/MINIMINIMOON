14   def _untie_named_tensors_map(
15       module: "torch.nn.Module",
16       parameters_and_buffers: dict[str, Tensor],
17   ) -> dict[str, Tensor]:
18       """ 
19       Unties all tied tensors in the module to parameters_and_buffers. 
20    
21       This function returns a new untied_parameters_and_buffers dictionary and leave the original 
22       untied_parameters_and_buffers dictionary unchanged. It adds new (missing) keys for tied tensors 
23       in the module to untied_parameters_and_buffers. The value of the new key is the user-given value 
24       in the original parameters_and_buffers dictionary. 
25    
26       If there are more than one user-given values for the same tied tensor, it will raise an error. 
27    
28       For example, if the module has two tied weights self.foo and self.tied_foo and the user passes 
29       {'foo': foo_value, ...}, this will return {'foo': foo_value, 'tied_foo': foo_value, ...}. If the 
30       user passes {'foo': foo_value, 'tied_foo': tied_foo_value, ...}, it will raise an error. If the 
31       user passes {'foo': foo_value, 'tied_foo': foo_value, ...}, it will not raise an error. 
32    
33       Args: 
34           module (torch.nn.Module): the module to determine which tensors are tied. 
35           parameters_and_buffers (Dict[str, Tensor]): a map of {name: tensor} for reparamaterizing the module. 
36    
37       Returns: 
38           A new untied version of the parameters_and_buffers dictionary. 
39    
40       Raises: 
41           ValueError: if there are more than one user-given values for the same tied tensor. 
42       """
43       # A map of {name: tensor} for all tensors (including tied ones) in the module.
44       all_named_tensors: dict[str, Tensor] = {}
45       all_named_tensors.update(module.named_parameters(remove_duplicate=False))
46       all_named_tensors.update(module.named_buffers(remove_duplicate=False))
47   
48       # A map of {tensor: set(all_tied_names)} for all tensor names in the module.
49       tensor_to_tied_names_map: dict[Tensor, set[str]] = {}
50       for name, tensor in all_named_tensors.items():
51           if tensor not in tensor_to_tied_names_map:
52               tensor_to_tied_names_map[tensor] = set()
53           tensor_to_tied_names_map[tensor].add(name)
54   
55       # A map of {tied_name: set(all_tied_names)} for all tensor names in the module.
56       # If a name is not tied, it will not be in this map.
57       tied_names_map: dict[str, set[str]] = {}
58       for tied_names in tensor_to_tied_names_map.values():
59           if len(tied_names) > 1:
60               for tied_name in tied_names:
61                   tied_names_map[tied_name] = tied_names
62   
63       # Make sure the user didn't pass multiple values for the same tied tensor.
64       given_names = set(parameters_and_buffers.keys())
65       # same as given_names.intersection(tied_names_map.keys()) but dynamo can't
66       # handle that
67       given_names_for_tied_tensors: set[str] = set()
68       for name in given_names:
69           if name in tied_names_map:
70               given_names_for_tied_tensors.add(name)
71   
72       for given_name in given_names_for_tied_tensors:
73           tied_names = tied_names_map[given_name]
74           if (
75               # Detect if there are multiple keys present for the same tied tensor.
76               len(tied_names.intersection(given_names_for_tied_tensors)) > 1
77               # Only raise an error if the user passed multiple values for the same tied tensor.
78               # If all given values are the same, don't raise.
79               and len({parameters_and_buffers[tied_name] for tied_name in tied_names})
80               != 1
81           ):
82               raise ValueError(
83                   f"functional_call got multiple values for keys {sorted(tied_names)}, "
84                   f"which are tied. Consider using tie_weights=False"
85               )
86   
87       # Untie the given named tensor map
88       # Make a copy for not modifying the original dict
89       untied_parameters_and_buffers = parameters_and_buffers.copy()
90       for given_name in given_names_for_tied_tensors:
91           for tied_name in tied_names_map[given_name]:
92               untied_parameters_and_buffers[tied_name] = parameters_and_buffers[
93                   given_name
94               ]
95       return untied_parameters_and_buffers

98   @contextlib.contextmanager
99   def _reparametrize_module(
100      module: "torch.nn.Module",
101      parameters_and_buffers: dict[str, Tensor],
102      tie_weights: bool = False,
103      strict: bool = False,
104      stack_weights: bool = False,
105  ):
106      parameters_and_buffers = parameters_and_buffers
107      stack_weights = stack_weights
108  
109      if tie_weights:
110          untied_parameters_and_buffers = _untie_named_tensors_map(
111              module, parameters_and_buffers
112          )
113      else:
114          untied_parameters_and_buffers = parameters_and_buffers
115  
116      accessor = NamedMemberAccessor(module)
117      if strict:
118          missing_keys, unexpected_keys = accessor.check_keys(
119              untied_parameters_and_buffers
120          )
121          error_msgs = []
122          if len(unexpected_keys) > 0:
123              error_msgs.append(
124                  f"Unexpected key(s): {', '.join(map(repr, unexpected_keys))}."
125              )
126          if len(missing_keys) > 0:
127              error_msgs.append(f"Missing key(s): {', '.join(map(repr, missing_keys))}.")
128          if len(error_msgs) > 0:
129              raise RuntimeError(
130                  "Error(s) in reparametrizing for {}:\n\t{}".format(
131                      module._get_name(), "\n\t".join(error_msgs)
132                  )
133              )
134  
135      orig_parameters_and_buffers: dict[str, Tensor] = {}
136      try:
137          orig_parameters_and_buffers, _ = accessor.swap_tensors_dict(
138              untied_parameters_and_buffers, allow_missing=True
139          )
140          yield
141      finally:
142          if stack_weights:
143              # When stacking is enabled, we will restore the weights in LIFO order.
144              orig_parameters_and_buffers = dict(
145                  reversed(orig_parameters_and_buffers.items())
146              )
147          new_parameters_and_buffers, _ = accessor.swap_tensors_dict(
148              orig_parameters_and_buffers, allow_missing=True
149          )
150          # Sometimes the module is not completely stateless and has some in-place modifications on
151          # the _parameters and _buffers dictionaries.
152          # Write the changed parameters and buffers back to the original dict.
153          parameters_and_buffers.update(
154              {
155                  k: new_parameters_and_buffers[k]
156                  for k in parameters_and_buffers
157                  if k in new_parameters_and_buffers
158              }
159          )

162  @deprecated(
163      "`torch.nn.utils.stateless.functional_call` is deprecated as of PyTorch 2.0 "
164      "and will be removed in a future version of PyTorch. "
165      "Please use `torch.func.functional_call` instead which is a drop-in replacement.",
166      category=FutureWarning,
167  )
168  def functional_call(
169      module: "torch.nn.Module",
170      parameters_and_buffers: dict[str, Tensor],
171      args: Optional[Union[Any, tuple]] = None,
172      kwargs: Optional[dict[str, Any]] = None,
173      *,
174      tie_weights: bool = True,
175      strict: bool = False,
176  ):
177      r"""Perform a functional call on the module by replacing the module parameters and buffers with the provided ones. 
178   
179      .. warning:: 
180   
181          This API is deprecated as of PyTorch 2.0 and will be removed in a future 
182          version of PyTorch. Please use :func:`torch.func.functional_call` instead, 
183          which is a drop-in replacement for this API. 
184   
185      .. note:: If the module has active parametrizations, passing a value in the 
186          :attr:`parameters_and_buffers` argument with the name set to the regular parameter 
187          name will completely disable the parametrization. 
188          If you want to apply the parametrization function to the value passed 
189          please set the key as ``{submodule_name}.parametrizations.{parameter_name}.original``. 
190   
191      .. note:: If the module performs in-place operations on parameters/buffers, these will be reflected 
192          in the `parameters_and_buffers` input. 
193   
194          Example:: 
195   
196              >>> a = {'foo': torch.zeros(())} 
197              >>> # xdoctest: +SKIP 
198              >>> mod = Foo()  # does self.foo = self.foo + 1 
199              >>> print(mod.foo)  # tensor(0.) 
200              >>> functional_call(mod, a, torch.ones(())) 
201              >>> print(mod.foo)  # tensor(0.) 
202              >>> print(a['foo'])  # tensor(1.) 
203   
204      .. note:: If the module has tied weights, whether or not functional_call respects the tying is determined by the 
205          tie_weights flag. 
206   
207          Example:: 
208   
209              >>> a = {'foo': torch.zeros(())} 
210              >>> # xdoctest: +SKIP 
211              >>> mod = Foo()  # has both self.foo and self.foo_tied which are tied. Returns x + self.foo + self.foo_tied 
212              >>> print(mod.foo)  # tensor(1.) 
213              >>> mod(torch.zeros(()))  # tensor(2.) 
214              >>> functional_call(mod, a, torch.zeros(()))  # tensor(0.) since it will change self.foo_tied too 
215              >>> functional_call(mod, a, torch.zeros(()), tie_weights=False)  # tensor(1.)--self.foo_tied is not updated 
216              >>> new_a = {'foo': torch.zeros(()), 'foo_tied': torch.zeros(())} 
217              >>> functional_call(mod, new_a, torch.zeros()) # tensor(0.) 
218   
219      Args: 
220          module (torch.nn.Module): the module to call 
221          parameters_and_buffers (dict of str and Tensor): the parameters that will be used in 
222              the module call. 
223          args (Any or tuple): arguments to be passed to the module call. If not a tuple, considered a single argument. 
224          kwargs (dict): keyword arguments to be passed to the module call 
225          tie_weights (bool, optional): If True, then parameters and buffers tied in the original model will be treated as 
226              tied in the reparamaterized version. Therefore, if True and different values are passed for the tied 
227              parameters and buffers, it will error. If False, it will not respect the originally tied parameters and 
228              buffers unless the values passed for both weights are the same. Default: True. 
229          strict (bool, optional): If True, then the parameters and buffers passed in must match the parameters and 
230              buffers in the original module. Therefore, if True and there are any missing or unexpected keys, it will 
231              error. Default: False. 
232   
233      Returns: 
234          Any: the result of calling ``module``. 
235      """
236      return _functional_call(
237          module,
238          parameters_and_buffers,
239          args,
240          kwargs,
241          tie_weights=tie_weights,
242          strict=strict,
243      )

246  def _functional_call(
247      module: "torch.nn.Module",
248      parameters_and_buffers: dict[str, Tensor],
249      args: Optional[Union[Any, tuple]] = None,
250      kwargs: Optional[dict[str, Any]] = None,
251      *,
252      tie_weights: bool = True,
253      strict: bool = False,
254  ):
255      # TODO allow kwargs such as unsafe and others for parametrization
256      if (
257          torch.jit.is_tracing()
258          or torch.jit.is_scripting()
259          or isinstance(
260              module,
261              (
262                  torch.jit.RecursiveScriptModule,
263                  torch.jit.ScriptModule,
264                  torch.jit.ScriptFunction,
265              ),
266          )
267      ):
268          raise RuntimeError("The stateless API can't be used with Jitted modules")
269      if isinstance(module, torch.nn.DataParallel):
270          raise RuntimeError(
271              "The stateless API can't be used with nn.DataParallel module"
272          )
273      if kwargs is None:
274          kwargs = {}
275      if args is None:
276          args = ()
277      elif not isinstance(args, tuple):
278          args = (args,)
279      with _reparametrize_module(
280          module, parameters_and_buffers, tie_weights=tie_weights, strict=strict
281      ):
282          return module(*args, **kwargs)
