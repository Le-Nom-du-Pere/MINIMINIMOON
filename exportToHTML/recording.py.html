28   # [Note: Recording ShapeEnv Events]
29   # =================================
30   #
31   # What is a ShapeEnv event?
32   # -------------------------
33   # We consider a ShapeEnv event every function call (ShapeEnv method or
34   # independent function) that modifies the state of the ShapeEnv instance.
35   # Such calls are recorded alongside their positional and keyword arguments,
36   # so that it may be replayed over a different ShapeEnv instance.
37   #
38   # See [Note: ShapeEnv State Equality] for what is considered the state
39   # of a ShapeEnv instance.
40   #
41   # What is it for?
42   # ---------------
43   # ShapeEnv events recording is used for reconstructing the ShapeEnv in an
44   # arbitrary state in time.
45   #
46   # Being able to arbitrarily replay events like so is useful, mainly for
47   # translation validation bisection. i.e. if a ValidationException has been
48   # raised, find the earliest point in time where the translation validation
49   # fails.
50   #
51   # Besides that, it also allows us to inspect the given instance and,
52   # for example, check the guards that would actually be issued at that point.
53   #
54   # What kind of arguments can be stored in an event?
55   # -------------------------------------------------
56   # There's no specific rule for what cannot be used as an argument.
57   # That said, pay special attention to the following cases:
58   #
59   #   1. Tensor inputs: there are some tests that check whether the inputs
60   #      were garbage collected after execution. These will fail if there's
61   #      an event that is holding a reference to those inputs.
62   #
63   #   2. ShapeEnv arguments: if there is an argument of ShapeEnv type, that
64   #      will be automatically replaced by the new given ShapeEnv instance.
65   #
66   #   3. SymTypes arguments: they also hold references to ShapeEnv. So,
67   #      whenever we see them, we create a new instance, replacing the
68   #      ShapeEnv reference.
69   #
70   #   4. FX nodes: specifically, FX nodes from the FX graph for symbolic
71   #      shapes. That argument must be replaced when replaying the event at
72   #      ShapeEnvEvent.run, since it has to reference a node from the given
73   #      instance, and not from the recorded instance.
74   
75   
76   # Event class for reconstructing ShapeEnv at arbitrary time.
77   #
78   # Represents a method call that mutates ShapeEnv in a way that affects the
79   # issued guards, when ShapeEnv.produce_guards is called.
80   @dataclass
81   class ShapeEnvEvent:
82       # ShapeEnv method.
83       f: Callable
84   
85       # Arguments and keyword arguments called with.
86       args: Optional[list[Any]] = None
87       kwargs: Optional[dict[str, Any]] = None
88   
89       # List of tracked_fakes at the time the method was called.
90       tracked_fakes: Optional[list[Any]] = None
91   
92       # Name of the captured event.
93       # Used for special handling of particular methods.
94       name: Optional[str] = None
95   
96       # Replay itself, but using shape_env as self.
97       def run(self, shape_env=None) -> Any:
98           from torch.fx.experimental.symbolic_shapes import (
99               is_symbolic,
100              ShapeEnv,
101              SymTypes,
102          )
103  
104          # Special handling for the constructor event.
105          if self.f is ShapeEnv:
106              assert shape_env is None and self.args is None and self.kwargs is not None
107              return ShapeEnv(**self.kwargs)
108  
109          assert shape_env is not None
110          args = list(self.args or [])
111          kwargs = dict(self.kwargs or {})
112  
113          # Replace any argument of type ShapeEnv by the given one.
114          args, kwargs = pytree.tree_map_only(
115              ShapeEnv, lambda _: shape_env, (args, kwargs)
116          )
117  
118          # Replace any argument of type SymTypes by a new instance,
119          # replacing its ShapeEnv reference.
120          args, kwargs = pytree.tree_map_only(
121              lambda x: isinstance(x, SymTypes) and is_symbolic(x),
122              lambda a: type(a)(a.node.with_shape_env(shape_env)),
123              (args, kwargs),
124          )
125  
126          # Converts FX nodes using the mapping argument.
127          def maybe_convert_node(x: Any) -> Any:
128              if not isinstance(x, torch.fx.Node):
129                  # Don't do anything to x if it's not an FX node.
130                  return x
131  
132              # If, at some point, we created an FX node, it means that translation validation is on.
133              # It also means we are building an FX graph for symbolic shapes at shape_env.graph, and
134              # we are tracking node names at shape_env.name_to_node.
135              assert hasattr(shape_env, "name_to_node")
136              name_to_node = shape_env.name_to_node  # type: ignore[attr-defined]
137              assert x.name in name_to_node
138              return name_to_node[x.name]
139  
140          # Replaces the value of an specific argument by the result of fn.
141          def replacearg(index: int, key: str, fn: Callable):
142              if index < len(args):
143                  args[index] = fn(args[index])
144              if key in kwargs:
145                  kwargs[key] = fn(kwargs[key])
146  
147          if self.is_create_fx_call_function():
148              # ShapeEnv.create_fx_call_function:
149              # "args" parameter is a tuple of FX nodes from the FX graph of the old ShapeEnv.
150              # They must be replaced, since a "call_function" FX node with this tuple as argument
151              # will be added to the FX graph of the new shape_env.
152              replacearg(
153                  index=2,
154                  key="args",
155                  fn=lambda args: tuple(maybe_convert_node(a) for a in args),
156              )
157          if self.is_evaluate_expr() or self.is_defer_runtime_assert():
158              # ShapeEnv.evaluate_expr and ShapeEnv.guard_or_defer_runtime_assert:
159              # "fx_node" parameter is an (optional) FX node that represents the evaluate expression.
160              # They must be replaced, since it will be part of a "call_function" FX node for
161              # torch._assert, which will be added to the FX graph of the new shape_env.
162              replacearg(index=3, key="fx_node", fn=maybe_convert_node)
163  
164          # Actually call the method with the converted arguments.
165          return self.f(*args, **kwargs)
166  
167      def __str__(self) -> str:
168          name = self.name if self.name is not None else self.f.__name__
169          return f"event: {name} ({self.args}, {self.kwargs})"
170  
171      def is_create_fx_call_function(self) -> bool:
172          return self.name == "_create_fx_call_function"
173  
174      def is_evaluate_expr(self) -> bool:
175          return self.name == "evaluate_expr"
176  
177      def is_defer_runtime_assert(self) -> bool:
178          return self.name == "guard_or_defer_runtime_assert"

184  # Extracts a ShapeEnv instance inside args and kwargs.
185  # Specifically, it looks for:
186  #   1. ShapeEnv arguments
187  #   2. SymInt, SymFloat, or SymBool arguments
188  # If we find more than one object of any of the above types, we
189  # also check that the ShapeEnv instance is the same for all of them.
190  def _extract_shape_env_and_assert_equal(args, kwargs):
191      from torch.fx.experimental.symbolic_shapes import is_symbolic, ShapeEnv, SymTypes
192  
193      def assert_equal(old: Optional[ShapeEnv], new: ShapeEnv) -> ShapeEnv:
194          if old is not None:
195              assert old is new, "call with different ShapeEnv"
196          return new
197  
198      shape_env = None
199      for val in itertools.chain(args, kwargs.values()):
200          if isinstance(val, ShapeEnv):
201              shape_env = assert_equal(shape_env, val)
202          if isinstance(val, SymTypes) and is_symbolic(val):
203              shape_env = assert_equal(shape_env, val.node.shape_env)
204  
205      return shape_env

208  # Decorator for recording the given function as a replayable event.
209  #
210  # This decorator should be used at every function that mutates the state of
211  # ShapeEnv in some way that affects the resulting issued guards (i.e. when
212  # ShapeEnv.produce_guards is called).
213  #
214  # save_tracked_fakes: saves a snapshot of the TrackedFake list.
215  # This is used when calling ShapeEnv.produce_guards at arbitrary points in time.
216  #
217  # name: the name of the function being recorded. Normally (and by default) this
218  # is taken from the decorated function but can be set if you need to override
219  # it.
220  #
221  # When to save the list of TrackedFake?
222  # =====================================
223  # We should save the list of TrackedFake whenever the translation validation
224  # bisection may actually stop and call the produce_guards method at the moment
225  # right after the recorded function was played. In other words, since the
226  # bisection bisects through torch._assert calls, we should save in all methods
227  # that adds a torch._assert call to the symbolic shapes FX graph.
228  #
229  # At the moment, there are 2 methods that save the list:
230  #   - ShapeEnv.evaluate_expr
231  #   - ShapeEnv.guard_or_defer_runtime_assert
232  def record_shapeenv_event(
233      *, save_tracked_fakes: bool = False, name: Optional[str] = None
234  ) -> Callable:
235      def decorator(fn: Callable) -> Callable:
236          assert callable(fn)
237          args = inspect.getfullargspec(fn).args
238          assert args and args[0] == "self", (
239              "record_shapeenv_event should only wrap methods on ShapeEnv; refactor your "
240              "code so that it calls into a method on ShapeEnv"
241          )
242          nonlocal name
243          if name is None:
244              name = fn.__name__
245  
246          @functools.wraps(fn)
247          def wrapper(*args, **kwargs):
248              from torch.fx.experimental.symbolic_shapes import ShapeEnv
249  
250              assert isinstance(args[0], ShapeEnv)
251  
252              global NEST
253  
254              trace_shape_events_log.debug(
255                  "%scall %s(*%r, **%r)", " " * NEST, name, args[1:], kwargs
256              )
257              NEST += 1
258  
259              def retlog(r):
260                  trace_shape_events_log.debug("%s-> %s", " " * (NEST - 1), r)
261                  return r
262  
263              shape_env = args[0]
264  
265              try:
266                  if not shape_env.should_record_events or shape_env.is_recording:  # type: ignore[has-type]
267                      # If ShapeEnv is already recording an event, call the wrapped
268                      # function directly.
269                      #
270                      # NB: here, we skip the check of whether all ShapeEnv instances
271                      # are equal, in favor of a faster dispatch.
272                      return retlog(fn(*args, **kwargs))
273  
274                  # Retrieve an instance of ShapeEnv.
275                  # Assumption: the collection of args and kwargs may not reference
276                  # different ShapeEnv instances.
277                  self = _extract_shape_env_and_assert_equal(args, kwargs)
278  
279                  # If we are calling this function without any ShapeEnv instance
280                  # alive in its arguments, we don't record and call the original.
281                  if self is None:
282                      return retlog(fn(*args, **kwargs))
283  
284                  # Otherwise, start recording and call the function.
285                  with self._recording():
286                      # Take a snapshot of the current tracked_fakes.
287                      tracked_fakes = (
288                          self._snapshot_tracked_fakes() if save_tracked_fakes else None
289                      )
290                      # Record the event for 'fn'.
291                      event = ShapeEnvEvent(
292                          fn,
293                          list(args),
294                          kwargs,
295                          tracked_fakes,
296                          name=name,
297                      )
298                      # Play the event on this ShapeEnv.
299                      # NB: It's important to put the event first, because running
300                      # the event can trigger internal events that must be ordered
301                      # after this event.  However, if an exception happens, we do
302                      # NOT want to have the event in the list, so pop it off from
303                      # the record if an error happened
304                      self.events.append(event)
305                      try:
306                          return retlog(event.run(self))
307                      except Exception:
308                          self.events.pop()
309                          raise
310  
311              except Exception:
312                  if not shape_env.should_record_events or shape_env.is_recording:
313                      # If ShapeEnv is disabled or already recording an event, re-raise the exception without logging.
314                      raise
315                  log.error(  # noqa: G201
316                      "failed while running %s(*%s, **%s)",
317                      name,
318                      args[1:],
319                      kwargs,
320                      exc_info=log.isEnabledFor(logging.INFO),
321                  )
322                  raise
323  
324              finally:
325                  NEST -= 1
326  
327          return wrapper
328  
329      return decorator

332  # Replays the ShapeEnvEvents list.
333  # It assumes the first event is the constructor call.
334  #
335  # fn: transforms an old FX node into one corresponding to the newly created ShapeEnv.
336  def replay_shape_env_events(events):
337      from torch.fx.experimental.symbolic_shapes import ShapeEnv
338  
339      constructor_event = events[0]
340      assert constructor_event.f == ShapeEnv
341  
342      # Constructs the new ShapeEnv.
343      shape_env = constructor_event.run()
344  
345      for event in events[1:]:
346          try:
347              # Actually replays each event.
348              # We need to call create_mapping_fn every time, since the node list might
349              # change after each event is replayed.
350              event.run(shape_env)
351          except Exception:
352              log.error("failed when running event: %s", event)
353              raise
354  
355      return shape_env

387  # [Note: ShapeEnv State Equality]
388  # ===============================
389  #
390  # What is considered ShapeEnv state?
391  # ----------------------------------
392  # We consider to be the state of a ShapeEnv instance everything that
393  # is not in the inline tuple inside remove_nonstate_variables function.
394  # That is: the fields within ShapeEnv that modify the flow of execution
395  # of the program.
396  #
397  # So, for example: the replacements field might influence on how an
398  # expression is simplified. That, in turn, may result in a guard being
399  # statically known (i.e. not added).
400  #
401  # On the other hand, var_to_stack serves only changes what is printed
402  # in the screen, i.e. used only for debugging purposes. Therefore, we
403  # should not consider it when comparing states.
404  #
405  # What to do on NotEqualError?
406  # ----------------------------
407  # Here are a few possible causes for getting a NotEqualError raised:
408  #
409  #   1. New field that does not belong in the ShapeEnv state.
410  #      For example: log field of type ShapeEnvLoggerAdapter. Different
411  #      ShapeEnv instances will always have different ShapeEnvLoggerAdapter
412  #      instances, i.e. equality comparison would fail.
413  #      Solution: add it to the inlined tuple inside remove_nonstate_variables
414  #      function inside check_equal method.
415  #
416  #   2. New field that is not directly comparable across instances.
417  #      For example: guards field of type List[ShapeGuard]. More specifically,
418  #      the ShapeGuard type holds an expression and a stack information
419  #      for debugging purposes. When replaying the even on a new ShapeEnv
420  #      instance, the stack would be different, which would trigger this error.
421  #      Solution: add a special case to the map_value function inside
422  #      check_equal function.
423  #
424  #   3. Mutation of ShapeEnv on some not recorded function.
425  #      If a mutation of the state of ShapeEnv happens inside a function
426  #      that is not recorded (or that no caller in the stack is recorded),
427  #      then, the replayed ShapeEnv won't catch that.
428  #      Solution: decorate the function with record_shape_env_event.
429

431  # Checks whether the state of two ShapeEnv are equal w.r.t. the guards
432  # returned by ShapeEnv.produce_guards.
433  def shape_env_check_state_equal(env1, env2, non_state_variable_names, map_value):
434      # Collect and remove variables that don't necessarily represent the state
435      # of a ShapeEnv. Note: we copy the dictionary so that we don't modify the
436      # instance itself.
437      env1_vars = vars(env1).copy()
438      env2_vars = vars(env2).copy()
439  
440      for v in non_state_variable_names:
441          if v in env1_vars:
442              env1_vars.pop(v)
443          if v in env2_vars:
444              env2_vars.pop(v)
445  
446      # Function for transforming the mismatched values into string.
447      # Needed, since dict and set entries order might not be the same every time.
448      def value_to_str(value: Any) -> str:
449          if isinstance(value, dict):
450              return (
451                  "{"
452                  + ", ".join(f"{k}: {value[k]}" for k in sorted(value.keys(), key=str))
453                  + "}"
454              )
455          if isinstance(value, set):
456              return "{" + ", ".join(f"{v}" for v in sorted(value)) + "}"
457          return str(value)
458  
459      # Compares env1_vars with env2_vars.
460      # Here, we allow the value of each field to be mapped, so that we appropriately
461      # compare the two values.
462      def compare_vars(
463          map_value: Callable[[str, Any], Any],
464      ) -> list[tuple[str, str, str]]:
465          env1_set, env2_set = set(env1_vars), set(env2_vars)
466  
467          # First, compare the set of keys in each vars dictionary.
468          if env1_set != env2_set:
469              raise NotEqualError(
470                  "field set mismatch:",
471                  [
472                      (
473                          "found unique fields:",
474                          str(sorted(env1_set - env2_set)),
475                          str(sorted(env2_set - env1_set)),
476                      ),
477                  ],
478              )
479  
480          # Then, sort the keys, and compare the mapped values of each key.
481          sorted_keys = list(env1_set)
482          sorted_keys.sort()
483  
484          mapped_dict = [
485              (k, map_value(k, env1_vars[k]), map_value(k, env2_vars[k]))
486              for k in sorted_keys
487          ]
488  
489          # Return a list of tuples representing the fields that did not match
490          # alongside their respective mapped values.
491          return [
492              (f"{k}: values don't match.", value_to_str(val1), value_to_str(val2))
493              for k, val1, val2 in mapped_dict
494              if val1 != val2
495          ]
496  
497      # Accumulate the mismatching fields.
498      errors = compare_vars(map_value)
499  
500      if len(errors) > 0:
501          raise NotEqualError("field values don't match:", errors)

504  class NotEqualError(Exception):
505      def __init__(
506          self,
507          msg: str,
508          mismatched: list[tuple[str, str, str]],
509      ) -> None:
510          details = "\n".join(
511              [
512                  "\n".join(
513                      [
514                          f"==> {inner_msg}",
515                          f"  >  Left: {str1}",
516                          f"  > Right: {str2}",
517                      ]
518                  )
519                  for inner_msg, str1, str2 in mismatched
520              ]
521          )
522  
523          super().__init__(
524              f"""\ 
525  ShapeEnv not equal: {msg}
526  
527  {details}
528  """
529          )
