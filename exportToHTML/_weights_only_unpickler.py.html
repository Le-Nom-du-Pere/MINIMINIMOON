1    # mypy: allow-untyped-defs
2    # Unpickler restricted to loading only state dicts
3    # Restrict constructing types to a list defined in _get_allowed_globals()
4    # Restrict BUILD operation to `Tensor`, `Parameter` and `OrderedDict` types only
5    # Restrict APPEND/APPENDS to `list`
6    # In `GLOBALS` operation do not do class lookup by name, but rather rely on dictionary
7    # defined by `_get_allowed_globals()` method, that contains:
8    # - torch types (Storage, dtypes, Tensor, `torch.Size`),
9    # - `torch._utils._rebuild` functions.
10   # - `torch.nn.Parameter`
11   # - `collections.Counter`
12   # - `collections.OrderedDict`
13   # Additionally, users can use an allowlist for adding classes they have deemed as safe using
14   # `_add_safe_globals()` (`torch.serialization.add_safe_globals`)
15   # `_clear_safe_globals()` (`torch.serialization.clear_safe_globals`)
16   # `_get_safe_globals()` (`torch.serialization.get_safe_globals`)
17   
18   # Based of https://github.com/python/cpython/blob/main/Lib/pickle.py
19   # Expected to be useful for loading PyTorch model weights
20   # For example:
21   # data = urllib.request.urlopen('https://download.pytorch.org/models/resnet50-0676ba61.pth').read()
22   # buf = io.BytesIO(data)
23   # weights = torch.load(buf, weights_only = True)

90       global _marked_safe_globals_set

104  def _remove_safe_globals(
105      globals_to_remove: list[Union[Callable, tuple[Callable, str]]],

111  class _safe_globals:
112      def __init__(self, safe_globals: list[Union[Callable, tuple[Callable, str]]]):
113          self.safe_globals = safe_globals
114  
115      def __enter__(self):
116          _add_safe_globals(self.safe_globals)
117  
118      def __exit__(self, type, value, tb):
119          _remove_safe_globals(self.safe_globals)

122  # Separate from _get_allowed_globals because of the lru_cache on _get_allowed_globals
123  # For example if user had a script like
124  #   torch.load(file_a)
125  #   torch.serialization._add_safe_globals([torch.foo])
126  #   torch.load(file_b)
127  # the dynamic additions to safe_globals would not be picked up by
128  # _get_allowed_globals due to the lru_cache
129  def _get_user_allowed_globals():
130      rc: dict[str, Any] = {}
131      for f in _marked_safe_globals_set:
132          if isinstance(f, tuple):
133              if len(f) != 2:
134                  raise ValueError(
135                      f"Expected tuple of length 2 (global, str of callable full path), but got tuple of length: {len(f)}"
136                  )
137              if type(f[1]) is not str:
138                  raise TypeError(
139                      f"Expected second item in tuple to be str of callable full path, but got: {type(f[1])}"
140                  )
141              f, name = f
142              rc[name] = f
143          else:
144              module, name = f.__module__, f.__qualname__
145              rc[f"{module}.{name}"] = f
146      return rc

171  # Unpickling machinery
172  @_functools.lru_cache(maxsize=1)
173  def _get_allowed_globals():
174      rc: dict[str, Any] = {
175          "collections.OrderedDict": OrderedDict,
176          "collections.Counter": Counter,
177          "torch.nn.parameter.Parameter": torch.nn.Parameter,
178          "torch.serialization._get_layout": torch.serialization._get_layout,
179          "torch.Size": torch.Size,
180          "torch.Tensor": torch.Tensor,
181          "torch.device": torch.device,
182          "_codecs.encode": encode,  # for bytes
183          "builtins.bytearray": bytearray,  # for bytearray
184          "builtins.set": set,  # for set
185          "builtins.complex": complex,  # for complex
186      }
187  
188      # dtype
189      for t in torch.storage._dtype_to_storage_type_map().keys():
190          rc[str(t)] = t
191      for t in torch.storage._new_dtypes():
192          rc[str(t)] = t
193      for t in [getattr(torch, f"uint{x}") for x in range(1, 8)]:
194          rc[str(t)] = t
195      for t in [getattr(torch, f"int{x}") for x in range(1, 8)]:
196          rc[str(t)] = t
197  
198      # Tensor classes
199      for tt in torch._tensor_classes:
200          rc[f"{tt.__module__}.{tt.__name__}"] = tt
201      # Storage classes
202      for ts in torch._storage_classes:
203          if ts not in (torch.storage.TypedStorage, torch.storage.UntypedStorage):
204              # Wrap legacy storage types in a dummy class
205              rc[f"{ts.__module__}.{ts.__name__}"] = torch.serialization.StorageType(
206                  ts.__name__
207              )
208          else:
209              rc[f"{ts.__module__}.{ts.__name__}"] = ts
210      # Quantization specific
211      for qt in [
212          torch.per_tensor_affine,
213          torch.per_tensor_symmetric,
214          torch.per_channel_affine,
215          torch.per_channel_symmetric,
216          torch.per_channel_affine_float_qparams,
217      ]:
218          rc[str(qt)] = qt
219      # Rebuild functions
220      for f in _tensor_rebuild_functions():
221          rc[f"torch._utils.{f.__name__}"] = f
222  
223      # Handles Tensor Subclasses, Tensor's with attributes.
224      # NOTE: It calls into above rebuild functions for regular Tensor types.
225      rc["torch._tensor._rebuild_from_type_v2"] = torch._tensor._rebuild_from_type_v2
226      return rc

229  def _read_global_instruction(readline: Callable) -> tuple[str, str]:
230      module = readline()[:-1].decode("utf-8")
231      name = readline()[:-1].decode("utf-8")
232      # Patch since torch.save default protocol is 2
233      # users will be running this code in python > 3
234      if (module, name) in NAME_MAPPING:
235          module, name = NAME_MAPPING[(module, name)]
236      elif module in IMPORT_MAPPING:
237          module = IMPORT_MAPPING[module]
238      return module, name

241  def get_globals_in_pkl(file) -> set[str]:
242      globals_in_checkpoint = set()
243      read = file.read
244      readline = file.readline
245      op_to_bytes_to_read = {
246          NEWOBJ[0]: 0,
247          REDUCE[0]: 0,
248          BUILD[0]: 0,
249          APPEND[0]: 0,
250          APPENDS[0]: 0,
251          SETITEM[0]: 0,
252          SETITEMS[0]: 0,
253          MARK[0]: 0,
254          TUPLE[0]: 0,
255          TUPLE1[0]: 0,
256          TUPLE2[0]: 0,
257          TUPLE3[0]: 0,
258          NONE[0]: 0,
259          NEWFALSE[0]: 0,
260          NEWTRUE[0]: 0,
261          EMPTY_TUPLE[0]: 0,
262          EMPTY_LIST[0]: 0,
263          EMPTY_DICT[0]: 0,
264          EMPTY_SET[0]: 0,
265          BINPERSID[0]: 0,
266          BININT[0]: 4,
267          BININT1[0]: 1,
268          BININT2[0]: 2,
269          BINFLOAT[0]: 8,
270          BINGET[0]: 1,
271          LONG_BINGET[0]: 4,
272          BINPUT[0]: 1,
273          LONG_BINPUT[0]: 4,
274      }
275      while True:
276          key = read(1)
277          if not key:
278              raise EOFError
279          assert isinstance(key, bytes_types)
280          if key[0] == GLOBAL[0]:
281              module, name = _read_global_instruction(readline)
282              globals_in_checkpoint.add(f"{module}.{name}")
283          elif key[0] in op_to_bytes_to_read:
284              bytes_to_read = op_to_bytes_to_read[key[0]]
285              if bytes_to_read:
286                  read(bytes_to_read)
287          # ops where bytes to read depends on the data
288          elif key[0] == BINUNICODE[0]:
289              strlen = unpack("<I", read(4))[0]
290              if strlen > maxsize:
291                  raise UnpicklingError("String is too long")
292              read(strlen)
293          elif key[0] in {SHORT_BINSTRING[0], LONG1[0]}:
294              strlen = read(1)[0]
295              read(strlen)
296          # first and last op
297          elif key[0] == PROTO[0]:
298              read(1)[0]
299          elif key[0] == STOP[0]:
300              return globals_in_checkpoint
301          else:
302              raise UnpicklingError(f"Unsupported operand {key[0]}")

305  class Unpickler:
306      def __init__(self, file, *, encoding: str = "bytes"):
307          self.encoding = encoding
308          self.readline = file.readline
309          self.read = file.read
310          self.memo: dict[int, Any] = {}
311          self.proto: int = -1
312  
313      def load(self):
314          """Read a pickled object representation from the open file. 
315   
316          Return the reconstituted object hierarchy specified in the file. 
317          """
318          self.metastack = []
319          self.stack: list[Any] = []
320          self.append = self.stack.append
321          read = self.read
322          while True:
323              key = read(1)
324              if not key:
325                  raise EOFError
326              assert isinstance(key, bytes_types)
327              # Risky operators
328              if key[0] == GLOBAL[0]:
329                  module, name = _read_global_instruction(self.readline)
330                  full_path = f"{module}.{name}"
331                  if module in _blocklisted_modules:
332                      raise UnpicklingError(
333                          f"Trying to load unsupported GLOBAL {full_path} whose module {module} is blocked."
334                      )
335                  if full_path in _get_allowed_globals():
336                      self.append(_get_allowed_globals()[full_path])
337                  elif full_path in _get_user_allowed_globals():
338                      self.append(_get_user_allowed_globals()[full_path])
339                  elif full_path in (
340                      [
341                          "torch.nested._internal.nested_tensor.NestedTensor",
342                          "torch.nested._internal.nested_tensor._rebuild_njt",
343                          "torch._dynamo.decorators._DimRange",
344                      ]
345                  ):
346                      raise UnpicklingError(
347                          "``torch.nested`` and ``torch._dynamo`` must be imported to load nested jagged tensors (NJTs)"
348                      )
349                  elif full_path in (
350                      [
351                          "torch.distributed.device_mesh.DeviceMesh",
352                          "torch.distributed.tensor._dtensor_spec.DTensorSpec",
353                          "torch.distributed.tensor._dtensor_spec.TensorMeta",
354                          "torch.distributed.tensor.DTensor",
355                          "torch.distributed.tensor.placement_types.Partial",
356                          "torch.distributed.tensor.placement_types.Replicate",
357                          "torch.distributed.tensor.placement_types.Shard",
358                      ]
359                  ):
360                      raise UnpicklingError(
361                          "``torch.distributed.tensor`` must be imported to load DTensors"
362                      )
363                  else:
364                      builtins_name = "builtins"
365                      if (
366                          builtins_name in full_path
367                          and builtins_name == full_path[: len(builtins_name)]
368                      ):
369                          full_path = full_path[len(builtins_name) :]
370                          full_path = (
371                              full_path[1:]
372                              if len(full_path) > 0 and full_path[0] == "."
373                              else builtins_name + full_path
374                          )
375                      raise UnpicklingError(
376                          f"Unsupported global: GLOBAL {full_path} was not an allowed global by default. "
377                          f"Please use `torch.serialization.add_safe_globals([{full_path}])` or the "
378                          f"`torch.serialization.safe_globals([{full_path}])` context manager to allowlist this global "
379                          "if you trust this class/function."
380                      )
381              elif key[0] == NEWOBJ[0]:
382                  args = self.stack.pop()
383                  cls = self.stack.pop()
384                  if cls is torch.nn.Parameter:
385                      self.append(torch.nn.Parameter(*args))
386                  elif (
387                      cls in _get_user_allowed_globals().values()
388                      or cls in _get_allowed_globals().values()
389                  ):
390                      result = cls.__new__(cls, *args)
391                      if cls in torch._tensor_classes and "sparse" in cls.__module__:
392                          _sparse_tensors_to_validate.append(result)
393                      self.append(result)
394                  else:
395                      raise UnpicklingError(
396                          "Can only create new object for nn.Parameter or classes allowlisted "
397                          f"via `add_safe_globals` but got {cls}"
398                      )
399              elif key[0] == REDUCE[0]:
400                  args = self.stack.pop()
401                  func = self.stack[-1]
402                  if (
403                      func not in _get_allowed_globals().values()
404                      and func not in _get_user_allowed_globals().values()
405                  ):
406                      raise UnpicklingError(
407                          f"Trying to call reduce for unrecognized function {func}"
408                      )
409                  result = func(*args)
410                  if func in torch._tensor_classes and "sparse" in func.__module__:
411                      _sparse_tensors_to_validate.append(result)
412                  self.stack[-1] = result
413              elif key[0] == BUILD[0]:
414                  state = self.stack.pop()
415                  inst = self.stack[-1]
416                  if type(inst) is torch.Tensor:
417                      # Legacy unpickling
418                      inst.set_(*state)
419                  elif type(inst) is torch.nn.Parameter:
420                      inst.__setstate__(state)
421                  elif type(inst) is OrderedDict:
422                      inst.__dict__.update(state)
423                  elif (
424                      type(inst) in _get_user_allowed_globals().values()
425                      or type(inst) in _get_allowed_globals().values()
426                  ):
427                      if hasattr(inst, "__setstate__"):
428                          inst.__setstate__(state)
429                      else:
430                          # mimics load_build in pickle
431                          # https://github.com/python/cpython/blob/f0c6fccd08904787a39269367f09f263d496114c/Lib/pickle.py#L1854-L1867
432                          slotstate = None
433                          if isinstance(state, tuple) and len(state) == 2:
434                              state, slotstate = state
435                          if state:
436                              inst.__dict__.update(state)
437                          if slotstate:
438                              for k, v in slotstate.items():
439                                  setattr(inst, k, v)
440                  else:
441                      raise UnpicklingError(
442                          "Can only build Tensor, Parameter, OrderedDict or types allowlisted "
443                          f"via `add_safe_globals`, but got {type(inst)}"
444                      )
445              # Stack manipulation
446              elif key[0] == APPEND[0]:
447                  item = self.stack.pop()
448                  list_obj = self.stack[-1]
449                  if type(list_obj) is not list:
450                      raise UnpicklingError(
451                          f"Can only append to lists, but got {type(list_obj)}"
452                      )
453                  list_obj.append(item)
454              elif key[0] == APPENDS[0]:
455                  items = self.pop_mark()
456                  list_obj = self.stack[-1]
457                  if type(list_obj) is not list:
458                      raise UnpicklingError(
459                          f"Can only extend lists, but got {type(list_obj)}"
460                      )
461                  list_obj.extend(items)
462              elif key[0] == SETITEM[0]:
463                  (v, k) = (self.stack.pop(), self.stack.pop())
464                  self.stack[-1][k] = v
465              elif key[0] == SETITEMS[0]:
466                  items = self.pop_mark()
467                  for i in range(0, len(items), 2):
468                      self.stack[-1][items[i]] = items[i + 1]
469              elif key[0] == MARK[0]:
470                  self.metastack.append(self.stack)
471                  self.stack = []
472                  self.append = self.stack.append
473              elif key[0] == TUPLE[0]:
474                  items = self.pop_mark()
475                  self.append(tuple(items))
476              elif key[0] == TUPLE1[0]:
477                  self.stack[-1] = (self.stack[-1],)
478              elif key[0] == TUPLE2[0]:
479                  self.stack[-2:] = [(self.stack[-2], self.stack[-1])]
480              elif key[0] == TUPLE3[0]:
481                  self.stack[-3:] = [(self.stack[-3], self.stack[-2], self.stack[-1])]
482              # Basic types construction
483              elif key[0] == NONE[0]:
484                  self.append(None)
485              elif key[0] == NEWFALSE[0]:
486                  self.append(False)
487              elif key[0] == NEWTRUE[0]:
488                  self.append(True)
489              elif key[0] == EMPTY_TUPLE[0]:
490                  self.append(())
491              elif key[0] == EMPTY_LIST[0]:
492                  self.append([])
493              elif key[0] == EMPTY_DICT[0]:
494                  self.append({})
495              elif key[0] == EMPTY_SET[0]:
496                  self.append(set())
497              elif key[0] == BININT[0]:
498                  self.append(unpack("<i", read(4))[0])
499              elif key[0] == BININT1[0]:
500                  self.append(self.read(1)[0])
501              elif key[0] == BININT2[0]:
502                  self.append(unpack("<H", read(2))[0])
503              elif key[0] == BINFLOAT[0]:
504                  self.append(unpack(">d", self.read(8))[0])
505              elif key[0] == BINUNICODE[0]:
506                  strlen = unpack("<I", read(4))[0]
507                  if strlen > maxsize:
508                      raise UnpicklingError("String is too long")
509                  strval = str(read(strlen), "utf-8", "surrogatepass")
510                  self.append(strval)
511              elif key[0] == SHORT_BINSTRING[0]:
512                  strlen = read(1)[0]
513                  strdata = read(strlen)
514                  if self.encoding != "bytes":
515                      strdata = strdata.decode(self.encoding, "strict")
516                  self.append(strdata)
517              elif key[0] == BINPERSID[0]:
518                  pid = self.stack.pop()
519                  # Only allow persistent load of storage
520                  if type(pid) is not tuple and not type(pid) is not int:
521                      raise UnpicklingError(
522                          f"persistent_load id must be tuple or int, but got {type(pid)}"
523                      )
524                  if (
525                      type(pid) is tuple
526                      and len(pid) > 0
527                      and torch.serialization._maybe_decode_ascii(pid[0]) != "storage"
528                  ):
529                      raise UnpicklingError(
530                          f"Only persistent_load of storage is allowed, but got {pid[0]}"
531                      )
532                  self.append(self.persistent_load(pid))
533              elif key[0] in [BINGET[0], LONG_BINGET[0]]:
534                  idx = (read(1) if key[0] == BINGET[0] else unpack("<I", read(4)))[0]
535                  self.append(self.memo[idx])
536              elif key[0] in [BINPUT[0], LONG_BINPUT[0]]:
537                  i = (read(1) if key[0] == BINPUT[0] else unpack("<I", read(4)))[0]
538                  if i < 0:
539                      raise ValueError("negative argument")
540                  self.memo[i] = self.stack[-1]
541              elif key[0] == LONG1[0]:
542                  n = read(1)[0]
543                  data = read(n)
544                  self.append(decode_long(data))
545              # First and last deserializer ops
546              elif key[0] == PROTO[0]:
547                  self.proto = read(1)[0]
548                  if self.proto != 2:
549                      warnings.warn(
550                          f"Detected pickle protocol {self.proto} in the checkpoint, which was "
551                          "not the default pickle protocol used by `torch.load` (2). The weights_only "
552                          "Unpickler might not support all instructions implemented by this protocol, "
553                          "please file an issue for adding support if you encounter this."
554                      )
555              elif key[0] == STOP[0]:
556                  rc = self.stack.pop()
557                  return rc
558              else:
559                  raise UnpicklingError(f"Unsupported operand {key[0]}")
560  
561      # Return a list of items pushed in the stack after last MARK instruction.
562      def pop_mark(self):
563          items = self.stack
564          self.stack = self.metastack.pop()
565          self.append = self.stack.append
566          return items
567  
568      def persistent_load(self, pid):
569          raise UnpicklingError("unsupported persistent id encountered")
