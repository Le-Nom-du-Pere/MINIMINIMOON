17   @runtime_checkable
18   class InputAdaptStep(Protocol):
19       """A protocol that defines a step in the input adapting process. 
20    
21       The input adapting process is a sequence of steps that are applied to the 
22       PyTorch model inputs to transform them into the inputs format expected by the 
23       exported ONNX model. Each step takes the PyTorch model inputs as arguments and 
24       returns the transformed inputs. 
25    
26       This serves as a base formalized construct for the transformation done to model 
27       input signature by any individual component in the exporter. 
28       """
29   
30       def apply(
31           self,
32           model_args: Sequence[Any],
33           model_kwargs: Mapping[str, Any],
34           model: torch.nn.Module | Callable | torch_export.ExportedProgram | None = None,
35       ) -> tuple[Sequence[Any], Mapping[str, Any]]: ...
36   
37   
38   class InputAdapter:
39       """A class that adapts the PyTorch model inputs to exported ONNX model inputs format."""
40   
41       def __init__(self, steps: list[InputAdaptStep] | None = None):
42           self._steps = steps or []
43   
44       def append_step(self, step: InputAdaptStep) -> None:
45           """Appends a step to the input adapt steps. 
46    
47           Args: 
48               step: The step to append. 
49           """
50           self._steps.append(step)
51   
52       def apply(
53           self,
54           *model_args,
55           model: torch.nn.Module | Callable | torch_export.ExportedProgram | None = None,
56           **model_kwargs,
57       ) -> Sequence[int | float | bool | str | torch.Tensor | torch.dtype | None]:
58           """Converts the PyTorch model inputs to exported ONNX model inputs format. 
59    
60           Args: 
61               model_args: The PyTorch model inputs. 
62               model: The PyTorch model. 
63               model_kwargs: The PyTorch model keyword inputs. 
64           Returns: 
65               A sequence of tensors converted from PyTorch model inputs. 
66           """
67           args: Sequence[Any] = model_args
68           kwargs: Mapping[str, Any] = model_kwargs
69           for step in self._steps:
70               args, kwargs = step.apply(args, kwargs, model=model)
71           assert not kwargs
72           return args

76   class OutputAdaptStep(Protocol):
77       """A protocol that defines a step in the output adapting process. 
78    
79       The output adapting process is a sequence of steps that are applied to the 
80       PyTorch model outputs to transform them into the outputs format produced by the 
81       exported ONNX model. Each step takes the PyTorch model outputs as arguments and 
82       returns the transformed outputs. 
83    
84       This serves as a base formalized construct for the transformation done to model 
85       output signature by any individual component in the exporter. 
86       """
87   
88       def apply(
89           self,
90           model_outputs: Any,
91           model: torch.nn.Module | Callable | torch_export.ExportedProgram | None = None,
92       ) -> Any: ...
93   
94   
95   class OutputAdapter:
96       """A class that adapts the PyTorch model outputs to exported ONNX model outputs format."""
97   
98       def __init__(self, steps: list[OutputAdaptStep] | None = None):
99           self._steps = steps or []
100  
101      def append_step(self, step: OutputAdaptStep) -> None:
102          """Appends a step to the output format steps. 
103   
104          Args: 
105              step: The step to append. 
106          """
107          self._steps.append(step)
108  
109      def apply(
110          self,
111          model_outputs: Any,
112          model: torch.nn.Module | Callable | torch_export.ExportedProgram | None = None,
113      ) -> Sequence[torch.Tensor | int | float | bool | str]:
114          """Converts the PyTorch model outputs to exported ONNX model outputs format. 
115   
116          Args: 
117              model_outputs: The PyTorch model outputs. 
118              model: The PyTorch model. 
119   
120          Returns: 
121              PyTorch model outputs in exported ONNX model outputs format. 
122          """
123          for step in self._steps:
124              model_outputs = step.apply(model_outputs, model=model)
125          return model_outputs

136  def _replace_list_with_tuple(spec: pytree.TreeSpec) -> pytree.TreeSpec:
137      def replace_list_with_tuple(x: Any) -> Any:
138          if type(x) is list:
139              return pytree.tree_map(
140                  replace_list_with_tuple,
141                  tuple(x),
142                  is_leaf=lambda x: type(x) is list,
143              )
144          return x
145  
146      dummy_leaf = _DummyLeaf()
147      dummy_tree = pytree.tree_unflatten([dummy_leaf] * spec.num_leaves, spec)
148      dummy_tree = pytree.tree_map(
149          replace_list_with_tuple,
150          dummy_tree,
151          is_leaf=lambda x: type(x) is list,
152      )
153      return pytree.tree_structure(dummy_tree)

156  def _open_top_level_sequence_if_single_element(
157      spec: pytree.TreeSpec,
158  ) -> pytree.TreeSpec:
159      if spec.type in (tuple, list) and spec.num_children == 1:
160          return spec.children_specs[0]
161      return spec

164  def _assert_identical_pytree_spec(
165      spec1: pytree.TreeSpec, spec2: pytree.TreeSpec, error_message: str
166  ) -> None:
167      """Assert the two `TreeSpec` objects are identical. 
168   
169      Args: 
170          spec1: The first `TreeSpec` object. 
171          spec2: The second `TreeSpec` object. 
172          error_message: The error message to raise if the two `TreeSpec` objects are not 
173              identical. 
174   
175      Raises: 
176          ValueError: If the two `TreeSpec` objects are not identical. 
177      """
178      pass_if_any_checks: Sequence[Callable[[], bool]] = [
179          lambda: spec1 == spec2,
180          # FIXME: Bug in `dynamo.export`. Sometimes outputs returned in 'list' instead of 'tuple'.
181          lambda: _replace_list_with_tuple(spec1) == _replace_list_with_tuple(spec2),
182          # FIXME: Bug in `dynamo.export`. Sometimes single function return is wrapped in list.
183          lambda: _open_top_level_sequence_if_single_element(spec1) == spec2,
184          lambda: spec1 == _open_top_level_sequence_if_single_element(spec2),
185      ]
186  
187      if not any(check() for check in pass_if_any_checks):
188          raise ValueError(f"{error_message}\nExpect {spec1}.\nActual {spec2}.")

191  class BindInputStep(InputAdaptStep):
192      """Bind the input arguments to the model signature."""
193  
194      def __init__(self, model_signature: inspect.Signature):
195          self._model_signature = model_signature
196  
197      def apply(
198          self,
199          model_args: Sequence[Any],
200          model_kwargs: Mapping[str, Any],
201          model: torch.nn.Module | Callable | torch_export.ExportedProgram | None = None,
202      ) -> tuple[Sequence[Any], Mapping[str, Any]]:
203          """Bind the input arguments to the model signature. 
204   
205          We hope the input kwargs will be mapped to bound.args after binding. 
206          If not, we will raise an error. 
207   
208          Args: 
209              model_args: The model args. 
210              model_kwargs: The model kwargs. 
211              model: The PyTorch model. 
212   
213          Returns: 
214              A tuple of the model args and kwargs. args is always empty. 
215   
216          Raises: 
217              ValueError: If there are keyword-only arguments left after binding args and 
218                  kwargs to model signature. 
219          """
220          bound = self._model_signature.bind(*model_args, **model_kwargs)
221          bound.apply_defaults()
222  
223          # keyword-only arguments are not handled.
224          # bound.kwargs only contains keyword-only arguments after calling
225          # bind & apply_defaults, so we raise if it's not empty.
226          if bound.kwargs:
227              raise ValueError("Keyword-only arguments are not supported.")
228          return (), bound.arguments

278  class ConvertComplexToRealRepresentationInputStep(InputAdaptStep):
279      """Convert complex dtype tensors to real representation tensors. 
280   
281      ONNX does not support complex dtype tensors. Thus, we convert complex dtype tensors 
282      to real representation tensors (i.e., float dtype tensors with an extra dimension 
283      representing the real and imaginary parts of the complex number). 
284   
285      """
286  
287      def apply(
288          self,
289          model_args: Sequence[Any],
290          model_kwargs: Mapping[str, Any],
291          model: torch.nn.Module | Callable | torch_export.ExportedProgram | None = None,
292      ) -> tuple[Sequence[Any], Mapping[str, Any]]:
293          """Convert complex tensors to float tensors. 
294   
295          Args: 
296              model_args: The model args. 
297              model_kwargs: The model kwargs. 
298              model: The PyTorch model. 
299   
300          Returns: 
301              A tuple of the model args and kwargs. 
302          """
303          return (
304              tuple(
305                  torch.view_as_real(arg.resolve_conj())
306                  if isinstance(arg, torch.Tensor) and arg.is_complex()
307                  else arg
308                  for arg in model_args
309              ),
310              model_kwargs,
311          )

344  class RemoveNonTensorInputStep(InputAdaptStep):
345      """Remove the non-tensor input arguments. 
346   
347      Dynamo does not support non-tensor input arguments (https://github.com/pytorch/pytorch/issues/99534). 
348   
349      Specifically, it does put the input into graph with an empty node, but consumed by no ones. 
350      The concrete value is embedded into the graph as a constant arg of a target node. Meta 
351      suggests in this case that one should rewrite the model code to make it tensor if the 
352      input value is supposed to change at runtime. We might need to further investigate 
353      the feasibility of that suggestion. 
354   
355      For example, 
356   
357          def func(x, b=1.0): 
358              y = x + b 
359              z = y.relu() 
360              return (y, z) 
361   
362          x = torch.randn(1, 1, 2, dtype=torch.float32) 
363          gm_fun, _ = dynamo.export(func, x, b=8.0, aten_graph=True, tracing_mode="real") 
364   
365          # class GraphModule(torch.nn.Module): 
366          #     def forward(self, x, b): 
367          #         arg0: f32[1, 1, 2], arg1, = fx_pytree.tree_flatten_spec(([x, b], {}), self._in_spec) 
368          #         # File: path/to/pytorch/test_constant_input.py:5, code: y = x + b 
369          #         add_tensor: f32[1, 1, 2] = torch.ops.aten.add.Tensor(arg0, 8.0);  arg0 = None 
370   
371          #         # File: path/to/pytorch/test_constant_input.py:6, code: z = y.relu() 
372          #         relu_default: f32[1, 1, 2] = torch.ops.aten.relu.default(add_tensor) 
373          #         return pytree.tree_unflatten([add_tensor, relu_default], self._out_spec) 
374   
375      Empty torch.fx.Node input leading to a mismatched number of input with PyTorch, as 
376      it's ignored in ONNX graph. Thus, we delete the useless input here. 
377   
378      """
379  
380      def apply(
381          self,
382          model_args: Sequence[Any],
383          model_kwargs: Mapping[str, Any],
384          model: torch.nn.Module | Callable | torch_export.ExportedProgram | None = None,
385      ) -> tuple[Sequence[Any], Mapping[str, Any]]:
386          """Remove Constant from arguments. 
387   
388          Args: 
389              model_args: The model args. 
390              model_kwargs: The model kwargs. 
391              model: The PyTorch model. 
392   
393          Returns: 
394              A tuple of the model args and kwargs. 
395   
396          Raises: 
397              ValueError: If `model_kwargs` is not empty. 
398          """
399          assert not model_kwargs
400          return (
401              tuple(
402                  arg
403                  for arg in model_args
404                  if not isinstance(arg, (int, float, bool, str))
405              ),
406              {},
407          )

410  class FlattenInputWithTreeSpecValidationInputStep(InputAdaptStep):
411      """Flatten nested collection types and return a flat list of elements. 
412   
413      ONNX can't represent collection types (e.g., dictionary, tuple of tuple of tensor, 
414      etc). 
415   
416      This class stores the `SpecTree` output produced when `adapt` was called the first 
417      time. It then validates the `SpecTree` output produced from later `adapt` calls. 
418      """
419  
420      _spec: pytree.TreeSpec | None = None
421  
422      def apply(
423          self,
424          model_args: Sequence[Any],
425          model_kwargs: Mapping[str, Any],
426          model: torch.nn.Module | Callable | torch_export.ExportedProgram | None = None,
427      ) -> tuple[Sequence[Any], Mapping[str, Any]]:
428          """Flatten the model args and kwargs and validate the `SpecTree` output. 
429   
430          Args: 
431              model_args: The model args. 
432              model_kwargs: The model kwargs. 
433              model: The PyTorch model. 
434   
435          Returns: 
436              A tuple of the flattened model args and kwargs. The kwargs is empty, because 
437              they are flattened and merged into the args. 
438   
439          Raises: 
440              ValueError: If the `SpecTree` output produced from the current `model_outputs` 
441                  is not identical to the `SpecTree` output produced from the first 
442                  `model_outputs` that was passed to this method. 
443          """
444          flattened_args, spec = pytree.tree_flatten((model_args, model_kwargs))
445          if self._spec is None:
446              self._spec = spec
447          else:
448              _assert_identical_pytree_spec(
449                  self._spec,
450                  spec,
451                  error_message="Model inputs incompatible with the format that was exported. ",
452              )
453          return flattened_args, {}

457      """Flatten nested collection types and return a flat list of elements.

558  class PrependParamsBuffersConstantAotAutogradInputStep(InputAdaptStep):
559      """Prepend model parameters, buffers and constants to the user input. 
560   
561      :func:`torch.export.export` lifts model parameters, buffers and constants as model input, thus, they 
562      must be added to the user input before the model is executed. 
563   
564      Args: 
565          model: The PyTorch model with embedded parameters and buffers. 
566      """
567  
568      def apply(
569          self,
570          model_args: Sequence[Any],
571          model_kwargs: Mapping[str, Any],
572          model: torch.nn.Module | Callable | torch_export.ExportedProgram | None = None,
573      ) -> tuple[Sequence[Any], Mapping[str, Any]]:
574          """Convert complex tensors to float tensors. 
575   
576          Args: 
577              model_args: The model args. 
578              model_kwargs: The model kwargs. 
579              model: The PyTorch model. 
580   
581          Returns: 
582              A tuple of the model args and kwargs. 
583          """
584          ordered_params = tuple(
585              model.state_dict[name]  # type: ignore[union-attr,index]
586              for name in model.graph_signature.parameters  # type: ignore[union-attr]
587          )
588          non_persistent_buffers = set(model.graph_signature.non_persistent_buffers)  # type: ignore[arg-type, union-attr]
589          ordered_buffers = []
590          for name in model.graph_signature.buffers:  # type: ignore[union-attr]
591              if name in non_persistent_buffers:
592                  ordered_buffers.append(model.constants[name])  # type: ignore[index, union-attr]
593              else:
594                  ordered_buffers.append(model.state_dict[name])  # type: ignore[union-attr,index]
595          ordered_constant_tensors = tuple(
596              model.constants[fqn]  # type: ignore[union-attr,index]
597              for fqn in model.graph_signature.lifted_tensor_constants  # type: ignore[union-attr]
598          )
599  
600          # NOTE: calling convention is first params, then buffers, then args as user supplied them.
601          # See: torch/_functorch/aot_autograd.py#L1034
602          updated_args = (
603              *ordered_params,
604              *ordered_buffers,
605              *ordered_constant_tensors,
606              *model_args,
607          )
608          if model_kwargs:
609              return MergeKwargsIntoArgsInputStep().apply(
610                  updated_args, model_kwargs, model=model
611              )
612          return updated_args, {}

615  class PrependParamsAndBuffersAotAutogradOutputStep(OutputAdaptStep):
616      """Prepend model's mutated buffers to the user output. 
617   
618      :func:`torch.export.export` lifts model's mutated buffers as outputs, thus, they 
619      must be added to the user output after the model is executed. 
620   
621      Args: 
622          model: The PyTorch model with mutated buffers. 
623      """
624  
625      def apply(
626          self,
627          model_outputs: Any,
628          model: torch.nn.Module | Callable | torch_export.ExportedProgram | None = None,
629      ) -> Sequence[Any]:
630          """Flatten the model outputs and validate the `SpecTree` output. 
631   
632          Args: 
633              model_outputs: The model outputs to flatten. 
634              model: The PyTorch model. 
635   
636          Returns: 
637              flattened_outputs: The flattened model outputs. 
638          """
639  
640          assert isinstance(model, torch_export.ExportedProgram), (
641              "'model' must be torch_export.ExportedProgram"
642          )
643          ordered_buffers = tuple(
644              model.state_dict[name]
645              if name in model.state_dict
646              else model.constants[name]
647              for name in model.graph_signature.buffers_to_mutate.values()
648          )
649  
650          # NOTE: calling convention is first mutated buffers, then outputs args as model returned them.
651          updated_outputs = (*ordered_buffers, *model_outputs)
652          return updated_outputs
