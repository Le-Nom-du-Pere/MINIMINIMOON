1    """ 
2    Profile Guided Optimization (PGO) implementation for Dynamo. 
3     
4    This module provides functionality for caching and managing code state profiles 
5    that guide optimization decisions in Dynamo. It implements both local and remote 
6    caching mechanisms for storing profile information across runs, handles profile 
7    merging across distributed ranks, and manages the lifecycle of profile data 
8    during compilation. The profiles track dynamic vs static properties of tensors 
9    and help Dynamo make better specialization decisions. 
10   """

63   # How does in memory representation work?  Concretely, this module is
64   # responsible for holding GLOBAL state representing the state it holds, no
65   # other copies permitted.  So we retire frame_state entirely and store it
66   # here.  This should be reset when Dynamo is reset.  We never GC information
67   # (similar to how the filesystem doesn't get cleaned up except by tmp
68   # cleaner), so the expectation is the information is relatively cheap and we
69   # don't mind leaking it.
70   
71   
72   # How exactly did we design the cache key?  Here are some of the questions:
73   #
74   # - JOB_ID: Do we have a unique identifier for the "training run"  (such that
75   #   it stays the same if we're running the same code, and changes if we're
76   #   running something different).
77   #
78   # - RANK: Are we sharing the cache across ranks, or does each rank get
79   #   an individual cache?
80   #
81   # We choose to require job_id for PGO cache.  This is to prevent
82   # situations where unrelated invocations of PyTorch unpredictably cause
83   # changes to each other's behavior.  With a job_id, at least you know there
84   # is some "state" associated with it.  (State dict might be another way to
85   # tell if a run is related or not.)  You can opt-in to YOLO everything
86   # aliases everything by passing a shared job_id for all your invocations.
87   #
88   # We choose to NOT share PGO cache across ranks.  With no RANK_SHARING, there
89   # is never contention between runs, so we can leisurely update a bundle with
90   # information we need.  Because we are grouped by job_id, we can have a single
91   # consolidated bundle for everything (or not; maybe worry about O(n^2) IO if
92   # we updated every compile--let's just instrument this.)  Can even take a
93   # filelock for extra safety (expect no contention); expect 50ns overhead from
94   # uncontended filelock.
95   #
96   # If we did share ranks, everyone is storming to modify the same cache files.
97   # We can do this by having folks atomic write to a CAS-store and then having
98   # readers do on-the-fly merging (this can be implemented in remote using
99   # prefix iteration).  As an optional optimization, one rank can be elected to
100  # handling bundling post facto (ideally, this is done async, after quiescence,
101  # without compiler collective need to wait for everyone to finish writing
102  # their bits.) Not sure how you can avoid a listdir because if some rank shows
103  # up with some new entries we need to pull them in ASAP (unless you want to
104  # delay bundling).
105  #
106  # But compiler collectives fill a similar niche:  compilers chat with each
107  # other so rank 0 has collected everything.  So elect rank 0 only to write the
108  # bundle.  Don't even need CAS-store atomic write; just one rank writing an
109  # updating bundles.  The point is that use compiler collectives to share
110  # profiles across ranks, but use the PGO cache to persist profiles per rank
111  # across attempts.  No need to have one mechanism to do everything.
112

114  @functools.cache
115  def _hash_containing_file(filepath: str) -> str:
116      # if the file does not exists we consider filepath to be the hash.
117      if not os.path.exists(filepath):
118          return filepath
119  
120      with open(filepath, "rb") as file:
121          content = file.read()
122          crc32_value = zlib.crc32(content)
123          hash = format(crc32_value & 0xFFFFFFFF, "08x")
124          return hash

127  @dataclasses.dataclass(frozen=True)
128  class CodeId:
129      filename: str
130      firstlineno: int
131      name: str
132      # When a job restart, the code can be copied to a different path than the previous attempt. In that case
133      # self.filename will have a different value,  we do not want to consider those differences. Instead we
134      # hash the content of the file and use it as an identifier of the file.
135      #
136      # self.filename is kept in the object to give readable information/pointer to the actual file, in a local
137      # code state it will refer to the first seen file path.
138      file_hash: str
139  
140      # Exclude file name.
141      def __eq__(self, other: object) -> bool:
142          if not isinstance(other, CodeId):
143              return False
144          return (
145              self.file_hash == other.file_hash
146              and self.firstlineno == other.firstlineno
147              and self.name == other.name
148          )
149  
150      # Ensure if two CodeIds are the same, then they have the same hash by excluding filename.
151      def __hash__(self) -> int:
152          return hash((self.file_hash, self.name, self.firstlineno))
153  
154      def __str__(self) -> str:
155          return f"hash({self.file_hash}){self.filename}:{self.firstlineno}:{self.name}"
156  
157      @staticmethod
158      def make(code: types.CodeType) -> CodeId:
159          return CodeId(
160              code.co_filename,
161              code.co_firstlineno,
162              code.co_name,
163              _hash_containing_file(code.co_filename),
164          )

178  @dataclasses.dataclass(frozen=True)
179  class InferStride:
180      """ 
181      Denotes the quantity stride[dim] * size[dim], which is what the stride would 
182      be for the next physical dimension that results in a contiguous layout. 
183   
184      For example, given size = [2, 3], stride = [3, 1], we can replace this with 
185      stride = [InferStride(1), 1], because InferStride(1) = stride[1] * size[1] = 1 * 3 = 3 
186   
187      Indirecting the representation in this way is important for the join operation 
188      on strides as if we join [2, 3][3, 1] and [2, 4][4, 1], 
189      we don't want [2, None][None, 1] which would get eventually symbolized into 
190      [2, s0][s1, 1] (notice that the relationship between s0 and s1 is broken). 
191      If we instead rewrite the expressions as InferStride so we have [2, 3][InferStride(1), 1] 
192      and [2, 4][InferStride(1), 1] we now join to [2, None][InferStride(1), 1] will 
193      result in [2, s0][s0, 1], as desired. 
194      """
195  
196      dim: int
197

226  @dataclasses.dataclass
227  class FrameStateSizeEntry:
228      scalar: Union[int, AutoDynamic, AutoUnset] = dataclasses.field(default=auto_unset)
229      # NB: We don't have cases where we have a known dimensionality but
230      # we know NOTHING about the individual sizes
231      size: Union[AutoDynamic, AutoUnset, tuple[Union[int, AutoDynamic], ...]] = (
232          dataclasses.field(default=auto_unset)
233      )
234      stride: Union[
235          AutoDynamic, AutoUnset, tuple[Union[int, AutoDynamic, InferStride], ...]
236      ] = dataclasses.field(default=auto_unset)
237  
238      def render(self) -> str:
239          # Special cases
240          def render_single(s: Union[int, AutoDynamic, AutoUnset, InferStride]) -> str:
241              if s is auto_dynamic:
242                  return "?"
243              elif s is auto_unset:
244                  # This basically shouldn't happen, this is for debugging
245                  return "auto unset"
246              elif isinstance(s, InferStride):
247                  return f"S({s.dim})"
248              else:
249                  return str(s)
250  
251          def render_tuple(ss: tuple[Union[int, AutoDynamic, InferStride], ...]) -> str:
252              return "[" + ", ".join(render_single(s) for s in ss) + "]"
253  
254          # Common cases
255          if self.size is auto_dynamic and self.stride is auto_dynamic:
256              if self.scalar is auto_dynamic:
257                  return "fully dynamic scalar or tensor"
258              else:
259                  return f"scalar {self.scalar}"
260          elif self.scalar is auto_dynamic:
261              if isinstance(self.size, tuple) and isinstance(self.stride, tuple):
262                  return f"tensor size={render_tuple(self.size)} stride={render_tuple(self.stride)}"
263  
264          # Fallback
265          return "unusual {repr(self)}"
266  
267      def __post_init__(self) -> None:
268          assert not isinstance(self.scalar, torch.SymInt), self.scalar
269          if isinstance(self.size, tuple):
270              for s in self.size:
271                  assert not isinstance(s, torch.SymInt), s
272          if isinstance(self.stride, tuple):
273              for s1 in self.stride:
274                  assert not isinstance(s1, torch.SymInt), s1
275  
276      def is_size_dynamic(self, dim: int) -> bool:
277          if self.size is auto_dynamic:
278              return True
279          if self.size is auto_unset:
280              return False
281          return self.size[dim] is auto_dynamic
282  
283      def is_stride_dynamic(self, dim: int) -> bool:
284          # At the moment, dynamic strides is a bit buggy.  Good test case
285          # here is `PYTORCH_TEST_WITH_DYNAMO=1 python test/test_autograd.py
286          # TestAutograd.test_gradcheck_jacobian_mismatch`
287          #
288          # This if statement preserves historical behavior, which is that we
289          # ONLY make strides dynamic if the size is exactly static everywhere.
290          # We could potentially relax this but in general we should be very
291          # careful about when to infer dynamic strides.
292          #
293          # Actually, the existing algorithm is already somewhat problematic.
294          # Suppose a tensor that is sometimes:
295          # f32[2, 3, 5][15, 5, 1] and other times
296          # f32[2, 3, 5][5, 10, 1] (specifically, dim 0 and 1 are physically transposed).
297          # If we infer strides should be (DYNAMIC, DYNAMIC, 1).  But this is
298          # silly: we really should have just guarded on dim order.
299          if not (
300              isinstance(self.size, tuple) and all(type(s) is int for s in self.size)
301          ):
302              return False
303          if self.stride is auto_dynamic:
304              return True
305          if self.stride is auto_unset:
306              return False
307          return self.stride[dim] is auto_dynamic
308  
309      @staticmethod
310      def _munge_symint(xs: tuple[int, ...]) -> tuple[Union[AutoDynamic, int], ...]:
311          return tuple(auto_dynamic if isinstance(x, torch.SymInt) else x for x in xs)
312  
313      @classmethod
314      def make_scalar(cls, x: int) -> FrameStateSizeEntry:
315          return FrameStateSizeEntry(scalar=x, size=auto_dynamic, stride=auto_dynamic)
316  
317      @classmethod
318      def make_tensor(
319          cls, size: tuple[int, ...], stride: tuple[int, ...]
320      ) -> FrameStateSizeEntry:
321          return FrameStateSizeEntry(
322              scalar=auto_dynamic,
323              size=cls._munge_symint(size),
324              stride=cls._munge_symint(stride),
325          )
326  
327      @classmethod
328      def make_size(cls, size: tuple[int, ...]) -> FrameStateSizeEntry:
329          return FrameStateSizeEntry(
330              scalar=auto_unset,
331              size=cls._munge_symint(size),
332              stride=auto_unset,
333          )
334  
335      @staticmethod
336      def _merge_atom(x: _T, y: _T) -> Union[AutoDynamic, _T]:
337          if x is auto_unset:
338              return y
339          if y is auto_unset:
340              return x
341          if x is auto_dynamic or y is auto_dynamic or x != y:
342              return auto_dynamic
343          return x
344  
345      @classmethod
346      def _merge_atom_tup(
347          cls,
348          xs: Union[AutoDynamic, AutoUnset, tuple[_T, ...]],
349          ys: Union[AutoDynamic, AutoUnset, tuple[_T, ...]],
350      ) -> Union[AutoDynamic, AutoUnset, tuple[Union[AutoDynamic, _T], ...]]:
351          if xs is auto_unset:
352              return ys
353          if ys is auto_unset:
354              return xs
355          if xs is auto_dynamic or ys is auto_dynamic:
356              return auto_dynamic
357          if len(xs) != len(ys):
358              return auto_dynamic
359          return tuple(cls._merge_atom(x, y) for x, y in zip(xs, ys))
360  
361      def __ior__(self, other: Self) -> Self:
362          self.scalar = self._merge_atom(self.scalar, other.scalar)
363          self.size = self._merge_atom_tup(self.size, other.size)
364          self.stride = self._merge_atom_tup(self.stride, other.stride)
365          return self

368  def update_automatic_dynamic(
369      tx: InstructionTranslator,
370      name: str,
371      entry: FrameStateSizeEntry,
372      *,
373      is_unspecialized_nn_module: bool = False,
374  ) -> FrameStateSizeEntry:
375      code_id = CodeId.make(tx.f_code)
376      frame_state = get_code_state()[code_id]
377      if torch._dynamo.config.automatic_dynamic_shapes:
378          is_update = name in frame_state.automatic_dynamic
379          mut_entry = frame_state.automatic_dynamic[name]
380          old_entry = copy.copy(mut_entry)
381          mut_entry |= entry
382  
383          # Do some logs (damn, I spend more code logging than I do actually doing
384          # the updates lol)
385          if is_update and old_entry.scalar != mut_entry.scalar:
386              log.debug(
387                  "automatic dynamic int %s val %s != %s",
388                  name,
389                  entry.scalar,
390                  old_entry.scalar,
391              )
392              CompileEventLogger.instant(
393                  "automatic_dynamic",
394                  {
395                      "name": name,
396                      "dim_changed": "scalar",
397                      "reason": "scalar change",
398                      "cached": str(old_entry.scalar),
399                      "new": str(entry.scalar),
400                  },
401              )
402              if is_unspecialized_nn_module:
403                  log.info(
404                      "%s is converted to a symbolic integer. It is an attribute of a "
405                      "user defined nn module class. If you wish to keep it static, you can "
406                      "mark the nn module class as `torch._dynamo.mark_static`.",
407                      name,
408                  )
409  
410          def log_tup(
411              tup_name: str, short_reason: str, long_reason: str, i: Optional[int] = None
412          ) -> None:
413              entry_tup = (
414                  getattr(entry, tup_name) if i is None else getattr(entry, tup_name)[i]
415              )
416              old_entry_tup = (
417                  getattr(old_entry, tup_name)
418                  if i is None
419                  else getattr(old_entry, tup_name)[i]
420              )
421              log.debug(
422                  "automatic dynamic %s %s %s %s != %s",
423                  tup_name,
424                  name,
425                  short_reason,
426                  # NB: We used to only report len(...) here for dim mismatch
427                  entry_tup,
428                  old_entry_tup,
429              )
430              CompileEventLogger.instant(
431                  "automatic_dynamic",
432                  {
433                      "name": name,
434                      "dim_changed": "all" if i is None else i,
435                      "reason": long_reason,
436                      "cached": str(old_entry_tup),
437                      "new": str(entry_tup),
438                  },
439              )
440  
441          if is_update and old_entry.size != mut_entry.size:
442              if isinstance(old_entry.size, tuple) and isinstance(entry.size, tuple):
443                  if len(old_entry.size) != len(entry.size):
444                      log_tup("size", "dim", "dimensionality change")
445                  else:
446                      for i in range(len(entry.size)):
447                          if old_entry.size[i] != entry.size[i]:
448                              log_tup("size", f"size({i})", "size change", i)
449              else:
450                  log_tup("size", "other", "other")
451  
452          if is_update and old_entry.stride != mut_entry.stride:
453              if isinstance(old_entry.stride, tuple) and isinstance(entry.stride, tuple):
454                  if len(old_entry.stride) != len(entry.stride):
455                      log_tup("stride", "dim", "dimensionality change")
456                  else:
457                      for i in range(len(entry.stride)):
458                          if old_entry.stride[i] != entry.stride[i]:
459                              log_tup("stride", f"stride({i})", "stride change", i)
460              else:
461                  log_tup("stride", "other", "other")
462      else:
463          old_entry = frame_state.automatic_dynamic[name]
464          log.debug(
465              "automatic dynamic is off, overwriting int %s val %s -> %s",
466              name,
467              old_entry.scalar,
468              entry.scalar,
469          )
470          frame_state.automatic_dynamic[name] = entry
471          mut_entry = entry
472  
473      return mut_entry

476  def process_automatic_dynamic(
477      tx: InstructionTranslator,
478      name: str,
479      entry: FrameStateSizeEntry,
480      *,
481      is_unspecialized_nn_module: bool = False,
482  ) -> FrameStateSizeEntry:
483      if (st := tx.distributed_state) is None:
484          return update_automatic_dynamic(
485              tx,
486              name,
487              entry,
488              is_unspecialized_nn_module=is_unspecialized_nn_module,
489          )
490      elif st.all_states is None:
491          # Preflight, always pretend as if it's static.  The point here
492          # is we want to get through the preflight quickly, and static
493          # will run faster.  The preexisting frame state will get
494          # applied anyway after we do compiler collectives.
495          # TODO: I'm not sure if we should just bong the entire pgo
496          # state here, it kind of depends if we're going to have other
497          # things that talk in compiler collective.  Also, the PGO
498          # state, if we've already inferred something is automatic
499          # dynamic, will have lost the actual input sizes, which might
500          # be useful for debugging purposes (e.g., observing 0/1
501          # specialization).  Bonging the entire PGO state here would
502          # let us delete this logic here; the compiler collective
503          # would just directly update_automatic_dynamic
504          st.local_state.automatic_dynamic[name] = entry
505          return entry
506      else:
507          # Apply the updates.  NB: all_states includes the local state
508          # too.
509          res = None
510          for sub_state in st.all_states:
511              if name in sub_state.automatic_dynamic:
512                  res = update_automatic_dynamic(
513                      tx,
514                      name,
515                      sub_state.automatic_dynamic[name],
516                      is_unspecialized_nn_module=is_unspecialized_nn_module,
517                  )
518          assert res is not None
519          return res

522  def get_cache_key() -> Optional[str]:
523      # TODO: info versions of these logs that log only once
524      if torch._inductor.config.force_disable_caches:
525          warn_once(
526              "dynamo_pgo force disabled by torch._inductor.config.force_disable_caches"
527          )
528          return None
529  
530      # NB: We always use global rank for keys, even though they are overkill
531      # for local only cache
532      rank = None
533      if dist.is_available() and dist.is_initialized():
534          rank = dist.get_rank()
535  
536      tag = torch.compiler.config.cache_key_tag
537  
538      # NB: We namespace the cache keys so that only user-specified job id
539      # can alias with each other.
540      if (r := torch.compiler.config.job_id) is not None:
541          if r.startswith("mast:"):
542              raise ReservedWorkflowIdUserError(
543                  "torch.compiler.config.job_id with prefix 'mast:' is reserved for "
544                  "automatically generated job id associated with a specific MAST job "
545                  "name and version."
546              )
547          return f"{r}:{rank}:{tag}"
548  
549      if (name_version := torch._utils_internal.get_mast_job_name_version()) is not None:
550          mast_job_name, mast_job_version = name_version
551          return f"mast:{mast_job_name}:{mast_job_version}:{rank}:{tag}"
552  
553      return None

565      return os.path.join(cache_dir(), "dynamo", code_state_key)

568  def should_use_remote_dynamo_pgo_cache() -> bool:
569      if torch._inductor.config.force_disable_caches:
570          return False
571  
572      if (r := torch._dynamo.config.automatic_dynamic_remote_pgo) is not None:
573          return r
574  
575      if not is_fbcode():
576          return False
577  
578      if torch._utils_internal.is_fb_unit_test():
579          return False
580  
581      try:
582          from torch._inductor.fb.remote_cache import REMOTE_CACHE_VERSION
583      except ModuleNotFoundError:
584          return False
585  
586      return REMOTE_CACHE_VERSION >= torch._utils_internal.justknobs_getval_int(
587          "pytorch/remote_cache:dynamo_pgo_version"
588      )

605  def _collect_dynamic_sources(code_state: CodeState) -> OrderedSet[str]:
606      dynamic_sources: OrderedSet[str] = OrderedSet()
607      for src, fs in code_state.automatic_dynamic.items():
608          dynamic = False
609          if isinstance(fs.size, tuple):
610              dynamic = auto_dynamic in fs.size  # type: ignore[operator]
611          elif fs.scalar == auto_dynamic:
612              dynamic = True
613          if dynamic:
614              dynamic_sources.add(src)
615      return dynamic_sources

629  def render_code_state(cs: defaultdict[CodeId, CodeState]) -> str:
630      code_state_str = "\n".join(
631          f"{k}:\n"
632          + "\n".join(
633              f"  {src}: {fs.render()}" for src, fs in v.automatic_dynamic.items()
634          )
635          for k, v in cs.items()
636      )
637      dynamic_sources: OrderedSet[str] = OrderedSet()
638      for state in cs.values():
639          dynamic_sources.update(_collect_dynamic_sources(state))
640      if dynamic_sources:
641          code_state_str += (
642              "\n\nPGO detected a recompilation due to dynamic shapes. "
643              "To reduce shape recompilations by compiling dynamically to start, "
644              f'set environment variable TORCH_COMPILE_DYNAMIC_SOURCES="{",".join(dynamic_sources)}"'
645          )
646      return code_state_str
647  
648  
649  @CacheArtifactFactory.register
650  class PGOCacheArtifact(CacheArtifact):
651      @override
652      def populate_cache(self) -> None:
653          meta = write_local_impl(
654              self._rewrite_cache_key_for_mega_cache(self.key), self.content
655          )
656          assert meta is not None
657  
658      @override
659      @staticmethod
660      def type() -> str:
661          return "pgo"
662  
663      @staticmethod
664      def _rewrite_cache_key_for_mega_cache(original_key: str) -> str:
665          """ 
666          The PGO cache artifact key for a MAST job contains the job name and the version. 
667          When we want to use the cache artifact on a different MAST job, we need to 
668          update the key to use the new MAST job's name and version. 
669          """
670          if not original_key.startswith("mast:"):
671              # if original_key is overridden, then dont change it
672              return original_key
673          if (new_key := get_cache_key()) is not None:
674              return new_key
675          return original_key
676  
677  
678  def get_code_state() -> defaultdict[CodeId, CodeState]:
679      global _CODE_STATE, _INIT_CODE_STATE
680      if _CODE_STATE is not None:
681          return _CODE_STATE
682  
683      # Initialize it (even if we don't look up profile)
684      _CODE_STATE = defaultdict(CodeState)
685  
686      cache_key = get_cache_key()
687      if cache_key is None:
688          return _CODE_STATE
689  
690      def hit(ty: str) -> defaultdict[CodeId, CodeState]:
691          global _INIT_CODE_STATE
692          assert isinstance(_CODE_STATE, defaultdict)
693          log.info("get_code_state %s hit %s, %d entries", path, ty, len(_CODE_STATE))
694          trace_structured_artifact(
695              f"get_{ty}_code_state",
696              "string",
697              lambda: render_code_state(_CODE_STATE),  # type: ignore[arg-type]
698          )
699          set_feature_use("pgo", True)
700          _INIT_CODE_STATE = copy.deepcopy(_CODE_STATE)
701          return _CODE_STATE
702  
703      # Attempt local
704      path = code_state_path(cache_key)
705      if path is not None and os.path.exists(path):
706          with dynamo_timed(
707              name := "pgo.get_local_code_state", log_pt2_compile_event=True
708          ):
709              CompileEventLogger.pt2_compile(name, cache_key=cache_key)
710              # Read lock not necessary as we always write atomically write to
711              # the actual location
712              with open(path, "rb") as f:
713                  try:
714                      content = f.read()
715                      _CODE_STATE = pickle.loads(content)
716                      CompileEventLogger.pt2_compile(name, cache_size_bytes=f.tell())
717                  except Exception:
718                      log.warning(
719                          "get_code_state failed while reading %s", path, exc_info=True
720                      )
721                  else:
722                      CacheArtifactManager.record_artifact(
723                          PGOCacheArtifact.type(), cache_key, content
724                      )
725                      return hit("local")
726  
727      # Attempt remote
728      remote_cache = get_remote_cache()
729      if remote_cache is not None:
730          with dynamo_timed(
731              name := "pgo.get_remote_code_state",
732              log_pt2_compile_event=True,
733              dynamo_compile_column_us="pgo_get_remote_code_state_time_us",
734          ):
735              CompileEventLogger.pt2_compile(name, cache_key=cache_key)
736              # TODO: I don't really understand why there's a JSON container format
737              try:
738                  cache_data = remote_cache.get(cache_key)
739              except Exception:
740                  log.warning(
741                      "get_code_state failed remote read on %s", cache_key, exc_info=True
742                  )
743              else:
744                  if cache_data is not None:
745                      try:
746                          assert isinstance(cache_data, dict)
747                          data = cache_data["data"]
748                          assert isinstance(data, str)
749                          payload = base64.b64decode(data)
750                          CompileEventLogger.pt2_compile(
751                              name, cache_size_bytes=len(payload)
752                          )
753                          _CODE_STATE = pickle.loads(payload)
754                      except Exception:
755                          log.warning(
756                              "get_code_state failed parsing remote result on %s",
757                              cache_key,
758                              exc_info=True,
759                          )
760                      else:
761                          CacheArtifactManager.record_artifact(
762                              PGOCacheArtifact.type(), cache_key, payload
763                          )
764                          return hit("remote")
765                  else:
766                      log.info("get_code_state remote miss on %s", cache_key)
767  
768      log.info("get_code_state using default")
769  
770      assert _CODE_STATE is not None
771      return _CODE_STATE
772  
773  
774  def put_code_state() -> None:
775      if _CODE_STATE is None:
776          log.info("put_code_state: never initialized, will not write")
777          return
778  
779      if _CODE_STATE == _INIT_CODE_STATE:
780          log.info("put_code_state: no change, skipping")
781          return
782  
783      cache_key = get_cache_key()
784      if cache_key is None:
785          log.info("put_code_state: no cache key, skipping")
786          return
787  
788      put_local_code_state(cache_key)
789      put_remote_code_state(cache_key)
790  
791  
792  def write_local_impl(cache_key: str, pickled_code: bytes) -> Optional[tuple[str, int]]:
793      path = code_state_path(cache_key)
794  
795      if path is None:
796          return None
797  
798      # If the user isn't misusing our API, we should have exclusive access to
799      # this directory.  But it's not too hard
800  
801      tmp_path = path + ".tmp"
802      lock_path = path + ".lock"
803      # We /mostly/ don't need the lock but the tmp file could be clobbered
804      # TODO: use a safe tempfile create to eliminate lock
805      from torch.utils._filelock import FileLock
806  
807      os.makedirs(os.path.dirname(path), exist_ok=True)
808  
809      with FileLock(lock_path, timeout=LOCK_TIMEOUT):
810          with open(tmp_path, "wb") as f:
811              f.write(pickled_code)
812              size = f.tell()
813          os.replace(tmp_path, path)
814      return path, size
815  
816  
817  def put_local_code_state(cache_key: str) -> None:
818      with dynamo_timed(name := "pgo.put_local_code_state", log_pt2_compile_event=True):
819          CompileEventLogger.pt2_compile(name, cache_key=cache_key)
820          assert _CODE_STATE is not None
821  
822          pickled_code = pickle.dumps(_CODE_STATE)
823  
824          CacheArtifactManager.record_artifact(
825              PGOCacheArtifact.type(), cache_key, pickled_code
826          )
827  
828          meta = write_local_impl(cache_key, pickled_code)
829          if meta is None:
830              log.info("put_code_state: local cache disabled")
831              return
832          path, size = meta
833  
834          CompileEventLogger.pt2_compile(name, cache_size_bytes=size)
835          log.info("put_code_state: wrote local %s, %d entries", path, len(_CODE_STATE))
836          trace_structured_artifact(
837              "put_local_code_state",
838              "string",
839              lambda: render_code_state(_CODE_STATE),
840          )
841  
842  
843  def put_remote_code_state(cache_key: str) -> None:
844      with dynamo_timed(
845          name := "pgo.put_remote_code_state",
846          log_pt2_compile_event=True,
847          dynamo_compile_column_us="pgo_put_remote_code_state_time_us",
848      ):
849          CompileEventLogger.pt2_compile(name, cache_key=cache_key)
850          assert _CODE_STATE is not None
851  
852          remote_cache = get_remote_cache()
853  
854          if remote_cache is None:
855              log.info("put_code_state: remote cache disabled")
856              return
857  
858          content = pickle.dumps(_CODE_STATE)
859          CompileEventLogger.pt2_compile(name, cache_size_bytes=len(content))
860          cache_data: JsonDataTy = {
861              "data": base64.b64encode(content).decode("ascii"),
862          }
863          remote_cache.put(cache_key, cache_data)
864          log.info(
865              "put_code_state: wrote remote %s, %d entries", cache_key, len(_CODE_STATE)
866          )
867          # TODO: don't log this multiple times
868          trace_structured_artifact(
869              "put_remote_code_state",
870              "string",
871              lambda: render_code_state(_CODE_STATE),
872          )

877      global _CODE_STATE, _INIT_CODE_STATE
878      _CODE_STATE = None
879      _INIT_CODE_STATE = None
