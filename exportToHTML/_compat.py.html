29   def _get_torch_export_args(
30       args: tuple[Any, ...],
31       kwargs: dict[str, Any] | None,
32   ) -> tuple[tuple[Any, ...], dict[str, Any] | None]:
33       """Obtain the arguments for torch.onnx.export from the model and the input arguments."""
34       if not kwargs and args and isinstance(args[-1], dict):
35           kwargs = args[-1]
36           args = args[:-1]
37       return args, kwargs

40   def export_compat(
41       model: torch.nn.Module
42       | torch.export.ExportedProgram
43       | torch.jit.ScriptModule
44       | torch.jit.ScriptFunction,
45       args: tuple[Any, ...],
46       f: str | os.PathLike | None = None,
47       *,
48       kwargs: dict[str, Any] | None = None,
49       export_params: bool = True,
50       verbose: bool | None = None,
51       input_names: Sequence[str] | None = None,
52       output_names: Sequence[str] | None = None,
53       opset_version: int | None = _constants.TORCHLIB_OPSET,
54       custom_translation_table: dict[Callable, Callable | Sequence[Callable]]
55       | None = None,
56       dynamic_axes: Mapping[str, Mapping[int, str]]
57       | Mapping[str, Sequence[int]]
58       | None = None,
59       dynamic_shapes: dict[str, Any] | tuple[Any, ...] | list[Any] | None = None,
60       keep_initializers_as_inputs: bool = False,
61       external_data: bool = True,
62       report: bool = False,
63       optimize: bool = False,
64       verify: bool = False,
65       profile: bool = False,
66       dump_exported_program: bool = False,
67       artifacts_dir: str | os.PathLike = ".",
68       fallback: bool = False,
69       # Legacy export parameters for fallback
70       legacy_export_kwargs: dict[str, Any] | None = None,
71   ) -> _onnx_program.ONNXProgram:
72       if opset_version is None:
73           opset_version = _constants.TORCHLIB_OPSET
74   
75       if isinstance(model, torch.export.ExportedProgram):
76           # We know the model is already exported program, so the args, kwargs, and dynamic_shapes
77           # are not used
78           dynamic_shapes = dynamic_shapes or {}
79       else:
80           args, kwargs = _get_torch_export_args(args, kwargs)
81           if dynamic_shapes is None and dynamic_axes is not None:
82               warnings.warn(
83                   "# 'dynamic_axes' is not recommended when dynamo=True, "
84                   "and may lead to 'torch._dynamo.exc.UserError: Constraints violated.' "
85                   "Supply the 'dynamic_shapes' argument instead if export is unsuccessful.",
86                   UserWarning,
87                   stacklevel=3,
88               )
89               try:
90                   dynamic_shapes, args, kwargs = (
91                       _dynamic_shapes.from_dynamic_axes_to_dynamic_shapes(
92                           model,
93                           args,
94                           kwargs,
95                           dynamic_axes=dynamic_axes,
96                           input_names=input_names,
97                           output_names=set(output_names or ()),
98                       )
99                   )
100              except Exception as e:
101                  raise RuntimeError(
102                      "# Failed to convert 'dynamic_axes' to 'dynamic_shapes'. "
103                      "Please provide 'dynamic_shapes' directly. "
104                      "Refer to the documentation for 'torch.export.export' for more information on dynamic shapes."
105                  ) from e
106  
107      dynamic_shapes_with_export_dim, need_axis_mapping = (
108          _dynamic_shapes.convert_str_to_export_dim(dynamic_shapes)
109      )
110      registry = _registration.ONNXRegistry().from_torchlib(opset_version=opset_version)
111      if custom_translation_table is not None:
112          for torch_op, onnx_ops in custom_translation_table.items():
113              # TODO(justinchuby): Support complex inputs with annotations
114              if not isinstance(onnx_ops, Sequence):
115                  onnx_ops = (onnx_ops,)
116              for op in reversed(onnx_ops):
117                  # register_op places the op in the front of all onnx variants,
118                  # so we reverse the list to maintain the order of the custom ops provided
119                  registry.register_op(torch_op, op, is_complex=False)
120      try:
121          onnx_program = _core.export(
122              model,
123              args,
124              kwargs,
125              registry=registry,
126              dynamic_shapes=dynamic_shapes_with_export_dim,
127              input_names=input_names,
128              output_names=output_names,
129              profile=profile,
130              report=report,
131              verify=verify,
132              dump_exported_program=dump_exported_program,
133              artifacts_dir=artifacts_dir,
134              verbose=verbose,
135          )
136  
137      except Exception as e:
138          if fallback:
139              if verbose is not False:
140                  print(
141                      "[torch.onnx] Falling back to legacy torch.onnx.export due "
142                      f"to the following error: {e}",
143                  )
144              if f is None:
145                  raise TypeError("f must be provided when fallback is enabled") from e
146              if dynamic_shapes is not None and dynamic_axes is None:
147                  if input_names is None:
148                      raise ValueError(
149                          "Failed to convert dynamic_shapes to dynamic_axes. "
150                          "Either input_names or dynamic_axes must be provided "
151                          "when dynamic is requested in fallback"
152                      ) from e
153                  dynamic_axes = _dynamic_shapes.from_dynamic_shapes_to_dynamic_axes(
154                      dynamic_shapes=dynamic_shapes, input_names=input_names, exception=e
155                  )
156              # Use the legacy export kwargs prepared in __init__.py
157              if legacy_export_kwargs is None:
158                  legacy_export_kwargs = {}
159  
160              torch.onnx.utils.export(
161                  model,  # type: ignore[arg-type]
162                  args,
163                  f,  # type: ignore[arg-type]
164                  kwargs=kwargs,
165                  export_params=export_params,
166                  input_names=input_names,
167                  output_names=output_names,
168                  opset_version=opset_version,
169                  dynamic_axes=dynamic_axes,
170                  keep_initializers_as_inputs=keep_initializers_as_inputs,
171                  **legacy_export_kwargs,
172              )
173              onnx_program = _onnx_program.ONNXProgram(ir.load(f), None)
174  
175              # NOTE: It it's falling back to the legacy exporter, we don't need to
176              # optimize the model, so we return it here. Users can still optimize
177              # the model using the optimize() if they want.
178              return onnx_program
179          else:
180              raise
181  
182      if need_axis_mapping and dynamic_shapes is not None:
183          onnx_program._rename_dynamic_axes(dynamic_shapes)
184  
185      # Converter opset version and optimize
186      onnx_program.model = onnxscript_apis.convert_version(
187          onnx_program.model, opset_version
188      )
189      if optimize:
190          onnx_program.optimize()
191  
192      if f is not None:
193          onnx_program.save(
194              f,
195              include_initializers=export_params,
196              keep_initializers_as_inputs=keep_initializers_as_inputs,
197              external_data=external_data,
198          )
199  
200      return onnx_program
