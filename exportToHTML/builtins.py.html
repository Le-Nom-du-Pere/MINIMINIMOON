15   if TYPE_CHECKING:
16       from collections.abc import Iterable

19   __all__ = [
20       "all",
21       "any",
22       "enumerate",
23       "sum",
24   ]

27   _T = TypeVar("_T")
28   
29   
30   @substitute_in_graph(builtins.all, can_constant_fold_through=True)
31   def all(iterable: Iterable[object], /) -> bool:
32       for elem in iterable:
33           if not elem:
34               return False
35       return True
36   
37   
38   @substitute_in_graph(builtins.any, can_constant_fold_through=True)
39   def any(iterable: Iterable[object], /) -> bool:
40       for elem in iterable:
41           if elem:
42               return True
43       return False
44   
45   
46   @substitute_in_graph(builtins.enumerate, is_embedded_type=True)  # type: ignore[arg-type]
47   def enumerate(iterable: Iterable[_T], start: int = 0) -> Iterable[tuple[int, _T]]:
48       if not isinstance(start, int):
49           raise TypeError(
50               f"{type(start).__name__!r} object cannot be interpreted as an integer"
51           )
52   
53       for x in iterable:
54           yield start, x
55           start += 1
56   
57   
58   @substitute_in_graph(builtins.sum, can_constant_fold_through=True)  # type: ignore[arg-type]
59   def sum(iterable: Iterable[_T], /, start: _T = 0) -> _T:  # type: ignore[assignment]
60       return functools.reduce(operator.add, iterable, start)
