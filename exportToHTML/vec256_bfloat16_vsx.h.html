13   inline std::tuple<Vectorized<float>, Vectorized<float>> convert_bfloat16_float(
14       const Vectorized<BFloat16>& a) {
15     constexpr int64_t K = Vectorized<BFloat16>::size();
16     __at_align__ float arr[K];
17     __at_align__ BFloat16 arr2[K];
18     a.store(arr2);
19     convert(arr2, arr, K);
20     return std::make_tuple(
21         Vectorized<float>::loadu(arr),
22         Vectorized<float>::loadu(arr + Vectorized<float>::size()));
23   }
24   
25   inline Vectorized<BFloat16> convert_float_bfloat16(
26       const Vectorized<float>& a,
27       const Vectorized<float>& b) {
28     constexpr int64_t K = Vectorized<BFloat16>::size();
29     __at_align__ float arr[K];
30     __at_align__ BFloat16 arr2[K];
31     a.store(arr);
32     b.store(arr + Vectorized<float>::size());
33     convert(arr, arr2, K);
34     return Vectorized<BFloat16>::loadu(arr2);
35   }

37   inline void load_fp32_from_bf16(
38       const c10::BFloat16* data,
39       Vectorized<float>& out) {
40     __at_align__ float values[Vectorized<float>::size()];
41     for (const auto k : c10::irange(Vectorized<float>::size())) {
42       values[k] = data[k];
43     }
44     out = Vectorized<float>::loadu(values);
45   }
46   
47   inline void load_fp32_from_bf16(
48       const c10::BFloat16* data,
49       Vectorized<float>& out1,
50       Vectorized<float>& out2) {
51     load_fp32_from_bf16(data, out1);
52     data += Vectorized<float>::size();
53     load_fp32_from_bf16(data, out2);
54   }
