137  /*! 
138   * \brief The data type the tensor can hold. The data type is assumed to follow the 
139   * native endian-ness. An explicit error message should be raised when attempting to 
140   * export an array with non-native endianness 
141   * 
142   *  Examples 
143   *   - float: type_code = 2, bits = 32, lanes = 1 
144   *   - float4(vectorized 4 float): type_code = 2, bits = 32, lanes = 4 
145   *   - int8: type_code = 0, bits = 8, lanes = 1 
146   *   - std::complex<float>: type_code = 5, bits = 64, lanes = 1 
147   *   - bool: type_code = 6, bits = 8, lanes = 1 (as per common array library convention, the underlying storage size of bool is 8 bits) 
148   */
149  typedef struct {
150    /*! 
151     * \brief Type code of base types. 
152     * We keep it uint8_t instead of DLDataTypeCode for minimal memory 
153     * footprint, but the value should be one of DLDataTypeCode enum values. 
154     * */
155    uint8_t code;
156    /*! 
157     * \brief Number of bits, common choices are 8, 16, 32. 
158     */
159    uint8_t bits;
160    /*! \brief Number of lanes in the type, used for vector types. */
161    uint16_t lanes;
162  } DLDataType;

164  /*! 
165   * \brief Plain C Tensor object, does not manage memory. 
166   */
167  typedef struct {
168    /*! 
169     * \brief The data pointer points to the allocated data. This will be CUDA 
170     * device pointer or cl_mem handle in OpenCL. It may be opaque on some device 
171     * types. This pointer is always aligned to 256 bytes as in CUDA. The 
172     * `byte_offset` field should be used to point to the beginning of the data. 
173     * 
174     * Note that as of Nov 2021, multiply libraries (CuPy, PyTorch, TensorFlow, 
175     * TVM, perhaps others) do not adhere to this 256 byte aligment requirement 
176     * on CPU/CUDA/ROCm, and always use `byte_offset=0`.  This must be fixed 
177     * (after which this note will be updated); at the moment it is recommended 
178     * to not rely on the data pointer being correctly aligned. 
179     * 
180     * For given DLTensor, the size of memory required to store the contents of 
181     * data is calculated as follows: 
182     * 
183     * \code{.c} 
184     * static inline size_t GetDataSize(const DLTensor* t) { 
185     *   size_t size = 1; 
186     *   for (tvm_index_t i = 0; i < t->ndim; ++i) { 
187     *     size *= t->shape[i]; 
188     *   } 
189     *   size *= (t->dtype.bits * t->dtype.lanes + 7) / 8; 
190     *   return size; 
191     * } 
192     * \endcode 
193     */
194    void* data;
195    /*! \brief The device of the tensor */
196    DLDevice device;
197    /*! \brief Number of dimensions */
198    int32_t ndim;
199    /*! \brief The data type of the pointer*/
200    DLDataType dtype;
201    /*! \brief The shape of the tensor */
202    const int64_t* shape;
203    /*! 
204     * \brief strides of the tensor (in number of elements, not bytes) 
205     *  can be NULL, indicating tensor is compact and row-majored. 
206     */
207    const int64_t* strides;
208    /*! \brief The offset in bytes to the beginning pointer to data */
209    uint64_t byte_offset;
210  } DLTensor;
211  
212  /*! 
213   * \brief C Tensor object, manage memory of DLTensor. This data structure is 
214   *  intended to facilitate the borrowing of DLTensor by another framework. It is 
215   *  not meant to transfer the tensor. When the borrowing framework doesn't need 
216   *  the tensor, it should call the deleter to notify the host that the resource 
217   *  is no longer needed. 
218   */
219  typedef struct DLManagedTensor {
220    /*! \brief DLTensor which is being memory managed */
221    DLTensor dl_tensor;
222    /*! \brief the context of the original host framework of DLManagedTensor in 
223     *   which DLManagedTensor is used in the framework. It can also be NULL. 
224     */
225    void * manager_ctx;
226    /*! \brief Destructor signature void (*)(void*) - this should be called 
227     *   to destruct manager_ctx which holds the DLManagedTensor. It can be NULL 
228     *   if there is no way for the caller to provide a reasonable destructor. 
229     *   The destructors deletes the argument self as well. 
230     */
231    void (*deleter)(struct DLManagedTensor * self);
232  } DLManagedTensor;
