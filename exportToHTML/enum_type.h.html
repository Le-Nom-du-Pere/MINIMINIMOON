9    struct EnumType;
10   using EnumTypePtr = std::shared_ptr<EnumType>;
11   using EnumNameValue = std::pair<std::string, IValue>;
12   struct TORCH_API EnumType : public NamedType {
13     friend struct Type;
14     static const TypeKind Kind = TypeKind::EnumType;
15   
16     static EnumTypePtr create(
17         const c10::QualifiedName& qualified_class_name,
18         TypePtr value,
19         std::vector<EnumNameValue> enum_names_values,
20         std::weak_ptr<::torch::jit::CompilationUnit> cu) {
21       switch (value->kind()) {
22         case TypeKind::IntType:
23         case TypeKind::FloatType:
24         case TypeKind::StringType:
25           return EnumTypePtr(new EnumType(
26               qualified_class_name,
27               std::move(value),
28               std::move(enum_names_values),
29               std::move(cu)));
30         default:
31           TORCH_CHECK(
32               false,
33               "Cannot create Enum with value type '",
34               value->str(),
35               "', only int, float and string are supported");
36       }
37     }
38   
39     std::string str() const override {
40       return "Enum<" + annotation_str() + ">";
41     }
42   
43     std::string repr_str() const override {
44       return str();
45     }
46   
47     const TypePtr& getValueType() const {
48       return value_type_;
49     }
50   
51     bool equals(const Type& rhs) const override {
52       if (auto* enum_rhs = rhs.castRaw<EnumType>()) {
53         return name().has_value() && name() == enum_rhs->name() &&
54             *getValueType() == *(enum_rhs->getValueType()) &&
55             this->compilation_unit() == enum_rhs->compilation_unit();
56       }
57       return false;
58     }
59   
60     bool isSubtypeOfExt(const Type& rhs, std::ostream* why_not) const override;
61   
62     std::shared_ptr<const ::torch::jit::CompilationUnit> compilation_unit()
63         const {
64       auto cu = cu_.lock();
65       return cu;
66     }
67   
68     const QualifiedName& qualifiedClassName() const {
69       // NOLINTNEXTLINE(bugprone-unchecked-optional-access)
70       return name().value();
71     }
72   
73     at::ArrayRef<TypePtr> containedTypes() const override {
74       return value_type_;
75     }
76   
77     const at::ArrayRef<EnumNameValue> enumNamesValues() const {
78       return enum_names_values_;
79     }
80   
81    private:
82     EnumType(
83         c10::QualifiedName qualified_class_name,
84         TypePtr value_type,
85         std::vector<EnumNameValue> enum_names_values,
86         std::weak_ptr<torch::jit::CompilationUnit> cu)
87         : NamedType(TypeKind::EnumType, std::move(qualified_class_name)),
88           value_type_(std::move(value_type)),
89           enum_names_values_(std::move(enum_names_values)),
90           cu_(std::move(cu)) {}
91   
92     std::string annotation_str_impl(
93         [[maybe_unused]] const TypePrinter& printer = nullptr) const override {
94       return qualifiedClassName().qualifiedName();
95     }
96   
97     TypePtr value_type_;
98     std::vector<EnumNameValue> enum_names_values_;
99     std::weak_ptr<::torch::jit::CompilationUnit> cu_;
100  };
