24   class C10_API SignalHandler {
25    public:
26     enum class Action { NONE, STOP };
27   
28     // Constructor. Specify what action to take when a signal is received.
29     SignalHandler(Action SIGINT_action, Action SIGHUP_action);
30   
31     SignalHandler(const SignalHandler&) = delete;
32     SignalHandler(SignalHandler&&) = delete;
33     SignalHandler& operator=(const SignalHandler&) = delete;
34     SignalHandler& operator=(SignalHandler&&) = delete;
35     ~SignalHandler();
36   
37     Action CheckForSignals();
38   
39     bool GotSIGINT();
40     bool GotSIGHUP();
41   
42     Action SIGINT_action_;
43     Action SIGHUP_action_;
44     std::atomic<uint64_t> my_sigint_count_;
45     std::atomic<uint64_t> my_sighup_count_;
46   };

48   #if defined(C10_SUPPORTS_FATAL_SIGNAL_HANDLERS)
49   class C10_API FatalSignalHandler {
50     // This works by setting up certain fatal signal handlers. Previous fatal
51     // signal handlers will still be called when the signal is raised. Defaults
52     // to being off.
53    public:
54     C10_API void setPrintStackTracesOnFatalSignal(bool print);
55     C10_API bool printStackTracesOnFatalSignal();
56     static FatalSignalHandler& getInstance();
57     FatalSignalHandler(const FatalSignalHandler&) = delete;
58     FatalSignalHandler(FatalSignalHandler&&) = delete;
59     FatalSignalHandler& operator=(const FatalSignalHandler&) = delete;
60     FatalSignalHandler& operator=(FatalSignalHandler&&) = delete;
61     virtual ~FatalSignalHandler() = default;
62   
63    protected:
64     explicit FatalSignalHandler();
65   
66    private:
67     void installFatalSignalHandlers();
68     void uninstallFatalSignalHandlers();
69     static void fatalSignalHandlerStatic(int signum);
70     void fatalSignalHandler(int signum);
71     virtual void fatalSignalHandlerPostProcess();
72     struct sigaction* getPreviousSigaction(int signum);
73     const char* getSignalName(int signum);
74     void callPreviousSignalHandler(
75         struct sigaction* action,
76         int signum,
77         siginfo_t* info,
78         void* ctx);
79     void stacktraceSignalHandler(bool needsLock);
80     static void stacktraceSignalHandlerStatic(
81         int signum,
82         siginfo_t* info,
83         void* ctx);
84     void stacktraceSignalHandler(int signum, siginfo_t* info, void* ctx);
85   
86     // The mutex protects the bool.
87     std::mutex fatalSignalHandlersInstallationMutex;
88     bool fatalSignalHandlersInstalled;
89     // We need to hold a reference to call the previous SIGUSR2 handler in case
90     // we didn't signal it
91     struct sigaction previousSigusr2{};
92     // Flag dictating whether the SIGUSR2 handler falls back to previous handlers
93     // or is intercepted in order to print a stack trace.
94     std::atomic<bool> fatalSignalReceived;
95     // Global state set when a fatal signal is received so that backtracing
96     // threads know why they're printing a stacktrace.
97     const char* fatalSignalName;
98     int fatalSignum = -1;
99     // This wait condition is used to wait for other threads to finish writing
100    // their stack trace when in fatal sig handler (we can't use pthread_join
101    // because there's no way to convert from a tid to a pthread_t).
102    std::condition_variable writingCond;
103    std::mutex writingMutex;
104    // used to indicate if the other thread responded to the signal
105    bool signalReceived;
