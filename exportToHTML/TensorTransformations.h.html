13   static inline Tensor roll_common(
14       const Tensor& self,
15       IntArrayRef shifts,
16       IntArrayRef dims) {
17     TORCH_CHECK(!shifts.empty(), "`shifts` required");
18     if (dims.empty() && shifts.size() == 1) {
19       auto flattened = self.contiguous().view(self.numel());
20       return roll(flattened, shifts[0], 0).view(self.sizes());
21     }
22     TORCH_CHECK(
23         shifts.size() == dims.size(),
24         "shifts and dimensions must align. shifts: ",
25         shifts.size(),
26         ", dims:",
27         dims.size());
28     AT_ASSERT(dims.size() > 1);
29     auto tail_shifts = shifts.slice(1);
30     auto tail_dims = dims.slice(1);
31     auto first_dim_rolled = roll(self, shifts[0], dims[0]);
32     return at::roll(first_dim_rolled, tail_shifts, tail_dims);
33   }
