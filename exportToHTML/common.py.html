3    """ 
4    This module provides common utilities and base classes for TorchDynamo backends. 
5     
6    Key components: 
7    - AotAutograd: Base class for implementing AOT (Ahead-of-Time) autograd backends 
8    - Backend utilities for handling: 
9      - Fake tensor conversion 
10     - Device/dtype detection from inputs 
11     - Memory efficient fusion 
12     - Graph flattening 
13     - Common compiler configurations 
14    
15   The utilities here are used by various backend implementations to handle 
16   common operations and provide consistent behavior across different backends. 
17   AOT autograd functionality is particularly important as it enables ahead-of-time 
18   optimization of both forward and backward passes. 
19   """

31       aot_module_simplified,

40   class AotAutograd:
41       def __init__(self, **kwargs) -> None:
42           self.__name__ = "compiler_fn"
43           self.kwargs = kwargs
44   
45       def __call__(self, gm: torch.fx.GraphModule, example_inputs, **kwargs):
46           if kwargs:
47               log.warning("aot_autograd-based backend ignoring extra kwargs %s", kwargs)
48   
49           if any(isinstance(x, (list, tuple, dict)) for x in example_inputs):
50               return flatten_graph_inputs(
51                   gm,
52                   example_inputs,
53                   self,
54               )
55   
56           # Hack to get around circular import problems with aot_eager_decomp_partition
57           if callable(self.kwargs.get("decompositions")):
58               self.kwargs["decompositions"] = self.kwargs["decompositions"]()
59   
60           # NB: dont delete counter increment
61           counters["aot_autograd"]["total"] += 1
62           use_fallback = False
63   
64           if use_fallback:
65               log.debug("Unable to use AOT Autograd because graph has mutation")
66               counters["aot_autograd"]["not_ok"] += 1
67               return gm
68   
69           def wrap_bw_compiler(bw_compiler_fn):
70               def _wrapped_bw_compiler(*args, **kwargs):
71                   # Note [Wrapping bw_compiler in disable]
72                   # The two disables here:
73                   # - stop TorchDynamo from trying to compile the bw_compiler function itself
74                   # - stop TorchDynamo from trying to compile our the generated backwards pass bw_compiler produces
75                   return disable(
76                       disable(
77                           bw_compiler_fn, reason="do not trace backward compiler function"
78                       )(*args, **kwargs),
79                       reason="do not trace generated backwards pass",
80                   )
81   
82               return _wrapped_bw_compiler
83   
84           bw_compiler = self.kwargs.get("bw_compiler") or self.kwargs["fw_compiler"]
85   
86           if isinstance(bw_compiler, SerializableAOTDispatchCompiler):
87               bw_compiler.compiler_fn = wrap_bw_compiler(bw_compiler.compiler_fn)
88           else:
89               bw_compiler = wrap_bw_compiler(bw_compiler)
90   
91           self.kwargs["bw_compiler"] = bw_compiler
92           self.kwargs["inference_compiler"] = (
93               self.kwargs.get("inference_compiler") or self.kwargs["fw_compiler"]
94           )
95   
96           from functorch.compile import nop
97           from torch._inductor.debug import enable_aot_logging
98   
99           # debug asserts slow down compile time noticeably,
100          # So only default them on when the aot_eager backend is used.
101          if self.kwargs.get("fw_compiler", None) == nop:
102              patch_config = patch("functorch.compile.config.debug_assert", True)
103          else:
104              patch_config = contextlib.nullcontext()
105  
106          try:
107              # NB: NOT cloned!
108              with enable_aot_logging(), patch_config:
109                  cg = aot_module_simplified(gm, example_inputs, **self.kwargs)
110                  counters["aot_autograd"]["ok"] += 1
111                  return disable(cg, reason="do not trace AOT-compiled graph")
112          except TensorifyScalarRestartAnalysis:
113              raise
114          except Exception:
115              counters["aot_autograd"]["not_ok"] += 1
116              raise
117

119  def aot_autograd(**kwargs) -> AotAutograd:
120      return AotAutograd(**kwargs)

123  def mem_efficient_fusion_kwargs(use_decomps):
124      from functorch.compile import (
125          default_decompositions,
126          min_cut_rematerialization_partition,
127          ts_compile,
128      )
129  
130      kwargs = {
131          # these are taken from memory_efficient_fusion()
132          "fw_compiler": ts_compile,
133          "bw_compiler": ts_compile,
134          "partition_fn": min_cut_rematerialization_partition,
135      }
136  
137      if use_decomps:
138          kwargs["decompositions"] = default_decompositions
139  
140      return kwargs

143  def fake_tensor_unsupported(fn):
144      """ 
145      Decorator for backends that need real inputs.  We swap out fake 
146      tensors for zero tensors. 
147      """
148  
149      @functools.wraps(fn)
150      def wrapper(model, inputs, **kwargs):
151          with _disable_current_modes():
152              inputs = list(map(defake, inputs))
153              return fn(model, inputs, **kwargs)
154  
155      return wrapper

158  def device_from_inputs(example_inputs) -> torch.device:
159      for x in example_inputs:
160          if hasattr(x, "device"):
161              return x.device
162  
163  
164  def dtype_from_inputs(example_inputs) -> torch.dtype:
165      for x in example_inputs:
166          if hasattr(x, "dtype"):
167              return x.dtype
