11   # This file contains utilities for ensuring dynamically compile()'d
12   # code fragments display their line numbers in backtraces.
13   #
14   # The constraints:
15   #
16   # - We don't have control over the user exception printer (in particular,
17   #   we cannot assume the linecache trick will work, c.f.
18   #   https://stackoverflow.com/q/50515651/23845 )
19   #
20   # - We don't want to create temporary files every time we compile()
21   #   some code; file creation should happen lazily only at exception
22   #   time.  Arguably, you *should* be willing to write out your
23   #   generated Python code to file system, but in some situations
24   #   (esp. library code) it would violate user expectation to write
25   #   to the file system, so we try to avoid it.  In particular, we'd
26   #   like to keep the files around, so users can open up the files
27   #   mentioned in the trace; if the file is invisible, we want to
28   #   avoid clogging up the filesystem.
29   #
30   #   If this is not a constraint for you, there is a substantially simpler
31   #   way to implement the functionality in this PR: instead of using
32   #   eval/exec directly, just always write a Python file to filesystem
33   #   and compile that.
34   #
35   # - You have control over a context where the compiled code will get
36   #   executed, so that we can interpose while the stack is unwinding
37   #   (otherwise, we have no way to interpose on the exception printing
38   #   process.)
39   #
40   # There are two things you have to do to make use of the utilities here:
41   #
42   # - When you compile your source code, you must save its string source
43   #   in its f_globals under the magic name "__compile_source__"
44   #
45   # - Before running the compiled code, enter the
46   #   report_compile_source_on_error() context manager.
47   
48   
49   @contextlib.contextmanager
50   def report_compile_source_on_error():
51       try:
52           yield
53       except Exception as exc:
54           tb = exc.__traceback__
55   
56           # Walk the traceback, looking for frames that have
57           # source attached
58           stack = []
59           while tb is not None:
60               filename = tb.tb_frame.f_code.co_filename
61               source = tb.tb_frame.f_globals.get("__compile_source__")
62   
63               if filename == "<string>" and source is not None:
64                   # What black magic are we doing here?  Intuitively, what
65                   # we would like to do is overwrite the co_filename on any
66                   # frames that were generated from exec/eval so that they
67                   # point to a temporary file that has the actual line
68                   # information, so Python's default error printer can print
69                   # useful line information on it.
70                   #
71                   # Writing out the temporary file is easy.  But overwriting
72                   # co_filename is not!  You can't modify the code object
73                   # associated with a frame.  You can, however, reconstruct
74                   # a traceback with entirely new frames from scratch, so that's
75                   # what we do.  But there's another problem, which is how to
76                   # make the frame?
77                   #
78                   # The black magic is we make a frankenstein frame and code
79                   # object which resembles the original frame/code enough so
80                   # that it will print properly under traceback and the default
81                   # error printer, but IT IS NOT THE ORIGINAL FRAME (you
82                   # couldn't, e.g., execute its code with different variables
83                   # and expect it to work.)
84   
85                   # Don't delete the temporary file so the user can inspect it
86                   # TODO: This creates a temporary file for every frame, but we
87                   # technically only need one per distinct __compile_source__
88                   with tempfile.NamedTemporaryFile(
89                       mode="w", delete=False, suffix=".py"
90                   ) as f:
91                       f.write(source)
92                   # Create a frame.  Python doesn't let you construct
93                   # FrameType directly, so just make one with compile
94                   frame = tb.tb_frame
95                   code = compile("__inspect_currentframe()", f.name, "eval")
96                   code = code.replace(co_name=frame.f_code.co_name)
97                   # Python 3.11 only
98                   if hasattr(frame.f_code, "co_linetable"):
99                       # We can't copy ALL of the metadata over, because you
100                      # can cause Python to segfault this way.  What exactly
101                      # do we need?  We need enough information for
102                      # traceback to be able to print the exception
103                      # correctly.  Code reading Lib/traceback.py reveals
104                      # that traceback calls code.co_positions() in order to
105                      # get the augmented line/col numbers.  Objects/codeobject.c,
106                      # specifically _PyCode_InitAddressRange, reveals that
107                      # this iterator is initialized from co_linetable and
108                      # co_firstfileno.  So copy these we must!
109                      code = code.replace(  # type: ignore[call-arg]
110                          co_linetable=frame.f_code.co_linetable,  # type: ignore[attr-defined]
111                          co_firstlineno=frame.f_code.co_firstlineno,  # type: ignore[attr-defined]
112                      )
113                  fake_frame = eval(
114                      code,
115                      frame.f_globals,
116                      {**frame.f_locals, "__inspect_currentframe": inspect.currentframe},
117                  )
118                  fake_tb = TracebackType(None, fake_frame, tb.tb_lasti, tb.tb_lineno)
119                  stack.append(fake_tb)
120              else:
121                  stack.append(tb)
122  
123              tb = tb.tb_next
124  
125          # Reconstruct the linked list
126          tb_next = None
127          for tb in reversed(stack):
128              tb.tb_next = tb_next
129              tb_next = tb
130  
131          raise exc.with_traceback(tb_next)  # noqa: B904

134  def shorten_filename(fn, *, base=None):
135      """Shorten a source filepath, with the assumption that torch/ subdirectories don't need to be shown to user."""
136      if base is None:
137          base = os.path.dirname(os.path.dirname(__file__))
138      # Truncate torch/foo.py to foo.py
139      try:
140          prefix = os.path.commonpath([fn, base])
141      except ValueError:
142          return fn
143      else:
144          return fn[len(prefix) + 1 :]

147  def format_frame(frame, *, base=None, line=False):
148      """ 
149      Format a FrameSummary in a short way, without printing full absolute path or code. 
150   
151      The idea is the result fits on a single line. 
152      """
153      extra_line = ""
154      if line:
155          extra_line = f"{frame.line}  # "
156      return f"{extra_line}{shorten_filename(frame.filename, base=base)}:{frame.lineno} in {frame.name}"

159  def format_traceback_short(tb):
160      """Format a TracebackType in a short way, printing only the inner-most frame."""
161      return format_frame(traceback.extract_tb(tb)[-1])

164  class CapturedTraceback:
165      __slots__ = ["tb", "skip"]
166  
167      def __init__(self, tb, skip=0):
168          self.tb = tb
169          self.skip = skip
170  
171      def cleanup(self):
172          self.tb = None
173  
174      def summary(self):
175          import torch._C._profiler
176  
177          if self.tb is None:
178              # TODO: Maybe indicate that the traceback was elided?
179              return traceback.StackSummary()
180  
181          return _extract_symbolized_tb(
182              torch._C._profiler.symbolize_tracebacks([self.tb])[0], self.skip
183          )
184  
185      def __getstate__(self):
186          return (
187              None,
188              {
189                  "tb": None,  # TB is not pickleable
190                  "skip": self.skip,
191              },
192          )
193  
194      @staticmethod
195      def extract(*, script=False, cpp=False, skip=0):
196          """ 
197          Like traceback.extract_stack(), but faster (approximately 20x faster); it 
198          is fast enough that you can unconditionally log stacks this way as part of 
199          normal execution.  It returns a torch._C._profiler.CapturedTraceback 
200          object that must be formatted specially with format_captured_tb. 
201   
202          By default, this only reports Python backtraces (like extract_stack).  You 
203          can set the script/cpp kwargs to also turn on TorchScript/C++ trace 
204          reporting. 
205          """
206          import torch._C._profiler
207  
208          if script or cpp:
209              assert skip == 0, "skip with script/cpp NYI"
210  
211          return CapturedTraceback(
212              torch._C._profiler.gather_traceback(python=True, script=script, cpp=cpp),
213              # Elide extract() frame if we don't have script/cpp frames.  If
214              # we do have those frames, it doesn't work so force zero.
215              0 if script or cpp else skip + 1,
216          )
217  
218      def format(self):
219          """ 
220          Formats a single torch._C._profiler.CapturedTraceback into a list of 
221          strings equivalent to the output of traceback.format_list.  Note that if 
222          pass it CapturedTraceback with C++ traces,  it is better not to use this 
223          function and use the batch formatting API format_captured_tbs to amortize 
224          the cost of symbolization 
225          """
226          return traceback.format_list(self.summary())
227  
228      @staticmethod
229      def format_all(tbs):
230          """ 
231          Bulk version of CapturedTraceback.format.  Returns a list of list of strings. 
232          """
233          import torch._C._profiler
234  
235          # Directly populate tracebacks that already have cached summaries
236          rs: list[Optional[list[str]]] = []
237          delayed_idxs = []
238          for i, tb in enumerate(tbs):
239              if tb.tb is None:
240                  rs.append([])
241              else:
242                  rs.append(None)
243                  delayed_idxs.append(i)
244  
245          torch._C._profiler.symbolize_tracebacks([tbs[i].tb for i in delayed_idxs])
246          for i in delayed_idxs:
247              rs[i] = traceback.format_list(tbs[i].summary())
248  
249          return rs

252  def _extract_symbolized_tb(tb, skip):
253      """ 
254      Given a symbolized traceback from symbolize_tracebacks, return a StackSummary object of 
255      pre-processed stack trace entries. 
256      """
257      stack = traceback.StackSummary()
258      for f in reversed(tb[skip:]):
259          stack.append(traceback.FrameSummary(f["filename"], f["line"], f["name"]))
260      return stack
