2    """ 
3    This module contains tooling to compare weights and activations 
4    across models. Example usage:: 
5     
6        import copy 
7        import torch 
8        import torch.ao.quantization.quantize_fx as quantize_fx 
9        import torch.ao.ns._numeric_suite_fx as ns 
10    
11       m = torch.nn.Sequential(torch.nn.Conv2d(1, 1, 1)).eval() 
12       mp = quantize_fx.prepare_fx(m, {"": torch.ao.quantization.default_qconfig}) 
13       # We convert a copy because we need the original prepared model 
14       # to be available for comparisons, and `quantize_fx.convert_fx` is inplace. 
15       mq = quantize_fx.convert_fx(copy.deepcopy(mp)) 
16    
17       # 
18       # Comparing weights 
19       # 
20    
21       # extract weight pairs 
22       weight_comparison = ns.extract_weights("a", mp, "b", mq) 
23    
24       # add SQNR for each comparison, inplace 
25       ns.extend_logger_results_with_comparison( 
26           weight_comparison, "a", "b", torch.ao.ns.fx.utils.compute_sqnr, "sqnr" 
27       ) 
28    
29       # weight_comparison contains the weights from `mp` and `mq` stored 
30       # in pairs, and can be used for further analysis. 
31    
32    
33       # 
34       # Comparing activations, with error propagation 
35       # 
36    
37       # add loggers 
38       mp_ns, mq_ns = ns.add_loggers( 
39           "a", copy.deepcopy(mp), "b", copy.deepcopy(mq), ns.OutputLogger 
40       ) 
41    
42       # send an example datum to capture intermediate activations 
43       datum = torch.randn(1, 1, 1, 1) 
44       mp_ns(datum) 
45       mq_ns(datum) 
46    
47       # extract intermediate activations 
48       act_comparison = ns.extract_logger_info(mp_ns, mq_ns, ns.OutputLogger, "b") 
49    
50       # add SQNR for each comparison, inplace 
51       ns.extend_logger_results_with_comparison( 
52           act_comparison, "a", "b", torch.ao.ns.fx.utils.compute_sqnr, "sqnr" 
53       ) 
54    
55       # act_comparison contains the activations from `mp_ns` and `mq_ns` stored 
56       # in pairs, and can be used for further analysis. 
57    
58       # 
59       # Comparing activations, without error propagation 
60       # 
61    
62       # create shadow model 
63       mp_shadows_mq = ns.add_shadow_loggers( 
64           "a", copy.deepcopy(mp), "b", copy.deepcopy(mq), ns.OutputLogger 
65       ) 
66    
67       # send an example datum to capture intermediate activations 
68       datum = torch.randn(1, 1, 1, 1) 
69       mp_shadows_mq(datum) 
70    
71       # extract intermediate activations 
72       shadow_act_comparison = ns.extract_shadow_logger_info( 
73           mp_shadows_mq, ns.OutputLogger, "b" 
74       ) 
75    
76       # add SQNR for each comparison, inplace 
77       ns.extend_logger_results_with_comparison( 
78           shadow_act_comparison, "a", "b", torch.ao.ns.fx.utils.compute_sqnr, "sqnr" 
79       ) 
80    
81       # shadow_act_comparison contains the activations from `mp_ns` and `mq_ns` stored 
82       # in pairs, and can be used for further analysis. 
83

136  class OutputLogger(nn.Module):
137      """ 
138      Base class for capturing intermediate values. 
139      """
140  
141      stats: list[torch.Tensor]
142      stats_rnn: list[RNNReturnType]
143  
144      # Mark as impure so that calls to it will not be removed during DCE.
145      _is_impure = True
146  
147      def __init__(
148          self,
149          ref_node_name: str,
150          prev_node_name: str,
151          model_name: str,
152          ref_name: str,
153          prev_node_target_type: str,
154          ref_node_target_type: str,
155          results_type: str,
156          index_within_arg: int,
157          index_of_arg: int,
158          fqn: Optional[str],
159          qconfig_str: Optional[str] = "",
160      ):
161          super().__init__()
162          self.stats: list[torch.Tensor] = []
163          self.stats_rnn: list[RNNReturnType] = []
164  
165          # name of the node which was responsible for adding this logger
166          # Note:
167          # - if we are logging node outputs, this is the same as prev_node_name
168          # - if we are logging node inputs, this is the name of the node
169          #   whose input this logger is logging.
170          #
171          # example, where logger1 is logging input of op1 and logger2 is logging
172          #    the output of op1:
173          #
174          #  x1 -> logger1 -> op1 -> logger2 -> x2
175          #
176          # in this example,
177          #   - logger1's prev_node_name is x1 and ref_node_name is op1
178          #   - logger2's prev_node_name is op1 and ref_node_name is op1
179          self.ref_node_name = ref_node_name
180          # name of the node whose output this Logger is capturing
181          self.prev_node_name = prev_node_name
182  
183          # name of the model from which the node originated from
184          self.model_name = model_name
185          # reference name, used to match loggers from separate models
186          # to each other
187          self.ref_name = ref_name
188          # type of the target of the node whose output this logger is logging
189          self.prev_node_target_type = prev_node_target_type
190          # type of the target of the node which was responsible for adding this
191          # logger
192          self.ref_node_target_type = ref_node_target_type
193          # what kind of values are inside of stats
194          self.results_type = results_type
195          # index of this node within the arg of the input/output node
196          # for example, in cat([x1, x2, x3], dim=0), x2 would have index_within_arg == 1
197          self.index_within_arg = index_within_arg
198          # index of this node within the args of the input/output node
199          # for example, in add(x1, x2), x2 would have index_of_arg == 1
200          self.index_of_arg = index_of_arg
201          # fully qualified name
202          self.fqn = fqn
203          # if loggers are added before prepare_fx, but we do not want
204          # collect results of calibration, only results after convert_fx
205          # so, we add a flag to control whether this logger collects data
206          self.enabled = True
207          # string representation of qconfig
208          self.qconfig_str = qconfig_str
209          # this can be turned off to reduce memory usage during calibration
210          self.save_activations = True
211  
212      # Note: cannot annotate the type of x because TorchScript does not support
213      #   the Union type.
214      def forward(self, x):
215          # fmt: off
216          """ 
217          """  # blank docblock to make autodoc happy
218          # fmt: on
219          # TODO(future PR): consider designing this better, as the difference
220          # between these two flags is subtle and not obvious.
221          if not self.enabled:
222              return x
223          if not self.save_activations:
224              return x
225          # TODO(future PR): consider refactoring this to better reuse the parent
226          # class
227          if isinstance(x, torch.Tensor):
228              self.stats.append(x.detach())
229          elif isinstance(x, tuple) and len(x) == 2 and len(x[1]) == 2:
230              new_res = (x[0].detach(), (x[1][0].detach(), x[1][1].detach()))
231              self.stats_rnn.append(new_res)
232          return x
233  
234      def __repr__(self):
235          clean_dict = {
236              k: v
237              for k, v in self.__dict__.items()
238              # skip nn.Module keys
239              if (k != "training") and not k.startswith("_")
240          }
241          return f"OutputLogger({clean_dict})"

244  class OutputComparisonLogger(OutputLogger):
245      """ 
246      Same as OutputLogger, but also requires the original activation 
247      in order to calculate the comparison at calibration time 
248      """
249  
250      def __init__(self, *args, **kwargs):
251          super().__init__(*args, **kwargs)
252          # TODO(future PR): make the comparison function configurable
253          self.comparison_fn = torch.ao.ns.fx.utils.compute_sqnr
254          self.comparison_fn_name = "sqnr"
255          # precalculated comparisons of logger output versus reference
256          self.comparisons = []
257          # precalculated comparisons function
258  
259      def forward(self, x, x_ref):  # type: ignore[override]
260          # fmt: off
261          """ 
262          """  # blank docblock to make autodoc happy
263          # fmt: on
264          if not self.enabled:
265              return x
266          assert isinstance(x, torch.Tensor), "non-tensor inputs not yet supported"
267          if self.save_activations:
268              # save the activation, for debugging
269              self.stats.append(x.detach())
270          # save the comparison
271          self.comparisons.append(self.comparison_fn(x, x_ref))
272          return x
273  
274      def __repr__(self):
275          clean_dict = {
276              k: v
277              for k, v in self.__dict__.items()
278              # skip nn.Module keys
279              if (k != "training") and not k.startswith("_")
280          }
281          return f"OutputComparisonLogger({clean_dict})"

302  def _extract_weights_one_model(
303      model_name: str,
304      model: GraphModule,
305      nodes_and_names_to_instrument: list[tuple[Node, str]],
306      results: NSResultsType,
307      op_to_type_to_weight_extraction_fn: Optional[
308          dict[str, dict[Callable, Callable]]
309      ] = None,
310  ) -> None:
311      torch._C._log_api_usage_once(
312          "quantization_api._numeric_suite_fx._extract_weights_one_model"
313      )
314      for node, ref_name in nodes_and_names_to_instrument:
315          res_type = NSSingleResultValuesType.WEIGHT.value
316          extracted_weight = extract_weight_from_node(
317              node, model, op_to_type_to_weight_extraction_fn
318          )
319          if extracted_weight:
320              if ref_name not in results:
321                  results[ref_name] = {res_type: {}}
322              results[ref_name][res_type][model_name] = [extracted_weight]

325  def _extract_weights_impl(
326      model_name_a: str,
327      gm_a: GraphModule,
328      model_name_b: str,
329      gm_b: GraphModule,
330      base_name_to_sets_of_related_ops: Optional[dict[str, set[NSNodeTargetType]]] = None,
331      unmatchable_types_map: Optional[dict[str, set[NSNodeTargetType]]] = None,
332      op_to_type_to_weight_extraction_fn: Optional[
333          dict[str, dict[Callable, Callable]]
334      ] = None,
335  ) -> NSResultsType:
336      torch._C._log_api_usage_once(
337          "quantization_api._numeric_suite_fx._extract_weights_impl"
338      )
339      matched_subgraph_pairs = get_matching_subgraph_pairs(
340          gm_a, gm_b, base_name_to_sets_of_related_ops, unmatchable_types_map
341      )
342  
343      # split the subgraph pairs into one data structure for each model
344      nodes_and_names_to_instrument_a: list[tuple[Node, str]] = []
345      nodes_and_names_to_instrument_b: list[tuple[Node, str]] = []
346      for match_name, match in matched_subgraph_pairs.items():
347          subgraph_a, subgraph_b = match
348          nodes_and_names_to_instrument_a.append((subgraph_a.base_op_node, match_name))
349          nodes_and_names_to_instrument_b.append((subgraph_b.base_op_node, match_name))
350  
351      # populate the results, one model at a time
352      results: NSResultsType = {}
353      _extract_weights_one_model(
354          model_name_a,
355          gm_a,
356          nodes_and_names_to_instrument_a,
357          results,
358          op_to_type_to_weight_extraction_fn,
359      )
360      _extract_weights_one_model(
361          model_name_b,
362          gm_b,
363          nodes_and_names_to_instrument_b,
364          results,
365          op_to_type_to_weight_extraction_fn,
366      )
367  
368      # fill in missing fqn entries
369      maybe_add_missing_fqns(results)
370  
371      # rekey on names of nodes in gm_b
372      results = rekey_logger_info_on_node_name_of_model(results, model_name_b)
373  
374      return results

377  def extract_weights(
378      model_name_a: str,
379      model_a: nn.Module,
380      model_name_b: str,
381      model_b: nn.Module,
382      base_name_to_sets_of_related_ops: Optional[dict[str, set[NSNodeTargetType]]] = None,
383      unmatchable_types_map: Optional[dict[str, set[NSNodeTargetType]]] = None,
384      op_to_type_to_weight_extraction_fn: Optional[
385          dict[str, dict[Callable, Callable]]
386      ] = None,
387  ) -> NSResultsType:
388      """ 
389      Extract weights from model A and model B, and return a comparison. 
390   
391      Args: 
392          model_name_a: string name of model A to use in results 
393          model_a: model A 
394          model_name_b: string name of model B to use in results 
395          model_b: model B 
396          base_name_to_sets_of_related_ops: optional override of subgraph base nodes, subject to change 
397          unmatchable_types_map: optional override of unmatchable types, subject to change 
398          op_to_type_to_weight_extraction_fn: optional override of function which extracts weight 
399              from a type, subject to change 
400   
401      Return: 
402          NSResultsType, containing the weight comparisons 
403      """
404  
405      torch._C._log_api_usage_once("quantization_api._numeric_suite_fx.extract_weights")
406      if base_name_to_sets_of_related_ops is None:
407          base_name_to_sets_of_related_ops = get_base_name_to_sets_of_related_ops()
408  
409      # TODO(future PR): expose these
410      skipped_module_names: list[str] = []
411      skipped_module_classes: list[Callable] = []
412      tracer_a = NSTracer(skipped_module_names, skipped_module_classes)
413      tracer_b = NSTracer(skipped_module_names, skipped_module_classes)
414      gm_a = GraphModule(model_a, tracer_a.trace(model_a))
415      maybe_model_a_node_name_to_scope = _get_observed_graph_module_attr(
416          model_a, "node_name_to_scope"
417      )
418      if maybe_model_a_node_name_to_scope is not None:
419          gm_a._node_name_to_scope = maybe_model_a_node_name_to_scope
420      gm_b = GraphModule(model_b, tracer_b.trace(model_b))
421      maybe_model_b_node_name_to_scope = _get_observed_graph_module_attr(
422          model_b, "node_name_to_scope"
423      )
424      if maybe_model_b_node_name_to_scope is not None:
425          gm_b._node_name_to_scope = maybe_model_b_node_name_to_scope
426      return _extract_weights_impl(
427          model_name_a,
428          gm_a,
429          model_name_b,
430          gm_b,
431          base_name_to_sets_of_related_ops,
432          unmatchable_types_map,
433          op_to_type_to_weight_extraction_fn,
434      )

438      model_name: str,
439      model: GraphModule,
440      nodes_and_names_to_instrument_inputs: list[tuple[Node, str, str]],
441      nodes_and_names_to_instrument_outputs: list[tuple[Node, str, str]],
442      logger_cls: Callable,
443  ) -> nn.Module:
444      torch._C._log_api_usage_once(
445          "quantization_api._numeric_suite_fx._add_loggers_one_model"
446      )
447  
448      # TODO(future PR): do not observe nodes we do not care
449      #   about (both fp32, denylist, etc)
450      node_to_instrument_inputs_to_ref_name: dict[Node, tuple[str, str]] = {}
451      node_to_instrument_outputs_to_ref_name: dict[Node, tuple[str, str]] = {}
452      for node, ref_name, ref_node_type in nodes_and_names_to_instrument_inputs:

456

458          model,
459          node_to_instrument_inputs_to_ref_name,
460          node_to_instrument_outputs_to_ref_name,
461          logger_cls,
462          model_name,
463      )
464      return model

467  def _add_loggers_impl(
468      name_a: str,
469      gm_a: GraphModule,
470      name_b: str,
471      gm_b: GraphModule,
472      logger_cls: Callable,
473      should_log_inputs: bool,
474      base_name_to_sets_of_related_ops: Optional[dict[str, set[NSNodeTargetType]]] = None,
475      unmatchable_types_map: Optional[dict[str, set[NSNodeTargetType]]] = None,
476  ) -> tuple[nn.Module, nn.Module]:
477      torch._C._log_api_usage_once("quantization_api._numeric_suite_fx._add_loggers_impl")
478      matched_subgraph_pairs = get_matching_subgraph_pairs(
479          gm_a, gm_b, base_name_to_sets_of_related_ops, unmatchable_types_map
480      )
481      nodes_and_names_to_instrument_inputs_a = []
482      nodes_and_names_to_instrument_inputs_b = []
483      nodes_and_names_to_instrument_outputs_a = []
484      nodes_and_names_to_instrument_outputs_b = []
485      for match_name, (subgraph_a, subgraph_b) in matched_subgraph_pairs.items():
486          ref_node_type_a = get_target_type_str(subgraph_a.base_op_node, gm_a)
487          ref_node_type_b = get_target_type_str(subgraph_b.base_op_node, gm_b)
488          # Note: for matching inputs we use start_node, such as observing
489          # the input of linear in linear-relu
490          if should_log_inputs:
491              nodes_and_names_to_instrument_inputs_a.append(
492                  (subgraph_a.start_node, match_name, ref_node_type_a)
493              )
494              nodes_and_names_to_instrument_inputs_b.append(
495                  (subgraph_b.start_node, match_name, ref_node_type_b)
496              )
497          # Note: for matching activations we always use end_node,
498          # such as observing the output of relu in linear-relu
499          nodes_and_names_to_instrument_outputs_a.append(
500              (subgraph_a.end_node, match_name, ref_node_type_a)
501          )
502          nodes_and_names_to_instrument_outputs_b.append(
503              (subgraph_b.end_node, match_name, ref_node_type_b)
504          )
505  
506      new_model_a = _add_loggers_one_model(
507          name_a,
508          gm_a,
509          nodes_and_names_to_instrument_inputs_a,
510          nodes_and_names_to_instrument_outputs_a,
511          logger_cls,
512      )
513      new_model_b = _add_loggers_one_model(
514          name_b,
515          gm_b,
516          nodes_and_names_to_instrument_inputs_b,
517          nodes_and_names_to_instrument_outputs_b,
518          logger_cls,
519      )
520      return (new_model_a, new_model_b)

523  def add_loggers(
524      name_a: str,
525      model_a: nn.Module,
526      name_b: str,
527      model_b: nn.Module,
528      logger_cls: Callable,
529      should_log_inputs: bool = False,
530      base_name_to_sets_of_related_ops: Optional[dict[str, set[NSNodeTargetType]]] = None,
531      unmatchable_types_map: Optional[dict[str, set[NSNodeTargetType]]] = None,
532  ) -> tuple[nn.Module, nn.Module]:
533      """ 
534      Instrument model A and model B with loggers. 
535   
536      Args: 
537          name_a: string name of model A to use in results 
538          model_a: model A 
539          name_b: string name of model B to use in results 
540          model_b: model B 
541          logger_cls: class of Logger to use 
542          base_name_to_sets_of_related_ops: optional override of subgraph base nodes, subject to change 
543          unmatchable_types_map: optional override of unmatchable types, subject to change 
544   
545      Return: 
546          Returns a tuple of (model_a_with_loggers, model_b_with_loggers).  Modifies both models inplace. 
547      """
548  
549      torch._C._log_api_usage_once("quantization_api._numeric_suite_fx.add_loggers")
550      # TODO(future PR): expose these
551      skipped_module_names: list[str] = []
552      skipped_module_classes: list[Callable] = []
553      tracer_a = NSTracer(skipped_module_names, skipped_module_classes)
554      tracer_b = NSTracer(skipped_module_names, skipped_module_classes)
555      gm_a = GraphModule(model_a, tracer_a.trace(model_a))
556      maybe_model_a_node_name_to_scope = _get_observed_graph_module_attr(
557          model_a, "node_name_to_scope"
558      )
559      if maybe_model_a_node_name_to_scope is not None:
560          gm_a._node_name_to_scope = maybe_model_a_node_name_to_scope
561      gm_b = GraphModule(model_b, tracer_b.trace(model_b))
562      maybe_model_b_node_name_to_scope = _get_observed_graph_module_attr(
563          model_b, "node_name_to_scope"
564      )
565      if maybe_model_b_node_name_to_scope is not None:
566          gm_b._node_name_to_scope = maybe_model_b_node_name_to_scope
567      return _add_loggers_impl(
568          name_a,
569          gm_a,
570          name_b,
571          gm_b,
572          logger_cls,
573          should_log_inputs=should_log_inputs,
574          base_name_to_sets_of_related_ops=base_name_to_sets_of_related_ops,

579  def _extract_logger_info_one_model(
580      model: nn.Module,
581      results: NSResultsType,
582      logger_cls: Callable,
583  ) -> None:
584      torch._C._log_api_usage_once(
585          "quantization_api._numeric_suite_fx._extract_logger_info_one_model"
586      )
587      for _gm_name, mod in model.named_modules():
588          # TODO(future PR): better check when scripted
589          is_logger = isinstance(mod, logger_cls) or (  # type: ignore[arg-type]
590              isinstance(mod, torch.jit.RecursiveScriptModule)
591              and mod.original_name == "OutputLogger"
592          )
593          if is_logger:
594              key = mod.ref_name
595              if key not in results:
596                  results[key] = {}
597              assert mod.model_name not in results[key], (
598                  f"{mod.model_name} is already present in results"
599              )
600              if mod.results_type not in results[key]:
601                  results[key][mod.results_type] = {}
602              if mod.model_name not in results[key][mod.results_type]:
603                  results[key][mod.results_type][mod.model_name] = []
604              stats_to_use = mod.stats
605              if len(mod.stats_rnn) > 0:
606                  stats_to_use = mod.stats_rnn
607              data = {
608                  "type": mod.results_type,
609                  "values": stats_to_use,
610                  "ref_node_name": mod.ref_node_name,
611                  "ref_node_target_type": mod.ref_node_target_type,
612                  "prev_node_name": mod.prev_node_name,
613                  "prev_node_target_type": mod.prev_node_target_type,
614                  "index_within_arg": mod.index_within_arg,
615                  "index_of_arg": mod.index_of_arg,
616                  "fqn": mod.fqn,
617                  "qconfig_str": mod.qconfig_str,
618              }
619              if hasattr(mod, "comparisons"):
620                  data["comparisons"] = mod.comparisons
621                  data["comparison_fn_name"] = mod.comparison_fn_name
622              else:
623                  data["comparisons"] = []
624                  data["comparison_fn_name"] = ""
625              results[key][mod.results_type][mod.model_name].append(data)
626              # ensure the list stays sorted
627              results[key][mod.results_type][mod.model_name].sort(
628                  key=lambda res: f"{res['index_of_arg']}:{res['index_within_arg']}"
629              )

788  def extend_logger_results_with_comparison(
789      results: NSResultsType,
790      model_name_1: str,
791      model_name_2: str,
792      comparison_fn: Callable[[torch.Tensor, torch.Tensor], torch.Tensor],
793      comparison_name: str,
794  ) -> None:
795      """ 
796      Compares the logged values from `model_name_2` against the corresponding 
797      values in `model_name_1`, using `comparison_fn`. Records the result 
798      in `model_name_2`'s results under `comparison_name`. Modifies `results` inplace. 
799   
800      Args: 
801          results: the result data structure from `extract_logger_info` or 
802            `extract_shadow_logger_info`. 
803          model_name_1: string name of model 1 
804          model_name_2: string name of model 2 
805          comparison_fn: function to compare two Tensors 
806          comparison_name: string name of model to use for 
807            layer names in the output 
808      """
809      for results_type_to_results in results.values():
810          for model_name_to_results in results_type_to_results.values():
811              assert model_name_1 in model_name_to_results, (
812                  f"{model_name_1} not found in results"
813              )
814              assert model_name_2 in model_name_to_results, (
815                  f"{model_name_2} not found in results"
816              )
817  
818              results_1 = model_name_to_results[model_name_1]
819              results_2 = model_name_to_results[model_name_2]
820  
821              for result_2 in results_2:
822                  index_within_arg_2 = result_2["index_within_arg"]
823                  index_of_arg_2 = result_2["index_of_arg"]
824                  # find corresponding result_1
825                  result_1 = None
826                  for cur_result_1 in results_1:
827                      index_within_arg_1 = cur_result_1["index_within_arg"]
828                      index_of_arg_1 = cur_result_1["index_of_arg"]
829                      if (index_within_arg_1 == index_within_arg_2) and (
830                          index_of_arg_1 == index_of_arg_2
831                      ):
832                          result_1 = cur_result_1
833                          break
834                  assert result_1 is not None
835  
836                  values_1 = result_1["values"]
837                  values_2 = result_2["values"]
838                  result_2[comparison_name] = []
839                  for value_1, value_2 in zip(values_1, values_2):
840                      comparison_result = comparison_fn(value_1, value_2)
841                      result_2[comparison_name].append(comparison_result)

844  def prepare_n_shadows_model(
845      model: torch.nn.Module,
846      example_inputs: Any,
847      qconfig_multi_mapping: QConfigMultiMapping,
848      backend_config: BackendConfig,
849      custom_prepare_fn: Optional[Callable] = None,
850      custom_prepare_kwargs: Optional[dict[str, Any]] = None,
851      custom_tracer: Any = None,
852  ) -> GraphModule:
853      """ 
854      Given a model with a graph with M ops such as 
855   
856   
857        args_kwargs_m -> op_m -> output_m 
858   
859   
860      And a set of N qconfigs for each op, creates a new model, with 
861      each of the subgraph of `op_m` transformed into 
862   
863      .. code:: 
864   
865             |---------> op_m_n -> log_m_n 
866             |                     / 
867        args_kwargs_m ---------> op_m -> log_m_0 
868   
869      Where op_m_n is op_m wrapped in a submodule and transformed with 
870      qconfig_n, and its inner graph looks like 
871   
872      .. code:: 
873   
874        args_m -------- op_m_prepared_with_qconfig_n -> out_m_n 
875                    / 
876        kwargs_m --- 
877   
878      This is useful for testing different quantization of multiple layers in 
879      a single pass through the model. 
880   
881      High level TODOs for future PRs: 
882      * figure out a better way to name the output structure 
883      * return a results data structure instead of printing it out 
884      * add examples to docblocks 
885      """
886  
887      if custom_tracer is None:
888          tracer = quantize_fx.QuantizationTracer([], [])
889      else:
890          tracer = custom_tracer
891      mt = torch.fx.GraphModule(model, tracer.trace(model))
892      # this is necessary to ensure logger FQNs get populated
893      mt._node_name_to_scope = tracer.node_name_to_scope  # type: ignore[assignment]
894  
895      # run example input propagation, we need this to call prepare_fx on
896      # individual subgraphs
897      output_prop = OutputProp(mt)
898      output_prop.propagate(*example_inputs)
899  
900      # Find the set of subgraphs in the original graph which we need to
901      # consider.
902      modules = dict(mt.named_modules(remove_duplicate=False))
903      patterns = _get_pattern_to_quantize_handlers(backend_config)
904      root_node_getter_mapping = get_fusion_pattern_to_root_node_getter(backend_config)
905      standalone_module_names: list[str] = []
906      standalone_module_classes: list[type] = []
907      custom_module_classes: list[type] = []
908      matches = _find_matches(
909          mt.graph,
910          modules,
911          patterns,
912          root_node_getter_mapping,
913          standalone_module_names,
914          standalone_module_classes,
915          custom_module_classes,
916      )
917      subgraphs_dedup: dict[str, list[Node]] = _get_dedup_subgraphs(matches)
918  
919      # generate node to qconfig for each subgraph
920      # TODO(future PR): deduplicate repeating entries
921      list_of_node_name_to_qconfig: list[dict[str, QConfigAny]] = []
922      for qconfig_mapping in qconfig_multi_mapping.qconfig_mappings_list:
923          node_name_to_qconfig = _generate_node_name_to_qconfig(
924              mt, modules, mt.graph, qconfig_mapping, tracer.node_name_to_scope
925          )
926          list_of_node_name_to_qconfig.append(node_name_to_qconfig)
927  
928      # For each region in the model, do the following:
929      #   For each qconfig for that region, do the following:
930      #     1. create a copy of the region wrapped in a module
931      #     2. pass original args, original kwargs, and expected output to module
932      #     3. add an output comparison logger and hook it up to compare
933      #        actual output to expected output
934      #     4. run `prepare_fx` on the module
935      for subgraph_idx, (match_name, nodes_in_this_subgraph) in enumerate(
936          subgraphs_dedup.items()
937      ):
938          create_n_transformed_and_logged_copies_of_subgraph(
939              mt,
940              subgraph_idx,
941              match_name,
942              nodes_in_this_subgraph,
943              qconfig_multi_mapping.qconfig_mappings_list,
944              list_of_node_name_to_qconfig,
945              custom_prepare_fn,
946              custom_prepare_kwargs,  # type: ignore[arg-type]
947          )
948  
949      return mt

953  def _prepare_n_shadows_add_loggers_model(
954      model: torch.nn.Module,
955      example_inputs: Any,
956      qconfig_mapping: QConfigMapping,
957      backend_config: BackendConfig,
958  ) -> torch.nn.Module:
959      r""" 
960      Note: this API is not recommended for wide usage, it is only 
961      provided for customers who need to migrate from the `add_loggers` 
962      API. 
963   
964      This creates a model which provides logging for the following 
965      problem: if we quantize `model` with `qconfig_mapping` and feed 
966      the same input through both models, log the comparisons of 
967      corresponding intermediate layers. 
968   
969      The problem is solved with a single model.  Specifically, we 
970      partition `model` into N subgraphs, create a copy of each relevant 
971      subgraph, wrap it in a module, apply the quantization API to that 
972      module, and hook up loggers to measure the comparisons. 
973   
974      Example starting graph: 
975   
976        x0 -> op0 -> x1 -> op1 -> x2 
977   
978      Example config: quantize op0 to int8, do nothing to op1. 
979      The following graph will be created: 
980   
981      .. code:: 
982   
983        x0_0 -> op0_0 -> x1_0 -> log -----> op1_0 -> x2_0 -> log 
984         \                        \                           \       # noqa: W605 
985           ---> op0_1 -> x1_1 ----> clog -> op1_0 -> x2_1 ----> clog 
986   
987      Where op0_0 is op0, op0_1 is op0 wrapped in a submodule and quantized 
988      to int8, op1_0 is op1 (appearing in the graph twice), log is a logger, 
989      and clog is a comparison logger. 
990      """
991  
992      tracer = quantize_fx.QuantizationTracer([], [])
993      mt = torch.fx.GraphModule(model, tracer.trace(model))
994      # this is necessary to ensure logger FQNs get populated
995      mt._node_name_to_scope = tracer.node_name_to_scope  # type: ignore[assignment]
996  
997      # run example input propagation, we need this to call prepare_fx on
998      # individual subgraphs
999      output_prop = OutputProp(mt)
1000     output_prop.propagate(*example_inputs)
1001 
1002     # Find the set of subgraphs in the original graph which we need to
1003     # consider.
1004     modules = dict(mt.named_modules(remove_duplicate=False))
1005     patterns = _get_pattern_to_quantize_handlers(backend_config)
1006     root_node_getter_mapping = get_fusion_pattern_to_root_node_getter(backend_config)
1007     standalone_module_names: list[str] = []
1008     standalone_module_classes: list[type] = []
1009     custom_module_classes: list[type] = []
1010     matches = _find_matches(
1011         mt.graph,
1012         modules,
1013         patterns,
1014         root_node_getter_mapping,
1015         standalone_module_names,
1016         standalone_module_classes,
1017         custom_module_classes,
1018     )
1019     subgraphs_dedup: dict[str, list[Node]] = _get_dedup_subgraphs(matches)
1020 
1021     # generate node to qconfig for each subgraph
1022     node_name_to_qconfig = _generate_node_name_to_qconfig(
1023         mt, modules, mt.graph, qconfig_mapping, tracer.node_name_to_scope
1024     )
1025 
1026     # Now, mutate the graph to be the add_loggers graph with propagation
1027     # error.
1028     create_add_loggers_graph(mt, subgraphs_dedup, qconfig_mapping, node_name_to_qconfig)
1029 
1030     return mt
