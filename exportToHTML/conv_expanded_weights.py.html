1    from typing import Any, Callable, TypeVar
2    from typing_extensions import ParamSpec

8    _P = ParamSpec("_P")
9    _R = TypeVar("_R")

11   from .conv_utils import (
12       conv_args_and_kwargs,
13       conv_backward,
14       conv_input_for_string_padding,
15       conv_picker,
16   )
17   from .expanded_weights_impl import ExpandedWeight, implements_per_sample_grads
18   from .expanded_weights_utils import forward_helper
19   
20   
21   @implements_per_sample_grads(F.conv1d)
22   @implements_per_sample_grads(F.conv2d)
23   @implements_per_sample_grads(F.conv3d)
24   class ConvPerSampleGrad(torch.autograd.Function):
25       @staticmethod
26       def forward(
27           ctx: Any,
28           kwarg_names: list[str],
29           conv_fn: Callable[_P, _R],
30           *expanded_args_and_kwargs: Any,
31       ) -> torch.Tensor:
32           expanded_args, expanded_kwargs = conv_args_and_kwargs(
33               kwarg_names, expanded_args_and_kwargs
34           )
35           orig_input = expanded_args[0]
36           was_same_padding = expanded_kwargs["padding"] == "same"
37   
38           if isinstance(expanded_kwargs["padding"], str):
39               # if padding is a string, we'll do the necessary padding (slowly) using F.pad
40               kernel_size = expanded_args[1].shape[2:]
41               padding, dilation = expanded_kwargs["padding"], expanded_kwargs["dilation"]
42               input = conv_input_for_string_padding(
43                   conv_fn, padding, expanded_args[0], dilation, kernel_size
44               )
45               expanded_args = (input, expanded_args[1])
46               # since we've already done the padding, don't need any more
47               expanded_kwargs["padding"] = 0
48   
49           output = forward_helper(conv_fn, expanded_args, expanded_kwargs)
50           input, weight = expanded_args
51           batched_dim_size = conv_picker(conv_fn, 3, 4, 5)
52           if input.dim() != batched_dim_size:
53               raise RuntimeError(
54                   f"Expanded Weights only support convolution with batched input, got {conv_fn} with an"
55                   f"unbatched input of dim {input.dim()}, expected input of dim {batched_dim_size}"
56               )
57   
58           ctx.conv_fn = conv_fn
59   
60           ctx.batch_size = orig_input.shape[0]
61           ctx.input_required_grad = orig_input.requires_grad
62           ctx.orig_input_shape = orig_input.shape
63           ctx.was_same_padding = was_same_padding
64           ctx.stride, ctx.padding = expanded_kwargs["stride"], expanded_kwargs["padding"]
65           ctx.dilation, ctx.groups = (
66               expanded_kwargs["dilation"],
67               expanded_kwargs["groups"],
68           )
69   
70           if isinstance(weight, ExpandedWeight):
71               ctx.input = input
72           ctx.weight = weight
73           ctx.bias = expanded_kwargs["bias"]
74   
75           return output
76   
77       @staticmethod
78       def backward(ctx: Any, *grad_outputs: Any) -> Any:
79           return conv_backward(ctx.conv_fn, ctx, grad_outputs[0])
