12   @slow
13   def test_bicycle():
14       # Code to get equations of motion for a bicycle modeled as in:
15       # J.P Meijaard, Jim M Papadopoulos, Andy Ruina and A.L Schwab. Linearized
16       # dynamics equations for the balance and steer of a bicycle: a benchmark
17       # and review. Proceedings of The Royal Society (2007) 463, 1955-1982
18       # doi: 10.1098/rspa.2007.1857
19   
20       # Note that this code has been crudely ported from Autolev, which is the
21       # reason for some of the unusual naming conventions. It was purposefully as
22       # similar as possible in order to aide debugging.
23   
24       # Declare Coordinates & Speeds
25       # Simple definitions for qdots - qd = u
26       # Speeds are:
27       # - u1: yaw frame ang. rate
28       # - u2: roll frame ang. rate
29       # - u3: rear wheel frame ang. rate (spinning motion)
30       # - u4: frame ang. rate (pitching motion)
31       # - u5: steering frame ang. rate
32       # - u6: front wheel ang. rate (spinning motion)
33       # Wheel positions are ignorable coordinates, so they are not introduced.
34       q1, q2, q4, q5 = dynamicsymbols('q1 q2 q4 q5')
35       q1d, q2d, q4d, q5d = dynamicsymbols('q1 q2 q4 q5', 1)
36       u1, u2, u3, u4, u5, u6 = dynamicsymbols('u1 u2 u3 u4 u5 u6')
37       u1d, u2d, u3d, u4d, u5d, u6d = dynamicsymbols('u1 u2 u3 u4 u5 u6', 1)
38   
39       # Declare System's Parameters
40       WFrad, WRrad, htangle, forkoffset = symbols('WFrad WRrad htangle forkoffset')
41       forklength, framelength, forkcg1 = symbols('forklength framelength forkcg1')
42       forkcg3, framecg1, framecg3, Iwr11 = symbols('forkcg3 framecg1 framecg3 Iwr11')
43       Iwr22, Iwf11, Iwf22, Iframe11 = symbols('Iwr22 Iwf11 Iwf22 Iframe11')
44       Iframe22, Iframe33, Iframe31, Ifork11 = symbols('Iframe22 Iframe33 Iframe31 Ifork11')
45       Ifork22, Ifork33, Ifork31, g = symbols('Ifork22 Ifork33 Ifork31 g')
46       mframe, mfork, mwf, mwr = symbols('mframe mfork mwf mwr')
47   
48       # Set up reference frames for the system
49       # N - inertial
50       # Y - yaw
51       # R - roll
52       # WR - rear wheel, rotation angle is ignorable coordinate so not oriented
53       # Frame - bicycle frame
54       # TempFrame - statically rotated frame for easier reference inertia definition
55       # Fork - bicycle fork
56       # TempFork - statically rotated frame for easier reference inertia definition
57       # WF - front wheel, again posses a ignorable coordinate
58       N = ReferenceFrame('N')
59       Y = N.orientnew('Y', 'Axis', [q1, N.z])
60       R = Y.orientnew('R', 'Axis', [q2, Y.x])
61       Frame = R.orientnew('Frame', 'Axis', [q4 + htangle, R.y])
62       WR = ReferenceFrame('WR')
63       TempFrame = Frame.orientnew('TempFrame', 'Axis', [-htangle, Frame.y])
64       Fork = Frame.orientnew('Fork', 'Axis', [q5, Frame.x])
65       TempFork = Fork.orientnew('TempFork', 'Axis', [-htangle, Fork.y])
66       WF = ReferenceFrame('WF')
67   
68       # Kinematics of the Bicycle First block of code is forming the positions of
69       # the relevant points
70       # rear wheel contact -> rear wheel mass center -> frame mass center +
71       # frame/fork connection -> fork mass center + front wheel mass center ->
72       # front wheel contact point
73       WR_cont = Point('WR_cont')
74       WR_mc = WR_cont.locatenew('WR_mc', WRrad * R.z)
75       Steer = WR_mc.locatenew('Steer', framelength * Frame.z)
76       Frame_mc = WR_mc.locatenew('Frame_mc', - framecg1 * Frame.x
77                                              + framecg3 * Frame.z)
78       Fork_mc = Steer.locatenew('Fork_mc', - forkcg1 * Fork.x
79                                            + forkcg3 * Fork.z)
80       WF_mc = Steer.locatenew('WF_mc', forklength * Fork.x + forkoffset * Fork.z)
81       WF_cont = WF_mc.locatenew('WF_cont', WFrad * (dot(Fork.y, Y.z) * Fork.y -
82                                                     Y.z).normalize())
83   
84       # Set the angular velocity of each frame.
85       # Angular accelerations end up being calculated automatically by
86       # differentiating the angular velocities when first needed.
87       # u1 is yaw rate
88       # u2 is roll rate
89       # u3 is rear wheel rate
90       # u4 is frame pitch rate
91       # u5 is fork steer rate
92       # u6 is front wheel rate
93       Y.set_ang_vel(N, u1 * Y.z)
94       R.set_ang_vel(Y, u2 * R.x)
95       WR.set_ang_vel(Frame, u3 * Frame.y)
96       Frame.set_ang_vel(R, u4 * Frame.y)
97       Fork.set_ang_vel(Frame, u5 * Fork.x)
98       WF.set_ang_vel(Fork, u6 * Fork.y)
99   
100      # Form the velocities of the previously defined points, using the 2 - point
101      # theorem (written out by hand here).  Accelerations again are calculated
102      # automatically when first needed.
103      WR_cont.set_vel(N, 0)
104      WR_mc.v2pt_theory(WR_cont, N, WR)
105      Steer.v2pt_theory(WR_mc, N, Frame)
106      Frame_mc.v2pt_theory(WR_mc, N, Frame)
107      Fork_mc.v2pt_theory(Steer, N, Fork)
108      WF_mc.v2pt_theory(Steer, N, Fork)
109      WF_cont.v2pt_theory(WF_mc, N, WF)
110  
111      # Sets the inertias of each body. Uses the inertia frame to construct the
112      # inertia dyadics. Wheel inertias are only defined by principle moments of
113      # inertia, and are in fact constant in the frame and fork reference frames;
114      # it is for this reason that the orientations of the wheels does not need
115      # to be defined. The frame and fork inertias are defined in the 'Temp'
116      # frames which are fixed to the appropriate body frames; this is to allow
117      # easier input of the reference values of the benchmark paper. Note that
118      # due to slightly different orientations, the products of inertia need to
119      # have their signs flipped; this is done later when entering the numerical
120      # value.
121  
122      Frame_I = (inertia(TempFrame, Iframe11, Iframe22, Iframe33, 0, 0, Iframe31), Frame_mc)
123      Fork_I = (inertia(TempFork, Ifork11, Ifork22, Ifork33, 0, 0, Ifork31), Fork_mc)
124      WR_I = (inertia(Frame, Iwr11, Iwr22, Iwr11), WR_mc)
125      WF_I = (inertia(Fork, Iwf11, Iwf22, Iwf11), WF_mc)
126  
127      # Declaration of the RigidBody containers. ::
128  
129      BodyFrame = RigidBody('BodyFrame', Frame_mc, Frame, mframe, Frame_I)
130      BodyFork = RigidBody('BodyFork', Fork_mc, Fork, mfork, Fork_I)
131      BodyWR = RigidBody('BodyWR', WR_mc, WR, mwr, WR_I)
132      BodyWF = RigidBody('BodyWF', WF_mc, WF, mwf, WF_I)
133  
134      # The kinematic differential equations; they are defined quite simply. Each
135      # entry in this list is equal to zero.
136      kd = [q1d - u1, q2d - u2, q4d - u4, q5d - u5]
137  
138      # The nonholonomic constraints are the velocity of the front wheel contact
139      # point dotted into the X, Y, and Z directions; the yaw frame is used as it
140      # is "closer" to the front wheel (1 less DCM connecting them). These
141      # constraints force the velocity of the front wheel contact point to be 0
142      # in the inertial frame; the X and Y direction constraints enforce a
143      # "no-slip" condition, and the Z direction constraint forces the front
144      # wheel contact point to not move away from the ground frame, essentially
145      # replicating the holonomic constraint which does not allow the frame pitch
146      # to change in an invalid fashion.
147  
148      conlist_speed = [WF_cont.vel(N) & Y.x, WF_cont.vel(N) & Y.y, WF_cont.vel(N) & Y.z]
149  
150      # The holonomic constraint is that the position from the rear wheel contact
151      # point to the front wheel contact point when dotted into the
152      # normal-to-ground plane direction must be zero; effectively that the front
153      # and rear wheel contact points are always touching the ground plane. This
154      # is actually not part of the dynamic equations, but instead is necessary
155      # for the lineraization process.
156  
157      conlist_coord = [WF_cont.pos_from(WR_cont) & Y.z]
158  
159      # The force list; each body has the appropriate gravitational force applied
160      # at its mass center.
161      FL = [(Frame_mc, -mframe * g * Y.z),
162          (Fork_mc, -mfork * g * Y.z),
163          (WF_mc, -mwf * g * Y.z),
164          (WR_mc, -mwr * g * Y.z)]
165      BL = [BodyFrame, BodyFork, BodyWR, BodyWF]
166  
167  
168      # The N frame is the inertial frame, coordinates are supplied in the order
169      # of independent, dependent coordinates, as are the speeds. The kinematic
170      # differential equation are also entered here.  Here the dependent speeds
171      # are specified, in the same order they were provided in earlier, along
172      # with the non-holonomic constraints.  The dependent coordinate is also
173      # provided, with the holonomic constraint.  Again, this is only provided
174      # for the linearization process.
175  
176      KM = KanesMethod(N, q_ind=[q1, q2, q5],
177              q_dependent=[q4], configuration_constraints=conlist_coord,
178              u_ind=[u2, u3, u5],
179              u_dependent=[u1, u4, u6], velocity_constraints=conlist_speed,
180              kd_eqs=kd,
181              constraint_solver="CRAMER")
182      (fr, frstar) = KM.kanes_equations(BL, FL)
183  
184      # This is the start of entering in the numerical values from the benchmark
185      # paper to validate the eigen values of the linearized equations from this
186      # model to the reference eigen values. Look at the aforementioned paper for
187      # more information. Some of these are intermediate values, used to
188      # transform values from the paper into the coordinate systems used in this
189      # model.
190      PaperRadRear                    =  0.3
191      PaperRadFront                   =  0.35
192      HTA                             =  (pi / 2 - pi / 10).evalf()
193      TrailPaper                      =  0.08
194      rake                            =  (-(TrailPaper*sin(HTA)-(PaperRadFront*cos(HTA)))).evalf()
195      PaperWb                         =  1.02
196      PaperFrameCgX                   =  0.3
197      PaperFrameCgZ                   =  0.9
198      PaperForkCgX                    =  0.9
199      PaperForkCgZ                    =  0.7
200      FrameLength                     =  (PaperWb*sin(HTA)-(rake-(PaperRadFront-PaperRadRear)*cos(HTA))).evalf()
201      FrameCGNorm                     =  ((PaperFrameCgZ - PaperRadRear-(PaperFrameCgX/sin(HTA))*cos(HTA))*sin(HTA)).evalf()
202      FrameCGPar                      =  (PaperFrameCgX / sin(HTA) + (PaperFrameCgZ - PaperRadRear - PaperFrameCgX / sin(HTA) * cos(HTA)) * cos(HTA)).evalf()
203      tempa                           =  (PaperForkCgZ - PaperRadFront)
204      tempb                           =  (PaperWb-PaperForkCgX)
205      tempc                           =  (sqrt(tempa**2+tempb**2)).evalf()
206      PaperForkL                      =  (PaperWb*cos(HTA)-(PaperRadFront-PaperRadRear)*sin(HTA)).evalf()
207      ForkCGNorm                      =  (rake+(tempc * sin(pi/2-HTA-acos(tempa/tempc)))).evalf()
208      ForkCGPar                       =  (tempc * cos((pi/2-HTA)-acos(tempa/tempc))-PaperForkL).evalf()
209  
210      # Here is the final assembly of the numerical values. The symbol 'v' is the
211      # forward speed of the bicycle (a concept which only makes sense in the
212      # upright, static equilibrium case?). These are in a dictionary which will
213      # later be substituted in. Again the sign on the *product* of inertia
214      # values is flipped here, due to different orientations of coordinate
215      # systems.
216      v = symbols('v')
217      val_dict = {WFrad: PaperRadFront,
218                  WRrad: PaperRadRear,
219                  htangle: HTA,
220                  forkoffset: rake,
221                  forklength: PaperForkL,
222                  framelength: FrameLength,
223                  forkcg1: ForkCGPar,
224                  forkcg3: ForkCGNorm,
225                  framecg1: FrameCGNorm,
226                  framecg3: FrameCGPar,
227                  Iwr11: 0.0603,
228                  Iwr22: 0.12,
229                  Iwf11: 0.1405,
230                  Iwf22: 0.28,
231                  Ifork11: 0.05892,
232                  Ifork22: 0.06,
233                  Ifork33: 0.00708,
234                  Ifork31: 0.00756,
235                  Iframe11: 9.2,
236                  Iframe22: 11,
237                  Iframe33: 2.8,
238                  Iframe31: -2.4,
239                  mfork: 4,
240                  mframe: 85,
241                  mwf: 3,
242                  mwr: 2,
243                  g: 9.81,
244                  q1: 0,
245                  q2: 0,
246                  q4: 0,
247                  q5: 0,
248                  u1: 0,
249                  u2: 0,
250                  u3: v / PaperRadRear,
251                  u4: 0,
252                  u5: 0,
253                  u6: v / PaperRadFront}
254  
255      # Linearizes the forcing vector; the equations are set up as MM udot =
256      # forcing, where MM is the mass matrix, udot is the vector representing the
257      # time derivatives of the generalized speeds, and forcing is a vector which
258      # contains both external forcing terms and internal forcing terms, such as
259      # centripital or coriolis forces.  This actually returns a matrix with as
260      # many rows as *total* coordinates and speeds, but only as many columns as
261      # independent coordinates and speeds.
262  
263      A, B, _ = KM.linearize(
264          A_and_B=True,
265          op_point={
266              # Operating points for the accelerations are required for the
267              # linearizer to eliminate u' terms showing up in the coefficient
268              # matrices.
269              u1.diff(): 0,
270              u2.diff(): 0,
271              u3.diff(): 0,
272              u4.diff(): 0,
273              u5.diff(): 0,
274              u6.diff(): 0,
275              u1: 0,
276              u2: 0,
277              u3: v / PaperRadRear,
278              u4: 0,
279              u5: 0,
280              u6: v / PaperRadFront,
281              q1: 0,
282              q2: 0,
283              q4: 0,
284              q5: 0,
285          },
286          linear_solver="CRAMER",
287      )
288      # As mentioned above, the size of the linearized forcing terms is expanded
289      # to include both q's and u's, so the mass matrix must have this done as
290      # well.  This will likely be changed to be part of the linearized process,
291      # for future reference.
292      A_s = A.xreplace(val_dict)
293      B_s = B.xreplace(val_dict)
294  
295      A_s = A_s.evalf()
296      B_s = B_s.evalf()
297  
298      # Finally, we construct an "A" matrix for the form xdot = A x (x being the
299      # state vector, although in this case, the sizes are a little off). The
300      # following line extracts only the minimum entries required for eigenvalue
301      # analysis, which correspond to rows and columns for lean, steer, lean
302      # rate, and steer rate.
303      A = A_s.extract([1, 2, 3, 5], [1, 2, 3, 5])
304  
305      # Precomputed for comparison
306      Res = Matrix([[               0,                                           0,                  1.0,                    0],
307                    [               0,                                           0,                    0,                  1.0],
308                    [9.48977444677355, -0.891197738059089*v**2 - 0.571523173729245, -0.105522449805691*v, -0.330515398992311*v],
309                    [11.7194768719633,   -1.97171508499972*v**2 + 30.9087533932407,   3.67680523332152*v,  -3.08486552743311*v]])
310  
311      # Actual eigenvalue comparison
312      eps = 1.e-12
313      for i in range(6):
314          error = Res.subs(v, i) - A.subs(v, i)
315          assert all(abs(x) < eps for x in error)
