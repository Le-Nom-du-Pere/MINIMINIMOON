47   def _add_metadata(to_node: torch.fx.Node, from_node: torch.fx.Node) -> None:
48       from_meta = from_node.meta
49       for meta_name in _METADATA_TO_PORT:
50           if meta_name in from_meta:
51               to_node.meta[meta_name] = from_meta[meta_name]

58   def _find_choose_qparams_node(node: torch.fx.Node) -> Optional[torch.fx.Node]:
59       # BFS to look for choose qparams
60       from collections import deque
61   
62       queue = deque(list(node.users.keys()))
63       while len(queue):
64           n = queue.popleft()
65           if n.op == "output":
66               continue
67           if n.op == "call_function" and n.target in _CHOOSE_QPARAMS_OPS:
68               return n
69           for k in n.users.keys():
70               queue.append(k)
71       return None

74   def _port_metadata_for_input_quant_nodes(
75       input_node: torch.fx.Node,
76       node: torch.fx.Node,
77       qspec: Optional[QuantizationSpecBase],
78   ):
79       if qspec is None:
80           return
81   
82       is_dynamic_quant = getattr(qspec, "is_dynamic", None)
83       if is_dynamic_quant is not None and is_dynamic_quant is True:
84           choose_qparams_node = _find_choose_qparams_node(input_node)
85           if choose_qparams_node is None:
86               raise ValueError(f"No chose qparams node found for {node}")
87           choose_qparam_users = _filter_sym_size_users(choose_qparams_node)
88           if len(choose_qparam_users) != 2:
89               raise InternalError(f"Expecting exactly two user for {choose_qparams_node}")
90           scale_node = choose_qparam_users.pop()
91           dynamic_q_node = next(iter(scale_node.users.keys()))
92           dynamic_q_node_users = _filter_sym_size_users(dynamic_q_node)
93           if len(dynamic_q_node_users) > 1:
94               raise InternalError(f"Expecting single user for {dynamic_q_node}")
95           dynamic_dq_node = dynamic_q_node_users.pop()
96           _add_metadata(choose_qparams_node, node)
97           _add_metadata(dynamic_q_node, node)
98           _add_metadata(dynamic_dq_node, node)
99       else:
100          q_node, dq_node = _find_q_dq_node_for_user(input_node, node)
101          if q_node is None or dq_node is None:
102              return
103          # add metadata for all the node between q_node and get_attr node
104          # if the q_node can be traced back to get_attr node
105          q_to_get_attr_nodes = [q_node]
106          q_node_input = q_node.args[0]
107          while (
108              isinstance(q_node_input, torch.fx.Node)
109              and q_node_input.op == "call_function"
110              and q_node_input.target
111              in [
112                  torch.ops.aten.flatten.using_ints,
113                  torch.ops.aten.permute.default,
114                  torch.ops.aten.permute_copy.default,
115                  torch.ops.aten.slice_copy.Tensor,
116                  torch.ops.aten.squeeze.dim,
117                  torch.ops.aten.squeeze_copy.dim,
118                  torch.ops.aten.transpose.Dimname,
119                  torch.ops.aten.transpose.int,
120                  torch.ops.aten.transpose_,
121                  torch.ops.aten.view_copy.default,
122                  torch.ops.aten.view.default,
123                  torch.ops.aten._mkldnn_transpose,
124              ]
125          ):
126              q_to_get_attr_nodes.append(q_node_input)
127              q_node_input = q_node_input.args[0]
128          if isinstance(q_node_input, torch.fx.Node) and q_node_input.op == "get_attr":
129              for n in q_to_get_attr_nodes:
130                  _add_metadata(n, q_node_input)
131          _add_metadata(dq_node, node)

134  def _port_metadata_for_output_quant_nodes(
135      node: torch.fx.Node, qspec: Optional[QuantizationSpecBase]
136  ):
137      if qspec is None:
138          return
139  
140      node_users = _filter_sym_size_users(node)
141      if len(node.users) == 0:
142          return
143      if len(node_users) != 1:
144          logger.warning(f"Expecting {node} to have single user")  # noqa: G004
145      q_node = node_users.pop()
146      if q_node.op != "call_function" or q_node.target not in _QUANTIZE_OPS:
147          logger.warning(
148              f"Expecting {node} user to be a quantized op but got {q_node}"  # noqa: G004
149          )  # noqa: G004
150          return
151  
152      _add_metadata(q_node, node)

155  class PortNodeMetaForQDQ(PassBase):
156      """ 
157      Port metadata for nodes added by quantization flow. 
158      For static quant these are: 
159      - quantizer_per_tensor.default, dequantize_per_tensor.default 
160      - quantizer_per_channel.default, dequantize_per_channel.default 
161      For dynamic quant these are: 
162      - choose_qparams.tensor 
163      - quantizer_per_tensor.tensor, dequantize_per_tensor.tensor 
164      - quantizer_per_channel.default, dequantize_per_channel.default 
165   
166      Rules of porting metadata: 
167      - Metadata to be ported: 
168        - nn_module_stack 
169        - stack_trace 
170        - quantization_tag 
171      - Metadata to NOT be ported: 
172        - Everything else 
173      - Rules: 
174        - Statically quantized patterns: 
175          - Dequantize nodes on the inputs to be quantized inherit metadata of the consumer node. 
176          - Quantize nodes on the outputs inherit metadata of the producer node. 
177          - Example 1: 
178            - Original: [Conv -> AvgPool -> Linear] 
179            - Quantized [Q-> DQ -> Conv -> Q -> DQ -> AvgPool -> Q -> DQ -> Linear -> Q -> DQ] 
180            - Inner brackets specify which nodes Q/DQ inherit metdata from 
181            - [Q-> [DQ -> Conv -> Q] -> [DQ -> AvgPool -> Q] -> [DQ -> Linear -> Q] -> DQ] 
182            - Note first Q and last DQ do not inherit metadata from any nodes 
183          - Example 2: 
184            - Original: [Conv -> AvgPool -> Linear] 
185            - AvgPool is not quantized 
186            - Quantized [Q-> DQ -> Conv -> Q -> DQ -> AvgPool -> Q -> DQ -> Linear -> Q -> DQ] 
187            - Inner brackets specify which nodes Q/DQ inherit metdata from 
188            - [Q-> [DQ -> Conv -> Q] -> DQ -> [AvgPool] -> Q -> [DQ -> Linear -> Q] -> DQ] 
189            - Note DQ and Q nodes around AvgPool do not inherit metadata from AvgPool because 
190              AvgPool was not supposed to be quantized. Metadata porting relies on quantization_annotation 
191              on the nodes (in this case AvgPool node) to conclude if the node or patter was 
192              supposed to be quantized. And subsequntly decide if the preceding Q, if any, should 
193              inherit metadata from AvgPool. 
194        - Dynamically quantized patterns: 
195          - Input that are dynamically quantized have choose_qparams, quantize and dequantize nodes 
196          - For example, below linear is dynamically quantized while rest statically: 
197            - Original: [Conv -> AvgPool -> Linear] 
198            - Quantized [Q-> DQ -> Conv -> Q -> DQ -> AvgPool -> Q -> DQ -> choose_params -> Q -> DQ -> Linear] 
199            - Quantized [Q-> [DQ -> Conv -> Q] -> [DQ -> AvgPool -> Q] -> DQ -> [choose_params -> Q -> DQ -> Linear]] 
200            - Note first Q does not inherit metadata from any nodes 
201      NB: 
202      - The best place for porting metadata is during observer conversion to q/dq. This is because it precisely 
203        knows which quantization spec is converted to q/dq and thus from where the metadata should be ported. 
204        However, since FX and PT2E quant workflow are on a common code-base, this hurts readability quite a bit. 
205        Doing it via a separate pass, helps readability of the code. Once we are able to refactor PT2E quant 
206        code, this pass should like to be integrated in the refactored variant of "convert" step. 
207      """
208  
209      def call(self, graph_module: torch.fx.GraphModule) -> PassResult:
210          for node in graph_module.graph.nodes:
211              annotation = node.meta.get("quantization_annotation", None)
212              if _is_valid_annotation(annotation):
213                  input_qspec_map = node.meta["quantization_annotation"].input_qspec_map
214                  output_qspec = node.meta["quantization_annotation"].output_qspec
215                  for input_node, qspec in input_qspec_map.items():
216                      _port_metadata_for_input_quant_nodes(input_node, node, qspec)
217                  _port_metadata_for_output_quant_nodes(node, output_qspec)
218          return PassResult(graph_module, True)
