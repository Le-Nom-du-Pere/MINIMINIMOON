24   class ActuatorBase(ABC):
25       """Abstract base class for all actuator classes to inherit from. 
26    
27       Notes 
28       ===== 
29    
30       Instances of this class cannot be directly instantiated by users. However, 
31       it can be used to created custom actuator types through subclassing. 
32    
33       """
34   
35       def __init__(self):
36           """Initializer for ``ActuatorBase``."""
37           pass
38   
39       @abstractmethod
40       def to_loads(self):
41           """Loads required by the equations of motion method classes. 
42    
43           Explanation 
44           =========== 
45    
46           ``KanesMethod`` requires a list of ``Point``-``Vector`` tuples to be 
47           passed to the ``loads`` parameters of its ``kanes_equations`` method 
48           when constructing the equations of motion. This method acts as a 
49           utility to produce the correctly-structred pairs of points and vectors 
50           required so that these can be easily concatenated with other items in 
51           the list of loads and passed to ``KanesMethod.kanes_equations``. These 
52           loads are also in the correct form to also be passed to the other 
53           equations of motion method classes, e.g. ``LagrangesMethod``. 
54    
55           """
56           pass
57   
58       def __repr__(self):
59           """Default representation of an actuator."""
60           return f'{self.__class__.__name__}()'

63   class ForceActuator(ActuatorBase):
64       """Force-producing actuator. 
65    
66       Explanation 
67       =========== 
68    
69       A ``ForceActuator`` is an actuator that produces a (expansile) force along 
70       its length. 
71    
72       A force actuator uses a pathway instance to determine the direction and 
73       number of forces that it applies to a system. Consider the simplest case 
74       where a ``LinearPathway`` instance is used. This pathway is made up of two 
75       points that can move relative to each other, and results in a pair of equal 
76       and opposite forces acting on the endpoints. If the positive time-varying 
77       Euclidean distance between the two points is defined, then the "extension 
78       velocity" is the time derivative of this distance. The extension velocity 
79       is positive when the two points are moving away from each other and 
80       negative when moving closer to each other. The direction for the force 
81       acting on either point is determined by constructing a unit vector directed 
82       from the other point to this point. This establishes a sign convention such 
83       that a positive force magnitude tends to push the points apart, this is the 
84       meaning of "expansile" in this context. The following diagram shows the 
85       positive force sense and the distance between the points:: 
86    
87          P           Q 
88          o<--- F --->o 
89          |           | 
90          |<--l(t)--->| 
91    
92       Examples 
93       ======== 
94    
95       To construct an actuator, an expression (or symbol) must be supplied to 
96       represent the force it can produce, alongside a pathway specifying its line 
97       of action. Let's also create a global reference frame and spatially fix one 
98       of the points in it while setting the other to be positioned such that it 
99       can freely move in the frame's x direction specified by the coordinate 
100      ``q``. 
101   
102      >>> from sympy import symbols 
103      >>> from sympy.physics.mechanics import (ForceActuator, LinearPathway, 
104      ...     Point, ReferenceFrame) 
105      >>> from sympy.physics.vector import dynamicsymbols 
106      >>> N = ReferenceFrame('N') 
107      >>> q = dynamicsymbols('q') 
108      >>> force = symbols('F') 
109      >>> pA, pB = Point('pA'), Point('pB') 
110      >>> pA.set_vel(N, 0) 
111      >>> pB.set_pos(pA, q*N.x) 
112      >>> pB.pos_from(pA) 
113      q(t)*N.x 
114      >>> linear_pathway = LinearPathway(pA, pB) 
115      >>> actuator = ForceActuator(force, linear_pathway) 
116      >>> actuator 
117      ForceActuator(F, LinearPathway(pA, pB)) 
118   
119      Parameters 
120      ========== 
121   
122      force : Expr 
123          The scalar expression defining the (expansile) force that the actuator 
124          produces. 
125      pathway : PathwayBase 
126          The pathway that the actuator follows. This must be an instance of a 
127          concrete subclass of ``PathwayBase``, e.g. ``LinearPathway``. 
128   
129      """
130  
131      def __init__(self, force, pathway):
132          """Initializer for ``ForceActuator``. 
133   
134          Parameters 
135          ========== 
136   
137          force : Expr 
138              The scalar expression defining the (expansile) force that the 
139              actuator produces. 
140          pathway : PathwayBase 
141              The pathway that the actuator follows. This must be an instance of 
142              a concrete subclass of ``PathwayBase``, e.g. ``LinearPathway``. 
143   
144          """
145          self.force = force
146          self.pathway = pathway
147  
148      @property
149      def force(self):
150          """The magnitude of the force produced by the actuator."""
151          return self._force
152  
153      @force.setter
154      def force(self, force):
155          if hasattr(self, '_force'):
156              msg = (
157                  f'Can\'t set attribute `force` to {repr(force)} as it is '
158                  f'immutable.'
159              )
160              raise AttributeError(msg)
161          self._force = sympify(force, strict=True)
162  
163      @property
164      def pathway(self):
165          """The ``Pathway`` defining the actuator's line of action."""
166          return self._pathway
167  
168      @pathway.setter
169      def pathway(self, pathway):
170          if hasattr(self, '_pathway'):
171              msg = (
172                  f'Can\'t set attribute `pathway` to {repr(pathway)} as it is '
173                  f'immutable.'
174              )
175              raise AttributeError(msg)
176          if not isinstance(pathway, PathwayBase):
177              msg = (
178                  f'Value {repr(pathway)} passed to `pathway` was of type '
179                  f'{type(pathway)}, must be {PathwayBase}.'
180              )
181              raise TypeError(msg)
182          self._pathway = pathway
183  
184      def to_loads(self):
185          """Loads required by the equations of motion method classes. 
186   
187          Explanation 
188          =========== 
189   
190          ``KanesMethod`` requires a list of ``Point``-``Vector`` tuples to be 
191          passed to the ``loads`` parameters of its ``kanes_equations`` method 
192          when constructing the equations of motion. This method acts as a 
193          utility to produce the correctly-structred pairs of points and vectors 
194          required so that these can be easily concatenated with other items in 
195          the list of loads and passed to ``KanesMethod.kanes_equations``. These 
196          loads are also in the correct form to also be passed to the other 
197          equations of motion method classes, e.g. ``LagrangesMethod``. 
198   
199          Examples 
200          ======== 
201   
202          The below example shows how to generate the loads produced by a force 
203          actuator that follows a linear pathway. In this example we'll assume 
204          that the force actuator is being used to model a simple linear spring. 
205          First, create a linear pathway between two points separated by the 
206          coordinate ``q`` in the ``x`` direction of the global frame ``N``. 
207   
208          >>> from sympy.physics.mechanics import (LinearPathway, Point, 
209          ...     ReferenceFrame) 
210          >>> from sympy.physics.vector import dynamicsymbols 
211          >>> q = dynamicsymbols('q') 
212          >>> N = ReferenceFrame('N') 
213          >>> pA, pB = Point('pA'), Point('pB') 
214          >>> pB.set_pos(pA, q*N.x) 
215          >>> pathway = LinearPathway(pA, pB) 
216   
217          Now create a symbol ``k`` to describe the spring's stiffness and 
218          instantiate a force actuator that produces a (contractile) force 
219          proportional to both the spring's stiffness and the pathway's length. 
220          Note that actuator classes use the sign convention that expansile 
221          forces are positive, so for a spring to produce a contractile force the 
222          spring force needs to be calculated as the negative for the stiffness 
223          multiplied by the length. 
224   
225          >>> from sympy import symbols 
226          >>> from sympy.physics.mechanics import ForceActuator 
227          >>> stiffness = symbols('k') 
228          >>> spring_force = -stiffness*pathway.length 
229          >>> spring = ForceActuator(spring_force, pathway) 
230   
231          The forces produced by the spring can be generated in the list of loads 
232          form that ``KanesMethod`` (and other equations of motion methods) 
233          requires by calling the ``to_loads`` method. 
234   
235          >>> spring.to_loads() 
236          [(pA, k*q(t)*N.x), (pB, - k*q(t)*N.x)] 
237   
238          A simple linear damper can be modeled in a similar way. Create another 
239          symbol ``c`` to describe the dampers damping coefficient. This time 
240          instantiate a force actuator that produces a force proportional to both 
241          the damper's damping coefficient and the pathway's extension velocity. 
242          Note that the damping force is negative as it acts in the opposite 
243          direction to which the damper is changing in length. 
244   
245          >>> damping_coefficient = symbols('c') 
246          >>> damping_force = -damping_coefficient*pathway.extension_velocity 
247          >>> damper = ForceActuator(damping_force, pathway) 
248   
249          Again, the forces produces by the damper can be generated by calling 
250          the ``to_loads`` method. 
251   
252          >>> damper.to_loads() 
253          [(pA, c*Derivative(q(t), t)*N.x), (pB, - c*Derivative(q(t), t)*N.x)] 
254   
255          """
256          return self.pathway.to_loads(self.force)
257  
258      def __repr__(self):
259          """Representation of a ``ForceActuator``."""
260          return f'{self.__class__.__name__}({self.force}, {self.pathway})'

263  class LinearSpring(ForceActuator):
264      """A spring with its spring force as a linear function of its length. 
265   
266      Explanation 
267      =========== 
268   
269      Note that the "linear" in the name ``LinearSpring`` refers to the fact that 
270      the spring force is a linear function of the springs length. I.e. for a 
271      linear spring with stiffness ``k``, distance between its ends of ``x``, and 
272      an equilibrium length of ``0``, the spring force will be ``-k*x``, which is 
273      a linear function in ``x``. To create a spring that follows a linear, or 
274      straight, pathway between its two ends, a ``LinearPathway`` instance needs 
275      to be passed to the ``pathway`` parameter. 
276   
277      A ``LinearSpring`` is a subclass of ``ForceActuator`` and so follows the 
278      same sign conventions for length, extension velocity, and the direction of 
279      the forces it applies to its points of attachment on bodies. The sign 
280      convention for the direction of forces is such that, for the case where a 
281      linear spring is instantiated with a ``LinearPathway`` instance as its 
282      pathway, they act to push the two ends of the spring away from one another. 
283      Because springs produces a contractile force and acts to pull the two ends 
284      together towards the equilibrium length when stretched, the scalar portion 
285      of the forces on the endpoint are negative in order to flip the sign of the 
286      forces on the endpoints when converted into vector quantities. The 
287      following diagram shows the positive force sense and the distance between 
288      the points:: 
289   
290         P           Q 
291         o<--- F --->o 
292         |           | 
293         |<--l(t)--->| 
294   
295      Examples 
296      ======== 
297   
298      To construct a linear spring, an expression (or symbol) must be supplied to 
299      represent the stiffness (spring constant) of the spring, alongside a 
300      pathway specifying its line of action. Let's also create a global reference 
301      frame and spatially fix one of the points in it while setting the other to 
302      be positioned such that it can freely move in the frame's x direction 
303      specified by the coordinate ``q``. 
304   
305      >>> from sympy import symbols 
306      >>> from sympy.physics.mechanics import (LinearPathway, LinearSpring, 
307      ...     Point, ReferenceFrame) 
308      >>> from sympy.physics.vector import dynamicsymbols 
309      >>> N = ReferenceFrame('N') 
310      >>> q = dynamicsymbols('q') 
311      >>> stiffness = symbols('k') 
312      >>> pA, pB = Point('pA'), Point('pB') 
313      >>> pA.set_vel(N, 0) 
314      >>> pB.set_pos(pA, q*N.x) 
315      >>> pB.pos_from(pA) 
316      q(t)*N.x 
317      >>> linear_pathway = LinearPathway(pA, pB) 
318      >>> spring = LinearSpring(stiffness, linear_pathway) 
319      >>> spring 
320      LinearSpring(k, LinearPathway(pA, pB)) 
321   
322      This spring will produce a force that is proportional to both its stiffness 
323      and the pathway's length. Note that this force is negative as SymPy's sign 
324      convention for actuators is that negative forces are contractile. 
325   
326      >>> spring.force 
327      -k*sqrt(q(t)**2) 
328   
329      To create a linear spring with a non-zero equilibrium length, an expression 
330      (or symbol) can be passed to the ``equilibrium_length`` parameter on 
331      construction on a ``LinearSpring`` instance. Let's create a symbol ``l`` 
332      to denote a non-zero equilibrium length and create another linear spring. 
333   
334      >>> l = symbols('l') 
335      >>> spring = LinearSpring(stiffness, linear_pathway, equilibrium_length=l) 
336      >>> spring 
337      LinearSpring(k, LinearPathway(pA, pB), equilibrium_length=l) 
338   
339      The spring force of this new spring is again proportional to both its 
340      stiffness and the pathway's length. However, the spring will not produce 
341      any force when ``q(t)`` equals ``l``. Note that the force will become 
342      expansile when ``q(t)`` is less than ``l``, as expected. 
343   
344      >>> spring.force 
345      -k*(-l + sqrt(q(t)**2)) 
346   
347      Parameters 
348      ========== 
349   
350      stiffness : Expr 
351          The spring constant. 
352      pathway : PathwayBase 
353          The pathway that the actuator follows. This must be an instance of a 
354          concrete subclass of ``PathwayBase``, e.g. ``LinearPathway``. 
355      equilibrium_length : Expr, optional 
356          The length at which the spring is in equilibrium, i.e. it produces no 
357          force. The default value is 0, i.e. the spring force is a linear 
358          function of the pathway's length with no constant offset. 
359   
360      See Also 
361      ======== 
362   
363      ForceActuator: force-producing actuator (superclass of ``LinearSpring``). 
364      LinearPathway: straight-line pathway between a pair of points. 
365   
366      """
367  
368      def __init__(self, stiffness, pathway, equilibrium_length=S.Zero):
369          """Initializer for ``LinearSpring``. 
370   
371          Parameters 
372          ========== 
373   
374          stiffness : Expr 
375              The spring constant. 
376          pathway : PathwayBase 
377              The pathway that the actuator follows. This must be an instance of 
378              a concrete subclass of ``PathwayBase``, e.g. ``LinearPathway``. 
379          equilibrium_length : Expr, optional 
380              The length at which the spring is in equilibrium, i.e. it produces 
381              no force. The default value is 0, i.e. the spring force is a linear 
382              function of the pathway's length with no constant offset. 
383   
384          """
385          self.stiffness = stiffness
386          self.pathway = pathway
387          self.equilibrium_length = equilibrium_length
388  
389      @property
390      def force(self):
391          """The spring force produced by the linear spring."""
392          return -self.stiffness*(self.pathway.length - self.equilibrium_length)
393  
394      @force.setter
395      def force(self, force):
396          raise AttributeError('Can\'t set computed attribute `force`.')
397  
398      @property
399      def stiffness(self):
400          """The spring constant for the linear spring."""
401          return self._stiffness
402  
403      @stiffness.setter
404      def stiffness(self, stiffness):
405          if hasattr(self, '_stiffness'):
406              msg = (
407                  f'Can\'t set attribute `stiffness` to {repr(stiffness)} as it '
408                  f'is immutable.'
409              )
410              raise AttributeError(msg)
411          self._stiffness = sympify(stiffness, strict=True)
412  
413      @property
414      def equilibrium_length(self):
415          """The length of the spring at which it produces no force."""
416          return self._equilibrium_length
417  
418      @equilibrium_length.setter
419      def equilibrium_length(self, equilibrium_length):
420          if hasattr(self, '_equilibrium_length'):
421              msg = (
422                  f'Can\'t set attribute `equilibrium_length` to '
423                  f'{repr(equilibrium_length)} as it is immutable.'
424              )
425              raise AttributeError(msg)
426          self._equilibrium_length = sympify(equilibrium_length, strict=True)
427  
428      def __repr__(self):
429          """Representation of a ``LinearSpring``."""
430          string = f'{self.__class__.__name__}({self.stiffness}, {self.pathway}'
431          if self.equilibrium_length == S.Zero:
432              string += ')'
433          else:
434              string += f', equilibrium_length={self.equilibrium_length})'
435          return string

438  class LinearDamper(ForceActuator):
439      """A damper whose force is a linear function of its extension velocity. 
440   
441      Explanation 
442      =========== 
443   
444      Note that the "linear" in the name ``LinearDamper`` refers to the fact that 
445      the damping force is a linear function of the damper's rate of change in 
446      its length. I.e. for a linear damper with damping ``c`` and extension 
447      velocity ``v``, the damping force will be ``-c*v``, which is a linear 
448      function in ``v``. To create a damper that follows a linear, or straight, 
449      pathway between its two ends, a ``LinearPathway`` instance needs to be 
450      passed to the ``pathway`` parameter. 
451   
452      A ``LinearDamper`` is a subclass of ``ForceActuator`` and so follows the 
453      same sign conventions for length, extension velocity, and the direction of 
454      the forces it applies to its points of attachment on bodies. The sign 
455      convention for the direction of forces is such that, for the case where a 
456      linear damper is instantiated with a ``LinearPathway`` instance as its 
457      pathway, they act to push the two ends of the damper away from one another. 
458      Because dampers produce a force that opposes the direction of change in 
459      length, when extension velocity is positive the scalar portions of the 
460      forces applied at the two endpoints are negative in order to flip the sign 
461      of the forces on the endpoints wen converted into vector quantities. When 
462      extension velocity is negative (i.e. when the damper is shortening), the 
463      scalar portions of the fofces applied are also negative so that the signs 
464      cancel producing forces on the endpoints that are in the same direction as 
465      the positive sign convention for the forces at the endpoints of the pathway 
466      (i.e. they act to push the endpoints away from one another). The following 
467      diagram shows the positive force sense and the distance between the 
468      points:: 
469   
470         P           Q 
471         o<--- F --->o 
472         |           | 
473         |<--l(t)--->| 
474   
475      Examples 
476      ======== 
477   
478      To construct a linear damper, an expression (or symbol) must be supplied to 
479      represent the damping coefficient of the damper (we'll use the symbol 
480      ``c``), alongside a pathway specifying its line of action. Let's also 
481      create a global reference frame and spatially fix one of the points in it 
482      while setting the other to be positioned such that it can freely move in 
483      the frame's x direction specified by the coordinate ``q``. The velocity 
484      that the two points move away from one another can be specified by the 
485      coordinate ``u`` where ``u`` is the first time derivative of ``q`` 
486      (i.e., ``u = Derivative(q(t), t)``). 
487   
488      >>> from sympy import symbols 
489      >>> from sympy.physics.mechanics import (LinearDamper, LinearPathway, 
490      ...     Point, ReferenceFrame) 
491      >>> from sympy.physics.vector import dynamicsymbols 
492      >>> N = ReferenceFrame('N') 
493      >>> q = dynamicsymbols('q') 
494      >>> damping = symbols('c') 
495      >>> pA, pB = Point('pA'), Point('pB') 
496      >>> pA.set_vel(N, 0) 
497      >>> pB.set_pos(pA, q*N.x) 
498      >>> pB.pos_from(pA) 
499      q(t)*N.x 
500      >>> pB.vel(N) 
501      Derivative(q(t), t)*N.x 
502      >>> linear_pathway = LinearPathway(pA, pB) 
503      >>> damper = LinearDamper(damping, linear_pathway) 
504      >>> damper 
505      LinearDamper(c, LinearPathway(pA, pB)) 
506   
507      This damper will produce a force that is proportional to both its damping 
508      coefficient and the pathway's extension length. Note that this force is 
509      negative as SymPy's sign convention for actuators is that negative forces 
510      are contractile and the damping force of the damper will oppose the 
511      direction of length change. 
512   
513      >>> damper.force 
514      -c*sqrt(q(t)**2)*Derivative(q(t), t)/q(t) 
515   
516      Parameters 
517      ========== 
518   
519      damping : Expr 
520          The damping constant. 
521      pathway : PathwayBase 
522          The pathway that the actuator follows. This must be an instance of a 
523          concrete subclass of ``PathwayBase``, e.g. ``LinearPathway``. 
524   
525      See Also 
526      ======== 
527   
528      ForceActuator: force-producing actuator (superclass of ``LinearDamper``). 
529      LinearPathway: straight-line pathway between a pair of points. 
530   
531      """
532  
533      def __init__(self, damping, pathway):
534          """Initializer for ``LinearDamper``. 
535   
536          Parameters 
537          ========== 
538   
539          damping : Expr 
540              The damping constant. 
541          pathway : PathwayBase 
542              The pathway that the actuator follows. This must be an instance of 
543              a concrete subclass of ``PathwayBase``, e.g. ``LinearPathway``. 
544   
545          """
546          self.damping = damping
547          self.pathway = pathway
548  
549      @property
550      def force(self):
551          """The damping force produced by the linear damper."""
552          return -self.damping*self.pathway.extension_velocity
553  
554      @force.setter
555      def force(self, force):
556          raise AttributeError('Can\'t set computed attribute `force`.')
557  
558      @property
559      def damping(self):
560          """The damping constant for the linear damper."""
561          return self._damping
562  
563      @damping.setter
564      def damping(self, damping):
565          if hasattr(self, '_damping'):
566              msg = (
567                  f'Can\'t set attribute `damping` to {repr(damping)} as it is '
568                  f'immutable.'
569              )
570              raise AttributeError(msg)
571          self._damping = sympify(damping, strict=True)
572  
573      def __repr__(self):
574          """Representation of a ``LinearDamper``."""
575          return f'{self.__class__.__name__}({self.damping}, {self.pathway})'

578  class TorqueActuator(ActuatorBase):
579      """Torque-producing actuator. 
580   
581      Explanation 
582      =========== 
583   
584      A ``TorqueActuator`` is an actuator that produces a pair of equal and 
585      opposite torques on a pair of bodies. 
586   
587      Examples 
588      ======== 
589   
590      To construct a torque actuator, an expression (or symbol) must be supplied 
591      to represent the torque it can produce, alongside a vector specifying the 
592      axis about which the torque will act, and a pair of frames on which the 
593      torque will act. 
594   
595      >>> from sympy import symbols 
596      >>> from sympy.physics.mechanics import (ReferenceFrame, RigidBody, 
597      ...     TorqueActuator) 
598      >>> N = ReferenceFrame('N') 
599      >>> A = ReferenceFrame('A') 
600      >>> torque = symbols('T') 
601      >>> axis = N.z 
602      >>> parent = RigidBody('parent', frame=N) 
603      >>> child = RigidBody('child', frame=A) 
604      >>> bodies = (child, parent) 
605      >>> actuator = TorqueActuator(torque, axis, *bodies) 
606      >>> actuator 
607      TorqueActuator(T, axis=N.z, target_frame=A, reaction_frame=N) 
608   
609      Note that because torques actually act on frames, not bodies, 
610      ``TorqueActuator`` will extract the frame associated with a ``RigidBody`` 
611      when one is passed instead of a ``ReferenceFrame``. 
612   
613      Parameters 
614      ========== 
615   
616      torque : Expr 
617          The scalar expression defining the torque that the actuator produces. 
618      axis : Vector 
619          The axis about which the actuator applies torques. 
620      target_frame : ReferenceFrame | RigidBody 
621          The primary frame on which the actuator will apply the torque. 
622      reaction_frame : ReferenceFrame | RigidBody | None 
623          The secondary frame on which the actuator will apply the torque. Note 
624          that the (equal and opposite) reaction torque is applied to this frame. 
625   
626      """
627  
628      def __init__(self, torque, axis, target_frame, reaction_frame=None):
629          """Initializer for ``TorqueActuator``. 
630   
631          Parameters 
632          ========== 
633   
634          torque : Expr 
635              The scalar expression defining the torque that the actuator 
636              produces. 
637          axis : Vector 
638              The axis about which the actuator applies torques. 
639          target_frame : ReferenceFrame | RigidBody 
640              The primary frame on which the actuator will apply the torque. 
641          reaction_frame : ReferenceFrame | RigidBody | None 
642             The secondary frame on which the actuator will apply the torque. 
643             Note that the (equal and opposite) reaction torque is applied to 
644             this frame. 
645   
646          """
647          self.torque = torque
648          self.axis = axis
649          self.target_frame = target_frame
650          self.reaction_frame = reaction_frame
651  
652      @classmethod
653      def at_pin_joint(cls, torque, pin_joint):
654          """Alternate constructor to instantiate from a ``PinJoint`` instance. 
655   
656          Examples 
657          ======== 
658   
659          To create a pin joint the ``PinJoint`` class requires a name, parent 
660          body, and child body to be passed to its constructor. It is also 
661          possible to control the joint axis using the ``joint_axis`` keyword 
662          argument. In this example let's use the parent body's reference frame's 
663          z-axis as the joint axis. 
664   
665          >>> from sympy.physics.mechanics import (PinJoint, ReferenceFrame, 
666          ...     RigidBody, TorqueActuator) 
667          >>> N = ReferenceFrame('N') 
668          >>> A = ReferenceFrame('A') 
669          >>> parent = RigidBody('parent', frame=N) 
670          >>> child = RigidBody('child', frame=A) 
671          >>> pin_joint = PinJoint( 
672          ...     'pin', 
673          ...     parent, 
674          ...     child, 
675          ...     joint_axis=N.z, 
676          ... ) 
677   
678          Let's also create a symbol ``T`` that will represent the torque applied 
679          by the torque actuator. 
680   
681          >>> from sympy import symbols 
682          >>> torque = symbols('T') 
683   
684          To create the torque actuator from the ``torque`` and ``pin_joint`` 
685          variables previously instantiated, these can be passed to the alternate 
686          constructor class method ``at_pin_joint`` of the ``TorqueActuator`` 
687          class. It should be noted that a positive torque will cause a positive 
688          displacement of the joint coordinate or that the torque is applied on 
689          the child body with a reaction torque on the parent. 
690   
691          >>> actuator = TorqueActuator.at_pin_joint(torque, pin_joint) 
692          >>> actuator 
693          TorqueActuator(T, axis=N.z, target_frame=A, reaction_frame=N) 
694   
695          Parameters 
696          ========== 
697   
698          torque : Expr 
699              The scalar expression defining the torque that the actuator 
700              produces. 
701          pin_joint : PinJoint 
702              The pin joint, and by association the parent and child bodies, on 
703              which the torque actuator will act. The pair of bodies acted upon 
704              by the torque actuator are the parent and child bodies of the pin 
705              joint, with the child acting as the reaction body. The pin joint's 
706              axis is used as the axis about which the torque actuator will apply 
707              its torque. 
708   
709          """
710          if not isinstance(pin_joint, PinJoint):
711              msg = (
712                  f'Value {repr(pin_joint)} passed to `pin_joint` was of type '
713                  f'{type(pin_joint)}, must be {PinJoint}.'
714              )
715              raise TypeError(msg)
716          return cls(
717              torque,
718              pin_joint.joint_axis,
719              pin_joint.child_interframe,
720              pin_joint.parent_interframe,
721          )

726          return self._torque

803                  f'of type {type(reaction_frame)}, must be {ReferenceFrame}.'

808      def to_loads(self):
809          """Loads required by the equations of motion method classes. 
810   
811          Explanation 
812          =========== 
813   
814          ``KanesMethod`` requires a list of ``Point``-``Vector`` tuples to be 
815          passed to the ``loads`` parameters of its ``kanes_equations`` method 
816          when constructing the equations of motion. This method acts as a 
817          utility to produce the correctly-structred pairs of points and vectors 
818          required so that these can be easily concatenated with other items in 
819          the list of loads and passed to ``KanesMethod.kanes_equations``. These 
820          loads are also in the correct form to also be passed to the other 
821          equations of motion method classes, e.g. ``LagrangesMethod``. 
822   
823          Examples 
824          ======== 
825   
826          The below example shows how to generate the loads produced by a torque 
827          actuator that acts on a pair of bodies attached by a pin joint. 
828   
829          >>> from sympy import symbols 
830          >>> from sympy.physics.mechanics import (PinJoint, ReferenceFrame, 
831          ...     RigidBody, TorqueActuator) 
832          >>> torque = symbols('T') 
833          >>> N = ReferenceFrame('N') 
834          >>> A = ReferenceFrame('A') 
835          >>> parent = RigidBody('parent', frame=N) 
836          >>> child = RigidBody('child', frame=A) 
837          >>> pin_joint = PinJoint( 
838          ...     'pin', 
839          ...     parent, 
840          ...     child, 
841          ...     joint_axis=N.z, 
842          ... ) 
843          >>> actuator = TorqueActuator.at_pin_joint(torque, pin_joint) 
844   
845          The forces produces by the damper can be generated by calling the 
846          ``to_loads`` method. 
847   
848          >>> actuator.to_loads() 
849          [(A, T*N.z), (N, - T*N.z)] 
850   
851          Alternatively, if a torque actuator is created without a reaction frame 
852          then the loads returned by the ``to_loads`` method will contain just 
853          the single load acting on the target frame. 
854   
855          >>> actuator = TorqueActuator(torque, N.z, N) 
856          >>> actuator.to_loads() 
857          [(N, T*N.z)] 
858   
859          """
860          loads = [
861              Torque(self.target_frame, self.torque*self.axis),
862          ]
863          if self.reaction_frame is not None:
864              loads.append(Torque(self.reaction_frame, -self.torque*self.axis))
865          return loads
866  
867      def __repr__(self):
868          """Representation of a ``TorqueActuator``."""
869          string = (
870              f'{self.__class__.__name__}({self.torque}, axis={self.axis}, '
871              f'target_frame={self.target_frame}'
872          )
873          if self.reaction_frame is not None:
874              string += f', reaction_frame={self.reaction_frame})'
875          else:
876              string += ')'
877          return string

995  class CoulombKineticFriction(ForceActuator):
996      r"""Coulomb kinetic friction with Stribeck and viscous effects. 
997   
998      Explanation 
999      =========== 
1000  
1001     This represents a Coulomb kinetic friction with the Stribeck and viscous effect, 
1002     described by the function: 
1003  
1004     .. math:: 
1005         F = (\mu_k f_n + (\mu_s - \mu_k) f_n e^{-(\frac{v}{v_s})^2}) \text{sign}(v) + \sigma  v 
1006  
1007     where :math:`\mu_k` is the coefficient of kinetic friction, :math:`\mu_s` is the 
1008     coefficient of static friction, :math:`f_n` is the normal force, :math:`v` is the 
1009     relative velocity, :math:`v_s` is the Stribeck friction coefficient, and 
1010     :math:`\sigma` is the viscous friction constant. 
1011  
1012     The default friction force is :math:`F = \mu_k f_n`. 
1013     When specified, the actuator includes: 
1014  
1015     - Stribeck effect: :math:`(\mu_s - \mu_k) f_n e^{-(\frac{v}{v_s})^2}` 
1016     - Viscous effect: :math:`\sigma v` 
1017  
1018     Notes 
1019     ===== 
1020  
1021     The actuator makes the following assumptions: 
1022  
1023     - The actuator assumes relative motion is non-zero. 
1024     - The normal force is assumed to be a non-negative scalar. 
1025     - The resultant friction force is opposite to the velocity direction. 
1026     - Each point in the pathway is fixed within separate objects that are sliding relative to each other. In other words, these two points are fixed in the mutually sliding objects. 
1027  
1028     This actuator has been tested for straightforward motions, like a block sliding 
1029     on a surface. 
1030  
1031     The friction force is defined to always oppose the direction of relative velocity :math:`v`. 
1032     Specifically: 
1033  
1034     - The default Coulomb friction force :math:`\mu_k f_n \text{sign}(v)` is opposite to :math:`v`. 
1035     - The Stribeck effect :math:`(\mu_s - \mu_k) f_n e^{-(\frac{v}{v_s})^2} \text{sign}(v)` is also opposite to :math:`v`. 
1036     - The viscous friction term :math:`\sigma v` is opposite to :math:`v`. 
1037  
1038     Examples 
1039     ======== 
1040  
1041     The below example shows how to generate the loads produced by a Coulomb kinetic 
1042     friction actuator in a mass-spring system with friction. 
1043  
1044     >>> import sympy as sm 
1045     >>> from sympy.physics.mechanics import (dynamicsymbols, ReferenceFrame, Point, 
1046     ...     LinearPathway, CoulombKineticFriction, LinearSpring, KanesMethod, Particle) 
1047  
1048     >>> x, v = dynamicsymbols('x, v', real=True) 
1049     >>> m, g, k, mu_k, mu_s, v_s, sigma = sm.symbols('m, g, k, mu_k, mu_s, v_s, sigma') 
1050  
1051     >>> N = ReferenceFrame('N') 
1052     >>> O, P = Point('O'), Point('P') 
1053     >>> O.set_vel(N, 0) 
1054     >>> P.set_pos(O, x*N.x) 
1055  
1056     >>> pathway = LinearPathway(O, P) 
1057     >>> friction = CoulombKineticFriction(mu_k, m*g, pathway, v_s=v_s, sigma=sigma, mu_s=mu_k) 
1058     >>> spring = LinearSpring(k, pathway) 
1059     >>> block = Particle('block', point=P, mass=m) 
1060  
1061     >>> kane = KanesMethod(N, (x,), (v,), kd_eqs=(x.diff() - v,)) 
1062     >>> friction.to_loads() 
1063         [(O, (g*m*mu_k*sign(sign(x(t))*Derivative(x(t), t)) + sigma*sign(x(t))*Derivative(x(t), t))*x(t)/Abs(x(t))*N.x), (P, (-g*m*mu_k*sign(sign(x(t))*Derivative(x(t), t)) - sigma*sign(x(t))*Derivative(x(t), t))*x(t)/Abs(x(t))*N.x)] 
1064     >>> loads = friction.to_loads() + spring.to_loads() 
1065     >>> fr, frstar = kane.kanes_equations([block], loads) 
1066     >>> eom = fr + frstar 
1067     >>> eom 
1068         Matrix([[-k*x(t) - m*Derivative(v(t), t) + (-g*m*mu_k*sign(v(t)*sign(x(t))) - sigma*v(t)*sign(x(t)))*x(t)/Abs(x(t))]]) 
1069  
1070     Parameters 
1071     ========== 
1072  
1073     f_n : sympifiable 
1074         The normal force between the surfaces. It should always be a non-negative scalar. 
1075     mu_k : sympifiable 
1076         The coefficient of kinetic friction. 
1077     pathway : PathwayBase 
1078         The pathway that the actuator follows. 
1079     v_s : sympifiable, optional 
1080         The Stribeck friction coefficient. 
1081     sigma : sympifiable, optional 
1082         The viscous friction coefficient. 
1083     mu_s : sympifiable, optional 
1084         The coefficient of static friction. Defaults to mu_k, meaning the Stribeck effect evaluates to 0 by default. 
1085  
1086     References 
1087     ========== 
1088  
1089     .. [Moore2022] https://moorepants.github.io/learn-multibody-dynamics/loads.html#friction. 
1090     .. [Flores2023] Paulo Flores, Jorge Ambrosio, Hamid M. Lankarani, 
1091             "Contact-impact events with friction in multibody dynamics: Back to basics", 
1092             Mechanism and Machine Theory, vol. 184, 2023. https://doi.org/10.1016/j.mechmachtheory.2023.105305. 
1093     .. [Rogner2017] I. Rogner, "Friction modelling for robotic applications with planar motion", 
1094             Chalmers University of Technology, Department of Electrical Engineering, 2017. 
1095  
1096     """
1097 
1098     def __init__(self, mu_k, f_n, pathway, *, v_s=None, sigma=None, mu_s=None):
1099         self._mu_k = sympify(mu_k, strict=True) if mu_k is not None else 1
1100         self._mu_s = sympify(mu_s, strict=True) if mu_s is not None else self._mu_k
1101         self._f_n = sympify(f_n, strict=True)
1102         self._sigma = sympify(sigma, strict=True) if sigma is not None else 0
1103         self._v_s = sympify(v_s, strict=True) if v_s is not None or v_s == 0 else 0.01
1104         self.pathway = pathway
1105 
1106     @property
1107     def mu_k(self):
1108         """The coefficient of kinetic friction."""
1109         return self._mu_k
1110 
1111     @property
1112     def mu_s(self):
1113         """The coefficient of static friction."""
1114         return self._mu_s
1115 
1116     @property
1117     def f_n(self):
1118         """The normal force between the surfaces."""
1119         return self._f_n
1120 
1121     @property
1122     def sigma(self):
1123         """The viscous friction coefficient."""
1124         return self._sigma
1125 
1126     @property
1127     def v_s(self):
1128         """The Stribeck friction coefficient."""
1129         return self._v_s
1130 
1131     @property
1132     def force(self):
1133         v = self.pathway.extension_velocity
1134         f_c = self.mu_k * self.f_n
1135         f_max = self.mu_s * self.f_n
1136         stribeck_term = (f_max - f_c) * exp(-(v / self.v_s)**2) if self.v_s is not None else 0
1137         viscous_term = self.sigma * v if self.sigma is not None else 0
1138         return (f_c + stribeck_term) * -sign(v) - viscous_term
1139 
1140     @force.setter
1141     def force(self, force):
1142         raise AttributeError('Can\'t set computed attribute `force`.')
1143 
1144     def __repr__(self):
1145         return (f'{self.__class__.__name__}({self._mu_k}, {self._mu_s} '
1146                 f'{self._f_n}, {self.pathway}, {self._v_s}, '
1147                 f'{self._sigma})')
