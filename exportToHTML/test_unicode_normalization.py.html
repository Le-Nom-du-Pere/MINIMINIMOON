111      def test_normalization_prevents_overcounting(self):
112          """Test that normalization prevents overcounting identical characters."""
113          # Create text with same character in different Unicode forms  
114          text1 = 'é'  # Composed form
115          text2 = 'e\u0301'  # Decomposed form (base + combining accent)
116          
117          # Before normalization, they might be different
118          self.assertNotEqual(text1, text2)
119          
120          # After normalization, they should be the same
121          norm1 = normalize_unicode(text1)
122          norm2 = normalize_unicode(text2)
123          self.assertEqual(norm1, norm2)
124          
125          # Word counts should be identical
126          count1 = count_words(text1 + ' word')
127          count2 = count_words(text2 + ' word')
128          self.assertEqual(count1, count2)
