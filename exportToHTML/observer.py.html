77   class _PartialWrapper:
78       def __init__(self, p):
79           self.p = p
80           self.callable_args = {}
81   
82       def __call__(self, *args, **keywords):
83           # call each arg in callable_args and add them partial, then run with keywords
84           # skip if arg_name in keywords so its possible to overwrite
85           for arg_name in self.callable_args:
86               if arg_name not in keywords:
87                   keywords = {**keywords, arg_name: self.callable_args[arg_name]()}
88           return self.p(*args, **keywords)
89   
90       def __repr__(self):
91           return self.p.__repr__() + self.callable_args.__repr__()
92   
93       def with_args(self, **kwargs):
94           return _with_args(self, **kwargs)
95   
96       def with_callable_args(self, **kwargs):
97           result = _PartialWrapper(p=self.p)
98           result.callable_args = {**self.callable_args, **kwargs}
99           return result

102  def _with_args(cls_or_self, **kwargs):
103      r"""Wrapper that allows creation of class factories. 
104   
105      This can be useful when there is a need to create classes with the same 
106      constructor arguments, but different instances. Can be used in conjunction with 
107      _callable_args 
108   
109      Example:: 
110   
111          >>> # xdoctest: +SKIP("Undefined vars") 
112          >>> Foo.with_args = classmethod(_with_args) 
113          >>> foo_builder = Foo.with_args(a=3, b=4).with_args(answer=42) 
114          >>> foo_instance1 = foo_builder() 
115          >>> foo_instance2 = foo_builder() 
116          >>> id(foo_instance1) == id(foo_instance2) 
117          False 
118      """
119      r = _PartialWrapper(partial(cls_or_self, **kwargs))
120      return r

123  def _with_callable_args(cls_or_self, **kwargs):
124      r"""Wrapper that allows creation of class factories args that need to be 
125      called at construction time. 
126   
127      This can be useful when there is a need to create classes with the same 
128      constructor arguments, but different instances and those arguments should only 
129      be calculated at construction time. Can be used in conjunction with _with_args 
130   
131      Example:: 
132   
133          >>> # xdoctest: +SKIP("Undefined vars") 
134          >>> Foo.with_callable_args = classmethod(_with_callable_args) 
135          >>> Foo.with_args = classmethod(_with_args) 
136          >>> foo_builder = Foo.with_callable_args(cur_time=get_time_func).with_args(name="dan") 
137          >>> foo_instance1 = foo_builder() 
138          >>> # wait 50 
139          >>> foo_instance2 = foo_builder() 
140          >>> id(foo_instance1.creation_time) == id(foo_instance2.creation_time) 
141          False 
142      """
143      r = _PartialWrapper(partial(cls_or_self))
144      return r.with_callable_args(**kwargs)

184      r"""Common base for all observers using uniform quantization to calculate 
185      scale and zero_point. 
186   
187      Args: 
188          dtype: dtype argument to the `quantize` node needed to implement the 
189                 reference model spec. 
190          qscheme: Quantization scheme to be used. 
191          reduce_range: Reduces the range of the quantized data type by 1 bit. 
192                        This is sometimes required to avoid instruction overflow. 
193          quant_min: Minimum quantization value. If unspecified, it will follow the 8-bit setup. 
194          quant_max: Maximum quantization value. If unspecified, it will follow the 8-bit setup. 
195          eps: Epsilon value for float32, Defaults to `torch.finfo(torch.float32).eps`. 
196   
197      .. warning:: 
198   
199          :attr:`dtype` can only take ``torch.qint8`` or ``torch.quint8``. 
200                 or `torch.int8` or `torch.uint8` 
201   
202      .. warning:: 
203   
204          :attr:`qscheme` can only take one of the following options: 
205   
206          - ``torch.per_tensor_affine`` 
207          - ``torch.per_tensor_symmetric`` 
208          - ``torch.per_channel_affine`` 
209          - ``torch.per_channel_symmetric``

319      @torch.jit.export
320      def _validate_qmin_qmax(self, quant_min: int, quant_max: int) -> None:
321          r"""Validates that the user-specified quantization range is properly initialized 
322          and within the given bound supported by the observer dtype. 
323   
324          To accommodate lower-bit quantization with respect to the existing torch.qint8 and 
325          torch.quint8 datatypes, the user can choose to use dynamic quantization range by passing 
326          in a tuple of initial qmin and qmax values. One use case is these customized qmin and qmax 
327          values are used to calculate static estimates of the scale and zero point for aggressive lower-bit 
328          fake quantization. These estimates are compared against parameters learned through backpropagation. 
329          The related literatures for scale and zero point via backpropagation are as follows: 
330   
331          Learned Step Size Quantization: https://openreview.net/pdf?id=rkgO66VKDS 
332          Trained Quantization Thresholds: https://arxiv.org/pdf/1903.08066.pdf 
333          """
334          # The variable names are prefixed with "initial" because their values (qmin and qmax) might be adjusted
335          # based on whether quantization range is reduced and the datatype (signed/unsigned) used by the observer.
336          assert quant_min <= 0 <= quant_max, (
337              "Used-specified quantization range must include 0."
338          )
339          assert quant_min < quant_max, (
340              "qmin must be strictly less than qmax for user-specified quantization range."
341          )

343      @torch.jit.export
344      def _calculate_qparams(
345          self, min_val: torch.Tensor, max_val: torch.Tensor
346      ) -> tuple[torch.Tensor, torch.Tensor]:
347          r"""Calculates the quantization parameters, given min and max 
348          value tensors. Works for both per tensor and per channel cases 
349   
350          Args: 
351              min_val: Minimum values per channel 
352              max_val: Maximum values per channel 
353   
354          Returns: 
355              scales: Scales tensor of shape (#channels,) 
356              zero_points: Zero points tensor of shape (#channels,) 
357          """
358          # Functionally equivalent to 'determine_qparams' in utils.py. Observers must be torchscriptable however and qscheme
359          # as far as I can tell is not allowed to passed as a parameter in torchscript functions. This makes refactoring observer
360          # to use this utility a massive pain and very gross. For now Im opting just to duplicate as this code
361          # seems unlikey to change (last update over 1 year ago) and when torchscript is fully deprecated we can refactor.
362          # TODO(jakeszwe, jerryzh168)
363          if not check_min_max_valid(min_val, max_val):
364              return torch.tensor([1.0], device=min_val.device.type), torch.tensor(
365                  [0], device=min_val.device.type
366              )
367  
368          quant_min, quant_max = self.quant_min, self.quant_max
369          min_val_neg = torch.min(min_val, torch.zeros_like(min_val))
370          max_val_pos = torch.max(max_val, torch.zeros_like(max_val))
371  
372          device = min_val_neg.device
373          scale = torch.ones(min_val_neg.size(), dtype=torch.float32, device=device)
374          zero_point = torch.zeros(min_val_neg.size(), dtype=torch.int64, device=device)
375  
376          if (
377              self.qscheme == torch.per_tensor_symmetric
378              or self.qscheme == torch.per_channel_symmetric
379          ):
380              max_val_pos = torch.max(-min_val_neg, max_val_pos)
381              scale = max_val_pos / (float(quant_max - quant_min) / 2)
382              scale = torch.max(scale, self.eps)
383              if self.dtype in [torch.quint8, torch.uint8]:
384                  if self.has_customized_qrange:
385                      # When customized quantization range is used, down-rounded midpoint of the range is chosen.
386                      zero_point = zero_point.new_full(
387                          zero_point.size(), (quant_min + quant_max) // 2
388                      )
389                  else:
390                      zero_point = zero_point.new_full(zero_point.size(), 128)
391              elif self.dtype in [torch.uint16]:
392                  zero_point = zero_point.new_full(zero_point.size(), 2**15)
393          elif self.qscheme == torch.per_channel_affine_float_qparams:
394              scale = (max_val - min_val) / float(quant_max - quant_min)
395              scale = torch.where(scale > self.eps, scale, torch.ones_like(scale))
396              # We use the quantize function
397              # xq = Round(Xf * inv_scale + zero_point),
398              # setting zero_point to (-1 * min *inv_scale) we get
399              # Xq = Round((Xf - min) * inv_scale)
400              zero_point = -1 * min_val / scale
401          else:
402              scale = (max_val_pos - min_val_neg) / float(quant_max - quant_min)
403              scale = torch.max(scale, self.eps)
404              zero_point = quant_min - torch.round(min_val_neg / scale).to(torch.int)
405              zero_point = torch.clamp(zero_point, quant_min, quant_max)
406  
407          # For scalar values, cast them to Tensors of size 1 to keep the shape
408          # consistent with default values in FakeQuantize.
409          if len(scale.shape) == 0:
410              # TODO: switch to scale.item() after adding JIT support
411              scale = torch.tensor([float(scale)], dtype=scale.dtype, device=device)
412          if len(zero_point.shape) == 0:
413              # TODO: switch to zero_point.item() after adding JIT support
414              zero_point = torch.tensor(
415                  [int(zero_point)], dtype=zero_point.dtype, device=device
416              )
417              if self.qscheme == torch.per_channel_affine_float_qparams:
418                  zero_point = torch.tensor(
419                      [float(zero_point)], dtype=zero_point.dtype, device=device
420                  )
421  
422          return scale, zero_point

435  class MinMaxObserver(UniformQuantizationObserverBase):
436      r"""Observer module for computing the quantization parameters based on the 
437      running min and max values. 
438   
439      This observer uses the tensor min/max statistics to compute the quantization 
440      parameters. The module records the running minimum and maximum of incoming 
441      tensors, and uses this statistic to compute the quantization parameters. 
442   
443      Args: 
444          dtype: dtype argument to the `quantize` node needed to implement the 
445                 reference model spec. 
446          qscheme: Quantization scheme to be used 
447          reduce_range: Reduces the range of the quantized data type by 1 bit 
448          quant_min: Minimum quantization value. If unspecified, it will follow the 8-bit setup. 
449          quant_max: Maximum quantization value. If unspecified, it will follow the 8-bit setup. 
450          eps: Epsilon value for float32, Defaults to `torch.finfo(torch.float32).eps`. 
451   
452      Given running min/max as :math:`x_\text{min}` and :math:`x_\text{max}`, 
453      scale :math:`s` and zero point :math:`z` are computed as: 
454   
455      The running minimum/maximum :math:`x_\text{min/max}` is computed as: 
456   
457      .. math:: 
458   
459          \begin{array}{ll} 
460          x_\text{min} &= \begin{cases} 
461              \min(X) & \text{if~}x_\text{min} = \text{None} \\ 
462              \min\left(x_\text{min}, \min(X)\right) & \text{otherwise} 
463          \end{cases}\\ 
464          x_\text{max} &= \begin{cases} 
465              \max(X) & \text{if~}x_\text{max} = \text{None} \\ 
466              \max\left(x_\text{max}, \max(X)\right) & \text{otherwise} 
467          \end{cases}\\ 
468          \end{array} 
469   
470      where :math:`X` is the observed tensor. 
471   
472      The scale :math:`s` and zero point :math:`z` are then computed as: 
473   
474      .. math:: 
475   
476          \begin{aligned} 
477              \text{if Symmetric:}&\\ 
478              &s = 2 \max(|x_\text{min}|, x_\text{max}) / 
479                  \left( Q_\text{max} - Q_\text{min} \right) \\ 
480              &z = \begin{cases} 
481                  0 & \text{if dtype is qint8} \\ 
482                  128 & \text{otherwise} 
483              \end{cases}\\ 
484              \text{Otherwise:}&\\ 
485                  &s = \left( x_\text{max} - x_\text{min}  \right ) / 
486                      \left( Q_\text{max} - Q_\text{min} \right ) \\ 
487                  &z = Q_\text{min} - \text{round}(x_\text{min} / s) 
488          \end{aligned} 
489   
490      where :math:`Q_\text{min}` and :math:`Q_\text{max}` are the minimum and 
491      maximum of the quantized data type. 
492   
493      .. warning:: :attr:`dtype` can only take ``torch.qint8`` or ``torch.quint8``. 
494   
495      .. note:: If the running minimum equals to the running maximum, the scale 
496                and zero_point are set to 1.0 and 0. 
497      """
498  
499      min_val: torch.Tensor
500      max_val: torch.Tensor
501  
502      def __init__(
503          self,
504          dtype=torch.quint8,
505          qscheme=torch.per_tensor_affine,
506          reduce_range=False,
507          quant_min=None,
508          quant_max=None,
509          factory_kwargs=None,
510          eps=torch.finfo(torch.float32).eps,
511          is_dynamic=False,
512          **kwargs,
513      ) -> None:
514          if not is_per_tensor(qscheme):
515              raise NotImplementedError(
516                  "MinMaxObserver's qscheme only support torch.per_tensor_symmetric \ 
517                      and torch.per_tensor_affine."
518              )
519          # TODO: MinMaxObserver by itself doesn't support dynamic quantization, but
520          # if it's inherited by MovingAverageObserver, and averaging_constant is 1, it
521          # supports dynamic quantization, we may need to better error checking here
522  
523          # For x86 quantized kernels, we need to ensure that the vpmaddubsw
524          # instruction does not overflow. We allow for a reduce_range argument to
525          # observers that reduces the quantized range to (0,127) or (-64, 63).
526          # For more details see aten/src/ATen/native/quantized/cpu/qconv.cpp
527          # This is not an optimal choice for non x86 backends as it loses a bit
528          # of precision for activations.
529          super().__init__(
530              dtype=dtype,
531              qscheme=qscheme,
532              reduce_range=reduce_range,
533              quant_min=quant_min,
534              quant_max=quant_max,
535              factory_kwargs=factory_kwargs,
536              eps=eps,
537              is_dynamic=is_dynamic,
538              **kwargs,
539          )
540          factory_kwargs = torch.nn.factory_kwargs(factory_kwargs)
541          self.register_buffer("min_val", torch.tensor(float("inf"), **factory_kwargs))
542          self.register_buffer("max_val", torch.tensor(float("-inf"), **factory_kwargs))
543          if (
544              self.qscheme == torch.per_tensor_symmetric
545              and self.reduce_range
546              and self.dtype == torch.quint8
547          ):
548              raise NotImplementedError(
549                  "Cannot reduce range for symmetric \ 
550                                         quantization for quint8"
551              )
552  
553      def forward(self, x_orig):
554          r"""Records the running minimum and maximum of ``x``."""
555          if x_orig.numel() == 0:
556              return x_orig
557          x = x_orig.detach()  # avoid keeping autograd tape
558          x = x.to(self.min_val.dtype)
559          min_val_cur, max_val_cur = torch.aminmax(x)
560          min_val = torch.min(min_val_cur, self.min_val)
561          max_val = torch.max(max_val_cur, self.max_val)
562          self.min_val.copy_(min_val)
563          self.max_val.copy_(max_val)
564          return x_orig

569          return self._calculate_qparams(self.min_val, self.max_val)

582  class MovingAverageMinMaxObserver(MinMaxObserver):
583      r"""Observer module for computing the quantization parameters based on the 
584      moving average of the min and max values. 
585   
586      This observer computes the quantization parameters based on the moving 
587      averages of minimums and maximums of the incoming tensors. The module 
588      records the average minimum and maximum of incoming tensors, and uses this 
589      statistic to compute the quantization parameters. 
590   
591      Args: 
592          averaging_constant: Averaging constant for min/max. 
593          dtype: dtype argument to the `quantize` node needed to implement the 
594                 reference model spec. 
595          qscheme: Quantization scheme to be used 
596          reduce_range: Reduces the range of the quantized data type by 1 bit 
597          quant_min: Minimum quantization value. If unspecified, it will follow the 8-bit setup. 
598          quant_max: Maximum quantization value. If unspecified, it will follow the 8-bit setup. 
599          eps: Epsilon value for float32, Defaults to `torch.finfo(torch.float32).eps`. 
600   
601      The moving average min/max is computed as follows 
602   
603      .. math:: 
604   
605          \begin{array}{ll} 
606                  x_\text{min} = \begin{cases} 
607                      \min(X) & \text{if~}x_\text{min} = \text{None} \\ 
608                      (1 - c) x_\text{min} + c \min(X) & \text{otherwise} 
609                  \end{cases}\\ 
610                  x_\text{max} = \begin{cases} 
611                      \max(X) & \text{if~}x_\text{max} = \text{None} \\ 
612                      (1 - c) x_\text{max} + c \max(X) & \text{otherwise} 
613                  \end{cases}\\ 
614          \end{array} 
615   
616      where :math:`x_\text{min/max}` is the running average min/max, :math:`X` is 
617      is the incoming tensor, and :math:`c` is the ``averaging_constant``. 
618   
619      The scale and zero point are then computed as in 
620      :class:`~torch.ao.quantization.observer.MinMaxObserver`. 
621   
622      .. note:: Only works with ``torch.per_tensor_affine`` quantization scheme. 
623   
624      .. note:: If the running minimum equals to the running maximum, the scale 
625                and zero_point are set to 1.0 and 0. 
626      """
627  
628      def __init__(
629          self,
630          averaging_constant=0.01,
631          dtype=torch.quint8,
632          qscheme=torch.per_tensor_affine,
633          reduce_range=False,
634          quant_min=None,
635          quant_max=None,
636          eps=torch.finfo(torch.float32).eps,
637          is_dynamic=False,
638          **kwargs,
639      ) -> None:
640          if not is_per_tensor(qscheme):
641              raise NotImplementedError(
642                  f"MovingAverageMinMaxObserver's qscheme only support \ 
643                  torch.per_tensor_symmetric and torch.per_tensor_affine. \ 
644                  but got: {qscheme}"
645              )
646          self.averaging_constant = averaging_constant
647          if is_dynamic and self.averaging_constant != 1:
648              raise NotImplementedError(
649                  "MovingAverageMinMaxObserver doesn't support dynamic quantization for "
650                  f"averaging constant of {self.averaging_constant}"
651              )
652          super().__init__(
653              dtype=dtype,
654              qscheme=qscheme,
655              reduce_range=reduce_range,
656              quant_min=quant_min,
657              quant_max=quant_max,
658              eps=eps,
659              is_dynamic=is_dynamic,
660              **kwargs,
661          )
662  
663      def forward(self, x_orig):
664          if x_orig.numel() == 0:
665              return x_orig
666          x = x_orig.detach()  # avoid keeping autograd tape
667          x = x.to(self.min_val.dtype)
668          min_val = self.min_val
669          max_val = self.max_val
670          if min_val == float("inf") and max_val == float("-inf"):
671              min_val, max_val = torch.aminmax(x)
672          else:
673              min_val_cur, max_val_cur = torch.aminmax(x)
674              min_val = min_val + self.averaging_constant * (min_val_cur - min_val)
675              max_val = max_val + self.averaging_constant * (max_val_cur - max_val)
676          self.min_val.copy_(min_val)
677          self.max_val.copy_(max_val)
678          return x_orig

980  class HistogramObserver(UniformQuantizationObserverBase):
981      r""" 
982      The module records the running histogram of tensor values along with 
983      min/max values. ``calculate_qparams`` will calculate scale and zero_point. 
984   
985      Args: 
986          bins: Number of bins to use for the histogram 
987          dtype: dtype argument to the `quantize` node needed to implement the 
988                 reference model spec 
989          qscheme: Quantization scheme to be used 
990          reduce_range: Reduces the range of the quantized data type by 1 bit 
991          eps: Epsilon value for float32, Defaults to `torch.finfo(torch.float32).eps`.

993      The scale and zero point are computed as follows: 
994   
995      1. Create the histogram of the incoming inputs. 
996          The histogram is computed continuously, and the ranges per bin change 
997          with every new tensor observed. 
998      2. Search the distribution in the histogram for optimal min/max values. 
999          The search for the min/max values ensures the minimization of the 
1000         quantization error with respect to the floating point model. 
1001     3. Compute the scale and zero point the same way as in the 
1002         :class:`~torch.ao.quantization.MinMaxObserver` 
1003     """

1005     histogram: torch.Tensor
1006     min_val: torch.Tensor
1007     max_val: torch.Tensor

1010         self,

1046         self.register_buffer("min_val", torch.tensor(float("inf"), **factory_kwargs))
1047         self.register_buffer("max_val", torch.tensor(float("-inf"), **factory_kwargs))

1053     def _get_norm(
1054         self, delta_begin: torch.Tensor, delta_end: torch.Tensor, density: torch.Tensor
1055     ) -> torch.Tensor:
1056         r""" 
1057         Compute the norm of the values uniformaly distributed between 
1058         delta_begin and delta_end. 
1059         Currently only L2 norm is supported. 
1060  
1061         norm = density * (integral_{begin, end} x^2) 
1062              = density * (end^3 - begin^3) / 3 
1063         """
1064         norm = (
1065             delta_end * delta_end * delta_end - delta_begin * delta_begin * delta_begin
1066         ) / 3
1067         return density * norm
1068 
1069     def _compute_quantization_error(self, next_start_bin: int, next_end_bin: int):
1070         r""" 
1071         Compute the quantization error if we use start_bin to end_bin as the 
1072         min and max to do the quantization. 
1073         """
1074         bin_width = (self.max_val.item() - self.min_val.item()) / self.bins
1075 
1076         dst_bin_width = bin_width * (next_end_bin - next_start_bin + 1) / self.dst_nbins
1077         if dst_bin_width == 0.0:
1078             return 0.0
1079 
1080         src_bin = torch.arange(self.bins, device=self.histogram.device)
1081         # distances from the beginning of first dst_bin to the beginning and
1082         # end of src_bin
1083         src_bin_begin = (src_bin - next_start_bin) * bin_width
1084         src_bin_end = src_bin_begin + bin_width
1085 
1086         # which dst_bins the beginning and end of src_bin belong to?
1087         dst_bin_of_begin = torch.clamp(
1088             torch.div(src_bin_begin, dst_bin_width, rounding_mode="floor"),
1089             0,
1090             self.dst_nbins - 1,
1091         )
1092         dst_bin_of_begin_center = (dst_bin_of_begin + 0.5) * dst_bin_width
1093 
1094         dst_bin_of_end = torch.clamp(
1095             torch.div(src_bin_end, dst_bin_width, rounding_mode="floor"),
1096             0,
1097             self.dst_nbins - 1,
1098         )
1099         density = self.histogram / bin_width
1100 
1101         norm = torch.zeros(self.bins, device=self.histogram.device)
1102 
1103         delta_begin = src_bin_begin - dst_bin_of_begin_center
1104         delta_end = dst_bin_width / 2
1105         norm += self._get_norm(
1106             delta_begin,
1107             torch.ones(self.bins, device=self.histogram.device) * delta_end,
1108             density,
1109         )
1110 
1111         norm += (dst_bin_of_end - dst_bin_of_begin - 1) * self._get_norm(
1112             torch.tensor(-dst_bin_width / 2), torch.tensor(dst_bin_width / 2), density
1113         )
1114 
1115         dst_bin_of_end_center = dst_bin_of_end * dst_bin_width + dst_bin_width / 2
1116 
1117         delta_begin = -dst_bin_width / 2
1118         delta_end = src_bin_end - dst_bin_of_end_center
1119         norm += self._get_norm(torch.tensor(delta_begin), delta_end, density)
1120 
1121         return norm.sum().item()
1122 
1123     def _non_linear_param_search(self) -> tuple[torch.Tensor, torch.Tensor]:
1124         r"""Non-linear parameter search. 
1125  
1126         An approximation for L2 error minimization for selecting min/max. 
1127         By selecting new min/max, we filter out outliers in input distribution. 
1128         This follows the implementation of NormMinimization::NonlinearQuantizationParamsSearch in 
1129         caffe2/quantization/server/norm_minimization.cc 
1130         """
1131         assert self.histogram.size()[0] == self.bins, "bins mismatch"
1132         bin_width = (self.max_val - self.min_val) / self.bins
1133 
1134         # cumulative sum
1135         total = torch.sum(self.histogram).item()
1136         cSum = torch.cumsum(self.histogram, dim=0)
1137 
1138         stepsize = 1e-5  # granularity
1139         alpha = 0.0  # lower bound
1140         beta = 1.0  # upper bound
1141         start_bin = 0
1142         end_bin = self.bins - 1
1143         norm_min = float("inf")
1144 
1145         while alpha < beta:
1146             # Find the next step
1147             next_alpha = alpha + stepsize
1148             next_beta = beta - stepsize
1149 
1150             # find the left and right bins between the quantile bounds
1151             l = start_bin
1152             r = end_bin
1153             while l < end_bin and cSum[l] < next_alpha * total:
1154                 l = l + 1
1155             while r > start_bin and cSum[r] > next_beta * total:
1156                 r = r - 1
1157 
1158             # decide the next move
1159             next_start_bin = start_bin
1160             next_end_bin = end_bin
1161             if (l - start_bin) > (end_bin - r):
1162                 # move the start bin
1163                 next_start_bin = l
1164                 alpha = next_alpha
1165             else:
1166                 # move the end bin
1167                 next_end_bin = r
1168                 beta = next_beta
1169 
1170             if next_start_bin == start_bin and next_end_bin == end_bin:
1171                 continue
1172 
1173             # calculate the quantization error using next_start_bin and next_end_bin
1174             norm = self._compute_quantization_error(next_start_bin, next_end_bin)
1175 
1176             if norm > norm_min:
1177                 break
1178             norm_min = norm
1179             start_bin = next_start_bin
1180             end_bin = next_end_bin
1181 
1182         new_min = self.min_val + bin_width * start_bin
1183         new_max = self.min_val + bin_width * (end_bin + 1)
1184         return new_min, new_max

1186     def _upscale_histogram(
1187         self,
1188         histogram: torch.Tensor,
1189         orig_min: torch.Tensor,
1190         orig_max: torch.Tensor,
1191         update_min: torch.Tensor,
1192         update_max: torch.Tensor,
1193     ):
1194         # this turns the histogram into a more fine-coarsed histogram to reduce
1195         # bin quantization errors
1196         histogram = histogram.repeat_interleave(self.upsample_rate) / self.upsample_rate
1197         bin_size = (orig_max - orig_min) / (self.bins * self.upsample_rate)
1198         mid_points_histogram = (
1199             torch.linspace(
1200                 orig_min,
1201                 orig_max,
1202                 self.bins * self.upsample_rate + 1,
1203                 device=orig_min.device,
1204             )[:-1].to(histogram.device)
1205             + 0.5 * bin_size
1206         )
1207         boundaries_new_histogram = torch.linspace(
1208             update_min, update_max, self.bins + 1, device=update_min.device
1209         ).to(histogram.device)
1210         # this maps the mid-poits of the histogram to the new histogram's space
1211         bucket_assignments = (
1212             torch.bucketize(mid_points_histogram, boundaries_new_histogram, right=True)
1213             - 1
1214         )
1215         # this then maps the histogram mid-points in the new space, weighted by the original histogram's values
1216         # this is just the old histogram in the new histogram's space
1217 
1218         # In case due to numerical issues the values land higher/lower than the maximum/minimum
1219         bucket_assignments[bucket_assignments >= self.bins] = self.bins - 1
1220         bucket_assignments[bucket_assignments < 0] = 0
1221 
1222         update_histogram = torch.bincount(
1223             bucket_assignments, weights=histogram, minlength=self.bins
1224         )
1225         return update_histogram
1226 
1227     def _combine_histograms(
1228         self,
1229         orig_hist: torch.Tensor,
1230         orig_min: torch.Tensor,
1231         orig_max: torch.Tensor,
1232         update_hist: torch.Tensor,
1233         update_min: torch.Tensor,
1234         update_max: torch.Tensor,
1235     ) -> torch.Tensor:
1236         # If the new min and max are the same as the current min and max,
1237         # we can just add the new histogram to the original histogram
1238         if update_min == orig_min and update_max == orig_max:
1239             return orig_hist + update_hist
1240 
1241         # If the orig hist only has one value (i.e., the min and max are the same)
1242         # we can just add it into new histogram
1243         if orig_min == orig_max:
1244             bin_value = torch.sum(update_hist)
1245             transformed_orig_hist = (
1246                 torch.histc(orig_min, bins=self.bins, min=update_min, max=update_max)  # type: ignore[arg-type]
1247                 * bin_value
1248             )
1249             return transformed_orig_hist + update_hist
1250 
1251         # We assume the update_hist is already in the target range, we will map the orig_max to it
1252         assert update_min <= orig_min
1253         assert update_max >= orig_max
1254 
1255         # Now we need to turn the old_histogram, into the range of the new histogram
1256         transformed_orig_hist = self._upscale_histogram(
1257             orig_hist,
1258             orig_min,
1259             orig_max,
1260             update_min,
1261             update_max,
1262         )
1263 
1264         return update_hist + transformed_orig_hist

1266     def reset_histogram(
1267         self, x: torch.Tensor, min_val: torch.Tensor, max_val: torch.Tensor
1268     ) -> None:
1269         self.min_val.resize_(min_val.shape)
1270         self.min_val.copy_(min_val)
1271         self.max_val.resize_(max_val.shape)
1272         self.max_val.copy_(max_val)
1273         assert min_val.numel() == 1 and max_val.numel() == 1, (
1274             "histogram min/max values must be scalar."
1275         )
1276         new_histogram = torch.histc(x, self.bins, min=min_val, max=max_val)  # type: ignore[arg-type]
1277         self.histogram.detach_().resize_(new_histogram.shape)
1278         self.histogram.copy_(new_histogram)
1279 
1280     def forward(self, x_orig: torch.Tensor) -> torch.Tensor:  # pyre-ignore[14]
1281         if x_orig.numel() == 0:
1282             return x_orig
1283         x = x_orig.detach()
1284         x_min, x_max = torch.aminmax(x)
1285         # want to ignore torch.inf since we don't actually
1286         # want to make our quantization range infinite
1287         # and in practice those values will be clamped
1288         if x_min == -torch.inf or x_max == torch.inf:
1289             warnings.warn("torch.inf detected in input tensor, ignoring input")
1290             x = x[x.abs() != torch.inf]
1291             if x.numel() == 0:
1292                 return x_orig
1293             x_min, x_max = torch.aminmax(x)
1294 
1295         current_min = self.min_val
1296         current_max = self.max_val
1297 
1298         is_uninitialized = self.min_val == float("inf") or self.max_val == float("-inf")
1299         if is_uninitialized:
1300             self.reset_histogram(x, x_min, x_max)
1301         else:
1302             update_min, update_max = x_min, x_max
1303             new_min = torch.min(current_min, update_min)
1304             new_max = torch.max(current_max, update_max)
1305 
1306             # TODO: For some reason, this is required for it to pass torchscript test
1307             # new_min and new_max should already have requires_grad set to False
1308             new_min, new_max = new_min.detach(), new_max.detach()
1309             update_histogram = torch.histc(
1310                 x,
1311                 self.bins,
1312                 min=new_min,  # type: ignore[arg-type]
1313                 max=new_max,  # type: ignore[arg-type]
1314             ).to(self.histogram.device)
1315             if new_min == current_min and new_max == current_max:
1316                 combined_histogram = self.histogram + update_histogram
1317                 self.histogram.detach_().resize_(combined_histogram.shape)
1318                 self.histogram.copy_(combined_histogram)
1319             else:
1320                 combined_histogram = self._combine_histograms(
1321                     self.histogram,
1322                     current_min,
1323                     current_max,
1324                     update_histogram,
1325                     new_min,
1326                     new_max,
1327                 )
1328                 self.histogram.detach_().resize_(combined_histogram.shape)
1329                 self.histogram.copy_(combined_histogram)
1330                 self.min_val.detach_().resize_(new_min.shape)
1331                 self.min_val.copy_(new_min)
1332                 self.max_val.detach_().resize_(new_max.shape)
1333                 self.max_val.copy_(new_max)
1334 
1335         return x_orig

1776 def get_block_size(
1777     input_shape: tuple[int, ...], granularity: Granularity
1778 ) -> tuple[int, ...]:
1779     """Get the block size based on the input shape and granularity type. 
1780  
1781     Args: 
1782         input_shape: The input tensor shape possibly more than 2 dimensions 
1783         granularity: The granularity type of the quantization 
1784     """
1785     assert isinstance(granularity, Granularity), (
1786         "Please provide an instance of Granularity, not subclass of it"
1787     )
1788     if isinstance(granularity, PerTensor):
1789         return input_shape
1790     elif isinstance(granularity, PerAxis):
1791         block_size = list(input_shape)
1792         block_size[granularity.axis] = 1
1793         return tuple(block_size)
1794     elif isinstance(granularity, PerRow):
1795         return (1,) * (len(input_shape) - 1) + (input_shape[-1],)
1796     elif isinstance(granularity, PerGroup):
1797         assert len(input_shape) == 2, (
1798             f"Expecting input shape dim to be 2 for per group quantization, gotinput shape: {input_shape}"
1799         )
1800         return (1, granularity.group_size)
1801     elif isinstance(granularity, PerToken):
1802         block_size = [1] * len(input_shape)
1803         block_size[-1] = input_shape[-1]
1804         return tuple(block_size)
1805     raise ValueError(f"Unsupported Granularity: {granularity}")

1870             observer_node: the observer node to convert 
1871         """
1872         from torch.ao.quantization.fx.utils import create_getattr_from_value
1873 
1874         with model.graph.inserting_before(observer_node):
1875             assert self.block_size is not None, "Expecting block_size to be populated"
1876             assert self.original_dtype is not None, (
1877                 "Expecting original_dtype to be populated"
1878             )
1879             if hasattr(self, "is_dynamic") and self.is_dynamic:
1880                 choose_qparams_affine = model.graph.call_function(
1881                     torch.ops.pt2e_quant.choose_qparams_affine,
1882                     (
1883                         observer_node.args[0],

1885                         self.block_size,
1886                         self.target_dtype,
1887                         self.quant_min,
1888                         self.quant_max,
1889                         self.eps,
1890                         self.scale_dtype,
1891                         self.zero_point_dtype,
1892                         self.preserve_zero,
1893                         self.zero_point_domain.name,
1894                     ),
1895                 )
1896                 scale_node = model.graph.call_function(
1897                     operator.getitem, (choose_qparams_affine, 0)
1898                 )
1899                 zero_point_node = model.graph.call_function(
1900                     operator.getitem, (choose_qparams_affine, 1)
1901                 )
1902             else:
1903                 scale, zero_point = self.calculate_qparams()
1904                 scale_node = create_getattr_from_value(
1905                     model, model.graph, "_scale", scale
1906                 )
1907                 zero_point_node = create_getattr_from_value(
1908                     model, model.graph, "_zero_point", zero_point
1909                 )
1910 
1911             q_node = model.graph.call_function(
1912                 torch.ops.pt2e_quant.quantize_affine,
1913                 (
1914                     observer_node.args[0],
1915                     self.block_size,
1916                     scale_node,
1917                     zero_point_node,
1918                     self.target_dtype,
1919                     self.quant_min,
1920                     self.quant_max,
1921                     self.zero_point_domain.name,
1922                 ),
1923                 {},
1924             )
1925             dq_node = model.graph.call_function(
1926                 torch.ops.pt2e_quant.dequantize_affine,
1927                 (
1928                     q_node,
1929                     self.block_size,
1930                     scale_node,
1931                     zero_point_node,
1932                     self.target_dtype,
1933                     self.quant_min,
1934                     self.quant_max,
1935                     self.zero_point_domain.name,
1936                 ),
1937                 {"output_dtype": self.original_dtype},
1938             )
1939             observer_node.replace_all_uses_with(dq_node)
1940             model.graph.erase_node(observer_node)

1944     """Returns true if given mod is an instance of Observer script module."""
1945     if isinstance(mod, torch.jit.RecursiveScriptModule):

1947         suffix = mod._c.qualified_name.split(".", 1)[1]

1977 def get_observer_state_dict(mod):
1978     r""" 
1979     Returns the state dict corresponding to the observer stats. 
1980     Traverse the model state_dict and extract out the stats. 
1981     """
1982     od = OrderedDict()
1983     if isinstance(mod, torch.jit.RecursiveScriptModule):
1984         for k, v in mod.state_dict().items():
1985             if "observer" in k:
1986                 od[k] = v
1987     else:
1988         # path for GraphModule and nn.Module (eager mode)
1989         for k, v in mod.state_dict().items():
1990             if "activation_post_process" in k:
1991                 od[k] = v
1992     od._metadata = mod.state_dict()._metadata  # type: ignore[attr-defined]
1993     return od

1996 def load_observer_state_dict(mod, obs_dict):
1997     r""" 
1998     Given input model and a state_dict containing model observer stats, 
1999     load the stats back into the model. The observer state_dict can be saved 
2000     using torch.ao.quantization.get_observer_state_dict 
2001     """
2002     missing_keys: list[str] = []
2003     unexpected_keys: list[str] = []
2004     for name, module in mod.named_modules():
2005         prefix = name + "."
2006         if _is_activation_post_process(module):
2007             if _is_per_channel_script_obs_instance(module):
2008                 # For per-channel observers we need to call a custom load_from_state_dict to resize the tensor.
2009                 # However this is not called when the module is scripted and we end up calling the default one in module.py
2010                 module._load_from_state_dict_script(
2011                     obs_dict, prefix, {}, True, missing_keys, unexpected_keys, []
2012                 )
2013             else:
2014                 module._load_from_state_dict(
2015                     obs_dict, prefix, {}, False, missing_keys, unexpected_keys, []
2016                 )
2017     for k in missing_keys:
2018         if "observer" in k or "activation_post_process" in k:
2019             raise Exception(  # noqa: TRY002
2020                 f"Missing keys for observer {k} in state_dict"
2021             )
2022     for k in unexpected_keys:
2023         if "observer" in k or "activation_post_process" in k:
2024             raise Exception(  # noqa: TRY002
2025                 f"Unexpected keys for observer {k} in state_dict"
2026             )
