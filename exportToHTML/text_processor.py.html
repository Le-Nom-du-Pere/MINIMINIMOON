7    def normalize_unicode(text: str) -> str:
8        """Normalize Unicode text using NFKC normalization."""
9        return unicodedata.normalize("NFKC", text)

12   def find_quotes(text: str) -> List[Match[str]]:
13       """Find all quote characters in text with Unicode normalization."""
14       normalized_text = normalize_unicode(text)
15       pattern = r'[""''"\']'
16       return re.findall(pattern, normalized_text)

19   def count_words(text: str) -> int:
20       """Count words in text with Unicode normalization."""
21       normalized_text = normalize_unicode(text)
22       pattern = r'\b\w+\b'
23       matches = re.findall(pattern, normalized_text)

63   def highlight_keywords(text: str, keywords: List[str]) -> str:
64       """Highlight keywords with Unicode normalization."""
65       normalized_text = normalize_unicode(text)
66       for keyword in keywords:
67           normalized_keyword = normalize_unicode(keyword)
68           pattern = re.escape(normalized_keyword)
69           normalized_text = re.sub(pattern, f"**{normalized_keyword}**", normalized_text, flags=re.IGNORECASE)
70       return normalized_text
