1    """Isolated calls to methods that may segfault."""
2    
3    from __future__ import annotations
4    
5    import multiprocessing
6    import os
7    import warnings
8    from typing import Any, Callable, TypeVar, TypeVarTuple, Union, Unpack
9    from typing_extensions import ParamSpec
10   
11   
12   _P = ParamSpec("_P")
13   _R = TypeVar("_R")
14   _Ts = TypeVarTuple("_Ts")
15   
16   _IS_WINDOWS = os.name == "nt"
17   
18   
19   def _call_function_and_return_exception(
20       func: Callable[[Unpack[_Ts]], _R], args: tuple[Unpack[_Ts]], kwargs: dict[str, Any]
21   ) -> Union[_R, Exception]:
22       """Call function and return a exception if there is one."""
23   
24       try:
25           return func(*args, **kwargs)
26       except Exception as e:
27           return e
28   
29   
30   def safe_call(func: Callable[_P, _R], *args: _P.args, **kwargs: _P.kwargs) -> _R:
31       """Call a function in a separate process. 
32    
33       Args: 
34           func: The function to call. 
35           args: The positional arguments to pass to the function. 
36           kwargs: The keyword arguments to pass to the function. 
37    
38       Returns: 
39           The return value of the function. 
40    
41       Raises: 
42           Exception: If the function raised an exception. 
43       """
44       if _IS_WINDOWS:
45           # On Windows, we cannot create a new process with fork.
46           warnings.warn(
47               f"A new process is not created for {func} on Windows.", stacklevel=1
48           )
49           return func(*args, **kwargs)
50   
51       with multiprocessing.get_context("fork").Pool(1) as pool:
52           # It is important to fork a process here to prevent the main logic from
53           # running again when the user does not place it under a `if __name__ == "__main__":`
54           # block.
55           result = pool.apply_async(
56               _call_function_and_return_exception, (func, args, kwargs)
57           )
58           result = result.get(timeout=5)
59       if isinstance(result, Exception):
60           raise result
61       return result
