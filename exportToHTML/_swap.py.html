22   def _get_getitem_users(node: torch.fx.Node) -> set[torch.fx.Node]:
23       node_users = list(node.users.keys())
24       getitem_users = set()
25       for user in node_users:
26           if user.op == "output":
27               continue
28   
29           assert user.op == "call_function" and user.target == operator.getitem, (
30               f"Expected getitem node as user for {node}, instead got {user}"
31           )
32           getitem_users.update(list(user.users.keys()))
33       return getitem_users

36   def _try_remove_connecting_pytrees(curr_module_node: torch.fx.Node) -> None:
37       """ 
38       We want to try to remove extraneous pytree flatten/unflatten calls between modules 
39       calls. Instead of having the following: 
40       graph(): 
41           ... 
42           %foo : [num_users=1] = call_module[target=foo](args = (%getitem_1, %getitem_2), kwargs = {}) 
43           %tree_flatten_spec : [num_users=1] = call_function[target=torch.fx._pytree.tree_flatten_spec](args = (%foo, %_spec_1), kwargs = {}) 
44           %getitem_4 : [num_users=1] = call_function[target=operator.getitem](args = (%tree_flatten_spec, 0), kwargs = {}) 
45           %tree_unflatten_1 : [num_users=2] = call_function[target=torch.utils._pytree.tree_unflatten](args = ([%getitem_4], %_spec_2), kwargs = {}) 
46           %getitem_5 : [num_users=1] = call_function[target=operator.getitem](args = (%tree_unflatten_1, 0), kwargs = {}) 
47           %getitem_7 : [num_users=0] = call_function[target=operator.getitem](args = (%tree_unflatten_1, 1), kwargs = {}) 
48           %getitem_6 : [num_users=1] = call_function[target=operator.getitem](args = (%getitem_5, 0), kwargs = {}) 
49           %bar : [num_users=1] = call_module[target=bar](args = (%getitem_6,), kwargs = {}) 
50           ... 
51    
52       We could do the following, if we know that all the outputs of `foo` feed into `bar`: 
53       graph(): 
54           ... 
55           %foo : [num_users=1] = call_module[target=foo](args = (%getitem_1, %getitem_2), kwargs = {}) 
56           %bar : [num_users=1] = call_module[target=bar](args = (%getitem_6,), kwargs = {}) 
57           ... 
58    
59       Currently this optimization only works for the case where all of the outputs 
60       of `foo` go directly into `bar`, and `bar` has no other inputs. 
61       """  # noqa: B950
62   
63       log.debug("Trying to remove pytrees for module call %s", curr_module_node)
64   
65       curr_module_users = list(curr_module_node.users.keys())
66       assert len(curr_module_users) == 1, (
67           f"Expected only one user for module node, instead got {list(curr_module_users)}"
68       )
69       flatten_node = curr_module_users[0]
70       assert (
71           flatten_node.op == "call_function"
72           and flatten_node.target == fx_pytree.tree_flatten_spec
73       )
74   
75       flatten_getitem_users = _get_getitem_users(flatten_node)
76       if len(flatten_getitem_users) != 1:
77           log.debug(
78               "More than one user found for flatten node, %s: %s. "
79               "Unable to fuse it with another unflatten call.",
80               flatten_node,
81               flatten_getitem_users,
82           )
83           return
84   
85       unflatten_node = next(iter(flatten_getitem_users))
86       if not (
87           unflatten_node.op == "call_function"
88           and unflatten_node.target == pytree.tree_unflatten
89       ):
90           log.debug(
91               "Flatten node %s's user is not a pytree.tree_unflatten. "
92               "Instead it is: %s. Passing...",
93               flatten_node,
94               unflatten_node,
95           )
96           return
97   
98       for i, arg in enumerate(unflatten_node.args[0]):  # type: ignore[union-attr,arg-type]
99           if arg not in flatten_node.users:
100              log.debug(
101                  "Module %s's outputs are not all directly used as inputs to "
102                  "the subsequent module. Unable to fuse the connecting "
103                  "flatten/unflatten. The inputs to the subsequent module are: %s. ",
104                  curr_module_node,
105                  unflatten_node.args[0],
106              )
107              return
108  
109          if not (
110              arg.op == "call_function"
111              and arg.target == operator.getitem
112              and arg.args[1] == i
113          ):
114              log.debug(
115                  "Module %s's outputs are not all directly used in the same "
116                  "order as outputted. Unable to fuse the connecting "
117                  "flatten/unflatten. The inputs to the "
118                  "subsequent module are: %s. ",
119                  curr_module_node,
120                  unflatten_node.args[0],
121              )
122              return
123  
124      # Unflatten has two levels of getitem, because it gets the args and kwargs
125      unflatten_getitem_getitem_users = set()
126      unflatten_getitem_users = _get_getitem_users(unflatten_node)
127      for unflatten_getitem_user in unflatten_getitem_users:
128          unflatten_getitem_getitem_users.update(
129              list(unflatten_getitem_user.users.keys())
130          )
131  
132      if len(unflatten_getitem_getitem_users) != 1:
133          log.debug(
134              "More than one user found for unflatten node, %s: %s. "
135              "Unable to fuse it with another flatten call.",
136              unflatten_node,
137              unflatten_getitem_getitem_users,
138          )
139          return
140  
141      next_module_node = next(iter(unflatten_getitem_getitem_users))
142      if not (next_module_node.op == "call_module"):
143          log.debug(
144              "Unflatten node %s's user is not a call_module. "
145              "Instead it is: %s. Passing...",
146              unflatten_node,
147              next_module_node,
148          )
149          return
150  
151      # Directly put the outputs of the current module into the next module
152      next_module_node.args = (curr_module_node,)

155  def _remove_extraneous_pytrees(gm: torch.fx.GraphModule) -> None:
156      """ 
157      Remove extraneous pytree flatten/unflatten calls. 
158   
159      We try a couple of optimizations here: 
160          1. Remove pytree flatten/unflatten calls between modules 
161          2. TODO: Remove module's in_spec + initial unflatten call 
162          3. TODO: Remove module's out_spec + final flatten call 
163      """
164  
165      for node in gm.graph.nodes:
166          if node.op == "call_module":
167              _try_remove_connecting_pytrees(node)
168  
169      gm.graph.eliminate_dead_code()
170

172  def _construct_inputs(
173      gm: torch.fx.GraphModule,
174      signature: ModuleCallSignature,
175      node_name_map: dict[str, torch.fx.Node],
176  ) -> tuple[list[torch.fx.Node], dict[str, torch.fx.Node]]:
177      tree_unflatten_args: list[Optional[torch.fx.Node]] = []
178      for input_ in signature.inputs:
179          if isinstance(input_, ConstantArgument) and input_.value is None:
180              # Constants should be directly embedded into the graph and not used
181              # as inputs
182              tree_unflatten_args.append(None)
183          elif input_.name not in node_name_map:
184              # For unused inputs
185              tree_unflatten_args.append(None)
186          else:
187              tree_unflatten_args.append(node_name_map[input_.name])
188  
189      # Insert unflatten call
190      from .unflatten import _generate_unflatten
191  
192      unflatten_node = _generate_unflatten(gm, tree_unflatten_args, signature.in_spec)
193  
194      assert signature.in_spec.num_children == 2
195  
196      args_spec = signature.in_spec.children_specs[0]
197      assert args_spec.context is None
198      args_node = gm.graph.call_function(operator.getitem, (unflatten_node, 0))
199      args_nodes = [
200          gm.graph.call_function(operator.getitem, (args_node, i))
201          for i in range(args_spec.num_children)
202      ]
203  
204      kwargs_spec = signature.in_spec.children_specs[1]
205      assert kwargs_spec.context is not None
206      kwargs_node = gm.graph.call_function(operator.getitem, (unflatten_node, 1))
207      kwargs_nodes = {
208          k: gm.graph.call_function(operator.getitem, (kwargs_node, k))
209          for k in kwargs_spec.context
210      }
211      return args_nodes, kwargs_nodes

214  def _insert_call_module(
215      gm: torch.fx.GraphModule,
216      args_nodes: list[torch.fx.Node],
217      kwargs_nodes: dict[str, torch.fx.Node],
218      module_to_swap: torch.nn.Module,
219      name: str,
220  ) -> torch.fx.Node:
221      from .unflatten import _assign_attr, _AttrKind
222  
223      _assign_attr(module_to_swap, gm, name, _AttrKind.MODULE)
224      module_node = gm.graph.call_module(name, tuple(args_nodes), kwargs_nodes)  # type: ignore[arg-type]
225      return module_node

228  def _deconstruct_outputs(
229      gm: torch.fx.GraphModule,
230      signature: ModuleCallSignature,
231      module_node: torch.fx.Node,
232      node_name_map: dict[str, torch.fx.Node],
233      orig_outputs: tuple[torch.fx.Node, ...],
234  ) -> None:
235      from .unflatten import _generate_flatten_spec
236  
237      flatten_node = _generate_flatten_spec(gm, module_node, signature.out_spec)
238  
239      for i, orig_output in enumerate(orig_outputs):
240          # Use Proxy to record getitem access.
241          proxy_out = torch.fx.Proxy(flatten_node)[i].node  # type: ignore[index]
242          orig_output.replace_all_uses_with(proxy_out, propagate_meta=True)
243  
244          node_name_map[orig_output.name] = proxy_out

247  def _swap_module_helper(
248      gm: torch.fx.GraphModule,
249      modules_to_swap: dict[str, torch.nn.Module],
250      module_call_graph: dict[str, ModuleCallSignature],
251  ) -> torch.fx.GraphModule:
252      log.debug("Starting graph:")
253      log.debug(gm.graph)
254  
255      legalize_graph(gm)
256  
257      partitions: dict[str, NodeList] = defaultdict(list)
258  
259      node_name_map: dict[str, torch.fx.Node] = {
260          node.name: node for node in gm.graph.nodes
261      }
262  
263      # TODO: Handle the duplicate module case
264      for node in gm.graph.nodes:
265          if nn_module_stack := node.meta.get("nn_module_stack"):
266              for path, _ in nn_module_stack.values():
267                  if path in modules_to_swap:
268                      partitions[path].append(node)
269                      break
270  
271      for name, nodes in partitions.items():
272          """ 
273          Given a graph like the following, and we want to swap out the submodule "foo": 
274          graph(): 
275              %x : [num_users=1] = placeholder[target=x] 
276              %y : [num_users=2] = placeholder[target=y] 
277              %add : [num_users=1] = call_function[target=torch.ops.aten.add.Tensor](args = (%y, %x), kwargs = {}), nn_module_stack = {"foo": ("foo", torch.nn.Module)} 
278              %sub : [num_users=1] = call_function[target=torch.ops.aten.sub.Tensor](args = (%y, %add), kwargs = {}), nn_module_stack = {"bar": ("bar", torch.nn.Module)} 
279              return (sub,) 
280   
281          We will first partition out foo's subgraph: 
282          graph(): 
283              %x : [num_users=1] = placeholder[target=x] 
284              %y : [num_users=2] = placeholder[target=y] 
285              %add : [num_users=1] = call_function[target=torch.ops.aten.add.Tensor](args = (%y, %x), kwargs = {}) 
286              return add 
287   
288          And then insert an unflatten + call_module + flatten to replace the subgraph: 
289          graph(): 
290              %x : [num_users=1] = placeholder[target=x] 
291              %y : [num_users=1] = placeholder[target=y] 
292   
293              %_spec_0 : [num_users=1] = get_attr[target=_spec_0] 
294              %tree_unflatten : [num_users=2] = call_function[target=torch.utils._pytree.tree_unflatten](args = ([%x, %y], %_spec_0), kwargs = {}) 
295              %getitem : [num_users=2] = call_function[target=operator.getitem](args = (%tree_unflatten, 0), kwargs = {}) 
296              %getitem_1 : [num_users=1] = call_function[target=operator.getitem](args = (%getitem, 0), kwargs = {}) 
297              %getitem_2 : [num_users=1] = call_function[target=operator.getitem](args = (%getitem, 1), kwargs = {}) 
298              %getitem_3 : [num_users=0] = call_function[target=operator.getitem](args = (%tree_unflatten, 1), kwargs = {}) 
299              %foo : [num_users=0] = call_module[target=foo](args = (%getitem_1, %getitem_2), kwargs = {}) 
300              %_spec_1 : [num_users=1] = get_attr[target=_spec_1] 
301              %tree_flatten_spec : [num_users=1] = call_function[target=torch.fx._pytree.tree_flatten_spec](args = (None, %_spec_1), kwargs = {}) 
302              %getitem_4 : [num_users=1] = call_function[target=operator.getitem](args = (%tree_flatten_spec, 0), kwargs = {}) 
303   
304              %sub : [num_users=1] = call_function[target=torch.ops.aten.sub.Tensor](args = (%y, %getitem_4), kwargs = {}) 
305              return (%sub,) 
306   
307          The `tree_unflatten` call will construct tensor inputs into the input 
308          format needed by the swapped eager module. 
309          The `call_module` node should now reference the swapped torch.nn.Module. 
310          The `tree_flatten_spec` call will deconstruct the eager outputs of the 
311          swapped module into tensors. 
312          """  # noqa: B950
313  
314          submod_name = name.replace(".", "_")
315          sub_gm, orig_inputs, orig_outputs = fuse_as_graphmodule(
316              gm, nodes, f"fused_{submod_name}"
317          )
318  
319          log.debug("Fused subgraph nodes:")
320          log.debug(sub_gm.graph)
321  
322          signature: ModuleCallSignature = module_call_graph[name]
323  
324          args_nodes, kwargs_nodes = _construct_inputs(gm, signature, node_name_map)
325          module_node = _insert_call_module(
326              gm, args_nodes, kwargs_nodes, modules_to_swap[name], name
327          )
328          _deconstruct_outputs(gm, signature, module_node, node_name_map, orig_outputs)
329  
330          erase_nodes(gm, nodes)
331  
332          log.debug("Swapped graph:")
333          log.debug(gm.graph)
334  
335      legalize_graph(gm)
336  
337      log.debug("Before removing extraneous pytrees:")
338      log.debug(gm.graph)
339  
340      _remove_extraneous_pytrees(gm)
341      log.debug("After removing extraneous pytrees:")
342      log.debug(gm.graph)
343  
344      gm.recompile()
345  
346      return gm

349  def _fix_input_output_signature(
350      gm: torch.fx.GraphModule, signature: ModuleCallSignature
351  ) -> None:
352      """ 
353      Given the unlifted module from calling ep.module(), we want to remove the 
354      pytree processing from the graph module's PyTreeCodeGen and instead make it 
355      nodes inside of the graph. This allows us to do some optimizations, like 
356      remove these pytree calls if it is unnecessary, and makes the PyTree part 
357      more obvious to graph passes. 
358      """
359      from torch.export.unflatten import _generate_flatten, _generate_unflatten
360  
361      # Remove the registered pytree codegen because we will take care of it
362      # through inserting pytree nodes into the graph
363      gm.graph._codegen = torch.fx.graph.CodeGen()
364  
365      old_placeholders = [node for node in gm.graph.nodes if node.op == "placeholder"]
366  
367      new_placeholders = []
368      forward_arg_names = signature.forward_arg_names
369      if forward_arg_names is None:
370          forward_arg_names = []
371          assert signature.in_spec.num_children == 2
372          arg_spec = signature.in_spec.children_specs[0]
373          kwarg_spec = signature.in_spec.children_specs[1]
374          assert arg_spec.type == tuple
375          assert kwarg_spec.type == dict
376          for i in range(arg_spec.num_children):
377              forward_arg_names.append(f"arg_{i}")
378          forward_arg_names.extend(kwarg_spec.context)
379  
380      for arg in forward_arg_names:
381          with gm.graph.inserting_before(old_placeholders[0]):
382              new_placeholders.append(gm.graph.placeholder(arg))
383  
384      # Insert flatten call for the inputs
385      with gm.graph.inserting_before(old_placeholders[0]):
386          flat_node = _generate_flatten(gm, tuple(new_placeholders))
387          for i, old_placeholder in enumerate(old_placeholders):
388              old_placeholder.op = "call_function"
389              old_placeholder.target = operator.getitem
390              old_placeholder.args = (flat_node, i)
391  
392      # Insert unflatten call for the outputs
393      output_node = next(node for node in gm.graph.nodes if node.op == "output")
394      with gm.graph.inserting_before(output_node):
395          unflat = _generate_unflatten(gm, output_node.args[0], signature.out_spec)
396          output_node.args = (unflat,)
397  
398      gm.recompile()

401  def _swap_modules(
402      ep: ExportedProgram, modules_to_swap: dict[str, torch.nn.Module]
403  ) -> torch.fx.GraphModule:
404      """ 
405      Unlifts the given ExportedProgram into a fx.GraphModule, and then swaps 
406      previously traced modules with new eager modules specified. Returns a 
407      fx.GraphModule with a custom forward function. 
408   
409      Args: 
410          ep (ExportedProgram): Exported program to modify 
411          modules_to_swap (Dict[str, torch.nn.Module]): Mapping from module fqn to 
412              eager module to swap with. The specified module fqn should have also 
413              been specified in the `preserve_module_call_signature` argument to 
414              torch.export so that we know how to restore the calling convention 
415              to this argument. 
416          run_with_interpreter: Whether or not to run the graph using 
417              fx.Interpreter. Setting to true will help result in better error 
418              messages and easier debugging, but it has found to result in a QPS 
419              drop. 
420      """
421      module_call_graph = {
422          entry.fqn: entry.signature for entry in ep.module_call_graph if entry.signature
423      }
424  
425      gm = ep.module()
426      gm.validate_inputs = False  # type: ignore[assignment]
427      gm.graph.eliminate_dead_code()  # type: ignore[operator, union-attr]
428      assert isinstance(gm, torch.fx.GraphModule)
429      _fix_input_output_signature(gm, ep.module_call_graph[0].signature)
430  
431      gm.module_call_graph = ep.module_call_graph
432      gm.train = types.MethodType(type(gm).train, gm)  # type: ignore[assignment]
433      gm.eval = types.MethodType(type(gm).eval, gm)  # type: ignore[assignment]
434  
435      assert isinstance(gm, torch.fx.GraphModule)
436      gm = _swap_module_helper(gm, modules_to_swap, module_call_graph)
437  
438      return gm
