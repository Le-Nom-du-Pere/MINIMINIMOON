12   class Mod(DefinedFunction):
13       """Represents a modulo operation on symbolic expressions. 
14    
15       Parameters 
16       ========== 
17    
18       p : Expr 
19           Dividend. 
20    
21       q : Expr 
22           Divisor. 
23    
24       Notes 
25       ===== 
26    
27       The convention used is the same as Python's: the remainder always has the 
28       same sign as the divisor. 
29    
30       Many objects can be evaluated modulo ``n`` much faster than they can be 
31       evaluated directly (or at all).  For this, ``evaluate=False`` is 
32       necessary to prevent eager evaluation: 
33    
34       >>> from sympy import binomial, factorial, Mod, Pow 
35       >>> Mod(Pow(2, 10**16, evaluate=False), 97) 
36       61 
37       >>> Mod(factorial(10**9, evaluate=False), 10**9 + 9) 
38       712524808 
39       >>> Mod(binomial(10**18, 10**12, evaluate=False), (10**5 + 3)**2) 
40       3744312326 
41    
42       Examples 
43       ======== 
44    
45       >>> from sympy.abc import x, y 
46       >>> x**2 % y 
47       Mod(x**2, y) 
48       >>> _.subs({x: 5, y: 6}) 
49       1 
50    
51       """
52   
53       kind = NumberKind
54   
55       @classmethod
56       def eval(cls, p, q):
57           def number_eval(p, q):
58               """Try to return p % q if both are numbers or +/-p is known 
59               to be less than or equal q. 
60               """
61   
62               if q.is_zero:
63                   raise ZeroDivisionError("Modulo by zero")
64               if p is S.NaN or q is S.NaN or p.is_finite is False or q.is_finite is False:
65                   return S.NaN
66               if p is S.Zero or p in (q, -q) or (p.is_integer and q == 1):
67                   return S.Zero
68   
69               if q.is_Number:
70                   if p.is_Number:
71                       return p%q
72                   if q == 2:
73                       if p.is_even:
74                           return S.Zero
75                       elif p.is_odd:
76                           return S.One
77   
78               if hasattr(p, '_eval_Mod'):
79                   rv = getattr(p, '_eval_Mod')(q)
80                   if rv is not None:
81                       return rv
82   
83               # by ratio
84               r = p/q
85               if r.is_integer:
86                   return S.Zero
87               try:
88                   d = int(r)
89               except TypeError:
90                   pass
91               else:
92                   if isinstance(d, int):
93                       rv = p - d*q
94                       if (rv*q < 0) == True:
95                           rv += q
96                       return rv
97   
98               # by difference
99               # -2|q| < p < 2|q|
100              if q.is_positive:
101                  comp1, comp2 = is_le, is_lt
102              elif q.is_negative:
103                  comp1, comp2 = is_ge, is_gt
104              else:
105                  return
106              ls = -2*q
107              r = p - q
108              for _ in range(4):
109                  if not comp1(ls, p):
110                      return
111                  if comp2(r, ls):
112                      return p - ls
113                  ls += q
114  
115          rv = number_eval(p, q)
116          if rv is not None:
117              return rv
118  
119          # denest
120          if isinstance(p, cls):
121              qinner = p.args[1]
122              if qinner % q == 0:
123                  return cls(p.args[0], q)
124              elif (qinner*(q - qinner)).is_nonnegative:
125                  # |qinner| < |q| and have same sign
126                  return p
127          elif isinstance(-p, cls):
128              qinner = (-p).args[1]
129              if qinner % q == 0:
130                  return cls(-(-p).args[0], q)
131              elif (qinner*(q + qinner)).is_nonpositive:
132                  # |qinner| < |q| and have different sign
133                  return p
134          elif isinstance(p, Add):
135              # separating into modulus and non modulus
136              both_l = non_mod_l, mod_l = [], []
137              for arg in p.args:
138                  both_l[isinstance(arg, cls)].append(arg)
139              # if q same for all
140              if mod_l and all(inner.args[1] == q for inner in mod_l):
141                  net = Add(*non_mod_l) + Add(*[i.args[0] for i in mod_l])
142                  return cls(net, q)
143  
144          elif isinstance(p, Mul):
145              # separating into modulus and non modulus
146              both_l = non_mod_l, mod_l = [], []
147              for arg in p.args:
148                  both_l[isinstance(arg, cls)].append(arg)
149  
150              if mod_l and all(inner.args[1] == q for inner in mod_l) and all(t.is_integer for t in p.args) and q.is_integer:
151                  # finding distributive term
152                  non_mod_l = [cls(x, q) for x in non_mod_l]
153                  mod = []
154                  non_mod = []
155                  for j in non_mod_l:
156                      if isinstance(j, cls):
157                          mod.append(j.args[0])
158                      else:
159                          non_mod.append(j)
160                  prod_mod = Mul(*mod)
161                  prod_non_mod = Mul(*non_mod)
162                  prod_mod1 = Mul(*[i.args[0] for i in mod_l])
163                  net = prod_mod1*prod_mod
164                  return prod_non_mod*cls(net, q)
165  
166              if q.is_Integer and q is not S.One:
167                  if all(t.is_integer for t in p.args):
168                      non_mod_l = [i % q if i.is_Integer else i for i in p.args]
169                      if any(iq is S.Zero for iq in non_mod_l):
170                          return S.Zero
171  
172              p = Mul(*(non_mod_l + mod_l))
173  
174          # XXX other possibilities?
175  
176          from sympy.polys.polyerrors import PolynomialError
177          from sympy.polys.polytools import gcd
178  
179          # extract gcd; any further simplification should be done by the user
180          try:
181              G = gcd(p, q)
182              if not equal_valued(G, 1):
183                  p, q = [gcd_terms(i/G, clear=False, fraction=False)
184                          for i in (p, q)]
185          except PolynomialError:  # issue 21373
186              G = S.One
187          pwas, qwas = p, q
188  
189          # simplify terms
190          # (x + y + 2) % x -> Mod(y + 2, x)
191          if p.is_Add:
192              args = []
193              for i in p.args:
194                  a = cls(i, q)
195                  if a.count(cls) > i.count(cls):
196                      args.append(i)
197                  else:
198                      args.append(a)
199              if args != list(p.args):
200                  p = Add(*args)
201  
202          else:
203              # handle coefficients if they are not Rational
204              # since those are not handled by factor_terms
205              # e.g. Mod(.6*x, .3*y) -> 0.3*Mod(2*x, y)
206              cp, p = p.as_coeff_Mul()
207              cq, q = q.as_coeff_Mul()
208              ok = False
209              if not cp.is_Rational or not cq.is_Rational:
210                  r = cp % cq
211                  if equal_valued(r, 0):
212                      G *= cq
213                      p *= int(cp/cq)
214                      ok = True
215              if not ok:
216                  p = cp*p
217                  q = cq*q
218  
219          # simple -1 extraction
220          if p.could_extract_minus_sign() and q.could_extract_minus_sign():
221              G, p, q = [-i for i in (G, p, q)]
222  
223          # check again to see if p and q can now be handled as numbers
224          rv = number_eval(p, q)
225          if rv is not None:
226              return rv*G
227  
228          # put 1.0 from G on inside
229          if G.is_Float and equal_valued(G, 1):
230              p *= G
231              return cls(p, q, evaluate=False)
232          elif G.is_Mul and G.args[0].is_Float and equal_valued(G.args[0], 1):
233              p = G.args[0]*p
234              G = Mul._from_args(G.args[1:])
235          return G*cls(p, q, evaluate=(p, q) != (pwas, qwas))

237      def _eval_is_integer(self):
238          p, q = self.args
239          if fuzzy_and([p.is_integer, q.is_integer, fuzzy_not(q.is_zero)]):
240              return True

242      def _eval_is_nonnegative(self):
243          if self.args[1].is_positive:
244              return True

247          if self.args[1].is_negative:

250      def _eval_rewrite_as_floor(self, a, b, **kwargs):
251          from sympy.functions.elementary.integers import floor
252          return a - b*floor(a/b)

254      def _eval_as_leading_term(self, x, logx, cdir):
255          from sympy.functions.elementary.integers import floor
256          return self.rewrite(floor)._eval_as_leading_term(x, logx=logx, cdir=cdir)
