23   # Generalization of getattr
24   def _get_module(model, submodule_key):
25       tokens = submodule_key.split(".")
26       cur_mod = model
27       for s in tokens:
28           cur_mod = getattr(cur_mod, s)
29       return cur_mod

32   # Generalization of setattr
33   def _set_module(model, submodule_key, module):
34       tokens = submodule_key.split(".")
35       sub_tokens = tokens[:-1]
36       cur_mod = model
37       for s in sub_tokens:
38           cur_mod = getattr(cur_mod, s)
39   
40       setattr(cur_mod, tokens[-1], module)

43   def fuse_known_modules(mod_list, is_qat, additional_fuser_method_mapping=None):
44       r"""Return a list of known fuse modules. 
45    
46       Returns a list of modules that fuses the operations specified 
47        in the input module list. 
48    
49       Fuses only the following sequence of modules: 
50       conv, bn 
51       conv, bn, relu 
52       conv, relu 
53       linear, bn 
54       linear, relu 
55       For these sequences, the first element in the output module list performs 
56       the fused operation. The rest of the elements are set to nn.Identity() 
57       """
58       types = tuple(type_before_parametrizations(m) for m in mod_list)
59       fuser_method = get_fuser_method(types, additional_fuser_method_mapping)
60       if fuser_method is None:
61           raise NotImplementedError(f"Cannot fuse modules: {types}")
62       new_mod: list[Optional[nn.Module]] = [None] * len(mod_list)
63       fused = fuser_method(is_qat, *mod_list)
64       # NOTE: forward hooks not processed in the two following for loops will be lost after the fusion
65       # Move pre forward hooks of the base module to resulting fused module
66       for pre_hook_fn in mod_list[0]._forward_pre_hooks.values():
67           fused.register_forward_pre_hook(pre_hook_fn)
68       mod_list[0]._forward_pre_hooks.clear()
69       # Move post forward hooks of the last module to resulting fused module
70       for hook_fn in mod_list[-1]._forward_hooks.values():
71           fused.register_forward_hook(hook_fn)
72       mod_list[-1]._forward_hooks.clear()
73       new_mod[0] = fused
74   
75       for i in range(1, len(mod_list)):
76           identity = nn.Identity()
77           identity.training = mod_list[0].training
78           new_mod[i] = identity
79   
80       return new_mod

83   def _fuse_modules_helper(
84       model,
85       modules_to_fuse,
86       is_qat,
87       fuser_func=fuse_known_modules,
88       fuse_custom_config_dict=None,
89   ):
90       if fuse_custom_config_dict is None:
91           fuse_custom_config_dict = {}
92       additional_fuser_method_mapping = fuse_custom_config_dict.get(
93           "additional_fuser_method_mapping", {}
94       )
95       mod_list = [_get_module(model, item) for item in modules_to_fuse]
96   
97       # Fuse list of modules
98       new_mod_list = fuser_func(mod_list, is_qat, additional_fuser_method_mapping)
99   
100      # Replace original module list with fused module list
101      for i, item in enumerate(modules_to_fuse):
102          _set_module(model, item, new_mod_list[i])

105  def _fuse_modules(
106      model,
107      modules_to_fuse,
108      is_qat,
109      inplace=False,
110      fuser_func=fuse_known_modules,
111      fuse_custom_config_dict=None,
112  ):
113      if not inplace:
114          model = copy.deepcopy(model)
115  
116      if all(isinstance(module_element, str) for module_element in modules_to_fuse):
117          # Handle case of modules_to_fuse being a list
118          _fuse_modules_helper(
119              model, modules_to_fuse, is_qat, fuser_func, fuse_custom_config_dict
120          )
121      else:
122          # Handle case of modules_to_fuse being a list of lists
123          for module_list in modules_to_fuse:
124              _fuse_modules_helper(
125                  model, module_list, is_qat, fuser_func, fuse_custom_config_dict
126              )
127      return model

130  def fuse_modules(
131      model,
132      modules_to_fuse,
133      inplace=False,
134      fuser_func=fuse_known_modules,
135      fuse_custom_config_dict=None,
136  ):
137      r"""Fuse a list of modules into a single module. 
138   
139      Fuses only the following sequence of modules: 
140      conv, bn 
141      conv, bn, relu 
142      conv, relu 
143      linear, relu 
144      bn, relu 
145      All other sequences are left unchanged. 
146      For these sequences, replaces the first item in the list 
147      with the fused module, replacing the rest of the modules 
148      with identity. 
149   
150      Args: 
151          model: Model containing the modules to be fused 
152          modules_to_fuse: list of list of module names to fuse. Can also be a list 
153                           of strings if there is only a single list of modules to fuse. 
154          inplace: bool specifying if fusion happens in place on the model, by default 
155                   a new model is returned 
156          fuser_func: Function that takes in a list of modules and outputs a list of fused modules 
157                      of the same length. For example, 
158                      fuser_func([convModule, BNModule]) returns the list [ConvBNModule, nn.Identity()] 
159                      Defaults to torch.ao.quantization.fuse_known_modules 
160          `fuse_custom_config_dict`: custom configuration for fusion 
161   
162      .. code-block:: python 
163   
164         # Example of fuse_custom_config_dict 
165         fuse_custom_config_dict = { 
166             # Additional fuser_method mapping 
167             "additional_fuser_method_mapping": { 
168                 (torch.nn.Conv2d, torch.nn.BatchNorm2d): fuse_conv_bn 
169             }, 
170         } 
171   
172      Returns: 
173          model with fused modules. A new copy is created if inplace=True. 
174   
175      Examples:: 
176   
177              >>> # xdoctest: +SKIP 
178              >>> m = M().eval() 
179              >>> # m is a module containing the sub-modules below 
180              >>> modules_to_fuse = [ ['conv1', 'bn1', 'relu1'], ['submodule.conv', 'submodule.relu']] 
181              >>> fused_m = torch.ao.quantization.fuse_modules(m, modules_to_fuse) 
182              >>> output = fused_m(input) 
183   
184              >>> m = M().eval() 
185              >>> # Alternately provide a single list of modules to fuse 
186              >>> modules_to_fuse = ['conv1', 'bn1', 'relu1'] 
187              >>> fused_m = torch.ao.quantization.fuse_modules(m, modules_to_fuse) 
188              >>> output = fused_m(input) 
189   
190      """
191      return _fuse_modules(
192          model,
193          modules_to_fuse,
194          is_qat=False,
195          inplace=inplace,
196          fuser_func=fuser_func,
197          fuse_custom_config_dict=fuse_custom_config_dict,
198      )
