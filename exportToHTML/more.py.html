6    def unifiable(cls):
7        """Register standard unify and reify operations on class 
8        This uses the type and __dict__ or __slots__ attributes to define the 
9        nature of the term 
10       See Also: 
11       >>> # xdoctest: +SKIP 
12       >>> class A(object): 
13       ...     def __init__(self, a, b): 
14       ...         self.a = a 
15       ...         self.b = b 
16       >>> unifiable(A) 
17       <class 'unification.more.A'> 
18       >>> x = var("x") 
19       >>> a = A(1, 2) 
20       >>> b = A(1, x) 
21       >>> unify(a, b, {}) 
22       {~x: 2} 
23       """
24       _unify.add((cls, cls, dict), unify_object)
25       _reify.add((cls, dict), reify_object)
26   
27       return cls

35   def reify_object(o, s):
36       """Reify a Python object with a substitution 
37       >>> # xdoctest: +SKIP 
38       >>> class Foo(object): 
39       ...     def __init__(self, a, b): 
40       ...         self.a = a 
41       ...         self.b = b 
42       ... 
43       ...     def __str__(self): 
44       ...         return "Foo(%s, %s)" % (str(self.a), str(self.b)) 
45       >>> x = var("x") 
46       >>> f = Foo(1, x) 
47       >>> print(f) 
48       Foo(1, ~x) 
49       >>> print(reify_object(f, {x: 2})) 
50       Foo(1, 2) 
51       """
52       if hasattr(o, "__slots__"):
53           return _reify_object_slots(o, s)
54       else:
55           return _reify_object_dict(o, s)
56

58   def _reify_object_dict(o, s):
59       obj = object.__new__(type(o))
60       d = reify(o.__dict__, s)
61       if d == o.__dict__:
62           return o
63       obj.__dict__.update(d)
64       return obj

67   def _reify_object_slots(o, s):
68       attrs = [getattr(o, attr) for attr in o.__slots__]
69       new_attrs = reify(attrs, s)
70       if attrs == new_attrs:
71           return o
72       else:
73           newobj = object.__new__(type(o))
74           for slot, attr in zip(o.__slots__, new_attrs):
75               setattr(newobj, slot, attr)
76           return newobj

80   def _reify(o, s):
81       """Reify a Python ``slice`` object"""
82       return slice(*reify((o.start, o.stop, o.step), s))

90   def unify_object(u, v, s):
91       """Unify two Python objects 
92       Unifies their type and ``__dict__`` attributes 
93       >>> # xdoctest: +SKIP 
94       >>> class Foo(object): 
95       ...     def __init__(self, a, b): 
96       ...         self.a = a 
97       ...         self.b = b 
98       ... 
99       ...     def __str__(self): 
100      ...         return "Foo(%s, %s)" % (str(self.a), str(self.b)) 
101      >>> x = var("x") 
102      >>> f = Foo(1, x) 
103      >>> g = Foo(1, 2) 
104      >>> unify_object(f, g, {}) 
105      {~x: 2} 
106      """
107      if type(u) != type(v):
108          return False
109      if hasattr(u, "__slots__"):
110          return unify(
111              [getattr(u, slot) for slot in u.__slots__],
112              [getattr(v, slot) for slot in v.__slots__],
113              s,
114          )
115      else:
116          return unify(u.__dict__, v.__dict__, s)

119  @dispatch(slice, slice, dict)
120  def _unify(u, v, s):
121      """Unify a Python ``slice`` object"""
122      return unify((u.start, u.stop, u.step), (v.start, v.stop, v.step), s)
