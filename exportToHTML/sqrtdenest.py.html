9    def is_sqrt(expr):
10       """Return True if expr is a sqrt, otherwise False."""
11   
12       return expr.is_Pow and expr.exp.is_Rational and abs(expr.exp) is S.Half

15   def sqrt_depth(p) -> int:
16       """Return the maximum depth of any square root argument of p. 
17    
18       >>> from sympy.functions.elementary.miscellaneous import sqrt 
19       >>> from sympy.simplify.sqrtdenest import sqrt_depth 
20    
21       Neither of these square roots contains any other square roots 
22       so the depth is 1: 
23    
24       >>> sqrt_depth(1 + sqrt(2)*(1 + sqrt(3))) 
25       1 
26    
27       The sqrt(3) is contained within a square root so the depth is 
28       2: 
29    
30       >>> sqrt_depth(1 + sqrt(2)*sqrt(1 + sqrt(3))) 
31       2 
32       """
33       if p is S.ImaginaryUnit:
34           return 1
35       if p.is_Atom:
36           return 0
37       if p.is_Add or p.is_Mul:
38           return max(sqrt_depth(x) for x in p.args)
39       if is_sqrt(p):
40           return sqrt_depth(p.base) + 1
41       return 0

44   def is_algebraic(p):
45       """Return True if p is comprised of only Rationals or square roots 
46       of Rationals and algebraic operations. 
47    
48       Examples 
49       ======== 
50    
51       >>> from sympy.functions.elementary.miscellaneous import sqrt 
52       >>> from sympy.simplify.sqrtdenest import is_algebraic 
53       >>> from sympy import cos 
54       >>> is_algebraic(sqrt(2)*(3/(sqrt(7) + sqrt(5)*sqrt(2)))) 
55       True 
56       >>> is_algebraic(sqrt(2)*(3/(sqrt(7) + sqrt(5)*cos(2)))) 
57       False 
58       """
59   
60       if p.is_Rational:
61           return True
62       elif p.is_Atom:
63           return False
64       elif is_sqrt(p) or p.is_Pow and p.exp.is_Integer:
65           return is_algebraic(p.base)
66       elif p.is_Add or p.is_Mul:
67           return all(is_algebraic(x) for x in p.args)
68       else:
69           return False

72   def _subsets(n):
73       """ 
74       Returns all possible subsets of the set (0, 1, ..., n-1) except the 
75       empty set, listed in reversed lexicographical order according to binary 
76       representation, so that the case of the fourth root is treated last. 
77    
78       Examples 
79       ======== 
80    
81       >>> from sympy.simplify.sqrtdenest import _subsets 
82       >>> _subsets(2) 
83       [[1, 0], [0, 1], [1, 1]] 
84    
85       """
86       if n == 1:
87           a = [[1]]
88       elif n == 2:
89           a = [[1, 0], [0, 1], [1, 1]]
90       elif n == 3:
91           a = [[1, 0, 0], [0, 1, 0], [1, 1, 0],
92                [0, 0, 1], [1, 0, 1], [0, 1, 1], [1, 1, 1]]
93       else:
94           b = _subsets(n - 1)
95           a0 = [x + [0] for x in b]
96           a1 = [x + [1] for x in b]
97           a = a0 + [[0]*(n - 1) + [1]] + a1
98       return a

101  def sqrtdenest(expr, max_iter=3):
102      """Denests sqrts in an expression that contain other square roots 
103      if possible, otherwise returns the expr unchanged. This is based on the 
104      algorithms of [1]. 
105   
106      Examples 
107      ======== 
108   
109      >>> from sympy.simplify.sqrtdenest import sqrtdenest 
110      >>> from sympy import sqrt 
111      >>> sqrtdenest(sqrt(5 + 2 * sqrt(6))) 
112      sqrt(2) + sqrt(3) 
113   
114      See Also 
115      ======== 
116   
117      sympy.solvers.solvers.unrad 
118   
119      References 
120      ========== 
121   
122      .. [1] https://web.archive.org/web/20210806201615/https://researcher.watson.ibm.com/researcher/files/us-fagin/symb85.pdf 
123   
124      .. [2] D. J. Jeffrey and A. D. Rich, 'Symplifying Square Roots of Square Roots 
125             by Denesting' (available at https://www.cybertester.com/data/denest.pdf) 
126   
127      """
128      expr = expand_mul(expr)
129      for i in range(max_iter):
130          z = _sqrtdenest0(expr)
131          if expr == z:
132              return expr
133          expr = z
134      return expr

137  def _sqrt_match(p):
138      """Return [a, b, r] for p.match(a + b*sqrt(r)) where, in addition to 
139      matching, sqrt(r) also has then maximal sqrt_depth among addends of p. 
140   
141      Examples 
142      ======== 
143   
144      >>> from sympy.functions.elementary.miscellaneous import sqrt 
145      >>> from sympy.simplify.sqrtdenest import _sqrt_match 
146      >>> _sqrt_match(1 + sqrt(2) + sqrt(2)*sqrt(3) +  2*sqrt(1+sqrt(5))) 
147      [1 + sqrt(2) + sqrt(6), 2, 1 + sqrt(5)] 
148      """
149      from sympy.simplify.radsimp import split_surds
150  
151      p = _mexpand(p)
152      if p.is_Number:
153          res = (p, S.Zero, S.Zero)
154      elif p.is_Add:
155          pargs = sorted(p.args, key=default_sort_key)
156          sqargs = [x**2 for x in pargs]
157          if all(sq.is_Rational and sq.is_positive for sq in sqargs):
158              r, b, a = split_surds(p)
159              res = a, b, r
160              return list(res)
161          # to make the process canonical, the argument is included in the tuple
162          # so when the max is selected, it will be the largest arg having a
163          # given depth
164          v = [(sqrt_depth(x), x, i) for i, x in enumerate(pargs)]
165          nmax = max(v, key=default_sort_key)
166          if nmax[0] == 0:
167              res = []
168          else:
169              # select r
170              depth, _, i = nmax
171              r = pargs.pop(i)
172              v.pop(i)
173              b = S.One
174              if r.is_Mul:
175                  bv = []
176                  rv = []
177                  for x in r.args:
178                      if sqrt_depth(x) < depth:
179                          bv.append(x)
180                      else:
181                          rv.append(x)
182                  b = Mul._from_args(bv)
183                  r = Mul._from_args(rv)
184              # collect terms containing r
185              a1 = []
186              b1 = [b]
187              for x in v:
188                  if x[0] < depth:
189                      a1.append(x[1])
190                  else:
191                      x1 = x[1]
192                      if x1 == r:
193                          b1.append(1)
194                      else:
195                          if x1.is_Mul:
196                              x1args = list(x1.args)
197                              if r in x1args:
198                                  x1args.remove(r)
199                                  b1.append(Mul(*x1args))
200                              else:
201                                  a1.append(x[1])
202                          else:
203                              a1.append(x[1])
204              a = Add(*a1)
205              b = Add(*b1)
206              res = (a, b, r**2)
207      else:
208          b, r = p.as_coeff_Mul()
209          if is_sqrt(r):
210              res = (S.Zero, b, r**2)
211          else:
212              res = []
213      return list(res)

220  def _sqrtdenest0(expr):
221      """Returns expr after denesting its arguments."""
222  
223      if is_sqrt(expr):
224          n, d = expr.as_numer_denom()
225          if d is S.One:  # n is a square root
226              if n.base.is_Add:
227                  args = sorted(n.base.args, key=default_sort_key)
228                  if len(args) > 2 and all((x**2).is_Integer for x in args):
229                      try:
230                          return _sqrtdenest_rec(n)
231                      except SqrtdenestStopIteration:
232                          pass
233                  expr = sqrt(_mexpand(Add(*[_sqrtdenest0(x) for x in args])))
234              return _sqrtdenest1(expr)
235          else:
236              n, d = [_sqrtdenest0(i) for i in (n, d)]
237              return n/d
238  
239      if isinstance(expr, Add):
240          cs = []
241          args = []
242          for arg in expr.args:
243              c, a = arg.as_coeff_Mul()
244              cs.append(c)
245              args.append(a)
246  
247          if all(c.is_Rational for c in cs) and all(is_sqrt(arg) for arg in args):
248              return _sqrt_ratcomb(cs, args)
249  
250      if isinstance(expr, Expr):
251          args = expr.args
252          if args:
253              return expr.func(*[_sqrtdenest0(a) for a in args])
254      return expr

257  def _sqrtdenest_rec(expr):
258      """Helper that denests the square root of three or more surds. 
259   
260      Explanation 
261      =========== 
262   
263      It returns the denested expression; if it cannot be denested it 
264      throws SqrtdenestStopIteration 
265   
266      Algorithm: expr.base is in the extension Q_m = Q(sqrt(r_1),..,sqrt(r_k)); 
267      split expr.base = a + b*sqrt(r_k), where `a` and `b` are on 
268      Q_(m-1) = Q(sqrt(r_1),..,sqrt(r_(k-1))); then a**2 - b**2*r_k is 
269      on Q_(m-1); denest sqrt(a**2 - b**2*r_k) and so on. 
270      See [1], section 6. 
271   
272      Examples 
273      ======== 
274   
275      >>> from sympy import sqrt 
276      >>> from sympy.simplify.sqrtdenest import _sqrtdenest_rec 
277      >>> _sqrtdenest_rec(sqrt(-72*sqrt(2) + 158*sqrt(5) + 498)) 
278      -sqrt(10) + sqrt(2) + 9 + 9*sqrt(5) 
279      >>> w=-6*sqrt(55)-6*sqrt(35)-2*sqrt(22)-2*sqrt(14)+2*sqrt(77)+6*sqrt(10)+65 
280      >>> _sqrtdenest_rec(sqrt(w)) 
281      -sqrt(11) - sqrt(7) + sqrt(2) + 3*sqrt(5) 
282      """
283      from sympy.simplify.radsimp import radsimp, rad_rationalize, split_surds
284      if not expr.is_Pow:
285          return sqrtdenest(expr)
286      if expr.base < 0:
287          return sqrt(-1)*_sqrtdenest_rec(sqrt(-expr.base))
288      g, a, b = split_surds(expr.base)
289      a = a*sqrt(g)
290      if a < b:
291          a, b = b, a
292      c2 = _mexpand(a**2 - b**2)
293      if len(c2.args) > 2:
294          g, a1, b1 = split_surds(c2)
295          a1 = a1*sqrt(g)
296          if a1 < b1:
297              a1, b1 = b1, a1
298          c2_1 = _mexpand(a1**2 - b1**2)
299          c_1 = _sqrtdenest_rec(sqrt(c2_1))
300          d_1 = _sqrtdenest_rec(sqrt(a1 + c_1))
301          num, den = rad_rationalize(b1, d_1)
302          c = _mexpand(d_1/sqrt(2) + num/(den*sqrt(2)))
303      else:
304          c = _sqrtdenest1(sqrt(c2))
305  
306      if sqrt_depth(c) > 1:
307          raise SqrtdenestStopIteration
308      ac = a + c
309      if len(ac.args) >= len(expr.args):
310          if count_ops(ac) >= count_ops(expr.base):
311              raise SqrtdenestStopIteration
312      d = sqrtdenest(sqrt(ac))
313      if sqrt_depth(d) > 1:
314          raise SqrtdenestStopIteration
315      num, den = rad_rationalize(b, d)
316      r = d/sqrt(2) + num/(den*sqrt(2))
317      r = radsimp(r)
318      return _mexpand(r)

321  def _sqrtdenest1(expr, denester=True):
322      """Return denested expr after denesting with simpler methods or, that 
323      failing, using the denester."""
324  
325      from sympy.simplify.simplify import radsimp
326  
327      if not is_sqrt(expr):
328          return expr
329  
330      a = expr.base
331      if a.is_Atom:
332          return expr
333      val = _sqrt_match(a)
334      if not val:
335          return expr
336  
337      a, b, r = val
338      # try a quick numeric denesting
339      d2 = _mexpand(a**2 - b**2*r)
340      if d2.is_Rational:
341          if d2.is_positive:
342              z = _sqrt_numeric_denest(a, b, r, d2)
343              if z is not None:
344                  return z
345          else:
346              # fourth root case
347              # sqrtdenest(sqrt(3 + 2*sqrt(3))) =
348              # sqrt(2)*3**(1/4)/2 + sqrt(2)*3**(3/4)/2
349              dr2 = _mexpand(-d2*r)
350              dr = sqrt(dr2)
351              if dr.is_Rational:
352                  z = _sqrt_numeric_denest(_mexpand(b*r), a, r, dr2)
353                  if z is not None:
354                      return z/root(r, 4)
355  
356      else:
357          z = _sqrt_symbolic_denest(a, b, r)
358          if z is not None:
359              return z
360  
361      if not denester or not is_algebraic(expr):
362          return expr
363  
364      res = sqrt_biquadratic_denest(expr, a, b, r, d2)
365      if res:
366          return res
367  
368      # now call to the denester
369      av0 = [a, b, r, d2]
370      z = _denester([radsimp(expr**2)], av0, 0, sqrt_depth(expr))[0]
371      if av0[1] is None:
372          return expr
373      if z is not None:
374          if sqrt_depth(z) == sqrt_depth(expr) and count_ops(z) > count_ops(expr):
375              return expr
376          return z
377      return expr

380  def _sqrt_symbolic_denest(a, b, r):
381      """Given an expression, sqrt(a + b*sqrt(b)), return the denested 
382      expression or None. 
383   
384      Explanation 
385      =========== 
386   
387      If r = ra + rb*sqrt(rr), try replacing sqrt(rr) in ``a`` with 
388      (y**2 - ra)/rb, and if the result is a quadratic, ca*y**2 + cb*y + cc, and 
389      (cb + b)**2 - 4*ca*cc is 0, then sqrt(a + b*sqrt(r)) can be rewritten as 
390      sqrt(ca*(sqrt(r) + (cb + b)/(2*ca))**2). 
391   
392      Examples 
393      ======== 
394   
395      >>> from sympy.simplify.sqrtdenest import _sqrt_symbolic_denest, sqrtdenest 
396      >>> from sympy import sqrt, Symbol 
397      >>> from sympy.abc import x 
398   
399      >>> a, b, r = 16 - 2*sqrt(29), 2, -10*sqrt(29) + 55 
400      >>> _sqrt_symbolic_denest(a, b, r) 
401      sqrt(11 - 2*sqrt(29)) + sqrt(5) 
402   
403      If the expression is numeric, it will be simplified: 
404   
405      >>> w = sqrt(sqrt(sqrt(3) + 1) + 1) + 1 + sqrt(2) 
406      >>> sqrtdenest(sqrt((w**2).expand())) 
407      1 + sqrt(2) + sqrt(1 + sqrt(1 + sqrt(3))) 
408   
409      Otherwise, it will only be simplified if assumptions allow: 
410   
411      >>> w = w.subs(sqrt(3), sqrt(x + 3)) 
412      >>> sqrtdenest(sqrt((w**2).expand())) 
413      sqrt((sqrt(sqrt(sqrt(x + 3) + 1) + 1) + 1 + sqrt(2))**2) 
414   
415      Notice that the argument of the sqrt is a square. If x is made positive 
416      then the sqrt of the square is resolved: 
417   
418      >>> _.subs(x, Symbol('x', positive=True)) 
419      sqrt(sqrt(sqrt(x + 3) + 1) + 1) + 1 + sqrt(2) 
420      """
421  
422      a, b, r = map(sympify, (a, b, r))
423      rval = _sqrt_match(r)
424      if not rval:
425          return None
426      ra, rb, rr = rval
427      if rb:
428          y = Dummy('y', positive=True)
429          try:
430              newa = Poly(a.subs(sqrt(rr), (y**2 - ra)/rb), y)
431          except PolynomialError:
432              return None
433          if newa.degree() == 2:
434              ca, cb, cc = newa.all_coeffs()
435              cb += b
436              if _mexpand(cb**2 - 4*ca*cc).equals(0):
437                  z = sqrt(ca*(sqrt(r) + cb/(2*ca))**2)
438                  if z.is_number:
439                      z = _mexpand(Mul._from_args(z.as_content_primitive()))
440                  return z

443  def _sqrt_numeric_denest(a, b, r, d2):
444      r"""Helper that denest 
445      $\sqrt{a + b \sqrt{r}}, d^2 = a^2 - b^2 r > 0$ 
446   
447      If it cannot be denested, it returns ``None``. 
448      """
449      d = sqrt(d2)
450      s = a + d
451      # sqrt_depth(res) <= sqrt_depth(s) + 1
452      # sqrt_depth(expr) = sqrt_depth(r) + 2
453      # there is denesting if sqrt_depth(s) + 1 < sqrt_depth(r) + 2
454      # if s**2 is Number there is a fourth root
455      if sqrt_depth(s) < sqrt_depth(r) + 1 or (s**2).is_Rational:
456          s1, s2 = sign(s), sign(b)
457          if s1 == s2 == -1:
458              s1 = s2 = 1
459          res = (s1 * sqrt(a + d) + s2 * sqrt(a - d)) * sqrt(2) / 2
460          return res.expand()

463  def sqrt_biquadratic_denest(expr, a, b, r, d2):
464      """denest expr = sqrt(a + b*sqrt(r)) 
465      where a, b, r are linear combinations of square roots of 
466      positive rationals on the rationals (SQRR) and r > 0, b != 0, 
467      d2 = a**2 - b**2*r > 0 
468   
469      If it cannot denest it returns None. 
470   
471      Explanation 
472      =========== 
473   
474      Search for a solution A of type SQRR of the biquadratic equation 
475      4*A**4 - 4*a*A**2 + b**2*r = 0                               (1) 
476      sqd = sqrt(a**2 - b**2*r) 
477      Choosing the sqrt to be positive, the possible solutions are 
478      A = sqrt(a/2 +/- sqd/2) 
479      Since a, b, r are SQRR, then a**2 - b**2*r is a SQRR, 
480      so if sqd can be denested, it is done by 
481      _sqrtdenest_rec, and the result is a SQRR. 
482      Similarly for A. 
483      Examples of solutions (in both cases a and sqd are positive): 
484   
485        Example of expr with solution sqrt(a/2 + sqd/2) but not 
486        solution sqrt(a/2 - sqd/2): 
487        expr = sqrt(-sqrt(15) - sqrt(2)*sqrt(-sqrt(5) + 5) - sqrt(3) + 8) 
488        a = -sqrt(15) - sqrt(3) + 8; sqd = -2*sqrt(5) - 2 + 4*sqrt(3) 
489   
490        Example of expr with solution sqrt(a/2 - sqd/2) but not 
491        solution sqrt(a/2 + sqd/2): 
492        w = 2 + r2 + r3 + (1 + r3)*sqrt(2 + r2 + 5*r3) 
493        expr = sqrt((w**2).expand()) 
494        a = 4*sqrt(6) + 8*sqrt(2) + 47 + 28*sqrt(3) 
495        sqd = 29 + 20*sqrt(3) 
496   
497      Define B = b/2*A; eq.(1) implies a = A**2 + B**2*r; then 
498      expr**2 = a + b*sqrt(r) = (A + B*sqrt(r))**2 
499   
500      Examples 
501      ======== 
502   
503      >>> from sympy import sqrt 
504      >>> from sympy.simplify.sqrtdenest import _sqrt_match, sqrt_biquadratic_denest 
505      >>> z = sqrt((2*sqrt(2) + 4)*sqrt(2 + sqrt(2)) + 5*sqrt(2) + 8) 
506      >>> a, b, r = _sqrt_match(z**2) 
507      >>> d2 = a**2 - b**2*r 
508      >>> sqrt_biquadratic_denest(z, a, b, r, d2) 
509      sqrt(2) + sqrt(sqrt(2) + 2) + 2 
510      """
511      from sympy.simplify.radsimp import radsimp, rad_rationalize
512      if r <= 0 or d2 < 0 or not b or sqrt_depth(expr.base) < 2:
513          return None
514      for x in (a, b, r):
515          for y in x.args:
516              y2 = y**2
517              if not y2.is_Integer or not y2.is_positive:
518                  return None
519      sqd = _mexpand(sqrtdenest(sqrt(radsimp(d2))))
520      if sqrt_depth(sqd) > 1:
521          return None
522      x1, x2 = [a/2 + sqd/2, a/2 - sqd/2]
523      # look for a solution A with depth 1
524      for x in (x1, x2):
525          A = sqrtdenest(sqrt(x))
526          if sqrt_depth(A) > 1:
527              continue
528          Bn, Bd = rad_rationalize(b, _mexpand(2*A))
529          B = Bn/Bd
530          z = A + B*sqrt(r)
531          if z < 0:
532              z = -z
533          return _mexpand(z)
534      return None

537  def _denester(nested, av0, h, max_depth_level):
538      """Denests a list of expressions that contain nested square roots. 
539   
540      Explanation 
541      =========== 
542   
543      Algorithm based on <http://www.almaden.ibm.com/cs/people/fagin/symb85.pdf>. 
544   
545      It is assumed that all of the elements of 'nested' share the same 
546      bottom-level radicand. (This is stated in the paper, on page 177, in 
547      the paragraph immediately preceding the algorithm.) 
548   
549      When evaluating all of the arguments in parallel, the bottom-level 
550      radicand only needs to be denested once. This means that calling 
551      _denester with x arguments results in a recursive invocation with x+1 
552      arguments; hence _denester has polynomial complexity. 
553   
554      However, if the arguments were evaluated separately, each call would 
555      result in two recursive invocations, and the algorithm would have 
556      exponential complexity. 
557   
558      This is discussed in the paper in the middle paragraph of page 179. 
559      """
560      from sympy.simplify.simplify import radsimp
561      if h > max_depth_level:
562          return None, None
563      if av0[1] is None:
564          return None, None
565      if (av0[0] is None and
566              all(n.is_Number for n in nested)):  # no arguments are nested
567          for f in _subsets(len(nested)):  # test subset 'f' of nested
568              p = _mexpand(Mul(*[nested[i] for i in range(len(f)) if f[i]]))
569              if f.count(1) > 1 and f[-1]:
570                  p = -p
571              sqp = sqrt(p)
572              if sqp.is_Rational:
573                  return sqp, f  # got a perfect square so return its square root.
574          # Otherwise, return the radicand from the previous invocation.
575          return sqrt(nested[-1]), [0]*len(nested)
576      else:
577          R = None
578          if av0[0] is not None:
579              values = [av0[:2]]
580              R = av0[2]
581              nested2 = [av0[3], R]
582              av0[0] = None
583          else:
584              values = list(filter(None, [_sqrt_match(expr) for expr in nested]))
585              for v in values:
586                  if v[2]:  # Since if b=0, r is not defined
587                      if R is not None:
588                          if R != v[2]:
589                              av0[1] = None
590                              return None, None
591                      else:
592                          R = v[2]
593              if R is None:
594                  # return the radicand from the previous invocation
595                  return sqrt(nested[-1]), [0]*len(nested)
596              nested2 = [_mexpand(v[0]**2) -
597                         _mexpand(R*v[1]**2) for v in values] + [R]
598          d, f = _denester(nested2, av0, h + 1, max_depth_level)
599          if not f:
600              return None, None
601          if not any(f[i] for i in range(len(nested))):
602              v = values[-1]
603              return sqrt(v[0] + _mexpand(v[1]*d)), f
604          else:
605              p = Mul(*[nested[i] for i in range(len(nested)) if f[i]])
606              v = _sqrt_match(p)
607              if 1 in f and f.index(1) < len(nested) - 1 and f[len(nested) - 1]:
608                  v[0] = -v[0]
609                  v[1] = -v[1]
610              if not f[len(nested)]:  # Solution denests with square roots
611                  vad = _mexpand(v[0] + d)
612                  if vad <= 0:

637  def _sqrt_ratcomb(cs, args):
638      """Denest rational combinations of radicals. 
639   
640      Based on section 5 of [1]. 
641   
642      Examples 
643      ======== 
644   
645      >>> from sympy import sqrt 
646      >>> from sympy.simplify.sqrtdenest import sqrtdenest 
647      >>> z = sqrt(1+sqrt(3)) + sqrt(3+3*sqrt(3)) - sqrt(10+6*sqrt(3)) 
648      >>> sqrtdenest(z) 
649      0 
650      """
651      from sympy.simplify.radsimp import radsimp
652  
653      # check if there exists a pair of sqrt that can be denested
654      def find(a):
655          n = len(a)
656          for i in range(n - 1):
657              for j in range(i + 1, n):
658                  s1 = a[i].base
659                  s2 = a[j].base
660                  p = _mexpand(s1 * s2)
661                  s = sqrtdenest(sqrt(p))
662                  if s != sqrt(p):
663                      return s, i, j
664  
665      indices = find(args)
666      if indices is None:
667          return Add(*[c * arg for c, arg in zip(cs, args)])
668  
669      s, i1, i2 = indices
670  
671      c2 = cs.pop(i2)
672      args.pop(i2)
673      a1 = args[i1]
674  
675      # replace a2 by s/a1
676      cs[i1] += radsimp(c2 * s / a1.base)
677  
678      return _sqrt_ratcomb(cs, args)
