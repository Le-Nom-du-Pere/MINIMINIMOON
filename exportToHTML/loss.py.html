62   class L1Loss(_Loss):
63       r"""Creates a criterion that measures the mean absolute error (MAE) between each element in 
64       the input :math:`x` and target :math:`y`. 
65    
66       The unreduced (i.e. with :attr:`reduction` set to ``'none'``) loss can be described as: 
67    
68       .. math:: 
69           \ell(x, y) = L = \{l_1,\dots,l_N\}^\top, \quad 
70           l_n = \left| x_n - y_n \right|, 
71    
72       where :math:`N` is the batch size. If :attr:`reduction` is not ``'none'`` 
73       (default ``'mean'``), then: 
74    
75       .. math:: 
76           \ell(x, y) = 
77           \begin{cases} 
78               \operatorname{mean}(L), & \text{if reduction} = \text{`mean';}\\ 
79               \operatorname{sum}(L),  & \text{if reduction} = \text{`sum'.} 
80           \end{cases} 
81    
82       :math:`x` and :math:`y` are tensors of arbitrary shapes with a total 
83       of :math:`N` elements each. 
84    
85       The sum operation still operates over all the elements, and divides by :math:`N`. 
86    
87       The division by :math:`N` can be avoided if one sets ``reduction = 'sum'``. 
88    
89       Supports real-valued and complex-valued inputs. 
90    
91       Args: 
92           size_average (bool, optional): Deprecated (see :attr:`reduction`). By default, 
93               the losses are averaged over each loss element in the batch. Note that for 
94               some losses, there are multiple elements per sample. If the field :attr:`size_average` 
95               is set to ``False``, the losses are instead summed for each minibatch. Ignored 
96               when :attr:`reduce` is ``False``. Default: ``True`` 
97           reduce (bool, optional): Deprecated (see :attr:`reduction`). By default, the 
98               losses are averaged or summed over observations for each minibatch depending 
99               on :attr:`size_average`. When :attr:`reduce` is ``False``, returns a loss per 
100              batch element instead and ignores :attr:`size_average`. Default: ``True`` 
101          reduction (str, optional): Specifies the reduction to apply to the output: 
102              ``'none'`` | ``'mean'`` | ``'sum'``. ``'none'``: no reduction will be applied, 
103              ``'mean'``: the sum of the output will be divided by the number of 
104              elements in the output, ``'sum'``: the output will be summed. Note: :attr:`size_average` 
105              and :attr:`reduce` are in the process of being deprecated, and in the meantime, 
106              specifying either of those two args will override :attr:`reduction`. Default: ``'mean'`` 
107   
108      Shape: 
109          - Input: :math:`(*)`, where :math:`*` means any number of dimensions. 
110          - Target: :math:`(*)`, same shape as the input. 
111          - Output: scalar. If :attr:`reduction` is ``'none'``, then 
112            :math:`(*)`, same shape as the input. 
113   
114      Examples: 
115   
116          >>> loss = nn.L1Loss() 
117          >>> input = torch.randn(3, 5, requires_grad=True) 
118          >>> target = torch.randn(3, 5) 
119          >>> output = loss(input, target) 
120          >>> output.backward() 
121      """
122  
123      __constants__ = ["reduction"]
124  
125      def __init__(self, size_average=None, reduce=None, reduction: str = "mean") -> None:
126          super().__init__(size_average, reduce, reduction)
127  
128      def forward(self, input: Tensor, target: Tensor) -> Tensor:
129          return F.l1_loss(input, target, reduction=self.reduction)

132  class NLLLoss(_WeightedLoss):
133      r"""The negative log likelihood loss. It is useful to train a classification 
134      problem with `C` classes. 
135   
136      If provided, the optional argument :attr:`weight` should be a 1D Tensor assigning 
137      weight to each of the classes. This is particularly useful when you have an 
138      unbalanced training set. 
139   
140      The `input` given through a forward call is expected to contain 
141      log-probabilities of each class. `input` has to be a Tensor of size either 
142      :math:`(minibatch, C)` or :math:`(minibatch, C, d_1, d_2, ..., d_K)` 
143      with :math:`K \geq 1` for the `K`-dimensional case. The latter is useful for 
144      higher dimension inputs, such as computing NLL loss per-pixel for 2D images. 
145   
146      Obtaining log-probabilities in a neural network is easily achieved by 
147      adding a  `LogSoftmax`  layer in the last layer of your network. 
148      You may use `CrossEntropyLoss` instead, if you prefer not to add an extra 
149      layer. 
150   
151      The `target` that this loss expects should be a class index in the range :math:`[0, C-1]` 
152      where `C = number of classes`; if `ignore_index` is specified, this loss also accepts 
153      this class index (this index may not necessarily be in the class range). 
154   
155      The unreduced (i.e. with :attr:`reduction` set to ``'none'``) loss can be described as: 
156   
157      .. math:: 
158          \ell(x, y) = L = \{l_1,\dots,l_N\}^\top, \\ 
159          l_n = - w_{y_n} x_{n,y_n}, \\ 
160          w_{c} = \text{weight}[c] \cdot \mathbb{1}\{c \not= \text{ignore\_index}\}, 
161   
162      where :math:`x` is the input, :math:`y` is the target, :math:`w` is the weight, and 
163      :math:`N` is the batch size. If :attr:`reduction` is not ``'none'`` 
164      (default ``'mean'``), then 
165   
166      .. math:: 
167          \ell(x, y) = \begin{cases} 
168              \sum_{n=1}^N \frac{1}{\sum_{n=1}^N w_{y_n}} l_n, & 
169              \text{if reduction} = \text{`mean';}\\ 
170              \sum_{n=1}^N l_n,  & 
171              \text{if reduction} = \text{`sum'.} 
172          \end{cases} 
173   
174      Args: 
175          weight (Tensor, optional): a manual rescaling weight given to each 
176              class. If given, it has to be a Tensor of size `C`. Otherwise, it is 
177              treated as if having all ones. 
178          size_average (bool, optional): Deprecated (see :attr:`reduction`). By default, 
179              the losses are averaged over each loss element in the batch. Note that for 
180              some losses, there are multiple elements per sample. If the field :attr:`size_average` 
181              is set to ``False``, the losses are instead summed for each minibatch. Ignored 
182              when :attr:`reduce` is ``False``. Default: ``None`` 
183          ignore_index (int, optional): Specifies a target value that is ignored 
184              and does not contribute to the input gradient. When 
185              :attr:`size_average` is ``True``, the loss is averaged over 
186              non-ignored targets. 
187          reduce (bool, optional): Deprecated (see :attr:`reduction`). By default, the 
188              losses are averaged or summed over observations for each minibatch depending 
189              on :attr:`size_average`. When :attr:`reduce` is ``False``, returns a loss per 
190              batch element instead and ignores :attr:`size_average`. Default: ``None`` 
191          reduction (str, optional): Specifies the reduction to apply to the output: 
192              ``'none'`` | ``'mean'`` | ``'sum'``. ``'none'``: no reduction will 
193              be applied, ``'mean'``: the weighted mean of the output is taken, 
194              ``'sum'``: the output will be summed. Note: :attr:`size_average` 
195              and :attr:`reduce` are in the process of being deprecated, and in 
196              the meantime, specifying either of those two args will override 
197              :attr:`reduction`. Default: ``'mean'`` 
198   
199      Shape:: 
200          - Input: :math:`(N, C)` or :math:`(C)`, where `C = number of classes`, `N = batch size`, or 
201            :math:`(N, C, d_1, d_2, ..., d_K)` with :math:`K \geq 1` 
202            in the case of `K`-dimensional loss. 
203          - Target: :math:`(N)` or :math:`()`, where each value is 
204            :math:`0 \leq \text{targets}[i] \leq C-1`, or 
205            :math:`(N, d_1, d_2, ..., d_K)` with :math:`K \geq 1` in the case of 
206            K-dimensional loss. 
207          - Output: If :attr:`reduction` is ``'none'``, shape :math:`(N)` or 
208            :math:`(N, d_1, d_2, ..., d_K)` with :math:`K \geq 1` in the case of K-dimensional loss. 
209            Otherwise, scalar. 
210   
211      Examples: 
212   
213          >>> log_softmax = nn.LogSoftmax(dim=1) 
214          >>> loss_fn = nn.NLLLoss() 
215          >>> # input to NLLLoss is of size N x C = 3 x 5 
216          >>> input = torch.randn(3, 5, requires_grad=True) 
217          >>> # each element in target must have 0 <= value < C 
218          >>> target = torch.tensor([1, 0, 4]) 
219          >>> loss = loss_fn(log_softmax(input), target) 
220          >>> loss.backward() 
221          >>> 
222          >>> 
223          >>> # 2D loss example (used, for example, with image inputs) 
224          >>> N, C = 5, 4 
225          >>> loss_fn = nn.NLLLoss() 
226          >>> data = torch.randn(N, 16, 10, 10) 
227          >>> conv = nn.Conv2d(16, C, (3, 3)) 
228          >>> log_softmax = nn.LogSoftmax(dim=1) 
229          >>> # output of conv forward is of shape [N, C, 8, 8] 
230          >>> output = log_softmax(conv(data)) 
231          >>> # each element in target must have 0 <= value < C 
232          >>> target = torch.empty(N, 8, 8, dtype=torch.long).random_(0, C) 
233          >>> # input to NLLLoss is of size N x C x height (8) x width (8) 
234          >>> loss = loss_fn(output, target) 
235          >>> loss.backward() 
236      """
237  
238      __constants__ = ["ignore_index", "reduction"]
239      ignore_index: int
240  
241      def __init__(
242          self,
243          weight: Optional[Tensor] = None,
244          size_average=None,
245          ignore_index: int = -100,
246          reduce=None,
247          reduction: str = "mean",
248      ) -> None:
249          super().__init__(weight, size_average, reduce, reduction)
250          self.ignore_index = ignore_index
251  
252      def forward(self, input: Tensor, target: Tensor) -> Tensor:
253          return F.nll_loss(
254              input,
255              target,
256              weight=self.weight,
257              ignore_index=self.ignore_index,
258              reduction=self.reduction,
259          )

280  class PoissonNLLLoss(_Loss):
281      r"""Negative log likelihood loss with Poisson distribution of target. 
282   
283      The loss can be described as: 
284   
285      .. math:: 
286          \text{target} \sim \mathrm{Poisson}(\text{input}) 
287   
288          \text{loss}(\text{input}, \text{target}) = \text{input} - \text{target} * \log(\text{input}) 
289                                      + \log(\text{target!}) 
290   
291      The last term can be omitted or approximated with Stirling formula. The 
292      approximation is used for target values more than 1. For targets less or 
293      equal to 1 zeros are added to the loss. 
294   
295      Args: 
296          log_input (bool, optional): if ``True`` the loss is computed as 
297              :math:`\exp(\text{input}) - \text{target}*\text{input}`, if ``False`` the loss is 
298              :math:`\text{input} - \text{target}*\log(\text{input}+\text{eps})`. 
299          full (bool, optional): whether to compute full loss, i. e. to add the 
300              Stirling approximation term 
301   
302              .. math:: 
303                  \text{target}*\log(\text{target}) - \text{target} + 0.5 * \log(2\pi\text{target}). 
304          size_average (bool, optional): Deprecated (see :attr:`reduction`). By default, 
305              the losses are averaged over each loss element in the batch. Note that for 
306              some losses, there are multiple elements per sample. If the field :attr:`size_average` 
307              is set to ``False``, the losses are instead summed for each minibatch. Ignored 
308              when :attr:`reduce` is ``False``. Default: ``True`` 
309          eps (float, optional): Small value to avoid evaluation of :math:`\log(0)` when 
310              :attr:`log_input = False`. Default: 1e-8 
311          reduce (bool, optional): Deprecated (see :attr:`reduction`). By default, the 
312              losses are averaged or summed over observations for each minibatch depending 
313              on :attr:`size_average`. When :attr:`reduce` is ``False``, returns a loss per 
314              batch element instead and ignores :attr:`size_average`. Default: ``True`` 
315          reduction (str, optional): Specifies the reduction to apply to the output: 
316              ``'none'`` | ``'mean'`` | ``'sum'``. ``'none'``: no reduction will be applied, 
317              ``'mean'``: the sum of the output will be divided by the number of 
318              elements in the output, ``'sum'``: the output will be summed. Note: :attr:`size_average` 
319              and :attr:`reduce` are in the process of being deprecated, and in the meantime, 
320              specifying either of those two args will override :attr:`reduction`. Default: ``'mean'`` 
321   
322      Examples: 
323   
324          >>> loss = nn.PoissonNLLLoss() 
325          >>> log_input = torch.randn(5, 2, requires_grad=True) 
326          >>> target = torch.randn(5, 2) 
327          >>> output = loss(log_input, target) 
328          >>> output.backward() 
329   
330      Shape: 
331          - Input: :math:`(*)`, where :math:`*` means any number of dimensions. 
332          - Target: :math:`(*)`, same shape as the input. 
333          - Output: scalar by default. If :attr:`reduction` is ``'none'``, then :math:`(*)`, 
334            the same shape as the input. 
335      """
336  
337      __constants__ = ["log_input", "full", "eps", "reduction"]
338      log_input: bool
339      full: bool
340      eps: float
341  
342      def __init__(
343          self,
344          log_input: bool = True,
345          full: bool = False,
346          size_average=None,
347          eps: float = 1e-8,
348          reduce=None,
349          reduction: str = "mean",
350      ) -> None:
351          super().__init__(size_average, reduce, reduction)
352          self.log_input = log_input
353          self.full = full
354          self.eps = eps
355  
356      def forward(self, log_input: Tensor, target: Tensor) -> Tensor:
357          return F.poisson_nll_loss(
358              log_input,
359              target,
360              log_input=self.log_input,
361              full=self.full,
362              eps=self.eps,
363              reduction=self.reduction,
364          )

367  class GaussianNLLLoss(_Loss):
368      r"""Gaussian negative log likelihood loss. 
369   
370      The targets are treated as samples from Gaussian distributions with 
371      expectations and variances predicted by the neural network. For a 
372      ``target`` tensor modelled as having Gaussian distribution with a tensor 
373      of expectations ``input`` and a tensor of positive variances ``var`` the loss is: 
374   
375      .. math:: 
376          \text{loss} = \frac{1}{2}\left(\log\left(\text{max}\left(\text{var}, 
377          \ \text{eps}\right)\right) + \frac{\left(\text{input} - \text{target}\right)^2} 
378          {\text{max}\left(\text{var}, \ \text{eps}\right)}\right) + \text{const.} 
379   
380      where :attr:`eps` is used for stability. By default, the constant term of 
381      the loss function is omitted unless :attr:`full` is ``True``. If ``var`` is not the same 
382      size as ``input`` (due to a homoscedastic assumption), it must either have a final dimension 
383      of 1 or have one fewer dimension (with all other sizes being the same) for correct broadcasting. 
384   
385      Args: 
386          full (bool, optional): include the constant term in the loss 
387              calculation. Default: ``False``. 
388          eps (float, optional): value used to clamp ``var`` (see note below), for 
389              stability. Default: 1e-6. 
390          reduction (str, optional): specifies the reduction to apply to the 
391              output:``'none'`` | ``'mean'`` | ``'sum'``. ``'none'``: no reduction 
392              will be applied, ``'mean'``: the output is the average of all batch 
393              member losses, ``'sum'``: the output is the sum of all batch member 
394              losses. Default: ``'mean'``. 
395   
396      Shape: 
397          - Input: :math:`(N, *)` or :math:`(*)` where :math:`*` means any number of additional 
398            dimensions 
399          - Target: :math:`(N, *)` or :math:`(*)`, same shape as the input, or same shape as the input 
400            but with one dimension equal to 1 (to allow for broadcasting) 
401          - Var: :math:`(N, *)` or :math:`(*)`, same shape as the input, or same shape as the input but 
402            with one dimension equal to 1, or same shape as the input but with one fewer 
403            dimension (to allow for broadcasting), or a scalar value 
404          - Output: scalar if :attr:`reduction` is ``'mean'`` (default) or 
405            ``'sum'``. If :attr:`reduction` is ``'none'``, then :math:`(N, *)`, same 
406            shape as the input 
407   
408      Examples: 
409          >>> loss = nn.GaussianNLLLoss() 
410          >>> input = torch.randn(5, 2, requires_grad=True) 
411          >>> target = torch.randn(5, 2) 
412          >>> var = torch.ones(5, 2, requires_grad=True)  # heteroscedastic 
413          >>> output = loss(input, target, var) 
414          >>> output.backward() 
415   
416          >>> loss = nn.GaussianNLLLoss() 
417          >>> input = torch.randn(5, 2, requires_grad=True) 
418          >>> target = torch.randn(5, 2) 
419          >>> var = torch.ones(5, 1, requires_grad=True)  # homoscedastic 
420          >>> output = loss(input, target, var) 
421          >>> output.backward() 
422   
423      Note: 
424          The clamping of ``var`` is ignored with respect to autograd, and so the 
425          gradients are unaffected by it. 
426   
427      Reference: 
428          Nix, D. A. and Weigend, A. S., "Estimating the mean and variance of the 
429          target probability distribution", Proceedings of 1994 IEEE International 
430          Conference on Neural Networks (ICNN'94), Orlando, FL, USA, 1994, pp. 55-60 
431          vol.1, doi: 10.1109/ICNN.1994.374138. 
432      """
433  
434      __constants__ = ["full", "eps", "reduction"]
435      full: bool
436      eps: float
437  
438      def __init__(
439          self, *, full: bool = False, eps: float = 1e-6, reduction: str = "mean"
440      ) -> None:
441          super().__init__(None, None, reduction)
442          self.full = full
443          self.eps = eps
444  
445      def forward(
446          self, input: Tensor, target: Tensor, var: Union[Tensor, float]
447      ) -> Tensor:
448          return F.gaussian_nll_loss(
449              input, target, var, full=self.full, eps=self.eps, reduction=self.reduction
450          )

453  class KLDivLoss(_Loss):
454      r"""The Kullback-Leibler divergence loss. 
455   
456      For tensors of the same shape :math:`y_{\text{pred}},\ y_{\text{true}}`, 
457      where :math:`y_{\text{pred}}` is the :attr:`input` and :math:`y_{\text{true}}` is the 
458      :attr:`target`, we define the **pointwise KL-divergence** as 
459   
460      .. math:: 
461   
462          L(y_{\text{pred}},\ y_{\text{true}}) 
463              = y_{\text{true}} \cdot \log \frac{y_{\text{true}}}{y_{\text{pred}}} 
464              = y_{\text{true}} \cdot (\log y_{\text{true}} - \log y_{\text{pred}}) 
465   
466      To avoid underflow issues when computing this quantity, this loss expects the argument 
467      :attr:`input` in the log-space. The argument :attr:`target` may also be provided in the 
468      log-space if :attr:`log_target`\ `= True`. 
469   
470      To summarise, this function is roughly equivalent to computing 
471   
472      .. code-block:: python 
473   
474          if not log_target:  # default 
475              loss_pointwise = target * (target.log() - input) 
476          else: 
477              loss_pointwise = target.exp() * (target - input) 
478   
479      and then reducing this result depending on the argument :attr:`reduction` as 
480   
481      .. code-block:: python 
482   
483          if reduction == "mean":  # default 
484              loss = loss_pointwise.mean() 
485          elif reduction == "batchmean":  # mathematically correct 
486              loss = loss_pointwise.sum() / input.size(0) 
487          elif reduction == "sum": 
488              loss = loss_pointwise.sum() 
489          else:  # reduction == "none" 
490              loss = loss_pointwise 
491   
492      .. note:: 
493          As all the other losses in PyTorch, this function expects the first argument, 
494          :attr:`input`, to be the output of the model (e.g. the neural network) 
495          and the second, :attr:`target`, to be the observations in the dataset. 
496          This differs from the standard mathematical notation :math:`KL(P\ ||\ Q)` where 
497          :math:`P` denotes the distribution of the observations and :math:`Q` denotes the model. 
498   
499      .. warning:: 
500          :attr:`reduction`\ `= "mean"` doesn't return the true KL divergence value, please use 
501          :attr:`reduction`\ `= "batchmean"` which aligns with the mathematical definition. 
502   
503      Args: 
504          size_average (bool, optional): Deprecated (see :attr:`reduction`). By default, 
505              the losses are averaged over each loss element in the batch. Note that for 
506              some losses, there are multiple elements per sample. If the field :attr:`size_average` 
507              is set to `False`, the losses are instead summed for each minibatch. Ignored 
508              when :attr:`reduce` is `False`. Default: `True` 
509          reduce (bool, optional): Deprecated (see :attr:`reduction`). By default, the 
510              losses are averaged or summed over observations for each minibatch depending 
511              on :attr:`size_average`. When :attr:`reduce` is `False`, returns a loss per 
512              batch element instead and ignores :attr:`size_average`. Default: `True` 
513          reduction (str, optional): Specifies the reduction to apply to the output. Default: `"mean"` 
514          log_target (bool, optional): Specifies whether `target` is the log space. Default: `False` 
515   
516      Shape: 
517          - Input: :math:`(*)`, where :math:`*` means any number of dimensions. 
518          - Target: :math:`(*)`, same shape as the input. 
519          - Output: scalar by default. If :attr:`reduction` is `'none'`, then :math:`(*)`, 
520            same shape as the input. 
521   
522      Examples: 
523          >>> kl_loss = nn.KLDivLoss(reduction="batchmean") 
524          >>> # input should be a distribution in the log space 
525          >>> input = F.log_softmax(torch.randn(3, 5, requires_grad=True), dim=1) 
526          >>> # Sample a batch of distributions. Usually this would come from the dataset 
527          >>> target = F.softmax(torch.rand(3, 5), dim=1) 
528          >>> output = kl_loss(input, target) 
529          >>> 
530          >>> kl_loss = nn.KLDivLoss(reduction="batchmean", log_target=True) 
531          >>> log_target = F.log_softmax(torch.rand(3, 5), dim=1) 
532          >>> output = kl_loss(input, log_target) 
533      """
534  
535      __constants__ = ["reduction"]
536  
537      def __init__(
538          self,
539          size_average=None,
540          reduce=None,
541          reduction: str = "mean",
542          log_target: bool = False,
543      ) -> None:
544          super().__init__(size_average, reduce, reduction)
545          self.log_target = log_target
546  
547      def forward(self, input: Tensor, target: Tensor) -> Tensor:
548          return F.kl_div(
549              input, target, reduction=self.reduction, log_target=self.log_target
550          )

553  class MSELoss(_Loss):
554      r"""Creates a criterion that measures the mean squared error (squared L2 norm) between 
555      each element in the input :math:`x` and target :math:`y`. 
556   
557      The unreduced (i.e. with :attr:`reduction` set to ``'none'``) loss can be described as: 
558

567          \ell(x, y) = 
568          \begin{cases}

570              \operatorname{sum}(L),  &  \text{if reduction} = \text{`sum'.}

598          - Input: :math:`(*)`, where :math:`*` means any number of dimensions.

619  class BCELoss(_WeightedLoss):
620      r"""Creates a criterion that measures the Binary Cross Entropy between the target and 
621      the input probabilities: 
622   
623      The unreduced (i.e. with :attr:`reduction` set to ``'none'``) loss can be described as: 
624   
625      .. math:: 
626          \ell(x, y) = L = \{l_1,\dots,l_N\}^\top, \quad 
627          l_n = - w_n \left[ y_n \cdot \log x_n + (1 - y_n) \cdot \log (1 - x_n) \right], 
628   
629      where :math:`N` is the batch size. If :attr:`reduction` is not ``'none'`` 
630      (default ``'mean'``), then 
631   
632      .. math:: 
633          \ell(x, y) = \begin{cases} 
634              \operatorname{mean}(L), & \text{if reduction} = \text{`mean';}\\ 
635              \operatorname{sum}(L),  & \text{if reduction} = \text{`sum'.} 
636          \end{cases} 
637   
638      This is used for measuring the error of a reconstruction in for example 
639      an auto-encoder. Note that the targets :math:`y` should be numbers 
640      between 0 and 1. 
641   
642      Notice that if :math:`x_n` is either 0 or 1, one of the log terms would be 
643      mathematically undefined in the above loss equation. PyTorch chooses to set 
644      :math:`\log (0) = -\infty`, since :math:`\lim_{x\to 0} \log (x) = -\infty`. 
645      However, an infinite term in the loss equation is not desirable for several reasons. 
646   
647      For one, if either :math:`y_n = 0` or :math:`(1 - y_n) = 0`, then we would be 
648      multiplying 0 with infinity. Secondly, if we have an infinite loss value, then 
649      we would also have an infinite term in our gradient, since 
650      :math:`\lim_{x\to 0} \frac{d}{dx} \log (x) = \infty`. 
651      This would make BCELoss's backward method nonlinear with respect to :math:`x_n`, 
652      and using it for things like linear regression would not be straight-forward. 
653   
654      Our solution is that BCELoss clamps its log function outputs to be greater than 
655      or equal to -100. This way, we can always have a finite loss value and a linear 
656      backward method. 
657   
658   
659      Args: 
660          weight (Tensor, optional): a manual rescaling weight given to the loss 
661              of each batch element. If given, has to be a Tensor of size `nbatch`. 
662          size_average (bool, optional): Deprecated (see :attr:`reduction`). By default, 
663              the losses are averaged over each loss element in the batch. Note that for 
664              some losses, there are multiple elements per sample. If the field :attr:`size_average` 
665              is set to ``False``, the losses are instead summed for each minibatch. Ignored 
666              when :attr:`reduce` is ``False``. Default: ``True`` 
667          reduce (bool, optional): Deprecated (see :attr:`reduction`). By default, the 
668              losses are averaged or summed over observations for each minibatch depending 
669              on :attr:`size_average`. When :attr:`reduce` is ``False``, returns a loss per 
670              batch element instead and ignores :attr:`size_average`. Default: ``True`` 
671          reduction (str, optional): Specifies the reduction to apply to the output: 
672              ``'none'`` | ``'mean'`` | ``'sum'``. ``'none'``: no reduction will be applied, 
673              ``'mean'``: the sum of the output will be divided by the number of 
674              elements in the output, ``'sum'``: the output will be summed. Note: :attr:`size_average` 
675              and :attr:`reduce` are in the process of being deprecated, and in the meantime, 
676              specifying either of those two args will override :attr:`reduction`. Default: ``'mean'`` 
677   
678      Shape: 
679          - Input: :math:`(*)`, where :math:`*` means any number of dimensions. 
680          - Target: :math:`(*)`, same shape as the input. 
681          - Output: scalar. If :attr:`reduction` is ``'none'``, then :math:`(*)`, same 
682            shape as input. 
683   
684      Examples: 
685   
686          >>> m = nn.Sigmoid() 
687          >>> loss = nn.BCELoss() 
688          >>> input = torch.randn(3, 2, requires_grad=True) 
689          >>> target = torch.rand(3, 2, requires_grad=False) 
690          >>> output = loss(m(input), target) 
691          >>> output.backward() 
692      """
693  
694      __constants__ = ["reduction"]
695  
696      def __init__(
697          self,
698          weight: Optional[Tensor] = None,
699          size_average=None,
700          reduce=None,
701          reduction: str = "mean",
702      ) -> None:
703          super().__init__(weight, size_average, reduce, reduction)
704  
705      def forward(self, input: Tensor, target: Tensor) -> Tensor:
706          return F.binary_cross_entropy(
707              input, target, weight=self.weight, reduction=self.reduction

711  class BCEWithLogitsLoss(_Loss):
712      r"""This loss combines a `Sigmoid` layer and the `BCELoss` in one single 
713      class. This version is more numerically stable than using a plain `Sigmoid` 
714      followed by a `BCELoss` as, by combining the operations into one layer, 
715      we take advantage of the log-sum-exp trick for numerical stability. 
716   
717      The unreduced (i.e. with :attr:`reduction` set to ``'none'``) loss can be described as: 
718   
719      .. math:: 
720          \ell(x, y) = L = \{l_1,\dots,l_N\}^\top, \quad 
721          l_n = - w_n \left[ y_n \cdot \log \sigma(x_n) 
722          + (1 - y_n) \cdot \log (1 - \sigma(x_n)) \right], 
723   
724      where :math:`N` is the batch size. If :attr:`reduction` is not ``'none'`` 
725      (default ``'mean'``), then 
726   
727      .. math:: 
728          \ell(x, y) = \begin{cases} 
729              \operatorname{mean}(L), & \text{if reduction} = \text{`mean';}\\ 
730              \operatorname{sum}(L),  & \text{if reduction} = \text{`sum'.} 
731          \end{cases} 
732   
733      This is used for measuring the error of a reconstruction in for example 
734      an auto-encoder. Note that the targets `t[i]` should be numbers 
735      between 0 and 1. 
736   
737      It's possible to trade off recall and precision by adding weights to positive examples. 
738      In the case of multi-label classification the loss can be described as: 
739   
740      .. math:: 
741          \ell_c(x, y) = L_c = \{l_{1,c},\dots,l_{N,c}\}^\top, \quad 
742          l_{n,c} = - w_{n,c} \left[ p_c y_{n,c} \cdot \log \sigma(x_{n,c}) 
743          + (1 - y_{n,c}) \cdot \log (1 - \sigma(x_{n,c})) \right], 
744   
745      where :math:`c` is the class number (:math:`c > 1` for multi-label binary classification, 
746      :math:`c = 1` for single-label binary classification), 
747      :math:`n` is the number of the sample in the batch and 
748      :math:`p_c` is the weight of the positive answer for the class :math:`c`. 
749   
750      :math:`p_c > 1` increases the recall, :math:`p_c < 1` increases the precision. 
751   
752      For example, if a dataset contains 100 positive and 300 negative examples of a single class, 
753      then ``pos_weight`` for the class should be equal to :math:`\frac{300}{100}=3`. 
754      The loss would act as if the dataset contains :math:`3\times 100=300` positive examples. 
755   
756      Examples: 
757   
758          >>> target = torch.ones([10, 64], dtype=torch.float32)  # 64 classes, batch size = 10 
759          >>> output = torch.full([10, 64], 1.5)  # A prediction (logit) 
760          >>> pos_weight = torch.ones([64])  # All weights are equal to 1 
761          >>> criterion = torch.nn.BCEWithLogitsLoss(pos_weight=pos_weight) 
762          >>> criterion(output, target)  # -log(sigmoid(1.5)) 
763          tensor(0.20...) 
764   
765      In the above example, the ``pos_weight`` tensor's elements correspond to the 64 distinct classes 
766      in a multi-label binary classification scenario. Each element in ``pos_weight`` is designed to adjust the 
767      loss function based on the imbalance between negative and positive samples for the respective class. 
768      This approach is useful in datasets with varying levels of class imbalance, ensuring that the loss 
769      calculation accurately accounts for the distribution in each class. 
770   
771      Args: 
772          weight (Tensor, optional): a manual rescaling weight given to the loss 
773              of each batch element. If given, has to be a Tensor of size `nbatch`. 
774          size_average (bool, optional): Deprecated (see :attr:`reduction`). By default, 
775              the losses are averaged over each loss element in the batch. Note that for 
776              some losses, there are multiple elements per sample. If the field :attr:`size_average` 
777              is set to ``False``, the losses are instead summed for each minibatch. Ignored 
778              when :attr:`reduce` is ``False``. Default: ``True`` 
779          reduce (bool, optional): Deprecated (see :attr:`reduction`). By default, the 
780              losses are averaged or summed over observations for each minibatch depending 
781              on :attr:`size_average`. When :attr:`reduce` is ``False``, returns a loss per 
782              batch element instead and ignores :attr:`size_average`. Default: ``True`` 
783          reduction (str, optional): Specifies the reduction to apply to the output: 
784              ``'none'`` | ``'mean'`` | ``'sum'``. ``'none'``: no reduction will be applied, 
785              ``'mean'``: the sum of the output will be divided by the number of 
786              elements in the output, ``'sum'``: the output will be summed. Note: :attr:`size_average` 
787              and :attr:`reduce` are in the process of being deprecated, and in the meantime, 
788              specifying either of those two args will override :attr:`reduction`. Default: ``'mean'`` 
789          pos_weight (Tensor, optional): a weight of positive examples to be broadcasted with target. 
790              Must be a tensor with equal size along the class dimension to the number of classes. 
791              Pay close attention to PyTorch's broadcasting semantics in order to achieve the desired 
792              operations. For a target of size [B, C, H, W] (where B is batch size) pos_weight of 
793              size [B, C, H, W] will apply different pos_weights to each element of the batch or 
794              [C, H, W] the same pos_weights across the batch. To apply the same positive weight 
795              along all spacial dimensions for a 2D multi-class target [C, H, W] use: [C, 1, 1]. 
796              Default: ``None`` 
797   
798      Shape: 
799          - Input: :math:`(*)`, where :math:`*` means any number of dimensions. 
800          - Target: :math:`(*)`, same shape as the input. 
801          - Output: scalar. If :attr:`reduction` is ``'none'``, then :math:`(*)`, same 
802            shape as input. 
803   
804      Examples: 
805   
806          >>> loss = nn.BCEWithLogitsLoss() 
807          >>> input = torch.randn(3, requires_grad=True) 
808          >>> target = torch.empty(3).random_(2) 
809          >>> output = loss(input, target) 
810          >>> output.backward() 
811      """
812  
813      def __init__(
814          self,
815          weight: Optional[Tensor] = None,
816          size_average=None,
817          reduce=None,
818          reduction: str = "mean",
819          pos_weight: Optional[Tensor] = None,
820      ) -> None:
821          super().__init__(size_average, reduce, reduction)
822          self.register_buffer("weight", weight)
823          self.register_buffer("pos_weight", pos_weight)
824          self.weight: Optional[Tensor]
825          self.pos_weight: Optional[Tensor]
826  
827      def forward(self, input: Tensor, target: Tensor) -> Tensor:
828          return F.binary_cross_entropy_with_logits(
829              input,
830              target,
831              self.weight,
832              pos_weight=self.pos_weight,
833              reduction=self.reduction,
834          )

971  class SmoothL1Loss(_Loss):
972      r"""Creates a criterion that uses a squared term if the absolute 
973      element-wise error falls below beta and an L1 term otherwise. 
974      It is less sensitive to outliers than :class:`torch.nn.MSELoss` and in some cases 
975      prevents exploding gradients (e.g. see the paper `Fast R-CNN`_ by Ross Girshick). 
976   
977      For a batch of size :math:`N`, the unreduced loss can be described as: 
978   
979      .. math:: 
980          \ell(x, y) = L = \{l_1, ..., l_N\}^T 
981   
982      with 
983   
984      .. math:: 
985          l_n = \begin{cases} 
986          0.5 (x_n - y_n)^2 / beta, & \text{if } |x_n - y_n| < beta \\ 
987          |x_n - y_n| - 0.5 * beta, & \text{otherwise } 
988          \end{cases} 
989   
990      If `reduction` is not `none`, then: 
991   
992      .. math:: 
993          \ell(x, y) = 
994          \begin{cases} 
995              \operatorname{mean}(L), &  \text{if reduction} = \text{`mean';}\\ 
996              \operatorname{sum}(L),  &  \text{if reduction} = \text{`sum'.} 
997          \end{cases} 
998   
999      .. note:: 
1000         Smooth L1 loss can be seen as exactly :class:`L1Loss`, but with the :math:`|x - y| < beta` 
1001         portion replaced with a quadratic function such that its slope is 1 at :math:`|x - y| = beta`. 
1002         The quadratic segment smooths the L1 loss near :math:`|x - y| = 0`. 
1003  
1004     .. note:: 
1005         Smooth L1 loss is closely related to :class:`HuberLoss`, being 
1006         equivalent to :math:`huber(x, y) / beta` (note that Smooth L1's beta hyper-parameter is 
1007         also known as delta for Huber). This leads to the following differences: 
1008  
1009         * As beta -> 0, Smooth L1 loss converges to :class:`L1Loss`, while :class:`HuberLoss` 
1010           converges to a constant 0 loss. When beta is 0, Smooth L1 loss is equivalent to L1 loss. 
1011         * As beta -> :math:`+\infty`, Smooth L1 loss converges to a constant 0 loss, while 
1012           :class:`HuberLoss` converges to :class:`MSELoss`. 
1013         * For Smooth L1 loss, as beta varies, the L1 segment of the loss has a constant slope of 1. 
1014           For :class:`HuberLoss`, the slope of the L1 segment is beta.

1016     .. _`Fast R-CNN`: https://arxiv.org/abs/1504.08083

1027             batch element instead and ignores :attr:`size_average`. Default: ``True``

1029             ``'none'`` | ``'mean'`` | ``'sum'``. ``'none'``: no reduction will be applied, 
1030             ``'mean'``: the sum of the output will be divided by the number of 
1031             elements in the output, ``'sum'``: the output will be summed. Note: :attr:`size_average` 
1032             and :attr:`reduce` are in the process of being deprecated, and in the meantime, 
1033             specifying either of those two args will override :attr:`reduction`. Default: ``'mean'`` 
1034         beta (float, optional): Specifies the threshold at which to change between L1 and L2 loss. 
1035             The value must be non-negative. Default: 1.0 
1036  
1037     Shape: 
1038         - Input: :math:`(*)`, where :math:`*` means any number of dimensions. 
1039         - Target: :math:`(*)`, same shape as the input. 
1040         - Output: scalar. If :attr:`reduction` is ``'none'``, then :math:`(*)`, same shape as the input.

1055 class HuberLoss(_Loss):
1056     r"""Creates a criterion that uses a squared term if the absolute 
1057     element-wise error falls below delta and a delta-scaled L1 term otherwise. 
1058     This loss combines advantages of both :class:`L1Loss` and :class:`MSELoss`; the 
1059     delta-scaled L1 region makes the loss less sensitive to outliers than :class:`MSELoss`, 
1060     while the L2 region provides smoothness over :class:`L1Loss` near 0. See 
1061     `Huber loss <https://en.wikipedia.org/wiki/Huber_loss>`_ for more information. 
1062  
1063     For a batch of size :math:`N`, the unreduced loss can be described as: 
1064  
1065     .. math:: 
1066         \ell(x, y) = L = \{l_1, ..., l_N\}^T 
1067  
1068     with 
1069  
1070     .. math:: 
1071         l_n = \begin{cases} 
1072         0.5 (x_n - y_n)^2, & \text{if } |x_n - y_n| < delta \\ 
1073         delta * (|x_n - y_n| - 0.5 * delta), & \text{otherwise } 
1074         \end{cases} 
1075  
1076     If `reduction` is not `none`, then: 
1077  
1078     .. math:: 
1079         \ell(x, y) = 
1080         \begin{cases} 
1081             \operatorname{mean}(L), &  \text{if reduction} = \text{`mean';}\\ 
1082             \operatorname{sum}(L),  &  \text{if reduction} = \text{`sum'.} 
1083         \end{cases} 
1084  
1085     .. note:: 
1086         When delta is set to 1, this loss is equivalent to :class:`SmoothL1Loss`. 
1087         In general, this loss differs from :class:`SmoothL1Loss` by a factor of delta (AKA beta 
1088         in Smooth L1). 
1089         See :class:`SmoothL1Loss` for additional discussion on the differences in behavior 
1090         between the two losses. 
1091  
1092     Args: 
1093         reduction (str, optional): Specifies the reduction to apply to the output: 
1094             ``'none'`` | ``'mean'`` | ``'sum'``. ``'none'``: no reduction will be applied, 
1095             ``'mean'``: the sum of the output will be divided by the number of 
1096             elements in the output, ``'sum'``: the output will be summed. Default: ``'mean'`` 
1097         delta (float, optional): Specifies the threshold at which to change between delta-scaled L1 and L2 loss. 
1098             The value must be positive.  Default: 1.0 
1099  
1100     Shape: 
1101         - Input: :math:`(*)` where :math:`*` means any number of dimensions. 
1102         - Target: :math:`(*)`, same shape as the input. 
1103         - Output: scalar. If :attr:`reduction` is ``'none'``, then :math:`(*)`, same shape as the input. 
1104     """
1105 
1106     __constants__ = ["reduction", "delta"]
1107 
1108     def __init__(self, reduction: str = "mean", delta: float = 1.0) -> None:
1109         super().__init__(reduction=reduction)
1110         self.delta = delta
1111 
1112     def forward(self, input: Tensor, target: Tensor) -> Tensor:
1113         return F.huber_loss(input, target, reduction=self.reduction, delta=self.delta)

1158 class CrossEntropyLoss(_WeightedLoss):
1159     r"""This criterion computes the cross entropy loss between input logits 
1160     and target. 
1161  
1162     It is useful when training a classification problem with `C` classes. 
1163     If provided, the optional argument :attr:`weight` should be a 1D `Tensor` 
1164     assigning weight to each of the classes. 
1165     This is particularly useful when you have an unbalanced training set. 
1166  
1167     The `input` is expected to contain the unnormalized logits for each class (which do `not` need 
1168     to be positive or sum to 1, in general). 
1169     `input` has to be a Tensor of size :math:`(C)` for unbatched input, 
1170     :math:`(minibatch, C)` or :math:`(minibatch, C, d_1, d_2, ..., d_K)` with :math:`K \geq 1` for the 
1171     `K`-dimensional case. The last being useful for higher dimension inputs, such 
1172     as computing cross entropy loss per-pixel for 2D images. 
1173  
1174     The `target` that this criterion expects should contain either: 
1175  
1176     - Class indices in the range :math:`[0, C)` where :math:`C` is the number of classes; if 
1177       `ignore_index` is specified, this loss also accepts this class index (this index 
1178       may not necessarily be in the class range). The unreduced (i.e. with :attr:`reduction` 
1179       set to ``'none'``) loss for this case can be described as: 
1180  
1181       .. math:: 
1182           \ell(x, y) = L = \{l_1,\dots,l_N\}^\top, \quad 
1183           l_n = - w_{y_n} \log \frac{\exp(x_{n,y_n})}{\sum_{c=1}^C \exp(x_{n,c})} 
1184           \cdot \mathbb{1}\{y_n \not= \text{ignore\_index}\} 
1185  
1186       where :math:`x` is the input, :math:`y` is the target, :math:`w` is the weight, 
1187       :math:`C` is the number of classes, and :math:`N` spans the minibatch dimension as well as 
1188       :math:`d_1, ..., d_k` for the `K`-dimensional case. If 
1189       :attr:`reduction` is not ``'none'`` (default ``'mean'``), then 
1190  
1191       .. math:: 
1192           \ell(x, y) = \begin{cases} 
1193               \sum_{n=1}^N \frac{1}{\sum_{n=1}^N w_{y_n} \cdot \mathbb{1}\{y_n \not= \text{ignore\_index}\}} l_n, & 
1194                \text{if reduction} = \text{`mean';}\\ 
1195                 \sum_{n=1}^N l_n,  & 
1196                 \text{if reduction} = \text{`sum'.} 
1197             \end{cases} 
1198  
1199       Note that this case is equivalent to applying :class:`~torch.nn.LogSoftmax` 
1200       on an input, followed by :class:`~torch.nn.NLLLoss`. 
1201  
1202     - Probabilities for each class; useful when labels beyond a single class per minibatch item 
1203       are required, such as for blended labels, label smoothing, etc. The unreduced (i.e. with 
1204       :attr:`reduction` set to ``'none'``) loss for this case can be described as: 
1205  
1206       .. math:: 
1207           \ell(x, y) = L = \{l_1,\dots,l_N\}^\top, \quad 
1208           l_n = - \sum_{c=1}^C w_c \log \frac{\exp(x_{n,c})}{\sum_{i=1}^C \exp(x_{n,i})} y_{n,c} 
1209  
1210       where :math:`x` is the input, :math:`y` is the target, :math:`w` is the weight, 
1211       :math:`C` is the number of classes, and :math:`N` spans the minibatch dimension as well as 
1212       :math:`d_1, ..., d_k` for the `K`-dimensional case. If 
1213       :attr:`reduction` is not ``'none'`` (default ``'mean'``), then 
1214  
1215       .. math:: 
1216           \ell(x, y) = \begin{cases} 
1217               \frac{\sum_{n=1}^N l_n}{N}, & 
1218                \text{if reduction} = \text{`mean';}\\ 
1219                 \sum_{n=1}^N l_n,  & 
1220                 \text{if reduction} = \text{`sum'.} 
1221             \end{cases} 
1222  
1223     .. note:: 
1224         The performance of this criterion is generally better when `target` contains class 
1225         indices, as this allows for optimized computation. Consider providing `target` as 
1226         class probabilities only when a single class label per minibatch item is too restrictive. 
1227  
1228     Args: 
1229         weight (Tensor, optional): a manual rescaling weight given to each class. 
1230             If given, has to be a Tensor of size `C`. 
1231         size_average (bool, optional): Deprecated (see :attr:`reduction`). By default, 
1232             the losses are averaged over each loss element in the batch. Note that for 
1233             some losses, there are multiple elements per sample. If the field :attr:`size_average` 
1234             is set to ``False``, the losses are instead summed for each minibatch. Ignored 
1235             when :attr:`reduce` is ``False``. Default: ``True`` 
1236         ignore_index (int, optional): Specifies a target value that is ignored 
1237             and does not contribute to the input gradient. When :attr:`size_average` is 
1238             ``True``, the loss is averaged over non-ignored targets. Note that 
1239             :attr:`ignore_index` is only applicable when the target contains class indices. 
1240         reduce (bool, optional): Deprecated (see :attr:`reduction`). By default, the

1256     Shape: 
1257         - Input: Shape :math:`(C)`, :math:`(N, C)` or :math:`(N, C, d_1, d_2, ..., d_K)` with :math:`K \geq 1` 
1258           in the case of `K`-dimensional loss.

1263           data type is required to be float when using class probabilities. 
1264         - Output: If reduction is 'none', shape :math:`()`, :math:`(N)` or :math:`(N, d_1, d_2, ..., d_K)` with :math:`K \geq 1`

1266  
1267  
1268         where: 
1269  
1270         .. math:: 
1271             \begin{aligned} 
1272                 C ={} & \text{number of classes} \\ 
1273                 N ={} & \text{batch size} \\ 
1274             \end{aligned} 
1275  
1276     Examples: 
1277  
1278         >>> # Example of target with class indices

1516     output :math:`y` (which is a 1D tensor of target class indices, 
1517     :math:`0 \leq y \leq \text{x.size}(1)-1`): 
1518  
1519     For each mini-batch sample, the loss in terms of the 1D input :math:`x` and scalar 
1520     output :math:`y` is: 
1521  
1522     .. math:: 
1523         \text{loss}(x, y) = \frac{\sum_i \max(0, \text{margin} - x[y] + x[i])^p}{\text{x.size}(0)} 
1524  
1525     where :math:`i \in \left\{0, \; \cdots , \; \text{x.size}(0) - 1\right\}` 
1526     and :math:`i \neq y`. 
1527  
1528     Optionally, you can give non-equal weighting on the classes by passing 
1529     a 1D :attr:`weight` tensor into the constructor. 
1530  
1531     The loss function then becomes: 
1532  
1533     .. math:: 
1534         \text{loss}(x, y) = \frac{\sum_i w[y] * \max(0, \text{margin} - x[y] + x[i])^p}{\text{x.size}(0)} 
1535  
1536     Args: 
1537         p (int, optional): Has a default value of :math:`1`. :math:`1` and :math:`2` 
1538             are the only supported values. 
1539         margin (float, optional): Has a default value of :math:`1`. 
1540         weight (Tensor, optional): a manual rescaling weight given to each 
1541             class. If given, it has to be a Tensor of size `C`. Otherwise, it is 
1542             treated as if having all ones. 
1543         size_average (bool, optional): Deprecated (see :attr:`reduction`). By default, 
1544             the losses are averaged over each loss element in the batch. Note that for 
1545             some losses, there are multiple elements per sample. If the field :attr:`size_average` 
1546             is set to ``False``, the losses are instead summed for each minibatch. Ignored 
1547             when :attr:`reduce` is ``False``. Default: ``True`` 
1548         reduce (bool, optional): Deprecated (see :attr:`reduction`). By default, the 
1549             losses are averaged or summed over observations for each minibatch depending 
1550             on :attr:`size_average`. When :attr:`reduce` is ``False``, returns a loss per 
1551             batch element instead and ignores :attr:`size_average`. Default: ``True`` 
1552         reduction (str, optional): Specifies the reduction to apply to the output: 
1553             ``'none'`` | ``'mean'`` | ``'sum'``. ``'none'``: no reduction will be applied, 
1554             ``'mean'``: the sum of the output will be divided by the number of 
1555             elements in the output, ``'sum'``: the output will be summed. Note: :attr:`size_average` 
1556             and :attr:`reduce` are in the process of being deprecated, and in the meantime, 
1557             specifying either of those two args will override :attr:`reduction`. Default: ``'mean'`` 
1558  
1559     Shape: 
1560         - Input: :math:`(N, C)` or :math:`(C)`, where :math:`N` is the batch size and :math:`C` is the number of classes. 
1561         - Target: :math:`(N)` or :math:`()`, where each value is :math:`0 \leq \text{targets}[i] \leq C-1`. 
1562         - Output: scalar. If :attr:`reduction` is ``'none'``, then same shape as the target. 
1563  
1564     Examples: 
1565  
1566         >>> loss = nn.MultiMarginLoss() 
1567         >>> x = torch.tensor([[0.1, 0.2, 0.4, 0.8]]) 
1568         >>> y = torch.tensor([3]) 
1569         >>> # 0.25 * ((1-(0.8-0.1)) + (1-(0.8-0.2)) + (1-(0.8-0.4))) 
1570         >>> loss(x, y) 
1571         tensor(0.32...) 
1572     """
1573 
1574     __constants__ = ["p", "margin", "reduction"]
1575     margin: float
1576     p: int
1577 
1578     def __init__(
1579         self,
1580         p: int = 1,
1581         margin: float = 1.0,
1582         weight: Optional[Tensor] = None,
1583         size_average=None,
1584         reduce=None,
1585         reduction: str = "mean",
1586     ) -> None:
1587         super().__init__(weight, size_average, reduce, reduction)
1588         if p != 1 and p != 2:
1589             raise ValueError("only p == 1 and p == 2 supported")
1590         if weight is not None and weight.dim() != 1:
1591             raise ValueError(
1592                 f"MultiMarginLoss: expected weight to be None or 1D tensor, got {weight.dim()}D instead"
1593             )
1594         self.p = p
1595         self.margin = margin
1596 
1597     def forward(self, input: Tensor, target: Tensor) -> Tensor:
1598         return F.multi_margin_loss(
1599             input,
1600             target,
1601             p=self.p,
1602             margin=self.margin,
1603             weight=self.weight,
1604             reduction=self.reduction,
1605         )

1608 class TripletMarginLoss(_Loss):
1609     r"""Creates a criterion that measures the triplet loss given an input 
1610     tensors :math:`x1`, :math:`x2`, :math:`x3` and a margin with a value greater than :math:`0`. 
1611     This is used for measuring a relative similarity between samples. A triplet 
1612     is composed by `a`, `p` and `n` (i.e., `anchor`, `positive examples` and `negative 
1613     examples` respectively). The shapes of all input tensors should be 
1614     :math:`(N, D)`. 
1615  
1616     The distance swap is described in detail in the paper `Learning shallow 
1617     convolutional feature descriptors with triplet losses`_ by 
1618     V. Balntas, E. Riba et al. 
1619  
1620     The loss function for each sample in the mini-batch is: 
1621  
1622     .. math:: 
1623         L(a, p, n) = \max \{d(a_i, p_i) - d(a_i, n_i) + {\rm margin}, 0\} 
1624  
1625  
1626     where 
1627  
1628     .. math:: 
1629         d(x_i, y_i) = \left\lVert {\bf x}_i - {\bf y}_i \right\rVert_p 
1630  
1631     The norm is calculated using the specified p value and a small constant :math:`\varepsilon` is 
1632     added for numerical stability. 
1633  
1634     See also :class:`~torch.nn.TripletMarginWithDistanceLoss`, which computes the 
1635     triplet margin loss for input tensors using a custom distance function. 
1636  
1637     Args: 
1638         margin (float, optional): Default: :math:`1`. 
1639         p (int, optional): The norm degree for pairwise distance. Default: :math:`2`. 
1640         eps (float, optional): Small constant for numerical stability. Default: :math:`1e-6`. 
1641         swap (bool, optional): The distance swap is described in detail in the paper 
1642             `Learning shallow convolutional feature descriptors with triplet losses` by 
1643             V. Balntas, E. Riba et al. Default: ``False``. 
1644         size_average (bool, optional): Deprecated (see :attr:`reduction`). By default, 
1645             the losses are averaged over each loss element in the batch. Note that for 
1646             some losses, there are multiple elements per sample. If the field :attr:`size_average` 
1647             is set to ``False``, the losses are instead summed for each minibatch. Ignored 
1648             when :attr:`reduce` is ``False``. Default: ``True`` 
1649         reduce (bool, optional): Deprecated (see :attr:`reduction`). By default, the 
1650             losses are averaged or summed over observations for each minibatch depending 
1651             on :attr:`size_average`. When :attr:`reduce` is ``False``, returns a loss per 
1652             batch element instead and ignores :attr:`size_average`. Default: ``True`` 
1653         reduction (str, optional): Specifies the reduction to apply to the output: 
1654             ``'none'`` | ``'mean'`` | ``'sum'``. ``'none'``: no reduction will be applied, 
1655             ``'mean'``: the sum of the output will be divided by the number of 
1656             elements in the output, ``'sum'``: the output will be summed. Note: :attr:`size_average` 
1657             and :attr:`reduce` are in the process of being deprecated, and in the meantime, 
1658             specifying either of those two args will override :attr:`reduction`. Default: ``'mean'`` 
1659  
1660     Shape: 
1661         - Input: :math:`(N, D)` or :math:`(D)` where :math:`D` is the vector dimension. 
1662         - Output: A Tensor of shape :math:`(N)` if :attr:`reduction` is ``'none'`` and 
1663           input shape is :math:`(N, D)`; a scalar otherwise. 
1664  
1665     Examples: 
1666  
1667     >>> triplet_loss = nn.TripletMarginLoss(margin=1.0, p=2, eps=1e-7) 
1668     >>> anchor = torch.randn(100, 128, requires_grad=True) 
1669     >>> positive = torch.randn(100, 128, requires_grad=True) 
1670     >>> negative = torch.randn(100, 128, requires_grad=True) 
1671     >>> output = triplet_loss(anchor, positive, negative) 
1672     >>> output.backward() 
1673  
1674     .. _Learning shallow convolutional feature descriptors with triplet losses: 
1675         https://bmva-archive.org.uk/bmvc/2016/papers/paper119/index.html 
1676     """
1677 
1678     __constants__ = ["margin", "p", "eps", "swap", "reduction"]
1679     margin: float
1680     p: float
1681     eps: float
1682     swap: bool
1683 
1684     def __init__(
1685         self,
1686         margin: float = 1.0,
1687         p: float = 2.0,
1688         eps: float = 1e-6,
1689         swap: bool = False,
1690         size_average=None,
1691         reduce=None,
1692         reduction: str = "mean",
1693     ):
1694         super().__init__(size_average, reduce, reduction)
1695         if margin <= 0:
1696             raise ValueError(
1697                 f"TripletMarginLoss: expected margin to be greater than 0, got {margin} instead"
1698             )
1699         self.margin = margin
1700         self.p = p
1701         self.eps = eps
1702         self.swap = swap
1703 
1704     def forward(self, anchor: Tensor, positive: Tensor, negative: Tensor) -> Tensor:
1705         return F.triplet_margin_loss(
1706             anchor,

1710             p=self.p,

1717 class TripletMarginWithDistanceLoss(_Loss):
1718     r"""Creates a criterion that measures the triplet loss given input 
1719     tensors :math:`a`, :math:`p`, and :math:`n` (representing anchor, 
1720     positive, and negative examples, respectively), and a nonnegative, 
1721     real-valued function ("distance function") used to compute the relationship 
1722     between the anchor and positive example ("positive distance") and the 
1723     anchor and negative example ("negative distance"). 
1724  
1725     The unreduced loss (i.e., with :attr:`reduction` set to ``'none'``) 
1726     can be described as: 
1727  
1728     .. math:: 
1729         \ell(a, p, n) = L = \{l_1,\dots,l_N\}^\top, \quad 
1730         l_i = \max \{d(a_i, p_i) - d(a_i, n_i) + {\rm margin}, 0\} 
1731  
1732     where :math:`N` is the batch size; :math:`d` is a nonnegative, real-valued function 
1733     quantifying the closeness of two tensors, referred to as the :attr:`distance_function`; 
1734     and :math:`margin` is a nonnegative margin representing the minimum difference 
1735     between the positive and negative distances that is required for the loss to 
1736     be 0.  The input tensors have :math:`N` elements each and can be of any shape 
1737     that the distance function can handle. 
1738  
1739     If :attr:`reduction` is not ``'none'`` 
1740     (default ``'mean'``), then: 
1741  
1742     .. math:: 
1743         \ell(x, y) = 
1744         \begin{cases} 
1745             \operatorname{mean}(L), &  \text{if reduction} = \text{`mean';}\\ 
1746             \operatorname{sum}(L),  &  \text{if reduction} = \text{`sum'.} 
1747         \end{cases} 
1748  
1749     See also :class:`~torch.nn.TripletMarginLoss`, which computes the triplet 
1750     loss for input tensors using the :math:`l_p` distance as the distance function. 
1751  
1752     Args: 
1753         distance_function (Callable, optional): A nonnegative, real-valued function that 
1754             quantifies the closeness of two tensors. If not specified, 
1755             `nn.PairwiseDistance` will be used.  Default: ``None`` 
1756         margin (float, optional): A nonnegative margin representing the minimum difference 
1757             between the positive and negative distances required for the loss to be 0. Larger 
1758             margins penalize cases where the negative examples are not distant enough from the 
1759             anchors, relative to the positives. Default: :math:`1`. 
1760         swap (bool, optional): Whether to use the distance swap described in the paper 
1761             `Learning shallow convolutional feature descriptors with triplet losses` by 
1762             V. Balntas, E. Riba et al. If True, and if the positive example is closer to the 
1763             negative example than the anchor is, swaps the positive example and the anchor in 
1764             the loss computation. Default: ``False``. 
1765         reduction (str, optional): Specifies the (optional) reduction to apply to the output: 
1766             ``'none'`` | ``'mean'`` | ``'sum'``. ``'none'``: no reduction will be applied, 
1767             ``'mean'``: the sum of the output will be divided by the number of 
1768             elements in the output, ``'sum'``: the output will be summed. Default: ``'mean'`` 
1769  
1770  
1771     Shape: 
1772         - Input: :math:`(N, *)` where :math:`*` represents any number of additional dimensions 
1773           as supported by the distance function. 
1774         - Output: A Tensor of shape :math:`(N)` if :attr:`reduction` is ``'none'``, or a scalar 
1775           otherwise. 
1776  
1777     Examples: 
1778  
1779     >>> # Initialize embeddings 
1780     >>> embedding = nn.Embedding(1000, 128) 
1781     >>> anchor_ids = torch.randint(0, 1000, (1,)) 
1782     >>> positive_ids = torch.randint(0, 1000, (1,)) 
1783     >>> negative_ids = torch.randint(0, 1000, (1,)) 
1784     >>> anchor = embedding(anchor_ids) 
1785     >>> positive = embedding(positive_ids) 
1786     >>> negative = embedding(negative_ids) 
1787     >>> 
1788     >>> # Built-in Distance Function 
1789     >>> triplet_loss = \ 
1790     >>>     nn.TripletMarginWithDistanceLoss(distance_function=nn.PairwiseDistance()) 
1791     >>> output = triplet_loss(anchor, positive, negative) 
1792     >>> output.backward() 
1793     >>> 
1794     >>> # Custom Distance Function 
1795     >>> def l_infinity(x1, x2): 
1796     >>>     return torch.max(torch.abs(x1 - x2), dim=1).values 
1797     >>> 
1798     >>> # xdoctest: +SKIP("FIXME: Would call backwards a second time") 
1799     >>> triplet_loss = ( 
1800     >>>     nn.TripletMarginWithDistanceLoss(distance_function=l_infinity, margin=1.5)) 
1801     >>> output = triplet_loss(anchor, positive, negative) 
1802     >>> output.backward() 
1803     >>> 
1804     >>> # Custom Distance Function (Lambda) 
1805     >>> triplet_loss = ( 
1806     >>>     nn.TripletMarginWithDistanceLoss( 
1807     >>>         distance_function=lambda x, y: 1.0 - F.cosine_similarity(x, y))) 
1808     >>> output = triplet_loss(anchor, positive, negative) 
1809     >>> output.backward() 
1810  
1811     Reference: 
1812         V. Balntas, et al.: Learning shallow convolutional feature descriptors with triplet losses: 
1813         https://bmva-archive.org.uk/bmvc/2016/papers/paper119/index.html 
1814     """
1815 
1816     __constants__ = ["margin", "swap", "reduction"]
1817     margin: float
1818     swap: bool
1819 
1820     def __init__(
1821         self,
1822         *,
1823         distance_function: Optional[Callable[[Tensor, Tensor], Tensor]] = None,
1824         margin: float = 1.0,
1825         swap: bool = False,
1826         reduction: str = "mean",
1827     ):
1828         super().__init__(size_average=None, reduce=None, reduction=reduction)
1829         if margin <= 0:
1830             raise ValueError(
1831                 f"TripletMarginWithDistanceLoss: expected margin to be greater than 0, got {margin} instead"
1832             )
1833         self.distance_function: Optional[Callable[[Tensor, Tensor], Tensor]] = (
1834             distance_function if distance_function is not None else PairwiseDistance()
1835         )
1836         self.margin = margin
1837         self.swap = swap
1838 
1839     def forward(self, anchor: Tensor, positive: Tensor, negative: Tensor) -> Tensor:
1840         return F.triplet_margin_with_distance_loss(
1841             anchor,
1842             positive,
1843             negative,
1844             distance_function=self.distance_function,
1845             margin=self.margin,
1846             swap=self.swap,
1847             reduction=self.reduction,
1848         )

1851 class CTCLoss(_Loss):
1852     r"""The Connectionist Temporal Classification loss. 
1853  
1854     Calculates loss between a continuous (unsegmented) time series and a target sequence. CTCLoss sums over the 
1855     probability of possible alignments of input to target, producing a loss value which is differentiable 
1856     with respect to each input node. The alignment of input to target is assumed to be "many-to-one", which 
1857     limits the length of the target sequence such that it must be :math:`\leq` the input length. 
1858  
1859     Args: 
1860         blank (int, optional): blank label. Default :math:`0`. 
1861         reduction (str, optional): Specifies the reduction to apply to the output: 
1862             ``'none'`` | ``'mean'`` | ``'sum'``. ``'none'``: no reduction will be applied, 
1863             ``'mean'``: the output losses will be divided by the target lengths and 
1864             then the mean over the batch is taken, ``'sum'``: the output losses will be summed. 
1865             Default: ``'mean'`` 
1866         zero_infinity (bool, optional): 
1867             Whether to zero infinite losses and the associated gradients. 
1868             Default: ``False`` 
1869             Infinite losses mainly occur when the inputs are too short 
1870             to be aligned to the targets. 
1871  
1872     Shape: 
1873         - Log_probs: Tensor of size :math:`(T, N, C)` or :math:`(T, C)`, 
1874           where :math:`T = \text{input length}`, 
1875           :math:`N = \text{batch size}`, and 
1876           :math:`C = \text{number of classes (including blank)}`. 
1877           The logarithmized probabilities of the outputs (e.g. obtained with 
1878           :func:`torch.nn.functional.log_softmax`). 
1879         - Targets: Tensor of size :math:`(N, S)` or 
1880           :math:`(\operatorname{sum}(\text{target\_lengths}))`, 
1881           where :math:`N = \text{batch size}` and 
1882           :math:`S = \text{max target length, if shape is } (N, S)`. 
1883           It represents the target sequences. Each element in the target 
1884           sequence is a class index. And the target index cannot be blank (default=0). 
1885           In the :math:`(N, S)` form, targets are padded to the 
1886           length of the longest sequence, and stacked. 
1887           In the :math:`(\operatorname{sum}(\text{target\_lengths}))` form, 
1888           the targets are assumed to be un-padded and 
1889           concatenated within 1 dimension. 
1890         - Input_lengths: Tuple or tensor of size :math:`(N)` or :math:`()`, 
1891           where :math:`N = \text{batch size}`. It represents the lengths of the 
1892           inputs (must each be :math:`\leq T`). And the lengths are specified 
1893           for each sequence to achieve masking under the assumption that sequences 
1894           are padded to equal lengths. 
1895         - Target_lengths: Tuple or tensor of size :math:`(N)` or :math:`()`, 
1896           where :math:`N = \text{batch size}`. It represents lengths of the targets. 
1897           Lengths are specified for each sequence to achieve masking under the 
1898           assumption that sequences are padded to equal lengths. If target shape is 
1899           :math:`(N,S)`, target_lengths are effectively the stop index 
1900           :math:`s_n` for each target sequence, such that ``target_n = targets[n,0:s_n]`` for 
1901           each target in a batch. Lengths must each be :math:`\leq S` 
1902           If the targets are given as a 1d tensor that is the concatenation of individual 
1903           targets, the target_lengths must add up to the total length of the tensor. 
1904         - Output: scalar if :attr:`reduction` is ``'mean'`` (default) or 
1905           ``'sum'``. If :attr:`reduction` is ``'none'``, then :math:`(N)` if input is batched or 
1906           :math:`()` if input is unbatched, where :math:`N = \text{batch size}`. 
1907  
1908     Examples: 
1909  
1910         >>> # Target are to be padded 
1911         >>> T = 50  # Input sequence length 
1912         >>> C = 20  # Number of classes (including blank) 
1913         >>> N = 16  # Batch size 
1914         >>> S = 30  # Target sequence length of longest target in batch (padding length) 
1915         >>> S_min = 10  # Minimum target length, for demonstration purposes 
1916         >>> 
1917         >>> # Initialize random batch of input vectors, for *size = (T,N,C) 
1918         >>> input = torch.randn(T, N, C).log_softmax(2).detach().requires_grad_() 
1919         >>> 
1920         >>> # Initialize random batch of targets (0 = blank, 1:C = classes) 
1921         >>> target = torch.randint(low=1, high=C, size=(N, S), dtype=torch.long) 
1922         >>> 
1923         >>> input_lengths = torch.full(size=(N,), fill_value=T, dtype=torch.long) 
1924         >>> target_lengths = torch.randint( 
1925         ...     low=S_min, 
1926         ...     high=S, 
1927         ...     size=(N,), 
1928         ...     dtype=torch.long, 
1929         ... ) 
1930         >>> ctc_loss = nn.CTCLoss() 
1931         >>> loss = ctc_loss(input, target, input_lengths, target_lengths) 
1932         >>> loss.backward() 
1933         >>> 
1934         >>> 
1935         >>> # Target are to be un-padded 
1936         >>> T = 50  # Input sequence length 
1937         >>> C = 20  # Number of classes (including blank) 
1938         >>> N = 16  # Batch size
