3    from sympy.core import Expr, sympify
4    from sympy.functions.elementary.complexes import adjoint
5    
6    __all__ = [
7        'Dagger'
8    ]
9    
10   
11   class Dagger(adjoint):
12       """General Hermitian conjugate operation. 
13    
14       Explanation 
15       =========== 
16    
17       Take the Hermetian conjugate of an argument [1]_. For matrices this 
18       operation is equivalent to transpose and complex conjugate [2]_. 
19    
20       Parameters 
21       ========== 
22    
23       arg : Expr 
24           The SymPy expression that we want to take the dagger of. 
25       evaluate : bool 
26           Whether the resulting expression should be directly evaluated. 
27    
28       Examples 
29       ======== 
30    
31       Daggering various quantum objects: 
32    
33           >>> from sympy.physics.quantum.dagger import Dagger 
34           >>> from sympy.physics.quantum.state import Ket, Bra 
35           >>> from sympy.physics.quantum.operator import Operator 
36           >>> Dagger(Ket('psi')) 
37           <psi| 
38           >>> Dagger(Bra('phi')) 
39           |phi> 
40           >>> Dagger(Operator('A')) 
41           Dagger(A) 
42    
43       Inner and outer products:: 
44    
45           >>> from sympy.physics.quantum import InnerProduct, OuterProduct 
46           >>> Dagger(InnerProduct(Bra('a'), Ket('b'))) 
47           <b|a> 
48           >>> Dagger(OuterProduct(Ket('a'), Bra('b'))) 
49           |b><a| 
50    
51       Powers, sums and products:: 
52    
53           >>> A = Operator('A') 
54           >>> B = Operator('B') 
55           >>> Dagger(A*B) 
56           Dagger(B)*Dagger(A) 
57           >>> Dagger(A+B) 
58           Dagger(A) + Dagger(B) 
59           >>> Dagger(A**2) 
60           Dagger(A)**2 
61    
62       Dagger also seamlessly handles complex numbers and matrices:: 
63    
64           >>> from sympy import Matrix, I 
65           >>> m = Matrix([[1,I],[2,I]]) 
66           >>> m 
67           Matrix([ 
68           [1, I], 
69           [2, I]]) 
70           >>> Dagger(m) 
71           Matrix([ 
72           [ 1,  2], 
73           [-I, -I]]) 
74    
75       References 
76       ========== 
77    
78       .. [1] https://en.wikipedia.org/wiki/Hermitian_adjoint 
79       .. [2] https://en.wikipedia.org/wiki/Hermitian_transpose 
80       """
81   
82       @property
83       def kind(self):
84           """Find the kind of a dagger of something (just the kind of the something)."""
85           return self.args[0].kind
86   
87       def __new__(cls, arg, evaluate=True):
88           if hasattr(arg, 'adjoint') and evaluate:
89               return arg.adjoint()
90           elif hasattr(arg, 'conjugate') and hasattr(arg, 'transpose') and evaluate:
91               return arg.conjugate().transpose()
92           return Expr.__new__(cls, sympify(arg))
93   
94   adjoint.__name__ = "Dagger"
95   adjoint._sympyrepr = lambda a, b: "Dagger(%s)" % b._print(a.args[0])
