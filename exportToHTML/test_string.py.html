40       @staticmethod
41       def _sint(s, start=0, end=None):
42           """Return the content of a string buffer as integer value. 
43    
44           For example: 
45             _sint('1234') -> 4321 
46             _sint('123A') -> 17321 
47           """
48           if isinstance(s, np.ndarray):
49               s = s.tobytes()
50           elif isinstance(s, str):
51               s = s.encode()
52           assert isinstance(s, bytes)
53           if end is None:
54               end = len(s)
55           i = 0
56           for j in range(start, min(end, len(s))):
57               i += s[j] * 10**j
58           return i

60       def _get_input(self, intent="in"):
61           if intent in ["in"]:
62               yield ""
63               yield "1"
64               yield "1234"
65               yield "12345"
66               yield b""
67               yield b"\0"
68               yield b"1"
69               yield b"\01"
70               yield b"1\0"
71               yield b"1234"
72               yield b"12345"
73           yield np.ndarray((), np.bytes_, buffer=b"")  # array(b'', dtype='|S0')
74           yield np.array(b"")  # array(b'', dtype='|S1')
75           yield np.array(b"\0")
76           yield np.array(b"1")
77           yield np.array(b"1\0")
78           yield np.array(b"\01")
79           yield np.array(b"1234")
80           yield np.array(b"123\0")
81           yield np.array(b"12345")

84           for s in self._get_input():
85               r = self.module.test_in_bytes4(s)
86               # also checks that s is not changed inplace

90       def test_intent_inout(self):
91           for s in self._get_input(intent="inout"):
92               rest = self._sint(s, start=4)
93               r = self.module.test_inout_bytes4(s)
94               expected = self._sint(s, end=4)
