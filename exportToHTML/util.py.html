11   def gen_tvar(curr):
12       """ 
13       Generate a tensor variable 
14       :param curr: The current counter 
15       :return: a tensor variable and the updated counter 
16       """

18       return TVar(curr), curr

41   def gen_tensor_dims(n, curr):
42       """ 
43       Generate a list of tensor dimensions 
44       :param n:  the number of dimensions 
45       :param curr: the current counter 
46       :return: a list of dimension variables and an updated counter 
47       """
48       dims = []
49       for _ in range(n):
50           dvar, curr = gen_dvar(curr)
51           dims.append(dvar)
52       return dims, curr

55   def gen_nat_constraints(list_of_dims):
56       """ 
57       Generate natural number constraints for dimensions 
58       """
59       return [BinConstraintD(0, d, op_leq) for d in list_of_dims]
