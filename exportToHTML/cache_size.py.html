13   """ 
14   [Note on cache size limit] 
15    
16   Background - TorchDynamo cache is a linked list. Each cache entry is a 
17   (guard_manager, out_code, next pointer). These are stored on the f_code's co_extra 
18   scratch space. When a frame is invoked, we walk this linked list and run 
19   guard_manager in each cache_entry to decide if the frame needs recompilation. If none 
20   of the guard_manager's returns True, we recompile and add a new entry. To ensure we 
21   don't end up recompiling infinitely, we put limits on the cache size. 
22    
23   There are two limits 
24   1) recompile_limit 
25   2) accumulated_recompile_limit 
26    
27    
28   Earlier we used to have only limit - maximum number of entries in 1 cache line 
29   (which is now represented by (2) above). So, why do we need two limits? Lets try 
30   to understand that. 
31    
32   In general, we want our cache limit value to be a small number (e.g. 8 or even 
33   lower). This ensures that for frames that cause too many recompilation fall to 
34   eager quickly. However, there is another problem that prevents us from lowering 
35   the value of recompile_limit. This is due to ID_MATCH'd guards. Today, we put 
36   ID_MATCH guards on nn module if there is a graph break. This means we will have 
37   many recompilations for the same code object because the ID_MATCH guard fails 
38   for different instances of the nn module. This is a common pattern in how models 
39   are authored. Therefore, this requires us to keep the recompile_limit high. 
40    
41   We resolve this by introducing these two limits. The first limit (1) limits the 
42   number of cache entries that have an ID_MATCH'd guard for an nn module instance. 
43   And, (2)nd limit becomes a safeguard mechanism to have a maximum compilations 
44   for a code object. One important question is - what is the limit for the code 
45   object that does not have any ID_MATCH guard? For such code objects, we choose 
46   (1) as the cache size limit. 
47    
48   Lets take an example to understand how these limits help. Suppose, we have 16 
49   instances of a nn module and we ID_MATCH on the self object. Further, suppose 
50   the inputs to these functions have varying batch size, leading to one 
51   recompilation. In total, there will be 32 recompilations, and therefore 32 cache 
52   entries on the forward code object. In the older case when we had only 1 limit, 
53   our cache size limit must be >= 32 to capture all these recompilations. Now, 
54   suppose there is a separate function in the same program which is very dynamic 
55   and unsuitable for compilation. Such a function will need to undergo 32 
56   compilations to burst the cache and fallback to eager. These 32 recompilations 
57   are too many and we want to fallback for these compilation-unfriendly functions 
58   sooner. 
59    
60   In the new scenario, we can have (1) recompile_limit = 2, (2) 
61   accumulated_recompile_limit = 32. This means that each ID_MATCH'd object can 
62   have maximum of two cache entries, and the maximum number of cache entries 
63   (irrespective of ID_MATCH obj) is 32. This covers the case of forward code 
64   object which has 32 recompilations. For the other function, the one unsuitable 
65   for recompilation, our limit is 2. So, we will burst the cache in just 2 
66   recompilations. In this manner, these 2 limits help us resolve the tension 
67   mentioned earlier. 
68   """

71   @dataclass
72   class CacheSizeRelevantForFrame:
73       """ 
74       We track the number of cache entries that have same id_match objects as the 
75       given frame. 
76    
77       TODO(janimesh) - Consider adding a map from tuple_of_match_ids to count - 
78       https://github.com/pytorch/pytorch/pull/107496#discussion_r1304564682 - this 
79       could be useful for debugging as well. 
80       """
81   
82       # Total number of CacheEntry objects in the Dynamo linked list
83       num_cache_entries: int = 0
84   
85       # Number of CacheEntry objects having same ID_MATCH'd objects as given frame.
86       num_cache_entries_with_same_id_matched_objs: int = 0
87   
88       def will_compilation_exceed(self, limit: int) -> bool:
89           # Checks if a compilation will exceed the given limit (that's why >=).
90           return (
91               self.will_compilation_exceed_accumulated_limit()
92               or self.will_compilation_exceed_specific_limit(limit)
93           )
94   
95       def will_compilation_exceed_accumulated_limit(self) -> bool:
96           return self.num_cache_entries >= config.accumulated_recompile_limit
97   
98       def will_compilation_exceed_specific_limit(self, limit: int) -> bool:
99           return self.num_cache_entries_with_same_id_matched_objs >= limit

102  def _get_weakref_from_f_locals(frame: DynamoFrameType, local_name: str):
103      obj = frame.f_locals.get(local_name, None)
104      weak_id = None
105      try:
106          weak_id = weakref.ref(obj)
107      except TypeError:
108          pass  # cannot weakref bool object
109      return weak_id
110  
111  
112  def _has_same_id_matched_objs(frame: DynamoFrameType, cache_entry) -> bool:
113      """ 
114      Checks if the ID_MATCH'd objects saved on cache_entry are same as the ones 
115      in frame.f_locals. 
116      """
117      if not cache_entry:
118          return False
119  
120      for (
121          local_name,
122          weakref_from_cache_entry,
123      ) in cache_entry.guard_manager.id_matched_objs.items():
124          if weakref_from_cache_entry() is not None:
125              weakref_from_frame = _get_weakref_from_f_locals(frame, local_name)
126              if weakref_from_frame is not weakref_from_cache_entry:
127                  return False
128  
129      # Also covers the case where no ID_MATCH objects are saved in frame.f_locals
130      return True
131

133  def compute_cache_size(
134      frame: DynamoFrameType, cache_entry
135  ) -> CacheSizeRelevantForFrame:
136      # Walk the linked list to calculate the cache size
137      num_cache_entries = 0
138      num_cache_entries_with_same_id_matched_objs = 0
139  
140      while cache_entry:
141          num_cache_entries += 1
142          # Track the number of cache entries having same ID_MATCH'd objects as
143          # that of frame.f_locals. This will be used later to compare against the
144          # recompile_limit.
145          if _has_same_id_matched_objs(frame, cache_entry):
146              num_cache_entries_with_same_id_matched_objs += 1
147          cache_entry = cache_entry.next
148  
149      return CacheSizeRelevantForFrame(
150          num_cache_entries, num_cache_entries_with_same_id_matched_objs
151      )

154  def is_recompilation(cache_size: CacheSizeRelevantForFrame) -> bool:
155      """ 
156      If the frame (earlier parsed by compute_cache_size) has more than 1 cache 
157      entry with same ID_MATCH'd objects, then its a recompilation. 
158      """
159      # Note that you can have multiple entries in the cache but still not a
160      # recompile, e.g., you can have 64 nn module instances, each one having an
161      # ID_MATCH guard, and each one having just 1 cache entry in the cache.  In
162      # this case, we can have 64 entries in the cache, but no recompilation
163      # because there is only one entry for each id_matched_obj.
164      return cache_size.will_compilation_exceed(1)
165  
166  
167  def exceeds_recompile_limit(
168      cache_size: CacheSizeRelevantForFrame, compile_id: CompileId
169  ) -> tuple[bool, str]:
170      """ 
171      Checks if we are exceeding the cache size limit. 
172      """
173      if cache_size.will_compilation_exceed_accumulated_limit():
174          return True, "accumulated_recompile_limit"
175      if cache_size.will_compilation_exceed_specific_limit(config.recompile_limit):
176          return True, "recompile_limit"
177      # NOTE this check is needed in the case that the frame's cache doesn't grow
178      # and we keep recompiling. This can happen if the guard guard_manager becomes invalidated,
179      # e.g. due to guarded objects being freed. This technically makes the
180      # will_compilation_exceed_accumulated_limit check unnecessary, but we will keep the
181      # check in case we have a better fix in the future.
182      assert compile_id.frame_compile_id is not None
183      if compile_id.frame_compile_id >= config.accumulated_recompile_limit:
184          return True, "accumulated_recompile_limit"
185      return False, ""
