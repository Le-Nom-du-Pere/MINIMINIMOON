1    """ 
2    This makes the functions in torch._C._VariableFunctions available as 
3        torch._VF.<funcname> 
4    without mypy being able to find them. 
5     
6    A subset of those functions are mapped to ATen functions in 
7    torch/jit/_builtins.py 
8     
9    See https://github.com/pytorch/pytorch/issues/21478 for the reason for 
10   introducing torch._VF 
11    
12   """
13   
14   import sys
15   import types
16   
17   import torch
18   
19   
20   class VFModule(types.ModuleType):
21       vf: types.ModuleType
22   
23       def __init__(self, name: str):
24           super().__init__(name)
25           self.vf = torch._C._VariableFunctions
26   
27       def __getattr__(self, name: str) -> object:
28           return getattr(self.vf, name)
29   
30   
31   sys.modules[__name__] = VFModule(__name__)
