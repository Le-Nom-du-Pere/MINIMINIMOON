90   def test_tensorproduct():
91       a = BosonOp("a")
92       b = BosonOp("b")
93       ket1 = TensorProduct(BosonFockKet(1), BosonFockKet(2))
94       ket2 = TensorProduct(BosonFockKet(0), BosonFockKet(0))
95       ket3 = TensorProduct(BosonFockKet(0), BosonFockKet(2))
96       bra1 = TensorProduct(BosonFockBra(0), BosonFockBra(0))
97       bra2 = TensorProduct(BosonFockBra(1), BosonFockBra(2))
98       assert qapply(TensorProduct(a, b ** 2) * ket1) == sqrt(2) * ket2
99       assert qapply(TensorProduct(a, Dagger(b) * b) * ket1) == 2 * ket3
100      assert qapply(bra1 * TensorProduct(a, b * b),
101                    dagger=True) == sqrt(2) * bra2
102      assert qapply(bra2 * ket1).doit() == S.One
103      assert qapply(TensorProduct(a, b * b) * ket1) == sqrt(2) * ket2
104      assert qapply(Dagger(TensorProduct(a, b * b) * ket1),
105                    dagger=True) == sqrt(2) * Dagger(ket2)

134  # Issue 24158: Tests whether qapply incorrectly evaluates some ket*op as op*ket
135  def test_issue24158_ket_times_op():
136      P = BosonFockKet(0) * BosonOp("a") # undefined term
137      # Does lhs._apply_operator_BosonOp(rhs) still evaluate ket*op as op*ket?
138      assert qapply(P) == P   # qapply(P) -> BosonOp("a")*BosonFockKet(0) = 0 before fix
139      P = Qubit(1) * XGate(0) # undefined term
140      # Does rhs._apply_operator_Qubit(lhs) still evaluate ket*op as op*ket?
141      assert qapply(P) == P   # qapply(P) -> Qubit(0) before fix
142      P1 = Mul(QubitBra(0), Mul(QubitBra(0), Qubit(0)), XGate(0)) # legal expr <0| * (<1|*|1>) * X
143      assert qapply(P1) == QubitBra(0) * XGate(0)     # qapply(P1) -> 0 before fix
144      P1 = qapply(P1, dagger = True)  # unsatisfactorily -> <0|*X(0), expect <1| since dagger=True
145      assert qapply(P1, dagger = True) == QubitBra(1) # qapply(P1, dagger=True) -> 0 before fix
146      P2 = QubitBra(0) * (QubitBra(0) * Qubit(0)) * XGate(0) # 'forgot' to set brackets
147      P2 = qapply(P2, dagger = True) # unsatisfactorily -> <0|*X(0), expect <1| since dagger=True
148      assert P2 == QubitBra(1) # qapply(P1) -> 0 before fix
149      # Pull Request 24237: IdentityOperator from the right without dagger=True option
150      with warns_deprecated_sympy():
151          assert qapply(QubitBra(1)*IdentityOperator()) == QubitBra(1)
152          assert qapply(IdentityGate(0)*(Qubit(0) + Qubit(1))) == Qubit(0) + Qubit(1)
