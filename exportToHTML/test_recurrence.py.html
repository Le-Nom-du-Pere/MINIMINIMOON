6    def test_RecurrenceOperator():
7        n = symbols('n', integer=True)
8        R, Sn = RecurrenceOperators(QQ.old_poly_ring(n), 'Sn')
9        assert Sn*n == (n + 1)*Sn
10       assert Sn*n**2 == (n**2+1+2*n)*Sn
11       assert Sn**2*n**2 == (n**2 + 4*n + 4)*Sn**2
12       p = (Sn**3*n**2 + Sn*n)**2
13       q = (n**2 + 3*n + 2)*Sn**2 + (2*n**3 + 19*n**2 + 57*n + 52)*Sn**4 + (n**4 + 18*n**3 + \
14           117*n**2 + 324*n + 324)*Sn**6
15       assert p == q

36       R, _ = RecurrenceOperators(QQ.old_poly_ring(n), 'Sn')
37       rr = RecurrenceOperator([n**2, 0, 0], R)
