86       def __init__(
87           self,
88           expr,
89           shape_env,
90           pytype,
91           hint: Optional[Union[int, float, bool]],
92           constant=None,
93           fx_node=None,
94           optimized_summation=False,
95       ):
96           self._expr = expr
97           self.shape_env = shape_env
98           self.pytype = pytype
99           self._optimized_summation = optimized_summation
100  
101          # What's the difference between hint and constant?
102          #
103          # - A constant is known to be invariant across invocations of the model;
104          #   it will always be this value.  We only really know this when we
105          #   encounter an honest-to-goodness literal (when wrapping it into
106          #   a SymNode, we set constant.)  Most of the time, constant is None
107          #
108          # - A hint is a *particular* value from the particular run we are
109          #   tracing, but it may vary the next time around.  It's useful to
110          #   keep this around, as if we need a concrete value from a SymNode,
111          #   we will return the hint and guard on the expression that produced
112          #   it giving the same hint next time around.  The hint is not
113          #   guaranteed to be set either: if you have an unbacked SymNode,
114          #   there won't be any hint; it was the result of some tensor-dependent
115          #   computation, but we don't know what it actually is because we
116          #   haven't actually run the tensor computation.
117          #
118          # If _hint is None, we will query maybe_evaluate_static(compute_hint=True)
119          # in hopes that we've learned enough about the unbacked symints to
120          # discharge the hint; otherwise, you're likely to just error out.
121          #
122          # (A previous version of this system had some optimizations to only
123          # recompute when it was possible we had learned enough about the
124          # unbacked symint that a hint was now possible, but as we added more
125          # potential refinements to unbacked symints this got harder to keep
126          # in sync, so we've deleted it for now.)
127  
128          def compute_hint():
129              from torch.fx.experimental.symbolic_shapes import has_free_unbacked_symbols
130  
131              # This occasionally gets exercised by, e.g.,
132              # convert_shape_to_symint.  It's just a nicety so you don't HAVE
133              # to have a correct hint on hand when making a SymNode.
134              # Don't attempt to compute for unbacked, this can be quite
135              # expensive.
136              if has_free_unbacked_symbols(self.expr):
137                  return None
138              hint = self.shape_env._maybe_evaluate_static(self.expr, compute_hint=True)
139              if hint is not None:
140                  hint = self.pytype(hint) if not isinstance(hint, SymTypes) else hint
141              return hint
142  
143          if hint is not None:
144              assert type(hint) is pytype or type(hint) is _to_symtype(pytype), (
145                  "Cannot create SymNode of type "
146                  f"{pytype} with incompatible hint of type {type(hint)}"
147              )
148              if self.shape_env and self.shape_env._translation_validation_enabled:
149                  # This is technically not TV, but this assert is expensive so
150                  # let's only do it when we're already doing expensive things
151                  computed_hint = compute_hint()
152                  assert hint == computed_hint, (
153                      f"{hint} != {computed_hint} (for {self.expr})"
154                  )
155          else:
156              hint = compute_hint()
157          self._hint = hint
158          self.constant: Optional[Union[int, float, bool]] = constant
159  
160          # Record the FX node of the current node if we are doing translation
161          # validation. They will be used for building the input assertions for
162          # the translation validation problem.
163          tx_validation_en = (
164              self.shape_env and self.shape_env._translation_validation_enabled
165          )
166          self.fx_node = tx_validation_en and fx_node

198      def require_hint(self, fallback=None):
199          from torch.fx.experimental.symbolic_shapes import free_unbacked_symbols
200  
201          if self._hint is None:
202              if fallback is not None:
203                  # Say we have some expr like 2*u0 + s0
204                  # The hint will be None, since the expr contains at least 1 unbacked.
205                  # We will:
206                  # - replace every backed free symbol with its corresponding hint
207                  # - replace every unbacked free symbol with the fallback
208                  # - regenerate the expression with those symbol replacements
209                  # Note: this is not really complete either, since right now
210                  # this logic does not take into account any value ranges
211                  # for the unbacked symints, we may need to beef it up at some point.
212                  unbacked_symbols = free_unbacked_symbols(self.expr)
213                  replacements = {
214                      s: 4096 if s in unbacked_symbols else self.shape_env.var_to_val[s]
215                      for s in self.expr.free_symbols
216                  }
217                  return self.expr.xreplace(replacements)
218              # NB: we expect this to raise
219              return self.shape_env.size_hint(self.expr)
220          return self._hint

474      def sym_sum(self, args) -> SymNode:
475          import sympy
476  
477          # Inner impl
478          from torch.fx.experimental.proxy_tensor import (
479              get_proxy_mode,
480              handle_sym_dispatch,
481          )
482  
483          if get_proxy_mode():
484              return to_node(
485                  self,
486                  handle_sym_dispatch(
487                      torch.sym_sum,
488                      (tuple(wrap_node(a) for a in args),),
489                      {},
490                  ),
491              )
492          exprs = [a.expr for a in args]
493          out = sympy.Add(*exprs)
494  
495          size_hints = []
496          out_hint = None
497          for a in args:
498              if a.hint is None:
499                  break
500              size_hints.append(a.hint)
501          else:
502              out_hint = sum(size_hints)
503  
504          fx_node, _ = self.shape_env._create_fx_call_function(
505              torch.sym_sum, (tuple(a.fx_node for a in args),)
506          )
507  
508          # NB: Only for integers!
509          return SymNode(out, self.shape_env, int, out_hint, fx_node=fx_node)

545      def expect_true(self, file, line):
546          from torch.fx.experimental.symbolic_shapes import free_unbacked_symbols
547  
548          if (
549              self.has_hint()
550              and not free_unbacked_symbols(self.expr)
551              and not self.shape_env.prefer_deferred_runtime_asserts_over_guards
552          ):
553              # OK to generate guards
554              return self.guard_bool(file, line)
555          # Generate a deferred runtime assert (this might actually end up doing
556          # a regular guard if we can!)
557          # TODO: file/line here is very important, because the assert has been
558          # deferred so you can't backtrace easily
559          return self.shape_env.guard_or_defer_runtime_assert(
560              self.expr, f"{file}:{line}", fx_node=self.fx_node
561          )

563      def expect_size(self, file, line):
564          from torch.fx.experimental.symbolic_shapes import _advise_is_size
565  
566          b = self.ge(self.wrap_int(0))
567          # Generate a deferred runtime assert
568          r = b.expect_true(file, line)
569          # Refine compile time range, but only if it's unbacked.
570          # If you refine range for hinted variables, you can end up making
571          # improper deductions since compile time reasoning may be
572          # incompatible with runtime reasoning.
573          if r and not self.has_hint():
574              _advise_is_size(SymInt(self))
575          return r

583      def guard_size_oblivious(self, file, line):
584          """ 
585          Like guard_bool, but if we encounter unbacked symbols, if those symbols 
586          are size-like, we will treat them as >= 2 for the purposes of the analysis. 
587   
588          This CHANGES the runtime semantics, but all size-oblivious sites have been 
589          audited to ensure that the runtime semantics don't change in a material way. 
590          Acceptable runtime semantic changes are, e.g., squeeze() no longer dropping 
591          an unbacked one size, or a tensor reporting as non-contiguous even if it's 
592          contiguous if it would have been reported contiguous due to being empty. 
593          """
594          # TODO: use the file/line for some useful diagnostic on why a
595          # guard occurred
596          r = self.evaluate(size_oblivious=True)
597          try:
598              return bool(r)
599          except Exception:
600              log.warning("Failed to convert to bool: %s", r)
601              raise

825  def _binary_search_insert_arg(ordered_args, new_arg):
826      """ 
827      If new_arg is found in ordered_args None is returned, else the new 
828      ordered_args with new_arg inserted 
829      """
830      if len(ordered_args) == 0:
831          return [new_arg]
832  
833      from sympy.core.basic import _args_sortkey as sort_key, Basic
834  
835      # Fast path when new_arg > ordered_args[-1].
836      if sort_key(ordered_args[-1]) < sort_key(new_arg):
837          return ordered_args + [new_arg]
838  
839      # Fast path when new_arg < ordered_args[0].
840      if sort_key(ordered_args[0]) > sort_key(new_arg):
841          return [new_arg] + ordered_args
842  
843      low, high = 0, len(ordered_args) - 1
844  
845      while low <= high:
846          mid = (low + high) // 2
847          compare_result = Basic.compare(ordered_args[mid], new_arg)
848          if compare_result == 0:
849              return None
850          elif compare_result < 0:
851              low = mid + 1
852          else:
853              high = mid - 1
854  
855      ordered_args.insert(low, new_arg)
856      return ordered_args

859  def _optimized_add(
860      lhs, rhs, lhs_is_optimized_summation=False, rhs_is_optimized_summation=False
861  ):
862      """ 
863      Custom optimization for Add used to optimize incremental binary summations of certain properties. The idea 
864      is when we know the expression is a summation of unique symbols all we need to know is the correct order of symbols, 
865      and no other optimizations are needed. We pass evaluate=false, with the correct order of args and save the following. 
866      1. Avoid running other optimizations when the Add is constructed. 
867      2. Manually figure out the order of the args for the new expression in log(n) comparisons instead of nLog(n) 
868      (comparing terms is expensive and shows in the profiles). 
869      The function returns a tuple of (1) a boolean that indicates whether the output is a summation of unique symbols, 
870      (2) the result sympy expression. 
871      """
872      import sympy
873      from sympy.core.basic import _args_sortkey as sortkey
874  
875      def make_optimized(ordered_args):
876          assert ordered_args is not None
877          result = sympy.Add(*ordered_args, evaluate=False)
878          return (True, result)
879  
880      from torch.utils._sympy.functions import _is_symbols_binary_summation
881  
882      lhs_is_optimized_summation |= _is_symbols_binary_summation(lhs)
883      rhs_is_optimized_summation |= _is_symbols_binary_summation(rhs)
884  
885      if lhs_is_optimized_summation and rhs_is_optimized_summation:
886          # (a0+a1..) + (a2+a3..) => (a0+a1+a2+a3)
887          if sortkey(lhs._args[-1]) < sortkey(rhs._args[0]):
888              return make_optimized(lhs._args + rhs._args)
889          #  (a2+a3..) + (a0+a1..) => (a0+a1+a2+a3)
890          if sortkey(lhs._args[0]) > sortkey(rhs._args[-1]):
891              return make_optimized(rhs._args + lhs._args)
892  
893          #  (a1+a3) + (a0+a2) => (a0+a1+a2+a3)
894          if len(lhs._args) <= 2 and len(rhs._args) <= 2:
895              new_args = list(lhs._args)
896              for a in rhs._args:
897                  new_args = _binary_search_insert_arg(new_args, a)
898                  if new_args is None:
899                      break
900              # None means an element already exists.
901              if new_args is not None:
902                  return make_optimized(new_args)
903  
904      # (a0+a2) + a1 => (a0+a1+a2)
905      if lhs_is_optimized_summation and rhs.is_symbol:
906          new_args = _binary_search_insert_arg(list(lhs._args), rhs)
907          # None means an element already exists.
908          if new_args is not None:
909              return make_optimized(new_args)
910  
911      # a1 + (a0+a2)=> (a0+a1+a2)
912      if rhs_is_optimized_summation and lhs.is_symbol:
913          new_args = _binary_search_insert_arg(list(rhs._args), lhs)
914          # None means an element already exists.
915          if new_args is not None:
916              return make_optimized(new_args)
917  
918      result = sympy.Add(lhs, rhs)
919      return (_is_symbols_binary_summation(result), result)

1134     import sympy
1135 
1136     dim = len(sizes)
1137 
1138     if len(dim_order) != dim:
1139         return sympy.false

1143     # Contiguous if the strides make sense (or the dim is size 1)

1165 def sympy_is_channels_last_strides_generic(sizes, strides, dim_order):
1166     import sympy
1167 
1168     from torch.utils._sympy.functions import Max
1169 
1170     dim = len(sizes)
1171 
1172     if dim != len(dim_order):
1173         return sympy.false
1174 
1175     m = sympy.S.Zero
1176     r = sympy.true
1177 
1178     # special case for trivial C dimension. default to NCHW
1179     r &= sympy.Ne(strides[1], 0)
1180 
1181     for d in dim_order:
1182         r &= sympy.Ne(sizes[d], 0) & (strides[d] >= m)
1183         # Fallback to NCHW as default layout for ambiguous cases
1184         # This is the flaw of implicit memory_format from strides.
1185         # N111 tensor with identical strides for size 1 dimension;
1186         # Two cases could lead us here:
1187         # a. N111 contiguous Tensor ([N,1,1,1]@[1,1,1,1])
1188         # b. N11W contiguous Tensor sliced on the W-dimension.
1189         # ([N,1,1,1]@[W,W,W,W])
1190         if d == 0:
1191             r &= sympy.Ne(m, strides[1])
1192         # This is necessary to:
1193         # 1. distinguish the memory_format of N1H1;
1194         #     [H, 1, 1, 1] channels_last stride
1195         #     [H, H, 1, 1] contiguous stride
1196         # 2. permutation of 1C1W:
1197         #     [1, C, 1, H]@[HC, H, H, 1] transpose(1, 3)
1198         #     [1, H, 1, C]@[HC, 1, H, H] shouldn't be identified as
1199         #     channels_last
1200         m = strides[d] * Max(sizes[d], 1)
1201 
1202     return r
