8    class VariadicSignatureType(type):
9        # checking if subclass is a subclass of self
10       def __subclasscheck__(cls, subclass):
11           other_type = subclass.variadic_type if isvariadic(subclass) else (subclass,)
12           return subclass is cls or all(
13               issubclass(other, cls.variadic_type)  # type: ignore[attr-defined]
14               for other in other_type
15           )
16   
17       def __eq__(cls, other):
18           """ 
19           Return True if other has the same variadic type 
20           Parameters 
21           ---------- 
22           other : object (type) 
23               The object (type) to check 
24           Returns 
25           ------- 
26           bool 
27               Whether or not `other` is equal to `self` 
28           """
29           return isvariadic(other) and set(cls.variadic_type) == set(other.variadic_type)  # type: ignore[attr-defined]
30   
31       def __hash__(cls):
32           return hash((type(cls), frozenset(cls.variadic_type)))  # type: ignore[attr-defined]

35   def isvariadic(obj):
36       """Check whether the type `obj` is variadic. 
37       Parameters 
38       ---------- 
39       obj : type 
40           The type to check 
41       Returns 
42       ------- 
43       bool 
44           Whether or not `obj` is variadic 
45       Examples 
46       -------- 
47       >>> # xdoctest: +SKIP 
48       >>> isvariadic(int) 
49       False 
50       >>> isvariadic(Variadic[int]) 
51       True 
52       """
53       return isinstance(obj, VariadicSignatureType)

56   class VariadicSignatureMeta(type):
57       """A metaclass that overrides ``__getitem__`` on the class. This is used to 
58       generate a new type for Variadic signatures. See the Variadic class for 
59       examples of how this behaves. 
60       """
61   
62       def __getitem__(cls, variadic_type):
63           if not (isinstance(variadic_type, (type, tuple)) or type(variadic_type)):
64               raise ValueError(
65                   "Variadic types must be type or tuple of types"
66                   " (Variadic[int] or Variadic[(int, float)]"
67               )
68   
69           if not isinstance(variadic_type, tuple):
70               variadic_type = (variadic_type,)
71           return VariadicSignatureType(
72               f"Variadic[{typename(variadic_type)}]",
73               (),
74               dict(variadic_type=variadic_type, __slots__=()),
75           )
76

78   class Variadic(metaclass=VariadicSignatureMeta):
79       """A class whose getitem method can be used to generate a new type 
80       representing a specific variadic signature. 
81       Examples 
82       -------- 
83       >>> # xdoctest: +SKIP 
84       >>> Variadic[int]  # any number of int arguments 
85       <class 'multipledispatch.variadic.Variadic[int]'> 
86       >>> Variadic[(int, str)]  # any number of one of int or str arguments 
87       <class 'multipledispatch.variadic.Variadic[(int, str)]'> 
88       >>> issubclass(int, Variadic[int]) 
89       True 
90       >>> issubclass(int, Variadic[(int, str)]) 
91       True 
92       >>> issubclass(str, Variadic[(int, str)]) 
93       True 
94       >>> issubclass(float, Variadic[(int, str)]) 
95       False 
96       """
