27   class ONNXTorchPatcher:
28       """Context manager to temporarily patch PyTorch during FX-to-ONNX export. 
29    
30       This class is a collection of "patches" required by FX-to-ONNX exporter. 
31    
32       This context overrides several torch functions to support symbolic 
33       export of large scale models. 
34    
35       torch.load: 
36           This function is patched to record the files PyTorch stores model 
37           parameters and buffers. Downstream FX-to-ONNX exporter can create 
38           initializers from these files. 
39       torch.fx._symbolic_trace._wrapped_methods_to_patch: 
40           This list is extended with (torch.Tensor, "__getitem__") so that 
41           weight[x, :, y] becomes exportable with torch.fx.symbolic_trace. 
42       safetensors.torch.load_file: 
43           This function is patched to allow safetensors to be loaded within 
44           FakeTensorMode. Remove after https://github.com/huggingface/safetensors/pull/318 
45    
46       Search for ONNXTorchPatcher in test_fx_to_onnx_with_onnxruntime.py for 
47       example usage. 
48    
49       TODO: Should this really be a global patcher? Can we make it a local patcher? 
50           A reason for splitting this into several patchers is to patch one part of the code 
51           as a collateral damage of patching another part of the code. For example, we 
52           for tracing model with torch._dynamo.export, we don't need to patch 
53           `torch.fx._symbolic_trace._wrapped_methods_to_patch` 
54       """
55   
56       def __init__(self) -> None:
57           # List of file paths processed by torch.load.
58           self.paths: list[Union[str, io.BufferedIOBase]] = []
59   
60           def torch_load_wrapper(f, *args, **kwargs):
61               # Record path for later serialization into ONNX proto
62               self.paths.append(f)
63               # Then, call the original torch.load.
64               return self.torch_load(f, *args, **kwargs)
65   
66           # Original version of torch.load.
67           self.torch_load = torch.load
68   
69           # Wrapper or modified version of torch functions.
70           self.torch_load_wrapper = torch_load_wrapper
71   
72           if has_safetensors_and_transformers():
73               import safetensors
74               import transformers
75   
76               def safetensors_load_file_wrapper(filename, device="cpu"):
77                   # Record path for later serialization into ONNX proto
78                   self.paths.append(filename)
79                   result = {}
80                   with safetensors.torch.safe_open(  # type: ignore[attr-defined]
81                       filename, framework="pt", device=device
82                   ) as f:
83                       for k in f.keys():
84                           fake_mode = torch._guards.detect_fake_mode()
85                           if not fake_mode:
86                               result[k] = f.get_tensor(k)
87                           else:
88                               empty_tensor = f.get_slice(k)
89                               result[k] = torch.empty(
90                                   tuple(empty_tensor.get_shape()),
91                                   dtype=safetensors.torch._getdtype(
92                                       empty_tensor.get_dtype()
93                                   ),
94                               )
95                   return result
96   
97               self.safetensors_torch_load_file = safetensors.torch.load_file
98               self.safetensors_torch_load_file_wrapper = safetensors_load_file_wrapper
99               self.transformers_modeling_utils_safe_load_file = (
100                  transformers.modeling_utils.safe_load_file
101              )
102  
103      def __enter__(self):
104          torch.load = self.torch_load_wrapper
105  
106          self.torch_fx__symbolic_trace__wrapped_methods_to_patch = (
107              torch.fx._symbolic_trace._wrapped_methods_to_patch
108          )
109          desired_wrapped_methods = copy.deepcopy(
110              torch.fx._symbolic_trace._wrapped_methods_to_patch
111          )
112          if (torch.Tensor, "__getitem__") not in desired_wrapped_methods:
113              # Adding `__getitem__` to the patching list will make tensor indexing traceable via
114              # torch.fx.symbolic_trace. Otherwise, `tensor[x, :, y]` cannot be traced.
115              # This happens because `__getitem__` is neither under torch domain nor an aten operator,
116              # so the patching (or similar Proxy-generating mechanism) doesn't happen automatically.
117              # Note that torch.fx.symbolic_trace defines FX_PATCH_GETITEM environment variable for
118              # enabling the line below for patching.
119              desired_wrapped_methods.append((torch.Tensor, "__getitem__"))
120          torch.fx._symbolic_trace._wrapped_methods_to_patch = desired_wrapped_methods
121  
122          if has_safetensors_and_transformers():
123              import safetensors
124              import transformers
125  
126              safetensors.torch.load_file = self.safetensors_torch_load_file_wrapper
127              transformers.modeling_utils.safe_load_file = (
128                  self.safetensors_torch_load_file_wrapper
129              )
130  
131      def __exit__(self, exc_type, exc_value, traceback):
132          torch.load = self.torch_load
133          torch.fx._symbolic_trace._wrapped_methods_to_patch = (
134              self.torch_fx__symbolic_trace__wrapped_methods_to_patch
135          )
136          if has_safetensors_and_transformers():
137              import safetensors
138              import transformers
139  
140              safetensors.torch.load_file = self.safetensors_torch_load_file
141              transformers.modeling_utils.safe_load_file = (
142                  self.transformers_modeling_utils_safe_load_file
143              )
