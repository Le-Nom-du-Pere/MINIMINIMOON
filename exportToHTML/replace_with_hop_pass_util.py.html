19   def _replace_with_hop_helper(
20       node: torch.fx.Node,
21       enter_block_node: torch.fx.Node,
22       wrap_hoo: HigherOrderOperator,
23   ) -> None:
24       graph: torch.fx.Graph = node.graph
25       assert graph.owning_module is not None
26       gm: torch.fx.GraphModule = graph.owning_module
27       assert isinstance(node.target, str)
28       sub_gm = getattr(gm, node.target)
29   
30       def set_hoo_node_meta(call_func_node):
31           call_func_node.meta["nn_module_stack"] = copy.copy(
32               enter_block_node.meta.get("nn_module_stack", {})
33           )
34           call_func_node.meta["torch_fn"] = (
35               f"{wrap_hoo.__name__}",
36               f"{wrap_hoo.__class__.__name__}.{wrap_hoo.__name__}",
37           )
38           if isinstance(output_args, (tuple, list)):
39               call_func_node.meta["val"] = tuple(arg.meta["val"] for arg in output_args)
40           elif isinstance(output_args, torch.fx.Node):
41               call_func_node.meta["val"] = (output_args.meta["val"],)
42   
43       with graph.inserting_before(node):
44           get_attr_node = graph.get_attr(node.target)
45           get_attr_node.meta["nn_module_stack"] = copy.copy(
46               enter_block_node.meta.get("nn_module_stack", {})
47           )
48           output_node = next(iter(reversed(sub_gm.graph.nodes)), None)
49           # Split_module pass intentially doesn't add output node
50           # if the graph doesn't return anything.
51           # TODO (tmanlaibaatar) Figure out if this is right behaviour
52           # for split_module
53           if isinstance(output_node, torch.fx.Node) and output_node.op != "output":
54               output_node = None
55           if output_node is not None:
56               assert len(output_node.args) == 1
57               output_args = output_node.args[0]
58               enter_block_node_args = enter_block_node.args
59               if isinstance(output_args, (tuple, list)):
60                   call_func_node = graph.call_function(
61                       wrap_hoo,
62                       (*enter_block_node_args, get_attr_node, *node.args),
63                       {},
64                   )
65                   # Create the metadata
66                   set_hoo_node_meta(call_func_node)
67                   node_replace_(node, call_func_node)
68   
69                   # Rename the name of getitem nodes to the actual name of its contents
70                   # for passing verifier and better readability, also propagate metadata
71                   for get_item_node in call_func_node.users.keys():
72                       idx: int = get_item_node.args[1]  # type: ignore[assignment]
73                       output_node = output_args[idx]
74                       get_item_node._rename(output_node.name)
75                       get_item_node.meta = output_node.meta
76   
77               elif isinstance(output_args, torch.fx.Node):
78                   call_func_node = graph.create_node(
79                       "call_function",
80                       wrap_hoo,
81                       (*enter_block_node_args, get_attr_node, *node.args),
82                       {},
83                       output_args.name,
84                   )
85                   # Modify the subgraph to output a singleton list.
86                   output_node.args = ((output_args,),)
87                   # Add in an extra `getitem(wrap_hoo, 0)` node to the toplevel graph.
88                   get_item_node = graph.create_node(
89                       "call_function",
90                       operator.getitem,
91                       (call_func_node, 0),
92                       {},
93                   )
94                   # Create the metadata
95                   get_item_node.meta = output_args.meta
96                   set_hoo_node_meta(call_func_node)
97                   node_replace_(node, get_item_node)
98               else:
99                   raise NotImplementedError(
100                      f"repalce_with_hop_pass doesnt' support output type {type(output_args)}"
101                  )
102          else:
103              # TODO (shangdiy): remove this line, since the export graph can be non-functional
104              node.graph.erase_node(node)

107  def _sequential_split_and_maybe_inline_subgraphs_helper(
108      new_gm: torch.fx.GraphModule,
109      graph_signature: Optional[ExportGraphSignature],
110      maybe_inline_or_replace_with_hop: Callable[[torch.fx.Node], None],
111  ) -> tuple[torch.fx.GraphModule, Optional[ExportGraphSignature]]:
112      """ 
113      Helper function for replacing graph nodse with higher order nodes. 
114      For each subgraph in `new_gm`, decides whether to construct a HOO subgraph, or inline the calls 
115      back into the parent graph module, depending on `maybe_inline_or_replace_with_hop`. 
116      """
117      # new_gm is a new graph module that could have different output args names.
118      # We need to fix the graph signature.
119      replace_ctx = contextlib.nullcontext()
120      new_signature = None
121      if graph_signature is not None:
122          # Cannot deep copy a real ScriptObject, which is referenced
123          # in the FakeScriptObject. Copy should be good enough to guard
124          # against accidental mutation to original graph_signature.
125          new_signature = copy.copy(graph_signature)
126          new_gm_out_node = next(reversed(new_gm.graph.find_nodes(op="output")))
127          assert new_gm_out_node.op == "output" and len(new_gm_out_node.args[0]) == len(
128              new_signature.output_specs
129          )
130          for arg_node, out_spec in zip(
131              new_gm_out_node.args[0], new_signature.output_specs
132          ):
133              if arg_node is None:
134                  assert out_spec.arg.value is None  # type: ignore[union-attr]
135              elif (
136                  isinstance(arg_node, torch.fx.Node)
137                  and out_spec.arg.name != arg_node.name
138              ):
139                  out_spec.arg.name = arg_node.name
140  
141          replace_ctx = new_gm._set_replace_hook(new_signature.get_replace_hook())  # type: ignore[assignment]
142  
143      with replace_ctx:
144          nodes_map(
145              list(new_gm.graph.nodes),
146              lambda node: (
147                  maybe_inline_or_replace_with_hop(node)
148                  if node.op == "call_module"
149                  else node
150              ),
151          )
152      new_gm.recompile()
153      new_gm.graph.lint()
154      return new_gm, new_signature

157  def _replace_with_hop_pass_helper(
158      gm: torch.fx.GraphModule,
159      graph_signature: Optional[ExportGraphSignature],
160      sequential_split_and_maybe_inline_subgraphs: Callable[
161          [torch.fx.GraphModule, Optional[ExportGraphSignature]],
162          tuple[torch.fx.GraphModule, Optional[ExportGraphSignature]],
163      ],
164  ) -> tuple[torch.fx.GraphModule, Optional[ExportGraphSignature]]:
165      """ 
166      Split gm into sub-graph-modules using `sequential_split_and_maybe_inline_subgraphs`, and 
167      then recursively call itself on each of the submodules. 
168      """
169      new_gm, new_signature = sequential_split_and_maybe_inline_subgraphs(
170          gm, graph_signature
171      )
172      # recursively call
173      for node in new_gm.graph.nodes:
174          if node.op == "get_attr":
175              subgm = getattr(new_gm, node.target)
176              if not isinstance(subgm, torch.fx.GraphModule):
177                  continue
178              new_subgm, _ = _replace_with_hop_pass_helper(
179                  subgm,
180                  None,
181                  sequential_split_and_maybe_inline_subgraphs,
182              )
183              setattr(new_gm, node.target, new_subgm)
184  
185      new_gm.recompile()
186      new_gm.graph.lint()
187      return new_gm, new_signature
