10   import logging
11   from collections.abc import Sequence
12   from typing import Any, Callable, TypeVar
13   
14   import onnxscript
15   
16   import torch
17   from torch.onnx._internal.exporter import _constants, _registration
18   
19   
20   _T = TypeVar("_T", bound=Callable)
21   
22   logger = logging.getLogger("__name__")
23   
24   
25   _registry: list[_registration.OnnxDecompMeta] = []
26   
27   
28   def onnx_impl(
29       target: _registration.TorchOp | tuple[_registration.TorchOp, ...],
30       *,
31       trace_only: bool = False,
32       complex: bool = False,
33       opset_introduced: int = 18,
34       no_compile: bool = False,
35       private: bool = False,
36   ) -> Callable[[_T], _T]:
37       """Register an ONNX implementation of a torch op."""
38   
39       if isinstance(target, torch._ops.OpOverloadPacket):
40           raise TypeError(
41               f"Target '{target}' should be provided as an OpOverload instead of an "
42               "OpOverloadPacket. You can get the default overload with "
43               "<op>.default"
44           )
45   
46       def wrapper(
47           func: _T,
48       ) -> _T:
49           processed_func: Any
50           if no_compile:
51               processed_func = func
52           else:
53               torchlib_opset = onnxscript.values.Opset(
54                   domain=_constants.TORCHLIB_DOMAIN, version=1
55               )
56   
57               if not trace_only:
58                   # Compile the function
59                   processed_func = onnxscript.script(opset=torchlib_opset)(func)
60               else:
61                   processed_func = onnxscript.TracedOnnxFunction(torchlib_opset, func)
62   
63           if not private:
64               # TODO(justinchuby): Simplify the logic and remove the private attribute
65               # Skip registration if private
66               if not isinstance(target, Sequence):
67                   targets = (target,)
68               else:
69                   targets = target  # type: ignore[assignment]
70   
71               for t in targets:
72                   _registry.append(
73                       _registration.OnnxDecompMeta(
74                           onnx_function=processed_func,
75                           fx_target=t,
76                           signature=None,
77                           is_complex=complex,
78                           opset_introduced=opset_introduced,
79                           skip_signature_inference=no_compile,
80                       )
81                   )
82           return processed_func  # type: ignore[return-value]
83   
84       return wrapper
85   
86   
87   def get_torchlib_ops() -> tuple[_registration.OnnxDecompMeta, ...]:
88       # Trigger op registration
89       from torch.onnx._internal.exporter._torchlib import ops
90   
91       del ops
92       assert len(_registry) != 0
93       return tuple(_registry)
