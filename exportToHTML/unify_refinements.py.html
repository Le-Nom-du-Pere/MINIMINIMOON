37   def convert_eq(list_of_eq):
38       """ 
39       Convert equality constraints in the right format 
40       to be used by unification library. 
41       """
42       lhs = []
43       rhs = []
44       for eq in list_of_eq:
45           lhs.append(eq.lhs)
46           rhs.append(eq.rhs)
47       return tuple(lhs), tuple(rhs)

50   def unify_eq(list_of_eq):
51       """ 
52       Apply unification to a set of 
53       equality constraints 
54       """
55       lhs, rhs = convert_eq(list_of_eq)
56       return unify(lhs, rhs)

59   def substitute_solution_one_type(mapping, t):
60       """ 
61       Apply the most general unifier to a type 
62       """
63       if isinstance(t, Var):
64           if t in mapping.keys():
65               return mapping[t]
66           else:
67               return t
68   
69       elif isinstance(t, TensorType):
70           new_type = []
71           for typ in t.__args__:
72               if typ in mapping.keys():
73                   new_type.append(mapping[typ])
74               else:
75                   new_type.append(typ)
76           return TensorType(tuple(new_type))
77   
78       elif isinstance(t, list):
79           new_type = []
80           for typ in t:
81               new_type.append(substitute_solution_one_type(mapping, typ))
82           return new_type
83   
84       elif isinstance(t, tuple):
85           new_type = []
86           for typ in t:
87               new_type.append(substitute_solution_one_type(mapping, typ))
88           return tuple(new_type)
89   
90       else:
91           return t

94   def substitute_all_types(graph, mapping):
95       """ 
96       Apply the most general unifier to all types in a graph 
97       till reaching a fixed point. If the input and output graph 
98       are the same, we converge. 
99       """
100      flag = True
101      while flag:
102          flag = False
103          for k in mapping:
104              old_mapping_val = mapping[k]
105              if mapping[k] in mapping.keys():
106                  new_key = mapping[k]
107                  mapping[k] = mapping[new_key]
108              if old_mapping_val != mapping[k]:
109                  flag = True
110  
111      for n in graph.nodes:
112          n.type = substitute_solution_one_type(mapping, n.type)

115  def check_for_type_equality(g1, g2):
116      """ 
117      A check equality to be used in fixed points. 
118      We do not use graph equality but instead type 
119      equality. 
120      """
121      for n, m in zip(g1.nodes, g2.nodes):
122          if n.type != m.type:
123              return False
124      return True
