10   def test_equal():

23   def test_global():
24       """Test for global assumptions"""
25       global_assumptions.add(x > 0)
26       assert (x > 0) in global_assumptions
27       global_assumptions.remove(x > 0)
28       assert not (x > 0) in global_assumptions
29       # same with multiple of assumptions
30       global_assumptions.add(x > 0, y > 0)
31       assert (x > 0) in global_assumptions
32       assert (y > 0) in global_assumptions
33       global_assumptions.clear()
34       assert not (x > 0) in global_assumptions
35       assert not (y > 0) in global_assumptions
