24   class Commutator(Expr):
25       """The standard commutator, in an unevaluated state. 
26    
27       Explanation 
28       =========== 
29    
30       Evaluating a commutator is defined [1]_ as: ``[A, B] = A*B - B*A``. This 
31       class returns the commutator in an unevaluated form. To evaluate the 
32       commutator, use the ``.doit()`` method. 
33    
34       Canonical ordering of a commutator is ``[A, B]`` for ``A < B``. The 
35       arguments of the commutator are put into canonical order using ``__cmp__``. 
36       If ``B < A``, then ``[B, A]`` is returned as ``-[A, B]``. 
37    
38       Parameters 
39       ========== 
40    
41       A : Expr 
42           The first argument of the commutator [A,B]. 
43       B : Expr 
44           The second argument of the commutator [A,B]. 
45    
46       Examples 
47       ======== 
48    
49       >>> from sympy.physics.quantum import Commutator, Dagger, Operator 
50       >>> from sympy.abc import x, y 
51       >>> A = Operator('A') 
52       >>> B = Operator('B') 
53       >>> C = Operator('C') 
54    
55       Create a commutator and use ``.doit()`` to evaluate it: 
56    
57       >>> comm = Commutator(A, B) 
58       >>> comm 
59       [A,B] 
60       >>> comm.doit() 
61       A*B - B*A 
62    
63       The commutator orders it arguments in canonical order: 
64    
65       >>> comm = Commutator(B, A); comm 
66       -[A,B] 
67    
68       Commutative constants are factored out: 
69    
70       >>> Commutator(3*x*A, x*y*B) 
71       3*x**2*y*[A,B] 
72    
73       Using ``.expand(commutator=True)``, the standard commutator expansion rules 
74       can be applied: 
75    
76       >>> Commutator(A+B, C).expand(commutator=True) 
77       [A,C] + [B,C] 
78       >>> Commutator(A, B+C).expand(commutator=True) 
79       [A,B] + [A,C] 
80       >>> Commutator(A*B, C).expand(commutator=True) 
81       [A,C]*B + A*[B,C] 
82       >>> Commutator(A, B*C).expand(commutator=True) 
83       [A,B]*C + B*[A,C] 
84    
85       Adjoint operations applied to the commutator are properly applied to the 
86       arguments: 
87    
88       >>> Dagger(Commutator(A, B)) 
89       -[Dagger(A),Dagger(B)] 
90    
91       References 
92       ========== 
93    
94       .. [1] https://en.wikipedia.org/wiki/Commutator 
95       """
96       is_commutative = False
97   
98       _kind_dispatcher = KindDispatcher("Commutator_kind_dispatcher", commutative=True)
99   
100      @property
101      def kind(self):
102          arg_kinds = (a.kind for a in self.args)
103          return self._kind_dispatcher(*arg_kinds)
104  
105      def __new__(cls, A, B):
106          r = cls.eval(A, B)
107          if r is not None:
108              return r
109          obj = Expr.__new__(cls, A, B)
110          return obj
111  
112      @classmethod
113      def eval(cls, a, b):
114          if not (a and b):
115              return S.Zero
116          if a == b:
117              return S.Zero
118          if a.is_commutative or b.is_commutative:
119              return S.Zero
120  
121          # [xA,yB]  ->  xy*[A,B]
122          ca, nca = a.args_cnc()
123          cb, ncb = b.args_cnc()
124          c_part = ca + cb
125          if c_part:
126              return Mul(Mul(*c_part), cls(Mul._from_args(nca), Mul._from_args(ncb)))
127  
128          # Canonical ordering of arguments
129          # The Commutator [A, B] is in canonical form if A < B.
130          if a.compare(b) == 1:
131              return S.NegativeOne*cls(b, a)
132  
133      def _expand_pow(self, A, B, sign):
134          exp = A.exp
135          if not exp.is_integer or not exp.is_constant() or abs(exp) <= 1:
136              # nothing to do
137              return self
138          base = A.base
139          if exp.is_negative:
140              base = A.base**-1
141              exp = -exp
142          comm = Commutator(base, B).expand(commutator=True)
143  
144          result = base**(exp - 1) * comm
145          for i in range(1, exp):
146              result += base**(exp - 1 - i) * comm * base**i
147          return sign*result.expand()
148  
149      def _eval_expand_commutator(self, **hints):
150          A = self.args[0]
151          B = self.args[1]
152  
153          if isinstance(A, Add):
154              # [A + B, C]  ->  [A, C] + [B, C]
155              sargs = []
156              for term in A.args:
157                  comm = Commutator(term, B)
158                  if isinstance(comm, Commutator):
159                      comm = comm._eval_expand_commutator()
160                  sargs.append(comm)
161              return Add(*sargs)
162          elif isinstance(B, Add):
163              # [A, B + C]  ->  [A, B] + [A, C]
164              sargs = []
165              for term in B.args:
166                  comm = Commutator(A, term)
167                  if isinstance(comm, Commutator):
168                      comm = comm._eval_expand_commutator()
169                  sargs.append(comm)
170              return Add(*sargs)
171          elif isinstance(A, Mul):
172              # [A*B, C] -> A*[B, C] + [A, C]*B
173              a = A.args[0]
174              b = Mul(*A.args[1:])
175              c = B
176              comm1 = Commutator(b, c)
177              comm2 = Commutator(a, c)
178              if isinstance(comm1, Commutator):
179                  comm1 = comm1._eval_expand_commutator()
180              if isinstance(comm2, Commutator):
181                  comm2 = comm2._eval_expand_commutator()
182              first = Mul(a, comm1)
183              second = Mul(comm2, b)
184              return Add(first, second)
185          elif isinstance(B, Mul):
186              # [A, B*C] -> [A, B]*C + B*[A, C]
187              a = A
188              b = B.args[0]
189              c = Mul(*B.args[1:])
190              comm1 = Commutator(a, b)
191              comm2 = Commutator(a, c)
192              if isinstance(comm1, Commutator):
193                  comm1 = comm1._eval_expand_commutator()
194              if isinstance(comm2, Commutator):
195                  comm2 = comm2._eval_expand_commutator()
196              first = Mul(comm1, c)
197              second = Mul(b, comm2)
198              return Add(first, second)
199          elif isinstance(A, Pow):
200              # [A**n, C] -> A**(n - 1)*[A, C] + A**(n - 2)*[A, C]*A + ... + [A, C]*A**(n-1)
201              return self._expand_pow(A, B, 1)
202          elif isinstance(B, Pow):
203              # [A, C**n] -> C**(n - 1)*[C, A] + C**(n - 2)*[C, A]*C + ... + [C, A]*C**(n-1)
204              return self._expand_pow(B, A, -1)
205  
206          # No changes, so return self
207          return self
208  
209      def doit(self, **hints):
210          """ Evaluate commutator """
211          # Keep the import of Operator here to avoid problems with
212          # circular imports.
213          from sympy.physics.quantum.operator import Operator
214          A = self.args[0]
215          B = self.args[1]
216          if isinstance(A, Operator) and isinstance(B, Operator):
217              try:
218                  comm = A._eval_commutator(B, **hints)
219              except NotImplementedError:
220                  try:
221                      comm = -1*B._eval_commutator(A, **hints)
222                  except NotImplementedError:
223                      comm = None
224              if comm is not None:
225                  return comm.doit(**hints)
226          return (A*B - B*A).doit(**hints)

228      def _eval_adjoint(self):
229          return Commutator(Dagger(self.args[1]), Dagger(self.args[0]))

241      def _pretty(self, printer, *args):
242          pform = printer._print(self.args[0], *args)
243          pform = prettyForm(*pform.right(prettyForm(',')))
244          pform = prettyForm(*pform.right(printer._print(self.args[1], *args)))
245          pform = prettyForm(*pform.parens(left='[', right=']'))
246          return pform

248      def _latex(self, printer, *args):
249          return "\\left[%s,%s\\right]" % tuple([
250              printer._print(arg, *args) for arg in self.args])
