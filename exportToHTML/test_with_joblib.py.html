10   def install_joblib():
11       """Install joblib for testing."""
12       try:
13           subprocess.check_call([sys.executable, "-m", "pip", "install", "joblib>=1.3.0"])
14           print("✓ joblib installed successfully")
15           return True
16       except subprocess.CalledProcessError as e:
17           print(f"✗ Failed to install joblib: {e}")
18           return False
19   
20   def test_parallel_with_joblib():
21       """Test parallel processing with joblib installed."""
22       try:
23           from feasibility_scorer import FeasibilityScorer
24           import joblib
25           
26           print(f"✓ joblib version: {joblib.__version__}")
27           
28           # Setup logging
29           logging.basicConfig(level=logging.INFO)
30           
31           # Create large enough batch to trigger parallel processing
32           indicators = [
33               "Incrementar la línea base de 65% de cobertura educativa a una meta de 85% para el año 2025",
34               "Reduce from baseline of 15.3 million people in poverty to target of 8 million by December 2024",
35               "Aumentar el valor inicial de 2.5 millones de beneficiarios hasta alcanzar el objetivo de 4 millones",
36               "Mejorar desde la situación inicial hasta el objetivo propuesto con incremento del 20%",
37               "Partir del nivel base actual para lograr la meta establecida en los próximos años"
38           ]
39           
40           # Create batch large enough for parallel processing
41           large_batch = indicators * 4  # 20 indicators
42           
43           print(f"\nTesting parallel processing with {len(large_batch)} indicators:")
44           print("-" * 60)
45           
46           # Test with loky backend
47           scorer = FeasibilityScorer(backend='loky', n_jobs=4)
48           print(f"Scorer config: backend={scorer.backend}, n_jobs={scorer.n_jobs}, parallel_enabled={scorer.enable_parallel}")
49           
50           results = scorer.batch_score(large_batch)
51           print(f"✓ Processed {len(results)} indicators with loky backend")
52           
53           # Test backend comparison
54           print("\nTesting backend comparison:")
55           comparison_results = scorer.batch_score(large_batch, compare_backends=True)
56           print(f"✓ Backend comparison completed with {len(comparison_results)} results")
57           
58           # Verify results are consistent
59           if len(results) == len(comparison_results):
60               score_diffs = [abs(r1.feasibility_score - r2.feasibility_score) 
61                             for r1, r2 in zip(results, comparison_results)]
62               max_diff = max(score_diffs)
63               print(f"✓ Results consistency check: max difference = {max_diff:.6f}")
64           
65           print("\n" + "=" * 60)
66           print("✓ All parallel processing tests passed with joblib!")
67           return True
68           
69       except Exception as e:
70           print(f"✗ Parallel processing test failed: {e}")
71           import traceback
72           traceback.print_exc()
73           return False
