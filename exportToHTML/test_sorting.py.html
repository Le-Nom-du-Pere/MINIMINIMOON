18   def test_ordered():
19       # Issue 7210 - this had been failing with python2/3 problems
20       assert (list(ordered([{1:3, 2:4, 9:10}, {1:3}])) == \
21                  [{1: 3}, {1: 3, 2: 4, 9: 10}])

24       assert list(ordered(l, warn=True)) == l

26       assert list(ordered(l, warn=True)) == [[1], [1], [2]]

28           default=False, warn=True)))
