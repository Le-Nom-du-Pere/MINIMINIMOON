16           "Reduce from baseline of 15.3 million people in poverty to target of 8 million by December 2024",

20           "Achieve target improvement from current baseline within the established timeframe",

22           "Improve from baseline to reach established goal"
23       ]

25       print("Testing Parallel Processing Implementation")

35       # Test with threading backend comparison

37       results_comparison = scorer_loky.batch_score(batch_large, compare_backends=True)
38       print(f"   Processed {len(results_comparison)} indicators with comparison")
39       
40       # Test sequential vs parallel for smaller batch
41       print("\n3. Testing sequential fallback for small batch:")
42       small_batch = indicators[:3]
43       results_small = scorer_loky.batch_score(small_batch)
44       print(f"   Processed {len(results_small)} indicators (should use sequential)")
45       
46       # Test with parallel disabled
47       print("\n4. Testing with parallel disabled:")
48       scorer_seq = FeasibilityScorer(enable_parallel=False)
49       results_seq = scorer_seq.batch_score(indicators)
50       print(f"   Processed {len(results_seq)} indicators sequentially")
51       
52       # Test n_jobs configuration

60       try:

66           print(f"   Pickled copy works: score = {test_result.feasibility_score:.2f}")
67       except Exception as e:
68           print(f"   Pickling test failed: {e}")
69
