19   /** 
20    * Note [Flags defining the behavior of events] 
21    * 
22    * PYTORCH_DEFAULT and BACKEND_DEFAULT are valid for all backends. The 
23    * BACKEND_DEFAULT is what a particular backend would select if no 
24    * flags were given. PYTORCH_DEFAULT is the PyTorch's framework default 
25    * choice for events on that backend, which may not be the same. 
26    * 
27    * The mapping of PYTORCH_DEFAULT and BACKEND_DEFAULT is done by each 
28    * backend implementation. 
29    */
30   enum class EventFlag {
31     // Disable timing
32     PYTORCH_DEFAULT,
33     // Enable timing
34     BACKEND_DEFAULT,
35     // FOR TESTING ONLY
36     INVALID
37   };

41   /** 
42    * DeviceGuardImplInterface represents the virtual interface which provides 
43    * functionality to provide an RAII class for device and stream switching, 
44    * via DeviceGuard.  Every distinct device type, e.g., CUDA and HIP, is 
45    * expected to implement and register an implementation of this interface. 
46    * All classes which inherit from DeviceGuardImplInterface should be declared 
47    * 'final'. 
48    * 
49    * This class exists because we provide a unified interface for performing 
50    * device guards via DeviceGuard, but we cannot assume that we have actually 
51    * compiled against the, e.g., CUDA library, which actually implements 
52    * this guard functionality.  In this case, a dynamic dispatch is required 
53    * to cross the library boundary. 
54    * 
55    * If possible, you should directly use implementations of this interface; 
56    * those uses will be devirtualized. 
57    */
58   struct C10_API DeviceGuardImplInterface {
59     DeviceGuardImplInterface() = default;
60     DeviceGuardImplInterface(const DeviceGuardImplInterface&) = default;
61     DeviceGuardImplInterface& operator=(const DeviceGuardImplInterface&) =
62         default;
63     DeviceGuardImplInterface(DeviceGuardImplInterface&&) noexcept = default;
64     DeviceGuardImplInterface& operator=(DeviceGuardImplInterface&&) noexcept =
65         default;
66   
67     /** 
68      * Return the type of device managed by this guard implementation. 
69      */
70     virtual DeviceType type() const = 0;
71   
72     /** 
73      * Set the current device to Device, and return the previous Device. 
74      */
75     virtual Device exchangeDevice(Device) const = 0;
76     // NB: Implementations of exchangeDevice can be a bit boilerplatey.  You might
77     // consider replacing exchangeDevice with a non-virtual function with a baked
78     // in implementation; however, note that this will triple the number of
79     // virtual calls (when you implement exchangeDevice in a final subclass,
80     // the compiler gets to devirtualize everything; it won't do that if you don't
81     // define it in the subclass!)  A common way to solve this problem is to use
82     // some sort of CRTP; however, we can template DeviceGuardImplInterface since
83     // we really *do* need it to be virtual.  A little boilerplate seems easiest
84     // to explain.  (Another way around this problem is to provide inline
85     // functions that provide the default implementations, but this seems a little
86     // hard to explain.  In any case, we're only going to have on order of ten
87     // implementations of this anyway.)
88   
89     /** 
90      * Get the current device. 
91      */
92     virtual Device getDevice() const = 0;
93   
94     /** 
95      * Set the current device to Device. 
96      */
97     virtual void setDevice(Device) const = 0;
98   
99     /** 
100     * Set the current device to Device, without checking for errors 
101     * (so, e.g., this can be called from a destructor). 
102     */
103    virtual void uncheckedSetDevice(Device) const noexcept = 0;
104  
105    /** 
106     * Get the current stream for a given device. 
107     */
108    virtual Stream getStream(Device) const = 0;
109  
110    /** 
111     * Get the default stream for a given device. 
112     */
113    virtual Stream getDefaultStream(Device) const {
114      TORCH_CHECK(false, "Backend doesn't support acquiring a default stream.")
115    }
116  
117    /** 
118     * Get a stream from the global pool for a given device. 
119     */
120    virtual Stream getStreamFromGlobalPool(Device, bool isHighPriority = false)
121        const {
122      (void)isHighPriority; // Suppress unused variable warning
123      TORCH_CHECK(false, "Backend doesn't support acquiring a stream from pool.")
124    }
125  
126    /** 
127     * Return a new stream for a given device and priority. The stream will be 
128     * copied and shared around, device backend should be able to correctly handle 
129     * the lifetime of the stream. 
130     */
131    virtual Stream getNewStream(Device, int priority = 0) const {
132      (void)priority;
133      TORCH_CHECK(false, "Backend doesn't support create a new Stream.")
134    }
135  
136    /** 
137     * Set a stream to be the thread local current stream for its device. 
138     * Return the previous stream for that device. You are NOT required 
139     * to set the current device to match the device of this stream. 
140     */
141    virtual Stream exchangeStream(Stream) const = 0;
142  
143    /** 
144     * Destroys the given event. 
145     */
146    virtual void destroyEvent(void* /*event*/, const DeviceIndex /*device_index*/)
147        const noexcept {}
148  
149    /** 
150     * Increments the event's version and enqueues a job with this version 
151     * in the stream's work queue. When the stream process that job 
152     * it notifies all streams waiting on / blocked by that version of the 
153     * event to continue and marks that version as recorded. 
154     * */
155    virtual void record(
156        void** /*event*/,
157        const Stream& /*stream*/,
158        const DeviceIndex /*device_index*/,
159        const c10::EventFlag /*flag*/) const {
160      TORCH_CHECK(false, "Backend doesn't support events.");
161    }
162  
163    /** 
164     * Does nothing if the event has not been scheduled to be recorded. 
165     * If the event was previously enqueued to be recorded, a command 
166     * to wait for the version of the event that exists at the time of this call 
167     * is inserted in the stream's work queue. 
168     * When the stream reaches this command it will stop processing 
169     * additional commands until that version of the event is marked as recorded. 
170     */
171    virtual void block(void* /*event*/, const Stream& /*stream*/) const {
172      TORCH_CHECK(false, "Backend doesn't support events.");
173    }
174  
175    /** 
176     * Returns true if (and only if) 
177     *  (1) the event has never been scheduled to be recorded 
178     *  (2) the current version is marked as recorded. 
179     * Returns false otherwise. 
180     */
181    virtual bool queryEvent(void* /*event*/) const {
182      TORCH_CHECK(false, "Backend doesn't support events.");
183    }
184  
185    /** 
186     * Get the number of devices.  WARNING: This is REQUIRED to not raise 
187     * an exception.  If there is some sort of problem, e.g., driver error, 
188     * you should report that there are zero available devices. 
189     */
190    virtual DeviceIndex deviceCount() const noexcept = 0;
191  
192    /** 
193     * Return true if all the work previously enqueued on the stream for 
194     * asynchronous execution has completed running on the device. 
195     */
196    virtual bool queryStream(const Stream& /*stream*/) const {
197      TORCH_CHECK(false, "Backend doesn't support querying streams.");
198    }
199  
200    /** 
201     * Wait (by blocking the calling thread) until all the work previously 
202     * enqueued on the stream has completed running on the device. 
203     */
204    virtual void synchronizeStream(const Stream& /*stream*/) const {
205      TORCH_CHECK(false, "Backend doesn't support synchronizing streams.");
206    }
207  
208    /** 
209     * Wait (by blocking the calling thread) until all the work previously 
210     * recorded on the event has completed running on the device. 
211     */
212    virtual void synchronizeEvent(void* /*event*/) const {
213      TORCH_CHECK(false, "Backend doesn't support synchronizing events.");
214    }
215  
216    /** 
217     * Wait (by blocking the calling thread) until all the work previously 
218     * enqueued on the device has been completed. 
219     */
220    virtual void synchronizeDevice(const DeviceIndex /*device_index*/) const {
221      TORCH_CHECK(
222          false, "Backend doesn't support synchronizing all streams on device.");
223    }
224  
225    /** 
226     * Ensure the caching allocator (if any) is aware that the given DataPtr is 
227     * being used on the given stream, and that it should thus avoid recycling the 
228     * DataPtr until all work on that stream is done. 
229     */
230    virtual void recordDataPtrOnStream(const c10::DataPtr&, const Stream&) const {
231    }
232  
233    /** 
234     * Fetch the elapsed time between two recorded events. 
235     */
236    virtual double elapsedTime(
237        void* /*event1*/,
238        void* /*event2*/,
239        const DeviceIndex /*device_index*/) const {
240      TORCH_CHECK(false, "Backend doesn't support elapsedTime.");
241    }
242  
243    /** 
244     * Intended use of this class is to leak the DeviceGuardImpl at program end. 
245     * So you better not call the destructor, buster! 
246     */
247    virtual ~DeviceGuardImplInterface() = default;
248  };

250  // A no-op device guard impl that doesn't do anything interesting.  Useful
251  // for devices that don't actually have a concept of device index.  Prominent
252  // examples are CPU and Meta.
253  template <DeviceType D>
254  struct NoOpDeviceGuardImpl final : public DeviceGuardImplInterface {
255    NoOpDeviceGuardImpl() = default;
256    DeviceType type() const override {
257      return D;
258    }
259    Device exchangeDevice(Device) const override {
260      return Device(D, -1); // no-op
261    }
262    Device getDevice() const override {
263      return Device(D, -1);
264    }
265    void setDevice(Device) const override {
266      // no-op
267    }
268    void uncheckedSetDevice(Device) const noexcept override {
269      // no-op
270    }
271    Stream getStream(Device) const noexcept override {
272      // no-op
273      return Stream(Stream::DEFAULT, Device(D, -1));
274    }
275  
276    Stream getNewStream(Device, int priority = 0) const override {
277      // no-op
278      (void)priority;
279      return Stream(Stream::DEFAULT, Device(D, -1));
280    }
281  
282    // NB: These do NOT set the current device
283    Stream exchangeStream(Stream) const noexcept override {
284      // no-op
285      return Stream(Stream::DEFAULT, Device(D, -1));
286    }
287    DeviceIndex deviceCount() const noexcept override {
288      return 1;
289    }
290  
291    // Event-related functions
292    void record(
293        void** /*event*/,
294        const Stream& /*stream*/,
295        const DeviceIndex /*device_index*/,
296        const EventFlag /*flag*/) const override {
297      TORCH_CHECK(false, D, " backend doesn't support events.");
298    }
299    void block(void* /*event*/, const Stream& /*stream*/) const override {
300      TORCH_CHECK(false, D, " backend doesn't support events.")
301    }
302    bool queryEvent(void* /*event*/) const override {
303      TORCH_CHECK(false, D, " backend doesn't support events.")
304    }
305    void destroyEvent(void* /*event*/, const DeviceIndex /*device_index*/)
306        const noexcept override {}
307  
308    // Stream-related functions
309    bool queryStream(const Stream& /*stream*/) const override {
310      return true;
311    }
312    void synchronizeStream(const Stream& /*stream*/) const override {
313      // Don't wait for anything.
314    }
315  };

317  // The registry is NON-owning.  Each stored pointer is std::atomic so
318  // that under all interleavings of registry calls the structure is
319  // race-free.  This doesn't cost us anything on reads in X86.  (An
320  // unsynchronized implementation probably is OK too, but I didn't want
321  // to prove that we never read from device_guard_impl_registry at the
322  // same time some registration is occurring.  Shiver.)
323  //
324  // I'd like this registry to be valid even at program destruction time
325  // (in case someone uses a DeviceGuard in a destructor to do some cleanup
326  // in the CUDA API.)  Since there are no direct accesses of the underlying
327  // owning objects which I can use to enforce initialization order (unlike
328  // in a Meyer singleton), it implies that you must *leak* objects when
329  // putting them in the registry.  This is done by deleting the destructor
330  // on DeviceGuardImplInterface.
331  extern C10_API std::array<
332      std::atomic<const DeviceGuardImplInterface*>,
333      static_cast<size_t>(DeviceType::COMPILE_TIME_MAX_DEVICE_TYPES)>
334      device_guard_impl_registry;
335  
336  // I can't conveniently use c10/util/Registry.h for the following reason:
337  // c10/util/Registry.h gives me a slow way of Create'ing a object of some
338  // interface from the registry, but no way of quickly accessing an already
339  // created object.  I'll be banging on getDeviceGuardImpl every time we do a
340  // DeviceGuard, so I really don't want to be doing an unordered_map lookup.
341  // Better if the registration mechanism directly drops its implementation
342  // into device_guard_impl_registry.
343  
344  class C10_API DeviceGuardImplRegistrar {
345   public:
346    DeviceGuardImplRegistrar(DeviceType, const DeviceGuardImplInterface*);
347  };
348  
349  #define C10_REGISTER_GUARD_IMPL(DevType, DeviceGuardImpl)              \
350    static ::c10::impl::DeviceGuardImplRegistrar C10_ANONYMOUS_VARIABLE( \
351        g_##DeviceType)(::c10::DeviceType::DevType, new DeviceGuardImpl());
352  
353  inline const DeviceGuardImplInterface* getDeviceGuardImpl(DeviceType type) {
354    // Two adjacent int16_t fields DeviceType and DeviceIndex has field access
355    // miscompiled on NVCC. To workaround this issue, we apply a mask to the
356    // DeviceType. First check if the DeviceType is 16-bit.
357    // FB employees can see
358    //   https://fb.workplace.com/groups/llvm.gcc/permalink/4053565044692080/
359    // for more details
360    static_assert(sizeof(DeviceType) == 1, "DeviceType is not 8-bit");
361    auto p = device_guard_impl_registry[static_cast<size_t>(type) & 0xFF].load();
362  
363    // This seems to be the first place where you make use of a device
364    // when you pass devices to factory functions.  Give a nicer error
365    // message in this case.
366    TORCH_CHECK(p, "PyTorch is not linked with support for ", type, " devices");
367    return p;
368  }
369  
370  inline bool hasDeviceGuardImpl(DeviceType type) {
371    return device_guard_impl_registry[static_cast<size_t>(type)].load();
372  }
