43   // A handle to a wait counter.
44   class C10_API WaitCounterHandle {
45    public:
46     explicit WaitCounterHandle(std::string_view key);
47   
48     class WaitGuard {
49      public:
50       WaitGuard(WaitGuard&& other) noexcept
51           : handle_{std::exchange(other.handle_, {})},
52             ctxs_{std::move(other.ctxs_)} {}
53       WaitGuard(const WaitGuard&) = delete;
54       WaitGuard& operator=(const WaitGuard&) = delete;
55       WaitGuard& operator=(WaitGuard&&) = delete;
56   
57       ~WaitGuard() {
58         stop();
59       }
60   
61       void stop() {
62         if (auto handle = std::exchange(handle_, nullptr)) {
63           handle->stop(ctxs_);
64         }
65       }
66   
67      private:
68       WaitGuard(WaitCounterHandle& handle, SmallVector<intptr_t>&& ctxs)
69           : handle_{&handle}, ctxs_{std::move(ctxs)} {}
70   
71       friend class WaitCounterHandle;
72   
73       WaitCounterHandle* handle_;
74       SmallVector<intptr_t> ctxs_;
75     };
76   
77     // Starts a waiter
78     WaitGuard start();
79   
80    private:
81     // Stops the waiter. Each start() call should be matched by exactly one stop()
82     // call.
83     void stop(const SmallVector<intptr_t>& ctxs);
84   
85     // NOLINTNEXTLINE(cppcoreguidelines-avoid-const-or-ref-data-members)
86     detail::WaitCounterImpl& impl_;
87   };
88   } // namespace c10::monitor
