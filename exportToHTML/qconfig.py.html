85   class QConfig(namedtuple("QConfig", ["activation", "weight"])):
86       """ 
87       Describes how to quantize a layer or a part of the network by providing 
88       settings (observer classes) for activations and weights respectively. 
89    
90    
91       Note that QConfig needs to contain observer **classes** (like MinMaxObserver) or a callable that returns 
92       instances on invocation, not the concrete observer instances themselves. 
93       Quantization preparation function will instantiate observers multiple times for each of the layers. 
94    
95    
96       Observer classes have usually reasonable default arguments, but they can be overwritten with `with_args` 
97       method (that behaves like functools.partial):: 
98    
99         my_qconfig = QConfig( 
100            activation=MinMaxObserver.with_args(dtype=torch.qint8), 
101            weight=default_observer.with_args(dtype=torch.qint8), 
102        ) 
103   
104      """
105  
106      __slots__ = ()
107  
108      def __new__(cls, activation, weight):
109          # catch common mistakes
110          if isinstance(activation, nn.Module) or isinstance(weight, nn.Module):
111              raise ValueError(
112                  "QConfig received observer instance, please pass observer class instead. "
113                  + "Use MyObserver.with_args(x=1) to override arguments to constructor if needed"
114              )
115          return super().__new__(cls, activation, weight)

123      """ 
124      Describes how to dynamically quantize a layer or a part of the network by providing 
125      settings (observer classes) for weights. 
126   
127      It's like QConfig, but for dynamic quantization. 
128   
129      Note that QConfigDynamic needs to contain observer **classes** (like MinMaxObserver) or a callable that returns 
130      instances on invocation, not the concrete observer instances themselves. 
131      Quantization function will instantiate observers multiple times for each of the layers. 
132   
133      Observer classes have usually reasonable default arguments, but they can be overwritten with `with_args` 
134      method (that behaves like functools.partial):: 
135   
136        my_qconfig = QConfigDynamic(weight=default_observer.with_args(dtype=torch.qint8)) 
137      """
138  
139      __slots__ = ()
140  
141      def __new__(cls, activation=torch.nn.Identity, weight=torch.nn.Identity):
142          # catch common mistakes
143          if isinstance(weight, nn.Module):
144              raise ValueError(
145                  "QConfigDynamic received observer instance, please pass observer class instead. "
146                  + "Use MyObserver.with_args(x=1) to override arguments to constructor if needed"
147              )
148          return super().__new__(cls, activation, weight)

257  def get_default_qconfig(backend="x86", version=0):
258      """ 
259      Returns the default PTQ qconfig for the specified backend. 
260   
261      Args: 
262        * `backend` (str): a string representing the target backend. Currently supports 
263          `x86` (default), `fbgemm`, `qnnpack` and `onednn`. 
264   
265      Return: 
266          qconfig 
267      """
268      supported_backends = ["fbgemm", "x86", "qnnpack", "onednn"]
269      if backend not in supported_backends:
270          raise AssertionError(
271              "backend: "
272              + str(backend)
273              + f" not supported. backend must be one of {supported_backends}"
274          )
275  
276      if version == 0:
277          if backend == "fbgemm":
278              qconfig = QConfig(
279                  activation=HistogramObserver.with_args(reduce_range=True),
280                  weight=default_per_channel_weight_observer,
281              )
282          elif backend == "qnnpack":
283              # TODO: make this compatible with xnnpack constraints
284              qconfig = QConfig(
285                  activation=HistogramObserver.with_args(reduce_range=False),
286                  weight=default_weight_observer,
287              )
288          elif backend == "onednn":
289              if not torch.cpu._is_vnni_supported():
290                  warnings.warn(
291                      "Default qconfig of oneDNN backend with reduce_range of false may have accuracy issues "
292                      "on CPU without Vector Neural Network Instruction support."
293                  )
294              qconfig = QConfig(
295                  activation=HistogramObserver.with_args(reduce_range=False),
296                  weight=default_per_channel_weight_observer,
297              )
298          elif backend == "x86":
299              qconfig = QConfig(
300                  activation=HistogramObserver.with_args(reduce_range=True),
301                  weight=default_per_channel_weight_observer,
302              )
303          else:
304              # won't reach
305              qconfig = default_qconfig
306      else:
307          raise AssertionError(
308              "Version number: "
309              + str(version)
310              + " in get_default_qconfig is not supported. Version number must be 0"
311          )
312  
313      return qconfig

316  """ 
317  Default, symmetric PTQ qconfig for the specified backend. And a per_channel 
318  variant of the same. 
319   
320  Symmetric here applies to signed weights with zero point = 0, and additional 
321  value restrictions. The activations are also signed 8-bit integers with this 
322  qconfig. 
323   
324      * Once this change is merged [as of 3/17/22], with backend or qengine = 
325      'qnnpack', some quantized operators with this symmetric qconfig may use 
326      operators from xnnpack library. 
327   
328          ** Support to use xnnpack ops with `qnnpack` backed for asymmetric 
329          qconfig (returned by get_default_qconfig()) is not available yet. 
330   
331      * This qconfig uses signed activations and weights. Weights have added 
332      restrictions such as zero point is forced to be 0, making the weights 
333      symmetric, hence the name. And the 8-bit quantized values are 
334      restricting to to [-127, +127], excluding -128. 
335   
336      * xnnpack has a requantization scale value restriction, 0x1p-32 <= 
337      requantization_scale < 256.0 where, `requantization_scale = (input_scale 
338      * kernel_scale) / (output_scale)`. Using this eps (w/ assumed max value 
339      of 256) is to prevent requantization_scale to go below xnnpack lower 
340      threshold. 
341  """
342  default_symmetric_qnnpack_qconfig = QConfig(
343      activation=HistogramObserver.with_args(
344          dtype=torch.qint8, reduce_range=False, eps=2**-12
345      ),

347  )

371  def get_default_qat_qconfig(backend="x86", version=1):
372      """ 
373      Returns the default QAT qconfig for the specified backend. 
374   
375      Args: 
376        * `backend` (str): a string representing the target backend. Currently supports 
377          `x86` (default), `fbgemm`, `qnnpack` and `onednn`. 
378        * `version`: version, for backwards compatibility. Can be `None` or `1`. 
379   
380      Return: 
381          qconfig 
382      """
383      supported_backends = ["fbgemm", "x86", "qnnpack", "onednn"]
384      if backend not in supported_backends:
385          raise AssertionError(
386              "backend: "
387              + str(backend)
388              + f" not supported. backend must be one of {supported_backends}"
389          )
390  
391      # Histogram observer is too slow for quantization aware training
392      if version == 0:
393          if backend == "fbgemm":
394              qconfig = QConfig(
395                  activation=FakeQuantize.with_args(
396                      observer=MovingAverageMinMaxObserver,
397                      quant_min=0,
398                      quant_max=255,
399                      reduce_range=True,
400                  ),
401                  weight=default_per_channel_weight_fake_quant,
402              )
403          elif backend == "qnnpack":
404              qconfig = QConfig(
405                  activation=FakeQuantize.with_args(
406                      observer=MovingAverageMinMaxObserver,
407                      quant_min=0,
408                      quant_max=255,
409                      reduce_range=False,
410                  ),
411                  weight=default_weight_fake_quant,
412              )
413          elif backend == "onednn":
414              qconfig = QConfig(
415                  activation=FakeQuantize.with_args(
416                      observer=MovingAverageMinMaxObserver, quant_min=0, quant_max=255
417                  ),
418                  weight=default_per_channel_weight_fake_quant,
419              )
420          elif backend == "x86":
421              qconfig = QConfig(
422                  activation=FakeQuantize.with_args(
423                      observer=MovingAverageMinMaxObserver,
424                      quant_min=0,
425                      quant_max=255,
426                      reduce_range=True,
427                  ),
428                  weight=default_per_channel_weight_fake_quant,
429              )
430          else:
431              qconfig = default_qat_qconfig
432      # Use the fused observe + fake_quant modules for doing QAT.
433      elif version == 1:
434          if backend == "fbgemm":
435              qconfig = QConfig(
436                  activation=FusedMovingAvgObsFakeQuantize.with_args(
437                      observer=MovingAverageMinMaxObserver,
438                      quant_min=0,
439                      quant_max=255,
440                      reduce_range=True,
441                  ),
442                  weight=default_fused_per_channel_wt_fake_quant,
443              )
444          elif backend == "qnnpack":
445              # TODO: make this compatible with xnnpack constraints
446              qconfig = QConfig(
447                  activation=FusedMovingAvgObsFakeQuantize.with_args(
448                      observer=MovingAverageMinMaxObserver,
449                      quant_min=0,
450                      quant_max=255,
451                      reduce_range=False,
452                  ),
453                  weight=default_fused_wt_fake_quant,
454              )
455          elif backend == "onednn":
456              qconfig = QConfig(
457                  activation=FusedMovingAvgObsFakeQuantize.with_args(
458                      observer=MovingAverageMinMaxObserver, quant_min=0, quant_max=255
459                  ),
460                  weight=default_fused_per_channel_wt_fake_quant,
461              )
462          elif backend == "x86":
463              qconfig = QConfig(
464                  activation=FusedMovingAvgObsFakeQuantize.with_args(
465                      observer=MovingAverageMinMaxObserver,
466                      quant_min=0,
467                      quant_max=255,
468                      reduce_range=True,
469                  ),
470                  weight=default_fused_per_channel_wt_fake_quant,
471              )
472          else:
473              qconfig = default_qat_qconfig_v2
474      else:
475          raise AssertionError(
476              "Version number: "
477              + str(version)
478              + "in get_default_qat_qconfig is not supported. Version number must be 0 or 1"
479          )
480  
481      return qconfig

543  def _assert_valid_qconfig(qconfig: Optional[QConfig], mod: torch.nn.Module) -> None:
544      """ 
545      Verifies that this `qconfig` is valid. 
546      """
547      if qconfig is None:
548          return
549      is_conv_transpose_mod = isinstance(
550          mod,
551          (torch.nn.ConvTranspose1d, torch.nn.ConvTranspose2d, torch.nn.ConvTranspose3d),
552      )
553      if is_conv_transpose_mod:
554          if qconfig.weight is None:
555              # for now, we assume that any qconfig for ConvTranspose without a weight is valid
556              return
557          example_observer = qconfig.weight()
558          is_per_channel = isinstance(
559              example_observer,
560              (
561                  torch.ao.quantization.PerChannelMinMaxObserver,
562                  torch.ao.quantization.MovingAveragePerChannelMinMaxObserver,
563              ),
564          )
565          assert not is_per_channel, (
566              "Per channel weight observer is not supported yet for ConvTranspose{n}d."
567          )

574  def _add_module_to_qconfig_obs_ctr(
575      qconfig: QConfigAny, module: Optional[nn.Module]
576  ) -> Any:
577      r"""This is a helper function for use in quantization prepare that updates a qconfig so that 
578      the constructors stored in the qconfig will create observers on the same device that 
579      'module' is on. This is intended to be used when the qconfigs are propagated to each 
580      module in order to avoid potential device alignment issues. 
581   
582      Args: 
583          qconfig: QConfig with obs constructors stored in activation and weight 
584          module: module which the qconfig is related to 
585   
586      Return: 
587          qconfig: configured so that obs constructors set to construct on the same device as module 
588      """
589  
590      if module is None or qconfig is None or qconfig._fields != ("activation", "weight"):
591          return qconfig
592  
593      def get_factory_kwargs_based_on_module_device():
594          assert isinstance(module, torch.nn.Module)
595          devices = {p.device for p in module.parameters()} | {
596              p.device for p in module.buffers()
597          }
598          device = next(iter(devices)) if len(devices) > 0 else None
599          return None if device is None else {"device": device}
600  
601      def configure_constructor_to_put_obs_on_module_device(original_constructor):
602          try:
603              # check if constructor can accept factory_kwargs
604              check = original_constructor.with_args(factory_kwargs=None)
605              check()
606              return original_constructor.with_callable_args(
607                  factory_kwargs=get_factory_kwargs_based_on_module_device
608              )
609          except AttributeError:  # qconfig doesn't have activation or weight
610              return original_constructor
611          except TypeError:  # the class doesn't accept factory_kwargs argument
612              return original_constructor
613  
614      activation = configure_constructor_to_put_obs_on_module_device(qconfig.activation)
615      weight = configure_constructor_to_put_obs_on_module_device(qconfig.weight)
616  
617      return QConfig(activation, weight)

625  def _obs_or_fq_ctr_equals(
626      obs_or_fq1: _ObserverOrFakeQuantizeConstructor,
627      obs_or_fq2: _ObserverOrFakeQuantizeConstructor,
628  ):
629      if isinstance(obs_or_fq1, _PartialWrapper) and isinstance(
630          obs_or_fq2, _PartialWrapper
631      ):
632          return _partial_wrapper_equals(obs_or_fq1, obs_or_fq2)
633      return obs_or_fq1 == obs_or_fq2

636  def _partial_wrapper_equals(obs_or_fq1: _PartialWrapper, obs_or_fq2: _PartialWrapper):
637      """ 
638      Return whether the two partial wrappers are equal, 
639      """
640      # functools.partial has no __eq__ operator defined so '==' defaults to 'is'
641      obs_or_fq1_keywords = copy.copy(obs_or_fq1.p.keywords)
642      obs_or_fq2_keywords = copy.copy(obs_or_fq2.p.keywords)
643      keywords_equal = True
644      # compare observer constructor with _obs_or_fq_ctr_equals since direct compare would fail
645      if "observer" in obs_or_fq1_keywords and "observer" in obs_or_fq2_keywords:
646          keywords_equal = keywords_equal and _obs_or_fq_ctr_equals(
647              obs_or_fq1_keywords["observer"], obs_or_fq2_keywords["observer"]
648          )
649          obs_or_fq1_keywords.pop("observer")
650          obs_or_fq2_keywords.pop("observer")
651      keywords_equal = keywords_equal and obs_or_fq1_keywords == obs_or_fq2_keywords
652      return (
653          obs_or_fq1.p.func == obs_or_fq2.p.func
654          and obs_or_fq1.p.args == obs_or_fq2.p.args
655          and keywords_equal
656      )

659  def qconfig_equals(q1: QConfigAny, q2: QConfigAny):
660      """ 
661      Returns `True` if `q1` equals `q2`, and `False` otherwise. 
662      """
663      if q1 is None or q2 is None:
664          return q1 == q2
665      else:
666          assert q1 is not None and q2 is not None
667          try:
668              # Qconfig weight and activation can be either a partial wrapper,
669              # or an observer class. Special handling is required (above) for
670              # comparing partial wrappers.
671              activation_same = _obs_or_fq_ctr_equals(q1.activation, q2.activation)
672              weight_same = _obs_or_fq_ctr_equals(q1.weight, q2.weight)
673              return activation_same and weight_same
674          except AttributeError:
675              return q1 == q2

678  def _activation_is_memoryless(qconfig: QConfig):
679      """ 
680      Return whether the observer for activations defined in the given QConfig is memoryless. 
681      This means a MovingAverage observer with averaging constant equal to 1. 
682      """
683  
684      def _is_memoryless(observer):
685          return (
686              hasattr(observer, "averaging_constant") and observer.averaging_constant == 1
687          )
688  
689      act = qconfig.activation()
690      if isinstance(act, FakeQuantizeBase) and hasattr(act, "activation_post_process"):
691          return _is_memoryless(act.activation_post_process)
692      else:
693          return _is_memoryless(act)

699          and isinstance(qconfig.activation(), ReuseInputObserver)
700          and isinstance(qconfig.weight(), NoopObserver)
