29   def test_dft():
30       n, i, j = symbols('n i j')
31       assert DFT(4).shape == (4, 4)
32       assert ask(Q.unitary(DFT(4)))
33       assert Abs(simplify(det(Matrix(DFT(4))))) == 1
34       assert DFT(n)*IDFT(n) == Identity(n)
35       assert DFT(n)[i, j] == exp(-2*S.Pi*I/n)**(i*j) / sqrt(n)
