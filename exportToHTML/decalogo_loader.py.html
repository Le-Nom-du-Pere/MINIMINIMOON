30   def load_decalogo_industrial(target_path: Optional[str] = None) -> str:
31       """ 
32       Load DECALOGO_INDUSTRIAL template, attempting to write to file first, 
33       then falling back to in-memory template on failure. 
34    
35       Args: 
36           target_path: Optional path where the template should be written 
37    
38       Returns: 
39           The DECALOGO_INDUSTRIAL template content 
40       """
41       template_content = DECALOGO_INDUSTRIAL_TEMPLATE.strip()
42   
43       if target_path:
44           target_file = Path(target_path)
45   
46           try:
47               # Create temporary file in the same directory as target
48               temp_dir = target_file.parent
49               temp_dir.mkdir(parents=True, exist_ok=True)
50   
51               with tempfile.NamedTemporaryFile(
52                   mode="w",
53                   dir=temp_dir,
54                   prefix=f".{target_file.name}_tmp_",
55                   suffix=".tmp",
56                   delete=False,
57                   encoding="utf-8",
58               ) as temp_file:
59                   temp_file.write(template_content)
60                   temp_file.flush()
61                   temp_path = Path(temp_file.name)
62   
63               # Atomically rename temporary file to target location
64               temp_path.rename(target_file)
65               log_info_with_text(
66                   logger,
67                   f"DECALOGO_INDUSTRIAL loaded and written to file: {target_path}",
68                   template_content,
69               )
70   
71           except (PermissionError, OSError, IOError) as e:
72               log_warning_with_text(
73                   logger,
74                   f"Failed to write DECALOGO_INDUSTRIAL to {target_path}: {e}. Using in-memory fallback template.",
75                   template_content,
76               )
77               # Clean up temp file if it exists
78               try:
79                   if "temp_path" in locals() and temp_path.exists():
80                       temp_path.unlink()
81               except Exception:
82                   pass  # Best effort cleanup
83   
84               log_info_with_text(
85                   logger,
86                   "DECALOGO_INDUSTRIAL loaded from in-memory fallback template",
87                   template_content,
88               )
89   
90           except Exception as e:
91               log_error_with_text(
92                   logger,
93                   f"Unexpected error writing DECALOGO_INDUSTRIAL to {target_path}: {e}. Using in-memory fallback template.",
94                   template_content,
95               )
96               # Clean up temp file if it exists
97               try:
98                   if "temp_path" in locals() and temp_path.exists():
99                       temp_path.unlink()
100              except Exception:
101                  pass  # Best effort cleanup
102  
103              log_info_with_text(
104                  logger,
105                  "DECALOGO_INDUSTRIAL loaded from in-memory fallback template",
106                  template_content,
107              )
108      else:
109          log_info_with_text(
110              logger,
111              "DECALOGO_INDUSTRIAL loaded from in-memory template (no target path specified)",
112              template_content,
113          )
114  
115      return template_content

118  def get_decalogo_industrial(
119      cache_path: Optional[str] = "decalogo_industrial.txt",
120  ) -> str:
121      """ 
122      Convenience function to get DECALOGO_INDUSTRIAL with optional caching. 
123   
124      Args: 
125          cache_path: Path where template should be cached, or None to skip caching 
126   
127      Returns: 
128          The DECALOGO_INDUSTRIAL template content 
129      """
130      return load_decalogo_industrial(cache_path)
