196  inline bool operator==(const Argument& lhs, const Argument& rhs) {
197    return lhs.name() == rhs.name()
198            && *lhs.type() == *rhs.type()
199            && lhs.N() == rhs.N()
200            && lhs.default_value() == rhs.default_value()
201            && lhs.kwarg_only() == rhs.kwarg_only()
202            && (lhs.alias_info() == rhs.alias_info()
203                || (lhs.alias_info() != nullptr && rhs.alias_info() != nullptr
204                     && *lhs.alias_info() == *rhs.alias_info()));
205  }
206  
207  inline bool operator!=(const Argument& lhs, const Argument& rhs) {
208    return !(lhs == rhs);
209  }

262    // Checks whether this schema is backward compatible with the old one.
263    // The following conditions must be true:
264    // [Function structure] The new schema's name, overload-name, varargs, and
265    //      return arity are the same.
266    // [Output Narrowing] The new schema's output type must be the same class
267    //      or inherit from the old schema's output type.
268    // [Argument count] The new schema must have at least as many arguments as
269    //      the old schema (considering the list of positional and kwargs).
270    // [Arg Compatibility] Every argument in the old schema has a corresponding
271    //      argument in the new schema that:
272    //        * is at the same position.
273    //        * has the same name.
274    //        * is either positional, or kwarg and the old argument was kwarg.
275    //        * has the same type, or the old argument's type inherits from the
276    //          new argument's type.
277    // [Default Values] Every new argument must have a default value.
278    // E.g.
279    //   OK    f_new(a, b, c=1) => f_old(a, b)
280    //   NOK   f_new(a, c=1, *, b) => f_old(a, *, b)
281    //   OK    f_new(a, b, *, c) => f_old(a, *, b, c)
282    //   NOK   f_new(a, *, b, c) -> f_old(a, b, *, c)
283    //   NOK   f_new(a, *, c, b) => f_old(a, *, b, c)
284    //   OK    f_new(a, *, b, c, d=1) => f_old(a, *, b, c)
285    bool isBackwardCompatibleWith(
286        const FunctionSchema& old,
287        std::ostream* why_not = nullptr) const;
288  
289    // Checks whether this schema is forward compatible with the old one.
290    // The following conditions must be true:
291    // [Function structure] The new schema's name, overload-name, varargs, and
292    //      return arity are the same.
293    // [Output Narrowing] The new schema's output type must be the same class
294    //      or inherit from the old schema's output type.
295    // [Arg Compatibility] Every argument in the old schema has a corresponding
296    //      argument in the new schema that:
297    //        * is at the same position.
298    //        * has the same name.
299    //        * is either positional, or kwarg and the old argument was kwarg.
300    //        * has the same type, or the old argument's type inherits from the
301    //          new argument's type.
302    // [Default Values] Every new argument must have a default value.
303    //         Each default value type should NOT be a container type.
304    // [Positioning] All defaults arguments MUST go after either old
305    //         default arguments or the end of positional arguments
306    //         and right BEFORE all out arguments
307    bool isForwardCompatibleWith(
308        const FunctionSchema& old,
309        std::ostringstream& why_not) const;

331    void checkSchema() const {
332      bool seen_default_arg = false;
333      for (const auto& arg : arguments()) {
334        if (arg.default_value()) {
335          seen_default_arg = true;
336        } else {
337          // we have historically serialized broadcasting lists wo/default values,
338          // so to not break BC allow lists here
339          if (arg.type()->kind() == ListType::Kind) {
340            continue;
341          }
342          TORCH_INTERNAL_ASSERT(
343              !seen_default_arg || arg.kwarg_only(),
344              "Non-default positional argument follows default argument. Parameter ",
345              arg.name(),
346              " in ",
347              *this);
348        }
349      }
350    }

548  // print out Argument, which is compatible with FunctionSchema parser
549  // full format: Type(alias)? name=default_value
550  inline std::ostream& operator<<(std::ostream& out, const Argument& arg) {
551  
552    // for adjusting the ? position.
553    // in schema, we have Tensor?(a!) input, and t(a!)?.
554    // however, t?(a!) doesn't work with schema parser.
555    // so we always use Type(alias)? format
556    // real_type versus fake_type: in order to be compatible with FunctionSchema
557    // parser, printing an argument with either MemoryFormat or Layout type should
558    // give us the original schema string, hence printing out real_type.
559    auto type = arg.real_type();
560    bool is_opt = type->kind() == OptionalType::Kind;
561    auto unopt_type = is_opt ? type->castRaw<OptionalType>()->getElementType() : type;
562  
563    if (unopt_type->kind() == ListType::Kind) {
564      // sized lists get size N from arg, not type
565      auto list = unopt_type->cast<c10::ListType>();
566      out << list->getElementType()->str();
567      if (arg.alias_info() && !arg.alias_info()->containedTypes().empty()){
568        out << arg.alias_info()->containedTypes()[0];
569      }
570      std::string N;
571      if (arg.N()) {
572          N = std::to_string(*arg.N());
573      }
574      out << "[" << N << "]";
575    } else {
576      out << unopt_type->str();
577    }
578  
579    // print alias info if it has beforeSets.
580    if (arg.alias_info() && !arg.alias_info()->beforeSets().empty()) {
581      out << *arg.alias_info();
582    }
583  
584    if (is_opt) {
585      out << "?";
586    }
587  
588    if (!arg.name().empty()) {
589      out << " " << arg.name();
590    }
591  
592    if (arg.default_value()) {
593      out << "=";
594      if ((type->kind() == c10::TypeKind::StringType ||
595          unopt_type->kind() == c10::TypeKind::StringType) &&
596          arg.default_value().value().isString()) {
597        printQuotedString(out, arg.default_value().value().toStringRef());
598      } else if (type->kind() == TypeKind::ListType && type->castRaw<ListType>()->getElementType()->kind() == c10::TypeKind::IntType) {
599        // We want to faithfully replicate JIT schema.
600        // in native_functions.yaml defaults for int arrays with a single value always look like
601        //   int[2] stride=1
602        // instead of
603        //   int[2] stride=[1, 1]
604        auto default_val = arg.default_value().value().toIntList();
605        if (default_val.size() > 1) {
606          auto all_defaults_the_same = true;
607          for (const auto i : c10::irange(1, default_val.size())) {
608            if (default_val[0] != default_val[i]) all_defaults_the_same = false;
609          }
610          if (all_defaults_the_same) {
611            out << default_val[0];
612          } else {
613            out << arg.default_value().value();
614          }
615        } else {
616          out << arg.default_value().value();
617        }
618      } else {
619        out << arg.default_value().value();
620      }
621    }
622  
623    return out;
624  }

646      size_t operator()(const c10::Argument& arg) const
647      {
648        auto hash = std::hash<std::string>{}(arg.name());
649        auto type_hash = std::hash<c10::TypePtr>{}(arg.type());
650        auto kwarg_only_hash = std::hash<bool>{}(arg.kwarg_only());
651        hash = c10::hash_combine(hash, type_hash);
652        hash = c10::hash_combine(hash, kwarg_only_hash);
653        // hashing optional fields if they exist
654        if (arg.default_value().has_value()) {
655          // NOLINTNEXTLINE(bugprone-unchecked-optional-access)
656          auto default_value_hash = c10::hash<c10::IValue>{}(*arg.default_value());
657          hash = c10::hash_combine(hash, default_value_hash);
658        }
659        if (arg.N().has_value()) {
660          // NOLINTNEXTLINE(bugprone-unchecked-optional-access)
661          auto N_hash = std::hash<int64_t>{}(*arg.N());
662          hash = c10::hash_combine(hash, N_hash);
663        }
664        if (arg.alias_info()) {
665          auto alias_info_hash = std::hash<c10::AliasInfo>{}(*arg.alias_info());
666          hash = c10::hash_combine(hash, alias_info_hash);
667        }
668        return hash;
669      }
670    };
671  template<>
672    struct hash<c10::FunctionSchema> {
673      size_t operator()(const c10::FunctionSchema& schema) const
674      {
675        auto hash = std::hash<c10::OperatorName>{}(schema.operator_name());
676        auto args_hash = c10::hash<std::vector<c10::Argument>>{}(schema.arguments());
677        auto returns_hash = c10::hash<std::vector<c10::Argument>>{}(schema.returns());
678        auto is_vararg_hash = std::hash<bool>{}(schema.is_vararg());
679        auto is_varret_hash = std::hash<bool>{}(schema.is_varret());
680        hash = c10::hash_combine(hash, args_hash);
681        hash = c10::hash_combine(hash, returns_hash);
682        hash = c10::hash_combine(hash, is_vararg_hash);
683        hash = c10::hash_combine(hash, is_varret_hash);
684        return hash;
685      }
686    };
