12   class FPGMPruner(BaseStructuredSparsifier):
13       r"""Filter Pruning via Geometric Median (FPGM) Structured Pruner 
14       This sparsifier prune fliter (row) in a tensor according to distances among filters according to 
15       `Filter Pruning via Geometric Median for Deep Convolutional Neural Networks Acceleration <https://arxiv.org/abs/1811.00250>`_. 
16    
17       This sparsifier is controlled by three variables: 
18       1. `sparsity_level` defines the number of filters (rows) that are zeroed-out. 
19       2. `dist` defines the distance measurement type. Default: 3 (L2 distance). 
20       Available options are: [1, 2, (custom callable distance function)]. 
21    
22       Note:: 
23           Inputs should be a 4D convolutional tensor of shape (N, C, H, W). 
24               - N: output channels size 
25               - C: input channels size 
26               - H: height of kernel 
27               - W: width of kernel 
28       """
29   
30       def __init__(
31           self, sparsity_level: float = 0.5, dist: Optional[Union[Callable, int]] = None
32       ):
33           defaults = {
34               "sparsity_level": sparsity_level,
35           }
36   
37           if dist is None:
38               dist = 2
39   
40           if callable(dist):
41               self.dist_fn = dist
42           elif dist == 1:
43               self.dist_fn = lambda x: torch.cdist(x, x, p=1)
44           elif dist == 2:
45               self.dist_fn = lambda x: torch.cdist(x, x, p=2)
46           else:
47               raise NotImplementedError("Distance function is not yet implemented.")
48           super().__init__(defaults=defaults)
49   
50       def _compute_distance(self, t):
51           r"""Compute distance across all entries in tensor `t` along all dimension 
52           except for the one identified by dim. 
53           Args: 
54               t (torch.Tensor): tensor representing the parameter to prune 
55           Returns: 
56               distance (torch.Tensor): distance computed across filtters 
57           """
58           dim = 0  # prune filter (row)
59   
60           size = t.size(dim)
61           slc = [slice(None)] * t.dim()
62   
63           # flatten the tensor along the dimension
64           t_flatten = [
65               t[tuple(slc[:dim] + [slice(i, i + 1)] + slc[dim + 1 :])].reshape(-1)
66               for i in range(size)
67           ]
68           t_flatten = torch.stack(t_flatten)
69   
70           # distance measurement
71           dist_matrix = self.dist_fn(t_flatten)
72   
73           # more similar with other filter indicates large in the sum of row
74           distance = torch.sum(torch.abs(dist_matrix), 1)
75   
76           return distance
77   
78       def update_mask(  # type: ignore[override]
79           self, module, tensor_name, sparsity_level, **kwargs
80       ):
81           tensor_weight = getattr(module, tensor_name)
82           mask = getattr(module.parametrizations, tensor_name)[0].mask
83   
84           if sparsity_level <= 0:
85               mask.data = torch.ones_like(mask).bool()
86           elif sparsity_level >= 1.0:
87               mask.data = torch.zeros_like(mask).bool()
88           else:
89               distance = self._compute_distance(tensor_weight)
90   
91               tensor_size = tensor_weight.shape[0]  # prune filter (row)
92               nparams_toprune = round(sparsity_level * tensor_size)
93               nparams_toprune = min(
94                   max(nparams_toprune, 0), tensor_size
95               )  # clamp to [0, tensor_size]
96               topk = torch.topk(distance, k=nparams_toprune, largest=False)
97               mask[topk.indices] = False
