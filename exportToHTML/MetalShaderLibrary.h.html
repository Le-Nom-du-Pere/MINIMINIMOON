35   class has_size_type {
36     template <typename U>
37     static constexpr std::true_type check(typename U::size_type*);
38     template <typename>
39     static constexpr std::false_type check(...);
40   
41    public:
42     static constexpr bool value = decltype(check<T>(nullptr))::value;
43   };

54    public:
55     MetalKernelFunction(MTLComputePipelineState_t cps_, MTLFunction_t f_);
56     ~MetalKernelFunction();
57     MetalKernelFunction(MetalKernelFunction&) = delete;
58     // Shader properties
59     uint64_t getMaxThreadsPerThreadgroup() const;
60     uint64_t getThreadExecutionWidth() const;
61     uint64_t getStaticThreadGroupMemoryLength() const;
62     void runCommandBlock(std::function<void(void)> f);
63     // Methods below should be called from runCommandBlock function
64     void startEncoding();
65     void setArg(unsigned idx, const at::TensorBase& t);
66     void setArg(unsigned idx, const void* ptr, uint64_t size);
67     template <
68         typename T,
69         typename = std::enable_if_t<
70             std::is_integral_v<T> || std::is_same_v<T, float> ||
71             (std::is_class_v<T> && std::is_trivially_copyable_v<T> &&
72              !detail::has_size_type_v<T>)>>
73     inline void setArg(unsigned idx, const T val) {
74       setArg(idx, &val, sizeof(T));
75     }
76   
77     template <
78         typename Container,
79         typename = std::enable_if_t<detail::has_size_type_v<Container>>>
80     inline void setArg(unsigned idx, const Container& values) {
81       setArg(
82           idx,
83           values.data(),
84           values.size() * sizeof(typename Container::value_type));
85     }
86     void dispatch(
87         uint64_t length,
88         std::optional<uint64_t> groupSize = std::nullopt);
89     void dispatch(
90         c10::ArrayRef<uint64_t> length,
91         c10::OptionalArrayRef<uint64_t> groupSize = std::nullopt);
92   
93    private:
94     MTLComputePipelineState_t cps;
95     MTLFunction_t func;
96     MTLComputeCommandEncoder_t encoder = nullptr;
97   };
