20   class AdaptiveLogSoftmaxWithLoss(Module):
21       (
22           """Efficient softmax approximation. 
23    
24       As described in 
25       `Efficient softmax approximation for GPUs by Edouard Grave, Armand Joulin, 
26       Moustapha Ciss\u00e9, David Grangier, and Herv\u00e9 J\u00e9gou 
27       <https://arxiv.org/abs/1609.04309>`__. 
28   """
29           r""" 
30       Adaptive softmax is an approximate strategy for training models with large 
31       output spaces. It is most effective when the label distribution is highly 
32       imbalanced, for example in natural language modelling, where the word 
33       frequency distribution approximately follows the `Zipf's law`_. 
34    
35       Adaptive softmax partitions the labels into several clusters, according to 
36       their frequency. These clusters may contain different number of targets 
37       each. 
38       Additionally, clusters containing less frequent labels assign lower 
39       dimensional embeddings to those labels, which speeds up the computation. 
40       For each minibatch, only clusters for which at least one target is 
41       present are evaluated. 
42    
43       The idea is that the clusters which are accessed frequently 
44       (like the first one, containing most frequent labels), should also be cheap 
45       to compute -- that is, contain a small number of assigned labels. 
46    
47       We highly recommend taking a look at the original paper for more details. 
48    
49       * :attr:`cutoffs` should be an ordered Sequence of integers sorted 
50         in the increasing order. 
51         It controls number of clusters and the partitioning of targets into 
52         clusters. For example setting ``cutoffs = [10, 100, 1000]`` 
53         means that first `10` targets will be assigned 
54         to the 'head' of the adaptive softmax, targets `11, 12, ..., 100` will be 
55         assigned to the first cluster, and targets `101, 102, ..., 1000` will be 
56         assigned to the second cluster, while targets 
57         `1001, 1002, ..., n_classes - 1` will be assigned 
58         to the last, third cluster. 
59    
60       * :attr:`div_value` is used to compute the size of each additional cluster, 
61         which is given as 
62         :math:`\left\lfloor\frac{\texttt{in\_features}}{\texttt{div\_value}^{idx}}\right\rfloor`, 
63         where :math:`idx` is the cluster index (with clusters 
64         for less frequent words having larger indices, 
65         and indices starting from :math:`1`). 
66    
67       * :attr:`head_bias` if set to True, adds a bias term to the 'head' of the 
68         adaptive softmax. See paper for details. Set to False in the official 
69         implementation. 
70    
71       .. warning:: 
72           Labels passed as inputs to this module should be sorted according to 
73           their frequency. This means that the most frequent label should be 
74           represented by the index `0`, and the least frequent 
75           label should be represented by the index `n_classes - 1`. 
76    
77       .. note:: 
78           This module returns a ``NamedTuple`` with ``output`` 
79           and ``loss`` fields. See further documentation for details. 
80    
81       .. note:: 
82           To compute log-probabilities for all classes, the ``log_prob`` 
83           method can be used. 
84    
85       Args: 
86           in_features (int): Number of features in the input tensor 
87           n_classes (int): Number of classes in the dataset 
88           cutoffs (Sequence): Cutoffs used to assign targets to their buckets 
89           div_value (float, optional): value used as an exponent to compute sizes 
90               of the clusters. Default: 4.0 
91           head_bias (bool, optional): If ``True``, adds a bias term to the 'head' of the 
92               adaptive softmax. Default: ``False`` 
93    
94       Returns: 
95           ``NamedTuple`` with ``output`` and ``loss`` fields: 
96               * **output** is a Tensor of size ``N`` containing computed target 
97                 log probabilities for each example 
98               * **loss** is a Scalar representing the computed negative 
99                 log likelihood loss 
100   
101      Shape: 
102          - input: :math:`(N, \texttt{in\_features})` or :math:`(\texttt{in\_features})` 
103          - target: :math:`(N)` or :math:`()` where each value satisfies :math:`0 <= \texttt{target[i]} <= \texttt{n\_classes}` 
104          - output1: :math:`(N)` or :math:`()` 
105          - output2: ``Scalar`` 
106   
107      .. _Zipf's law: https://en.wikipedia.org/wiki/Zipf%27s_law 
108      """
109      )
110  
111      in_features: int
112      n_classes: int
113      cutoffs: list[int]
114      div_value: float
115      head_bias: bool
116      head: Linear
117      tail: ModuleList

119      def __init__(
120          self,
121          in_features: int,
122          n_classes: int,
123          cutoffs: Sequence[int],
124          div_value: float = 4.0,
125          head_bias: bool = False,
126          device=None,
127          dtype=None,
128      ) -> None:
129          factory_kwargs = {"device": device, "dtype": dtype}
130          super().__init__()
131  
132          cutoffs = list(cutoffs)
133  
134          if len(cutoffs) == 0:
135              raise ValueError("cutoffs should be a sequence of length larger than 0")
136  
137          if (
138              (cutoffs != sorted(cutoffs))
139              or (min(cutoffs) <= 0)
140              or (max(cutoffs) > (n_classes - 1))
141              or (len(set(cutoffs)) != len(cutoffs))
142              or any(int(c) != c for c in cutoffs)
143          ):
144              raise ValueError(
145                  "cutoffs should be a sequence of unique, positive "
146                  "integers sorted in an increasing order, where "
147                  "each value is between 1 and n_classes-1"
148              )
149  
150          self.in_features = in_features
151          self.n_classes = n_classes
152          self.cutoffs = cutoffs + [n_classes]
153          self.div_value = div_value
154          self.head_bias = head_bias
155  
156          self.shortlist_size = self.cutoffs[0]
157          self.n_clusters = len(self.cutoffs) - 1
158          self.head_size = self.shortlist_size + self.n_clusters
159  
160          self.head = Linear(
161              self.in_features, self.head_size, bias=self.head_bias, **factory_kwargs
162          )
163          self.tail = ModuleList()
164  
165          for i in range(self.n_clusters):
166              hsz = int(self.in_features // (self.div_value ** (i + 1)))
167              osz = self.cutoffs[i + 1] - self.cutoffs[i]
168  
169              projection = Sequential(
170                  Linear(self.in_features, hsz, bias=False, **factory_kwargs),
171                  Linear(hsz, osz, bias=False, **factory_kwargs),
172              )
173  
174              self.tail.append(projection)
175  
176      def reset_parameters(self) -> None:
177          self.head.reset_parameters()
178          for i2h, h2o in self.tail:  # type: ignore[misc]
179              i2h.reset_parameters()  # type: ignore[has-type]
180              h2o.reset_parameters()  # type: ignore[has-type]

182      def forward(self, input_: Tensor, target_: Tensor) -> _ASMoutput:
183          targ_dim = target_.dim()
184  
185          if targ_dim == 1:
186              if input_.size(0) != target_.size(0):
187                  raise RuntimeError(
188                      "Input and target should have the same size in the batch dimension."
189                  )
190              if input_.dim() != 2:
191                  raise RuntimeError(
192                      "1D target tensor expects 2D input tensors, "
193                      "but found inputs with size",
194                      input_.size(),
195                  )
196          elif targ_dim == 0:
197              if input_.dim() != 1:
198                  raise RuntimeError(
199                      "0D target tensor expects 1D input tensors, "
200                      "but found inputs with size",
201                      input_.size(),
202                  )
203          else:
204              raise RuntimeError(
205                  "0D or 1D target tensor expected, multi-target not supported"
206              )
207  
208          is_batched = targ_dim > 0
209          input = input_ if is_batched else input_.unsqueeze(0)
210          target = target_ if is_batched else target_.unsqueeze(0)
211  
212          used_rows = 0
213          batch_size = target.size(0)
214  
215          output = input.new_zeros(batch_size)
216          gather_inds = target.new_empty(batch_size)
217  
218          cutoff_values = [0] + self.cutoffs
219          for i in range(len(cutoff_values) - 1):
220              low_idx = cutoff_values[i]
221              high_idx = cutoff_values[i + 1]
222  
223              target_mask = (target >= low_idx) & (target < high_idx)
224              row_indices = target_mask.nonzero().squeeze()
225  
226              if row_indices.numel() == 0:
227                  continue
228  
229              if i == 0:
230                  gather_inds.index_copy_(0, row_indices, target[target_mask])
231  
232              else:
233                  relative_target = target[target_mask] - low_idx
234                  input_subset = input.index_select(0, row_indices)
235  
236                  cluster_output = self.tail[i - 1](input_subset)
237                  cluster_index = self.shortlist_size + i - 1
238  
239                  gather_inds.index_fill_(0, row_indices, cluster_index)
240                  cluster_logprob = F.log_softmax(cluster_output, dim=1)
241                  local_logprob = cluster_logprob.gather(1, relative_target.unsqueeze(1))
242                  output.index_copy_(0, row_indices, local_logprob.squeeze(1))
243  
244              used_rows += row_indices.numel()
245  
246          if used_rows != batch_size:
247              raise RuntimeError(
248                  f"Target values should be in [0, {self.n_classes - 1}], "
249                  f"but values in range [{target.min().item()}, {target.max().item()}] "
250                  "were found. "
251              )
252  
253          head_output = self.head(input)
254          head_logprob = F.log_softmax(head_output, dim=1)
255          output += head_logprob.gather(1, gather_inds.unsqueeze(1)).squeeze()
256          loss = (-output).mean()
257  
258          if not is_batched:
259              output = output.squeeze(0)
260  
261          return _ASMoutput(output, loss)
262  
263      def _get_full_log_prob(self, input, head_output):
264          """Given input tensor, and output of ``self.head``, compute the log of the full distribution."""
265          out = input.new_empty((head_output.size(0), self.n_classes))
266          head_logprob = F.log_softmax(head_output, dim=1)
267  
268          out[:, : self.shortlist_size] = head_logprob[:, : self.shortlist_size]
269  
270          for i, (start_idx, stop_idx) in enumerate(zip(self.cutoffs, self.cutoffs[1:])):
271              cluster_output = self.tail[i](input)
272              cluster_logprob = F.log_softmax(cluster_output, dim=1)
273              output_logprob = cluster_logprob + head_logprob[
274                  :, self.shortlist_size + i
275              ].unsqueeze(1)
276  
277              out[:, start_idx:stop_idx] = output_logprob
278  
279          return out

281      def log_prob(self, input: Tensor) -> Tensor:
282          r"""Compute log probabilities for all :math:`\texttt{n\_classes}`. 
283   
284          Args: 
285              input (Tensor): a minibatch of examples 
286   
287          Returns: 
288              log-probabilities of for each class :math:`c` 
289              in range :math:`0 <= c <= \texttt{n\_classes}`, where :math:`\texttt{n\_classes}` is a 
290              parameter passed to ``AdaptiveLogSoftmaxWithLoss`` constructor. 
291   
292          Shape: 
293              - Input: :math:`(N, \texttt{in\_features})` 
294              - Output: :math:`(N, \texttt{n\_classes})` 
295   
296          """
297          head_output = self.head(input)
298          return self._get_full_log_prob(input, head_output)

300      def predict(self, input: Tensor) -> Tensor:
301          r"""Return the class with the highest probability for each example in the input minibatch. 
302   
303          This is equivalent to ``self.log_prob(input).argmax(dim=1)``, but is more efficient in some cases. 
304   
305          Args: 
306              input (Tensor): a minibatch of examples 
307   
308          Returns: 
309              output (Tensor): a class with the highest probability for each example 
310   
311          Shape: 
312              - Input: :math:`(N, \texttt{in\_features})` 
313              - Output: :math:`(N)` 
314          """
315          head_output = self.head(input)
316          output = torch.argmax(head_output, dim=1)
317          not_in_shortlist = output >= self.shortlist_size
318          all_in_shortlist = not (not_in_shortlist.any())
319  
320          if all_in_shortlist:
321              return output
322  
323          elif not_in_shortlist.all():
324              log_prob = self._get_full_log_prob(input, head_output)
325              return torch.argmax(log_prob, dim=1)
326  
327          else:
328              log_prob = self._get_full_log_prob(
329                  input[not_in_shortlist], head_output[not_in_shortlist]
330              )
331              output[not_in_shortlist] = torch.argmax(log_prob, dim=1)
332              return output
