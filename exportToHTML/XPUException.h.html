8    static inline sycl::async_handler asyncHandler = [](sycl::exception_list el) {
9      if (el.size() == 0) {
10       return;
11     }
12     for (const auto& e : el) {
13       try {
14         std::rethrow_exception(e);
15       } catch (sycl::exception& e) {
16         TORCH_WARN("SYCL Exception: ", e.what());
17       }
18     }
19     throw;
20   };
