3    """ 
4    Utilities for reproducing and debugging issues in PyTorch's Dynamo AOT compilation. 
5     
6    This module provides tools and infrastructure for: 
7    1. Generating minimal reproducible test cases ("repros") from failing compilations 
8    2. Analyzing accuracy issues between eager and compiled execution 
9    3. Minifying large models/inputs to isolate problematic patterns 
10   4. Debugging compiler errors and accuracy divergences 
11    
12   The main components include: 
13   - Repro generation: Creates standalone Python files that reproduce compiler issues 
14   - Minification: Reduces large graphs to minimal failing examples 
15   - Accuracy analysis: Compares compiled vs eager execution, with fp64 reference 
16   - Debug tools: Dumps graph state, tracks intermediates, analyzes divergences 
17    
18   This is primarily used by PyTorch developers and researchers to debug issues in 
19   the Dynamo AOT compilation pipeline, particularly for the Inductor backend. 
20   """

91   def wrap_compiler_debug(
92       unconfigured_compiler_fn: "_CompileFxCallable",
93       compiler_name: str,
94   ) -> "_CompileFxCallable":
95       """ 
96       Minifier for Fx Graph modules after Aot Autograd has finished. We wrap both 
97       forward and backward call separately with the backend compiler_fn - like 
98       inductor or nvfuser. Intercepting after Aot Autograd presents neat 
99       abstraction, where all the params are lifted as graph inputs, making it easy 
100      to save the graph as a string. 
101      """
102  
103      @functools.wraps(unconfigured_compiler_fn)
104      def debug_wrapper(
105          gm: torch.fx.GraphModule,
106          example_inputs: Sequence["InputType"],
107          **kwargs: Unpack["_CompileFxKwargs"],
108      ) -> OutputCode:
109          from torch._subclasses import FakeTensorMode
110  
111          compiler_fn = functools.partial(unconfigured_compiler_fn, **kwargs)
112  
113          from torch._functorch.aot_autograd import get_aot_graph_name
114  
115          graph_name = get_aot_graph_name()
116  
117          # TODO: why do we need to deepcopy the original graph?
118          orig_graph = copy.deepcopy(gm.graph)
119          assert config.repro_after in ("dynamo", "aot", None)
120  
121          try:
122              # Call the compiler_fn - which is either aot_autograd or inductor
123              # with fake inputs
124              inner_compiled_fn = compiler_fn(gm, example_inputs)
125          except Exception:
126              # TODO: Failures here are troublesome because no real inputs,
127              # need a different serialization strategy
128              if config.repro_after == "aot":
129                  if config.repro_level == 1:
130                      dump_compiler_graph_state(
131                          fx.GraphModule(gm, orig_graph),
132                          example_inputs,
133                          compiler_name,
134                      )
135                  elif config.repro_level == 2:
136                      dump_to_minify(
137                          fx.GraphModule(gm, orig_graph),
138                          example_inputs,
139                          compiler_name,
140                      )
141                  log.error("CompilerError")
142              raise
143  
144          # We may run regular PyTorch compute that may trigger Dynamo, do NOT
145          # recursively attempt to accuracy minify in that case!
146          def deferred_for_real_inputs(
147              real_inputs: Sequence["InputType"], **_kwargs: object
148          ) -> Any:
149              # This is a bit obscure: if we recursively try to accuracy minify
150              # the SAME function, this would trigger.  But most of the time
151              # we should never hit this branch
152              assert not _kwargs
153              if config.repro_after != "aot":
154                  assert not isinstance(inner_compiled_fn, str)
155                  return inner_compiled_fn(real_inputs)
156              with config.patch(repro_after=None):
157                  return inner_debug_fn(real_inputs)
158  
159          def inner_debug_fn(real_inputs):
160              """ 
161              Aot Autograd fw_compiler and bw_compiler can have fake tensors. So, 
162              example_inputs can be fake tensors. We can call compiler_fn (which is 
163              inductor or nvfuser) with fake tensors but the actually compiled_fn 
164              should be called with real tensors. Therefore, the actual invocation 
165              is deferred. 
166              """
167              # Copy the tensor attrs like shape, stride etc by converting to Fake Tensor
168              # because inductor clears the tensor list in its codegen. And example_inputs
169              # are available only for the first invocation.
170              fake_mode = FakeTensorMode()
171              copy_tensor_attrs = [
172                  fake_mode.from_tensor(x) if isinstance(x, torch.Tensor) else x
173                  for x in real_inputs
174              ]
175              if config.repro_level == 3:
176                  # Always dump the original module in case we have segfaults
177                  dump_to_minify(
178                      fx.GraphModule(gm, orig_graph), real_inputs, compiler_name
179                  )
180  
181              if config.repro_level == 4:
182                  if compiler_name != "inductor":
183                      raise NotImplementedError(
184                          "Accuracy minification is supported for inductor only"
185                      )
186                  failed = not same_two_models(
187                      gm,
188                      inner_compiled_fn,
189                      real_inputs,
190                      only_fwd=True,
191                      ignore_non_fp=config.repro_ignore_non_fp,
192                  )
193  
194                  if failed:
195                      log.warning(
196                          "Accuracy failed for the AOT Autograd graph %s", graph_name
197                      )
198                      dump_compiler_graph_state(
199                          fx.GraphModule(gm, orig_graph),
200                          real_inputs,
201                          f"{compiler_name}_accuracy",
202                      )
203                      dump_to_minify(
204                          fx.GraphModule(gm, orig_graph),
205                          real_inputs,
206                          f"{compiler_name}_accuracy",
207                      )
208                      raise AccuracyError("Bad accuracy detected")
209                  else:
210                      # Call the compiled function with real inputs
211                      return inner_compiled_fn(real_inputs)  # type: ignore[operator]
212              else:
213                  try:
214                      # Call the compiled function with real inputs
215                      out = inner_compiled_fn(real_inputs)  # type: ignore[operator]
216                      # sync cuda kernels to ensure IMA detection
217                      for arg in example_inputs:
218                          if isinstance(arg, torch.Tensor) and arg.is_cuda:
219                              torch.cuda.synchronize()
220                              break
221                      return out
222                  except Exception:
223                      if config.repro_level == 1:
224                          dump_compiler_graph_state(
225                              fx.GraphModule(gm, orig_graph),
226                              copy_tensor_attrs,
227                              compiler_name,
228                          )
229                      elif config.repro_level == 2:
230                          dump_to_minify(
231                              fx.GraphModule(gm, orig_graph),
232                              copy_tensor_attrs,
233                              compiler_name,
234                          )
235                      raise
236  
237          if config.repro_after == "aot":
238              compiled_fn = deferred_for_real_inputs
239              compiled_fn._boxed_call = True  # type: ignore[attr-defined]
240              return compiled_fn  # type: ignore[return-value]
241          else:
242              return inner_compiled_fn
243  
244      return debug_wrapper

284  def generate_compiler_repro_string(
285      gm, args, *, stable_output=False, save_dir=None, stable_hash=False
286  ):
287      model_str = textwrap.dedent(
288          f"""
289  {generate_env_vars_string(stable_output=stable_output)}
290  import torch
291  from torch import tensor, device
292  import torch.fx as fx
293  from torch._dynamo.testing import rand_strided
294  from math import inf
295  import torch._inductor.inductor_prims
296  
297  {generate_config_string(stable_output=stable_output)}
298  
299  isolate_fails_code_str = None
300  
301  {extra_imports}
302  
303  {maybe_fbcode_instructions()}
304          """
305      )
306      if not stable_output:
307          model_str += f"# torch version: {torch.version.__version__}\n"
308          if hasattr(torch.version, "cuda"):
309              model_str += f"# torch cuda version: {torch.version.cuda}\n"
310          if hasattr(torch.version, "git_version"):
311              model_str += f"# torch git version: {torch.version.git_version}\n\n\n"
312          model_str += _cuda_system_info_comment()
313  
314      model_str += NNModuleToString.convert(gm)
315  
316      # get hint shape/stride when dynamic shape enabled
317      def hint_if_symint(x):
318          return tuple(i.node.hint if isinstance(i, torch.SymInt) else i for i in x)
319  
320      writer = InputWriter(save_dir, stable_hash=stable_hash)
321      for placeholder, arg in zip(fx_placeholder_targets(gm), args):
322          if isinstance(arg, (int, torch.SymInt)):
323              writer.symint(placeholder, arg)
324          elif isinstance(arg, torch.Tensor):
325              # TODO: improve these names with FQN
326              writer.tensor(placeholder, arg)
327          elif arg is None:
328              writer.const(placeholder)
329          else:
330              # It's better to produce a slightly wrong repro string than none
331              # at all
332              writer.unsupported(placeholder, arg)
333  
334      model_str += "\n".join(writer.lines()) + "\n"
335  
336      model_str += "mod = Repro()\n"
337      return model_str

340  def save_graph_repro(
341      fd,
342      gm,
343      args,
344      compiler_name,
345      *,
346      stable_output=False,
347      save_dir=None,
348      command="run",
349      accuracy=None,
350      tracing_mode=None,
351      check_str=None,
352      stable_hash=False,
353  ):
354      if any(
355          isinstance(arg, torch.fx.experimental._backward_state.BackwardState)
356          for arg in args
357      ):
358          fd.write(
359              "Repro is not generated due to existence of BackwardState in graph input"
360          )
361          return
362  
363      fd.write(
364          generate_compiler_repro_string(
365              gm,
366              args,
367              stable_output=stable_output,
368              save_dir=save_dir,
369              stable_hash=stable_hash,
370          )
371      )
372      if accuracy is None:
373          accuracy = "_accuracy" in compiler_name
374      if tracing_mode is None:
375          tracing_mode = "real"
376          if any(
377              has_free_symbols(a) for a in args if not isinstance(a, FakeScriptObject)
378          ):
379              tracing_mode = "symbolic"
380      fd.write("if __name__ == '__main__':\n")
381      fd.write("    from torch._dynamo.repro.after_aot import run_repro\n")
382      fd.write(
383          f"    with torch.no_grad():\n"
384          f"        run_repro(mod, load_args, accuracy={accuracy!r}, command={command!r}, "
385          f"save_dir={save_dir!r}, tracing_mode={tracing_mode!r}, check_str={check_str!r})\n"
386          f"        # To run it separately, do \n"
387          f"        # mod, args = run_repro(mod, load_args, accuracy={accuracy!r}, command='get_args', "
388          f"save_dir={save_dir!r}, tracing_mode={tracing_mode!r}, check_str={check_str!r})\n"
389          f"        # mod(*args)"
390      )

393  def dump_compiler_graph_state(gm, args, compiler_name, *, accuracy=None):
394      subdir = os.path.join(minifier_dir(), "checkpoints")
395      if not os.path.exists(subdir):
396          os.makedirs(subdir, exist_ok=True)
397      file_name = os.path.join(subdir, f"{len(gm.graph.nodes)}.py")
398      log.warning(
399          "Writing checkpoint with %s nodes to %s", len(gm.graph.nodes), file_name
400      )
401      with open(file_name, "w") as fd:
402          save_graph_repro(
403              fd, gm, args, compiler_name, save_dir=subdir, accuracy=accuracy
404          )
405      curdir = os.getcwd()
406      repro_path = os.path.join(curdir, "repro.py")
407      try:
408          shutil.copyfile(file_name, repro_path)
409          log.warning("Copying repro file for convenience to %s", repro_path)
410          if use_buck:
411              BuckTargetWriter(file_name).write()
412      except OSError:
413          log.warning("No write permissions for %s", repro_path)

432      fx_g,
433      args,
434      compiler_name: str,
435      env=None,
436      save_dir=None,
437      accuracy=None,
438      tracing_mode=None,
439      check_str=None,
440  ):
441      if env is None:
442          env = {}
443      subdir = os.path.join(os.getcwd(), "isolate")
444      if not os.path.exists(subdir):
445          os.makedirs(subdir, exist_ok=True)
446      file_name = os.path.join(subdir, f"{str(uuid.uuid4())[:5]}.py")
447      with open(file_name, "w") as fd:
448          save_graph_repro(
449              fd,
450              fx_g,
451              args,
452              compiler_name,
453              save_dir=save_dir,
454              command="minifier-query",
455              accuracy=accuracy,
456              tracing_mode=tracing_mode,
457              check_str=check_str,
458          )
459      # with open(file_name, "r") as fd:
460      #     print(fd.read())
461      new_env = os.environ.copy()
462      new_env = {**new_env, **env}
463      stdout, stderr = TemporaryFile(), TemporaryFile()
464  
465      if use_buck:
466          cmd = BuckTargetWriter(file_name).write(print_msg=False)
467      else:
468          cmd = [sys.executable, file_name]
469  
470      p = subprocess.Popen(
471          cmd,
472          cwd=subdir,
473          stdout=stdout,
474          stderr=stderr,
475          env=new_env,
476      )
477      p.wait()
478  
479      stdout.seek(0)
480      stderr.seek(0)
481      print(
482          textwrap.indent(stdout.read().decode("utf-8"), prefix=">>  "), file=sys.stdout
483      )
484      print(
485          textwrap.indent(stderr.read().decode("utf-8"), prefix=">>  "), file=sys.stderr
486      )
487      # print(f"Isolated test failed - {file_name}")
488      return p.returncode != 0

496  def inductor_fails(fx_g, args, check_str=None):
497      has_cuda = False
498      for arg in args:
499          if isinstance(arg, torch.Tensor) and arg.is_cuda:
500              has_cuda = True
501              break
502  
503      def sync():
504          if has_cuda:
505              # Ensures that segfaults are surfaced
506              torch.cuda.synchronize()
507  
508      from torch._inductor.compile_fx import compile_fx_inner
509  
510      try:
511          result = fx_g(*args)
512          assert isinstance(result, (tuple, list))
513          assert not any(isinstance(x, (tuple, list)) for x in result)
514      except Exception:
515          return False
516  
517      sync()
518  
519      try:
520          compile_mod = compile_fx_inner(fx_g, args)
521          assert not isinstance(compile_mod, str)
522          compile_mod(args)
523          sync()
524      except Exception as e:
525          if check_str is not None and check_str not in repr(e):
526              return False
527          print(repr(e))
528          return True
529      return False

554  def repro_common(options, mod, load_args):
555      # Invariant for graphs we generate with the repro script
556      assert not any(mod.named_parameters())
557      for n, b in mod.named_buffers():
558          if b.numel() > MAX_CONSTANT_NUMEL_INLINE:
559              log.warning(
560                  "Constant %s was not serialized, generated random data instead. "
561                  "If you think this is affecting you, please comment on "
562                  "https://github.com/pytorch/pytorch/issues/100468",
563                  n,
564              )
565  
566      if not hasattr(load_args, "_version"):
567          log.warning(
568              "load_args does not have a _version attribute, please file a bug to PyTorch "
569              "and describe how you generate this repro script"
570          )
571      else:
572          if load_args._version > 0:
573              log.warning(
574                  "load_args is version %s, but this version of PyTorch only supports "
575                  "version 0.  We will try to run it anyway but there may be an incompatibility; "
576                  "if so, try upgrading your version of PyTorch.",
577                  load_args._version,
578              )
579  
580      nop_reader = NopInputReader()
581      load_args(nop_reader)
582  
583      with tqdm(desc="Loading inputs", total=nop_reader.total) as pbar:
584          input_reader = InputReader(save_dir=options.save_dir, pbar=pbar)
585          load_args(input_reader)
586          args = input_reader.args
587  
588      # Turn mod into a GraphModule the slow way
589      # TODO: speed this up
590      mod = make_fx(mod, tracing_mode=options.tracing_mode)(*args)
591  
592      torch._inductor.config.generate_intermediate_hooks = True
593  
594      return mod, args

597  ACCURACY_FAILS: dict[str, Callable[[nn.Module, Any], bool]] = {
598      "": inductor_fails,
599      # This might look inverted but it's not.  strict_accuracy means "we will
600      # minify any time we see anything that diverges", whereas accuracy is more
601      # conservative, and will only minify if there is a meaningful fp64
602      # divergence
603      "accuracy": functools.partial(
604          inductor_accuracy_fails, require_fp64=True, ignore_non_fp=True
605      ),
606      "strict_accuracy": inductor_accuracy_fails,
607  }

610  def repro_minifier_query(options, mod, load_args):
611      mod, args = repro_common(options, mod, load_args)
612      fail_fn = functools.partial(
613          ACCURACY_FAILS[options.accuracy],
614          check_str=options.check_str,  # type: ignore[call-arg]
615      )
616      if fail_fn(mod, args):
617          sys.exit(1)
618      else:
619          sys.exit(0)

622  def repro_minify(options, mod, load_args):
623      from functorch.compile import minifier
624  
625      mod, args = repro_common(options, mod, load_args)
626      compiler_name = "inductor_accuracy" if options.accuracy != "" else "inductor"
627  
628      favored_device = 1 if torch.cuda.device_count() >= 2 else 0
629      env_variables = {"CUDA_VISIBLE_DEVICES": str(favored_device)}
630  
631      module_fails: Any
632      if options.isolate:
633          module_fails = functools.partial(
634              isolate_fails,
635              env=env_variables,
636              compiler_name=compiler_name,
637              save_dir=options.save_dir,
638              accuracy=options.accuracy,
639              tracing_mode=options.tracing_mode,
640          )
641      else:
642          module_fails = ACCURACY_FAILS[options.accuracy]
643  
644      minifier(
645          mod,
646          args,
647          module_fails=functools.partial(module_fails, check_str=options.check_str),
648          dump_state=functools.partial(
649              dump_compiler_graph_state, compiler_name=compiler_name
650          ),
651          save_dir=options.save_dir,
652          offload_to_disk=options.offload_to_disk,
653          skip_offload=options.skip_saving_eager_intermediates,
654          skip_sanity=options.skip_sanity,
655          max_granularity=options.max_granularity,
656      )

659  def repro_analyze(options, mod, load_args):
660      from torch._inductor.compile_fx import compile_fx_inner
661      from torch._inductor.hooks import intermediate_hook
662  
663      mod, args = repro_common(options, mod, load_args)
664  
665      # TODO: The logic for cloning inputs/models here is intentionally
666      # modeled off of run_fwd_maybe_bwd, but arguably it is better not to
667      # clone inputs (as you are doubling your effective GPU memory usage).
668      # It is certainly faster though!  It probably makes sense to let the
669      # user specify the offload strategy.
670  
671      with tqdm(desc="Compiling"):
672          compiled = compile_fx_inner(mod, args)
673      total = counters["inductor"]["intermediate_hooks"]
674  
675      known_names = set()
676  
677      def save_hook(name, val):
678          known_names.add(name)
679          if not options.skip_saving_inductor_intermediates:
680              writer.write_tensor(os.path.join("inductor", name), val)
681          pbar.update(1)  # type: ignore[has-type]
682  
683      writer = torch.utils._content_store.ContentStoreWriter(
684          options.save_dir, stable_hash=options.stable_hash
685      )
686      reader = torch.utils._content_store.ContentStoreReader(options.save_dir)
687  
688      new_args = clone_inputs(args)
689      with (
690          intermediate_hook(save_hook),
691          tqdm(desc="Saving inductor intermediates", total=total) as pbar,
692      ):
693          assert not isinstance(compiled, str)
694          compiled(new_args)
695          assert not new_args
696  
697      def compare_tuples(tuple1, tuple2):
698          diff_indices = [i for i in range(len(tuple1)) if tuple1[i] != tuple2[i]]
699          diff_values = [(tuple1[i], tuple2[i]) for i in diff_indices]
700  
701          if not diff_values:
702              return None
703          else:
704              return " and ".join(f"{a} != {b}" for a, b in diff_values)
705  
706      def check_hook(name, val):
707          meta = writer.compute_tensor_metadata(val)
708          meta2 = reader.read_tensor_metadata(os.path.join("inductor", name))
709          reason = compare_tuples(meta, meta2)
710          if reason is not None:
711              pbar.write(f"NONDETERMINISTIC INDUCTOR at {name} ({reason})")
712          pbar.update(1)
713  
714      if not options.skip_check_deterministic:
715          new_args = clone_inputs(args)
716          with (
717              intermediate_hook(check_hook),
718              tqdm(desc="Checking inductor determinism", total=total) as pbar,
719          ):
720              compiled(new_args)
721              assert not new_args
722  
723      class WriterInterp(fx.Interpreter):
724          def __init__(self, mod, subdir) -> None:
725              super().__init__(mod)
726              self.subdir = subdir
727  
728          def run_node(self, n):
729              r = super().run_node(n)
730              name = n.name
731              if name in known_names:
732                  pbar.update(1)
733                  writer.write_tensor(os.path.join(self.subdir, name), r)
734              return r
735  
736      # NB: the module cast doesn't actually do anything, since there are no
737      # parameters/buffers on the module
738      if not options.skip_saving_float64_intermediates:
739          new_mod, new_args = cast_to_fp64(copy.deepcopy(mod), clone_inputs(args))
740          with tqdm(desc="Saving float64 intermediates", total=total) as pbar:
741              WriterInterp(new_mod, "float64").boxed_run(new_args)
742          assert not new_args
743  
744      class ExactReaderInterp(fx.Interpreter):
745          def run_node(self, n):
746              r = super().run_node(n)
747              name = n.name
748              if name in known_names:
749                  meta = writer.compute_tensor_metadata(r)
750                  meta2 = reader.read_tensor_metadata(os.path.join("float64", name))
751                  reason = compare_tuples(meta, meta2)
752                  if reason is not None:
753                      pbar.write(f"NONDETERMINISTIC FLOAT64 at {name} ({reason})")
754                  pbar.update(1)
755              return r
756  
757      # TODO: check eager determinism
758  
759      if not options.skip_check_deterministic:
760          new_mod, new_args = cast_to_fp64(copy.deepcopy(mod), clone_inputs(args))
761          with tqdm(desc="Checking float64 determinism", total=total) as pbar:
762              ExactReaderInterp(new_mod).boxed_run(new_args)
763              assert not new_args
764  
765      # Now that we've saved everything, interp through the eager graph
766      # and do comparisons
767      class ReaderInterp(fx.Interpreter):
768          def run_node(self, n):
769              r = super().run_node(n)
770              name = n.name
771              if name in known_names:
772                  inductor = reader.read_tensor(os.path.join("inductor", name))
773                  float64 = reader.read_tensor(os.path.join("float64", name))
774                  logged = False
775  
776                  def log_error(msg, *args):
777                      nonlocal logged
778                      logged = True
779                      pbar.write(f"DIVERGED at {name}: {msg % args}")
780  
781                  if not same(
782                      r,
783                      inductor,
784                      float64,
785                      tol=torch._dynamo.config.repro_tolerance,
786                      equal_nan=True,
787                      log_error=log_error,
788                  ):
789                      assert logged
790                  pbar.update(1)
791              return r
792  
793      with tqdm(desc="Checking divergence", total=total) as pbar:
794          ReaderInterp(mod).boxed_run(args)
795      assert not args
796

798  def repro_get_args(options, mod, load_args):
799      mod, args = repro_common(options, mod, load_args)
800      return mod, args

803  def repro_run(options, mod, load_args):
804      from torch._inductor.compile_fx import compile_fx_inner
805  
806      mod, args = repro_common(options, mod, load_args)
807  
808      from torch.cuda import synchronize
809  
810      compiled = compile_fx_inner(mod, args)
811      assert not isinstance(compiled, str)
812  
813      if options.accuracy != "":
814          # We don't really respect --accuracy vs --strict-accuracy here, it
815          # seems counterintuitive
816          if not same_two_models(
817              mod,
818              compiled,
819              args,
820              only_fwd=True,
821              ignore_non_fp=config.repro_ignore_non_fp,
822          ):
823              raise AccuracyError("Bad accuracy detected")
824      else:
825          need_sync = False
826  
827          for arg in args:
828              if isinstance(arg, torch.Tensor) and arg.is_cuda:
829                  need_sync = True
830                  break
831  
832          compiled(list(args))
833  
834          if need_sync:
835              synchronize()  # ensure segfaults are surfaced
