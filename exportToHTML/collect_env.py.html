90   def run(command):
91       """Return (return-code, stdout, stderr)."""
92       shell = True if type(command) is str else False
93       p = subprocess.Popen(command, stdout=subprocess.PIPE,
94                            stderr=subprocess.PIPE, shell=shell)
95       raw_output, raw_err = p.communicate()
96       rc = p.returncode
97       if get_platform() == 'win32':
98           enc = 'oem'
99       else:
100          enc = locale.getpreferredencoding()
101      output = raw_output.decode(enc)
102      err = raw_err.decode(enc)
103      return rc, output.strip(), err.strip()

106  def run_and_read_all(run_lambda, command):
107      """Run command using run_lambda; reads and returns entire output if rc is 0."""
108      rc, out, _ = run_lambda(command)
109      if rc != 0:
110          return None
111      return out

114  def run_and_parse_first_match(run_lambda, command, regex):
115      """Run command using run_lambda, returns the first regex match if it exists."""
116      rc, out, _ = run_lambda(command)
117      if rc != 0:
118          return None
119      match = re.search(regex, out)
120      if match is None:
121          return None
122      return match.group(1)

125      """Run command using run_lambda and returns first line if output is not empty."""
126      rc, out, _ = run_lambda(command)
127      if rc != 0:
128          return None
129      return out.split('\n')[0]

167  def get_gpu_info(run_lambda):
168      if get_platform() == 'darwin' or (TORCH_AVAILABLE and hasattr(torch.version, 'hip') and torch.version.hip is not None):
169          if TORCH_AVAILABLE and torch.cuda.is_available():
170              if torch.version.hip is not None:
171                  prop = torch.cuda.get_device_properties(0)
172                  if hasattr(prop, "gcnArchName"):
173                      gcnArch = " ({})".format(prop.gcnArchName)
174                  else:
175                      gcnArch = "NoGCNArchNameOnOldPyTorch"
176              else:
177                  gcnArch = ""
178              return torch.cuda.get_device_name(None) + gcnArch
179          return None
180      smi = get_nvidia_smi()
181      uuid_regex = re.compile(r' \(UUID: .+?\)')
182      rc, out, _ = run_lambda(smi + ' -L')
183      if rc != 0:
184          return None
185      # Anonymize GPUs by removing their UUID
186      return re.sub(uuid_regex, '', out)

193  def get_cudnn_version(run_lambda):
194      """Return a list of libcudnn.so; it's hard to tell which one is being used."""
195      if get_platform() == 'win32':
196          system_root = os.environ.get('SYSTEMROOT', 'C:\\Windows')
197          cuda_path = os.environ.get('CUDA_PATH', "%CUDA_PATH%")
198          where_cmd = os.path.join(system_root, 'System32', 'where')
199          cudnn_cmd = '{} /R "{}\\bin" cudnn*.dll'.format(where_cmd, cuda_path)
200      elif get_platform() == 'darwin':
201          # CUDA libraries and drivers can be found in /usr/local/cuda/. See
202          # https://docs.nvidia.com/cuda/archive/9.0/cuda-installation-guide-mac-os-x/index.html#installation
203          # https://docs.nvidia.com/deeplearning/cudnn/installation/latest/
204          # Use CUDNN_LIBRARY when cudnn library is installed elsewhere.
205          cudnn_cmd = 'ls /usr/local/cuda/lib/libcudnn*'
206      else:
207          cudnn_cmd = 'ldconfig -p | grep libcudnn | rev | cut -d" " -f1 | rev'
208      rc, out, _ = run_lambda(cudnn_cmd)
209      # find will return 1 if there are permission errors or if not found
210      if len(out) == 0 or (rc != 1 and rc != 0):
211          l = os.environ.get('CUDNN_LIBRARY')
212          if l is not None and os.path.isfile(l):
213              return os.path.realpath(l)
214          return None
215      files_set = set()
216      for fn in out.split('\n'):
217          fn = os.path.realpath(fn)  # eliminate symbolic links
218          if os.path.isfile(fn):
219              files_set.add(fn)
220      if not files_set:
221          return None
222      # Alphabetize the result because the order is non-deterministic otherwise
223      files = sorted(files_set)
224      if len(files) == 1:
225          return files[0]
226      result = '\n'.join(files)
227      return 'Probably one of the following:\n{}'.format(result)

230  def get_nvidia_smi():
231      # Note: nvidia-smi is currently available only on Windows and Linux
232      smi = 'nvidia-smi'
233      if get_platform() == 'win32':
234          system_root = os.environ.get('SYSTEMROOT', 'C:\\Windows')
235          program_files_root = os.environ.get('PROGRAMFILES', 'C:\\Program Files')
236          legacy_path = os.path.join(program_files_root, 'NVIDIA Corporation', 'NVSMI', smi)
237          new_path = os.path.join(system_root, 'System32', smi)
238          smis = [new_path, legacy_path]
239          for candidate_smi in smis:
240              if os.path.exists(candidate_smi):
241                  smi = '"{}"'.format(candidate_smi)
242                  break
243      return smi

321  def get_cpu_info(run_lambda):
322      rc, out, err = 0, '', ''
323      if get_platform() == 'linux':
324          rc, out, err = run_lambda('lscpu')
325      elif get_platform() == 'win32':
326          rc, out, err = run_lambda(
327              'powershell.exe "gwmi -Class Win32_Processor | Select-Object -Property Name,Manufacturer,Family,\ 
328              Architecture,ProcessorType,DeviceID,CurrentClockSpeed,MaxClockSpeed,L2CacheSize,L2CacheSpeed,Revision\ 
329              | ConvertTo-Json"'
330          )
331          if rc == 0:
332              lst = []
333              try:
334                  obj = json.loads(out)
335                  if type(obj) is list:
336                      for o in obj:
337                          lst.append("----------------------")
338                          lst.extend([f"{k}: {v}" for (k, v) in o.items()])
339                  else:
340                      lst.extend([f"{k}: {v}" for (k, v) in obj.items()])
341              except ValueError as e:
342                  lst.append(out)
343                  lst.append(str(e))
344              out = "\n".join(lst)
345      elif get_platform() == 'darwin':
346          rc, out, err = run_lambda("sysctl -n machdep.cpu.brand_string")
347      cpu_info = 'None'
348      if rc == 0:
349          cpu_info = out
350      else:
351          cpu_info = err
352      return cpu_info

355  def get_platform():
356      if sys.platform.startswith('linux'):
357          return 'linux'
358      elif sys.platform.startswith('win32'):
359          return 'win32'
360      elif sys.platform.startswith('cygwin'):
361          return 'cygwin'
362      elif sys.platform.startswith('darwin'):
363          return 'darwin'
364      else:
365          return sys.platform

395  def get_os(run_lambda):
396      from platform import machine
397      platform = get_platform()
398  
399      if platform == 'win32' or platform == 'cygwin':
400          return get_windows_version(run_lambda)
401  
402      if platform == 'darwin':
403          version = get_mac_version(run_lambda)
404          if version is None:
405              return None
406          return 'macOS {} ({})'.format(version, machine())
407  
408      if platform == 'linux':
409          # Ubuntu/Debian based
410          desc = get_lsb_version(run_lambda)
411          if desc is not None:
412              return '{} ({})'.format(desc, machine())
413  
414          # Try reading /etc/*-release
415          desc = check_release_file(run_lambda)
416          if desc is not None:
417              return '{} ({})'.format(desc, machine())
418  
419          return '{} ({})'.format(platform, machine())
420  
421      # Unknown platform
422      return platform

437  def get_pip_packages(run_lambda, patterns=None):
438      """Return `pip list` output. Note: will also find conda-installed pytorch and numpy packages."""
439      if patterns is None:
440          patterns = PIP_PATTERNS + COMMON_PATTERNS + NVIDIA_PATTERNS
441  
442      pip_version = 'pip3' if sys.version_info.major == 3 else 'pip'
443  
444      os.environ['PIP_DISABLE_PIP_VERSION_CHECK'] = '1'
445      # People generally have pip as `pip` or `pip3`
446      # But here it is invoked as `python -mpip`
447      out = run_and_read_all(run_lambda, [sys.executable, '-mpip', 'list', '--format=freeze'])
448      if out is None:
449          return pip_version, out
450  
451      filtered_out = '\n'.join(
452          line
453          for line in out.splitlines()
454          if any(name in line for name in patterns)
455      )
456  
457      return pip_version, filtered_out

483  def get_env_info():
484      """ 
485      Collects environment information to aid in debugging. 
486   
487      The returned environment information contains details on torch version, is debug build 
488      or not, cuda compiled version, gcc version, clang version, cmake version, operating 
489      system, libc version, python version, python platform, CUDA availability, CUDA 
490      runtime version, CUDA module loading config, GPU model and configuration, Nvidia 
491      driver version, cuDNN version, pip version and versions of relevant pip and 
492      conda packages, HIP runtime version, MIOpen runtime version, 
493      Caching allocator config, XNNPACK availability and CPU information. 
494   
495      Returns: 
496          SystemEnv (namedtuple): A tuple containining various environment details 
497              and system information. 
498      """
499      run_lambda = run
500      pip_version, pip_list_output = get_pip_packages(run_lambda)
501  
502      if TORCH_AVAILABLE:
503          version_str = torch.__version__
504          debug_mode_str = str(torch.version.debug)
505          cuda_available_str = str(torch.cuda.is_available())
506          cuda_version_str = torch.version.cuda
507          if not hasattr(torch.version, 'hip') or torch.version.hip is None:  # cuda version
508              hip_compiled_version = hip_runtime_version = miopen_runtime_version = 'N/A'
509          else:  # HIP version
510              def get_version_or_na(cfg, prefix):
511                  _lst = [s.rsplit(None, 1)[-1] for s in cfg if prefix in s]
512                  return _lst[0] if _lst else 'N/A'
513  
514              cfg = torch._C._show_config().split('\n')
515              hip_runtime_version = get_version_or_na(cfg, 'HIP Runtime')
516              miopen_runtime_version = get_version_or_na(cfg, 'MIOpen')
517              cuda_version_str = 'N/A'
518              hip_compiled_version = torch.version.hip
519      else:
520          version_str = debug_mode_str = cuda_available_str = cuda_version_str = 'N/A'
521          hip_compiled_version = hip_runtime_version = miopen_runtime_version = 'N/A'
522  
523      sys_version = sys.version.replace("\n", " ")
524  
525      conda_packages = get_conda_packages(run_lambda)
526  
527      return SystemEnv(
528          torch_version=version_str,
529          is_debug_build=debug_mode_str,
530          python_version='{} ({}-bit runtime)'.format(sys_version, sys.maxsize.bit_length() + 1),
531          python_platform=get_python_platform(),
532          is_cuda_available=cuda_available_str,
533          cuda_compiled_version=cuda_version_str,
534          cuda_runtime_version=get_running_cuda_version(run_lambda),
535          cuda_module_loading=get_cuda_module_loading_config(),
536          nvidia_gpu_models=get_gpu_info(run_lambda),
537          nvidia_driver_version=get_nvidia_driver_version(run_lambda),
538          cudnn_version=get_cudnn_version(run_lambda),
539          hip_compiled_version=hip_compiled_version,
540          hip_runtime_version=hip_runtime_version,
541          miopen_runtime_version=miopen_runtime_version,
542          pip_version=pip_version,
543          pip_packages=pip_list_output,
544          conda_packages=conda_packages,
545          os=get_os(run_lambda),
546          libc_version=get_libc_version(),
547          gcc_version=get_gcc_version(run_lambda),
548          clang_version=get_clang_version(run_lambda),
549          cmake_version=get_cmake_version(run_lambda),
550          caching_allocator_config=get_cachingallocator_config(),
551          is_xnnpack_available=is_xnnpack_available(),
552          cpu_info=get_cpu_info(run_lambda),
553      )

588  def pretty_str(envinfo):
589      def replace_nones(dct, replacement='Could not collect'):
590          for key in dct.keys():
591              if dct[key] is not None:
592                  continue
593              dct[key] = replacement
594          return dct
595  
596      def replace_bools(dct, true='Yes', false='No'):
597          for key in dct.keys():
598              if dct[key] is True:
599                  dct[key] = true
600              elif dct[key] is False:
601                  dct[key] = false
602          return dct
603  
604      def prepend(text, tag='[prepend]'):
605          lines = text.split('\n')
606          updated_lines = [tag + line for line in lines]
607          return '\n'.join(updated_lines)
608  
609      def replace_if_empty(text, replacement='No relevant packages'):
610          if text is not None and len(text) == 0:
611              return replacement
612          return text
613  
614      def maybe_start_on_next_line(string):
615          # If `string` is multiline, prepend a \n to it.
616          if string is not None and len(string.split('\n')) > 1:
617              return '\n{}\n'.format(string)
618          return string
619  
620      mutable_dict = envinfo._asdict()
621  
622      # If nvidia_gpu_models is multiline, start on the next line
623      mutable_dict['nvidia_gpu_models'] = \
624          maybe_start_on_next_line(envinfo.nvidia_gpu_models)
625  
626      # If the machine doesn't have CUDA, report some fields as 'No CUDA'
627      dynamic_cuda_fields = [
628          'cuda_runtime_version',
629          'nvidia_gpu_models',
630          'nvidia_driver_version',
631      ]
632      all_cuda_fields = dynamic_cuda_fields + ['cudnn_version']
633      all_dynamic_cuda_fields_missing = all(
634          mutable_dict[field] is None for field in dynamic_cuda_fields)
635      if TORCH_AVAILABLE and not torch.cuda.is_available() and all_dynamic_cuda_fields_missing:
636          for field in all_cuda_fields:
637              mutable_dict[field] = 'No CUDA'
638          if envinfo.cuda_compiled_version is None:
639              mutable_dict['cuda_compiled_version'] = 'None'
640  
641      # Replace True with Yes, False with No
642      mutable_dict = replace_bools(mutable_dict)
643  
644      # Replace all None objects with 'Could not collect'
645      mutable_dict = replace_nones(mutable_dict)
646  
647      # If either of these are '', replace with 'No relevant packages'
648      mutable_dict['pip_packages'] = replace_if_empty(mutable_dict['pip_packages'])
649      mutable_dict['conda_packages'] = replace_if_empty(mutable_dict['conda_packages'])
650  
651      # Tag conda and pip packages with a prefix
652      # If they were previously None, they'll show up as ie '[conda] Could not collect'
653      if mutable_dict['pip_packages']:
654          mutable_dict['pip_packages'] = prepend(mutable_dict['pip_packages'],
655                                                 '[{}] '.format(envinfo.pip_version))
656      if mutable_dict['conda_packages']:
657          mutable_dict['conda_packages'] = prepend(mutable_dict['conda_packages'],
658                                                   '[conda] ')
659      mutable_dict['cpu_info'] = envinfo.cpu_info
660      return env_info_fmt.format(**mutable_dict)
661  
662  
663  def get_pretty_env_info():
664      """ 
665      Returns a pretty string of environment information. 
666   
667      This function retrieves environment information by calling the `get_env_info` function 
668      and then formats the information into a human-readable string. The retrieved environment 
669      information is listed in the document of `get_env_info`. 
670      This function is used in `python collect_env.py` that should be executed when reporting a bug. 
671   
672      Returns: 
673          str: A pretty string of the environment information. 
674      """
675      return pretty_str(get_env_info())
