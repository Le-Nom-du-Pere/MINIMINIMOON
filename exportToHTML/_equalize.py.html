31   def reshape_scale(scale: torch.Tensor, axis: int, input: torch.Tensor) -> torch.Tensor:
32       """Reshapes the scale so that we can multiply it to the input by the given axis."""
33       new_shape = [1] * input.ndim
34       new_shape[axis] = input.size(axis)
35       return scale.view(new_shape)

44   class _InputEqualizationObserver(nn.Module):
45       r"""Observer for tracking the running min/max values of input columns, and 
46       computing the quantization parameters for the overall min/max input values. 
47    
48       Args: 
49           dtype: Quantized data type 
50           qscheme: Quantization scheme 
51           quant_min: Minimum quantization value. If unspecified, it will 
52               follow the 8-bit setup. 
53           quant_max: Maximum quantization value. If unspecified, it will 
54               follow the 8-bit setup. 
55    
56       The running minimum/maximum :math:`x_\text{min/max}` are computed in the 
57       same way as :class:`~torch.ao.quantization.observer.PerChannelMinMaxObserver`, 
58       with the difference that the running min/max values are stored per column. 
59       This observer is intended to be used along with a WeightEqualizationObserver 
60       to calculate the equalization scale. 
61       """
62   
63       def __init__(
64           self,
65           dtype=torch.quint8,
66           qscheme=torch.per_tensor_affine,
67           quant_min=None,
68           quant_max=None,
69           factory_kwargs=None,
70       ) -> None:
71           super().__init__()
72   
73           if qscheme not in {torch.per_tensor_affine, torch.per_tensor_symmetric}:
74               raise TypeError("Input qscheme must be per-tensor")
75   
76           self.dtype = dtype
77           self.qscheme = qscheme
78   
79           per_channel_qscheme = qsheme_mapping_per_tensor_to_per_channel[qscheme]
80           self.input_obs = PerChannelMinMaxObserver(
81               ch_axis=1,
82               dtype=dtype,
83               qscheme=per_channel_qscheme,
84               quant_min=quant_min,
85               quant_max=quant_max,
86               factory_kwargs=factory_kwargs,
87           )
88   
89           self.equalization_scale = torch.tensor(1)
90           self.equalization_shape: list[int] = []
91   
92       def forward(self, x_orig):
93           if not (x_orig.ndim >= 2 and x_orig.ndim <= 5):
94               raise ValueError(
95                   "InputEqualizationObserver only supports Linear and Conv layers"
96               )
97   
98           # Calculate the shape needed to reshape the equalization scale later (needed for Conv layers)
99           self.equalization_shape = [1] * x_orig.ndim
100          self.equalization_shape[1] = x_orig.size(1)
101  
102          return self.input_obs(x_orig)
103  
104      def get_input_minmax(self):
105          return (self.input_obs.min_val, self.input_obs.max_val)
106  
107      def set_equalization_scale(self, equalization_scale):
108          # Reshape the equalization scale along axis=1 so that it can be
109          # multiplied with the input along axis=1
110          if equalization_scale.nelement() == 1 and equalization_scale == torch.tensor(1):
111              return
112          self.equalization_scale = torch.reshape(
113              equalization_scale, self.equalization_shape
114          )
115  
116      def calculate_scaled_minmax(self):
117          r"""Returns the scaled min/max inputs"""
118          if (
119              self.equalization_scale.nelement() == 1
120              and self.equalization_scale == torch.tensor(1)
121          ):
122              warnings.warn(
123                  "Must call calculate_equalization_scale before calling calculate_scaled_minmax. "
124                  + "Will not scale the next quantization observer."
125              )
126              return None, None
127  
128          # Calculate qparams for the scaled min/max inputs
129          # Scale the input by the equalization scale located at the same column
130          # index
131          (min_inputs, max_inputs) = self.get_input_minmax()
132          equalization_scale_reshaped = reshape_scale(
133              self.equalization_scale, 0, min_inputs
134          )
135          min_input_scaled = torch.min(torch.mul(min_inputs, equalization_scale_reshaped))
136          max_input_scaled = torch.max(torch.mul(max_inputs, equalization_scale_reshaped))
137  
138          return min_input_scaled, max_input_scaled
139  
140      with_args = classmethod(_with_args)

143  class _WeightEqualizationObserver(nn.Module):
144      r"""Observer for tracking the running min/max values of weight columns and 
145      rows, and computing the quantization parameters for the weight rows. 
146   
147      Args: 
148          dtype: Quantized data type 
149          qscheme: Quantization scheme 
150          quant_min: Minimum quantization value. If unspecified, it will 
151              follow the 8-bit setup. 
152          quant_max: Maximum quantization value. If unspecified, it will 
153              follow the 8-bit setup. 
154   
155      This observer is made up of 1 PerChannelMinMaxObserver `weight_col_obs` used 
156      to record the running minimum and maximum of columns of incoming weight 
157      tensors. This observer is intended to be used along with an 
158      InputEqualizationObserver to calculate the equalization scale. 
159   
160      The running minimum/maximum :math:`w_\text{min/max}` are computed in the 
161      same way as :class:`~torch.ao.quantization.observer.PerChannelMinMaxObserver`. 
162      """
163  
164      def __init__(
165          self,
166          dtype=torch.qint8,
167          qscheme=torch.per_tensor_affine,
168          quant_min=None,
169          quant_max=None,
170          factory_kwargs=None,
171      ) -> None:
172          super().__init__()
173  
174          self.dtype = dtype
175          self.qscheme = qscheme
176          self.ch_axis = 1
177  
178          per_channel_qscheme = qscheme
179          if qscheme in {torch.per_tensor_affine, torch.per_tensor_symmetric}:
180              per_channel_qscheme = qsheme_mapping_per_tensor_to_per_channel[qscheme]
181          self.weight_col_obs = PerChannelMinMaxObserver(
182              ch_axis=1,
183              dtype=dtype,
184              qscheme=per_channel_qscheme,
185              quant_min=quant_min,
186              quant_max=quant_max,
187              factory_kwargs=factory_kwargs,
188          )
189  
190          self.equalization_scale = torch.tensor(1)
191  
192      def forward(self, w_orig):
193          if not (w_orig.ndim >= 2 and w_orig.ndim <= 5):
194              raise ValueError(
195                  "InputEqualizationObserver only supports Linear and Conv layers"
196              )
197  
198          return self.weight_col_obs(w_orig)
199  
200      def get_weight_col_minmax(self):
201          return (self.weight_col_obs.min_val, self.weight_col_obs.max_val)
202  
203      def set_equalization_scale(self, equalization_scale):
204          self.equalization_scale = equalization_scale
205  
206      with_args = classmethod(_with_args)

209  def calculate_equalization_scale(
210      input_obs: _InputEqualizationObserver, weight_obs: _WeightEqualizationObserver
211  ) -> torch.Tensor:
212      r"""Calculates the equalization scale and sets the equalization_scale value 
213      in the observers. 
214   
215      Args: 
216          input_obs: Observer that tracks the ranges for the input columns 
217          weight_obs: Observer that tracks the ranges for the weight columns 
218      """
219  
220      (min_inputs, max_inputs) = input_obs.get_input_minmax()
221      (min_weights, max_weights) = weight_obs.get_weight_col_minmax()
222  
223      if not (
224          check_min_max_valid(min_inputs, max_inputs)
225          and check_min_max_valid(min_weights, max_weights)
226      ):
227          warnings.warn(
228              "Must run observer before calling calculate_equalization_scale. "
229              + "Returning default equalization scale torch.tensor(1)."
230          )
231          return torch.tensor(1)
232  
233      if not (min_inputs.shape == min_weights.shape):
234          raise ValueError(
235              "Input and Weight must have the same column dimension. "
236              + f"Found {min_inputs.shape} and {min_weights.shape} shapes instead."
237          )
238  
239      equalization_scale = torch.sqrt(
240          (max_weights - min_weights) / (max_inputs - min_inputs)
241      )
242      # Replace all 'inf', 'nan', 0's with 1s to prevent errors
243      equalization_scale[equalization_scale == 0.0] = 1
244      equalization_scale = torch.nan_to_num(equalization_scale, nan=1, posinf=1, neginf=1)
245      return equalization_scale

248  class EqualizationQConfig(
249      namedtuple("EqualizationQConfig", ["input_activation", "weight"])
250  ):
251      """ 
252      Describes how to quantize a layer or a part of the network specifically for 
253      input-weight equalization by providing settings (observer classes) for 
254      inputs, outputs, and weights. 
255   
256      Note that EqualizationQConfig needs to contain observer **classes** (like 
257      MinMaxObserver) or a callable that returns instances on invocation, not the 
258      concrete observer instances themselves. 
259      Quantization function will instantiate observers multiple times for each of 
260      the layers. 
261   
262      Observer classes have usually reasonable default arguments, but they can be 
263      overwritten with `with_args` method (that behaves like functools.partial): 
264   
265      my_qconfig = EqualizationQConfig(input_activation=_InputEqualizationObserver.with_args(dtype=torch.qint8), 
266                                      weight=_WeightEqualizationObserver.with_args(dtype=torch.qint8)) 
267      """
268  
269      __slots__ = ()
270  
271      def __new__(cls, input_activation=torch.nn.Identity, weight=torch.nn.Identity):
272          if isinstance(input_activation, nn.Module) or isinstance(weight, nn.Module):
273              raise ValueError(
274                  "EqualizationQConfig received observer instance, please pass observer class instead. "
275                  + "Use MyObserver.with_args(x=1) to override arguments to constructor if needed"
276              )
277          self = super().__new__(cls, input_activation, weight)
278          return self

293      """Checks if the fused node supports equalization."""

296          nni.ConvReLU1d,
297          nni.ConvReLU2d,
298          nni.ConvReLU3d,

312  def node_supports_equalization(node: Node, modules) -> bool:
313      """Checks if the current node supports equalization 
314      Currently we only support nn.Linear/F.Linear and nn.Conv/F.conv layers 
315      """
316      if node.op == "call_module":
317          return (
318              nn_module_supports_equalization(modules[str(node.target)])
319              or fused_module_supports_equalization(modules[str(node.target)])
320              or custom_module_supports_equalization(modules[str(node.target)])
321          )
322      elif node.op == "call_function":
323          return node.target in [F.linear, F.conv1d, F.conv2d, F.conv3d]
324      return False

338  def get_op_node_and_weight_eq_obs(
339      input_eq_obs_node: Node, model: GraphModule, modules: dict[str, nn.Module]
340  ) -> tuple[Optional[Node], Optional[_WeightEqualizationObserver]]:
341      """Gets the following weight equalization observer. There should always 
342      exist a weight equalization observer after an input equalization observer. 
343   
344      Returns the operation node that follows the input equalization observer node 
345      and the weight equalization observer 
346      """
347  
348      # Find the op node that comes directly after the input equalization observer
349      op_node = None
350      for user in input_eq_obs_node.users.keys():
351          if node_supports_equalization(user, modules):
352              op_node = user
353              break
354  
355      assert op_node is not None
356      if op_node.op == "call_module":
357          # If the op_node is a nn.Linear layer, then it must have a
358          # WeightEqualizationObserver configuration
359          maybe_equalization_node_name_to_config = _get_observed_graph_module_attr(
360              model, "equalization_node_name_to_qconfig"
361          )
362          assert maybe_equalization_node_name_to_config is not None
363          equalization_node_name_to_qconfig: dict[str, Any] = (
364              maybe_equalization_node_name_to_config  # type: ignore[assignment]
365          )
366          assert equalization_node_name_to_qconfig.get(op_node.name, None) is not None
367          weight_eq_obs = equalization_node_name_to_qconfig.get(  # type: ignore[union-attr]
368              op_node.name, None
369          ).weight()
370  
371          assert isinstance(weight_eq_obs, _WeightEqualizationObserver)
372          return op_node, weight_eq_obs
373  
374      elif op_node.op == "call_function":
375          weight_node = maybe_get_weight_eq_obs_node(op_node, modules)
376          if weight_node is not None:
377              weight_eq_obs = modules[str(weight_node.target)]
378              assert isinstance(weight_eq_obs, _WeightEqualizationObserver)
379              return op_node, weight_eq_obs
380  
381      return None, None

384  def maybe_get_weight_eq_obs_node(
385      op_node: Node, modules: dict[str, nn.Module]
386  ) -> Optional[Node]:
387      """Gets the weight equalization observer node if it exists."""
388      assert op_node.op == "call_function"
389      for node_arg in op_node.args:
390          if node_arg_is_weight(op_node, node_arg):
391              assert (
392                  isinstance(node_arg, Node)
393                  and node_arg.op == "call_module"
394                  and isinstance(
395                      modules[str(node_arg.target)], _WeightEqualizationObserver
396                  )
397              )
398              return node_arg
399      return None

402  def maybe_get_next_input_eq_obs(
403      node: Node, modules: dict[str, nn.Module]
404  ) -> Optional[_InputEqualizationObserver]:
405      """Gets the following input equalization observer if it exists. 
406   
407      For example, in the case of connecting linear layers: 
408          x -> inp_obs1 -> eq_obs1 -> linear1 -> out_obs1 -> eq_obs2 -> linear2 -> out_obs2 
409      If the node being passed in is the linear1 node, then we want to return eq_obs2, 
410      the following equalization observer for linear2. 
411   
412      However, if there are no connecting layers: 
413          x -> inp_obs1 -> eq_obs1 -> linear1 -> out_obs1 -> add 
414      Then we want to return None. 
415   
416      In the case of an unfused linear-relu layer with a connecting linear layer: 
417          linear1 -> relu -> out_obs1 -> eq_obs2 -> linear2 -> out_obs2 
418      Since it is unfused, we want to skip over the relu layer and return eq_obs2, 
419      the following equalization observer for linear2. 
420      """
421  
422      assert node_supports_equalization(node, modules)
423  
424      # Locate the following nn.ReLU or F.relu node if it exists
425      maybe_relu_node = maybe_get_next_module(node, modules, nn.ReLU)
426      if maybe_relu_node is None:
427          maybe_relu_node = maybe_get_next_module(
428              node, modules, target_functional_type=F.relu
429          )
430  
431      # Locate the following output observer if it exists.
432      # We will skip the relu node if it exists.
433      maybe_obs_node = (
434          maybe_get_next_module(node, modules, ObserverBase)
435          if maybe_relu_node is None
436          else maybe_get_next_module(maybe_relu_node, modules, ObserverBase)
437      )
438      if maybe_obs_node is None:
439          return None
440  
441      maybe_eq_obs_node = maybe_get_next_module(
442          maybe_obs_node, modules, _InputEqualizationObserver
443      )
444      if maybe_eq_obs_node is None:
445          return None
446  
447      maybe_eq_obs = modules[str(maybe_eq_obs_node)]
448      assert isinstance(maybe_eq_obs, _InputEqualizationObserver)
449      return maybe_eq_obs

452  def maybe_get_next_equalization_scale(
453      node: Node, modules: dict[str, nn.Module]
454  ) -> Optional[torch.Tensor]:
455      """If the next next node is an InputEqualizationObserver then we want to 
456      return its equalization scale, else we return 1 
457   
458      This is used in the case where there are two connecting linear layers: 
459          linear1 -> LinearOutObs -> InputEqObs -> linear2 
460      In this case, the node given is linear1 and we want to locate the InputEqObs. 
461      """
462      next_inp_eq_obs = maybe_get_next_input_eq_obs(node, modules)
463      if next_inp_eq_obs:
464          if (
465              next_inp_eq_obs.equalization_scale.nelement() == 1
466              and next_inp_eq_obs.equalization_scale == torch.tensor(1)
467          ):
468              return None
469          return next_inp_eq_obs.equalization_scale
470      return None

473  def scale_input_observer(node: Node, modules: dict[str, nn.Module]) -> None:
474      """Scales the following input quantization observer's min/max values by 
475      updating the values with the scaled min/max values calculated by the input 
476      equalization observer 
477      """
478      input_eq_obs = modules[str(node.target)]
479      assert isinstance(input_eq_obs, _InputEqualizationObserver)
480  
481      input_quant_obs_node = node.args[0]
482      assert isinstance(input_quant_obs_node, Node)
483  
484      input_quant_obs = modules[str(input_quant_obs_node.target)]
485      if not isinstance(input_quant_obs, ObserverBase):
486          return
487  
488      min_input_scaled, max_input_scaled = input_eq_obs.calculate_scaled_minmax()
489      if min_input_scaled is None and max_input_scaled is None:
490          return
491      input_quant_obs.min_val = min_input_scaled
492      input_quant_obs.max_val = max_input_scaled

495  def scale_weight_node(
496      node: Node,
497      modules: dict[str, nn.Module],
498      equalization_scale: torch.Tensor,
499      next_equalization_scale: Optional[torch.Tensor],
500  ) -> None:
501      """Scale the weights for input-weight equalization by multiplying the 
502      weight by 1/equalization_scale and next_equalization_scale 
503   
504      Args: 
505          node: Current node whose weights we want to scale 
506          equalization_scale: Current node's calculated equalization scale 
507          next_equalization_scale: Next node's calculated equalization scale if 
508             the following node needs to be equalized, 1 otherwise 
509      """
510      if equalization_scale is None:
511          return
512  
513      if fused_module_supports_equalization(modules[str(node.target)]):
514          op_module = modules[str(node.target)][0]  # type: ignore[index]
515      else:
516          op_module = modules[str(node.target)]
517      assert nn_module_supports_equalization(
518          op_module
519      ) or custom_module_supports_equalization(op_module)
520  
521      # Scale the weights for input-weight equalization
522      # If the following layer needs to be equalized then we will multiply its scale
523      weight = op_module.weight
524      assert isinstance(weight, torch.Tensor)
525  
526      # Scale the weights by the reciprocal of the equalization scale
527      # Reshape the equalization scale so that we can multiply it to the weight along axis=1
528      equalization_scale_reshaped = reshape_scale(equalization_scale, 1, weight)
529      scaled_weight = torch.mul(weight, torch.reciprocal(equalization_scale_reshaped))
530  
531      if next_equalization_scale is None:
532          op_module.weight = nn.Parameter(scaled_weight)
533          return
534  
535      # Multiply the weights row wise by the next equalization scale
536      # Reshape the equalization scale so that we can multiply it to the weight along axis=0
537      next_equalization_scale_reshaped = reshape_scale(next_equalization_scale, 0, weight)
538      scaled_weight = torch.mul(scaled_weight, next_equalization_scale_reshaped)
539  
540      op_module.weight = nn.Parameter(scaled_weight)
541  
542      # Multiply the bias element wise by the next equalization scale
543      bias = op_module.bias
544      if bias is None:
545          return
546      assert isinstance(bias, torch.Tensor)
547  
548      # Reshape the equalization scale so that we can multiply it element-wise to the bias
549      next_equalization_scale_reshaped = reshape_scale(next_equalization_scale, 0, bias)
550      scaled_bias = torch.mul(bias, next_equalization_scale_reshaped)
551      op_module.bias = nn.Parameter(scaled_bias)
552

554  def scale_weight_functional(
555      op_node: Node,
556      model: GraphModule,
557      modules: dict[str, nn.Module],
558      equalization_scale: torch.Tensor,
559      next_equalization_scale: Optional[torch.Tensor],
560  ) -> None:
561      """Scales the weight value for functional layers"""
562      if equalization_scale is None:
563          return
564  
565      # From the given op_node, the path looks like:
566      #   get_attr(weight) -> weight_quant_obs -> weight_eq_obs -> op_node
567      # So we want to trace back from the op_node to get the equalization observer
568      # node, then the quantization observer node, and then finally the weight
569      # node which contains the weight values.
570  
571      # Get the equalization observer node
572      weight_eq_obs_node = maybe_get_weight_eq_obs_node(op_node, modules)
573      if weight_eq_obs_node is None:
574          return
575  
576      # Get the quantization observer node
577      weight_quant_obs_node = weight_eq_obs_node.args[0]
578      if weight_quant_obs_node is None:
579          return
580      assert isinstance(weight_quant_obs_node, Node) and isinstance(
581          modules[str(weight_quant_obs_node.target)], ObserverBase
582      )
583  
584      # Get the get_attr(weight) node
585      weight_node = weight_quant_obs_node.args[0]
586      if weight_node is None:
587          return
588      assert isinstance(weight_node, Node) and weight_node.op == "get_attr"
589  
590      weight_parent_name, weight_name = _parent_name(weight_node.target)
591      weight = getattr(modules[weight_parent_name], weight_name)
592  
593      # Scale the weights for input-weight equalization
594      # If the following layer needs to be equalized then we will multiply its scale
595      # Reshape the equalization scale so that we can multiply it to the weight along axis=1
596      equalization_scale_reshaped = reshape_scale(equalization_scale, 1, weight)
597      scaled_weight = torch.mul(weight, torch.reciprocal(equalization_scale_reshaped))
598  
599      if next_equalization_scale is None:
600          setattr(modules[weight_parent_name], weight_name, scaled_weight)
601          return
602  
603      # Multiply the weights row wise by the next equalization scale
604      # Reshape the equalization scale so that we can multiply it to the weight along axis=1
605      next_equalization_scale_reshaped = reshape_scale(
606          next_equalization_scale, 0, scaled_weight
607      )
608      scaled_weight = torch.mul(scaled_weight, next_equalization_scale_reshaped)
609  
610      setattr(modules[weight_parent_name], weight_name, scaled_weight)
611      assert torch.allclose(model.get_buffer(str(weight_node.target)), scaled_weight)
612  
613      # Multiply the bias element wise by the next equalization scale
614      bias_node = None
615      for node in op_node.args:
616          # Find the node containing the weight values
617          if isinstance(node, Node) and node.op == "get_attr" and "bias" in node.name:
618              bias_node = node
619              break
620      if bias_node is None:
621          return
622  
623      bias_parent_name, bias_name = _parent_name(bias_node.target)
624      bias = getattr(modules[bias_parent_name], bias_name)
625  
626      # Reshape the equalization scale so that we can multiply it element-wise to the bias
627      next_equalization_scale_reshaped = reshape_scale(next_equalization_scale, 0, bias)
628      scaled_bias = torch.mul(bias, next_equalization_scale_reshaped)
629      setattr(modules[bias_parent_name], bias_name, scaled_bias)
630

632  def clear_weight_quant_obs_node(op_node: Node, modules: dict[str, nn.Module]) -> None:
633      """Given the operation node, we want find the corresponding quantization 
634      observer and reset its min/max values 
635      """
636      weight_eq_obs_node = maybe_get_weight_eq_obs_node(op_node, modules)
637      if weight_eq_obs_node is None:
638          return
639  
640      weight_quant_obs_node = weight_eq_obs_node.args[0]
641      if weight_quant_obs_node is None:
642          return
643      assert isinstance(weight_quant_obs_node, Node)
644  
645      weight_quant_obs = modules[str(weight_quant_obs_node.target)]
646      assert isinstance(modules[str(weight_quant_obs_node.target)], ObserverBase)
647      weight_quant_obs.reset_min_max_vals()  # type: ignore[operator]

650  def remove_node(model: GraphModule, node: Node, prev_node: Node):
651      """Removes the given node from the model by replacing all of its users with 
652      the given previous node 
653      """
654      # For all of the current node's users, replace the current node with
655      # the input quantization observer node
656      orig_users = list(node.users.keys())
657      for user_node in orig_users:
658          user_node.replace_input_with(node, prev_node)
659  
660      # Erase the InputEqualizationObserver node
661      model.graph.erase_node(node)

664  def update_obs_for_equalization(
665      model: GraphModule, modules: dict[str, nn.Module]
666  ) -> dict[str, _WeightEqualizationObserver]:
667      """Update all of the observer's equalization scale. For each 
668      InputEqualizationObserver, we will find the location of the next 
669      WeightEqualizationObserver, create it, and calculate the equalization scale 
670      based on the two observers. 
671   
672      We will then return a dictionary mapping operation node names to 
673      the corresponding WeightEqualizationObservers for that operation. 
674      """
675      weight_eq_obs_dict = {}
676      for node in model.graph.nodes:
677          if node.op == "call_module" and isinstance(
678              modules[node.target], _InputEqualizationObserver
679          ):
680              input_eq_obs = modules[node.target]
681              assert isinstance(input_eq_obs, _InputEqualizationObserver)
682              op_node, weight_eq_obs = get_op_node_and_weight_eq_obs(node, model, modules)
683  
684              if op_node is None or weight_eq_obs is None:
685                  continue
686  
687              if op_node.op == "call_module":
688                  # Calibrate the weight equalization observer since it has just
689                  # been created
690                  if fused_module_supports_equalization(modules[str(op_node.target)]):
691                      module = modules[str(op_node.target)][0]  # type: ignore[index]
692                      assert nn_module_supports_equalization(module)
693                      weight_eq_obs(module.weight)
694                  else:
695                      weight_eq_obs(modules[str(op_node.target)].weight)
696  
697              # Calculate and set the equalization scale values
698              equalization_scale = calculate_equalization_scale(
699                  input_eq_obs, weight_eq_obs
700              )
701              input_eq_obs.set_equalization_scale(equalization_scale)
702              weight_eq_obs.set_equalization_scale(equalization_scale)
703  
704              weight_eq_obs_dict[op_node.name] = weight_eq_obs
705  
706      return weight_eq_obs_dict

709  def convert_eq_obs(
710      model: GraphModule,
711      modules: dict[str, nn.Module],
712      weight_eq_obs_dict: dict[str, _WeightEqualizationObserver],
713  ) -> None:
714      """Converts the equalization operations and updates the other nodes in the 
715      following way: 
716          - Removes the input equalization observers and inserts a mul operator 
717            along with an equalization scale node wherever applicable (we do not 
718            want to insert a mul operator between connecting linear layers). 
719          - Updates the input quantization observers with the scaled input min/max 
720            values. 
721          - Scales the weights by the current and next equalization scales. 
722          - Removes the weight equalization observer node if it exists. 
723   
724      Before (after prepare): 
725                                      weight values 
726                                            | 
727                                      WeightQuantObs 
728                                            | 
729                                        WeightEqObs 
730                                            | 
731          x -> InpQuantObs -> InpEqObs -> linear -> OutQuantObs 
732   
733      After this function: 
734                                                scaled weight values 
735                                                        | 
736         equalization scale                       WeightQuantObs 
737                |                                       | 
738          x -> mul -> InpQuantObs (scaled min/max) -> linear -> OutQuantObs 
739   
740      After convert: 
741         equalization scale                 scaled weight values 
742                |                                    | 
743          x -> mul -> quantize_per_tensor -> quantized::linear 
744   
745      Note that although the equalization observer appeared after the quantization 
746      observer after prepare_fx, the mul node appears before the quantization node 
747      after convert_fx. This is because placing the equalization observer after 
748      the quantization observer in prepare_fx would allow us to keep the invariant 
749      that the graph before the current node inserts its observers is not 
750      modified. 
751   
752      Having the equalization observer before the quantization observer would also 
753      cause some inconsistences between the ordering of the quantization and 
754      equalization observers. 
755      For example, a single linear layer would look like: 
756          x -> InpEqObs1 -> InpQuantObs1 -> linear1 -> OutQuantObs1 
757      But between two connected linear layers, it would look like: 
758          linear1 -> OutQuantObs1 -> InpEqObs2 -> linear2 -> OutQuantObs2 
759      """
760      for node in model.graph.nodes:
761          if node.op == "call_module" and isinstance(
762              modules[node.target], _InputEqualizationObserver
763          ):
764              inp_quant_obs_node = node.args[0]
765              prev_node = inp_quant_obs_node.args[0]
766  
767              # If the previous node is a layer that needs to be equalized, then
768              # we will remove the current node because we do not need to add any
769              # equalization nodes between two layers that need to be equalized
770  
771              # Before: linear1/relu (prev_node) -> output_quant_obs1 (inp_quant_obs_node) -> input_eq_obs2 (node) -> linear2
772              # After: linear1/relu (prev_node) -> output_quant_obs1 (inp_quant_obs_node) -> linear2
773              if (
774                  node_supports_equalization(prev_node, modules)
775                  or "relu" in prev_node.name
776              ):
777                  remove_node(model, node, inp_quant_obs_node)
778                  continue
779  
780              # Update the following input quantization observer's min/max values
781              scale_input_observer(node, modules)
782  
783              # Remove the InputEqualization node and add a mul operator before
784              # the quantization observer node that appears before the equalization node
785              # Before: x -> input_quant_obs -> input_eq_obs -> linear
786              # After: x -> mul -> input_quant_obs -> linear
787  
788              # Create a node containing the equalization scale
789              with model.graph.inserting_before(inp_quant_obs_node):
790                  get_new_eq_scale_name = get_new_attr_name_with_prefix(
791                      prev_node.name + "_equalization_scale"
792                  )
793                  name = get_new_eq_scale_name(modules)
794                  setattr(model, name, modules[node.target].equalization_scale)
795                  eq_scale_node = model.graph.create_node("get_attr", name)
796  
797              # Create a node multiplying the input with the equalization scale
798              with model.graph.inserting_after(eq_scale_node):
799                  inputs = (prev_node, eq_scale_node)
800                  mul_node = model.graph.create_node("call_function", torch.mul, inputs)
801  
802              # Set the mul nod to be the input_quant_obs_node's input instead of
803              # the previous node
804              inp_quant_obs_node.replace_input_with(prev_node, mul_node)
805              remove_node(model, node, inp_quant_obs_node)
806  
807          elif weight_eq_obs_dict.get(node.name, None) is not None:
808              weight_eq_obs = weight_eq_obs_dict.get(node.name)
809              assert isinstance(weight_eq_obs, _WeightEqualizationObserver)
810              equalization_scale = weight_eq_obs.equalization_scale
811  
812              if (
813                  equalization_scale.nelement() == 1
814                  and equalization_scale == torch.tensor(1)
815              ):
816                  equalization_scale = None  # type: ignore[assignment]
817              maybe_next_equalization_scale = maybe_get_next_equalization_scale(
818                  node, modules
819              )
820  
821              # Scale the weight nodes
822              if node.op == "call_module":
823                  scale_weight_node(
824                      node, modules, equalization_scale, maybe_next_equalization_scale
825                  )
826              elif node.op == "call_function":
827                  scale_weight_functional(
828                      node,
829                      model,
830                      modules,
831                      equalization_scale,
832                      maybe_next_equalization_scale,
833                  )
834  
835                  weight_eq_obs_node = maybe_get_weight_eq_obs_node(node, modules)
836                  if weight_eq_obs_node is None:
837                      return
838                  assert isinstance(
839                      modules[str(weight_eq_obs_node.target)], _WeightEqualizationObserver
840                  )
841  
842                  # Clear the quantization observer's min/max values so that they
843                  # can get updated later based on the new scale values
844                  clear_weight_quant_obs_node(node, modules)
845  
846                  # Erase the weight equalization observer node
847                  prev_node = weight_eq_obs_node.args[0]
848                  remove_node(model, weight_eq_obs_node, prev_node)  # type: ignore[arg-type]
849              else:
850                  raise ValueError(
851                      "Expected operation node to be 'call_module' or 'call_function"
852                      + f"Instead got node {node.name} as '{node.op}'."
853                  )
854

856  def _convert_equalization_ref(model: GraphModule):
857      """Reference function which applies changes needed for equalization, but 
858      does not quantize the nodes 
859      """
860      modules = dict(model.named_modules(remove_duplicate=False))
861  
862      # Calculate the equalization scale, update the observers with the scaled
863      # inputs, and scale the weight
864      weight_eq_obs_dict = update_obs_for_equalization(model, modules)
865      convert_eq_obs(model, modules, weight_eq_obs_dict)
866  
867      return GraphModule(model, model.graph)

875  def get_layer_sqnr_dict(
876      model_a: nn.Module, model_b: nn.Module, x: torch.Tensor
877  ) -> dict[str, float]:
878      """Runs the Numeric Suite on model_a and model_b and returns a dictionary 
879      containing the SQNR between layers in model_a and model_b. 
880   
881      Note: In order to support equalized models, this function has a hacky fix in 
882      which we do not match any torch.mul operators. This is because equalized 
883      models contain extra mul operators to scale the input by the equalization 
884      scale, but this edge case has not been resolved yet within the numeric suite code. 
885   
886      Args: 
887          model_a: A float model 
888          model_b: A quantized model 
889          x: Inputs to use during calibration 
890      """
891      import torch.ao.ns._numeric_suite_fx as ns
892      from torch.ao.ns.fx.mappings import get_unmatchable_types_map
893  
894      unmatchable_types_map = get_unmatchable_types_map()
895      unmatchable_types_map["funs_unmatchable"].add(torch.mul)
896  
897      model_a_ns, model_b_ns = ns.add_loggers(
898          "fp32",
899          model_a,
900          "int8",
901          model_b,
902          ns.OutputLogger,
903          unmatchable_types_map=unmatchable_types_map,
904      )
905  
906      model_a_ns(x)
907      model_b_ns(x)
908  
909      activation_comparison_dict = ns.extract_logger_info(
910          model_a_ns, model_b_ns, ns.OutputLogger, "int8"
911      )
912      ns.extend_logger_results_with_comparison(
913          activation_comparison_dict,
914          "fp32",
915          "int8",
916          torch.ao.ns.fx.utils.compute_sqnr,
917          "sqnr",
918      )
919  
920      # Construct a dictionary mapping layer names to the SQNR values
921      layer_sqnr_dict = {}
922      for key in activation_comparison_dict:
923          layer = activation_comparison_dict[key]["node_output"]["int8"][0]["fqn"]
924          sqnr = activation_comparison_dict[key]["node_output"]["int8"][0]["sqnr"][0]
925          layer_sqnr_dict[layer] = sqnr
926  
927      return layer_sqnr_dict

930  def get_equalization_qconfig_dict(
931      layer_sqnr_dict: dict[str, float], num_layers_to_equalize: int
932  ) -> Any:
933      """Given the layer to SQNR dictionary, find the layers with the highest 
934      quantization errors, and return an equalization_qconfig_dict 
935      specifying to only equalize those top layers. 
936   
937      Args: 
938          layer_sqnr_dict: Dictionary mapping layer names to SQNR values (found 
939              when comparing an equalized model against a float model) 
940          num_layers_to_equalize: Number of layers with the highest quantization 
941             errors to equalize 
942      """
943  
944      # Sort the layer_sqnr_dictionary values and get the layers with the lowest
945      # SQNR values (aka highest quantization errors)
946      layer_sqnr_sorted = sorted(layer_sqnr_dict.items(), key=operator.itemgetter(1))
947      layers_to_equalize = layer_sqnr_sorted[:num_layers_to_equalize]
948  
949      # Constructs an equalization_qconfig_dict that specifies to only equalize
950      # the layers with the highest quantization errors
951      module_to_qconfig_list = [
952          (item[0], default_equalization_qconfig) for item in layers_to_equalize
953      ]
954      equalization_qconfig_dict = {"module_name": module_to_qconfig_list}
955      return equalization_qconfig_dict
