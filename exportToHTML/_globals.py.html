10   import torch._C._onnx as _C_onnx
11   
12   # This module should only depend on _constants and nothing else in torch.onnx to keep
13   # dependency direction clean.
14   from torch.onnx import _constants
15   
16   
17   class _InternalGlobals:
18       """Globals used internally by ONNX exporter. 
19    
20       NOTE: Be very judicious when adding any new variables. Do not create new 
21       global variables unless they are absolutely necessary. 
22       """
23   
24       def __init__(self) -> None:
25           self._export_onnx_opset_version = _constants.ONNX_DEFAULT_OPSET
26           self._training_mode: _C_onnx.TrainingMode = _C_onnx.TrainingMode.EVAL
27           self._in_onnx_export: bool = False
28           # Whether the user's model is training during export
29           self.export_training: bool = False
30           self.operator_export_type: _C_onnx.OperatorExportTypes = (
31               _C_onnx.OperatorExportTypes.ONNX
32           )
33           self.onnx_shape_inference: bool = True
34           self._autograd_inlining: bool = True
35   
36       @property
37       def training_mode(self):
38           """The training mode for the exporter."""
39           return self._training_mode
40   
41       @training_mode.setter
42       def training_mode(self, training_mode: _C_onnx.TrainingMode):
43           if not isinstance(training_mode, _C_onnx.TrainingMode):
44               raise TypeError(
45                   "training_mode must be of type 'torch.onnx.TrainingMode'. This is "
46                   "likely a bug in torch.onnx."
47               )
48           self._training_mode = training_mode
49   
50       @property
51       def export_onnx_opset_version(self) -> int:
52           """Opset version used during export."""
53           return self._export_onnx_opset_version
54   
55       @export_onnx_opset_version.setter
56       def export_onnx_opset_version(self, value: int):
57           self._export_onnx_opset_version = value
58   
59       @property
60       def in_onnx_export(self) -> bool:
61           """Whether it is in the middle of ONNX export."""
62           return self._in_onnx_export
63   
64       @in_onnx_export.setter
65       def in_onnx_export(self, value: bool):
66           if type(value) is not bool:
67               raise TypeError("in_onnx_export must be a boolean")
68           self._in_onnx_export = value
69   
70       @property
71       def autograd_inlining(self) -> bool:
72           """Whether Autograd must be inlined."""
73           return self._autograd_inlining
74   
75       @autograd_inlining.setter
76       def autograd_inlining(self, value: bool):
77           if type(value) is not bool:
78               raise TypeError("autograd_inlining must be a boolean")
79           self._autograd_inlining = value
80   
81
