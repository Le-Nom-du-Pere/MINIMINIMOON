15   from sympy.functions.special.bessel import jn_zeros
16   
17   
18   def eq(a, b, tol=1e-6):
19       for x, y in zip(a, b):
20           if not (abs(x - y) < tol):
21               return False
22       return True
23   
24   
25   def test_jn_zeros():
26       assert eq(jn_zeros(0, 4, method="scipy"),
27               [3.141592, 6.283185, 9.424777, 12.566370])
28       assert eq(jn_zeros(1, 4, method="scipy"),
29               [4.493409, 7.725251, 10.904121, 14.066193])
30       assert eq(jn_zeros(2, 4, method="scipy"),
31               [5.763459, 9.095011, 12.322940, 15.514603])
32       assert eq(jn_zeros(3, 4, method="scipy"),
33               [6.987932, 10.417118, 13.698023, 16.923621])
34       assert eq(jn_zeros(4, 4, method="scipy"),
35               [8.182561, 11.704907, 15.039664, 18.301255])
