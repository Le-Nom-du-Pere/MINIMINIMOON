10   def decompose_with_registry(
11       exported_program: torch.export.ExportedProgram, registry: _registration.ONNXRegistry
12   ) -> torch.export.ExportedProgram:
13       """Decompose the exported program with the given registry. 
14    
15       This function is needed so it shows clearly on the profiler results. 
16       """
17       onnx_registered_ops = set(_decomp.get_onnx_implemented_overloads(registry))
18       decomp_table = _decomp.create_onnx_friendly_decomposition_table(onnx_registered_ops)
19       return exported_program.run_decompositions(decomp_table)

22   def insert_type_promotion_nodes(
23       graph_module: torch.fx.GraphModule,
24   ) -> None:
25       """Inplace pass to insert explicit type promotion nodes, recursively through nested modules."""
26       for module in graph_module.modules():
27           assert isinstance(module, torch.fx.GraphModule)
28           passes.InsertTypePromotion(module).run()

31   def remove_assertion_nodes(graph_module: torch.fx.GraphModule) -> torch.fx.GraphModule:
32       """Remove all assertion and check nodes from the FX graph"""
33       aten_assertion_targets = {
34           torch.ops.aten.sym_constrain_range_for_size.default,
35           torch.ops.aten._assert_async.default,
36           torch.ops.aten._assert_async.msg,
37           torch.ops.aten._assert_scalar.default,
38           torch.ops.aten._assert_tensor_metadata.default,
39       }
40       for node in graph_module.graph.nodes:
41           if node.op == "call_function" and node.target in aten_assertion_targets:
42               graph_module.graph.erase_node(node)
43       graph_module.recompile()
44       return graph_module
