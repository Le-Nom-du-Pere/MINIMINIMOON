25   class PrecompileCacheArtifact(CacheArtifact, Generic[T]):
26       """ 
27       Data for each cache artifact that will be serialized and deserialized by 
28       PrecompileContext, rather than CacheArtifactManager. 
29       T represents the deserialized type of the artifact, i.e. the return type of after_deserialization 
30    
31       PrecompileCacheArtifact is a frozen dataclass - you can add new serializable fields and metadata specific to your own artifacts 
32       as needed, and use them in after_deserialization. 
33    
34       Example implementation: 
35    
36       class MyPrecompileCacheArtifact(PrecompileCacheArtifact[MySerializableType]): 
37           my_field: int 
38    
39           def after_deserialization(self) -> MySerializableType: 
40               result = pickle.loads(self.content) 
41               # Do some extra work post deserialization 
42               result.my_post_deserialization_function(self.my_field) 
43               return result 
44       """

48           raise RuntimeError("Precompile cache artifacts do not populate caches")

63   class PrecompileContext(CacheArtifactManager):
64       """ 
65       PrecompileContext is a special CacheArtifactManager for handling precompilation 
66       It uses the same interface as CacheArtifactManager, but handles deserialization differently: instead 
67       of placing each artifact into respective caches, it will stitch all the cache artifacts for a single key 
68       together and place it into a global Precompile Cache. 
69    
70       The following artifact types are supported by PrecompileContext: 
71        - BundledAOTAutogradCacheArtifact 
72        - CodeStateArtifact (from torch._dynamo.package once available) 
73       """
74   
75       # Protected by the compile_lock
76       # _new_cache_artifacts_by_key organizes results by the key of each artifact.
77       # This allows us to implement serialize_by_key easily.
78       # On call to `serialize()`, all cache artifacts in _new_cache_artifacts_by_key
79       # are transferred to _new_cache_artifacts before serialization.
80       _new_cache_artifacts_by_key: dict[str, CacheArtifact] = {}
81       _new_cache_artifacts: CacheArtifactsResult = defaultdict(list)
82       # Keep a separate seen artifacts list to make avoid unnecessary duplicates
83       # This list will not be cleared between serialize() calls
84       _seen_artifacts: OrderedSet[CacheArtifact] = OrderedSet()
85       # When serialize() is called, artifacts are transferred from _cache_artifacts to
86       # internal data structure of the _serializer
87       # This allows us to only pay the cost of serialization if serialize() is called
88       _serializer: AppendingByteSerializer[tuple[str, list[CacheArtifact]]] = (
89           AppendingByteSerializer(serialize_fn=_serialize_single_cache)
90       )
91       _cache_info: CacheInfo = CacheInfo()
92   
93       @classmethod
94       def clear(cls) -> None:
95           cls._new_cache_artifacts_by_key.clear()
96           super().clear()
97   
98       @override
99       @classmethod
100      def record_artifact(
101          cls,
102          artifact_type: str,
103          key: str,
104          content: Any,
105      ) -> None:
106          """ 
107          Called from each caching operation to record the artifact in this 
108          "mega" list 
109          """
110          artifact = CacheArtifactFactory.encode_create(artifact_type, key, content)
111          # TODO: although this covers completely same artifacts, it's possible
112          # with AOTAutogradCacheEntries to have multiple artifacts whose keys
113          # (i.e. backend_ids) are different, but whose contents are equal.
114          # In those cases, it would be much better if we only serialize once instead
115          # of N times.
116          if artifact in cls._seen_artifacts:
117              return
118  
119          cls._new_cache_artifacts_by_key[key] = artifact
120          cls._seen_artifacts.add(artifact)
121  
122      @classmethod
123      def _save_artifacts_by_type(cls) -> None:
124          """ 
125          We normally record artifacts by key, but serialization expects them to be organized 
126          by artifact type. This function transfers artifacts from _new_cache_artifacts_by_key to _new_cache_artifacts 
127          """
128          for artifact in cls._new_cache_artifacts_by_key.values():
129              cls._new_cache_artifacts[artifact.__class__.type()].append(artifact)
130          cls._new_cache_artifacts_by_key.clear()
131  
132      @classmethod
133      def serialize_artifact_by_key(cls, key: str) -> Optional[CacheArtifact]:
134          """ 
135          Serialize all artifacts with the given key returned in a list. 
136          """
137          return cls._new_cache_artifacts_by_key.get(key, None)
138  
139      @classmethod
140      def serialize(cls) -> Optional[tuple[bytes, CacheInfo]]:
141          cls._save_artifacts_by_type()
142          return super().serialize()
