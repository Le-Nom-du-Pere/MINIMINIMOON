47   template <typename T>
48   std::ostream& operator<<(std::ostream& stream, const Vectorized<T>& vec) {
49     T buf[Vectorized<T>::size()];
50     vec.store(buf);
51     stream << "vec[";
52     for (int i = 0; i != Vectorized<T>::size(); i++) {
53       if (i != 0) {
54         stream << ", ";
55       }
56       stream << buf[i];
57     }
58     stream << "]";
59     return stream;
60   }

113          const Vectorized<double>& src,
114          const double* base_addr,

117    auto all_ones = _mm512_castsi512_pd(_mm512_set1_epi64(0xFFFFFFFFFFFFFFFF));
118    auto mask_ = _mm512_cmp_pd_mask(all_ones, mask.values, _CMP_EQ_OQ);
119    return _mm512_mask_i64gather_pd(src, mask_, vindex, base_addr, scale);
120  }

162  template <>
163  std::pair<Vectorized<double>, Vectorized<double>> inline interleave2<double>(
164      const Vectorized<double>& a,
165      const Vectorized<double>& b) {
166    // inputs:
167    //   a = {a0, a1, a3, a3, a4, a5, a6, a7}
168    //   b = {b0, b1, b2, b3, b4, b5, b6, b7}
169    // group cols crossing lanes:
170    //   return {a0, b0, a1, b1, a2, b2, a3, b3}
171    //          {a4, b4, a5, b5, a6, b6, a7, b7}
172    __m512i idx1 = _mm512_set_epi64(11, 3, 10, 2, 9, 1, 8, 0);
173    __m512i idx2 = _mm512_set_epi64(15, 7, 14, 6, 13, 5, 12, 4);
174    return std::make_pair(
175        _mm512_mask_permutex2var_pd(a, 0xff, idx1, b),
176        _mm512_mask_permutex2var_pd(a, 0xff, idx2, b));
177  }

179  template <>
180  std::pair<Vectorized<float>, Vectorized<float>> inline interleave2<float>(
181      const Vectorized<float>& a,
182      const Vectorized<float>& b) {
183    // inputs:
184    //   a = {a0, a1, a2, a3, a4, a5, a6, a7, a8, a9, a10, a11, a12, a13, a14,
185    //   a15} b = {b0, b1, b2, b3, b4, b5, b6, b7, b8, b9, b10, b11, b12, b13,
186    //   b14, b15}
187    //
188    //  return:
189    //    {a0, b0, a1, b1, a2, b2, a3, b3, a4, b4, a5, b5, a6, b6, a7, b7}
190    //    {a8, b8, a9, b9, a10, b10, a11, b11, a12, b12, a13, b13, a14, b14, a15,
191    //    b15}
192    __m512i idx1 =
193        _mm512_set_epi32(23, 7, 22, 6, 21, 5, 20, 4, 19, 3, 18, 2, 17, 1, 16, 0);
194    __m512i idx2 = _mm512_set_epi32(
195        31, 15, 30, 14, 29, 13, 28, 12, 27, 11, 26, 10, 25, 9, 24, 8);
196    return std::make_pair(
197        _mm512_mask_permutex2var_ps(a, 0xffff, idx1, b),
198        _mm512_mask_permutex2var_ps(a, 0xffff, idx2, b));
199  }

203  template <>
204  std::pair<Vectorized<double>, Vectorized<double>> inline deinterleave2<double>(
205      const Vectorized<double>& a,
206      const Vectorized<double>& b) {
207    // inputs:
208    //   a = {a0, b0, a1, b1, a2, b2, a3, b3}
209    //   b = {a4, b4, a5, b5, a6, b6, a7, b7}
210    // output:
211    //   return {a0, a1, a2, a3, a4, a5, a6, a7}
212    //          {b0, b1, b2, b3, b4, b5, b6, b7}
213    // The members of indices have been written in binary format for better
214    // understandability
215    __m512i idx1 = _mm512_set_epi64(14, 12, 10, 8, 6, 4, 2, 0);
216    __m512i idx2 = _mm512_set_epi64(15, 13, 11, 9, 7, 5, 3, 1);
217  
218    return std::make_pair(
219        _mm512_mask_permutex2var_pd(a, 0xff, idx1, b),
220        _mm512_mask_permutex2var_pd(a, 0xff, idx2, b));
221  }

238    __m512i idx2 = _mm512_set_epi32(
239        31, 29, 27, 25, 23, 21, 19, 17, 15, 13, 11, 9, 7, 5, 3, 1);

242        _mm512_mask_permutex2var_ps(a, 0xffff, idx1, b),
243        _mm512_mask_permutex2var_ps(a, 0xffff, idx2, b));
244  }

393  inline Vectorized<bool> operator&&(
394      const Vectorized<bool>& self,
395      const Vectorized<bool>& other) {
396    const __m512i* self_ = reinterpret_cast<const __m512i*>(self.as_bytes());
397    const __m512i* other_ = reinterpret_cast<const __m512i*>(other.as_bytes());
398    __m512i out = _mm512_and_si512(*self_, *other_);
399    Vectorized<bool> ret;
400    // We do not have a constructer that takes __m512i, so we need to memcpy
401    std::memcpy(ret, &out, ret.size() * sizeof(bool));
402    return ret;
403  }
