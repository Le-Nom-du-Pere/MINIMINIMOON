98   /// Runs the matrix multiplication microkernel followed by a clamp operation.
99   ///
100  /// The pointer of each buffers (packed LHS, packed RHS and output) needs to be added with offset
101  /// calculated using the following functions:
102  ///
103  ///   * Packed LHS: @ref kai_get_lhs_packed_offset_matmul_clamp_qai8_qai8p2vlx4_qsi8cxpsb2vlx4_2vlx2vl_sme2_mopa.

105  ///   * Output: @ref kai_get_dst_offset_matmul_clamp_qai8_qai8p2vlx4_qsi8cxpsb2vlx4_2vlx2vl_sme2_mopa.
106  ///
107  /// @param[in] m Number of output rows to be computed.
108  /// @param[in] n Number of output columns to be computed.
109  /// @param[in] k Common dimension of the LHS and RHS operands.
110  /// @param[in] packed_lhs Packed LHS matrix buffer.
111  /// @param[in] packed_rhs Packed RHS matrix buffer.
112  /// @param[out] dst Output matrix buffer.
113  /// @param[in] dst_stride_row Row stride in bytes of the output matrix.
114  /// @param[in] dst_stride_col Column stride in bytes of the output matrix.
115  /// @param[in] params Requantization and clamp parmaters.
