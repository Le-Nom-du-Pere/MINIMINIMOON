29   # Checks whether a class is defind in `torch.*` modules
30   def is_torch_native_class(cls):
31       if not hasattr(cls, "__module__"):
32           return False
33   
34       parent_modules = cls.__module__.split(".")
35       if not parent_modules:
36           return False
37   
38       root_module = sys.modules.get(parent_modules[0])
39       return root_module is torch

42   def get_type(type):
43       """Convert the given type to a torchScript acceptable format."""
44       if isinstance(type, str):
45           return type
46       elif inspect.getmodule(type) == typing:
47           # If the type is a type imported from typing
48           # like Tuple, List, Dict then replace `typing.`
49           # with a null string. This needs to be done since
50           # typing.List is not accepted by TorchScript.
51           type_to_string = str(type)
52           return type_to_string.replace(type.__module__ + ".", "")
53       elif is_torch_native_class(type):
54           # If the type is a subtype of torch module, then TorchScript expects a fully qualified name
55           # for the type which is obtained by combining the module name and type name.
56           return type.__module__ + "." + type.__name__
57       else:
58           # For all other types use the name for the type.
59           return type.__name__

62   def get_optional_of_element_type(types):
63       """Extract element type, return as `Optional[element type]` from consolidated types. 
64    
65       Helper function to extracts the type of the element to be annotated to Optional 
66       from the list of consolidated types and returns `Optional[element type]`. 
67       TODO: To remove this check once Union support lands. 
68       """
69       elem_type = types[1] if type(None) == types[0] else types[0]
70       elem_type = get_type(elem_type)
71   
72       # Optional type is internally converted to Union[type, NoneType], which
73       # is not supported yet in TorchScript. Hence, representing the optional type as string.
74       return "Optional[" + elem_type + "]"

83       class JitTypeTraceStoreLogger(CallTraceStoreLogger):
84           """A JitTypeCallTraceLogger that stores logged traces in a CallTraceStore."""
85

87               super().__init__(store)

92       class JitTypeTraceStore(CallTraceStore):
93           def __init__(self) -> None:
94               super().__init__()
95               # A dictionary keeping all collected CallTrace
96               # key is fully qualified name of called function
97               # value is list of all CallTrace
98               self.trace_records: dict[str, list] = defaultdict(list)
99   
100          def add(self, traces: Iterable[CallTrace]):
101              for t in traces:
102                  qualified_name = get_qualified_name(t.func)
103                  self.trace_records[qualified_name].append(t)
104  
105          def filter(
106              self,
107              qualified_name: str,
108              qualname_prefix: Optional[str] = None,
109              limit: int = 2000,
110          ) -> list[CallTraceThunk]:
111              return self.trace_records[qualified_name]
112  
113          def analyze(self, qualified_name: str) -> dict:
114              # Analyze the types for the given module
115              # and create a dictionary of all the types
116              # for arguments.
117              records = self.trace_records[qualified_name]
118              all_args = defaultdict(set)
119              for record in records:
120                  for arg, arg_type in record.arg_types.items():
121                      all_args[arg].add(arg_type)
122              return all_args
123  
124          def consolidate_types(self, qualified_name: str) -> dict:
125              all_args = self.analyze(qualified_name)
126              # If there are more types for an argument,
127              # then consolidate the type to `Any` and replace the entry
128              # by type `Any`.
129              for arg, types in all_args.items():
130                  types = list(types)
131                  type_length = len(types)
132                  if type_length == 2 and type(None) in types:
133                      # TODO: To remove this check once Union suppport in TorchScript lands.
134                      all_args[arg] = get_optional_of_element_type(types)
135                  elif type_length > 1:
136                      all_args[arg] = "Any"
137                  elif type_length == 1:
138                      all_args[arg] = get_type(types[0])
139              return all_args
140  
141          def get_args_types(self, qualified_name: str) -> dict:
142              return self.consolidate_types(qualified_name)

177  def jit_code_filter(code: CodeType) -> bool:
178      """Codefilter for Torchscript to trace forward calls. 
179   
180      The custom CodeFilter is required while scripting a FX Traced forward calls. 
181      FX Traced forward calls have `code.co_filename` start with '<' which is used 
182      to exclude tracing of stdlib and site-packages in the default code filter. 
183      Since we need all forward calls to be traced, this custom code filter 
184      checks for code.co_name to be 'forward' and enables tracing for all such calls. 
185      The code filter is similar to default code filter for monkeytype and 
186      excludes tracing of stdlib and site-packages. 
187      """
188      # Filter code without a source file and exclude this check for 'forward' calls.
189      if code.co_name != "forward" and (
190          not code.co_filename or code.co_filename[0] == "<"
191      ):
192          return False
193  
194      filename = Path(code.co_filename).resolve()
195      return not any(_startswith(filename, lib_path) for lib_path in LIB_PATHS)
