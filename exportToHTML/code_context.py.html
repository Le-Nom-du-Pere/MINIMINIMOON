1    """ 
2    This module provides thread-safe code context management for TorchDynamo using weak references. 
3     
4    The CodeContextDict class maintains a mapping between Python code objects and their associated 
5    context data, using weak references to automatically clean up entries when code objects are 
6    garbage collected. This prevents memory leaks while allowing context data to be associated 
7    with code objects throughout their lifecycle. 
8     
9    Key features: 
10   - Thread-safe context storage and retrieval 
11   - Automatic cleanup using weak references 
12   - Safe context management for Python code objects 
13   - Memory-leak prevention 
14    
15   Example usage: 
16       code_obj = compile('x = 1', '<string>', 'exec') 
17    
18       # Store context 
19       context = code_context.get_context(code_obj) 
20       context['metadata'] = {'optimized': True} 
21    
22       # Retrieve context 
23       if code_context.has_context(code_obj): 
24           ctx = code_context.get_context(code_obj) 
25           # Use context data... 
26    
27       # Remove context 
28       ctx = code_context.pop_context(code_obj) 
29   """
30   
31   import types
32   from typing import Any
33   
34   from .utils import ExactWeakKeyDictionary
35   
36   
37   class CodeContextDict:
38       def __init__(self) -> None:
39           self.code_context: ExactWeakKeyDictionary = ExactWeakKeyDictionary()
40   
41       def has_context(self, code: types.CodeType) -> bool:
42           return code in self.code_context
43   
44       def get_context(self, code: types.CodeType) -> dict[str, Any]:
45           ctx = self.code_context.get(code)
46           if ctx is None:
47               ctx = {}
48               self.code_context[code] = ctx
49           return ctx
50   
51       def pop_context(self, code: types.CodeType) -> dict[str, Any]:
52           ctx = self.get_context(code)
53           self.code_context._remove_id(id(code))
54           return ctx
55   
56       def clear(self) -> None:
57           self.code_context.clear()
58   
59   
60   code_context: CodeContextDict = CodeContextDict()
