95   def _find_cuda_home() -> Optional[str]:
96       """Find the CUDA install path."""
97       # Guess #1
98       cuda_home = os.environ.get('CUDA_HOME') or os.environ.get('CUDA_PATH')
99       if cuda_home is None:
100          # Guess #2
101          nvcc_path = shutil.which("nvcc")
102          if nvcc_path is not None:
103              cuda_home = os.path.dirname(os.path.dirname(nvcc_path))
104          else:
105              # Guess #3
106              if IS_WINDOWS:
107                  cuda_homes = glob.glob(
108                      'C:/Program Files/NVIDIA GPU Computing Toolkit/CUDA/v*.*')
109                  if len(cuda_homes) == 0:
110                      cuda_home = ''
111                  else:
112                      cuda_home = cuda_homes[0]
113              else:
114                  cuda_home = '/usr/local/cuda'
115              if not os.path.exists(cuda_home):
116                  cuda_home = None
117      if cuda_home and not torch.cuda.is_available():
118          logger.warning("No CUDA runtime is found, using CUDA_HOME='%s'", cuda_home)
119      return cuda_home

121  def _find_rocm_home() -> Optional[str]:
122      """Find the ROCm install path."""
123      # Guess #1
124      rocm_home = os.environ.get('ROCM_HOME') or os.environ.get('ROCM_PATH')
125      if rocm_home is None:
126          # Guess #2
127          hipcc_path = shutil.which('hipcc')
128          if hipcc_path is not None:
129              rocm_home = os.path.dirname(os.path.dirname(
130                  os.path.realpath(hipcc_path)))
131              # can be either <ROCM_HOME>/hip/bin/hipcc or <ROCM_HOME>/bin/hipcc
132              if os.path.basename(rocm_home) == 'hip':
133                  rocm_home = os.path.dirname(rocm_home)
134          else:
135              # Guess #3
136              fallback_path = '/opt/rocm'
137              if os.path.exists(fallback_path):
138                  rocm_home = fallback_path
139      if rocm_home and torch.version.hip is None:
140          logger.warning("No ROCm runtime is found, using ROCM_HOME='%s'", rocm_home)
141      return rocm_home

143  def _find_sycl_home() -> Optional[str]:
144      sycl_home = None
145      icpx_path = shutil.which('icpx')
146      # Guess 1: for source code build developer/user, we'll have icpx in PATH,
147      # which will tell us the SYCL_HOME location.
148      if icpx_path is not None:
149          sycl_home = os.path.dirname(os.path.dirname(
150              os.path.realpath(icpx_path)))
151  
152      # Guess 2: for users install Pytorch with XPU support, the sycl runtime is
153      # inside intel-sycl-rt, which is automatically installed via pip dependency.
154      else:
155          try:
156              files = importlib.metadata.files('intel-sycl-rt') or []
157              for f in files:
158                  if f.name == "libsycl.so":
159                      sycl_home = os.path.dirname(Path(f.locate()).parent.resolve())
160                      break
161          except importlib.metadata.PackageNotFoundError:
162              logger.warning("Trying to find SYCL_HOME from intel-sycl-rt package, but it is not installed.")
163      return sycl_home

350  def _maybe_write(filename, new_content):
351      r''' 
352      Equivalent to writing the content into the file but will not touch the file 
353      if it already had the right content (to avoid triggering recompile). 
354      '''
355      if os.path.exists(filename):
356          with open(filename) as f:
357              content = f.read()
358  
359          if content == new_content:
360              # The file already contains the right thing!
361              return
362  
363      with open(filename, 'w') as source_file:
364          source_file.write(new_content)

381  def check_compiler_ok_for_platform(compiler: str) -> bool:
382      """ 
383      Verify that the compiler is the expected one for the current platform. 
384   
385      Args: 
386          compiler (str): The compiler executable to check. 
387   
388      Returns: 
389          True if the compiler is gcc/g++ on Linux or clang/clang++ on macOS, 
390          and always True for Windows. 
391      """
392      if IS_WINDOWS:
393          return True
394      compiler_path = shutil.which(compiler)
395      if compiler_path is None:
396          return False
397      # Use os.path.realpath to resolve any symlinks, in particular from 'c++' to e.g. 'g++'.
398      compiler_path = os.path.realpath(compiler_path)
399      # Check the compiler name
400      if any(name in compiler_path for name in _accepted_compilers_for_platform()):
401          return True
402      # If compiler wrapper is used try to infer the actual compiler by invoking it with -v flag
403      env = os.environ.copy()
404      env['LC_ALL'] = 'C'  # Don't localize output
405      try:
406          version_string = subprocess.check_output([compiler, '-v'], stderr=subprocess.STDOUT, env=env).decode(*SUBPROCESS_DECODE_ARGS)
407      except subprocess.CalledProcessError:
408          # If '-v' fails, try '--version'
409          version_string = subprocess.check_output([compiler, '--version'], stderr=subprocess.STDOUT, env=env).decode(*SUBPROCESS_DECODE_ARGS)
410      if IS_LINUX:
411          # Check for 'gcc' or 'g++' for sccache wrapper
412          pattern = re.compile("^COLLECT_GCC=(.*)$", re.MULTILINE)
413          results = re.findall(pattern, version_string)
414          if len(results) != 1:
415              # Clang is also a supported compiler on Linux
416              # Though on Ubuntu it's sometimes called "Ubuntu clang version"
417              return 'clang version' in version_string
418          compiler_path = os.path.realpath(results[0].strip())
419          # On RHEL/CentOS c++ is a gcc compiler wrapper
420          if os.path.basename(compiler_path) == 'c++' and 'gcc version' in version_string:
421              return True
422          return any(name in compiler_path for name in _accepted_compilers_for_platform())
423      if IS_MACOS:
424          # Check for 'clang' or 'clang++'
425          return version_string.startswith("Apple clang")
426      return False

429  def get_compiler_abi_compatibility_and_version(compiler) -> tuple[bool, TorchVersion]:
430      """ 
431      Determine if the given compiler is ABI-compatible with PyTorch alongside its version. 
432   
433      Args: 
434          compiler (str): The compiler executable name to check (e.g. ``g++``). 
435              Must be executable in a shell process. 
436   
437      Returns: 
438          A tuple that contains a boolean that defines if the compiler is (likely) ABI-incompatible with PyTorch, 
439          followed by a `TorchVersion` string that contains the compiler version separated by dots. 
440      """
441      if not _is_binary_build():
442          return (True, TorchVersion('0.0.0'))
443      if os.environ.get('TORCH_DONT_CHECK_COMPILER_ABI') in ['ON', '1', 'YES', 'TRUE', 'Y']:
444          return (True, TorchVersion('0.0.0'))
445  
446      # First check if the compiler is one of the expected ones for the particular platform.
447      if not check_compiler_ok_for_platform(compiler):
448          logger.warning(WRONG_COMPILER_WARNING, compiler, _accepted_compilers_for_platform()[0], sys.platform, _accepted_compilers_for_platform()[0])
449          return (False, TorchVersion('0.0.0'))
450  
451      if IS_MACOS:
452          # There is no particular minimum version we need for clang, so we're good here.
453          return (True, TorchVersion('0.0.0'))
454      try:
455          if IS_LINUX:
456              minimum_required_version = MINIMUM_GCC_VERSION
457              versionstr = subprocess.check_output([compiler, '-dumpfullversion', '-dumpversion'])
458              version = versionstr.decode(*SUBPROCESS_DECODE_ARGS).strip().split('.')
459          else:
460              minimum_required_version = MINIMUM_MSVC_VERSION
461              compiler_info = subprocess.check_output(compiler, stderr=subprocess.STDOUT)
462              match = re.search(r'(\d+)\.(\d+)\.(\d+)', compiler_info.decode(*SUBPROCESS_DECODE_ARGS).strip())
463              version = ['0', '0', '0'] if match is None else list(match.groups())
464      except Exception:
465          _, error, _ = sys.exc_info()
466          logger.warning('Error checking compiler version for %s: %s', compiler, error)
467          return (False, TorchVersion('0.0.0'))
468  
469      # convert alpha-numeric string to numeric string
470      # amdclang++ returns str like 0.0.0git, others return 0.0.0
471      numeric_version = [re.sub(r'\D', '', v) for v in version]
472  
473      if tuple(map(int, numeric_version)) >= minimum_required_version:
474          return (True, TorchVersion('.'.join(numeric_version)))
475  
476      compiler = f'{compiler} {".".join(numeric_version)}'
477      logger.warning(ABI_INCOMPATIBILITY_WARNING, compiler)
478  
479      return (False, TorchVersion('.'.join(numeric_version)))

482  def _check_cuda_version(compiler_name: str, compiler_version: TorchVersion) -> None:
483      if not CUDA_HOME:
484          raise RuntimeError(CUDA_NOT_FOUND_MESSAGE)
485  
486      nvcc = os.path.join(CUDA_HOME, 'bin', 'nvcc.exe' if IS_WINDOWS else 'nvcc')
487      if not os.path.exists(nvcc):
488          raise FileNotFoundError(f"nvcc not found at '{nvcc}'. Ensure CUDA path '{CUDA_HOME}' is correct.")
489  
490      cuda_version_str = subprocess.check_output([nvcc, '--version']).strip().decode(*SUBPROCESS_DECODE_ARGS)
491      cuda_version = re.search(r'release (\d+[.]\d+)', cuda_version_str)
492      if cuda_version is None:
493          return
494  
495      cuda_str_version = cuda_version.group(1)
496      cuda_ver = Version(cuda_str_version)
497      if torch.version.cuda is None:
498          return
499  
500      torch_cuda_version = Version(torch.version.cuda)
501      if cuda_ver != torch_cuda_version:
502          # major/minor attributes are only available in setuptools>=49.4.0
503          if getattr(cuda_ver, "major", None) is None:
504              raise ValueError("setuptools>=49.4.0 is required")
505          if cuda_ver.major != torch_cuda_version.major:
506              raise RuntimeError(CUDA_MISMATCH_MESSAGE, cuda_str_version, torch.version.cuda)
507          logger.warning(CUDA_MISMATCH_WARN, cuda_str_version, torch.version.cuda)
508  
509      if not (sys.platform.startswith('linux') and
510              os.environ.get('TORCH_DONT_CHECK_COMPILER_ABI') not in ['ON', '1', 'YES', 'TRUE', 'Y'] and
511              _is_binary_build()):
512          return
513  
514      cuda_compiler_bounds: VersionMap = CUDA_CLANG_VERSIONS if compiler_name.startswith('clang') else CUDA_GCC_VERSIONS
515  
516      if cuda_str_version not in cuda_compiler_bounds:
517          logger.warning('There are no %s version bounds defined for CUDA version %s', compiler_name, cuda_str_version)
518      else:
519          min_compiler_version, max_excl_compiler_version = cuda_compiler_bounds[cuda_str_version]
520          # Special case for 11.4.0, which has lower compiler bounds than 11.4.1
521          if "V11.4.48" in cuda_version_str and cuda_compiler_bounds == CUDA_GCC_VERSIONS:

532                  f'Please make sure to use an adequate version of {compiler_name} ({version_bound_str}).'
533              )
534          if compiler_version >= TorchVersion(max_excl_compiler_version_str):
535              raise RuntimeError(
536                  f'The current installed version of {compiler_name} ({compiler_version}) is greater '
537                  f'than the maximum required version by CUDA {cuda_str_version}. '
538                  f'Please make sure to use an adequate version of {compiler_name} ({version_bound_str}).'

568  class BuildExtension(build_ext):
569      """ 
570      A custom :mod:`setuptools` build extension . 
571   
572      This :class:`setuptools.build_ext` subclass takes care of passing the 
573      minimum required compiler flags (e.g. ``-std=c++17``) as well as mixed 
574      C++/CUDA/SYCL compilation (and support for CUDA/SYCL files in general). 
575   
576      When using :class:`BuildExtension`, it is allowed to supply a dictionary 
577      for ``extra_compile_args`` (rather than the usual list) that maps from 
578      languages/compilers (the only expected values are ``cxx``, ``nvcc`` or 
579      ``sycl``) to a list of additional compiler flags to supply to the compiler. 
580      This makes it possible to supply different flags to the C++, CUDA and SYCL 
581      compiler during mixed compilation. 
582   
583      ``use_ninja`` (bool): If ``use_ninja`` is ``True`` (default), then we 
584      attempt to build using the Ninja backend. Ninja greatly speeds up 
585      compilation compared to the standard ``setuptools.build_ext``. 
586      Fallbacks to the standard distutils backend if Ninja is not available. 
587   
588      .. note:: 
589          By default, the Ninja backend uses #CPUS + 2 workers to build the 
590          extension. This may use up too many resources on some systems. One 
591          can control the number of workers by setting the `MAX_JOBS` environment 
592          variable to a non-negative number. 
593      """
594  
595      @classmethod
596      def with_options(cls, **options):
597          """Return a subclass with alternative constructor that extends any original keyword arguments to the original constructor with the given options."""
598          class cls_with_options(cls):  # type: ignore[misc, valid-type]
599              def __init__(self, *args, **kwargs):
600                  kwargs.update(options)
601                  super().__init__(*args, **kwargs)
602  
603          return cls_with_options

605      def __init__(self, *args, **kwargs) -> None:
606          super().__init__(*args, **kwargs)
607          self.no_python_abi_suffix = kwargs.get("no_python_abi_suffix", False)
608  
609          self.use_ninja = kwargs.get('use_ninja', True)
610          if self.use_ninja:
611              # Test if we can use ninja. Fallback otherwise.
612              msg = ('Attempted to use ninja as the BuildExtension backend but '
613                     '%s. Falling back to using the slow distutils backend.')
614              if not is_ninja_available():
615                  logger.warning(msg, 'we could not find ninja.')
616                  self.use_ninja = False

623      def build_extensions(self) -> None:
624          compiler_name, compiler_version = self._check_abi()
625  
626          cuda_ext = False
627          sycl_ext = False
628          extension_iter = iter(self.extensions)
629          extension = next(extension_iter, None)
630          while not (cuda_ext and sycl_ext) and extension:
631              for source in extension.sources:
632                  _, ext = os.path.splitext(source)
633                  if ext == '.cu':
634                      cuda_ext = True
635                  elif ext == '.sycl':
636                      sycl_ext = True
637  
638                  # This check accounts on a case when cuda and sycl sources
639                  # are mixed in the same extension. We can stop checking
640                  # sources if both are found or there is no more sources.
641                  if cuda_ext and sycl_ext:
642                      break
643  
644              extension = next(extension_iter, None)
645  
646          if sycl_ext:
647              assert self.use_ninja, "ninja is required to build sycl extensions."
648  
649          if cuda_ext and not IS_HIP_EXTENSION:
650              _check_cuda_version(compiler_name, compiler_version)
651  
652          for extension in self.extensions:
653              # Ensure at least an empty list of flags for 'cxx', 'nvcc' and 'sycl' when
654              # extra_compile_args is a dict. Otherwise, default torch flags do
655              # not get passed. Necessary when only one of 'cxx', 'nvcc' or 'sycl' is
656              # passed to extra_compile_args in CUDAExtension or SyclExtension, i.e.
657              #   CUDAExtension(..., extra_compile_args={'cxx': [...]})
658              # or
659              #   CUDAExtension(..., extra_compile_args={'nvcc': [...]})
660              if isinstance(extension.extra_compile_args, dict):
661                  for ext in ['cxx', 'nvcc', 'sycl']:
662                      if ext not in extension.extra_compile_args:
663                          extension.extra_compile_args[ext] = []
664  
665              self._add_compile_flag(extension, '-DTORCH_API_INCLUDE_EXTENSION_H')
666  
667              if IS_HIP_EXTENSION:
668                  self._hipify_compile_flags(extension)
669  
670              if extension.py_limited_api:
671                  # compile any extension that has passed in py_limited_api to the
672                  # Extension constructor with the Py_LIMITED_API flag set to our
673                  # min supported CPython version.
674                  # See https://docs.python.org/3/c-api/stable.html#c.Py_LIMITED_API
675                  self._add_compile_flag(extension, f'-DPy_LIMITED_API={min_supported_cpython}')
676              else:
677                  # pybind11 is not CPython API stable so don't add these flags used when
678                  # compiling pybind11 when pybind11 is not even used. otherwise, the build
679                  # logs are confusing.
680                  # See note [Pybind11 ABI constants]
681                  for name in ["COMPILER_TYPE", "STDLIB", "BUILD_ABI"]:
682                      val = getattr(torch._C, f"_PYBIND11_{name}")
683                      if val is not None and not IS_WINDOWS:
684                          self._add_compile_flag(extension, f'-DPYBIND11_{name}="{val}"')
685              self._define_torch_extension_name(extension)
686  
687              if 'nvcc_dlink' in extension.extra_compile_args:
688                  assert self.use_ninja, f"With dlink=True, ninja is required to build cuda extension {extension.name}."
689  
690          # Register .cu, .cuh, .hip, .mm and .sycl as valid source extensions.
691          # NOTE: At the moment .sycl is not a standard extension for SYCL supported
692          # by compiler. Here we introduce a torch level convention that SYCL sources
693          # should have .sycl file extension.
694          self.compiler.src_extensions += ['.cu', '.cuh', '.hip', '.sycl']
695          if torch.backends.mps.is_built():
696              self.compiler.src_extensions += ['.mm']
697          # Save the original _compile method for later.
698          if self.compiler.compiler_type == 'msvc':
699              self.compiler._cpp_extensions += ['.cu', '.cuh']
700              original_compile = self.compiler.compile
701              original_spawn = self.compiler.spawn
702          else:
703              original_compile = self.compiler._compile

705          def append_std17_if_no_std_present(cflags) -> None:
706              # NVCC does not allow multiple -std to be passed, so we avoid
707              # overriding the option if the user explicitly passed it.
708              cpp_format_prefix = '/{}:' if self.compiler.compiler_type == 'msvc' else '-{}='
709              cpp_flag_prefix = cpp_format_prefix.format('std')
710              cpp_flag = cpp_flag_prefix + 'c++17'
711              if not any(flag.startswith(cpp_flag_prefix) for flag in cflags):
712                  cflags.append(cpp_flag)

719              # NVCC does not allow multiple -ccbin/--compiler-bindir to be passed, so we avoid
720              # overriding the option if the user explicitly passed it.
721              _ccbin = os.getenv("CC")

723                  _ccbin is not None
724                  and not any(flag.startswith(('-ccbin', '--compiler-bindir')) for flag in cflags)

762          def unix_wrap_ninja_compile(sources,

764                                      macros=None,
765                                      include_dirs=None,
766                                      debug=0,
767                                      extra_preargs=None,
768                                      extra_postargs=None,
769                                      depends=None):
770              r"""Compiles sources by outputting a ninja file and running it."""
771              # NB: I copied some lines from self.compiler (which is an instance
772              # of distutils.UnixCCompiler). See the following link.
773              # https://github.com/python/cpython/blob/f03a8f8d5001963ad5b5b28dbd95497e9cc15596/Lib/distutils/ccompiler.py#L564-L567
774              # This can be fragile, but a lot of other repos also do this
775              # (see https://github.com/search?q=_setup_compile&type=Code)
776              # so it is probably OK; we'll also get CI signal if/when
777              # we update our python version (which is when distutils can be
778              # upgraded)

781              # (`objects`) get generated with absolute paths.
782              output_dir = os.path.abspath(output_dir)

792              extra_cc_cflags = self.compiler.compiler_so[1:]
793              with_cuda = any(map(_is_cuda_file, sources))
794              with_sycl = any(map(_is_sycl_file, sources))
795  
796              # extra_postargs can be either:
797              # - a dict mapping cxx/nvcc/sycl to extra flags
798              # - a list of extra flags.
799              if isinstance(extra_postargs, dict):
800                  post_cflags = extra_postargs['cxx']
801              else:

811                  if isinstance(extra_postargs, dict):
812                      cuda_post_cflags = extra_postargs['nvcc']
813                  else:
814                      cuda_post_cflags = list(extra_postargs)
815                  if IS_HIP_EXTENSION:
816                      cuda_post_cflags = cuda_post_cflags + _get_rocm_arch_flags(cuda_post_cflags)
817                      cuda_post_cflags = COMMON_HIP_FLAGS + COMMON_HIPCC_FLAGS + cuda_post_cflags
818                  else:
819                      cuda_post_cflags = unix_cuda_flags(cuda_post_cflags)
820                  append_std17_if_no_std_present(cuda_post_cflags)
821                  cuda_cflags = [shlex.quote(f) for f in cuda_cflags]
822                  cuda_post_cflags = [shlex.quote(f) for f in cuda_post_cflags]
823  
824              if isinstance(extra_postargs, dict) and 'nvcc_dlink' in extra_postargs:
825                  cuda_dlink_post_cflags = unix_cuda_flags(extra_postargs['nvcc_dlink'])
826                  cuda_dlink_post_cflags = [shlex.quote(f) for f in cuda_dlink_post_cflags]
827              else:
828                  cuda_dlink_post_cflags = None
829  
830              sycl_post_cflags = None
831              sycl_cflags = None
832              sycl_dlink_post_cflags = None
833              if with_sycl:
834                  sycl_cflags = extra_cc_cflags + common_cflags + _COMMON_SYCL_FLAGS
835                  if isinstance(extra_postargs, dict):
836                      sycl_post_cflags = extra_postargs['sycl']
837                  else:
838                      sycl_post_cflags = list(extra_postargs)
839                  _append_sycl_targets_if_missing(sycl_post_cflags)
840                  append_std17_if_no_std_present(sycl_cflags)
841                  _append_sycl_std_if_no_std_present(sycl_cflags)
842                  host_cflags = extra_cc_cflags + common_cflags + post_cflags
843                  append_std17_if_no_std_present(host_cflags)
844                  # escaping quoted arguments to pass them thru SYCL compiler
845                  host_cflags = [item.replace('"', '\\\\"') for item in host_cflags]
846                  host_cflags = ' '.join(host_cflags)
847                  # Note the order: shlex.quote sycl_flags first, _wrap_sycl_host_flags
848                  # second. Reason is that sycl host flags are quoted, space containing
849                  # strings passed to SYCL compiler.
850                  sycl_cflags = [shlex.quote(f) for f in sycl_cflags]
851                  sycl_cflags += _wrap_sycl_host_flags(host_cflags)
852                  sycl_dlink_post_cflags = _SYCL_DLINK_FLAGS.copy()
853                  sycl_dlink_post_cflags += _get_sycl_device_flags(sycl_post_cflags)
854                  sycl_post_cflags = [shlex.quote(f) for f in sycl_post_cflags]
855  
856              _write_ninja_file_and_compile_objects(
857                  sources=sources,
858                  objects=objects,
859                  cflags=[shlex.quote(f) for f in extra_cc_cflags + common_cflags],
860                  post_cflags=[shlex.quote(f) for f in post_cflags],
861                  cuda_cflags=cuda_cflags,
862                  cuda_post_cflags=cuda_post_cflags,
863                  cuda_dlink_post_cflags=cuda_dlink_post_cflags,
864                  sycl_cflags=sycl_cflags,
865                  sycl_post_cflags=sycl_post_cflags,
866                  sycl_dlink_post_cflags=sycl_dlink_post_cflags,
867                  build_directory=output_dir,
868                  verbose=True,
869                  with_cuda=with_cuda,
870                  with_sycl=with_sycl)
871  
872              # Return *all* object filenames, not just the ones we just built.
873              return objects

882          def win_wrap_single_compile(sources,
883                                      output_dir=None,
884                                      macros=None,
885                                      include_dirs=None,
886                                      debug=0,
887                                      extra_preargs=None,
888                                      extra_postargs=None,
889                                      depends=None):
890  
891              self.cflags = copy.deepcopy(extra_postargs)
892              extra_postargs = None
893  
894              def spawn(cmd):
895                  # Using regex to match src, obj and include files
896                  src_regex = re.compile('/T(p|c)(.*)')
897                  src_list = [
898                      m.group(2) for m in (src_regex.match(elem) for elem in cmd)
899                      if m
900                  ]
901  
902                  obj_regex = re.compile('/Fo(.*)')
903                  obj_list = [
904                      m.group(1) for m in (obj_regex.match(elem) for elem in cmd)
905                      if m
906                  ]
907  
908                  include_regex = re.compile(r'((\-|\/)I.*)')
909                  include_list = [
910                      m.group(1)
911                      for m in (include_regex.match(elem) for elem in cmd) if m
912                  ]
913  
914                  if len(src_list) >= 1 and len(obj_list) >= 1:
915                      src = src_list[0]
916                      obj = obj_list[0]
917                      if _is_cuda_file(src):
918                          if IS_HIP_EXTENSION:
919                              nvcc = _get_hipcc_path()
920                          else:
921                              nvcc = _join_cuda_home('bin', 'nvcc')
922                          if isinstance(self.cflags, dict):
923                              cflags = self.cflags['nvcc']
924                          elif isinstance(self.cflags, list):
925                              cflags = self.cflags
926                          else:
927                              cflags = []
928  
929                          if IS_HIP_EXTENSION:
930                              cflags = win_hip_flags(cflags)
931                          else:
932                              cflags = win_cuda_flags(cflags) + ['-std=c++17', '--use-local-env']
933                              for ignore_warning in MSVC_IGNORE_CUDAFE_WARNINGS:
934                                  cflags = ['-Xcudafe', '--diag_suppress=' + ignore_warning] + cflags
935                          for flag in COMMON_MSVC_FLAGS:
936                              cflags = ['-Xcompiler', flag] + cflags
937                          cmd = [nvcc, '-c', src, '-o', obj] + include_list + cflags
938                      elif isinstance(self.cflags, dict):
939                          cflags = COMMON_MSVC_FLAGS + self.cflags['cxx']
940                          append_std17_if_no_std_present(cflags)
941                          cmd += cflags
942                      elif isinstance(self.cflags, list):
943                          cflags = COMMON_MSVC_FLAGS + self.cflags
944                          append_std17_if_no_std_present(cflags)
945                          cmd += cflags
946  
947                  return original_spawn(cmd)
948  
949              try:
950                  self.compiler.spawn = spawn
951                  return original_compile(sources, output_dir, macros,
952                                          include_dirs, debug, extra_preargs,
953                                          extra_postargs, depends)
954              finally:
955                  self.compiler.spawn = original_spawn

1572 def load(name,
1573          sources: Union[str, list[str]],
1574          extra_cflags=None,
1575          extra_cuda_cflags=None,
1576          extra_sycl_cflags=None,
1577          extra_ldflags=None,
1578          extra_include_paths=None,
1579          build_directory=None,
1580          verbose=False,
1581          with_cuda: Optional[bool] = None,
1582          with_sycl: Optional[bool] = None,
1583          is_python_module=True,
1584          is_standalone=False,
1585          keep_intermediates=True):
1586     """ 
1587     Load a PyTorch C++ extension just-in-time (JIT). 
1588  
1589     To load an extension, a Ninja build file is emitted, which is used to 
1590     compile the given sources into a dynamic library. This library is 
1591     subsequently loaded into the current Python process as a module and 
1592     returned from this function, ready for use. 
1593  
1594     By default, the directory to which the build file is emitted and the 
1595     resulting library compiled to is ``<tmp>/torch_extensions/<name>``, where 
1596     ``<tmp>`` is the temporary folder on the current platform and ``<name>`` 
1597     the name of the extension. This location can be overridden in two ways. 
1598     First, if the ``TORCH_EXTENSIONS_DIR`` environment variable is set, it 
1599     replaces ``<tmp>/torch_extensions`` and all extensions will be compiled 
1600     into subfolders of this directory. Second, if the ``build_directory`` 
1601     argument to this function is supplied, it overrides the entire path, i.e. 
1602     the library will be compiled into that folder directly. 
1603  
1604     To compile the sources, the default system compiler (``c++``) is used, 
1605     which can be overridden by setting the ``CXX`` environment variable. To pass 
1606     additional arguments to the compilation process, ``extra_cflags`` or 
1607     ``extra_ldflags`` can be provided. For example, to compile your extension 
1608     with optimizations, pass ``extra_cflags=['-O3']``. You can also use 
1609     ``extra_cflags`` to pass further include directories. 
1610  
1611     CUDA support with mixed compilation is provided. Simply pass CUDA source 
1612     files (``.cu`` or ``.cuh``) along with other sources. Such files will be 
1613     detected and compiled with nvcc rather than the C++ compiler. This includes 
1614     passing the CUDA lib64 directory as a library directory, and linking 
1615     ``cudart``. You can pass additional flags to nvcc via 
1616     ``extra_cuda_cflags``, just like with ``extra_cflags`` for C++. Various 
1617     heuristics for finding the CUDA install directory are used, which usually 
1618     work fine. If not, setting the ``CUDA_HOME`` environment variable is the 
1619     safest option. 
1620  
1621     SYCL support with mixed compilation is provided. Simply pass SYCL source 
1622     files (``.sycl``) along with other sources. Such files will be detected 
1623     and compiled with SYCL compiler (such as Intel DPC++ Compiler) rather 
1624     than the C++ compiler. You can pass additional flags to SYCL compiler 
1625     via ``extra_sycl_cflags``, just like with ``extra_cflags`` for C++. 
1626     SYCL compiler is expected to be found via system PATH environment 
1627     variable. 
1628  
1629     Args: 
1630         name: The name of the extension to build. This MUST be the same as the 
1631             name of the pybind11 module! 
1632         sources: A list of relative or absolute paths to C++ source files. 
1633         extra_cflags: optional list of compiler flags to forward to the build. 
1634         extra_cuda_cflags: optional list of compiler flags to forward to nvcc 
1635             when building CUDA sources. 
1636         extra_sycl_cflags: optional list of compiler flags to forward to SYCL 
1637             compiler when building SYCL sources. 
1638         extra_ldflags: optional list of linker flags to forward to the build. 
1639         extra_include_paths: optional list of include directories to forward 
1640             to the build. 
1641         build_directory: optional path to use as build workspace. 
1642         verbose: If ``True``, turns on verbose logging of load steps. 
1643         with_cuda: Determines whether CUDA headers and libraries are added to 
1644             the build. If set to ``None`` (default), this value is 
1645             automatically determined based on the existence of ``.cu`` or 
1646             ``.cuh`` in ``sources``. Set it to `True`` to force CUDA headers 
1647             and libraries to be included. 
1648         with_sycl: Determines whether SYCL headers and libraries are added to 
1649             the build. If set to ``None`` (default), this value is 
1650             automatically determined based on the existence of ``.sycl`` in 
1651             ``sources``. Set it to `True`` to force SYCL headers and 
1652             libraries to be included. 
1653         is_python_module: If ``True`` (default), imports the produced shared 
1654             library as a Python module. If ``False``, behavior depends on 
1655             ``is_standalone``. 
1656         is_standalone: If ``False`` (default) loads the constructed extension 
1657             into the process as a plain dynamic library. If ``True``, build a 
1658             standalone executable. 
1659  
1660     Returns: 
1661         If ``is_python_module`` is ``True``: 
1662             Returns the loaded PyTorch extension as a Python module. 
1663  
1664         If ``is_python_module`` is ``False`` and ``is_standalone`` is ``False``: 
1665             Returns nothing. (The shared library is loaded into the process as 
1666             a side effect.) 
1667  
1668         If ``is_standalone`` is ``True``. 
1669             Return the path to the executable. (On Windows, TORCH_LIB_PATH is 
1670             added to the PATH environment variable as a side effect.) 
1671  
1672     Example: 
1673         >>> # xdoctest: +SKIP 
1674         >>> from torch.utils.cpp_extension import load 
1675         >>> module = load( 
1676         ...     name='extension', 
1677         ...     sources=['extension.cpp', 'extension_kernel.cu'], 
1678         ...     extra_cflags=['-O2'], 
1679         ...     verbose=True) 
1680     """
1681     return _jit_compile(
1682         name,
1683         [sources] if isinstance(sources, str) else sources,
1684         extra_cflags,
1685         extra_cuda_cflags,
1686         extra_sycl_cflags,
1687         extra_ldflags,
1688         extra_include_paths,
1689         build_directory or _get_build_directory(name, verbose),
1690         verbose,
1691         with_cuda,
1692         with_sycl,
1693         is_python_module,
1694         is_standalone,
1695         keep_intermediates=keep_intermediates)

1697 def _get_pybind11_abi_build_flags():
1698     # Note [Pybind11 ABI constants]
1699     #
1700     # Pybind11 before 2.4 used to build an ABI strings using the following pattern:
1701     # f"__pybind11_internals_v{PYBIND11_INTERNALS_VERSION}{PYBIND11_INTERNALS_KIND}{PYBIND11_BUILD_TYPE}__"
1702     # Since 2.4 compier type, stdlib and build abi parameters are also encoded like this:
1703     # f"__pybind11_internals_v{PYBIND11_INTERNALS_VERSION}{PYBIND11_INTERNALS_KIND}{PYBIND11_COMPILER_TYPE}{PYBIND11_STDLIB}{PYBIND11_BUILD_ABI}{PYBIND11_BUILD_TYPE}__"
1704     #
1705     # This was done in order to further narrow down the chances of compiler ABI incompatibility
1706     # that can cause a hard to debug segfaults.
1707     # For PyTorch extensions we want to relax those restrictions and pass compiler, stdlib and abi properties
1708     # captured during PyTorch native library compilation in torch/csrc/Module.cpp
1709 
1710     abi_cflags = []
1711     for pname in ["COMPILER_TYPE", "STDLIB", "BUILD_ABI"]:
1712         pval = getattr(torch._C, f"_PYBIND11_{pname}")
1713         if pval is not None and not IS_WINDOWS:
1714             abi_cflags.append(f'-DPYBIND11_{pname}=\\"{pval}\\"')
1715     return abi_cflags

1717 def check_compiler_is_gcc(compiler):
1718     if not IS_LINUX:
1719         return False
1720 
1721     env = os.environ.copy()
1722     env['LC_ALL'] = 'C'  # Don't localize output
1723     try:
1724         version_string = subprocess.check_output([compiler, '-v'], stderr=subprocess.STDOUT, env=env).decode(*SUBPROCESS_DECODE_ARGS)
1725     except Exception:
1726         try:
1727             version_string = subprocess.check_output([compiler, '--version'], stderr=subprocess.STDOUT, env=env).decode(*SUBPROCESS_DECODE_ARGS)
1728         except Exception:
1729             return False
1730     # Check for 'gcc' or 'g++' for sccache wrapper
1731     pattern = re.compile("^COLLECT_GCC=(.*)$", re.MULTILINE)
1732     results = re.findall(pattern, version_string)
1733     if len(results) != 1:
1734         return False
1735     compiler_path = os.path.realpath(results[0].strip())
1736     # On RHEL/CentOS c++ is a gcc compiler wrapper
1737     if os.path.basename(compiler_path) == 'c++' and 'gcc version' in version_string:
1738         return True
1739     return False

1741 def _check_and_build_extension_h_precompiler_headers(

1746     Precompiled Headers(PCH) can pre-build the same headers and reduce build time for pytorch load_inline modules. 
1747     GCC offical manual: https://gcc.gnu.org/onlinedocs/gcc-4.0.4/gcc/Precompiled-Headers.html 
1748     PCH only works when built pch file(header.h.gch) and build target have the same build parameters. So, We need 
1749     add a signature file to record PCH file parameters. If the build parameters(signature) changed, it should rebuild 
1750     PCH file. 
1751  
1752     Note: 
1753     1. Windows and MacOS have different PCH mechanism. We only support Linux currently. 
1754     2. It only works on GCC/G++. 
1755     '''
1756     if not IS_LINUX:
1757         return
1758 
1759     compiler = get_cxx_compiler()
1760 
1761     b_is_gcc = check_compiler_is_gcc(compiler)
1762     if b_is_gcc is False:
1763         return

1766     head_file_pch = os.path.join(_TORCH_PATH, 'include', 'torch', 'extension.h.gch')
1767     head_file_signature = os.path.join(_TORCH_PATH, 'include', 'torch', 'extension.h.sign')

1769     def listToString(s):
1770         # initialize an empty string
1771         string = ""
1772         if s is None:
1773             return string
1774 
1775         # traverse in the string
1776         for element in s:
1777             string += (element + ' ')
1778         # return string
1779         return string

1782         return re.sub(
1783             r"[ \n]+",
1784             " ",
1785             f"""
1786                 {compiler} -x c++-header {head_file} -o {head_file_pch} {torch_include_dirs} {extra_include_paths} {extra_cflags} {common_cflags}
1787             """,
1788         ).strip()

1791         signature = cmd.replace(' ', '_')

1799         with open(file_path) as file:
1800             # read all content of a file

1872 def load_inline(name,
1873                 cpp_sources,
1874                 cuda_sources=None,
1875                 sycl_sources=None,
1876                 functions=None,
1877                 extra_cflags=None,
1878                 extra_cuda_cflags=None,
1879                 extra_sycl_cflags=None,
1880                 extra_ldflags=None,
1881                 extra_include_paths=None,
1882                 build_directory=None,
1883                 verbose=False,
1884                 with_cuda=None,
1885                 with_sycl=None,
1886                 is_python_module=True,
1887                 with_pytorch_error_handling=True,
1888                 keep_intermediates=True,
1889                 use_pch=False,
1890                 no_implicit_headers=False):
1891     r''' 
1892     Load a PyTorch C++ extension just-in-time (JIT) from string sources. 
1893  
1894     This function behaves exactly like :func:`load`, but takes its sources as 
1895     strings rather than filenames. These strings are stored to files in the 
1896     build directory, after which the behavior of :func:`load_inline` is 
1897     identical to :func:`load`. 
1898  
1899     See `the 
1900     tests <https://github.com/pytorch/pytorch/blob/master/test/test_cpp_extensions_jit.py>`_ 
1901     for good examples of using this function. 
1902  
1903     Sources may omit two required parts of a typical non-inline C++ extension: 
1904     the necessary header includes, as well as the (pybind11) binding code. More 
1905     precisely, strings passed to ``cpp_sources`` are first concatenated into a 
1906     single ``.cpp`` file. This file is then prepended with ``#include 
1907     <torch/extension.h>`` 
1908  
1909     Furthermore, if the ``functions`` argument is supplied, bindings will be 
1910     automatically generated for each function specified. ``functions`` can 
1911     either be a list of function names, or a dictionary mapping from function 
1912     names to docstrings. If a list is given, the name of each function is used 
1913     as its docstring. 
1914  
1915     The sources in ``cuda_sources`` are concatenated into a separate ``.cu`` 
1916     file and  prepended with ``torch/types.h``, ``cuda.h`` and 
1917     ``cuda_runtime.h`` includes. The ``.cpp`` and ``.cu`` files are compiled 
1918     separately, but ultimately linked into a single library. Note that no 
1919     bindings are generated for functions in ``cuda_sources`` per se. To bind 
1920     to a CUDA kernel, you must create a C++ function that calls it, and either 
1921     declare or define this C++ function in one of the ``cpp_sources`` (and 
1922     include its name in ``functions``). 
1923  
1924     The sources in ``sycl_sources`` are concatenated into a separate ``.sycl`` 
1925     file and  prepended with ``torch/types.h``, ``sycl/sycl.hpp`` includes. 
1926     The ``.cpp`` and ``.sycl`` files are compiled separately, but ultimately 
1927     linked into a single library. Note that no bindings are generated for 
1928     functions in ``sycl_sources`` per se. To bind to a SYCL kernel, you must 
1929     create a C++ function that calls it, and either declare or define this 
1930     C++ function in one of the ``cpp_sources`` (and include its name 
1931     in ``functions``). 
1932  
1933  
1934  
1935     See :func:`load` for a description of arguments omitted below. 
1936  
1937     Args: 
1938         cpp_sources: A string, or list of strings, containing C++ source code. 
1939         cuda_sources: A string, or list of strings, containing CUDA source code. 
1940         sycl_sources: A string, or list of strings, containing SYCL source code. 
1941         functions: A list of function names for which to generate function 
1942             bindings. If a dictionary is given, it should map function names to 
1943             docstrings (which are otherwise just the function names). 
1944         with_cuda: Determines whether CUDA headers and libraries are added to 
1945             the build. If set to ``None`` (default), this value is 
1946             automatically determined based on whether ``cuda_sources`` is 
1947             provided. Set it to ``True`` to force CUDA headers 
1948             and libraries to be included. 
1949         with_sycl: Determines whether SYCL headers and libraries are added to 
1950             the build. If set to ``None`` (default), this value is 
1951             automatically determined based on whether ``sycl_sources`` is 
1952             provided. Set it to ``True`` to force SYCL headers 
1953             and libraries to be included. 
1954         with_pytorch_error_handling: Determines whether pytorch error and 
1955             warning macros are handled by pytorch instead of pybind. To do 
1956             this, each function ``foo`` is called via an intermediary ``_safe_foo`` 
1957             function. This redirection might cause issues in obscure cases 
1958             of cpp. This flag should be set to ``False`` when this redirect 
1959             causes issues. 
1960         no_implicit_headers: If ``True``, skips automatically adding headers, most notably 
1961             ``#include <torch/extension.h>`` and ``#include <torch/types.h>`` lines. 
1962             Use this option to improve cold start times when you 
1963             already include the necessary headers in your source code. Default: ``False``. 
1964  
1965     Example: 
1966         >>> # xdoctest: +REQUIRES(env:TORCH_DOCTEST_CPP_EXT) 
1967         >>> from torch.utils.cpp_extension import load_inline 
1968         >>> source = """ 
1969         at::Tensor sin_add(at::Tensor x, at::Tensor y) { 
1970           return x.sin() + y.sin(); 
1971         } 
1972         """ 
1973         >>> module = load_inline(name='inline_extension', 
1974         ...                      cpp_sources=[source], 
1975         ...                      functions=['sin_add']) 
1976  
1977     .. note:: 
1978         Since load_inline will just-in-time compile the source code, please ensure 
1979         that you have the right toolchains installed in the runtime. For example, 
1980         when loading C++, make sure a C++ compiler is available. If you're loading 
1981         a CUDA extension, you will need to additionally install the corresponding CUDA 
1982         toolkit (nvcc and any other dependencies your code has). Compiling toolchains 
1983         are not included when you install torch and must be additionally installed. 
1984  
1985         During compiling, by default, the Ninja backend uses #CPUS + 2 workers to build 
1986         the extension. This may use up too many resources on some systems. One 
1987         can control the number of workers by setting the `MAX_JOBS` environment 
1988         variable to a non-negative number. 
1989     '''
1990     build_directory = build_directory or _get_build_directory(name, verbose)
1991 
1992     if isinstance(cpp_sources, str):
1993         cpp_sources = [cpp_sources]
1994     cuda_sources = cuda_sources or []
1995     if isinstance(cuda_sources, str):
1996         cuda_sources = [cuda_sources]
1997     sycl_sources = sycl_sources or []
1998     if isinstance(sycl_sources, str):
1999         sycl_sources = [sycl_sources]
2000 
2001     if not no_implicit_headers:
2002         cpp_sources.insert(0, '#include <torch/extension.h>')
2003 
2004     if use_pch is True:
2005         # Using PreCompile Header('torch/extension.h') to reduce compile time.
2006         _check_and_build_extension_h_precompiler_headers(extra_cflags, extra_include_paths)
2007     else:
2008         remove_extension_h_precompiler_headers()
2009 
2010     # If `functions` is supplied, we create the pybind11 bindings for the user.
2011     # Here, `functions` is (or becomes, after some processing) a map from
2012     # function names to function docstrings.
2013     if functions is not None:
2014         module_def = []
2015         module_def.append('PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {')
2016         if isinstance(functions, str):
2017             functions = [functions]
2018         if isinstance(functions, list):
2019             # Make the function docstring the same as the function name.
2020             functions = {f: f for f in functions}
2021         elif not isinstance(functions, dict):
2022             raise ValueError(f"Expected 'functions' to be a list or dict, but was {type(functions)}")
2023         for function_name, docstring in functions.items():
2024             if with_pytorch_error_handling:
2025                 module_def.append(f'm.def("{function_name}", torch::wrap_pybind_function({function_name}), "{docstring}");')
2026             else:
2027                 module_def.append(f'm.def("{function_name}", {function_name}, "{docstring}");')
2028         module_def.append('}')
2029         cpp_sources += module_def
2030 
2031     cpp_source_path = os.path.join(build_directory, 'main.cpp')
2032     _maybe_write(cpp_source_path, "\n".join(cpp_sources))
2033 
2034     sources = [cpp_source_path]
2035 
2036     if cuda_sources:
2037         if not no_implicit_headers:
2038             cuda_sources.insert(0, '#include <torch/types.h>')
2039             cuda_sources.insert(1, '#include <cuda.h>')
2040             cuda_sources.insert(2, '#include <cuda_runtime.h>')

2371 def _get_cuda_arch_flags(cflags: Optional[list[str]] = None) -> list[str]:
2372     """ 
2373     Determine CUDA arch flags to use. 
2374  
2375     For an arch, say "6.1", the added compile flag will be 
2376     ``-gencode=arch=compute_61,code=sm_61``. 
2377     For an added "+PTX", an additional 
2378     ``-gencode=arch=compute_xx,code=compute_xx`` is added. 
2379  
2380     See select_compute_arch.cmake for corresponding named and supported arches 
2381     when building with CMake. 
2382     """
2383     # If cflags is given, there may already be user-provided arch flags in it
2384     # (from `extra_compile_args`)
2385     if cflags is not None:
2386         for flag in cflags:
2387             if 'TORCH_EXTENSION_NAME' in flag:
2388                 continue
2389             if 'arch' in flag:
2390                 return []
2391 
2392     # Note: keep combined names ("arch1+arch2") above single names, otherwise
2393     # string replacement may not do the right thing
2394     named_arches = collections.OrderedDict([
2395         ('Kepler+Tesla', '3.7'),
2396         ('Kepler', '3.5+PTX'),
2397         ('Maxwell+Tegra', '5.3'),
2398         ('Maxwell', '5.0;5.2+PTX'),
2399         ('Pascal', '6.0;6.1+PTX'),
2400         ('Volta+Tegra', '7.2'),
2401         ('Volta', '7.0+PTX'),
2402         ('Turing', '7.5+PTX'),
2403         ('Ampere+Tegra', '8.7'),
2404         ('Ampere', '8.0;8.6+PTX'),
2405         ('Ada', '8.9+PTX'),
2406         ('Hopper', '9.0+PTX'),
2407         ('Blackwell+Tegra', '10.1'),
2408         ('Blackwell', '10.0;10.3;12.0;12.1+PTX'),
2409     ])
2410 
2411     supported_arches = ['3.5', '3.7', '5.0', '5.2', '5.3', '6.0', '6.1', '6.2',
2412                         '7.0', '7.2', '7.5', '8.0', '8.6', '8.7', '8.9', '9.0', '9.0a',
2413                         '10.0', '10.0a', '10.1', '10.1a', '10.3', '10.3a', '12.0',
2414                         '12.0a', '12.1', '12.1a']
2415     valid_arch_strings = supported_arches + [s + "+PTX" for s in supported_arches]
2416 
2417     # The default is sm_30 for CUDA 9.x and 10.x
2418     # First check for an env var (same as used by the main setup.py)
2419     # Can be one or more architectures, e.g. "6.1" or "3.5;5.2;6.0;6.1;7.0+PTX"
2420     # See cmake/Modules_CUDA_fix/upstream/FindCUDA/select_compute_arch.cmake
2421     _arch_list = os.environ.get('TORCH_CUDA_ARCH_LIST', None)
2422 
2423     # If not given, determine what's best for the GPU / CUDA version that can be found
2424     if not _arch_list:
2425         logger.warning(
2426             "TORCH_CUDA_ARCH_LIST is not set, all archs for visible cards are included for compilation. \n"
2427             "If this is not desired, please set os.environ['TORCH_CUDA_ARCH_LIST'] to specific architectures.")
2428         arch_list = []
2429         # the assumption is that the extension should run on any of the currently visible cards,
2430         # which could be of different types - therefore all archs for visible cards should be included
2431         for i in range(torch.cuda.device_count()):
2432             capability = torch.cuda.get_device_capability(i)
2433             supported_sm = [int("".join(re.findall(r"\d+", arch.split('_')[1])))
2434                             for arch in torch.cuda.get_arch_list() if 'sm_' in arch]
2435             max_supported_sm = max((sm // 10, sm % 10) for sm in supported_sm)
2436             # Capability of the device may be higher than what's supported by the user's
2437             # NVCC, causing compilation error. User's NVCC is expected to match the one
2438             # used to build pytorch, so we use the maximum supported capability of pytorch
2439             # to clamp the capability.
2440             capability = min(max_supported_sm, capability)
2441             arch = f'{capability[0]}.{capability[1]}'
2442             if arch not in arch_list:
2443                 arch_list.append(arch)
2444         arch_list = sorted(arch_list)
2445         arch_list[-1] += '+PTX'
2446     else:
2447         # Deal with lists that are ' ' separated (only deal with ';' after)
2448         _arch_list = _arch_list.replace(' ', ';')
2449         # Expand named arches
2450         for named_arch, archval in named_arches.items():
2451             _arch_list = _arch_list.replace(named_arch, archval)
2452 
2453         arch_list = _arch_list.split(';')
2454 
2455     flags = []
2456     for arch in arch_list:
2457         if arch not in valid_arch_strings:
2458             raise ValueError(f"Unknown CUDA arch ({arch}) or GPU not supported")
2459         else:
2460             # Handle both single and double-digit architecture versions
2461             version = arch.split('+')[0]  # Remove "+PTX" if present
2462             major, minor = version.split('.')
2463             num = f"{major}{minor}"
2464             flags.append(f'-gencode=arch=compute_{num},code=sm_{num}')
2465             if arch.endswith('+PTX'):
2466                 flags.append(f'-gencode=arch=compute_{num},code=compute_{num}')
2467 
2468     return sorted(set(flags))

2780 def _write_ninja_file(path,
2781                       cflags,
2782                       post_cflags,
2783                       cuda_cflags,
2784                       cuda_post_cflags,
2785                       cuda_dlink_post_cflags,
2786                       sycl_cflags,
2787                       sycl_post_cflags,
2788                       sycl_dlink_post_cflags,
2789                       sources,
2790                       objects,
2791                       ldflags,
2792                       library_target,
2793                       with_cuda,
2794                       with_sycl) -> None:
2795     r"""Write a ninja file that does the desired compiling and linking. 
2796  
2797     `path`: Where to write this file 
2798     `cflags`: list of flags to pass to $cxx. Can be None. 
2799     `post_cflags`: list of flags to append to the $cxx invocation. Can be None. 
2800     `cuda_cflags`: list of flags to pass to $nvcc. Can be None. 
2801     `cuda_post_cflags`: list of flags to append to the $nvcc invocation. Can be None. 
2802     `cuda_dlink_post_cflags`: list of flags to append to the $nvcc device code link invocation. Can be None. 
2803     `sycl_cflags`: list of flags to pass to SYCL compiler. Can be None. 
2804     `sycl_post_cflags`: list of flags to append to the SYCL compiler invocation. Can be None. 
2805     `sycl_dlink_post_cflags`: list of flags to append to the SYCL compiler device code link invocation. Can be None. 
2806 e. 
2807     `sources`: list of paths to source files 
2808     `objects`: list of desired paths to objects, one per source. 
2809     `ldflags`: list of flags to pass to linker. Can be None. 
2810     `library_target`: Name of the output library. Can be None; in that case, 
2811                       we do no linking. 
2812     `with_cuda`: If we should be compiling with CUDA. 
2813     """
2814     def sanitize_flags(flags):
2815         if flags is None:
2816             return []
2817         else:
2818             return [flag.strip() for flag in flags]
2819 
2820     cflags = sanitize_flags(cflags)
2821     post_cflags = sanitize_flags(post_cflags)
2822     cuda_cflags = sanitize_flags(cuda_cflags)
2823     cuda_post_cflags = sanitize_flags(cuda_post_cflags)
2824     cuda_dlink_post_cflags = sanitize_flags(cuda_dlink_post_cflags)
2825     sycl_cflags = sanitize_flags(sycl_cflags)
2826     sycl_post_cflags = sanitize_flags(sycl_post_cflags)
2827     sycl_dlink_post_cflags = sanitize_flags(sycl_dlink_post_cflags)
2828     ldflags = sanitize_flags(ldflags)
2829 
2830     # Sanity checks...
2831     assert len(sources) == len(objects)
2832     assert len(sources) > 0
2833 
2834     compiler = get_cxx_compiler()
2835 
2836     # Version 1.3 is required for the `deps` directive.
2837     config = ['ninja_required_version = 1.3']
2838     config.append(f'cxx = {compiler}')
2839     if with_cuda or cuda_dlink_post_cflags:
2840         if "PYTORCH_NVCC" in os.environ:
2841             nvcc = os.getenv("PYTORCH_NVCC")    # user can set nvcc compiler with ccache using the environment variable here
2842         else:
2843             if IS_HIP_EXTENSION:
2844                 nvcc = _get_hipcc_path()
2845             else:
2846                 nvcc = _join_cuda_home('bin', 'nvcc')
2847         config.append(f'nvcc = {nvcc}')
2848     if with_sycl or sycl_dlink_post_cflags:
2849         sycl = 'icx' if IS_WINDOWS else 'icpx'
2850         config.append(f'sycl = {sycl}')
2851 
2852     if IS_HIP_EXTENSION:
2853         post_cflags = COMMON_HIP_FLAGS + post_cflags
2854     flags = [f'cflags = {" ".join(cflags)}']
2855     flags.append(f'post_cflags = {" ".join(post_cflags)}')
2856     if with_cuda:
2857         flags.append(f'cuda_cflags = {" ".join(cuda_cflags)}')
2858         flags.append(f'cuda_post_cflags = {" ".join(cuda_post_cflags)}')
2859     flags.append(f'cuda_dlink_post_cflags = {" ".join(cuda_dlink_post_cflags)}')
2860     if with_sycl:
2861         flags.append(f'sycl_cflags = {" ".join(sycl_cflags)}')
2862         flags.append(f'sycl_post_cflags = {" ".join(sycl_post_cflags)}')
2863     flags.append(f'sycl_dlink_post_cflags = {" ".join(sycl_dlink_post_cflags)}')
2864     flags.append(f'ldflags = {" ".join(ldflags)}')
2865 
2866     # Turn into absolute paths so we can emit them into the ninja build
2867     # file wherever it is.
2868     sources = [os.path.abspath(file) for file in sources]
2869 
2870     # See https://ninja-build.org/build.ninja.html for reference.
2871     compile_rule = ['rule compile']
2872     if IS_WINDOWS:
2873         compiler_name = "$cxx" if IS_HIP_EXTENSION else "cl"
2874         compile_rule.append(
2875             f'  command = {compiler_name} /showIncludes $cflags -c $in /Fo$out $post_cflags')
2876         if not IS_HIP_EXTENSION:
2877             compile_rule.append('  deps = msvc')
2878     else:
2879         compile_rule.append(
2880             '  command = $cxx -MMD -MF $out.d $cflags -c $in -o $out $post_cflags')
2881         compile_rule.append('  depfile = $out.d')
2882         compile_rule.append('  deps = gcc')
2883 
2884     if with_cuda:
2885         cuda_compile_rule = ['rule cuda_compile']
2886         nvcc_gendeps = ''
2887         # --generate-dependencies-with-compile is not supported by ROCm
2888         # Nvcc flag `--generate-dependencies-with-compile` is not supported by sccache, which may increase build time.
2889         if torch.version.cuda is not None and os.getenv('TORCH_EXTENSION_SKIP_NVCC_GEN_DEPENDENCIES', '0') != '1':
2890             cuda_compile_rule.append('  depfile = $out.d')
2891             cuda_compile_rule.append('  deps = gcc')
2892             # Note: non-system deps with nvcc are only supported
2893             # on Linux so use --generate-dependencies-with-compile
2894             # to make this work on Windows too.
2895             nvcc_gendeps = '--generate-dependencies-with-compile --dependency-output $out.d'
2896         cuda_compile_rule.append(
2897             f'  command = $nvcc {nvcc_gendeps} $cuda_cflags -c $in -o $out $cuda_post_cflags')
2898 
2899     if with_sycl:
2900         sycl_compile_rule = ['rule sycl_compile']
2901         # SYCL compiler does not recognize .sycl extension automatically,
2902         # so we pass '-x c++' explicitly notifying compiler of file format
2903         sycl_compile_rule.append(
2904             '  command = $sycl $sycl_cflags -c -x c++ $in -o $out $sycl_post_cflags')
2905 
2906 
2907     # Emit one build rule per source to enable incremental build.
2908     build = []
2909     for source_file, object_file in zip(sources, objects):
2910         is_cuda_source = _is_cuda_file(source_file) and with_cuda
2911         is_sycl_source = _is_sycl_file(source_file) and with_sycl
2912         if is_cuda_source:
2913             rule = 'cuda_compile'
2914         elif is_sycl_source:
2915             rule = 'sycl_compile'
2916         else:
2917             rule = 'compile'
2918         if IS_WINDOWS:
2919             source_file = source_file.replace(':', '$:')
2920             object_file = object_file.replace(':', '$:')
2921         source_file = source_file.replace(" ", "$ ")
2922         object_file = object_file.replace(" ", "$ ")
2923         build.append(f'build {object_file}: {rule} {source_file}')
2924 
2925     if cuda_dlink_post_cflags:
2926         cuda_devlink_out = os.path.join(os.path.dirname(objects[0]), 'dlink.o')
2927         cuda_devlink_rule = ['rule cuda_devlink']
2928         cuda_devlink_rule.append('  command = $nvcc $in -o $out $cuda_dlink_post_cflags')
2929         cuda_devlink = [f'build {cuda_devlink_out}: cuda_devlink {" ".join(objects)}']
2930         objects += [cuda_devlink_out]
2931     else:
2932         cuda_devlink_rule, cuda_devlink = [], []
2933 
2934     if sycl_dlink_post_cflags:
2935         sycl_devlink_out = os.path.join(os.path.dirname(objects[0]), 'sycl_dlink.o')
2936         sycl_devlink_rule = ['rule sycl_devlink']
2937         sycl_devlink_rule.append('  command = $sycl $in -o $out $sycl_dlink_post_cflags')
2938         sycl_devlink = [f'build {sycl_devlink_out}: sycl_devlink {" ".join(objects)}']
2939         objects += [sycl_devlink_out]
2940     else:
2941         sycl_devlink_rule, sycl_devlink = [], []
2942 
2943     if library_target is not None:
2944         link_rule = ['rule link']
2945         if IS_WINDOWS:
2946             cl_paths = subprocess.check_output(['where',
2947                                                 'cl']).decode(*SUBPROCESS_DECODE_ARGS).split('\r\n')
2948             if len(cl_paths) >= 1:
2949                 cl_path = os.path.dirname(cl_paths[0]).replace(':', '$:')
2950             else:
2951                 raise RuntimeError("MSVC is required to load C++ extensions")
2952             link_rule.append(f'  command = "{cl_path}/link.exe" $in /nologo $ldflags /out:$out')
2953         else:
2954             link_rule.append('  command = $cxx $in $ldflags -o $out')
2955 
2956         link = [f'build {library_target}: link {" ".join(objects)}']
2957 
2958         default = [f'default {library_target}']
2959     else:
2960         link_rule, link, default = [], [], []
2961 
2962     # 'Blocks' should be separated by newlines, for visual benefit.
2963     blocks = [config, flags, compile_rule]
2964     if with_cuda:
2965         blocks.append(cuda_compile_rule)  # type: ignore[possibly-undefined]
2966     if with_sycl:
2967         blocks.append(sycl_compile_rule)  # type: ignore[possibly-undefined]
