9    inline std::string shapes_as_str(TensorList tensors) {
10     std::ostringstream os;
11     bool first = true;
12     for (auto& tensor : tensors) {
13       if (tensor.defined()) {
14         if (!first) {
15           os << ", ";
16         }
17         os << tensor.sizes();
18         first = false;
19       }
20     }
21     return os.str();
22   }

26   inline std::tuple<bool, Tensor> canDispatchToMaskedFill(
27       const Tensor& self,
28       const torch::List<std::optional<at::Tensor>>& indices,
29       const Tensor& value) {
30     if (!(value.numel() == 1 && value.device().is_cpu())) {
31       return std::make_tuple(false, Tensor());
32     }
33     int64_t num_ind = 0;
34     Tensor mask;
35     auto self_device = self.device();
36     for (const std::optional<Tensor>& i : indices) {
37       if (!i.has_value() || !(*i).defined()) {
38         num_ind++;
39       } else {
40         const Tensor& index = *i;
41         if ((index.scalar_type() != kByte && index.scalar_type() != kBool) ||
42             index.device() != self_device || mask.defined()) {
43           return std::make_tuple(false, Tensor());
44         } else {
45           mask = index;
46           for (const auto j : c10::irange(index.dim())) {
47             int64_t srcIdx = num_ind + j;
48             TORCH_CHECK_INDEX(
49                 index.size(j) == self.size(srcIdx),
50                 "The shape of the mask ",
51                 index.sizes(),
52                 " at index ",
53                 j,
54                 " does not match the shape of the indexed tensor ",
55                 self.sizes(),
56                 " at index ",
57                 srcIdx);
58           }
59           num_ind += mask.ndimension();
60         }
61       }
62     }
63     for ([[maybe_unused]] const auto i :
64          c10::irange(num_ind, self.ndimension())) {
65       mask = mask.unsqueeze(-1);
66     }
67     return std::make_tuple(true, mask);
68   }

70   inline AdvancedIndex make_info(Tensor self, IOptTensorListRef orig) {
71     checkIndexTensorTypes(orig, /*allow_int*/ true);
72     // first expand BoolTensor (masks) or ByteTensor (masks) into 1 or more
73     // LongTensors
74     auto indices = expandTensors(self, orig);
75     // next broadcast all index tensors together
76     try {
77       indices = expand_outplace(indices);
78     } catch (std::exception& e) {
79       TORCH_CHECK_INDEX(
80           false,
81           "shape mismatch: indexing tensors could not be broadcast together"
82           " with shapes ",
83           shapes_as_str(indices));
84     }
85     // add missing null Tensors so that it matches self.dim()
86     while (indices.size() < (size_t)self.dim()) {
87       indices.emplace_back();
88     }
89     // if the non-null indices are not all adjacent, transpose self and indices
90     // together so that they're adjacent at the front
91     if (!hasContiguousSubspace(indices)) {
92       std::tie(self, indices) = transposeToFront(self, indices);
93     }
94     // Ensure indices are on the same device as self
95     for (auto& indice : indices) {
96       if (indice.defined() && indice.device() != self.device()) {
97         indice = indice.to(self.device());
98       }
99     }
100    for (auto& indice : indices) {
101      if (indice.defined() && indice.dtype() == at::kInt) {
102        indice = indice.to(at::kLong);
103      }
104    }
105  
106    return AdvancedIndex(self, indices);
107  }
