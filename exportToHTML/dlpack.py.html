30   torch._C._add_docstr(to_dlpack, r"""to_dlpack(tensor) -> PyCapsule 
31    
32   Returns an opaque object (a "DLPack capsule") representing the tensor. 
33    
34   .. note:: 
35     ``to_dlpack`` is a legacy DLPack interface. The capsule it returns 
36     cannot be used for anything in Python other than use it as input to 
37     ``from_dlpack``. The more idiomatic use of DLPack is to call 
38     ``from_dlpack`` directly on the tensor object - this works when that 
39     object has a ``__dlpack__`` method, which PyTorch and most other 
40     libraries indeed have now. 
41    
42   .. warning:: 
43     Only call ``from_dlpack`` once per capsule produced with ``to_dlpack``. 
44     Behavior when a capsule is consumed multiple times is undefined. 
45    
46   Args: 
47       tensor: a tensor to be exported 
48    
49   The DLPack capsule shares the tensor's memory. 
50   """)

55   def from_dlpack(ext_tensor: Any) -> 'torch.Tensor':
56       """from_dlpack(ext_tensor) -> Tensor 
57    
58       Converts a tensor from an external library into a ``torch.Tensor``. 
59    
60       The returned PyTorch tensor will share the memory with the input tensor 
61       (which may have come from another library). Note that in-place operations 
62       will therefore also affect the data of the input tensor. This may lead to 
63       unexpected issues (e.g., other libraries may have read-only flags or 
64       immutable data structures), so the user should only do this if they know 
65       for sure that this is fine. 
66    
67       Args: 
68           ext_tensor (object with ``__dlpack__`` attribute, or a DLPack capsule): 
69               The tensor or DLPack capsule to convert. 
70    
71               If ``ext_tensor`` is a tensor (or ndarray) object, it must support 
72               the ``__dlpack__`` protocol (i.e., have a ``ext_tensor.__dlpack__`` 
73               method). Otherwise ``ext_tensor`` may be a DLPack capsule, which is 
74               an opaque ``PyCapsule`` instance, typically produced by a 
75               ``to_dlpack`` function or method. 
76    
77       Examples:: 
78    
79           >>> import torch.utils.dlpack 
80           >>> t = torch.arange(4) 
81    
82           # Convert a tensor directly (supported in PyTorch >= 1.10) 
83           >>> t2 = torch.from_dlpack(t) 
84           >>> t2[:2] = -1  # show that memory is shared 
85           >>> t2 
86           tensor([-1, -1,  2,  3]) 
87           >>> t 
88           tensor([-1, -1,  2,  3]) 
89    
90           # The old-style DLPack usage, with an intermediate capsule object 
91           >>> capsule = torch.utils.dlpack.to_dlpack(t) 
92           >>> capsule 
93           <capsule object "dltensor" at ...> 
94           >>> t3 = torch.from_dlpack(capsule) 
95           >>> t3 
96           tensor([-1, -1,  2,  3]) 
97           >>> t3[0] = -9  # now we're sharing memory between 3 tensors 
98           >>> t3 
99           tensor([-9, -1,  2,  3]) 
100          >>> t2 
101          tensor([-9, -1,  2,  3]) 
102          >>> t 
103          tensor([-9, -1,  2,  3]) 
104   
105      """
106      if hasattr(ext_tensor, '__dlpack__'):
107          device = ext_tensor.__dlpack_device__()
108          # device is either CUDA or ROCm, we need to pass the current
109          # stream
110          if device[0] in (DLDeviceType.kDLGPU, DLDeviceType.kDLROCM):
111              stream = torch.cuda.current_stream(f'cuda:{device[1]}')
112              # cuda_stream is the pointer to the stream and it is a public
113              # attribute, but it is not documented
114              # The array API specify that the default legacy stream must be passed
115              # with a value of 1 for CUDA
116              # https://data-apis.org/array-api/latest/API_specification/array_object.html?dlpack-self-stream-none#dlpack-self-stream-none
117              is_cuda = device[0] == DLDeviceType.kDLGPU
118              # Since pytorch is not using PTDS by default, lets directly pass
119              # the legacy stream
120              stream_ptr = 1 if is_cuda and stream.cuda_stream == 0 else stream.cuda_stream
121              dlpack = ext_tensor.__dlpack__(stream=stream_ptr)
122          else:
123              dlpack = ext_tensor.__dlpack__()
124      else:
125          # Old versions just call the converter
126          dlpack = ext_tensor
127      return _from_dlpack(dlpack)
