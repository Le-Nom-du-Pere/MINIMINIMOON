11   def _pre():
12       maxn = 10**5
13       global _factor, _totient
14       _factor = [0]*maxn
15       _totient = [1]*maxn
16       lim = int(maxn**0.5) + 5
17       for i in range(2, lim):
18           if _factor[i] == 0:
19               for j in range(i*i, maxn, i):
20                   if _factor[j] == 0:
21                       _factor[j] = i
22       for i in range(2, maxn):
23           if _factor[i] == 0:
24               _factor[i] = i
25               _totient[i] = i-1
26               continue
27           x = _factor[i]
28           y = i//x
29           if y % x == 0:
30               _totient[i] = _totient[y]*x
31           else:
32               _totient[i] = _totient[y]*(x - 1)

34   def _a(n, k, prec):
35       """ Compute the inner sum in HRR formula [1]_ 
36    
37       References 
38       ========== 
39    
40       .. [1] https://msp.org/pjm/1956/6-1/pjm-v6-n1-p18-p.pdf 
41    
42       """
43       if k == 1:
44           return fone
45   
46       k1 = k
47       e = 0
48       p = _factor[k]
49       while k1 % p == 0:
50           k1 //= p
51           e += 1
52       k2 = k//k1 # k2 = p^e
53       v = 1 - 24*n
54       pi = mpf_pi(prec)
55   
56       if k1 == 1:
57           # k  = p^e
58           if p == 2:
59               mod = 8*k
60               v = mod + v % mod
61               v = (v*pow(9, k - 1, mod)) % mod
62               m = _sqrt_mod_prime_power(v, 2, e + 3)[0]
63               arg = mpf_div(mpf_mul(
64                   from_int(4*m), pi, prec), from_int(mod), prec)
65               return mpf_mul(mpf_mul(
66                   from_int((-1)**e*(2 - (m % 4))),
67                   mpf_sqrt(from_int(k), prec), prec),
68                   mpf_sin(arg, prec), prec)
69           if p == 3:
70               mod = 3*k
71               v = mod + v % mod
72               if e > 1:
73                   v = (v*pow(64, k//3 - 1, mod)) % mod
74               m = _sqrt_mod_prime_power(v, 3, e + 1)[0]
75               arg = mpf_div(mpf_mul(from_int(4*m), pi, prec),
76                   from_int(mod), prec)
77               return mpf_mul(mpf_mul(
78                   from_int(2*(-1)**(e + 1)*(3 - 2*(m % 3))),
79                   mpf_sqrt(from_int(k//3), prec), prec),
80                   mpf_sin(arg, prec), prec)
81           v = k + v % k
82           jacobi3 = -1 if k % 12 in [5, 7] else 1
83           if v % p == 0:
84               if e == 1:
85                   return mpf_mul(
86                       from_int(jacobi3),
87                       mpf_sqrt(from_int(k), prec), prec)
88               return fzero
89           if not is_quad_residue(v, p):
90               return fzero
91           _phi = p**(e - 1)*(p - 1)
92           v = (v*pow(576, _phi - 1, k))
93           m = _sqrt_mod_prime_power(v, p, e)[0]
94           arg = mpf_div(
95               mpf_mul(from_int(4*m), pi, prec),
96               from_int(k), prec)
97           return mpf_mul(mpf_mul(
98               from_int(2*jacobi3),
99               mpf_sqrt(from_int(k), prec), prec),
100              mpf_cos(arg, prec), prec)
101  
102      if p != 2 or e >= 3:
103          d1, d2 = math.gcd(k1, 24), math.gcd(k2, 24)
104          e = 24//(d1*d2)
105          n1 = ((d2*e*n + (k2**2 - 1)//d1)*
106              pow(e*k2*k2*d2, _totient[k1] - 1, k1)) % k1
107          n2 = ((d1*e*n + (k1**2 - 1)//d2)*
108              pow(e*k1*k1*d1, _totient[k2] - 1, k2)) % k2
109          return mpf_mul(_a(n1, k1, prec), _a(n2, k2, prec), prec)
110      if e == 2:
111          n1 = ((8*n + 5)*pow(128, _totient[k1] - 1, k1)) % k1
112          n2 = (4 + ((n - 2 - (k1**2 - 1)//8)*(k1**2)) % 4) % 4
113          return mpf_mul(mpf_mul(
114              from_int(-1),
115              _a(n1, k1, prec), prec),
116              _a(n2, k2, prec))
117      n1 = ((8*n + 1)*pow(32, _totient[k1] - 1, k1)) % k1
118      n2 = (2 + (n - (k1**2 - 1)//8) % 2) % 2
119      return mpf_mul(_a(n1, k1, prec), _a(n2, k2, prec), prec)

121  def _d(n, j, prec, sq23pi, sqrt8):
122      """ 
123      Compute the sinh term in the outer sum of the HRR formula. 
124      The constants sqrt(2/3*pi) and sqrt(8) must be precomputed. 
125      """
126      j = from_int(j)
127      pi = mpf_pi(prec)
128      a = mpf_div(sq23pi, j, prec)
129      b = mpf_sub(from_int(n), from_rational(1, 24, prec), prec)
130      c = mpf_sqrt(b, prec)
131      ch, sh = mpf_cosh_sinh(mpf_mul(a, c), prec)
132      D = mpf_div(
133          mpf_sqrt(j, prec),
134          mpf_mul(mpf_mul(sqrt8, b), pi), prec)
135      E = mpf_sub(mpf_mul(a, ch), mpf_div(sh, c, prec), prec)
136      return mpf_mul(D, E)

139  @recurrence_memo([1, 1])
140  def _partition_rec(n: int, prev) -> int:
141      """ Calculate the partition function P(n) 
142   
143      Parameters 
144      ========== 
145   
146      n : int 
147          nonnegative integer 
148   
149      """
150      v = 0
151      penta = 0 # pentagonal number: 1, 5, 12, ...
152      for i in count():
153          penta += 3*i + 1
154          np = n - penta
155          if np < 0:
156              break
157          s = prev[np]
158          np -= i + 1
159          # np = n - gp where gp = generalized pentagonal: 2, 7, 15, ...
160          if 0 <= np:
161              s += prev[np]
162          v += -s if i % 2 else s
163      return v

166  def _partition(n: int) -> int:
167      """ Calculate the partition function P(n) 
168   
169      Parameters 
170      ========== 
171   
172      n : int 
173   
174      """
175      if n < 0:
176          return 0
177      if (n <= 200_000 and n - _partition_rec.cache_length() < 70 or
178              _partition_rec.cache_length() == 2 and n < 14_400):
179          # There will be 2*10**5 elements created here
180          # and n elements created by partition, so in case we
181          # are going to be working with small n, we just
182          # use partition to calculate (and cache) the values
183          # since lookup is used there while summation, using
184          # _factor and _totient, will be used below. But we
185          # only do so if n is relatively close to the length
186          # of the cache since doing 1 calculation here is about
187          # the same as adding 70 elements to the cache. In addition,
188          # the startup here costs about the same as calculating the first
189          # 14,400 values via partition, so we delay startup here unless n
190          # is smaller than that.
191          return _partition_rec(n)
192      if '_factor' not in globals():
193          _pre()
194      # Estimate number of bits in p(n). This formula could be tidied
195      pbits = int((
196          math.pi*(2*n/3.)**0.5 -
197          math.log(4*n))/math.log(10) + 1) * \
198          math.log2(10)
199      prec = p = int(pbits*1.1 + 100)
200  
201      # find the number of terms needed so rounded sum will be accurate
202      # using Rademacher's bound M(n, N) for the remainder after a partial
203      # sum of N terms (https://arxiv.org/pdf/1205.5991.pdf, (1.8))
204      c1 = 44*math.pi**2/(225*math.sqrt(3))
205      c2 = math.pi*math.sqrt(2)/75
206      c3 = math.pi*math.sqrt(2/3)
207      def _M(n, N):
208          sqrt = math.sqrt
209          return c1/sqrt(N) + c2*sqrt(N/(n - 1))*math.sinh(c3*sqrt(n)/N)
210      big = max(9, math.ceil(n**0.5))  # should be too large (for n > 65, ceil should work)
211      assert _M(n, big) < 0.5  # else double big until too large
212      while big > 40 and _M(n, big) < 0.5:
213          big //= 2
214      small = big
215      big = small*2
216      while big - small > 1:
217          N = (big + small)//2
218          if (er := _M(n, N)) < 0.5:
219              big = N
220          elif er >= 0.5:
221              small = N
222      M = big  # done with function M; now have value
223  
224      # sanity check for expected size of answer
225      if M > 10**5:  # i.e. M > maxn
226          raise ValueError("Input too big")  # i.e. n > 149832547102
227  
228      # calculate it
229      s = fzero
230      sq23pi = mpf_mul(mpf_sqrt(from_rational(2, 3, p), p), mpf_pi(p), p)
231      sqrt8 = mpf_sqrt(from_int(8), p)
232      for q in range(1, M):
233          a = _a(n, q, p)
234          d = _d(n, q, p, sq23pi, sqrt8)
235          s = mpf_add(s, mpf_mul(a, d), prec)
236          # On average, the terms decrease rapidly in magnitude.
237          # Dynamically reducing the precision greatly improves
238          # performance.
239          p = bitcount(abs(to_int(d))) + 50
240      return int(to_int(mpf_add(s, fhalf, prec)))
