11   // NOTE: don't create instances of this class directly.
12   // Use MPSEventPool to acquire instances of MPSEvent.
13   class MPSEvent {
14    public:
15     explicit MPSEvent(id_t ID, MPSStream* stream, bool enable_timing);
16     ~MPSEvent();
17   
18     // records an event on the stream
19     void record(bool needsLock, bool syncEvent = false);
20     // makes all future work submitted to the stream wait for this event.
21     bool wait(bool needsLock, bool syncEvent = false);
22     // schedules a notifyListener callback for the event.
23     bool notify(bool needsLock, MTLSharedEventNotificationBlock block);
24     // checks if events are already signaled.
25     bool query() const;
26     // blocks the CPU thread until all the GPU work that were scheduled
27     // prior to recording this event are completed.
28     bool synchronize();
29     // resets this event with new parameters in case it gets reused from the event
30     // pool
31     void reset(MPSStream* stream, bool enable_timing);
32     // returns the unique ID of the event instance
33     id_t getID() const {
34       return m_id;
35     }
36     // returns the completion timestamp of the event
37     uint64_t getCompletionTime() const {
38       return m_completion_time;
39     }
40     // if already recorded, waits for cpu_sync_cv to be signaled
41     void waitForCpuSync();
42   
43    private:
44     id_t m_id;
45     // enables measuring the completion time of the notifyListener of this event
46     bool m_enable_timing;
47     uint64_t m_signalCounter = 0;
48     MPSStream* m_stream = nullptr;
49     MTLSharedEvent_t m_event = nullptr;
50     MTLSharedEventListener* m_listener = nullptr;
51     // used to sync the events created on this Stream with CPU
52     std::mutex m_cpu_sync_mutex{};
53     std::condition_variable m_cpu_sync_cv{};
54     // CondVar predicate to sync the events created on this Stream with CPU
55     bool m_cpu_sync_completed = false;
56     // used to compute elapsed time
57     uint64_t m_completion_time = 0;
58   
59     void recordLocked(bool syncEvent);
60     bool waitLocked(bool syncEvent);
61     bool notifyLocked(MTLSharedEventNotificationBlock block);
62     void notifyCpuSync();
63     static uint64_t getTime() {
64       return clock_gettime_nsec_np(CLOCK_MONOTONIC_RAW);
65     }
66   };
67   
68   typedef std::unique_ptr<MPSEvent, std::function<void(MPSEvent*)>> MPSEventPtr;
69   
70   class MPSEventPool {
71    public:
72     explicit MPSEventPool(MPSStream* default_stream);
73     ~MPSEventPool();
74   
75     MPSEventPtr acquireEvent(bool enable_timing, MPSStream* stream);
76     void emptyCache();
77   
78     // these are mainly used for MPSHooks and torch.mps.Event() bindings
79     id_t acquireEvent(bool enable_timing);
80     void releaseEvent(id_t event_id);
81     void recordEvent(id_t event_id, bool syncEvent);
82     void waitForEvent(id_t event_id, bool syncEvent);
83     void synchronizeEvent(id_t event_id);
84     bool queryEvent(id_t event_id);
85     // returns elapsed time between two recorded events in milliseconds
86     double elapsedTime(id_t start_event_id, id_t end_event_id);
87   
88    private:
89     MPSStream* m_default_stream = nullptr;
90     std::recursive_mutex m_mutex;
91     std::stack<std::unique_ptr<MPSEvent>> m_pool{};
92     // dictionary to associate event IDs with event objects
93     // used to retain in-use events out of the pool
94     // for torch.mps.Event() bindings.
95     std::unordered_map<id_t, MPSEventPtr> m_in_use_events{};
96     uint64_t m_event_counter = 0;
97     std::function<void(MPSEvent*)> m_default_deleter;
98   
99     MPSEvent* getInUseEvent(id_t event_id, bool locked = true);
100  };
