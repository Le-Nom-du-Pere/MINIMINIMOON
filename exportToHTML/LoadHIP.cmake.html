66   macro(find_package_and_print_version PACKAGE_NAME) 
67     find_package("${PACKAGE_NAME}" ${ARGN}) 
68     if(NOT ${PACKAGE_NAME}_FOUND) 
69       message("Optional package ${PACKAGE_NAME} not found") 
70     else() 
71       message("${PACKAGE_NAME} VERSION: ${${PACKAGE_NAME}_VERSION}") 
72       if(${PACKAGE_NAME}_INCLUDE_DIR) 
73         list(APPEND ROCM_INCLUDE_DIRS ${${PACKAGE_NAME}_INCLUDE_DIR}) 
74       endif() 
75     endif() 
76   endmacro()

125    # Below we use a RegEx to find ROCM version numbers.
126    # Note that CMake does not support \s for blank space. That is
127    # why in the regular expressions below we have a blank space in
128    # the square brackets.
129    # There are three steps:
130    # 1. Match regular expression
131    # 2. Strip the non-numerical part of the string
132    # 3. Strip leading and trailing spaces
133  
134    string(REGEX MATCH "${ROCM_LIB_NAME}_VERSION_MAJOR[ ]+[0-9]+" TEMP1 ${ROCM_HEADER_CONTENT}) 
135    string(REPLACE "${ROCM_LIB_NAME}_VERSION_MAJOR" "" TEMP2 ${TEMP1}) 
136    string(STRIP ${TEMP2} ROCM_VERSION_DEV_MAJOR) 
137    string(REGEX MATCH "${ROCM_LIB_NAME}_VERSION_MINOR[ ]+[0-9]+" TEMP1 ${ROCM_HEADER_CONTENT}) 
138    string(REPLACE "${ROCM_LIB_NAME}_VERSION_MINOR" "" TEMP2 ${TEMP1}) 
139    string(STRIP ${TEMP2} ROCM_VERSION_DEV_MINOR) 
140    string(REGEX MATCH "${ROCM_LIB_NAME}_VERSION_PATCH[ ]+[0-9]+" TEMP1 ${ROCM_HEADER_CONTENT}) 
141    string(REPLACE "${ROCM_LIB_NAME}_VERSION_PATCH" "" TEMP2 ${TEMP1}) 
142    string(STRIP ${TEMP2} ROCM_VERSION_DEV_PATCH)

144    # Create ROCM_VERSION_DEV_INT which is later used as a preprocessor macros

146    math(EXPR ROCM_VERSION_DEV_INT "(${ROCM_VERSION_DEV_MAJOR}*10000) + (${ROCM_VERSION_DEV_MINOR}*100) + ${ROCM_VERSION_DEV_PATCH}")
