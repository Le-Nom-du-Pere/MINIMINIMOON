5    from dag_validation import create_sample_causal_graph

7    def main():
8        # Test reproducibility across multiple runs

10       
11       plan_name = "reproducibility_verification_test"
12       
13       print(f"Testing reproducibility for plan: {plan_name}")
14       print("=" * 50)
15       
16       # Run the same test multiple times
17       results = []
18       for i in range(5):
19           result = validator.calculate_acyclicity_pvalue(plan_name, 200)
20           results.append(result)
21           print(f"Run {i+1}: seed={result.seed}, p-value={result.p_value:.4f}, acyclic_count={result.acyclic_count}")
22       
23       # Verify all results are identical
24       first_result = results[0]
25       all_identical = all(
26           r.seed == first_result.seed and 
27           r.p_value == first_result.p_value and 
28           r.acyclic_count == first_result.acyclic_count and
29           r.subgraph_sizes == first_result.subgraph_sizes
30           for r in results[1:]
31       )
32       
33       print(f"\nAll results identical: {all_identical}")
34       
35       # Test different plan names produce different results
36       print("\nTesting different plan names:")
37       plan_names = ["plan_a", "plan_b", "plan_c"]
38       different_results = []
39       
40       for plan in plan_names:
41           result = validator.calculate_acyclicity_pvalue(plan, 100)
42           different_results.append(result)
43           print(f"{plan}: seed={result.seed}, p-value={result.p_value:.4f}")
44       
45       # Verify seeds are different
46       seeds = [r.seed for r in different_results]
47       all_seeds_different = len(set(seeds)) == len(seeds)
48       
49       print(f"All seeds different: {all_seeds_different}")
50       
51       return all_identical and all_seeds_different
