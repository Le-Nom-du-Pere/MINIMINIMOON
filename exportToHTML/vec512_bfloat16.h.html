33   static inline void cvtbf16_fp32(const __m256i& a, __m512& o) {
34     o = _mm512_castsi512_ps(_mm512_slli_epi32(_mm512_cvtepu16_epi32(a), 16));

38     __m256i lo = _mm512_extracti32x8_epi32(a, 0);
39     __m256i hi = _mm512_extracti32x8_epi32(a, 1);
40     cvtbf16_fp32(lo, o1);
41     cvtbf16_fp32(hi, o2);
42   }

44   static inline __m256i cvtfp32_bf16(const __m512& src) {
45     __m512i value = _mm512_castps_si512(src);
46     __m512i nan = _mm512_set1_epi32(0xffff);
47     auto mask_value = _mm512_cmp_ps_mask(src, src, _CMP_ORD_Q);
48     __m512i ones = _mm512_set1_epi32(0x1);
49     __m512i vec_bias = _mm512_set1_epi32(0x7fff);
50     // uint32_t lsb = (input >> 16) & 1;
51     auto t_value = _mm512_and_si512(_mm512_srli_epi32(value, 16), ones);
52     // uint32_t rounding_bias = 0x7fff + lsb;
53     t_value = _mm512_add_epi32(t_value, vec_bias);
54     // input += rounding_bias;
55     t_value = _mm512_add_epi32(t_value, value);
56     // input = input >> 16;
57     t_value = _mm512_srli_epi32(t_value, 16);
58     // Check NaN before converting back to bf16
59     t_value = _mm512_mask_blend_epi32(mask_value, nan, t_value);
60     return _mm512_cvtusepi32_epi16(t_value);
61   }

63   static inline __m512i cvtfp32_bf16(const __m512& a, const __m512& b) {
64     __m512i lo = _mm512_castps_si512(a);
65     __m512i hi = _mm512_castps_si512(b);
66     __m512i nan = _mm512_set1_epi32(0xffff);
67     auto mask_lo = _mm512_cmp_ps_mask(a, a, _CMP_ORD_Q);
68     auto mask_hi = _mm512_cmp_ps_mask(b, b, _CMP_ORD_Q);
69     __m512i ones = _mm512_set1_epi32(0x1);
70     __m512i vec_bias = _mm512_set1_epi32(0x7fff);
71     // uint32_t lsb = (input >> 16) & 1;
72     auto t_lo = _mm512_and_si512(_mm512_srli_epi32(lo, 16), ones);
73     auto t_hi = _mm512_and_si512(_mm512_srli_epi32(hi, 16), ones);
74     // uint32_t rounding_bias = 0x7fff + lsb;
75     t_lo = _mm512_add_epi32(t_lo, vec_bias);
76     t_hi = _mm512_add_epi32(t_hi, vec_bias);
77     // input += rounding_bias;
78     t_lo = _mm512_add_epi32(t_lo, lo);
79     t_hi = _mm512_add_epi32(t_hi, hi);
80     // input = input >> 16;
81     t_lo = _mm512_srli_epi32(t_lo, 16);
82     t_hi = _mm512_srli_epi32(t_hi, 16);
83     // Check NaN before converting back to bf16
84     t_lo = _mm512_mask_blend_epi32(mask_lo, nan, t_lo);
85     t_hi = _mm512_mask_blend_epi32(mask_hi, nan, t_hi);
86   
87     t_lo = _mm512_packus_epi32(
88         t_lo, t_hi); // t_hi[4-7] t_lo[4-7] t_hi[0-4] t_lo[0-4]
89     __m512i idx = _mm512_set_epi64(7, 5, 3, 1, 6, 4, 2, 0);
90     return _mm512_permutexvar_epi64(idx, t_lo);
91   }

93   static inline __m512i merge_compare_result(const __m512& a, const __m512& b) {
94     __m512i lo = _mm512_castps_si512(a);
95     __m512i hi = _mm512_castps_si512(b);
96     lo = _mm512_srli_epi32(lo, 16);
97     hi = _mm512_srli_epi32(hi, 16);
98     auto out = _mm512_packus_epi32(lo, hi);
99     __m512i idx = _mm512_set_epi64(7, 5, 3, 1, 6, 4, 2, 0);
100    return _mm512_permutexvar_epi64(idx, out);
101  }

119  static inline __m512i cvtfp32_fp16(const __m512& a, const __m512& b) {
120    __m256i lo =
121        _mm512_cvtps_ph(a, (_MM_FROUND_TO_NEAREST_INT | _MM_FROUND_NO_EXC));
122    __m256i hi =
123        _mm512_cvtps_ph(b, (_MM_FROUND_TO_NEAREST_INT | _MM_FROUND_NO_EXC));
124    __m512 t_lo = _mm512_castsi512_ps(_mm512_castsi256_si512(lo));
125    __m256 t_hi = _mm256_castsi256_ps(hi);
126    return _mm512_castps_si512(_mm512_insertf32x8(t_lo, t_hi, 1));
127  }

425    Vectorized<T> isnan() const {
426      __m512 lo, hi;
427      cvt_to_fp32<T>(values, lo, hi);
428      __mmask16 lo_mask, hi_mask;
429      __m512 zero = _mm512_set1_ps(0.0);

431      lo_mask = _mm512_cmp_ps_mask(lo, zero, _CMP_UNORD_Q);

433          _mm512_mask_set1_epi32(zeroi, lo_mask, 0xFFFF'FFFF));

436          _mm512_mask_set1_epi32(zeroi, hi_mask, 0xFFFF'FFFF));

443    Vectorized<T> angle() const {
444      __m512 lo, hi;
445      cvt_to_fp32<T>(values, lo, hi);
446      auto angle_lambda = [](__m512 values) {
447        const auto zero_vec = _mm512_set1_ps(0.f);
448        const auto nan_vec = _mm512_set1_ps(NAN);
449        const auto not_nan_mask = _mm512_cmp_ps_mask(values, values, _CMP_EQ_OQ);
450        const auto non_nan_mask_vec = _mm512_mask_set1_epi32(
451            _mm512_castps_si512(zero_vec), not_nan_mask, 0xFFFFFFFF);
452        const auto nan_mask = _mm512_cmp_ps_mask(
453            _mm512_castsi512_ps(non_nan_mask_vec), zero_vec, _CMP_EQ_OQ);
454        const auto pi = _mm512_set1_ps(c10::pi<float>);
455  
456        const auto neg_mask = _mm512_cmp_ps_mask(values, zero_vec, _CMP_LT_OQ);
457        auto angle = _mm512_mask_blend_ps(neg_mask, zero_vec, pi);
458        angle = _mm512_mask_blend_ps(nan_mask, angle, nan_vec);
459        return angle;
460      };
461      auto o1 = angle_lambda(lo);
462      auto o2 = angle_lambda(hi);
463      return cvt_from_fp32<T>(o1, o2);

1681 // Implements the IEEE 754 201X `maximum` operation, which propagates NaN if
1682 // either input is a NaN.
1683 template <>
1684 Vectorized<Half> inline maximum(
1685     const Vectorized<Half>& a,
1686     const Vectorized<Half>& b) {
1687   __m512 a_lo, a_hi;
1688   __m512 b_lo, b_hi;
1689   cvtfp16_fp32(__m512i(a), a_lo, a_hi);
1690   cvtfp16_fp32(__m512i(b), b_lo, b_hi);
1691   auto max_lo = _mm512_max_ps(a_lo, b_lo);
1692   auto max_hi = _mm512_max_ps(a_hi, b_hi);
1693   auto nan_lo_mask = _mm512_cmp_ps_mask(a_lo, b_lo, _CMP_UNORD_Q);
1694   auto nan_hi_mask = _mm512_cmp_ps_mask(a_hi, b_hi, _CMP_UNORD_Q);
1695   auto nan_lo = _mm512_castsi512_ps(_mm512_set1_epi32(nan_lo_mask));
1696   auto nan_hi = _mm512_castsi512_ps(_mm512_set1_epi32(nan_hi_mask));
1697   // Exploit the fact that all-ones is a NaN.
1698   auto o1 = _mm512_or_ps(max_lo, nan_lo);
1699   auto o2 = _mm512_or_ps(max_hi, nan_hi);
1700   return cvtfp32_fp16(o1, o2);
