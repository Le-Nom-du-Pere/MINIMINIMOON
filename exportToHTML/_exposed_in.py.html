1    from typing import Callable, TypeVar
2    
3    
4    F = TypeVar("F")
5    
6    
7    # Allows one to expose an API in a private submodule publicly as per the definition
8    # in PyTorch's public api policy.
9    #
10   # It is a temporary solution while we figure out if it should be the long-term solution
11   # or if we should amend PyTorch's public api policy. The concern is that this approach
12   # may not be very robust because it's not clear what __module__ is used for.
13   # However, both numpy and jax overwrite the __module__ attribute of their APIs
14   # without problem, so it seems fine.
15   def exposed_in(module: str) -> Callable[[F], F]:
16       def wrapper(fn: F) -> F:
17           fn.__module__ = module
18           return fn
19   
20       return wrapper
