6    #pragma once
7    
8    #include <stddef.h>
9    
10   #ifdef __cplusplus
11   extern "C" {
12   #endif
13   
14   /// Gets the m step value.
15   /// The micro-kernel can process any M values. However, the starting M index to
16   /// be processed must be a multiple of m step.
17   ///
18   /// @param[in] mr The number of M rows to interleave on the same output row.
19   ///
20   /// @return the m step value
21   size_t kai_get_m_step_lhs_quant_pack_qsi8d32p_f32_neon(size_t mr);
22   
23   /// Gets the offset in bytes for the LHS matrix (not packed)
24   ///
25   /// This function should be called before passing the pointer to the LHS matrix to the micro-kernel.
26   ///
27   /// @param[in] m_idx      Row index in the LHS matrix (not packed).
28   /// @param[in] lhs_stride The number of bytes in in each row of the LHS matrix (not packed)
29   ///
30   /// @return the offset in bytes to the LHS matrix
31   size_t kai_get_lhs_offset_lhs_quant_pack_qsi8d32p_f32_neon(size_t m_idx, size_t lhs_stride);
32   
33   /// Gets the offset in bytes for the packed LHS matrix,
34   /// which contains the packed 8-bit quantized symmetric per-block (qsi8d32) values.
35   ///
36   /// This function should be called before passing the pointer to the packed LHS matrix to the micro-kernel.
37   ///
38   /// @param[in] m_idx Row index in the LHS matrix (not packed). It must be a multiple of mr.
39   /// @param[in] k     Total number of columns in the LHS matrix (not packed).
40   /// @param[in] bl    The block length.
41   /// @param[in] mr    The number of M rows to interleave on the same output row.
42   /// @param[in] kr    The number of columns loaded in the single inner most loop of the matmul micro-kernel.
43   /// @param[in] sr    The number of kr splits. It can be 1 (no splits) up to kr.
44   ///
45   /// @return the offset in bytes to the packed LHS matrix
46   size_t kai_get_lhs_packed_offset_lhs_quant_pack_qsi8d32p_f32_neon(
47       size_t m_idx, size_t k, size_t bl, size_t mr, size_t kr, size_t sr);
48   
49   /// Gets the size in bytes for the quantized and packed LHS matrix
50   ///
51   /// @param[in] m  Total number of rows in the LHS matrix (not packed).
52   /// @param[in] k  Total number of columns in the LHS matrix (not packed).
53   /// @param[in] bl The block length, which defines the number of K values stored in a single block. It must be a
54   /// multiple of 32.
55   /// @param[in] mr The number of M rows to interleave on the same output row.
56   /// @param[in] kr The number of columns loaded in the single inner most loop of the matmul micro-kernel.
57   /// @param[in] sr The number of kr splits. It can be 1 (no splits) up to kr.
58   ///
59   /// @return the packed LHS matrix size in bytes
60   size_t kai_get_lhs_packed_size_lhs_quant_pack_qsi8d32p_f32_neon(
61       size_t m, size_t k, size_t bl, size_t mr, size_t kr, size_t sr);
62   
63   /// Run the micro-kernel to quantize and pack the LHS matrix.
64   ///
65   /// @param[in]  m           The number of output rows written.
66   /// @param[in]  k           The number of channels. The common dimension of LHS & RHS. It must be multiple of 8.
67   /// @param[in]  bl          The block length, which defines the number of K values stored in a single block. It must be
68   /// a multiple of 32.
69   /// @param[in]  mr          The number of M rows to interleave on the same output row.
70   /// @param[in]  kr          The number of columns loaded in the single inner most loop of the matmul micro-kernel.
71   /// @param[in]  sr          The number of kr splits. It can be 1 (no splits) up to kr.
72   ///                         However, kr must be multiple of sr.
73   /// @param[in]  m_idx_start The starting M index.
74   /// @param[in]  lhs         LHS matrix.
75   /// @param[in]  lhs_stride  Stride in bytes between two rows of LHS.
76   /// @param[out] lhs_packed  The quantized and packed LHS matrix.
77   void kai_run_lhs_quant_pack_qsi8d32p_f32_neon(
78       size_t m, size_t k, size_t bl, size_t mr, size_t kr, size_t sr, size_t m_idx_start, const float* lhs,
79       size_t lhs_stride, void* lhs_packed);
80   
81   #ifdef __cplusplus
82   }
83   #endif
