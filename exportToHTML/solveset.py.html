73   def _masked(f, *atoms):
74       """Return ``f``, with all objects given by ``atoms`` replaced with 
75       Dummy symbols, ``d``, and the list of replacements, ``(d, e)``, 
76       where ``e`` is an object of type given by ``atoms`` in which 
77       any other instances of atoms have been recursively replaced with 
78       Dummy symbols, too. The tuples are ordered so that if they are 
79       applied in sequence, the origin ``f`` will be restored. 
80    
81       Examples 
82       ======== 
83    
84       >>> from sympy import cos 
85       >>> from sympy.abc import x 
86       >>> from sympy.solvers.solveset import _masked 
87    
88       >>> f = cos(cos(x) + 1) 
89       >>> f, reps = _masked(cos(1 + cos(x)), cos) 
90       >>> f 
91       _a1 
92       >>> reps 
93       [(_a1, cos(_a0 + 1)), (_a0, cos(x))] 
94       >>> for d, e in reps: 
95       ...     f = f.xreplace({d: e}) 
96       >>> f 
97       cos(cos(x) + 1) 
98       """
99       sym = numbered_symbols('a', cls=Dummy, real=True)
100      mask = []
101      for a in ordered(f.atoms(*atoms)):
102          for i in mask:
103              a = a.replace(*i)
104          mask.append((a, next(sym)))
105      for i, (o, n) in enumerate(mask):
106          f = f.replace(o, n)
107          mask[i] = (n, o)
108      mask = list(reversed(mask))
109      return f, mask

112  def _invert(f_x, y, x, domain=S.Complexes):
113      r""" 
114      Reduce the complex valued equation $f(x) = y$ to a set of equations 
115   
116      $$\left\{g(x) = h_1(y),\  g(x) = h_2(y),\ \dots,\  g(x) = h_n(y) \right\}$$ 
117   
118      where $g(x)$ is a simpler function than $f(x)$.  The return value is a tuple 
119      $(g(x), \mathrm{set}_h)$, where $g(x)$ is a function of $x$ and $\mathrm{set}_h$ is 
120      the set of function $\left\{h_1(y), h_2(y), \dots, h_n(y)\right\}$. 
121      Here, $y$ is not necessarily a symbol. 
122   
123      $\mathrm{set}_h$ contains the functions, along with the information 
124      about the domain in which they are valid, through set 
125      operations. For instance, if :math:`y = |x| - n` is inverted 
126      in the real domain, then $\mathrm{set}_h$ is not simply 
127      $\{-n, n\}$ as the nature of `n` is unknown; rather, it is: 
128   
129      $$ \left(\left[0, \infty\right) \cap \left\{n\right\}\right) \cup 
130                         \left(\left(-\infty, 0\right] \cap \left\{- n\right\}\right)$$ 
131   
132      By default, the complex domain is used which means that inverting even 
133      seemingly simple functions like $\exp(x)$ will give very different 
134      results from those obtained in the real domain. 
135      (In the case of $\exp(x)$, the inversion via $\log$ is multi-valued 
136      in the complex domain, having infinitely many branches.) 
137   
138      If you are working with real values only (or you are not sure which 
139      function to use) you should probably set the domain to 
140      ``S.Reals`` (or use ``invert_real`` which does that automatically). 
141   
142   
143      Examples 
144      ======== 
145   
146      >>> from sympy.solvers.solveset import invert_complex, invert_real 
147      >>> from sympy.abc import x, y 
148      >>> from sympy import exp 
149   
150      When does exp(x) == y? 
151   
152      >>> invert_complex(exp(x), y, x) 
153      (x, ImageSet(Lambda(_n, I*(2*_n*pi + arg(y)) + log(Abs(y))), Integers)) 
154      >>> invert_real(exp(x), y, x) 
155      (x, Intersection({log(y)}, Reals)) 
156   
157      When does exp(x) == 1? 
158   
159      >>> invert_complex(exp(x), 1, x) 
160      (x, ImageSet(Lambda(_n, 2*_n*I*pi), Integers)) 
161      >>> invert_real(exp(x), 1, x) 
162      (x, {0}) 
163   
164      See Also 
165      ======== 
166      invert_real, invert_complex 
167      """
168      x = sympify(x)
169      if not x.is_Symbol:
170          raise ValueError("x must be a symbol")
171      f_x = sympify(f_x)
172      if x not in f_x.free_symbols:
173          raise ValueError("Inverse of constant function doesn't exist")
174      y = sympify(y)
175      if x in y.free_symbols:
176          raise ValueError("y should be independent of x ")
177  
178      if domain.is_subset(S.Reals):
179          x1, s = _invert_real(f_x, FiniteSet(y), x)
180      else:
181          x1, s = _invert_complex(f_x, FiniteSet(y), x)
182  
183      # f couldn't be inverted completely; return unmodified.
184      if  x1 != x:
185          return x1, s
186  
187      # Avoid adding gratuitous intersections with S.Complexes. Actual
188      # conditions should be handled by the respective inverters.
189      if domain is S.Complexes:
190          return x1, s
191  
192      if isinstance(s, FiniteSet):
193          return x1, s.intersect(domain)
194  
195      # "Fancier" solution sets like those obtained by inversion of trigonometric
196      # functions already include general validity conditions (i.e. conditions on
197      # the domain of the respective inverse functions), so we should avoid adding
198      # blanket intersections with S.Reals. But subsets of R (or C) must still be
199      # accounted for.
200      if domain is S.Reals:
201          return x1, s
202      else:
203          return x1, s.intersect(domain)

209  def invert_real(f_x, y, x):
210      """ 
211      Inverts a real-valued function. Same as :func:`invert_complex`, but sets 
212      the domain to ``S.Reals`` before inverting. 
213      """
214      return _invert(f_x, y, x, S.Reals)

217  def _invert_real(f, g_ys, symbol):
218      """Helper function for _invert."""
219  
220      if f == symbol or g_ys is S.EmptySet:
221          return (symbol, g_ys)
222  
223      n = Dummy('n', real=True)
224  
225      if isinstance(f, exp) or (f.is_Pow and f.base == S.Exp1):
226          return _invert_real(f.exp,
227                              imageset(Lambda(n, log(n)), g_ys),
228                              symbol)
229  
230      if hasattr(f, 'inverse') and f.inverse() is not None and not isinstance(f, (
231              TrigonometricFunction,
232              HyperbolicFunction,
233              )):
234          if len(f.args) > 1:
235              raise ValueError("Only functions with one argument are supported.")
236          return _invert_real(f.args[0],
237                              imageset(Lambda(n, f.inverse()(n)), g_ys),
238                              symbol)
239  
240      if isinstance(f, Abs):
241          return _invert_abs(f.args[0], g_ys, symbol)
242  
243      if f.is_Add:
244          # f = g + h
245          g, h = f.as_independent(symbol)
246          if g is not S.Zero:
247              return _invert_real(h, imageset(Lambda(n, n - g), g_ys), symbol)
248  
249      if f.is_Mul:
250          # f = g*h
251          g, h = f.as_independent(symbol)
252  
253          if g is not S.One:
254              return _invert_real(h, imageset(Lambda(n, n/g), g_ys), symbol)
255  
256      if f.is_Pow:
257          base, expo = f.args
258          base_has_sym = base.has(symbol)
259          expo_has_sym = expo.has(symbol)
260  
261          if not expo_has_sym:
262  
263              if expo.is_rational:
264                  num, den = expo.as_numer_denom()
265  
266                  if den % 2 == 0 and num % 2 == 1 and den.is_zero is False:
267                      # Here we have f(x)**(num/den) = y
268                      # where den is nonzero and even and y is an element
269                      # of the set g_ys.
270                      # den is even, so we are only interested in the cases
271                      # where both f(x) and y are positive.
272                      # Restricting y to be positive (using the set g_ys_pos)
273                      # means that y**(den/num) is always positive.
274                      # Therefore it isn't necessary to also constrain f(x)
275                      # to be positive because we are only going to
276                      # find solutions of f(x) = y**(d/n)
277                      # where the rhs is already required to be positive.
278                      root = Lambda(n, real_root(n, expo))
279                      g_ys_pos = g_ys & Interval(0, oo)
280                      res = imageset(root, g_ys_pos)
281                      _inv, _set = _invert_real(base, res, symbol)
282                      return (_inv, _set)
283  
284                  if den % 2 == 1:
285                      root = Lambda(n, real_root(n, expo))
286                      res = imageset(root, g_ys)
287                      if num % 2 == 0:
288                          neg_res = imageset(Lambda(n, -n), res)
289                          return _invert_real(base, res + neg_res, symbol)
290                      if num % 2 == 1:
291                          return _invert_real(base, res, symbol)
292  
293              elif expo.is_irrational:
294                  root = Lambda(n, real_root(n, expo))
295                  g_ys_pos = g_ys & Interval(0, oo)
296                  res = imageset(root, g_ys_pos)
297                  return _invert_real(base, res, symbol)
298  
299              else:
300                  # indeterminate exponent, e.g. Float or parity of
301                  # num, den of rational could not be determined
302                  pass  # use default return
303  
304          if not base_has_sym:
305              rhs = g_ys.args[0]
306              if base.is_positive:
307                  return _invert_real(expo,
308                      imageset(Lambda(n, log(n, base, evaluate=False)), g_ys), symbol)
309              elif base.is_negative:
310                  s, b = integer_log(rhs, base)
311                  if b:
312                      return _invert_real(expo, FiniteSet(s), symbol)
313                  else:
314                      return (expo, S.EmptySet)
315              elif base.is_zero:
316                  one = Eq(rhs, 1)
317                  if one == S.true:
318                      # special case: 0**x - 1
319                      return _invert_real(expo, FiniteSet(0), symbol)
320                  elif one == S.false:
321                      return (expo, S.EmptySet)
322  
323      if isinstance(f, (TrigonometricFunction, HyperbolicFunction)):
324          return _invert_trig_hyp_real(f, g_ys, symbol)
325  
326      return (f, g_ys)

333  def _invert_trig_hyp_real(f, g_ys, symbol):
334      """Helper function for inverting trigonometric and hyperbolic functions. 
335   
336      This helper only handles inversion over the reals. 
337   
338      For trigonometric functions only finite `g_ys` sets are implemented. 
339   
340      For hyperbolic functions the set `g_ys` is checked against the domain of the 
341      respective inverse functions. Infinite `g_ys` sets are also supported. 
342      """
343  
344      if isinstance(f, HyperbolicFunction):
345          n = Dummy('n', real=True)
346  
347          if isinstance(f, sinh):
348              # asinh is defined over R.
349              return _invert_real(f.args[0], imageset(n, asinh(n), g_ys), symbol)
350  
351          if isinstance(f, cosh):
352              g_ys_dom = g_ys.intersect(Interval(1, oo))
353              if isinstance(g_ys_dom, Intersection):
354                  # could not properly resolve domain check
355                  if isinstance(g_ys, FiniteSet):
356                      # If g_ys is a `FiniteSet`` it should be sufficient to just
357                      # let the calling `_invert_real()` add an intersection with
358                      # `S.Reals` (or a subset `domain`) to ensure that only valid
359                      # (real) solutions are returned.
360                      # This avoids adding "too many" Intersections or
361                      # ConditionSets in the returned set.
362                      g_ys_dom = g_ys
363                  else:
364                      return (f, g_ys)
365              return _invert_real(f.args[0], Union(
366                  imageset(n, acosh(n), g_ys_dom),
367                  imageset(n, -acosh(n), g_ys_dom)), symbol)
368  
369          if isinstance(f, sech):
370              g_ys_dom = g_ys.intersect(Interval.Lopen(0, 1))
371              if isinstance(g_ys_dom, Intersection):
372                  if isinstance(g_ys, FiniteSet):
373                      g_ys_dom = g_ys
374                  else:
375                      return (f, g_ys)
376              return _invert_real(f.args[0], Union(
377                  imageset(n, asech(n), g_ys_dom),
378                  imageset(n, -asech(n), g_ys_dom)), symbol)
379  
380          if isinstance(f, tanh):
381              g_ys_dom = g_ys.intersect(Interval.open(-1, 1))
382              if isinstance(g_ys_dom, Intersection):
383                  if isinstance(g_ys, FiniteSet):
384                      g_ys_dom = g_ys
385                  else:
386                      return (f, g_ys)
387              return _invert_real(f.args[0],
388                  imageset(n, atanh(n), g_ys_dom), symbol)
389  
390          if isinstance(f, coth):
391              g_ys_dom = g_ys - Interval(-1, 1)
392              if isinstance(g_ys_dom, Complement):
393                  if isinstance(g_ys, FiniteSet):
394                      g_ys_dom = g_ys
395                  else:
396                      return (f, g_ys)
397              return _invert_real(f.args[0],
398                  imageset(n, acoth(n), g_ys_dom), symbol)
399  
400          if isinstance(f, csch):
401              g_ys_dom = g_ys - FiniteSet(0)
402              if isinstance(g_ys_dom, Complement):
403                  if isinstance(g_ys, FiniteSet):
404                      g_ys_dom = g_ys
405                  else:
406                      return (f, g_ys)
407              return _invert_real(f.args[0],
408                  imageset(n, acsch(n), g_ys_dom), symbol)
409  
410      elif isinstance(f, TrigonometricFunction) and isinstance(g_ys, FiniteSet):
411          def _get_trig_inverses(func):
412              global _trig_inverses
413              if _trig_inverses is None:
414                  _trig_inverses = {
415                      sin : ((asin, lambda y: pi-asin(y)), 2*pi, Interval(-1, 1)),
416                      cos : ((acos, lambda y: -acos(y)), 2*pi, Interval(-1, 1)),
417                      tan : ((atan,), pi, S.Reals),
418                      cot : ((acot,), pi, S.Reals),
419                      sec : ((asec, lambda y: -asec(y)), 2*pi,
420                          Union(Interval(-oo, -1), Interval(1, oo))),
421                      csc : ((acsc, lambda y: pi-acsc(y)), 2*pi,
422                          Union(Interval(-oo, -1), Interval(1, oo)))}
423              return _trig_inverses[func]
424  
425          invs, period, rng = _get_trig_inverses(f.func)
426          n = Dummy('n', integer=True)
427          def create_return_set(g):
428              # returns ConditionSet that will be part of the final (x, set) tuple
429              invsimg = Union(*[
430                  imageset(n, period*n + inv(g), S.Integers) for inv in invs])
431              inv_f, inv_g_ys = _invert_real(f.args[0], invsimg, symbol)
432              if inv_f == symbol:     # inversion successful
433                  conds = rng.contains(g)
434                  return ConditionSet(symbol, conds, inv_g_ys)
435              else:
436                  return ConditionSet(symbol, Eq(f, g), S.Reals)
437  
438          retset = Union(*[create_return_set(g) for g in g_ys])
439          return (symbol, retset)
440  
441      else:
442          return (f, g_ys)

445  def _invert_trig_hyp_complex(f, g_ys, symbol):
446      """Helper function for inverting trigonometric and hyperbolic functions. 
447   
448      This helper only handles inversion over the complex numbers. 
449      Only finite `g_ys` sets are implemented. 
450   
451      Handling of singularities is only implemented for hyperbolic equations. 
452      In case of a symbolic element g in g_ys a ConditionSet may be returned. 
453      """
454  
455      if isinstance(f, TrigonometricFunction) and isinstance(g_ys, FiniteSet):
456          def inv(trig):
457              if isinstance(trig, (sin, csc)):
458                  F = asin if isinstance(trig, sin) else acsc
459                  return (
460                      lambda a: 2*n*pi + F(a),
461                      lambda a: 2*n*pi + pi - F(a))
462              if isinstance(trig, (cos, sec)):
463                  F = acos if isinstance(trig, cos) else asec
464                  return (
465                      lambda a: 2*n*pi + F(a),
466                      lambda a: 2*n*pi - F(a))
467              if isinstance(trig, (tan, cot)):
468                  return (lambda a: n*pi + trig.inverse()(a),)
469  
470          n = Dummy('n', integer=True)
471          invs = S.EmptySet
472          for L in inv(f):
473              invs += Union(*[imageset(Lambda(n, L(g)), S.Integers) for g in g_ys])
474          return _invert_complex(f.args[0], invs, symbol)
475  
476      elif isinstance(f, HyperbolicFunction) and isinstance(g_ys, FiniteSet):
477          # There are two main options regarding singularities / domain checking
478          # for symbolic elements in g_ys:
479          # 1. Add a "catch-all" intersection with S.Complexes.
480          # 2. ConditionSets.
481          # At present ConditionSets seem to work better and have the additional
482          # benefit of representing the precise conditions that must be satisfied.
483          # The conditions are also rather straightforward. (At most two isolated
484          # points.)
485          def _get_hyp_inverses(func):
486              global _hyp_inverses
487              if _hyp_inverses is None:
488                  _hyp_inverses = {
489                      sinh : ((asinh, lambda y: I*pi-asinh(y)), 2*I*pi, ()),
490                      cosh : ((acosh, lambda y: -acosh(y)), 2*I*pi, ()),
491                      tanh : ((atanh,), I*pi, (-1, 1)),
492                      coth : ((acoth,), I*pi, (-1, 1)),
493                      sech : ((asech, lambda y: -asech(y)), 2*I*pi, (0, )),
494                      csch : ((acsch, lambda y: I*pi-acsch(y)), 2*I*pi, (0, ))}
495              return _hyp_inverses[func]
496  
497          # invs: iterable of main inverses, e.g. (acosh, -acosh).
498          # excl: iterable of singularities to be checked for.
499          invs, period, excl = _get_hyp_inverses(f.func)
500          n = Dummy('n', integer=True)
501          def create_return_set(g):
502              # returns ConditionSet that will be part of the final (x, set) tuple
503              invsimg = Union(*[
504                  imageset(n, period*n + inv(g), S.Integers) for inv in invs])
505              inv_f, inv_g_ys = _invert_complex(f.args[0], invsimg, symbol)
506              if inv_f == symbol:     # inversion successful
507                  conds = And(*[Ne(g, e) for e in excl])
508                  return ConditionSet(symbol, conds, inv_g_ys)
509              else:
510                  return ConditionSet(symbol, Eq(f, g), S.Complexes)
511  
512          retset = Union(*[create_return_set(g) for g in g_ys])
513          return (symbol, retset)
514  
515      else:
516          return (f, g_ys)

519  def _invert_complex(f, g_ys, symbol):
520      """Helper function for _invert."""
521  
522      if f == symbol or g_ys is S.EmptySet:
523          return (symbol, g_ys)
524  
525      n = Dummy('n')
526  
527      if f.is_Add:
528          # f = g + h
529          g, h = f.as_independent(symbol)
530          if g is not S.Zero:
531              return _invert_complex(h, imageset(Lambda(n, n - g), g_ys), symbol)
532  
533      if f.is_Mul:
534          # f = g*h
535          g, h = f.as_independent(symbol)
536  
537          if g is not S.One:
538              if g in {S.NegativeInfinity, S.ComplexInfinity, S.Infinity}:
539                  return (h, S.EmptySet)
540              return _invert_complex(h, imageset(Lambda(n, n/g), g_ys), symbol)
541  
542      if f.is_Pow:
543          base, expo = f.args
544          # special case: g**r = 0
545          # Could be improved like `_invert_real` to handle more general cases.
546          if expo.is_Rational and g_ys == FiniteSet(0):
547              if expo.is_positive:
548                  return _invert_complex(base, g_ys, symbol)
549  
550      if hasattr(f, 'inverse') and f.inverse() is not None and \
551         not isinstance(f, TrigonometricFunction) and \
552         not isinstance(f, HyperbolicFunction) and \
553         not isinstance(f, exp):
554          if len(f.args) > 1:
555              raise ValueError("Only functions with one argument are supported.")
556          return _invert_complex(f.args[0],
557                                 imageset(Lambda(n, f.inverse()(n)), g_ys), symbol)
558  
559      if isinstance(f, exp) or (f.is_Pow and f.base == S.Exp1):
560          if isinstance(g_ys, ImageSet):
561              # can solve up to `(d*exp(exp(...(exp(a*x + b))...) + c)` format.
562              # Further can be improved to `(d*exp(exp(...(exp(a*x**n + b*x**(n-1) + ... + f))...) + c)`.
563              g_ys_expr = g_ys.lamda.expr
564              g_ys_vars = g_ys.lamda.variables
565              k = Dummy('k{}'.format(len(g_ys_vars)))
566              g_ys_vars_1 = (k,) + g_ys_vars
567              exp_invs = Union(*[imageset(Lambda((g_ys_vars_1,), (I*(2*k*pi + arg(g_ys_expr))
568                                           + log(Abs(g_ys_expr)))), S.Integers**(len(g_ys_vars_1)))])
569              return _invert_complex(f.exp, exp_invs, symbol)
570  
571          elif isinstance(g_ys, FiniteSet):
572              exp_invs = Union(*[imageset(Lambda(n, I*(2*n*pi + arg(g_y)) +
573                                                 log(Abs(g_y))), S.Integers)
574                                 for g_y in g_ys if g_y != 0])
575              return _invert_complex(f.exp, exp_invs, symbol)
576  
577      if isinstance(f, (TrigonometricFunction, HyperbolicFunction)):
578          return _invert_trig_hyp_complex(f, g_ys, symbol)
579  
580      return (f, g_ys)

583  def _invert_abs(f, g_ys, symbol):
584      """Helper function for inverting absolute value functions. 
585   
586      Returns the complete result of inverting an absolute value 
587      function along with the conditions which must also be satisfied. 
588   
589      If it is certain that all these conditions are met, a :class:`~.FiniteSet` 
590      of all possible solutions is returned. If any condition cannot be 
591      satisfied, an :class:`~.EmptySet` is returned. Otherwise, a 
592      :class:`~.ConditionSet` of the solutions, with all the required conditions 
593      specified, is returned. 
594   
595      """
596      if not g_ys.is_FiniteSet:
597          # this could be used for FiniteSet, but the
598          # results are more compact if they aren't, e.g.
599          # ConditionSet(x, Contains(n, Interval(0, oo)), {-n, n}) vs
600          # Union(Intersection(Interval(0, oo), {n}), Intersection(Interval(-oo, 0), {-n}))
601          # for the solution of abs(x) - n
602          pos = Intersection(g_ys, Interval(0, S.Infinity))
603          parg = _invert_real(f, pos, symbol)
604          narg = _invert_real(-f, pos, symbol)
605          if parg[0] != narg[0]:
606              raise NotImplementedError
607          return parg[0], Union(narg[1], parg[1])
608  
609      # check conditions: all these must be true. If any are unknown
610      # then return them as conditions which must be satisfied
611      unknown = []
612      for a in g_ys.args:
613          ok = a.is_nonnegative if a.is_Number else a.is_positive
614          if ok is None:
615              unknown.append(a)
616          elif not ok:
617              return symbol, S.EmptySet
618      if unknown:
619          conditions = And(*[Contains(i, Interval(0, oo))
620              for i in unknown])
621      else:
622          conditions = True
623      n = Dummy('n', real=True)
624      # this is slightly different than above: instead of solving
625      # +/-f on positive values, here we solve for f on +/- g_ys
626      g_x, values = _invert_real(f, Union(
627          imageset(Lambda(n, n), g_ys),
628          imageset(Lambda(n, -n), g_ys)), symbol)
629      return g_x, ConditionSet(g_x, conditions, values)

632  def domain_check(f, symbol, p):
633      """Returns False if point p is infinite or any subexpression of f 
634      is infinite or becomes so after replacing symbol with p. If none of 
635      these conditions is met then True will be returned. 
636   
637      Examples 
638      ======== 
639   
640      >>> from sympy import Mul, oo 
641      >>> from sympy.abc import x 
642      >>> from sympy.solvers.solveset import domain_check 
643      >>> g = 1/(1 + (1/(x + 1))**2) 
644      >>> domain_check(g, x, -1) 
645      False 
646      >>> domain_check(x**2, x, 0) 
647      True 
648      >>> domain_check(1/x, x, oo) 
649      False 
650   
651      * The function relies on the assumption that the original form 
652        of the equation has not been changed by automatic simplification. 
653   
654      >>> domain_check(x/x, x, 0) # x/x is automatically simplified to 1 
655      True 
656   
657      * To deal with automatic evaluations use evaluate=False: 
658   
659      >>> domain_check(Mul(x, 1/x, evaluate=False), x, 0) 
660      False 
661      """
662      f, p = sympify(f), sympify(p)
663      if p.is_infinite:
664          return False
665      return _domain_check(f, symbol, p)

668  def _domain_check(f, symbol, p):
669      # helper for domain check
670      if f.is_Atom and f.is_finite:
671          return True
672      elif f.subs(symbol, p).is_infinite:
673          return False
674      elif isinstance(f, Piecewise):
675          # Check the cases of the Piecewise in turn. There might be invalid
676          # expressions in later cases that don't apply e.g.
677          #    solveset(Piecewise((0, Eq(x, 0)), (1/x, True)), x)
678          for expr, cond in f.args:
679              condsubs = cond.subs(symbol, p)
680              if condsubs is S.false:
681                  continue
682              elif condsubs is S.true:
683                  return _domain_check(expr, symbol, p)
684              else:
685                  # We don't know which case of the Piecewise holds. On this
686                  # basis we cannot decide whether any solution is in or out of
687                  # the domain. Ideally this function would allow returning a
688                  # symbolic condition for the validity of the solution that
689                  # could be handled in the calling code. In the mean time we'll
690                  # give this particular solution the benefit of the doubt and
691                  # let it pass.
692                  return True
693      else:
694          # TODO : We should not blindly recurse through all args of arbitrary expressions like this
695          return all(_domain_check(g, symbol, p)
696                     for g in f.args)

700      """ 
701      Return True if the given expression is finite. For symbols that 
702      do not assign a value for `complex` and/or `real`, the domain will 
703      be used to assign a value; symbols that do not assign a value 
704      for `finite` will be made finite. All other assumptions are

709          A.setdefault('finite', A.get('finite', True))
710          if domain.is_subset(S.Reals):

712              A.setdefault('real', True)
713          else:
714              # don't change 'real' because being complex implies

716              A.setdefault('complex', True)

723  def _is_function_class_equation(func_class, f, symbol):
724      """ Tests whether the equation is an equation of the given function class. 
725   
726      The given equation belongs to the given function class if it is 
727      comprised of functions of the function class which are multiplied by 
728      or added to expressions independent of the symbol. In addition, the 
729      arguments of all such functions must be linear in the symbol as well. 
730   
731      Examples 
732      ======== 
733   
734      >>> from sympy.solvers.solveset import _is_function_class_equation 
735      >>> from sympy import tan, sin, tanh, sinh, exp 
736      >>> from sympy.abc import x 
737      >>> from sympy.functions.elementary.trigonometric import TrigonometricFunction 
738      >>> from sympy.functions.elementary.hyperbolic import HyperbolicFunction 
739      >>> _is_function_class_equation(TrigonometricFunction, exp(x) + tan(x), x) 
740      False 
741      >>> _is_function_class_equation(TrigonometricFunction, tan(x) + sin(x), x) 
742      True 
743      >>> _is_function_class_equation(TrigonometricFunction, tan(x**2), x) 
744      False 
745      >>> _is_function_class_equation(TrigonometricFunction, tan(x + 2), x) 
746      True 
747      >>> _is_function_class_equation(HyperbolicFunction, tanh(x) + sinh(x), x) 
748      True 
749      """
750      if f.is_Mul or f.is_Add:
751          return all(_is_function_class_equation(func_class, arg, symbol)
752                     for arg in f.args)
753  
754      if f.is_Pow:
755          if not f.exp.has(symbol):
756              return _is_function_class_equation(func_class, f.base, symbol)
757          else:
758              return False
759  
760      if not f.has(symbol):
761          return True
762  
763      if isinstance(f, func_class):
764          try:
765              g = Poly(f.args[0], symbol)
766              return g.degree() <= 1
767          except PolynomialError:
768              return False
769      else:
770          return False

773  def _solve_as_rational(f, symbol, domain):
774      """ solve rational functions"""
775      f = together(_mexpand(f, recursive=True), deep=True)
776      g, h = fraction(f)
777      if not h.has(symbol):
778          try:
779              return _solve_as_poly(g, symbol, domain)
780          except NotImplementedError:
781              # The polynomial formed from g could end up having
782              # coefficients in a ring over which finding roots
783              # isn't implemented yet, e.g. ZZ[a] for some symbol a
784              return ConditionSet(symbol, Eq(f, 0), domain)
785          except CoercionFailed:
786              # contained oo, zoo or nan
787              return S.EmptySet
788      else:
789          valid_solns = _solveset(g, symbol, domain)
790          invalid_solns = _solveset(h, symbol, domain)
791          return valid_solns - invalid_solns

797  def _solve_trig(f, symbol, domain):
798      """Function to call other helpers to solve trigonometric equations """
799      # If f is composed of a single trig function (potentially appearing multiple
800      # times) we should solve by either inverting directly or inverting after a
801      # suitable change of variable.
802      #
803      # _solve_trig is currently only called by _solveset for trig/hyperbolic
804      # functions of an argument linear in x. Inverting a symbolic argument should
805      # include a guard against division by zero in order to have a result that is
806      # consistent with similar processing done by _solve_trig1.
807      # (Ideally _invert should add these conditions by itself.)
808      trig_expr, count = None, 0
809      for expr in preorder_traversal(f):
810          if isinstance(expr, (TrigonometricFunction,
811                              HyperbolicFunction)) and expr.has(symbol):
812              if not trig_expr:
813                  trig_expr, count = expr, 1
814              elif expr == trig_expr:
815                  count += 1
816              else:
817                  trig_expr, count = False, 0
818                  break
819      if count == 1:
820          # direct inversion
821          x, sol = _invert(f, 0, symbol, domain)
822          if x == symbol:
823              cond = True
824              if trig_expr.free_symbols - {symbol}:
825                  a, h = trig_expr.args[0].as_independent(symbol, as_Add=True)
826                  m, h = h.as_independent(symbol, as_Add=False)
827                  num, den = m.as_numer_denom()
828                  cond = Ne(num, 0) & Ne(den, 0)
829              return ConditionSet(symbol, cond, sol)
830          else:
831              return ConditionSet(symbol, Eq(f, 0), domain)
832      elif count:
833          # solve by change of variable
834          y = Dummy('y')
835          f_cov = f.subs(trig_expr, y)
836          sol_cov = solveset(f_cov, y, domain)
837          if isinstance(sol_cov, FiniteSet):
838              return Union(
839                  *[_solve_trig(trig_expr-s, symbol, domain) for s in sol_cov])
840  
841      sol = None
842      try:
843          # multiple trig/hyp functions; solve by rewriting to exp
844          sol = _solve_trig1(f, symbol, domain)
845      except _SolveTrig1Error:
846          try:
847              # multiple trig/hyp functions; solve by rewriting to tan(x/2)
848              sol = _solve_trig2(f, symbol, domain)
849          except ValueError:
850              raise NotImplementedError(filldedent(''' 
851                  Solution to this kind of trigonometric equations 
852                  is yet to be implemented'''))
853      return sol

856  def _solve_trig1(f, symbol, domain):
857      """Primary solver for trigonometric and hyperbolic equations 
858   
859      Returns either the solution set as a ConditionSet (auto-evaluated to a 
860      union of ImageSets if no variables besides 'symbol' are involved) or 
861      raises _SolveTrig1Error if f == 0 cannot be solved. 
862   
863      Notes 
864      ===== 
865      Algorithm: 
866      1. Do a change of variable x -> mu*x in arguments to trigonometric and 
867      hyperbolic functions, in order to reduce them to small integers. (This 
868      step is crucial to keep the degrees of the polynomials of step 4 low.) 
869      2. Rewrite trigonometric/hyperbolic functions as exponentials. 
870      3. Proceed to a 2nd change of variable, replacing exp(I*x) or exp(x) by y. 
871      4. Solve the resulting rational equation. 
872      5. Use invert_complex or invert_real to return to the original variable. 
873      6. If the coefficients of 'symbol' were symbolic in nature, add the 
874      necessary consistency conditions in a ConditionSet. 
875   
876      """
877      # Prepare change of variable
878      x = Dummy('x')
879      if _is_function_class_equation(HyperbolicFunction, f, symbol):
880          cov = exp(x)
881          inverter = invert_real if domain.is_subset(S.Reals) else invert_complex
882      else:
883          cov = exp(I*x)
884          inverter = invert_complex
885  
886      f = trigsimp(f)
887      f_original = f
888      trig_functions = f.atoms(TrigonometricFunction, HyperbolicFunction)
889      trig_arguments = [e.args[0] for e in trig_functions]
890      # trigsimp may have reduced the equation to an expression
891      # that is independent of 'symbol' (e.g. cos**2+sin**2)
892      if not any(a.has(symbol) for a in trig_arguments):
893          return solveset(f_original, symbol, domain)
894  
895      denominators = []
896      numerators = []
897      for ar in trig_arguments:
898          try:
899              poly_ar = Poly(ar, symbol)
900          except PolynomialError:
901              raise _SolveTrig1Error("trig argument is not a polynomial")
902          if poly_ar.degree() > 1:  # degree >1 still bad
903              raise _SolveTrig1Error("degree of variable must not exceed one")
904          if poly_ar.degree() == 0:  # degree 0, don't care
905              continue
906          c = poly_ar.all_coeffs()[0]   # got the coefficient of 'symbol'
907          numerators.append(fraction(c)[0])
908          denominators.append(fraction(c)[1])
909  
910      mu = lcm(denominators)/gcd(numerators)
911      f = f.subs(symbol, mu*x)
912      f = f.rewrite(exp)
913      f = together(f)
914      g, h = fraction(f)
915      y = Dummy('y')
916      g, h = g.expand(), h.expand()
917      g, h = g.subs(cov, y), h.subs(cov, y)
918      if g.has(x) or h.has(x):
919          raise _SolveTrig1Error("change of variable not possible")
920  
921      solns = solveset_complex(g, y) - solveset_complex(h, y)
922      if isinstance(solns, ConditionSet):
923          raise _SolveTrig1Error("polynomial has ConditionSet solution")
924  
925      if isinstance(solns, FiniteSet):
926          if any(isinstance(s, RootOf) for s in solns):
927              raise _SolveTrig1Error("polynomial results in RootOf object")
928          # revert the change of variable
929          cov = cov.subs(x, symbol/mu)
930          result = Union(*[inverter(cov, s, symbol)[1] for s in solns])
931          # In case of symbolic coefficients, the solution set is only valid
932          # if numerator and denominator of mu are non-zero.
933          if mu.has(Symbol):
934              syms = (mu).atoms(Symbol)
935              munum, muden = fraction(mu)
936              condnum = munum.as_independent(*syms, as_Add=False)[1]
937              condden = muden.as_independent(*syms, as_Add=False)[1]
938              cond = And(Ne(condnum, 0), Ne(condden, 0))
939          else:
940              cond = True
941          # Actual conditions are returned as part of the ConditionSet. Adding an
942          # intersection with C would only complicate some solution sets due to
943          # current limitations of intersection code. (e.g. #19154)
944          if domain is S.Complexes:
945              # This is a slight abuse of ConditionSet. Ideally this should
946              # be some kind of "PiecewiseSet". (See #19507 discussion)
947              return ConditionSet(symbol, cond, result)
948          else:
949              return ConditionSet(symbol, cond, Intersection(result, domain))
950      elif solns is S.EmptySet:
951          return S.EmptySet
952      else:
953          raise _SolveTrig1Error("polynomial solutions must form FiniteSet")

964      numerators = []

975      for ar in trig_arguments:

982          if poly_ar.degree() == 0:  # degree 0, don't care

996      f = expand_trig(f)
997      f = together(f)

1001     g, h = g.expand(), h.expand()
1002     g, h = g.subs(tan(x), y), h.subs(tan(x), y)

1005         return ConditionSet(symbol, Eq(f_original, 0), domain)

1010                        for s in solns])

1013             result = Union(result, dsol)            # would be another sol at Lim(denom-->oo)
1014         return Intersection(result, domain)
1015     elif solns is S.EmptySet:
1016         return S.EmptySet
1017     else:
1018         return ConditionSet(symbol, Eq(f_original, 0), S.Reals)

1021 def _solve_as_poly(f, symbol, domain=S.Complexes):
1022     """ 
1023     Solve the equation using polynomial techniques if it already is a 
1024     polynomial equation or, with a change of variables, can be made so. 
1025     """
1026     result = None
1027     if f.is_polynomial(symbol):
1028         solns = roots(f, symbol, cubics=True, quartics=True,
1029                       quintics=True, domain='EX')
1030         num_roots = sum(solns.values())
1031         if degree(f, symbol) <= num_roots:
1032             result = FiniteSet(*solns.keys())
1033         else:
1034             poly = Poly(f, symbol)
1035             solns = poly.all_roots()
1036             if poly.degree() <= len(solns):
1037                 result = FiniteSet(*solns)
1038             else:
1039                 result = ConditionSet(symbol, Eq(f, 0), domain)
1040     else:
1041         poly = Poly(f)
1042         if poly is None:
1043             result = ConditionSet(symbol, Eq(f, 0), domain)
1044         gens = [g for g in poly.gens if g.has(symbol)]
1045 
1046         if len(gens) == 1:
1047             poly = Poly(poly, gens[0])
1048             gen = poly.gen
1049             deg = poly.degree()
1050             poly = Poly(poly.as_expr(), poly.gen, composite=True)
1051             poly_solns = FiniteSet(*roots(poly, cubics=True, quartics=True,
1052                                           quintics=True).keys())
1053 
1054             if len(poly_solns) < deg:
1055                 result = ConditionSet(symbol, Eq(f, 0), domain)
1056 
1057             if gen != symbol:
1058                 y = Dummy('y')
1059                 inverter = invert_real if domain.is_subset(S.Reals) else invert_complex
1060                 lhs, rhs_s = inverter(gen, y, symbol)
1061                 if lhs == symbol:
1062                     result = Union(*[rhs_s.subs(y, s) for s in poly_solns])
1063                     if isinstance(result, FiniteSet) and isinstance(gen, Pow
1064                             ) and gen.base.is_Rational:
1065                         result = FiniteSet(*[expand_log(i) for i in result])
1066                 else:
1067                     result = ConditionSet(symbol, Eq(f, 0), domain)
1068         else:
1069             result = ConditionSet(symbol, Eq(f, 0), domain)
1070 
1071     if result is not None:
1072         if isinstance(result, FiniteSet):
1073             # this is to simplify solutions like -sqrt(-I) to sqrt(2)/2
1074             # - sqrt(2)*I/2. We are not expanding for solution with symbols
1075             # or undefined functions because that makes the solution more complicated.
1076             # For example, expand_complex(a) returns re(a) + I*im(a)
1077             if all(s.atoms(Symbol, AppliedUndef) == set() and not isinstance(s, RootOf)
1078                    for s in result):
1079                 s = Dummy('s')
1080                 result = imageset(Lambda(s, expand_complex(s)), result)
1081         if isinstance(result, FiniteSet) and domain != S.Complexes:
1082             # Avoid adding gratuitous intersections with S.Complexes. Actual
1083             # conditions should be handled elsewhere.
1084             result = result.intersection(domain)
1085         return result
1086     else:
1087         return ConditionSet(symbol, Eq(f, 0), domain)

1091     """ Helper function to solve equations with radicals """
1092     res = unradf
1093     eq, cov = res if res else (f, [])
1094     if not cov:
1095         result = solveset_solver(eq, symbol) - \
1096             Union(*[solveset_solver(g, symbol) for g in denoms(f, symbol)])

1098         y, yeq = cov
1099         if not solveset_solver(y - I, y):
1100             yreal = Dummy('yreal', real=True)

1102             eq = eq.xreplace({y: yreal})
1103             y = yreal

1105         f_y_sols = solveset_solver(eq, y)

1107                          for g_y in g_y_s])

1121             return solutions
1122         elif isinstance(solutions, ConditionSet):
1123             # XXX: Maybe the base set should be checked?
1124             return solutions

1132         else:
1133             # XXX: There should be more cases checked here. The cases above

1142 def _solve_abs(f, symbol, domain):
1143     """ Helper function to solve equation involving absolute value function """
1144     if not domain.is_subset(S.Reals):
1145         raise ValueError(filldedent(''' 
1146             Absolute values cannot be inverted in the 
1147             complex domain.'''))
1148     p, q, r = Wild('p'), Wild('q'), Wild('r')
1149     pattern_match = f.match(p*Abs(q) + r) or {}
1150     f_p, f_q, f_r = [pattern_match.get(i, S.Zero) for i in (p, q, r)]
1151 
1152     if not (f_p.is_zero or f_q.is_zero):
1153         domain = continuous_domain(f_q, symbol, domain)
1154         from .inequalities import solve_univariate_inequality
1155         q_pos_cond = solve_univariate_inequality(f_q >= 0, symbol,
1156                                                  relational=False, domain=domain, continuous=True)
1157         q_neg_cond = q_pos_cond.complement(domain)
1158 
1159         sols_q_pos = solveset_real(f_p*f_q + f_r,
1160                                            symbol).intersect(q_pos_cond)
1161         sols_q_neg = solveset_real(f_p*(-f_q) + f_r,
1162                                            symbol).intersect(q_neg_cond)
1163         return Union(sols_q_pos, sols_q_neg)
1164     else:
1165         return ConditionSet(symbol, Eq(f, 0), domain)

1168 def solve_decomposition(f, symbol, domain):
1169     """ 
1170     Function to solve equations via the principle of "Decomposition 
1171     and Rewriting". 
1172  
1173     Examples 
1174     ======== 
1175     >>> from sympy import exp, sin, Symbol, pprint, S 
1176     >>> from sympy.solvers.solveset import solve_decomposition as sd 
1177     >>> x = Symbol('x') 
1178     >>> f1 = exp(2*x) - 3*exp(x) + 2 
1179     >>> sd(f1, x, S.Reals) 
1180     {0, log(2)} 
1181     >>> f2 = sin(x)**2 + 2*sin(x) + 1 
1182     >>> pprint(sd(f2, x, S.Reals), use_unicode=False) 
1183               3*pi 
1184     {2*n*pi + ---- | n in Integers} 
1185                2 
1186     >>> f3 = sin(x + 2) 
1187     >>> pprint(sd(f3, x, S.Reals), use_unicode=False) 
1188     {2*n*pi - 2 | n in Integers} U {2*n*pi - 2 + pi | n in Integers} 
1189  
1190     """
1191     from sympy.solvers.decompogen import decompogen
1192     # decompose the given function
1193     g_s = decompogen(f, symbol)
1194     # `y_s` represents the set of values for which the function `g` is to be
1195     # solved.
1196     # `solutions` represent the solutions of the equations `g = y_s` or
1197     # `g = 0` depending on the type of `y_s`.
1198     # As we are interested in solving the equation: f = 0
1199     y_s = FiniteSet(0)
1200     for g in g_s:
1201         frange = function_range(g, symbol, domain)
1202         y_s = Intersection(frange, y_s)
1203         result = S.EmptySet
1204         if isinstance(y_s, FiniteSet):
1205             for y in y_s:
1206                 solutions = solveset(Eq(g, y), symbol, domain)
1207                 if not isinstance(solutions, ConditionSet):
1208                     result += solutions
1209 
1210         else:
1211             if isinstance(y_s, ImageSet):
1212                 iter_iset = (y_s,)
1213 
1214             elif isinstance(y_s, Union):
1215                 iter_iset = y_s.args
1216 
1217             elif y_s is S.EmptySet:
1218                 # y_s is not in the range of g in g_s, so no solution exists
1219                 #in the given domain
1220                 return S.EmptySet
1221 
1222             for iset in iter_iset:
1223                 new_solutions = solveset(Eq(iset.lamda.expr, g), symbol, domain)
1224                 dummy_var = tuple(iset.lamda.expr.free_symbols)[0]
1225                 (base_set,) = iset.base_sets
1226                 if isinstance(new_solutions, FiniteSet):
1227                     new_exprs = new_solutions
1228 
1229                 elif isinstance(new_solutions, Intersection):
1230                     if isinstance(new_solutions.args[1], FiniteSet):
1231                         new_exprs = new_solutions.args[1]
1232 
1233                 for new_expr in new_exprs:
1234                     result += ImageSet(Lambda(dummy_var, new_expr), base_set)
1235 
1236         if result is S.EmptySet:
1237             return ConditionSet(symbol, Eq(f, 0), domain)
1238 
1239         y_s = result
1240 
1241     return y_s

1244 def _solveset(f, symbol, domain, _check=False):
1245     """Helper for solveset to return a result from an expression 
1246     that has already been sympify'ed and is known to contain the 
1247     given symbol."""
1248     # _check controls whether the answer is checked or not
1249     from sympy.simplify.simplify import signsimp
1250 
1251     if isinstance(f, BooleanTrue):
1252         return domain
1253 
1254     orig_f = f
1255     if f.is_Mul:
1256         coeff, f = f.as_independent(symbol, as_Add=False)
1257         if coeff in {S.ComplexInfinity, S.NegativeInfinity, S.Infinity}:
1258             f = together(orig_f)
1259     elif f.is_Add:
1260         a, h = f.as_independent(symbol)
1261         m, h = h.as_independent(symbol, as_Add=False)
1262         if m not in {S.ComplexInfinity, S.Zero, S.Infinity,
1263                               S.NegativeInfinity}:
1264             f = a/m + h  # XXX condition `m != 0` should be added to soln
1265 
1266     # assign the solvers to use
1267     solver = lambda f, x, domain=domain: _solveset(f, x, domain)
1268     inverter = lambda f, rhs, symbol: _invert(f, rhs, symbol, domain)
1269 
1270     result = S.EmptySet
1271 
1272     if f.expand().is_zero:
1273         return domain
1274     elif not f.has(symbol):
1275         return S.EmptySet
1276     elif f.is_Mul and all(_is_finite_with_finite_vars(m, domain)
1277             for m in f.args):
1278         # if f(x) and g(x) are both finite we can say that the solution of
1279         # f(x)*g(x) == 0 is same as Union(f(x) == 0, g(x) == 0) is not true in
1280         # general. g(x) can grow to infinitely large for the values where
1281         # f(x) == 0. To be sure that we are not silently allowing any
1282         # wrong solutions we are using this technique only if both f and g are
1283         # finite for a finite input.
1284         result = Union(*[solver(m, symbol) for m in f.args])
1285     elif (_is_function_class_equation(TrigonometricFunction, f, symbol) or \
1286             _is_function_class_equation(HyperbolicFunction, f, symbol)):
1287         result = _solve_trig(f, symbol, domain)
1288     elif isinstance(f, arg):
1289         a = f.args[0]
1290         result = Intersection(_solveset(re(a) > 0, symbol, domain),
1291                               _solveset(im(a), symbol, domain))
1292     elif f.is_Piecewise:
1293         expr_set_pairs = f.as_expr_set_pairs(domain)
1294         for (expr, in_set) in expr_set_pairs:
1295             if in_set.is_Relational:
1296                 in_set = in_set.as_set()
1297             solns = solver(expr, symbol, in_set)
1298             result += solns
1299     elif isinstance(f, Eq):
1300         result = solver(Add(f.lhs, -f.rhs, evaluate=False), symbol, domain)
1301 
1302     elif f.is_Relational:
1303         from .inequalities import solve_univariate_inequality
1304         try:
1305             result = solve_univariate_inequality(
1306             f, symbol, domain=domain, relational=False)
1307         except NotImplementedError:
1308             result = ConditionSet(symbol, f, domain)
1309         return result
1310     elif _is_modular(f, symbol):
1311         result = _solve_modular(f, symbol, domain)
1312     else:
1313         lhs, rhs_s = inverter(f, 0, symbol)
1314         if lhs == symbol:
1315             # do some very minimal simplification since
1316             # repeated inversion may have left the result
1317             # in a state that other solvers (e.g. poly)
1318             # would have simplified; this is done here
1319             # rather than in the inverter since here it
1320             # is only done once whereas there it would
1321             # be repeated for each step of the inversion
1322             if isinstance(rhs_s, FiniteSet):
1323                 rhs_s = FiniteSet(*[Mul(*
1324                     signsimp(i).as_content_primitive())
1325                     for i in rhs_s])
1326             result = rhs_s
1327 
1328         elif isinstance(rhs_s, FiniteSet):
1329             for equation in [lhs - rhs for rhs in rhs_s]:
1330                 if equation == f:
1331                     u = unrad(f, symbol)
1332                     if u:
1333                         result += _solve_radical(equation, u,
1334                                                  symbol,
1335                                                  solver)
1336                     elif equation.has(Abs):
1337                         result += _solve_abs(f, symbol, domain)
1338                     else:
1339                         result_rational = _solve_as_rational(equation, symbol, domain)
1340                         if not isinstance(result_rational, ConditionSet):
1341                             result += result_rational
1342                         else:
1343                             # may be a transcendental type equation
1344                             t_result = _transolve(equation, symbol, domain)
1345                             if isinstance(t_result, ConditionSet):
1346                                 # might need factoring; this is expensive so we
1347                                 # have delayed until now. To avoid recursion
1348                                 # errors look for a non-trivial factoring into
1349                                 # a product of symbol dependent terms; I think
1350                                 # that something that factors as a Pow would
1351                                 # have already been recognized by now.
1352                                 factored = equation.factor()
1353                                 if factored.is_Mul and equation != factored:
1354                                     _, dep = factored.as_independent(symbol)
1355                                     if not dep.is_Add:
1356                                         # non-trivial factoring of equation
1357                                         # but use form with constants
1358                                         # in case they need special handling
1359                                         t_results = []
1360                                         for fac in Mul.make_args(factored):
1361                                             if fac.has(symbol):
1362                                                 t_results.append(solver(fac, symbol))
1363                                         t_result = Union(*t_results)
1364                             result += t_result
1365                 else:
1366                     result += solver(equation, symbol)
1367 
1368         elif rhs_s is not S.EmptySet:
1369             result = ConditionSet(symbol, Eq(f, 0), domain)
1370 
1371     if isinstance(result, ConditionSet):
1372         if isinstance(f, Expr):
1373             num, den = f.as_numer_denom()
1374             if den.has(symbol):
1375                 _result = _solveset(num, symbol, domain)
1376                 if not isinstance(_result, ConditionSet):
1377                     singularities = _solveset(den, symbol, domain)
1378                     result = _result - singularities
1379 
1380     if _check:
1381         if isinstance(result, ConditionSet):
1382             # it wasn't solved or has enumerated all conditions
1383             # -- leave it alone
1384             return result
1385 
1386         # whittle away all but the symbol-containing core
1387         # to use this for testing
1388         if isinstance(orig_f, Expr):
1389             fx = orig_f.as_independent(symbol, as_Add=True)[1]
1390             fx = fx.as_independent(symbol, as_Add=False)[1]
1391         else:
1392             fx = orig_f
1393 
1394         if isinstance(result, FiniteSet):
1395             # check the result for invalid solutions
1396             result = FiniteSet(*[s for s in result
1397                       if isinstance(s, RootOf)
1398                       or domain_check(fx, symbol, s)])
1399 
1400     return result

1403 def _is_modular(f, symbol):
1404     """ 
1405     Helper function to check below mentioned types of modular equations. 
1406     ``A - Mod(B, C) = 0`` 
1407  
1408     A -> This can or cannot be a function of symbol. 
1409     B -> This is surely a function of symbol. 
1410     C -> It is an integer. 
1411  
1412     Parameters 
1413     ========== 
1414  
1415     f : Expr 
1416         The equation to be checked. 
1417  
1418     symbol : Symbol 
1419         The concerned variable for which the equation is to be checked. 
1420  
1421     Examples 
1422     ======== 
1423  
1424     >>> from sympy import symbols, exp, Mod 
1425     >>> from sympy.solvers.solveset import _is_modular as check 
1426     >>> x, y = symbols('x y') 
1427     >>> check(Mod(x, 3) - 1, x) 
1428     True 
1429     >>> check(Mod(x, 3) - 1, y) 
1430     False 
1431     >>> check(Mod(x, 3)**2 - 5, x) 
1432     False 
1433     >>> check(Mod(x, 3)**2 - y, x) 
1434     False 
1435     >>> check(exp(Mod(x, 3)) - 1, x) 
1436     False 
1437     >>> check(Mod(3, y) - 1, y) 
1438     False 
1439     """
1440 
1441     if not f.has(Mod):
1442         return False
1443 
1444     # extract modterms from f.
1445     modterms = list(f.atoms(Mod))
1446 
1447     return (len(modterms) == 1 and  # only one Mod should be present
1448             modterms[0].args[0].has(symbol) and  # B-> function of symbol
1449             modterms[0].args[1].is_integer and  # C-> to be an integer.
1450             any(isinstance(term, Mod)
1451             for term in list(_term_factors(f)))  # free from other funcs
1452             )

1455 def _invert_modular(modterm, rhs, n, symbol):
1456     """ 
1457     Helper function to invert modular equation. 
1458     ``Mod(a, m) - rhs = 0`` 
1459  
1460     Generally it is inverted as (a, ImageSet(Lambda(n, m*n + rhs), S.Integers)). 
1461     More simplified form will be returned if possible. 
1462  
1463     If it is not invertible then (modterm, rhs) is returned. 
1464  
1465     The following cases arise while inverting equation ``Mod(a, m) - rhs = 0``: 
1466  
1467     1. If a is symbol then  m*n + rhs is the required solution. 
1468  
1469     2. If a is an instance of ``Add`` then we try to find two symbol independent 
1470        parts of a and the symbol independent part gets transferred to the other 
1471        side and again the ``_invert_modular`` is called on the symbol 
1472        dependent part. 
1473  
1474     3. If a is an instance of ``Mul`` then same as we done in ``Add`` we separate 
1475        out the symbol dependent and symbol independent parts and transfer the 
1476        symbol independent part to the rhs with the help of invert and again the 
1477        ``_invert_modular`` is called on the symbol dependent part. 
1478  
1479     4. If a is an instance of ``Pow`` then two cases arise as following: 
1480  
1481         - If a is of type (symbol_indep)**(symbol_dep) then the remainder is 
1482           evaluated with the help of discrete_log function and then the least 
1483           period is being found out with the help of totient function. 
1484           period*n + remainder is the required solution in this case. 
1485           For reference: (https://en.wikipedia.org/wiki/Euler's_theorem) 
1486  
1487         - If a is of type (symbol_dep)**(symbol_indep) then we try to find all 
1488           primitive solutions list with the help of nthroot_mod function. 
1489           m*n + rem is the general solution where rem belongs to solutions list 
1490           from nthroot_mod function. 
1491  
1492     Parameters 
1493     ========== 
1494  
1495     modterm, rhs : Expr 
1496         The modular equation to be inverted, ``modterm - rhs = 0`` 
1497  
1498     symbol : Symbol 
1499         The variable in the equation to be inverted. 
1500  
1501     n : Dummy 
1502         Dummy variable for output g_n. 
1503  
1504     Returns 
1505     ======= 
1506  
1507     A tuple (f_x, g_n) is being returned where f_x is modular independent function 
1508     of symbol and g_n being set of values f_x can have. 
1509  
1510     Examples 
1511     ======== 
1512  
1513     >>> from sympy import symbols, exp, Mod, Dummy, S 
1514     >>> from sympy.solvers.solveset import _invert_modular as invert_modular 
1515     >>> x, y = symbols('x y') 
1516     >>> n = Dummy('n') 
1517     >>> invert_modular(Mod(exp(x), 7), S(5), n, x) 
1518     (Mod(exp(x), 7), 5) 
1519     >>> invert_modular(Mod(x, 7), S(5), n, x) 
1520     (x, ImageSet(Lambda(_n, 7*_n + 5), Integers)) 
1521     >>> invert_modular(Mod(3*x + 8, 7), S(5), n, x) 
1522     (x, ImageSet(Lambda(_n, 7*_n + 6), Integers)) 
1523     >>> invert_modular(Mod(x**4, 7), S(5), n, x) 
1524     (x, EmptySet) 
1525     >>> invert_modular(Mod(2**(x**2 + x + 1), 7), S(2), n, x) 
1526     (x**2 + x + 1, ImageSet(Lambda(_n, 3*_n + 1), Naturals0)) 
1527  
1528     """
1529     a, m = modterm.args
1530 
1531     if rhs.is_integer is False:
1532         return symbol, S.EmptySet
1533 
1534     if rhs.is_real is False or any(term.is_real is False
1535             for term in list(_term_factors(a))):
1536         # Check for complex arguments
1537         return modterm, rhs
1538 
1539     if abs(rhs) >= abs(m):
1540         # if rhs has value greater than value of m.
1541         return symbol, S.EmptySet
1542 
1543     if a == symbol:
1544         return symbol, ImageSet(Lambda(n, m*n + rhs), S.Integers)
1545 
1546     if a.is_Add:
1547         # g + h = a
1548         g, h = a.as_independent(symbol)
1549         if g is not S.Zero:
1550             x_indep_term = rhs - Mod(g, m)
1551             return _invert_modular(Mod(h, m), Mod(x_indep_term, m), n, symbol)
1552 
1553     if a.is_Mul:
1554         # g*h = a
1555         g, h = a.as_independent(symbol)
1556         if g is not S.One:
1557             x_indep_term = rhs*invert(g, m)
1558             return _invert_modular(Mod(h, m), Mod(x_indep_term, m), n, symbol)
1559 
1560     if a.is_Pow:
1561         # base**expo = a
1562         base, expo = a.args
1563         if expo.has(symbol) and not base.has(symbol):
1564             # remainder -> solution independent of n of equation.
1565             # m, rhs are made coprime by dividing number_gcd(m, rhs)
1566             if not m.is_Integer and rhs.is_Integer and a.base.is_Integer:
1567                 return modterm, rhs
1568 
1569             mdiv = m.p // number_gcd(m.p, rhs.p)
1570             try:
1571                 remainder = discrete_log(mdiv, rhs.p, a.base.p)
1572             except ValueError:  # log does not exist
1573                 return modterm, rhs
1574             # period -> coefficient of n in the solution and also referred as
1575             # the least period of expo in which it is repeats itself.
1576             # (a**(totient(m)) - 1) divides m. Here is link of theorem:
1577             # (https://en.wikipedia.org/wiki/Euler's_theorem)
1578             period = totient(m)
1579             for p in divisors(period):
1580                 # there might a lesser period exist than totient(m).
1581                 if pow(a.base, p, m / number_gcd(m.p, a.base.p)) == 1:
1582                     period = p
1583                     break
1584             # recursion is not applied here since _invert_modular is currently
1585             # not smart enough to handle infinite rhs as here expo has infinite
1586             # rhs = ImageSet(Lambda(n, period*n + remainder), S.Naturals0).
1587             return expo, ImageSet(Lambda(n, period*n + remainder), S.Naturals0)
1588         elif base.has(symbol) and not expo.has(symbol):
1589             try:
1590                 remainder_list = nthroot_mod(rhs, expo, m, all_roots=True)
1591                 if remainder_list == []:
1592                     return symbol, S.EmptySet
1593             except (ValueError, NotImplementedError):
1594                 return modterm, rhs
1595             g_n = S.EmptySet
1596             for rem in remainder_list:
1597                 g_n += ImageSet(Lambda(n, m*n + rem), S.Integers)
1598             return base, g_n
1599 
1600     return modterm, rhs

1603 def _solve_modular(f, symbol, domain):
1604     r""" 
1605     Helper function for solving modular equations of type ``A - Mod(B, C) = 0``, 
1606     where A can or cannot be a function of symbol, B is surely a function of 
1607     symbol and C is an integer. 
1608  
1609     Currently ``_solve_modular`` is only able to solve cases 
1610     where A is not a function of symbol. 
1611  
1612     Parameters 
1613     ========== 
1614  
1615     f : Expr 
1616         The modular equation to be solved, ``f = 0`` 
1617  
1618     symbol : Symbol 
1619         The variable in the equation to be solved. 
1620  
1621     domain : Set 
1622         A set over which the equation is solved. It has to be a subset of 
1623         Integers. 
1624  
1625     Returns 
1626     ======= 
1627  
1628     A set of integer solutions satisfying the given modular equation. 
1629     A ``ConditionSet`` if the equation is unsolvable. 
1630  
1631     Examples 
1632     ======== 
1633  
1634     >>> from sympy.solvers.solveset import _solve_modular as solve_modulo 
1635     >>> from sympy import S, Symbol, sin, Intersection, Interval, Mod 
1636     >>> x = Symbol('x') 
1637     >>> solve_modulo(Mod(5*x - 8, 7) - 3, x, S.Integers) 
1638     ImageSet(Lambda(_n, 7*_n + 5), Integers) 
1639     >>> solve_modulo(Mod(5*x - 8, 7) - 3, x, S.Reals)  # domain should be subset of integers. 
1640     ConditionSet(x, Eq(Mod(5*x + 6, 7) - 3, 0), Reals) 
1641     >>> solve_modulo(-7 + Mod(x, 5), x, S.Integers) 
1642     EmptySet 
1643     >>> solve_modulo(Mod(12**x, 21) - 18, x, S.Integers) 
1644     ImageSet(Lambda(_n, 6*_n + 2), Naturals0) 
1645     >>> solve_modulo(Mod(sin(x), 7) - 3, x, S.Integers) # not solvable 
1646     ConditionSet(x, Eq(Mod(sin(x), 7) - 3, 0), Integers) 
1647     >>> solve_modulo(3 - Mod(x, 5), x, Intersection(S.Integers, Interval(0, 100))) 
1648     Intersection(ImageSet(Lambda(_n, 5*_n + 3), Integers), Range(0, 101, 1)) 
1649     """
1650     # extract modterm and g_y from f
1651     unsolved_result = ConditionSet(symbol, Eq(f, 0), domain)
1652     modterm = list(f.atoms(Mod))[0]
1653     rhs = -S.One*(f.subs(modterm, S.Zero))
1654     if f.as_coefficients_dict()[modterm].is_negative:
1655         # checks if coefficient of modterm is negative in main equation.
1656         rhs *= -S.One
1657 
1658     if not domain.is_subset(S.Integers):
1659         return unsolved_result
1660 
1661     if rhs.has(symbol):
1662         # TODO Case: A-> function of symbol, can be extended here
1663         # in future.
1664         return unsolved_result
1665 
1666     n = Dummy('n', integer=True)
1667     f_x, g_n = _invert_modular(modterm, rhs, n, symbol)
1668 
1669     if f_x == modterm and g_n == rhs:
1670         return unsolved_result
1671 
1672     if f_x == symbol:
1673         if domain is not S.Integers:
1674             return domain.intersect(g_n)
1675         return g_n
1676 
1677     if isinstance(g_n, ImageSet):
1678         lamda_expr = g_n.lamda.expr
1679         lamda_vars = g_n.lamda.variables
1680         base_sets = g_n.base_sets
1681         sol_set = _solveset(f_x - lamda_expr, symbol, S.Integers)
1682         if isinstance(sol_set, FiniteSet):
1683             tmp_sol = S.EmptySet
1684             for sol in sol_set:
1685                 tmp_sol += ImageSet(Lambda(lamda_vars, sol), *base_sets)
1686             sol_set = tmp_sol
1687         else:
1688             sol_set =  ImageSet(Lambda(lamda_vars, sol_set), *base_sets)
1689         return domain.intersect(sol_set)
1690 
1691     return unsolved_result

1721 def _solve_exponential(lhs, rhs, symbol, domain):
1722     r""" 
1723     Helper function for solving (supported) exponential equations. 
1724  
1725     Exponential equations are the sum of (currently) at most 
1726     two terms with one or both of them having a power with a 
1727     symbol-dependent exponent. 
1728  
1729     For example 
1730  
1731     .. math:: 5^{2x + 3} - 5^{3x - 1} 
1732  
1733     .. math:: 4^{5 - 9x} - e^{2 - x} 
1734  
1735     Parameters 
1736     ========== 
1737  
1738     lhs, rhs : Expr 
1739         The exponential equation to be solved, `lhs = rhs` 
1740  
1741     symbol : Symbol 
1742         The variable in which the equation is solved 
1743  
1744     domain : Set 
1745         A set over which the equation is solved. 
1746  
1747     Returns 
1748     ======= 
1749  
1750     A set of solutions satisfying the given equation. 
1751     A ``ConditionSet`` if the equation is unsolvable or 
1752     if the assumptions are not properly defined, in that case 
1753     a different style of ``ConditionSet`` is returned having the 
1754     solution(s) of the equation with the desired assumptions. 
1755  
1756     Examples 
1757     ======== 
1758  
1759     >>> from sympy.solvers.solveset import _solve_exponential as solve_expo 
1760     >>> from sympy import symbols, S 
1761     >>> x = symbols('x', real=True) 
1762     >>> a, b = symbols('a b') 
1763     >>> solve_expo(2**x + 3**x - 5**x, 0, x, S.Reals)  # not solvable 
1764     ConditionSet(x, Eq(2**x + 3**x - 5**x, 0), Reals) 
1765     >>> solve_expo(a**x - b**x, 0, x, S.Reals)  # solvable but incorrect assumptions 
1766     ConditionSet(x, (a > 0) & (b > 0), {0}) 
1767     >>> solve_expo(3**(2*x) - 2**(x + 3), 0, x, S.Reals) 
1768     {-3*log(2)/(-2*log(3) + log(2))} 
1769     >>> solve_expo(2**x - 4**x, 0, x, S.Reals) 
1770     {0} 
1771  
1772     * Proof of correctness of the method 
1773  
1774     The logarithm function is the inverse of the exponential function. 
1775     The defining relation between exponentiation and logarithm is: 
1776  
1777     .. math:: {\log_b x} = y \enspace if \enspace b^y = x 
1778  
1779     Therefore if we are given an equation with exponent terms, we can 
1780     convert every term to its corresponding logarithmic form. This is 
1781     achieved by taking logarithms and expanding the equation using 
1782     logarithmic identities so that it can easily be handled by ``solveset``. 
1783  
1784     For example: 
1785  
1786     .. math:: 3^{2x} = 2^{x + 3} 
1787  
1788     Taking log both sides will reduce the equation to 
1789  
1790     .. math:: (2x)\log(3) = (x + 3)\log(2) 
1791  
1792     This form can be easily handed by ``solveset``. 
1793     """
1794     unsolved_result = ConditionSet(symbol, Eq(lhs - rhs, 0), domain)
1795     newlhs = powdenest(lhs)
1796     if lhs != newlhs:
1797         # it may also be advantageous to factor the new expr
1798         neweq = factor(newlhs - rhs)
1799         if neweq != (lhs - rhs):
1800             return _solveset(neweq, symbol, domain)  # try again with _solveset
1801 
1802     if not (isinstance(lhs, Add) and len(lhs.args) == 2):
1803         # solving for the sum of more than two powers is possible
1804         # but not yet implemented
1805         return unsolved_result
1806 
1807     if rhs != 0:
1808         return unsolved_result
1809 
1810     a, b = list(ordered(lhs.args))
1811     a_term = a.as_independent(symbol)[1]
1812     b_term = b.as_independent(symbol)[1]
1813 
1814     a_base, a_exp = a_term.as_base_exp()
1815     b_base, b_exp = b_term.as_base_exp()
1816 
1817     if domain.is_subset(S.Reals):
1818         conditions = And(
1819             a_base > 0,
1820             b_base > 0,
1821             Eq(im(a_exp), 0),
1822             Eq(im(b_exp), 0))
1823     else:
1824         conditions = And(
1825             Ne(a_base, 0),
1826             Ne(b_base, 0))
1827 
1828     L, R = (expand_log(log(i), force=True) for i in (a, -b))
1829     solutions = _solveset(L - R, symbol, domain)
1830 
1831     return ConditionSet(symbol, conditions, solutions)

1836     Return ``True`` if one or more terms contain ``symbol`` only in 
1837     exponents, else ``False``. 
1838  
1839     Parameters 
1840     ==========

1844  
1845     symbol : Symbol 
1846         The variable in which the equation is checked

1883 def _solve_logarithm(lhs, rhs, symbol, domain):
1884     r""" 
1885     Helper to solve logarithmic equations which are reducible 
1886     to a single instance of `\log`. 
1887  
1888     Logarithmic equations are (currently) the equations that contains 
1889     `\log` terms which can be reduced to a single `\log` term or 
1890     a constant using various logarithmic identities. 
1891  
1892     For example: 
1893  
1894     .. math:: \log(x) + \log(x - 4) 
1895  
1896     can be reduced to: 
1897  
1898     .. math:: \log(x(x - 4)) 
1899  
1900     Parameters 
1901     ========== 
1902  
1903     lhs, rhs : Expr 
1904         The logarithmic equation to be solved, `lhs = rhs` 
1905  
1906     symbol : Symbol 
1907         The variable in which the equation is solved 
1908  
1909     domain : Set 
1910         A set over which the equation is solved. 
1911  
1912     Returns 
1913     ======= 
1914  
1915     A set of solutions satisfying the given equation. 
1916     A ``ConditionSet`` if the equation is unsolvable. 
1917  
1918     Examples 
1919     ======== 
1920  
1921     >>> from sympy import symbols, log, S 
1922     >>> from sympy.solvers.solveset import _solve_logarithm as solve_log 
1923     >>> x = symbols('x') 
1924     >>> f = log(x - 3) + log(x + 3) 
1925     >>> solve_log(f, 0, x, S.Reals) 
1926     {-sqrt(10), sqrt(10)} 
1927  
1928     * Proof of correctness 
1929  
1930     A logarithm is another way to write exponent and is defined by 
1931  
1932     .. math:: {\log_b x} = y \enspace if \enspace b^y = x 
1933  
1934     When one side of the equation contains a single logarithm, the 
1935     equation can be solved by rewriting the equation as an equivalent 
1936     exponential equation as defined above. But if one side contains 
1937     more than one logarithm, we need to use the properties of logarithm 
1938     to condense it into a single logarithm. 
1939  
1940     Take for example 
1941  
1942     .. math:: \log(2x) - 15 = 0 
1943  
1944     contains single logarithm, therefore we can directly rewrite it to 
1945     exponential form as 
1946  
1947     .. math:: x = \frac{e^{15}}{2} 
1948  
1949     But if the equation has more than one logarithm as 
1950  
1951     .. math:: \log(x - 3) + \log(x + 3) = 0 
1952  
1953     we use logarithmic identities to convert it into a reduced form 
1954  
1955     Using, 
1956  
1957     .. math:: \log(a) + \log(b) = \log(ab) 
1958  
1959     the equation becomes, 
1960  
1961     .. math:: \log((x - 3)(x + 3)) 
1962  
1963     This equation contains one logarithm and can be solved by rewriting 
1964     to exponents. 
1965     """
1966     new_lhs = logcombine(lhs, force=True)
1967     new_f = new_lhs - rhs
1968 
1969     return _solveset(new_f, symbol, domain)

2030 def _is_lambert(f, symbol):
2031     r""" 
2032     If this returns ``False`` then the Lambert solver (``_solve_lambert``) will not be called. 
2033  
2034     Explanation 
2035     =========== 
2036  
2037     Quick check for cases that the Lambert solver might be able to handle. 
2038  
2039     1. Equations containing more than two operands and `symbol`s involving any of 
2040        `Pow`, `exp`, `HyperbolicFunction`,`TrigonometricFunction`, `log` terms. 
2041  
2042     2. In `Pow`, `exp` the exponent should have `symbol` whereas for 
2043        `HyperbolicFunction`,`TrigonometricFunction`, `log` should contain `symbol`. 
2044  
2045     3. For `HyperbolicFunction`,`TrigonometricFunction` the number of trigonometric functions in 
2046        equation should be less than number of symbols. (since `A*cos(x) + B*sin(x) - c` 
2047        is not the Lambert type). 
2048  
2049     Some forms of lambert equations are: 
2050         1. X**X = C 
2051         2. X*(B*log(X) + D)**A = C 
2052         3. A*log(B*X + A) + d*X = C 
2053         4. (B*X + A)*exp(d*X + g) = C 
2054         5. g*exp(B*X + h) - B*X = C 
2055         6. A*D**(E*X + g) - B*X = C 
2056         7. A*cos(X) + B*sin(X) - D*X = C 
2057         8. A*cosh(X) + B*sinh(X) - D*X = C 
2058  
2059     Where X is any variable, 
2060           A, B, C, D, E are any constants, 
2061           g, h are linear functions or log terms. 
2062  
2063     Parameters 
2064     ========== 
2065  
2066     f : Expr 
2067         The equation to be checked 
2068  
2069     symbol : Symbol 
2070         The variable in which the equation is checked 
2071  
2072     Returns 
2073     ======= 
2074  
2075     If this returns ``False`` then the Lambert solver (``_solve_lambert``) will not be called. 
2076  
2077     Examples 
2078     ======== 
2079  
2080     >>> from sympy.solvers.solveset import _is_lambert 
2081     >>> from sympy import symbols, cosh, sinh, log 
2082     >>> x = symbols('x') 
2083  
2084     >>> _is_lambert(3*log(x) - x*log(3), x) 
2085     True 
2086     >>> _is_lambert(log(log(x - 3)) + log(x-3), x) 
2087     True 
2088     >>> _is_lambert(cosh(x) - sinh(x), x) 
2089     False 
2090     >>> _is_lambert((x**2 - 2*x + 1).subs(x, (log(x) + 3*x)**2 - 1), x) 
2091     True 
2092  
2093     See Also 
2094     ======== 
2095  
2096     _solve_lambert 
2097  
2098     """
2099     term_factors = list(_term_factors(f.expand()))
2100 
2101     # total number of symbols in equation
2102     no_of_symbols = len([arg for arg in term_factors if arg.has(symbol)])
2103     # total number of trigonometric terms in equation
2104     no_of_trig = len([arg for arg in term_factors \
2105         if arg.has(HyperbolicFunction, TrigonometricFunction)])
2106 
2107     if f.is_Add and no_of_symbols >= 2:
2108         # `log`, `HyperbolicFunction`, `TrigonometricFunction` should have symbols
2109         # and no_of_trig < no_of_symbols
2110         lambert_funcs = (log, HyperbolicFunction, TrigonometricFunction)
2111         if any(isinstance(arg, lambert_funcs)\
2112             for arg in term_factors if arg.has(symbol)):
2113                 if no_of_trig < no_of_symbols:
2114                     return True
2115         # here, `Pow`, `exp` exponent should have symbols
2116         elif any(isinstance(arg, (Pow, exp)) \
2117             for arg in term_factors if (arg.as_base_exp()[1]).has(symbol)):
2118             return True
2119     return False

2122 def _transolve(f, symbol, domain):
2123     r""" 
2124     Function to solve transcendental equations. It is a helper to 
2125     ``solveset`` and should be used internally. ``_transolve`` 
2126     currently supports the following class of equations: 
2127  
2128         - Exponential equations 
2129         - Logarithmic equations 
2130  
2131     Parameters 
2132     ========== 
2133  
2134     f : Any transcendental equation that needs to be solved. 
2135         This needs to be an expression, which is assumed 
2136         to be equal to ``0``. 
2137  
2138     symbol : The variable for which the equation is solved. 
2139         This needs to be of class ``Symbol``. 
2140  
2141     domain : A set over which the equation is solved. 
2142         This needs to be of class ``Set``. 
2143  
2144     Returns 
2145     ======= 
2146  
2147     Set 
2148         A set of values for ``symbol`` for which ``f`` is equal to 
2149         zero. An ``EmptySet`` is returned if ``f`` does not have solutions 
2150         in respective domain. A ``ConditionSet`` is returned as unsolved 
2151         object if algorithms to evaluate complete solution are not 
2152         yet implemented. 
2153  
2154     How to use ``_transolve`` 
2155     ========================= 
2156  
2157     ``_transolve`` should not be used as an independent function, because 
2158     it assumes that the equation (``f``) and the ``symbol`` comes from 
2159     ``solveset`` and might have undergone a few modification(s). 
2160     To use ``_transolve`` as an independent function the equation (``f``) 
2161     and the ``symbol`` should be passed as they would have been by 
2162     ``solveset``. 
2163  
2164     Examples 
2165     ======== 
2166  
2167     >>> from sympy.solvers.solveset import _transolve as transolve 
2168     >>> from sympy.solvers.solvers import _tsolve as tsolve 
2169     >>> from sympy import symbols, S, pprint 
2170     >>> x = symbols('x', real=True) # assumption added 
2171     >>> transolve(5**(x - 3) - 3**(2*x + 1), x, S.Reals) 
2172     {-(log(3) + 3*log(5))/(-log(5) + 2*log(3))} 
2173  
2174     How ``_transolve`` works 
2175     ======================== 
2176  
2177     ``_transolve`` uses two types of helper functions to solve equations 
2178     of a particular class: 
2179  
2180     Identifying helpers: To determine whether a given equation 
2181     belongs to a certain class of equation or not. Returns either 
2182     ``True`` or ``False``. 
2183  
2184     Solving helpers: Once an equation is identified, a corresponding 
2185     helper either solves the equation or returns a form of the equation 
2186     that ``solveset`` might better be able to handle. 
2187  
2188     * Philosophy behind the module 
2189  
2190     The purpose of ``_transolve`` is to take equations which are not 
2191     already polynomial in their generator(s) and to either recast them 
2192     as such through a valid transformation or to solve them outright. 
2193     A pair of helper functions for each class of supported 
2194     transcendental functions are employed for this purpose. One 
2195     identifies the transcendental form of an equation and the other 
2196     either solves it or recasts it into a tractable form that can be 
2197     solved by  ``solveset``. 
2198     For example, an equation in the form `ab^{f(x)} - cd^{g(x)} = 0` 
2199     can be transformed to 
2200     `\log(a) + f(x)\log(b) - \log(c) - g(x)\log(d) = 0` 
2201     (under certain assumptions) and this can be solved with ``solveset`` 
2202     if `f(x)` and `g(x)` are in polynomial form. 
2203  
2204     How ``_transolve`` is better than ``_tsolve`` 
2205     ============================================= 
2206  
2207     1) Better output 
2208  
2209     ``_transolve`` provides expressions in a more simplified form. 
2210  
2211     Consider a simple exponential equation 
2212  
2213     >>> f = 3**(2*x) - 2**(x + 3) 
2214     >>> pprint(transolve(f, x, S.Reals), use_unicode=False) 
2215         -3*log(2) 
2216     {------------------} 
2217      -2*log(3) + log(2) 
2218     >>> pprint(tsolve(f, x), use_unicode=False) 
2219          /   3     \ 
2220          | --------| 
2221          | log(2/9)| 
2222     [-log\2         /] 
2223  
2224     2) Extensible 
2225  
2226     The API of ``_transolve`` is designed such that it is easily 
2227     extensible, i.e. the code that solves a given class of 
2228     equations is encapsulated in a helper and not mixed in with 
2229     the code of ``_transolve`` itself. 
2230  
2231     3) Modular 
2232  
2233     ``_transolve`` is designed to be modular i.e, for every class of 
2234     equation a separate helper for identification and solving is 
2235     implemented. This makes it easy to change or modify any of the 
2236     method implemented directly in the helpers without interfering 
2237     with the actual structure of the API. 
2238  
2239     4) Faster Computation 
2240  
2241     Solving equation via ``_transolve`` is much faster as compared to 
2242     ``_tsolve``. In ``solve``, attempts are made computing every possibility 
2243     to get the solutions. This series of attempts makes solving a bit 
2244     slow. In ``_transolve``, computation begins only after a particular 
2245     type of equation is identified. 
2246  
2247     How to add new class of equations 
2248     ================================= 
2249  
2250     Adding a new class of equation solver is a three-step procedure: 
2251  
2252     - Identify the type of the equations 
2253  
2254       Determine the type of the class of equations to which they belong: 
2255       it could be of ``Add``, ``Pow``, etc. types. Separate internal functions 
2256       are used for each type. Write identification and solving helpers 
2257       and use them from within the routine for the given type of equation 
2258       (after adding it, if necessary). Something like: 
2259  
2260       .. code-block:: python 
2261  
2262         def add_type(lhs, rhs, x): 
2263             .... 
2264             if _is_exponential(lhs, x): 
2265                 new_eq = _solve_exponential(lhs, rhs, x) 
2266         .... 
2267         rhs, lhs = eq.as_independent(x) 
2268         if lhs.is_Add: 
2269             result = add_type(lhs, rhs, x) 
2270  
2271     - Define the identification helper. 
2272  
2273     - Define the solving helper. 
2274  
2275     Apart from this, a few other things needs to be taken care while 
2276     adding an equation solver: 
2277  
2278     - Naming conventions: 
2279       Name of the identification helper should be as 
2280       ``_is_class`` where class will be the name or abbreviation 
2281       of the class of equation. The solving helper will be named as 
2282       ``_solve_class``. 
2283       For example: for exponential equations it becomes 
2284       ``_is_exponential`` and ``_solve_expo``. 
2285     - The identifying helpers should take two input parameters, 
2286       the equation to be checked and the variable for which a solution 
2287       is being sought, while solving helpers would require an additional 
2288       domain parameter. 
2289     - Be sure to consider corner cases. 
2290     - Add tests for each helper. 
2291     - Add a docstring to your helper that describes the method 
2292       implemented. 
2293       The documentation of the helpers should identify: 
2294  
2295       - the purpose of the helper, 
2296       - the method used to identify and solve the equation, 
2297       - a proof of correctness 
2298       - the return values of the helpers 
2299     """

2302         """ 
2303         Helper for ``_transolve`` to handle equations of 
2304         ``Add`` type, i.e. equations taking the form as 
2305         ``a*f(x) + b*g(x) + .... = c``. 
2306         For example: 4**x + 8**x = 0 
2307         """
2308         result = ConditionSet(symbol, Eq(lhs - rhs, 0), domain)
2309 
2310         # check if it is exponential type equation
2311         if _is_exponential(lhs, symbol):
2312             result = _solve_exponential(lhs, rhs, symbol, domain)
2313         # check if it is logarithmic type equation
2314         elif _is_logarithmic(lhs, symbol):
2315             result = _solve_logarithm(lhs, rhs, symbol, domain)
2316 
2317         return result
2318

2320 
2321     # invert_complex handles the call to the desired inverter based

2326         assert (len(rhs_s.args)) == 1

2328 
2329         if lhs.is_Add:
2330             result = add_type(lhs, rhs, symbol, domain)

2333

2598     Others      | None

2602     ======

2604     NotImplementedError

2629 
2630     elif isinstance(solution_set, ConditionSet):
2631         raise NotImplementedError('solveset is unable to solve this equation.')
2632

2635 
2636     else:
2637         period = periodicity(f, symbol)

2640             iter_solutions = ()
2641             if isinstance(solution_set, ImageSet):

2650             if isinstance(solutions, FiniteSet):

2663

2677     corresponding symbols in the sum of terms in  ``eq``. 
2678     The additive constant is returned as the last element of the 
2679     list. 
2680  
2681     Raises

2683

2686     ValueError 
2687         duplicate or unordered symbols are passed

2690     ========== 
2691  
2692     dict - (default False) when True, return coefficients as a 
2693         dictionary with coefficients keyed to syms that were present; 
2694         key 1 gives the constant term

2697     ========

2700     >>> from sympy.abc import x, y, z

2702     [3, 2, -1]

2705  
2706         >>> linear_coeffs(x + y*(z*(x*3 + 2) + 3), x) 
2707         [3*y*z + 1, y*(2*z + 3)]

2710

2712         situation does not pass silently past the caller's

2717         [0, 1]

2720         ...

2722         nonlinear in given generators 
2723

2725  
2726         >>> linear_coeffs(x*(y + 1), x, y) 
2727         Traceback (most recent call last): 
2728         ... 
2729         NonlinearError: 
2730         symbol-dependent cross-terms encountered 
2731  
2732         * when there are terms that contain an expression 
2733         dependent on the symbols that is not linear 
2734  
2735         >>> linear_coeffs(x**2, x) 
2736         Traceback (most recent call last): 
2737         ... 
2738         NonlinearError: 
2739         nonlinear in given generators 
2740     """
2741     eq = _sympify(eq)
2742     if len(syms) == 1 and iterable(syms[0]) and not isinstance(syms[0], Basic):

2760         if k not in d:
2761             continue
2762         rv[i] = d[k]
2763     return rv

2766 def linear_eq_to_matrix(equations, *symbols):
2767     r""" 
2768     Converts a given System of Equations into Matrix form. Here ``equations`` 
2769     must be a linear system of equations in ``symbols``. Element ``M[i, j]`` 
2770     corresponds to the coefficient of the jth symbol in the ith equation. 
2771  
2772     The Matrix form corresponds to the augmented matrix form. For example: 
2773  
2774     .. math:: 
2775  
2776        4x + 2y + 3z & = 1 \\ 
2777        3x +  y +  z & = -6 \\ 
2778        2x + 4y + 9z & = 2 
2779  
2780     This system will return :math:`A` and :math:`b` as: 
2781  
2782     .. math:: 
2783  
2784        A = \left[\begin{array}{ccc} 
2785        4 & 2 & 3 \\ 
2786        3 & 1 & 1 \\ 
2787        2 & 4 & 9 
2788        \end{array}\right] \\ 
2789  
2790     .. math:: 
2791  
2792        b = \left[\begin{array}{c} 
2793        1 \\ -6 \\ 2 
2794        \end{array}\right] 
2795  
2796     The only simplification performed is to convert 
2797     ``Eq(a, b)`` :math:`\Rightarrow a - b`. 
2798  
2799     Raises 
2800     ====== 
2801  
2802     NonlinearError 
2803         The equations contain a nonlinear term. 
2804     ValueError 
2805         The symbols are not given or are not unique. 
2806  
2807     Examples 
2808     ======== 
2809  
2810     >>> from sympy import linear_eq_to_matrix, symbols 
2811     >>> c, x, y, z = symbols('c, x, y, z') 
2812  
2813     The coefficients (numerical or symbolic) of the symbols will 
2814     be returned as matrices: 
2815  
2816     >>> eqns = [c*x + z - 1 - c, y + z, x - y] 
2817     >>> A, b = linear_eq_to_matrix(eqns, [x, y, z]) 
2818     >>> A 
2819     Matrix([ 
2820     [c,  0, 1], 
2821     [0,  1, 1], 
2822     [1, -1, 0]]) 
2823     >>> b 
2824     Matrix([ 
2825     [c + 1], 
2826     [    0], 
2827     [    0]]) 
2828  
2829     This routine does not simplify expressions and will raise an error 
2830     if nonlinearity is encountered: 
2831  
2832     >>> eqns = [ 
2833     ...     (x**2 - 3*x)/(x - 3) - 3, 
2834     ...     y**2 - 3*y - y*(y - 4) + x - 4] 
2835     >>> linear_eq_to_matrix(eqns, [x, y]) 
2836     Traceback (most recent call last): 
2837     ... 
2838     NonlinearError: 
2839     symbol-dependent term can be ignored using `strict=False` 
2840  
2841     Simplifying these equations will discard the removable singularity in the 
2842     first and reveal the linear structure of the second: 
2843  
2844     >>> [e.simplify() for e in eqns] 
2845     [x - 3, x + y - 4] 
2846  
2847     Any such simplification needed to eliminate nonlinear terms must be done 
2848     *before* calling this routine. 
2849  
2850     """
2851     if not symbols:
2852         raise ValueError(filldedent(''' 
2853             Symbols must be given, for which coefficients 
2854             are to be found. 
2855             '''))
2856 
2857     # Check if 'symbols' is a set and raise an error if it is
2858     if isinstance(symbols[0], set):
2859         raise TypeError(
2860             "Unordered 'set' type is not supported as input for symbols.")
2861 
2862     if hasattr(symbols[0], '__iter__'):
2863         symbols = symbols[0]
2864 
2865     if has_dups(symbols):
2866         raise ValueError('Symbols must be unique')
2867 
2868     equations = sympify(equations)
2869     if isinstance(equations, MatrixBase):
2870         equations = list(equations)
2871     elif isinstance(equations, (Expr, Eq)):
2872         equations = [equations]
2873     elif not is_sequence(equations):
2874         raise ValueError(filldedent(''' 
2875             Equation(s) must be given as a sequence, Expr, 
2876             Eq or Matrix. 
2877             '''))
2878 
2879     # construct the dictionaries
2880     try:
2881         eq, c = _linear_eq_to_dict(equations, symbols)
2882     except PolyNonlinearError as err:
2883         raise NonlinearError(str(err))
2884     # prepare output matrices
2885     n, m = shape = len(eq), len(symbols)
2886     ix = dict(zip(symbols, range(m)))
2887     A = zeros(*shape)
2888     for row, d in enumerate(eq):
2889         for k in d:
2890             col = ix[k]
2891             A[row, col] = d[k]
2892     b = Matrix(n, 1, [-i for i in c])
2893     return A, b

2896 def linsolve(system, *symbols):
2897     r""" 
2898     Solve system of $N$ linear equations with $M$ variables; both 
2899     underdetermined and overdetermined systems are supported. 
2900     The possible number of solutions is zero, one or infinite. 
2901     Zero solutions throws a ValueError, whereas infinite 
2902     solutions are represented parametrically in terms of the given 
2903     symbols. For unique solution a :class:`~.FiniteSet` of ordered tuples 
2904     is returned. 
2905  
2906     All standard input formats are supported: 
2907     For the given set of equations, the respective input types 
2908     are given below: 
2909  
2910     .. math:: 3x + 2y -   z = 1 
2911     .. math:: 2x - 2y + 4z = -2 
2912     .. math:: 2x -   y + 2z = 0 
2913  
2914     * Augmented matrix form, ``system`` given below: 
2915  
2916     $$ \text{system} = \left[{array}{cccc} 
2917         3 &  2 & -1 &  1\\ 
2918         2 & -2 &  4 & -2\\ 
2919         2 & -1 &  2 &  0 
2920         \end{array}\right] $$ 
2921  
2922     :: 
2923  
2924         system = Matrix([[3, 2, -1, 1], [2, -2, 4, -2], [2, -1, 2, 0]]) 
2925  
2926     * List of equations form 
2927  
2928     :: 
2929  
2930         system  =  [3x + 2y - z - 1, 2x - 2y + 4z + 2, 2x - y + 2z] 
2931  
2932     * Input $A$ and $b$ in matrix form (from $Ax = b$) are given as: 
2933  
2934     $$ A = \left[\begin{array}{ccc} 
2935         3 &  2 & -1 \\ 
2936         2 & -2 &  4 \\ 
2937         2 & -1 &  2 
2938         \end{array}\right] \ \  b = \left[\begin{array}{c} 
2939         1 \\ -2 \\ 0 
2940         \end{array}\right] $$ 
2941  
2942     :: 
2943  
2944         A = Matrix([[3, 2, -1], [2, -2, 4], [2, -1, 2]]) 
2945         b = Matrix([[1], [-2], [0]]) 
2946         system = (A, b) 
2947  
2948     Symbols can always be passed but are actually only needed 
2949     when 1) a system of equations is being passed and 2) the 
2950     system is passed as an underdetermined matrix and one wants 
2951     to control the name of the free variables in the result. 
2952     An error is raised if no symbols are used for case 1, but if 
2953     no symbols are provided for case 2, internally generated symbols 
2954     will be provided. When providing symbols for case 2, there should 
2955     be at least as many symbols are there are columns in matrix A. 
2956  
2957     The algorithm used here is Gauss-Jordan elimination, which 
2958     results, after elimination, in a row echelon form matrix. 
2959  
2960     Returns 
2961     ======= 
2962  
2963     A FiniteSet containing an ordered tuple of values for the 
2964     unknowns for which the `system` has a solution. (Wrapping 
2965     the tuple in FiniteSet is used to maintain a consistent 
2966     output format throughout solveset.) 
2967  
2968     Returns EmptySet, if the linear system is inconsistent. 
2969  
2970     Raises 
2971     ====== 
2972  
2973     ValueError 
2974         The input is not valid. 
2975         The symbols are not given. 
2976  
2977     Examples 
2978     ======== 
2979  
2980     >>> from sympy import Matrix, linsolve, symbols 
2981     >>> x, y, z = symbols("x, y, z") 
2982     >>> A = Matrix([[1, 2, 3], [4, 5, 6], [7, 8, 10]]) 
2983     >>> b = Matrix([3, 6, 9]) 
2984     >>> A 
2985     Matrix([ 
2986     [1, 2,  3], 
2987     [4, 5,  6], 
2988     [7, 8, 10]]) 
2989     >>> b 
2990     Matrix([ 
2991     [3], 
2992     [6], 
2993     [9]]) 
2994     >>> linsolve((A, b), [x, y, z]) 
2995     {(-1, 2, 0)} 
2996  
2997     * Parametric Solution: In case the system is underdetermined, the 
2998       function will return a parametric solution in terms of the given 
2999       symbols. Those that are free will be returned unchanged. e.g. in 
3000       the system below, `z` is returned as the solution for variable z; 
3001       it can take on any value. 
3002  
3003     >>> A = Matrix([[1, 2, 3], [4, 5, 6], [7, 8, 9]]) 
3004     >>> b = Matrix([3, 6, 9]) 
3005     >>> linsolve((A, b), x, y, z) 
3006     {(z - 1, 2 - 2*z, z)} 
3007  
3008     If no symbols are given, internally generated symbols will be used. 
3009     The ``tau0`` in the third position indicates (as before) that the third 
3010     variable -- whatever it is named -- can take on any value: 
3011  
3012     >>> linsolve((A, b)) 
3013     {(tau0 - 1, 2 - 2*tau0, tau0)} 
3014  
3015     * List of equations as input 
3016  
3017     >>> Eqns = [3*x + 2*y - z - 1, 2*x - 2*y + 4*z + 2, - x + y/2 - z] 
3018     >>> linsolve(Eqns, x, y, z) 
3019     {(1, -2, -2)} 
3020  
3021     * Augmented matrix as input 
3022  
3023     >>> aug = Matrix([[2, 1, 3, 1], [2, 6, 8, 3], [6, 8, 18, 5]])

3807 def _handle_poly(polys, symbols):
3808     # _handle_poly(polys, symbols) -> (poly_sol, poly_eqs)
3809     #
3810     # We will return possible solution information to nonlinsolve as well as a
3811     # new system of polynomial equations to be solved if we cannot solve
3812     # everything directly here. The new system of polynomial equations will be
3813     # a lex-order Groebner basis for the original system. The lex basis
3814     # hopefully separate some of the variables and equations and give something
3815     # easier for substitution to work with.
3816 
3817     # The format for representing solution sets in nonlinsolve and substitution
3818     # is a list of dicts. These are the special cases:
3819     no_information = [{}]   # No equations solved yet
3820     no_solutions = []       # The system is inconsistent and has no solutions.
3821 
3822     # If there is no need to attempt further solution of these equations then
3823     # we return no equations:
3824     no_equations = []
3825 
3826     inexact = any(not p.domain.is_Exact for p in polys)
3827     if inexact:
3828         # The use of Groebner over RR is likely to result incorrectly in an
3829         # inconsistent Groebner basis. So, convert any float coefficients to
3830         # Rational before computing the Groebner basis.
3831         polys = [poly(nsimplify(p, rational=True)) for p in polys]
3832 
3833     # Compute a Groebner basis in grevlex order wrt the ordering given. We will
3834     # try to convert this to lex order later. Usually it seems to be more
3835     # efficient to compute a lex order basis by computing a grevlex basis and
3836     # converting to lex with fglm.
3837     basis = groebner(polys, symbols, order='grevlex', polys=False)
3838 
3839     #
3840     # No solutions (inconsistent equations)?
3841     #
3842     if 1 in basis:
3843 
3844         # No solutions:
3845         poly_sol = no_solutions
3846         poly_eqs = no_equations
3847 
3848     #
3849     # Finite number of solutions (zero-dimensional case)
3850     #
3851     elif basis.is_zero_dimensional:
3852 
3853         # Convert Groebner basis to lex ordering
3854         basis = basis.fglm('lex')
3855 
3856         # Convert polynomial coefficients back to float before calling
3857         # solve_poly_system
3858         if inexact:
3859             basis = [nfloat(p) for p in basis]
3860 
3861         # Solve the zero-dimensional case using solve_poly_system if possible.
3862         # If some polynomials have factors that cannot be solved in radicals
3863         # then this will fail. Using solve_poly_system(..., strict=True)
3864         # ensures that we either get a complete solution set in radicals or
3865         # UnsolvableFactorError will be raised.
3866         try:
3867             result = solve_poly_system(basis, *symbols, strict=True)
3868         except UnsolvableFactorError:
3869             # Failure... not fully solvable in radicals. Return the lex-order
3870             # basis for substitution to handle.
3871             poly_sol = no_information
3872             poly_eqs = list(basis)
3873         else:
3874             # Success! We have a finite solution set and solve_poly_system has
3875             # succeeded in finding all solutions. Return the solutions and also
3876             # an empty list of remaining equations to be solved.
3877             poly_sol = [dict(zip(symbols, res)) for res in result]
3878             poly_eqs = no_equations
3879 
3880     #
3881     # Infinite families of solutions (positive-dimensional case)
3882     #
3883     else:
3884         # In this case the grevlex basis cannot be converted to lex using the
3885         # fglm method and also solve_poly_system cannot solve the equations. We
3886         # would like to return a lex basis but since we can't use fglm we
3887         # compute the lex basis directly here. The time required to recompute
3888         # the basis is generally significantly less than the time required by
3889         # substitution to solve the new system.
3890         poly_sol = no_information
3891         poly_eqs = list(groebner(polys, symbols, order='lex', polys=False))
3892 
3893         if inexact:
3894             poly_eqs = [nfloat(p) for p in poly_eqs]
3895 
3896     return poly_sol, poly_eqs

3899 def nonlinsolve(system, *symbols):
3900     r""" 
3901     Solve system of $N$ nonlinear equations with $M$ variables, which means both 
3902     under and overdetermined systems are supported. Positive dimensional 
3903     system is also supported (A system with infinitely many solutions is said 
3904     to be positive-dimensional). In a positive dimensional system the solution will 
3905     be dependent on at least one symbol. Returns both real solution 
3906     and complex solution (if they exist). 
3907  
3908     Parameters 
3909     ========== 
3910  
3911     system : list of equations 
3912         The target system of equations 
3913     symbols : list of Symbols 
3914         symbols should be given as a sequence eg. list 
3915  
3916     Returns 
3917     ======= 
3918  
3919     A :class:`~.FiniteSet` of ordered tuple of values of `symbols` for which the `system` 
3920     has solution. Order of values in the tuple is same as symbols present in 
3921     the parameter `symbols`. 
3922  
3923     Please note that general :class:`~.FiniteSet` is unordered, the solution 
3924     returned here is not simply a :class:`~.FiniteSet` of solutions, rather it 
3925     is a :class:`~.FiniteSet` of ordered tuple, i.e. the first and only 
3926     argument to :class:`~.FiniteSet` is a tuple of solutions, which is 
3927     ordered, and, hence ,the returned solution is ordered. 
3928  
3929     Also note that solution could also have been returned as an ordered tuple, 
3930     FiniteSet is just a wrapper ``{}`` around the tuple. It has no other 
3931     significance except for the fact it is just used to maintain a consistent 
3932     output format throughout the solveset. 
3933  
3934     For the given set of equations, the respective input types 
3935     are given below: 
3936  
3937     .. math:: xy - 1 = 0 
3938     .. math:: 4x^2 + y^2 - 5 = 0 
3939  
3940     :: 
3941  
3942        system  = [x*y - 1, 4*x**2 + y**2 - 5] 
3943        symbols = [x, y] 
3944  
3945     Raises 
3946     ====== 
3947  
3948     ValueError 
3949         The input is not valid. 
3950         The symbols are not given. 
3951     AttributeError 
3952         The input symbols are not `Symbol` type. 
3953  
3954     Examples 
3955     ======== 
3956  
3957     >>> from sympy import symbols, nonlinsolve 
3958     >>> x, y, z = symbols('x, y, z', real=True) 
3959     >>> nonlinsolve([x*y - 1, 4*x**2 + y**2 - 5], [x, y]) 
3960     {(-1, -1), (-1/2, -2), (1/2, 2), (1, 1)} 
3961  
3962     1. Positive dimensional system and complements: 
3963  
3964     >>> from sympy import pprint 
3965     >>> from sympy.polys.polytools import is_zero_dimensional 
3966     >>> a, b, c, d = symbols('a, b, c, d', extended_real=True) 
3967     >>> eq1 =  a + b + c + d 
3968     >>> eq2 = a*b + b*c + c*d + d*a 
3969     >>> eq3 = a*b*c + b*c*d + c*d*a + d*a*b 
3970     >>> eq4 = a*b*c*d - 1 
3971     >>> system = [eq1, eq2, eq3, eq4] 
3972     >>> is_zero_dimensional(system) 
3973     False 
3974     >>> pprint(nonlinsolve(system, [a, b, c, d]), use_unicode=False) 
3975       -1       1               1      -1 
3976     {(---, -d, -, {d} \ {0}), (-, -d, ---, {d} \ {0})} 
3977        d       d               d       d 
3978     >>> nonlinsolve([(x+y)**2 - 4, x + y - 2], [x, y]) 
3979     {(2 - y, y)} 
3980  
3981     2. If some of the equations are non-polynomial then `nonlinsolve` 
3982     will call the ``substitution`` function and return real and complex solutions, 
3983     if present. 
3984  
3985     >>> from sympy import exp, sin 
3986     >>> nonlinsolve([exp(x) - sin(y), y**2 - 4], [x, y]) 
3987     {(ImageSet(Lambda(_n, I*(2*_n*pi + pi) + log(sin(2))), Integers), -2), 
3988      (ImageSet(Lambda(_n, 2*_n*I*pi + log(sin(2))), Integers), 2)} 
3989  
3990     3. If system is non-linear polynomial and zero-dimensional then it 
3991     returns both solution (real and complex solutions, if present) using 
3992     :func:`~.solve_poly_system`: 
3993  
3994     >>> from sympy import sqrt 
3995     >>> nonlinsolve([x**2 - 2*y**2 -2, x*y - 2], [x, y]) 
3996     {(-2, -1), (2, 1), (-sqrt(2)*I, sqrt(2)*I), (sqrt(2)*I, -sqrt(2)*I)} 
3997  
3998     4. ``nonlinsolve`` can solve some linear (zero or positive dimensional) 
3999     system (because it uses the :func:`sympy.polys.polytools.groebner` function to get the 
4000     groebner basis and then uses the ``substitution`` function basis as the 
4001     new `system`). But it is not recommended to solve linear system using 
4002     ``nonlinsolve``, because :func:`~.linsolve` is better for general linear systems. 
4003  
4004     >>> nonlinsolve([x + 2*y -z - 3, x - y - 4*z + 9, y + z - 4], [x, y, z]) 
4005     {(3*z - 5, 4 - z, z)} 
4006  
4007     5. System having polynomial equations and only real solution is 
4008     solved using :func:`~.solve_poly_system`: 
4009  
4010     >>> e1 = sqrt(x**2 + y**2) - 10 
4011     >>> e2 = sqrt(y**2 + (-x + 10)**2) - 3 
4012     >>> nonlinsolve((e1, e2), (x, y)) 
4013     {(191/20, -3*sqrt(391)/20), (191/20, 3*sqrt(391)/20)} 
4014     >>> nonlinsolve([x**2 + 2/y - 2, x + y - 3], [x, y]) 
4015     {(1, 2), (1 - sqrt(5), 2 + sqrt(5)), (1 + sqrt(5), 2 - sqrt(5))} 
4016     >>> nonlinsolve([x**2 + 2/y - 2, x + y - 3], [y, x]) 
4017     {(2, 1), (2 - sqrt(5), 1 + sqrt(5)), (2 + sqrt(5), 1 - sqrt(5))} 
4018  
4019     6. It is better to use symbols instead of trigonometric functions or 
4020     :class:`~.Function`. For example, replace $\sin(x)$ with a symbol, replace 
4021     $f(x)$ with a symbol and so on. Get a solution from ``nonlinsolve`` and then 
4022     use :func:`~.solveset` to get the value of $x$. 
4023  
4024     How nonlinsolve is better than old solver ``_solve_system`` : 
4025     ============================================================= 
4026  
4027     1. A positive dimensional system solver: nonlinsolve can return 
4028     solution for positive dimensional system. It finds the 
4029     Groebner Basis of the positive dimensional system(calling it as 
4030     basis) then we can start solving equation(having least number of 
4031     variable first in the basis) using solveset and substituting that 
4032     solved solutions into other equation(of basis) to get solution in 
4033     terms of minimum variables. Here the important thing is how we 
4034     are substituting the known values and in which equations. 
4035  
4036     2. Real and complex solutions: nonlinsolve returns both real 
4037     and complex solution. If all the equations in the system are polynomial 
4038     then using :func:`~.solve_poly_system` both real and complex solution is returned. 
4039     If all the equations in the system are not polynomial equation then goes to 
4040     ``substitution`` method with this polynomial and non polynomial equation(s), 
4041     to solve for unsolved variables. Here to solve for particular variable 
4042     solveset_real and solveset_complex is used. For both real and complex 
4043     solution ``_solve_using_known_values`` is used inside ``substitution`` 
4044     (``substitution`` will be called when any non-polynomial equation is present). 
4045     If a solution is valid its general solution is added to the final result. 
4046  
4047     3. :class:`~.Complement` and :class:`~.Intersection` will be added: 
4048     nonlinsolve maintains dict for complements and intersections. If solveset 
4049     find complements or/and intersections with any interval or set during the 
4050     execution of ``substitution`` function, then complement or/and 
4051     intersection for that variable is added before returning final solution. 
4052  
4053     """
4054     if not system:
4055         return S.EmptySet
4056 
4057     if not symbols:
4058         msg = ('Symbols must be given, for which solution of the '
4059                'system is to be found.')
4060         raise ValueError(filldedent(msg))
4061 
4062     if hasattr(symbols[0], '__iter__'):
4063         symbols = symbols[0]
4064 
4065     if not is_sequence(symbols) or not symbols:
4066         msg = ('Symbols must be given, for which solution of the '
4067                'system is to be found.')
4068         raise IndexError(filldedent(msg))
4069 
4070     symbols = list(map(_sympify, symbols))
4071     system, symbols, swap = recast_to_symbols(system, symbols)
4072     if swap:
4073         soln = nonlinsolve(system, symbols)
4074         return FiniteSet(*[tuple(i.xreplace(swap) for i in s) for s in soln])

4076     if len(system) == 1 and len(symbols) == 1:
4077         return _solveset_work(system, symbols)
4078 
4079     # main code of def nonlinsolve() starts from here
4080 
4081     polys, polys_expr, nonpolys, denominators, unrad_changed = \
4082         _separate_poly_nonpoly(system, symbols)
4083 
4084     poly_eqs = []
4085     poly_sol = [{}]
4086 
4087     if polys:
4088         poly_sol, poly_eqs = _handle_poly(polys, symbols)
4089         if poly_sol and poly_sol[0]:
4090             poly_syms = set().union(*(eq.free_symbols for eq in polys))
4091             unrad_syms = set().union(*(eq.free_symbols for eq in unrad_changed))
4092             if unrad_syms == poly_syms and unrad_changed:
4093                 # if all the symbols have been solved by _handle_poly
4094                 # and unrad has been used then check solutions
4095                 poly_sol = [sol for sol in poly_sol if checksol(unrad_changed, sol)]
4096 
4097     # Collect together the unsolved polynomials with the non-polynomial
4098     # equations.
4099     remaining = poly_eqs + nonpolys
4100

4102     # value for each symbol
4103     to_tuple = lambda sol: tuple(sol[s] for s in symbols)

4107         # solve_poly_system directly.
4108         return FiniteSet(*map(to_tuple, poly_sol))

4111         #
4112         #  1. The Groebner basis if solve_poly_system failed.
4113         #  2. The Groebner basis in the positive-dimensional case.
4114         #  3. Any non-polynomial equations

4117         # preliminary results.

4119

4121             return subs_res
4122

4124         # only those solutions which have non-Set variable values.
4125         if unrad_changed:
4126             result = [dict(zip(symbols, sol)) for sol in subs_res.args]
4127             correct_sols = [sol for sol in result if any(isinstance(v, Set) for v in sol)
4128                             or checksol(unrad_changed, sol) != False]
4129             return FiniteSet(*map(to_tuple, correct_sols))
4130         else:
4131             return subs_res
