52   COMPILE_ID_PATTERN = re.compile(r"^(?P<frame_id>\d+)/(?P<frame_compile_id>\d+)$")
53   CA_COMPILE_ID_PATTERN = re.compile(
54       r"^!(?P<compiled_autograd_id>\d+)(?:/(?P<frame_id>\d+)/(?P<frame_compile_id>\d+))?$"
55   )
56   
57   # [Note: Updating CompiledId]
58   #
59   # CompiledId represents a unique program-level identifier, and we want to keep that
60   # property as the codebase evolves. This property is relied on even outside of the pytorch
61   # repo, e.g. tlparse or other internal tooling. The in-memory format can be freely changed,
62   # as those dependencies only consume the string serialization.
63   #
64   # The string form should be:
65   # 1. Program-level uid: CompileId can uniquely identify a compiled graph.
66   # 2. Storage efficient: This object is logged in nearly every entry. We should elide symbols when possible.
67   # 3. Compact: The string form is directly displayed by some tools. Special symbols are okay.
68   
69   
70   # TODO: mark as kw_only=True once we drop support for <Python 3.10
71   @dataclass(frozen=True)
72   class CompileId:
73       frame_id: Optional[int]
74       # This id is per-frame, and counts how many times we've compiled this
75       # frame.  This could have been a global id but having this be per-frame
76       # gives you a better intuitive sense for how many recompiles have occurred
77       # so far.
78       frame_compile_id: Optional[int]
79   
80       # torch.compiling a compiled autograd graph
81       compiled_autograd_id: Optional[int] = None
82   
83       # TODO: consider also tracking the recompilation count
84       # See Note: Updating CompileId
85   
86       def __str__(self):
87           # NOTE: Keep this in sync with both from_string and the tlparse repo
88           if self.compiled_autograd_id is not None:
89               assert (self.frame_id is None) == (self.frame_compile_id is None)
90               frame_str = ""
91               if self.frame_id is not None:
92                   frame_str = f"/{self.frame_id}/{self.frame_compile_id}"
93   
94               return f"!{self.compiled_autograd_id}{frame_str}"
95           else:
96               assert self.frame_id is not None and self.frame_compile_id is not None
97               return f"{self.frame_id}/{self.frame_compile_id}"
98   
99       @classmethod
100      def from_string(cls, compile_id: Optional[str]):
101          """ 
102          Factory method that creates a CompileId from its string representation. 
103          Keep this in sync with the __str__ method. 
104          """
105          if compile_id is None:
106              return None
107          try:
108              for pattern in (COMPILE_ID_PATTERN, CA_COMPILE_ID_PATTERN):
109                  if match := pattern.match(compile_id):
110                      groups = match.groupdict()
111                      for k, v in groups.items():
112                          if v is not None:
113                              groups[k] = int(v)
114                      return cls(**groups)  # type: ignore[arg-type]
115              else:
116                  raise ValueError
117  
118          except Exception as e:
119              raise ValueError(f"Invalid compile_id '{compile_id}'") from e

239  @dataclass_slots
240  @dataclasses.dataclass
241  class Guard:
242      # originating_source is the source that called the make_guard method to
243      # construct this guard object. The property name specifies what exactly it
244      # is the guard is guarding on.  The meaning of the name is dependent on the
245      # create_fn; you must look at the use-site inside create_fn to know what
246      # name means.
247      #
248      # That being said, although you might think this is just a "name", name is
249      # usually an arbitrary Python expression that will be evaluated with all
250      # globals (and locals, if you create a LOCAL guard) to extract the Python
251      # object that we want to perform guard tests on.  This evaluation
252      # typically happens in GuardBuilder.eval.  In these cases, name is
253      # typically produced by originating_source.name() (not to be confused with
254      # GuardSource - the property source).
255      #
256      # Occasionally, name is not a valid Python expression; sometimes
257      # it is meaningless.  Example create_fns that are like this include
258      # GRAD_MODE and SHAPE_ENV.
259      originating_source: Source
260      create_fn: Callable[[GuardBuilderBase, Guard], None]
261  
262      # Export only. These values are written to at time of guard check_fn creation.
263      guard_types: Optional[list[str]] = None
264      code_list: Optional[list[str]] = None
265      obj_weakref: Optional[object] = None
266      guarded_class_weakref: Optional[type] = None
267  
268      stack: Optional[CapturedTraceback] = None
269      user_stack: Optional[traceback.StackSummary] = None
270      _hash: Optional[int] = None
271  
272      def __hash__(self):
273          if self._hash is None:
274              self._hash = hash((self.name, self.source, id(self.create_fn)))
275          return self._hash
276  
277      def sort_key(self):
278          # Put the duplicate input guards at the end. The duplicate guards have
279          # two sources while guard.name only considers one source.
280  
281          is_duplicate_input = (
282              isinstance(self.create_fn, functools.partial)
283              and self.create_fn.func is torch._dynamo.guards.GuardBuilder.DUPLICATE_INPUT
284          )
285          return (
286              is_duplicate_input,
287              self.source.value if self.source else -1,
288              len(self.name),
289              self.name,
290              self.inner_create_fn().__code__.co_firstlineno,
291          )
292  
293      def __lt__(self, other):
294          return self.sort_key() < other.sort_key()
295  
296      def inner_create_fn(self):
297          if isinstance(self.create_fn, functools.partial):
298              return self.create_fn.func
299          else:
300              return self.create_fn
301  
302      @property
303      def name(self) -> str:
304          return self.originating_source.name()
305  
306      @property
307      def source(self) -> GuardSource:
308          return self.originating_source.guard_source()
309  
310      @staticmethod
311      def weakref_to_str(obj_weakref):
312          """ 
313          This is a workaround of a Python weakref bug. 
314   
315          `obj_weakref` is instance returned by `weakref.ref`, 
316          `str(obj_weakref)` is buggy if the original obj overrides __getattr__, e.g: 
317   
318              class MyConfig(dict): 
319                  def __getattr__(self, x): 
320                      return self[x] 
321   
322              obj = MyConfig(offset=5) 
323              obj_weakref = weakref.ref(obj) 
324              str(obj_weakref)  # raise error: KeyError: '__name__' 
325          """
326          if isinstance(obj_weakref, weakref.ReferenceType):
327              obj = obj_weakref()
328              if obj is not None:
329                  return f"<weakref at {hex(id(obj_weakref))}; to '{obj.__class__.__name__}' at {hex(id(obj))}>"
330              else:
331                  return f"<weakref at {hex(id(obj_weakref))}; dead>"
332          else:
333              return str(obj_weakref)
334  
335      def __repr__(self):
336          s = f"""
337          {self.source.name.lower() if self.source else ""} {repr(self.name)} {self.inner_create_fn().__name__}
338          {{
339              'guard_types': {self.guard_types},
340              'code': {self.code_list},
341              'obj_weakref': {self.weakref_to_str(self.obj_weakref)}
342              'guarded_class': {self.guarded_class_weakref}
343          }}
344          """
345          return s
346  
347      def __str__(self):
348          output = f"Name: {repr(self.name)}\n"
349          source = self.source.name.lower() if self.source else ""
350          output += f"    Source: {source}\n"
351          output += f"    Create Function: {self.inner_create_fn().__name__}\n"
352          output += f"    Guard Types: {self.guard_types}\n"
353          output += f"    Code List: {self.code_list}\n"
354          output += f"    Object Weakref: {self.weakref_to_str(self.obj_weakref)}\n"
355          output += f"    Guarded Class Weakref: {self.guarded_class_weakref}\n"
356          return output
357  
358      def create(self, builder: GuardBuilderBase):
359          try:
360              return self.create_fn(builder, self)
361          except Exception:
362              log.exception("Error while creating guard:\n%s", str(self).rstrip())
363              if self.stack:
364                  log.error("Created at:\n%s", "".join(self.stack.format()[-4:]).rstrip())
365              raise
366  
367      def is_specialized_nn_module(self):
368          return self.source.is_specialized_nn_module()
369  
370      def is_fsdp_module(self):
371          return self.source.is_fsdp_module()
372  
373      def is_local(self):
374          return self.source.is_local()
375  
376      def set_export_info(self, guard_type, guarded_class, code_list, obj_weakref):
377          if not self.guard_types:
378              self.guard_types = []
379  
380          self.guard_types.append(guard_type)
381  
382          assert self.guarded_class_weakref in (
383              guarded_class,
384              None,
385          ), "Guarded class id must be identical, or None"
386          self.guarded_class_weakref = guarded_class
387  
388          if not self.code_list:
389              self.code_list = code_list
390          else:
391              self.code_list.extend(code_list)
392  
393          # Some objects are ephemeral, e.g., list[slice(1, 2)]. If we have
394          # multiple guards on the same object, the weakref can die between the
395          # invocation of set_export_info calls. So a dead weakref is also
396          # acceptable.
397          assert (
398              self.obj_weakref in (obj_weakref, None)
399              or callable(self.obj_weakref)
400              and self.obj_weakref() is None
401          ), "Guarded object must be identical, None or ephemeral (dead weakref)"
402          self.obj_weakref = obj_weakref

455  """ 
456  Checkpointable is an interface for driving state snapshotting, left purposely vague for now. 
457   
458  copy_graphstate() -> T, a somewhat legacy name, is expected to emit a snapshot of any type that 
459  can also be taken in at restore_graphstate(T) calls. 
460   
461  When to snapshot, is, at the moment, an implementation detail of upstream callers. Checkpointable 
462  does not provide any garuantees around consistency, idempotency, or safety of calling its APIs, yet. 
463   
464  In the future, it will have a closer coupling to a generic Checkpoint management system. 
465  """
466  
467  
468  class Checkpointable(Generic[T]):
469      @abstractmethod
470      def copy_graphstate(self) -> T: ...
471  
472      @abstractmethod
473      def restore_graphstate(self, state: T): ...

476  class GuardsCheckpointState:
477      """ 
478      The GuardCheckpointState - it is the T of Checkpointable[T] for GuardsContext 
479      """
480  
481      dynamo_guards: set[Guard] = set()
482  
483      def __init__(self, dynamo_guards):
484          self.dynamo_guards = dynamo_guards
485  
486      def diff(self, other):
487          """ 
488          Produces a delta against another GuardsCheckpointState. 
489   
490          Returns None if no delta is found, otherwise, return a set() of mismatched 
491          Guard type objects. 
492          """
493          r = self.dynamo_guards.difference(other.dynamo_guards)
494          if len(r) == 0:
495              return None
496          return r
497  
498      def __eq__(self, other):
499          return self.diff(other) is None

590  # Like a Set[Guard] but will record the user stack on all guards at the
591  # time they were installed at their destination
592  class GuardsSet:
593      def __init__(self, inner=None):
594          if inner is None:
595              inner = set()
596          self.inner = inner
597  
598      def __iter__(self):
599          return iter(self.inner)
600  
601      def __len__(self):
602          return len(self.inner)
603  
604      # Subtraction along with bool is typically used to determine the delta of
605      # added guards between checkpoints for higher order ops
606      def __sub__(self, other):
607          return GuardsSet(self.inner - other.inner)
608  
609      def __bool__(self):
610          return bool(self.inner)
611  
612      def add(self, guard: Guard, *, collect_debug_stack=True, skip=0):
613          if guard in self.inner:
614              return
615          if collect_debug_stack:
616              if guard.stack is None:
617                  guard.stack = CapturedTraceback.extract(skip=1 + skip)
618              if guard.user_stack is None:
619                  guard.user_stack = TracingContext.extract_stack()
620          self.inner.add(guard)
621  
622      def update(self, *others: set[Guard]):
623          for o in others:
624              for g in o:
625                  self.add(g, skip=1)
626  
627      def remove_guards_with_source(self, source):
628          """Delete all guards that contains a given source"""
629          from ._dynamo.source import is_from_source
630  
631          self.inner = {
632              g for g in self.inner if not is_from_source(g.originating_source, source)
633          }

753  """ 
754  TracingContext is the source of truth for all currently accumulated information 
755  needed to trace. Its lifecycle is kept 1:1 when using TorchDynamo, but other systems 
756  are open to managing their own TracingContext with that in mind. 
757   
758  The purpose of TracingContext is not to be a dumping ground, or god object, but rather to avoid 
759  having to plumb complex subsystems across multiple verticals. 
760   
761  Ex: A common example is guard accumulation between dynamo, shape_env, aot_autograd, and inductor. 
762  Accessing the current tracing context via 
763  TracingContext.get() allows users to accumulate their own guards for processing, without needing to know how 
764  to plumb objects back up to where frame interpretation happened. 
765   
766  Note that you can end up with multiple TracingContext for a single compilation 
767  of a frame, as we reset the TracingContext whenever we restart analysis. 
768  CompileContext is a more overarching context that encompasses multiple restarts. 
769  """

807      """ 
808      Provides the currently installed TracingContext, or None. 
809   
810      Note that it is a staticmethod, and invocations outside of `with tracing()` (see below), are valid but 
811      will return None.

814      @staticmethod
815      def try_get() -> Optional[TracingContext]:
816          return getattr(_TLS, "tracing_context", None)

818      @staticmethod
819      def get() -> TracingContext:
820          if ctx := TracingContext.try_get():
821              return ctx
822          raise RuntimeError(
823              "TracingContext.get() must be called within an ongoing trace."

826      def __init__(self, fake_mode):
827          self.guards_context = GuardsContext()

832          self.fake_mode = fake_mode
833          self.frame_summary_stack = []
834          # This is morally part of frame_summary_stack, but it is kept separate
835          # for clarity.  As we process a frame, this variable gets updated
836          # to keep track of what line we are in the function.  We make a
837          # function call, this gets cleared and the frame location is pushed
838          # to frame_summary_stack (prepping this variable for the inner frame's
839          # progress)
840          self.loc_in_frame = None
841          # this is only set after aot_autograd
842          self.fw_metadata = None
843          # this is only set after aot_autograd
844          self.aot_graph_name = None
845          self.params_flat = None
846          self.params_flat_unwrap_subclasses = None
847          self.params_unwrapped_to_flat_index = None
848          # this is for extended return calling convention from backend
849          # compiler to aot_autograd
850          # Per output, what the compiler specified stride of the output is,
851          # or None if no stride is known.  This is always the HINT, it
852          # is never a SymInt (it would be better if it was a SymInt, but
853          # I can't conveniently get this from Inductor atm.  Also, be
854          # careful not to accidentally induce guards on the SymInt if
855          # you ever do change this in aot_autograd.py; you should check
856          # on permutations preferentially.)
857          self.output_strides: Optional[list[Optional[tuple[int, ...]]]] = None
858          # When this is True, whenever we encounter an int in Dynamo tracing,
859          # we will (1) force unspec it and (2) force it as a size-like unbacked
860          # integer.  This is currently used when processing certain lists of
861          # ints that are known to be size-like and may have 0/1 entries that we
862          # must not specialize on.

865          self.tensor_to_context = WeakTensorKeyDictionary()

867          # If this true, Aot Autograd will return output Fake Tensors with appropiate
868          # meta on the first invocation
869          # see note: [Returning Fake Tensors on First AOT Autograd Call]
870          self.fakify_first_call = False
871          self.hop_dispatch_set_cache = HopDispatchSetCache()
872          # list of code objects for inlined functions
873          self.traced_code: list[CodeType] = []

875      def clear(self):

877          # for the context on clearing global context.
878          self.global_context.global_state = {}
879          self.previously_inlined_functions.clear()
880          self.previously_cleaned_instructions.clear()

882      @staticmethod
883      @contextmanager
884      def patch(**kwargs):
885          prior = {}
886          ctx = TracingContext.get()
887  
888          for key in kwargs.keys():
889              # KeyError on invalid entry
890              prior[key] = getattr(ctx, key)
891          for key, val in kwargs.items():
892              setattr(ctx, key, val)
893          try:
894              yield
895          finally:
896              for key, val in prior.items():
897                  setattr(ctx, key, val)

899      @staticmethod
900      def extract_stack():
901          self = TracingContext.try_get()
902          if self is None:
903              return traceback.StackSummary()
904          stack = self.frame_summary_stack
905          if self.loc_in_frame is not None:
906              stack = stack + [self._populate_loc_in_frame_summary()]
907          return traceback.StackSummary.from_list(stack)
908  
909      def _populate_loc_in_frame_summary(self):
910          assert self.loc_in_frame is not None
911          filename, lineno, frame_name = self.loc_in_frame
912          return traceback.FrameSummary(filename, lineno, frame_name, lookup_line=False)

914      # Call this when you want to call into some code that isn't necessarily
915      # associated with the current frame state
916      @staticmethod
917      @contextlib.contextmanager
918      def clear_frame():
919          tc = TracingContext.get()
920          with (
921              unittest.mock.patch.object(tc, "frame_summary_stack", []),
922              unittest.mock.patch.object(tc, "loc_in_frame", None),
923          ):
924              try:
925                  yield
926              except Exception as e:
927                  # Prevent real_stack from getting attached
928                  #
929                  # The invariant is that if an Exception as real_stack, we've
930                  # appropriately attached a user stack and we no longer need to
931                  # attach anything. Because we cannot conveniently interpose
932                  # when an exception is thrown, we instead interpose everywhere
933                  # we set what the user stack is set (using the context
934                  # manager). However, our compiler stack does "tail calls"
935                  # (when it calls into user compiler), at which point the
936                  # parent exception frames would incorrectly attach an
937                  # incorrect frame.
938                  #
939                  # However, if, somehow, someone raised an exception with this
940                  # scope that had a stack (for example, because they are
941                  # restoring the user stack state appropriately as they process
942                  # node by node), we should respect it. Thus, we cannot
943                  # unconditionally set None.
944                  if not hasattr(e, "real_stack"):
945                      e.real_stack = None  # type: ignore[attr-defined]
946                  raise

948      @staticmethod
949      @contextlib.contextmanager
950      def current_frame(frame_summary):
951          # frame_summary can be None to solely take advantage of real_stack
952          # attachment to thrown exceptions
953          tc = TracingContext.get()
954          if frame_summary is not None:
955              tc.frame_summary_stack.append(frame_summary)
956          old = tc.loc_in_frame
957          tc.loc_in_frame = None
958          try:
959              yield
960          except Exception as e:
961              if not hasattr(e, "real_stack"):
962                  e.real_stack = tc.extract_stack()  # type: ignore[attr-defined]
963              raise
964          finally:
965              if frame_summary is not None:
966                  tc.frame_summary_stack.pop()
967              tc.loc_in_frame = old

972          tc = TracingContext.try_get()
973          if tc is None:
974              yield None
975              return
976          old_output_strides = tc.output_strides
977          tc.output_strides = []
978          try:

1009     """ 
1010     This function installs the passed in tracing context as a dynamic scoped 
1011     global variable. 
1012  
1013     Calls to TracingContext.get() while not under a `with tracing()` context

1019         yield context

1022             e.real_stack = context.extract_stack()  # type: ignore[attr-defined]

1030             context.fake_mode.shape_env.cleanup()
1031         _TLS.tracing_context = old_context

1085 def detect_fake_mode(inputs: Any = None):
1086     """ 
1087     Attempts to "detect" what the current fake mode is.  If there is one ambiently 
1088     available from TracingContext, we preferentially use that.  Otherwise, we 
1089     heuristically detect the fake mode via the following sources, in order of 
1090     priority: 
1091  
1092         - Currently active fake mode on stack 
1093         - Fake mode associated with passed in tensors (inputs does not 
1094           have to be flattened) 
1095     """
1096     from torch._subclasses.fake_tensor import FakeTensor, FakeTensorMode
1097 
1098     fake_modes = []
1099 
1100     if context := TracingContext.try_get():
1101         fake_mode = context.fake_mode
1102         if fake_mode is not None:
1103             fake_modes.append((fake_mode, "tracing context", 0))
1104 
1105     from torch.utils._python_dispatch import _get_current_dispatch_mode_stack
1106 
1107     for i, m in enumerate(reversed(_get_current_dispatch_mode_stack())):
1108         if isinstance(m, FakeTensorMode):
1109             fake_modes.append((m, "active fake mode", i))
1110 
1111     flat_inputs = pytree.tree_leaves(inputs)
1112     for i, flat_input in enumerate(flat_inputs):
1113         if isinstance(flat_input, FakeTensor):
1114             fake_modes.append((flat_input.fake_mode, "fake tensor input", i))
1115 
1116     if fake_modes:
1117         fake_mode, desc1, i1 = fake_modes[0]
1118         for m, desc2, i2 in fake_modes[1:]:
1119             assert fake_mode is m, (
1120                 f"fake mode ({fake_mode}) from {desc1} {i1} doesn't match mode ({m}) from {desc2} {i2}\n\n"
1121                 f"fake mode from {desc1} {i1} allocated at:\n{fake_mode.stack}\n"
1122                 f"fake mode from {desc2} {i2} allocated at:\n{m.stack}"
1123             )
1124         return fake_mode
1125     else:
1126         return None

1129 def active_fake_mode():
1130     """ 
1131     Inspects the dispatch mode stack for an active fake mode and returns it. 
1132     Returns None if no fake mode is active. 
1133     """
1134     from torch._subclasses.fake_tensor import FakeTensorMode
1135     from torch.utils._python_dispatch import _get_current_dispatch_mode_stack
