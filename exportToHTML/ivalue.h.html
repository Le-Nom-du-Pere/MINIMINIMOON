65   // We need a ComplexHolder because currently the payloads in the Union
66   // only take 64 bits. Since ComplexDouble takes up 128 bits, and is too big
67   // to fit in the IValue directly, we indirect complex numbers through an
68   // intrusive pointer to ComplexHolder (which contains a c10::complex).

89   // This is an owning wrapper for a std::optional<std::vector<T>>
90   // that can be implicitly converted to a (non-owning) std::optional<ArrayRef<T>>.
91   // Its purpose is to be used in generated code to keep the vector alive
92   // either until the end of a statement (as a temporary), or as a saved arg
93   // in autograd.
94   template <typename T>
95   struct OptionalArray {
96     std::optional<std::vector<T>> list;
97   
98     OptionalArray() = default;
99     OptionalArray(std::vector<T> val) : list(std::move(val)) {}
100  
101    // Used when saving an argument for the backwards pass.
102    OptionalArray& operator=(std::optional<ArrayRef<T>> ref) {
103      if (ref) {
104        list = std::vector<T>(ref->begin(), ref->end());
105      } else {
106        list = std::nullopt;
107      }
108      return *this;
109    }
110  
111    // Used when saving an argument for the backwards pass.
112    OptionalArray& operator=(c10::OptionalArrayRef<T> ref) {
113      if (ref) {
114        list = std::vector<T>(ref->begin(), ref->end());
115      } else {
116        list = std::nullopt;
117      }
118      return *this;
119    }
120  
121    operator std::optional<c10::ArrayRef<T>>() {
122      if (!list) {
123        return std::nullopt;
124      }
125      return *list;
126    }
127  
128    operator c10::OptionalArrayRef<T>() {
129      if (!list) {
130        return std::nullopt;
131      }
132      return *list;

190  /// IValue (Interpreter Value) is a tagged union over the types
191  /// supported by the TorchScript interpreter. IValues contain their
192  /// values as an `IValue::Payload`, which holds primitive types
193  /// (`int64_t`, `bool`, `double`, `Device`) and `Tensor` as values,
194  /// and all other types as a `c10::intrusive_ptr`. In order to
195  /// optimize performance of the destructor and related operations by
196  /// making the `Tensor` and `c10::intrusive_ptr` paths generate the
197  /// same code, we represent a null `c10::intrusive_ptr` as
198  /// `UndefinedTensorImpl::singleton()`, *not* `nullptr`.
199  ///
200  /// IValues are used as inputs to and outputs from the TorchScript interpreter.
201  /// To retrieve the value contained within an IValue, use the `.toX()` methods,
202  /// where `X` is the type you are trying to get. Note that neither the `.toX()`
203  /// methods nor the templated `.to<T>` functions do any kind of casting, they
204  /// only unwrap the contained value. For example:
205  ///
206  /// \rst
207  /// .. code-block:: cpp
208  ///
209  ///   // Make the IValue
210  ///   torch::IValue my_ivalue(26);
211  ///   std::cout << my_ivalue << "\n";
212  ///
213  ///   // Unwrap the IValue
214  ///   int64_t my_int = my_ivalue.toInt();
215  ///   std::cout << my_int << "\n";
216  ///
217  ///   // This will throw an error!
218  ///   // `my_ivalue` is tagged as an int and cannot be used as another type
219  ///   torch::Tensor my_tensor = my_ivalue.toTensor();
220  /// \endrst
221  struct TORCH_API IValue final {
222    IValue(const IValue& rhs) : IValue(rhs.payload, rhs.tag) {
223      if (isIntrusivePtr() &&
224          payload.u.as_intrusive_ptr != c10::UndefinedTensorImpl::singleton()) {
225        c10::raw::intrusive_ptr::incref(payload.u.as_intrusive_ptr);
226      }
227    }
228  
229    IValue(IValue&& rhs) noexcept : tag(rhs.tag) {
230      moveFrom(std::move(rhs));
231    }

234    ~IValue() {
235      destroy();
236    }

255    /** 
256     * Equality comparison. The semantics are the same as Python's `==`: 
257     * 1. Numerical types are compared by value. 
258     * 2. Tensors compute element-wise equality, returning a BoolTensor (see: 
259     * `torch.eq()`) 
260     * 3. Strings are compared by value. 
261     * 4. Sequence types (list, tuple) are compared lexicographically by 
262     *    comparing their elements. Different sequence types never compare equal. 
263     * 5. Mappings (dict) must have equal (key, value) pairs. 
264     * 6. If not listed above, the default behavior for is to test identity 
265     * equality (e.g. pointer equality). 
266     * 
267     * Why does this return an IValue instead of a bool? Because in PyTorch, 
268     * `tensor1 == tensor2` returns a `BoolTensor`, not a bool. 
269     * 
270     * NOTE: we (like Python) assume that identity equality implies value equality 
271     * for efficiency. 
272     * TODO: need to support customizing equality 
273     */
274    IValue equals(const IValue& rhs) const;
275    /** 
276     * This implements the same semantics as `bool(lhs == rhs)` in Python. which 
277     * is the same as `equals()` except for Tensor types. 
278     */
279    TORCH_API friend bool operator==(const IValue& lhs, const IValue& rhs);
280    TORCH_API friend bool operator!=(const IValue& lhs, const IValue& rhs);

283     * Identity comparison. Checks if `this` is the same object as `rhs`. The 
284     * semantics are the same as Python's `is` operator. 
285     * 
286     * NOTE: Like in Python, this operation is poorly defined for primitive types 
287     * like numbers and strings. Prefer to use `==` unless you really want to 
288     * check identity equality.

326    static bool isAliasOf(const at::Tensor& a, const at::Tensor& b) {
327      if (a.is_sparse()) {
328        return isAliasOf(a._values(), b) || isAliasOf(a._indices(), b);
329      }
330      if (b.is_sparse()) {
331        return isAliasOf(a, b._values()) || isAliasOf(a, b._indices());
332      }
333      if (a.is_sparse_csr()) {
334        return isAliasOf(a.values(), b) || isAliasOf(a.crow_indices(), b) ||
335            isAliasOf(a.col_indices(), b);
336      }
337      if (b.is_sparse_csr()) {
338        return isAliasOf(a, b.values()) || isAliasOf(a, b.crow_indices()) ||
339            isAliasOf(a, b.col_indices());
340      }
341  
342      // Opaque tensors such as the ones constructed by the MKL-DNN backend
343      // don't have storage so we just compare their TensorImpls.
344      // TODO: Find way to expose alias info for opaque tensors.
345      if (!a.has_storage() || !b.has_storage()) {
346        return a.unsafeGetTensorImpl() == b.unsafeGetTensorImpl();
347      }
348  
349      return a.is_alias_of(b);
350    }

356    /// @private [doxygen private]
357    bool isAliasOf(const IValue& rhs) const {
358      if (this->tag != rhs.tag) {
359        // Trivially don't alias if the type is different
360        return false;
361      }
362  
363      // Tensors should be compared based on internal storage
364      if (this->isTensor()) {
365        return isAliasOf(this->toTensor(), rhs.toTensor());
366      }
367  
368      if (!isIntrusivePtr()) {
369        // Primitive types don't alias anything
370        return false;
371      }
372  
373      AT_ASSERT(rhs.isIntrusivePtr());
374  
375      // Other types can be compared by their ptr value
376      return this->payload.u.as_intrusive_ptr == rhs.payload.u.as_intrusive_ptr;
377    }

380    size_t use_count() const noexcept {
381      if (isTensor()) {
382        return payload.as_tensor.use_count();
383      }
384  
385      if (!isIntrusivePtrLegacyBehavior()) {
386        return 1;
387      }
388  
389      if (payload.u.as_intrusive_ptr == c10::UndefinedTensorImpl::singleton()) {
390        return 0;
391      }
392      return c10::raw::intrusive_ptr::use_count(payload.u.as_intrusive_ptr);
393    }

395    /// @private [doxygen private]
396    void swap(IValue& rhs) noexcept {
397      if (isTensor() && rhs.isTensor()) {
398        std::swap(payload.as_tensor, rhs.payload.as_tensor);
399      } else if (isTensor()) {
400        at::Tensor t = std::move(payload.as_tensor);
401        // As far as I can tell, omitting the usual explicit destructor call
402        // is not UB in and of itself, and it's a slight perf win. The
403        // destructor is a no-op, because the moved-from Tensor is
404        // effectively an intrusive_ptr in the null state, so we don't need
405        // the behavior for correctness reasons either. Leaving this
406        // explanatory comment, including commented-out destructor call, to
407        // make this abundantly clear.
408        //
409        // payload.as_tensor.~Tensor();
410        payload.u = rhs.payload.u;
411        new (&rhs.payload.as_tensor) at::Tensor(std::move(t));
412      } else if (rhs.isTensor()) {
413        rhs.swap(*this);
414        return;
415      } else {
416        std::swap(payload.u, rhs.payload.u);
417      }
418      std::swap(tag, rhs.tag);
419    }

872    // Scalar, which gets encoded as either an Int, a Double or a ComplexDouble
873    IValue(const at::Scalar& s) : IValue() {
874      // NB: do the symbolic versions first, as isFloatingPoint is true
875      // for both SymFloat and double
876      if (s.isSymInt()) {
877        tag = Tag::SymInt;
878        payload.u.as_intrusive_ptr = s.toSymInt().toSymNode().release();
879      } else if (s.isSymFloat()) {
880        tag = Tag::SymFloat;
881        payload.u.as_intrusive_ptr = s.toSymFloat().toSymNodeImpl().release();
882      } else if (s.isSymBool()) {
883        tag = Tag::SymBool;
884        payload.u.as_intrusive_ptr = s.toSymBool().toSymNodeImpl().release();
885      } else if (s.isFloatingPoint()) {
886        tag = Tag::Double;
887        payload.u.as_double = s.toDouble();
888      } else if (s.isComplex()) {
889        *this = s.toComplexDouble();
890      } else if (s.isBoolean()) {
891        tag = Tag::Bool;
892        payload.u.as_bool = s.toBool();
893      } else {
894        TORCH_INTERNAL_ASSERT_DEBUG_ONLY(
895            s.isIntegral(false), "Unknown type in Scalar");
896        tag = Tag::Int;
897        payload.u.as_int = s.toLong();
898      }
899    }
900  
901    bool isScalar() const {
902      return isDouble() || isInt() || isComplexDouble() || isBool() ||
903          isSymInt() || isSymFloat() || isSymBool();
904    }

910        return toInt();

912        return toComplexDouble();
913      else if (isBool())
914        return toBool();
915      else if (isSymInt())
916        return toSymInt();
917      else if (isSymFloat())

919      else if (isSymBool())

921      TORCH_CHECK(false, "IValue is not a Scalar");

1079   // Detect aliased tensors.
1080   struct HashAliasedIValue {
1081     size_t hashTensor(const at::Tensor& ten) const {
1082       if (ten.is_sparse()) {
1083         // COO sparse tensors have a "values" tensor and an "indices" tensor
1084         // so this will detect overlap of sparse tensors that share a values
1085         // tensor, but not sparse tensors that share an indices tensor.
1086         return hashTensor(ten._values());
1087       } else if (ten.is_sparse_csr()) {
1088         // COO sparse tensors have a "values" tensor and an "indices" tensor
1089         // so this will detect overlap of sparse tensors that share a values
1090         // tensor, but not sparse tensors that share an indices tensor.
1091         return hashTensor(ten.values());
1092       } else if (!ten.has_storage()) {
1093         // Opaque tensors such as the ones constructed by the MKL-DNN backend
1094         // don't have storage so we just use their TensorImpls.
1095         // TODO: Find way to expose alias info for opaque tensors.
1096         return reinterpret_cast<size_t>(ten.unsafeGetTensorImpl());
1097       } else {
1098         return reinterpret_cast<size_t>(ten.storage().unsafeGetStorageImpl());
1099       }
1100     }
1101     size_t operator()(const IValue& val) const {
1102       if (val.isTensor()) {
1103         return hashTensor(val.toTensor());
1104       }
1105       // If it is not a Tensor, then two mutable IValues alias each other only
1106       // if they are the same pointer.
1107       return val.payload.u.as_int;
1108     }
1109   };

1189   void destroy() {
1190     // We carefully construct this call to both 1) avoid UB by using
1191     // the "wrong" one of as_tensor and as_intrusive_ptr and 2) enable
1192     // the compiler to generate the same code for each case. It is
1193     // surprisingly difficult to get this right.
1194     if (isTensor() || isIntrusivePtr()) {
1195       c10::intrusive_ptr_target* p = isTensor()
1196           ? payload.as_tensor.unsafeGetTensorImpl()
1197           : payload.u.as_intrusive_ptr;
1198       c10::intrusive_ptr<intrusive_ptr_target, c10::UndefinedTensorImpl>::
1199           reclaim(p);
1200       // No need to make this destructor call!
1201       // payload.as_tensor.~Tensor();
1202     }
1203   }

1205   // NOLINTNEXTLINE(cppcoreguidelines-rvalue-reference-param-not-moved)
1206   C10_ALWAYS_INLINE void moveFrom(IValue&& rhs) noexcept {
1207     if (rhs.isTensor()) {
1208       new (&payload.as_tensor) at::Tensor(std::move(rhs.payload.as_tensor));
1209       // As far as I can tell, omitting the usual explicit destructor call
1210       // is not UB in and of itself, and it's a slight perf win. The
1211       // destructor is a no-op, because the moved-from Tensor is
1212       // effectively an intrusive_ptr in the null state, so we don't need
1213       // the behavior for correctness reasons either. Leaving this
1214       // explanatory comment, including commented-out destructor call, to
1215       // make this abundantly clear.
1216       //
1217       // rhs.payload.as_tensor.~Tensor();
1218     } else {
1219       payload.u = rhs.payload.u;
1220     }
1221     tag = rhs.tag;
1222     rhs.clearToNone();
1223   }

1295  public:
1296   // Don't edit this just to add results for new tags; edit
1297   // isIntrusivePtrConstexpr above.
1298   bool isIntrusivePtr() const {
1299     // Implementation NOTE: the switch in isIntrusivePtrConstexpr
1300     // above is the previous production implementation of this
1301     // function. We observed that, at least on x86_64, the generated
1302     // instruction sequence was a similar bit vector test to what we
1303     // have manually implemented below, except that there was an extra
1304     // "bounds check" branch confirming, essentially, that `tag <
1305     // kNumTags` and providing a consistent result in that case. We
1306     // don't care about the result if tag is out of bounds, so we'd
1307     // like to eliminate that comparison and branch; manually
1308     // implementing this function as a bit test is the simplest way I
1309     // could find to accomplish that elimination.
1310     static constexpr uint32_t kTruthTableBitVector =
1311 #define TRUTH_TABLE_ENTRY(tag) \
1312   (uint32_t(isIntrusivePtrConstexpr(Tag::tag)) << uint32_t(Tag::tag)) |
1313         TORCH_FORALL_TAGS(TRUTH_TABLE_ENTRY)
1314 #undef TRUTH_TABLE_ENTRY
1315             0;
1316 
1317     TORCH_INTERNAL_ASSERT_DEBUG_ONLY(
1318         static_cast<uint32_t>(tag) < kNumTags,
1319         "unexpected tag ",
1320         static_cast<int>(tag));
1321     return kTruthTableBitVector & (1 << (uint32_t(tag) % 32));
1322   }

1324   // Storage and Generator were treated specially when
1325   // is_intrusive_ptr was stored as explicit state. This getter
1326   // preserves the old behavior for use with WeakIValue for now.
1327   bool isIntrusivePtrLegacyBehavior() const {
1328     if (tag == Tag::Storage || tag == Tag::Generator) {
1329       return payload.u.as_intrusive_ptr !=
1330           c10::UndefinedTensorImpl::singleton();
1331     } else {
1332       return isIntrusivePtr();
1333     }
1334   }

1336   union Payload {
1337     // [TriviallyCopyablePayload]
1338     // We use a nested union here so that we can make the copy easy
1339     // and efficient in the non-tensor (i.e., trivially copyable)
1340     // case. Specifically, we do not have to do a switch-on-tag to
1341     // figure out which union member to assign; we can just use
1342     // TriviallyCopyablePayload::operator=.
1343     union TriviallyCopyablePayload {
1344       TriviallyCopyablePayload() : as_int(0) {}
1345       int64_t as_int;
1346       double as_double;
1347       bool as_bool;
1348       // Invariant: never nullptr; null state is represented as
1349       // c10::UndefinedTensorImpl::singleton() for consistency of
1350       // representation with Tensor.
1351       c10::intrusive_ptr_target* as_intrusive_ptr;
1352       struct {
1353         c10::DeviceType type;
1354         DeviceIndex index;
1355       } as_device;
1356     } u;
1357     static_assert(std::is_trivially_copyable_v<TriviallyCopyablePayload>);
1358     at::Tensor as_tensor;
1359     Payload() : u() {}
1360     Payload(const Payload&) = delete;
1361     Payload(Payload&&) = delete;
1362     Payload& operator=(const Payload&) = delete;
1363     Payload& operator=(Payload&&) = delete;
1364     // NOLINTNEXTLINE(modernize-use-equals-default)
1365     ~Payload() {}
1366   };

1389   WeakIValue(const WeakIValue& rhs)
1390       : payload(rhs.payload),
1391         tag(rhs.tag),
1392         is_intrusive_ptr(rhs.is_intrusive_ptr) {
1393     if (is_intrusive_ptr &&

1398   WeakIValue(const IValue& rhs)
1399       : tag(rhs.tag), is_intrusive_ptr(rhs.isIntrusivePtrLegacyBehavior()) {
1400     if (rhs.isTensor()) {
1401       payload.as_intrusive_ptr = rhs.unsafeToTensorImpl();
1402       is_intrusive_ptr = true;
1403     } else {
1404       payload = rhs.payload.u;
1405     }
1406     if (is_intrusive_ptr) {
1407       if (payload.as_intrusive_ptr != c10::UndefinedTensorImpl::singleton()) {
1408         c10::raw::weak_intrusive_ptr::incref(payload.as_intrusive_ptr);
1409       }
1410     }
1411   }
1412   WeakIValue(WeakIValue&& rhs) noexcept : WeakIValue() {
1413     swap(rhs);
1414   }
1415   ~WeakIValue() {
1416     if (is_intrusive_ptr &&
1417         payload.as_intrusive_ptr != c10::UndefinedTensorImpl::singleton()) {
1418       c10::raw::weak_intrusive_ptr::decref(payload.as_intrusive_ptr);
1419     }
1420   }
1421   WeakIValue& operator=(WeakIValue&& rhs) & noexcept {
1422     WeakIValue(std::move(rhs)).swap(*this); // this also sets rhs to None
1423     return *this;
1424   }

1426     WeakIValue(rhs).swap(*this);
1427     return *this;
1428   }
1429   void swap(WeakIValue& rhs) noexcept {
1430     std::swap(payload, rhs.payload);
1431     std::swap(is_intrusive_ptr, rhs.is_intrusive_ptr);
1432     std::swap(tag, rhs.tag);
1433   }

1435   bool isSameIdentity(const WeakIValue& rhs) const {
1436     return payload.as_int == rhs.payload.as_int && tag == rhs.tag &&
1437         is_intrusive_ptr == rhs.is_intrusive_ptr;
1438   }

1440   IValue lock() const {
1441     if (!is_intrusive_ptr) {
1442       IValue::Payload newPayload;
1443       newPayload.u = payload;
1444       return IValue(newPayload, tag);
1445     }
1446     if (IValue::Tag::Tensor == tag) {
1447       auto temp =
1448           c10::weak_intrusive_ptr<at::TensorImpl, c10::UndefinedTensorImpl>::
1449               reclaim(static_cast<at::TensorImpl*>(payload.as_intrusive_ptr));
1450       c10::intrusive_ptr<at::TensorImpl, c10::UndefinedTensorImpl> ip(
1451           temp.lock());
1452       temp.release();
1453       if (!ip) {
1454         return IValue();
1455       } else {
1456         return IValue(at::Tensor(std::move(ip)));
1457       }
1458     } else {
1459       auto temp = c10::weak_intrusive_ptr<c10::intrusive_ptr_target>::reclaim(
1460           payload.as_intrusive_ptr == c10::UndefinedTensorImpl::singleton()
1461               ? nullptr
1462               : payload.as_intrusive_ptr);
1463       IValue::Payload pl;
1464       pl.u.as_intrusive_ptr = temp.lock().release();
1465       temp.release();
1466       if (!pl.u.as_intrusive_ptr) {
1467         return IValue();
1468       } else {
1469         return IValue(pl, tag);
1470       }
1471     }
1472   }

1474   size_t use_count() const noexcept {
1475     if (!is_intrusive_ptr) {
1476       return 1;
1477     }
1478     auto temp = c10::weak_intrusive_ptr<
1479         c10::intrusive_ptr_target,
1480         c10::UndefinedTensorImpl>::reclaim(payload.as_intrusive_ptr);
1481     size_t result = temp.use_count();
1482     temp.release();

1493     size_t result = temp.weak_use_count();
