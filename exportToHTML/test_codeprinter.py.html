44   def test_lambdify_LaTeX_symbols_issue_23374():
45       # Create symbols with Latex style names
46       x1, x2 = symbols("x_{1} x_2")
47   
48       # Lambdify the function
49       f1 = lambdify([x1, x2], cos(x1 ** 2 + x2 ** 2))
50   
51       # Test that the function works correctly (numerically)
52       assert f1(1, 2) == math_cos(1 ** 2 + 2 ** 2)
53   
54       # Explicitly generate a custom printer to verify the naming convention
55       p = LambdaPrinter()
56       expr_str = p.doprint(cos(x1 ** 2 + x2 ** 2))
57       assert 'x_1' in expr_str
58       assert 'x_2' in expr_str
