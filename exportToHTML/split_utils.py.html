17   @compatibility(is_backward_compatible=False)
18   def getattr_recursive(obj, name):
19       for layer in name.split("."):
20           if hasattr(obj, layer):
21               obj = getattr(obj, layer)
22           else:
23               return None
24       return obj
25   
26   
27   @compatibility(is_backward_compatible=False)
28   def setattr_recursive(obj, attr, value):
29       if "." not in attr:
30           setattr(obj, attr, value)
31       else:
32           layer = attr.split(".")
33           setattr_recursive(getattr(obj, layer[0]), ".".join(layer[1:]), value)

37   @dataclass
38   class Component:
39       """ 
40       A component serves as a container for a subgraph we want to create afterwards. 
41       """
42   
43       graph: torch.fx.Graph
44       order: int
45       name: str
46   
47       # Stores the placeholder nodes in `graph`.
48       input_placeholders: list = field(default_factory=list)
49   
50       # Store the nodes in original graph that are placeholder in `graph`.
51       orig_inputs: list = field(default_factory=list)
52   
53       # Store the nodes in original graph that are outputs in `graph`.
54       orig_outputs: list = field(default_factory=list)

56       # Mapping from get_attr node in original graph to get_attr node in `graph`.
57       getattr_maps: dict[torch.fx.Node, torch.fx.Node] = field(default_factory=dict)

62   @compatibility(is_backward_compatible=False)
63   def split_by_tags(
64       gm: torch.fx.GraphModule,
65       tags: list[str],
66       return_fqn_mapping: bool = False,
67       return_tuple: bool = False,
68       GraphModuleCls: type[torch.fx.GraphModule] = torch.fx.GraphModule,
69   ) -> Union[torch.fx.GraphModule, tuple[torch.fx.GraphModule, dict[str, str]]]:
70       """ 
71       Splits a GraphModule using tags on its graph nodes. We honor the order of 
72       tags. For example, we have tags = ["a", "b", "c"], the function will create 
73       the initial submodules in the order of "a", "b", "c". 
74    
75       To set a tag: 
76       gm.graph.nodes[idx].tag = "mytag" 
77    
78       This will result in all nodes with the same tag being extracted and placed in their 
79       own submodule. For placeholder, output and get_attr node, the tag is ignored. placeholder 
80       and output nodes are created when needed while get_attr nodes get copied to submodules 
81       where they are used. 
82    
83       Given the following module def: 
84    
85       class SimpleModule(torch.nn.Module): 
86           def __init__(self) -> None: 
87               super().__init__() 
88               self.linear1 = torch.nn.Linear(...) 
89               self.linear2 = torch.nn.Linear(...) 
90               self.linear3 = torch.nn.Linear(...) 
91    
92           def forward(self, in1, in2): 
93               r1 = self.linear1(in1) 
94               r2 = self.linear2(in2) 
95               r3 = torch.cat([r1, r2]) 
96               return self.linear3(r3) 
97    
98       Marking the node corresponding to in1 with the tag sc.REQUEST_ONLY.lower() results in the following split: 
99    
100      ro: 
101      def forward(self, in1): 
102          self = self.root 
103          linear1 = self.linear1(in1) 
104          return linear1 
105   
106      main: 
107      def forward(self, in2, linear1): 
108          self = self.root 
109          linear2 = self.linear2(in2) 
110          cat_1 = torch.cat([linear1, linear2]) 
111          linear3 = self.linear3(cat_1) 
112          return linear3 
113   
114      main: 
115      def forward(self, in1, in2): 
116          self = self.root 
117          ro_0 = self.ro_0(in1) 
118          main_1 = self.main_1(in2, ro_0) 
119          return main_1 
120   
121      Returns: 
122          split_gm: torch fx graph after split 
123          orig_to_split_fqn_mapping: a map between the original fqn and the fqn 
124              after split for call_module and get_attr. 
125      """
126  
127      def flatten(x: torch.fx.node.Argument) -> NodeList:
128          """ 
129          Stores nodes in x to a list and returns the list. 
130          """
131          r: NodeList = []
132          map_arg(x, r.append)
133          return r
134  
135      # Mapping from node in original module to node in created submodule.
136      node_remapping: dict[torch.fx.Node, torch.fx.Node] = {}
137  
138      # Mapping from node in original module or created submodules to
139      # corresponding component.
140      node_to_component: dict[torch.fx.Node, Component] = {}
141  
142      # Mapping from tag to the corresponding component.
143      tag_to_component: dict[str, Component] = {}
144  
145      # Stores all components.
146      all_components: list[Component] = []
147  
148      # Stores nodes that will be used in main graph.
149      used_in_main: dict[torch.fx.Node, None] = {}
150  
151      # Main graph after split.
152      main_g = torch.fx.Graph()
153  
154      # Mapping from node in original module to node in main graph after split.
155      main_remapping: dict[torch.fx.Node, torch.fx.Node] = {}
156  
157      # Output node of original module.
158      output_node: Optional[torch.fx.Node] = None
159  
160      # Create a component for each tag, we don't expect to create other components afterwards.
161      for tag in tags:
162          comp = Component(torch.fx.Graph(), len(all_components), f"{tag}")
163          all_components.append(comp)
164          tag_to_component[tag] = comp
165  
166      # Traverse the nodes in original graph and take care of them.
167      for node in gm.graph.nodes:
168          if node.op == "output":
169              if output_node is not None:
170                  raise RuntimeError("Multiple output nodes in graph!")
171              output_node = node
172              continue
173  
174          # Placeholders in the original graph get copied to main graph.
175          if node.op == "placeholder":
176              main_remapping[node] = main_g.placeholder(node.name, type_expr=node.type)
177              main_remapping[node].meta = copy.copy(node.meta)
178              continue
179  
180          # Get_attr nodes are ignored because we are not tagging them.
181          # Instead, we copy them directly to the submodules use them afterwards.
182          if node.op == "get_attr":
183              continue
184  
185          # Now we process callable nodes which are nodes with op of call_module,
186          # call_function or call_method. Every callable nodes should be tagged.
187          assert hasattr(node, "tag"), f"Node does not have tag: {node.format_node()}"
188  
189          upstream_components = [
190              node_to_component[x]
191              for x in flatten(node.args) + flatten(node.kwargs)
192              if x.op not in {"placeholder", "get_attr"}
193          ]
194  
195          comp = tag_to_component[node.tag]
196          node_to_component[node] = comp
197  
198          # Max order of upperstream components.
199          mx = max((c.order for c in upstream_components), default=0)
200  
201          # Expect the component for `node` has higher order then its upstream components.
202          assert comp.order >= mx, (
203              f"Component {comp.name} order must be >= max of its upstream components, order={comp.order} and max={mx}"
204          )
205  
206          # Map a input of `node` to nodes in the component's graph.
207          def remap_func(x):
208              # If input is a get_attr node, copy it to current component's graph.
209              # Returns the get_attr node in current component's graph.
210              if x.op == "get_attr":
211                  if x not in comp.getattr_maps:
212                      comp.getattr_maps[x] = comp.graph.get_attr(
213                          x.target, type_expr=x.type
214                      )
215                      comp.getattr_maps[x].meta = copy.copy(x.meta)
216                  return comp.getattr_maps[x]
217  
218              # If input is not a placeholder, it should have been put into a component
219              # already. If it's the current component then we return the corresponding
220              # node in the component.
221              if x.op != "placeholder" and node_to_component[x] == comp:
222                  return node_remapping[x]
223  
224              # If input is a placeholder or it's in other components, we want to make it
225              # as a placeholder in current component's graph.
226              if x not in comp.orig_inputs:
227                  comp.orig_inputs.append(x)
228                  placeholder = comp.graph.placeholder(x.name, type_expr=x.type)
229                  placeholder.meta = copy.copy(x.meta)
230                  comp.input_placeholders.append(placeholder)
231                  used_in_main[x] = None
232  
233              return comp.input_placeholders[comp.orig_inputs.index(x)]
234  
235          n = comp.graph.node_copy(node, remap_func)
236          n.tag = node.tag  # type: ignore[attr-defined]
237          node_remapping[node] = n
238          node_to_component[n] = comp
239  
240      if output_node is None:
241          raise RuntimeError("Graph had no output node!")
242  
243      for x in flatten(output_node.args[0]):
244          if x.op == "get_attr":
245              # We don't need components mapping for nodes of type "get_attr"
246              # that are consumed by the output. Only need to make sure we create
247              # corresponding counterparts in the resulting graph.
248              main_remapping[x] = main_g.get_attr(x.name, type_expr=x.type)
249          else:
250              # All component results consumed by the output node should be
251              # marked as "used in main".
252              used_in_main[x] = None
253  
254      # If a node is used in main graph then we mark it as an output in the component
255      # it belongs to.
256      for n in used_in_main:
257          if n.op != "placeholder":
258              node_to_component[n].orig_outputs.append(n)
259  
260      # Now we create a graphmodule for each component.
261      orig_to_split_fqn_mapping: dict[str, str] = {}
262      for comp in all_components:
263          outs = tuple(map(node_remapping.__getitem__, comp.orig_outputs))
264  
265          if return_tuple:
266              comp.graph.output(outs)
267          else:
268              # Take care of the args of FX output node. If there's a single
269              # output then the output node args is like (output_single), else
270              # if there're multiple outputs then the output node args is like
271              # ((output_0, output_1, ...)).
272              comp.graph.output(outs[0] if len(outs) == 1 else outs)
273  
274          comp.gm, comp_orig_to_split_fqn_mapping = lift_subgraph_as_module(
275              gm, subgraph=comp.graph, comp_name=comp.name
276          )
277          orig_to_split_fqn_mapping.update(comp_orig_to_split_fqn_mapping)
278  
279          # Create a call_module node in main graph.
280          main_node = main_g.call_module(
281              comp.name,
282              args=tuple(map(main_remapping.__getitem__, comp.orig_inputs)),
283              kwargs=None,
284          )
285  
286          if len(outs) == 1 and not return_tuple:
287              main_remapping[comp.orig_outputs[0]] = main_node
288          else:
289              for i, o in enumerate(comp.orig_outputs):
290                  # Use Proxy to record getitem access.
291                  main_remapping[o] = torch.fx.Proxy(main_node)[i].node  # type: ignore[index]
292  
293      main_g.output(map_arg(output_node.args[0], main_remapping.__getitem__))
294      main_root = HolderModule({comp.name: comp.gm for comp in all_components})
295      main_g._codegen = gm.graph._codegen
296  
297      # If the output nodes consumes get_attr directly in the original graph,
298      # then we need to make sure get_attr is copied to the new graph.
299      for x in flatten(output_node.args[0]):
300          if x.op == "get_attr":
301              setattr(main_root, x.name, getattr_recursive(gm, x.target))  # type: ignore[arg-type]
302  
303      result_gm = GraphModuleCls(main_root, main_g)
304      if return_fqn_mapping:
305          return result_gm, orig_to_split_fqn_mapping
306  
307      return result_gm
