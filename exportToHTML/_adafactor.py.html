23   class Adafactor(Optimizer):
24       def __init__(
25           self,
26           params: ParamsT,
27           lr: Union[float, Tensor] = 1e-2,
28           beta2_decay: float = -0.8,
29           eps: tuple[Optional[float], float] = (None, 1e-3),
30           d: float = 1.0,
31           weight_decay: float = 0.0,
32           *,
33           foreach: Optional[bool] = None,
34           maximize: bool = False,
35       ):
36           if isinstance(lr, Tensor) and lr.numel() != 1:
37               raise ValueError("Tensor lr must be 1-element")
38           if not 0.0 <= lr:
39               raise ValueError(f"Learning rate should be >= 0 but is: {lr}")
40           if not 0.0 >= beta2_decay:
41               raise ValueError(f"beta2_decay should be <= 0 but is: {beta2_decay}")
42           if eps[0] is not None and not 0.0 <= eps[0]:
43               raise ValueError(f"epsilon1 should be >= 0 but is: {eps[0]}")
44           if not 0.0 <= eps[1]:
45               raise ValueError(f"epsilon2 should be >= 0 but is: {eps[1]}")
46           if not 1.0 <= d:
47               raise ValueError(f"Clipping threshold d should be >= 1 but is: {d}")
48           if not 0.0 <= weight_decay:
49               raise ValueError(f"weight_decay should be >= 0 but is: {weight_decay}")
50           defaults = dict(
51               lr=lr,
52               beta2_decay=beta2_decay,
53               eps=eps,
54               d=d,
55               weight_decay=weight_decay,
56               foreach=foreach,
57               maximize=maximize,
58           )
59           super().__init__(params, defaults)
60   
61       def __setstate__(self, state):
62           super().__setstate__(state)
63           for group in self.param_groups:
64               group.setdefault("foreach", None)
65               for p in group["params"]:
66                   p_state = self.state.get(p, [])
67                   if len(p_state) != 0 and not torch.is_tensor(p_state["step"]):
68                       step_val = float(p_state["step"])
69                       p_state["step"] = torch.tensor(step_val, dtype=_get_scalar_dtype())
70   
71       def _init_group(
72           self,
73           group,
74           params_with_grad,
75           grads,
76           row_vars,
77           col_vars,
78           variances,
79           state_steps,
80       ):
81           for p in group["params"]:
82               if p.grad is None:
83                   continue
84               if torch.is_complex(p):
85                   raise RuntimeError("Adafactor does not support complex parameters")
86               if p.grad.is_sparse:
87                   raise RuntimeError("Adafactor does not support sparse gradients")
88   
89               params_with_grad.append(p)
90               grads.append(p.grad)
91   
92               state = self.state[p]
93   
94               # State initialization
95               if len(state) == 0:
96                   # note(crcrpar): Deliberately host `step` on CPU if both capturable and fused are off.
97                   # This is because kernel launches are costly on CUDA and XLA.
98                   state["step"] = torch.tensor(0.0, dtype=_get_scalar_dtype())
99   
100                  if p.grad.dim() > 1:
101                      row_shape = list(p.grad.shape)
102                      row_shape[-1] = 1
103                      # Row factor of variance, NOT the same shape as grads (will be reduced along last dim)
104                      state["row_var"] = p.grad.new_zeros(row_shape)
105  
106                      col_shape = list(p.grad.shape)
107                      col_shape[-2] = 1
108                      # Col factor of variance, NOT the same shape as grads (will be reduced along penultimate dim)
109                      state["col_var"] = p.grad.new_zeros(col_shape)
110                  else:
111                      state["variance"] = torch.zeros_like(
112                          p.grad, memory_format=torch.preserve_format
113                      )
114  
115              row_vars.append(state.get("row_var", None))
116              col_vars.append(state.get("col_var", None))
117              variances.append(state.get("variance", None))
118              state_steps.append(state["step"])
119          return False  # has_complex

121      @torch.no_grad()
122      def step(self, closure=None):
123          r"""Perform a single optimization step. 
124   
125          Args: 
126              closure (Callable, optional): A closure that reevaluates the model 
127                  and returns the loss. 
128          """
129          self._cuda_graph_capture_health_check()
130  
131          loss = None
132          if closure is not None:
133              with torch.enable_grad():
134                  loss = closure()
135  
136          for group in self.param_groups:
137              params_with_grad: list[Tensor] = []
138              grads: list[Tensor] = []
139              row_vars: list[Optional[Tensor]] = []
140              col_vars: list[Optional[Tensor]] = []
141              variances: list[Optional[Tensor]] = []
142              state_steps: list[Tensor] = []
143              eps1, eps2 = group["eps"]
144  
145              has_complex = self._init_group(
146                  group,
147                  params_with_grad,
148                  grads,
149                  row_vars,
150                  col_vars,
151                  variances,
152                  state_steps,
153              )
154  
155              adafactor(
156                  params_with_grad,
157                  grads,
158                  row_vars,
159                  col_vars,
160                  variances,
161                  state_steps,
162                  d=group["d"],
163                  lr=group["lr"],
164                  beta2_decay=group["beta2_decay"],
165                  weight_decay=group["weight_decay"],
166                  eps1=eps1,
167                  eps2=eps2,
168                  foreach=group["foreach"],
169                  maximize=group["maximize"],
170                  grad_scale=getattr(self, "grad_scale", None),
171                  found_inf=getattr(self, "found_inf", None),
172                  has_complex=has_complex,
173              )
174  
175          return loss

178  Adafactor.__doc__ = (
179      r"""Implements Adafactor algorithm. 
180   
181      .. math:: 
182          \begin{aligned} 
183              &\rule{110mm}{0.4pt}                                                                 \\ 
184              &\textbf{input}      : \gamma \text{(lr)}, \: \tau 
185                  \text{(}\beta_2\text{ decay)}, \: \theta_0 \text{(params)}, \: f(\theta) \text{(objective)},    \\ 
186              &\hspace{15mm}      \: \epsilon_1, \epsilon_2 \text{ (epsilons)}, \: d \text{(clipping threshold)}, \\ 
187              &\hspace{15mm}      \: \lambda \text{(weight decay)}, 
188                  \: \textit{maximize}                                                             \\ 
189              &\textbf{initialize} : \: R_0 \leftarrow 0 \text{ (second moment row factor)},       \\ 
190              &\hspace{23mm} \: C_0 \leftarrow 0 \text{ (second moment col factor)},               \\ 
191              &\hspace{23mm} \: \widehat{V}_0 \leftarrow 0 \text{ (second moment for vectors)}     \\[-1.ex] 
192              &\rule{110mm}{0.4pt}                                                                 \\ 
193              &\textbf{for} \: t=1 \: \textbf{to} \: \ldots \: \textbf{do}                         \\ 
194   
195              &\hspace{5mm}\textbf{if} \: \textit{maximize}:                                       \\ 
196              &\hspace{10mm}G_t           \leftarrow   -\nabla_{\theta} f_t (\theta_{t-1})         \\ 
197              &\hspace{5mm}\textbf{else}                                                           \\ 
198              &\hspace{10mm}G_t           \leftarrow   \nabla_{\theta} f_t (\theta_{t-1})          \\ 
199              &\hspace{5mm}\widehat{\beta}_{2_t} \leftarrow 1 - t^{\tau}                           \\ 
200              &\hspace{5mm}\rho_t         \leftarrow min(lr, \frac{1}{\sqrt{t}})                   \\ 
201              &\hspace{5mm}\alpha_t       \leftarrow max(\epsilon_2, 
202                  \text{RMS}(\theta_{t-1}))\rho_t                                                  \\ 
203              &\hspace{5mm}\theta_t       \leftarrow \theta_{t-1} - \gamma \lambda \theta_{t-1}    \\ 
204              &\hspace{5mm}\textbf{if} \: \text{dim}(G_t) > 1:                                     \\ 
205              &\hspace{10mm}R_t           \leftarrow \widehat{\beta}_{2_t}R_{t-1}+ 
206                  (1-\widehat{\beta}_{2_t})(G_t \odot G_t) \cdot 1_m                               \\ 
207              &\hspace{10mm}C_t           \leftarrow \widehat{\beta}_{2_t}C_{t-1}+ 
208                  (1-\widehat{\beta}_{2_t}) 1^\top_n \cdot (G_t \odot G_t)                         \\ 
209              &\hspace{10mm}\widehat{V}_t \leftarrow 
210                  \frac{R_t \cdot C_t}{max(1^\top_n \cdot R_t, \epsilon_1)}                        \\ 
211              &\hspace{5mm}\textbf{else}                                                           \\ 
212              &\hspace{10mm}\widehat{V}_t \leftarrow \widehat{\beta}_{2_t}\widehat{V}_{t-1}+ 
213                  (1-\widehat{\beta}_{2_t}) \cdot (G_t \odot G_t)                                  \\ 
214              &\hspace{5mm}U_t            \leftarrow 
215                  \frac{G_t}{max(\sqrt{\widehat{V}_t}, \epsilon_1)}                                \\ 
216              &\hspace{5mm}\widehat{U}_t  \leftarrow \frac{U_t}{max(1, \frac{\text{RMS}(U_t)}{d})} \\ 
217              &\hspace{5mm}\theta_t       \leftarrow \theta_{t-1} - \alpha_t \widehat{U}_t         \\ 
218   
219              &\rule{110mm}{0.4pt}                                                          \\[-1.ex] 
220              &\bf{return} \:  \theta_t                                                     \\[-1.ex] 
221              &\rule{110mm}{0.4pt}                                                          \\[-1.ex] 
222          \end{aligned} 
223   
224      For further details regarding the algorithm we refer to `Adafactor: Adaptive Learning Rates with Sublinear Memory Cost`_. 
225      """
226      + rf"""
227      Args:
228          {_params_doc}
229          lr (float, Tensor, optional): unlike other optimizers, Adafactor does not require a
230              learning rate, and Noam Shazeer and Mitchell Stern do not use lr at all.
231              Deviating from the paper, this implementation uses lr for applying weight
232              decay and as the maximum value for relative step size rho_t. Note that in
233              the paper, a constant of 0.01 is used as the maximum value for relative
234              step size, and so we set 0.01 as the default value. (default: 1e-2)
235          beta2_decay (float, optional): the decay rate of beta2. beta2 standardly refers
236              to the coefficient used for computing the running average of the gradient
237              squared. (default: -0.8)
238          eps (Tuple[float, float], optional): epsilon1 is the term added to the denominator
239              of the update calculation to improve numerical stability. This use of epsilon1
240              deviates from the algorithm written in the paper! See note below for more details.
241              epsilon2 is the term used to avoid having too small a weight update when applying
242              parameter scaling. (default: (None, 1e-3))
243          d (float, optional): the clipping threshold, used to avoid larger-than-desired
244              updates.
245          weight_decay (float, optional): weight decay coefficient (default: 1e-2)
246          foreach (bool, optional): whether foreach implementation of optimizer is used. Note
247              that the foreach implementation uses ~ sizeof(params) more peak memory than the
248              for-loop version due to the intermediates being a tensorlist vs just one tensor.
249              As Adafactor is commonly used when memory is prohibitive, Adafactor will default
250              to the slower single tensor for-loop implementation unless this flag is explicitly
251              True. This behavior is contrary to other optimizers, which will attempt defaulting
252              to foreach on CUDA for faster runtime. (default: None)
253          {_maximize_doc}"""
254      + r""" 
255      .. Note:: 
256          The implementation of Adafactor subtly differs from Noam Shazeer and Mitchell Stern 
257          and implementations in some other frameworks with its use of learning rate and 
258          :math:`\epsilon_1`. 
259   
260          Regarding the learning rate hyperparameter: Noam Shazeer and Mitchell Stern do not 
261          use lr at all, as the stated algorithm uses :math:`\rho_t` and update clipping to 
262          affect the step size. 
263   
264          This implementation allows `lr` to influence the maximum value for :math:`\rho_t`: 
265   
266          .. math:: 
267              \begin{aligned} 
268                  &\hspace{5mm}\rho_t \leftarrow min(lr, \frac{1}{\sqrt{t}}) 
269              \end{aligned} 
270   
271          This differs from Noam Shazeer and Mitchell Stern, who use a constant of 0.01 as 
272          the maximum value of :math:`\rho_t` 
273   
274          .. math:: 
275              \begin{aligned} 
276                  &\hspace{5mm}\rho_t \leftarrow min(0.01, \frac{1}{\sqrt{t}}) 
277              \end{aligned} 
278   
279          Noam Shazeer and Mitchell Stern do not enforce an opinion on how weight decay should 
280          be computed, and so we use the learning rate as a coefficient for decoupled weight 
281          decay, similar to what is suggested in `Decoupled Weight Decay Regularization`_. 
282   
283          Regarding the use of :math:`\epsilon_1`: The implementation attempts to replicate the 
284          presumed intention of Noam Shazeer and Mitchell Stern to use :math:`\epsilon_1` as 
285          a stabilizing term when the squared gradient becomes small. 
286   
287          This stabilization can be written as 
288   
289          .. math:: 
290              \begin{aligned} 
291                  &\hspace{5mm}R_t \leftarrow \widehat{\beta}_{2_t}R_{t-1}+ 
292                      (1-\widehat{\beta}_{2_t})(G_t \odot G_t + 1_n \cdot 1^\top_m) \cdot 1_m          \\ 
293                  &\hspace{5mm}C_t \leftarrow \widehat{\beta}_{2_t}C_{t-1}+ 
294                      (1-\widehat{\beta}_{2_t}) 1^\top_n \cdot (G_t \odot G_t + 1_n \cdot 1^\top_m)    \\ 
295                  &\hspace{5mm}\widehat{V}_t \leftarrow 
296                      \frac{R_t \cdot C_t}{max(1^\top_n \cdot R_t, \epsilon_1)}                        \\ 
297                  &\hspace{5mm}U_t \leftarrow \frac{G_t}{max(\sqrt{\widehat{V}_t}, \epsilon_1)}        \\ 
298              \end{aligned} 
299   
300          where the row and column factors of gradient squared :math:`R_t` and :math:`C_t` 
301          are left alone, and we apply :math:`\epsilon_1` at the final calculation of 
302          the variance estimate :math:`\widehat{V}_t` and for the update :math:`U_t`. 
303   
304          This is in contrast to Noam Shazeer and Mitchell Stern and other frameworks which 
305          apply :math:`\epsilon_1` to both row and column factors of the squared gradient, but 
306          not in the calculations after: 
307   
308          .. math:: 
309              \begin{aligned} 
310                  &\hspace{5mm}R_t \leftarrow \widehat{\beta}_{2_t}R_{t-1}+ 
311                              (1-\widehat{\beta}_{2_t})(G_t \odot G_t + \epsilon_1 1_n \cdot 1^\top_m) \cdot 1_m          \\ 
312                  &\hspace{5mm}C_t \leftarrow \widehat{\beta}_{2_t}C_{t-1}+ 
313                              (1-\widehat{\beta}_{2_t}) 1^\top_n \cdot (G_t \odot G_t + \epsilon_1 1_n \cdot 1^\top_m)    \\ 
314                  &\hspace{5mm}\widehat{V}_t \leftarrow \frac{R_t \cdot C_t}{1^\top_n \cdot R_t}                          \\ 
315                  &\hspace{5mm}U_t \leftarrow \frac{G_t}{\sqrt{\widehat{V}_t}}                                            \\ 
316              \end{aligned} 
317   
318   
319      .. _Adafactor\: Adaptive Learning Rates with Sublinear Memory Cost: 
320          https://arxiv.org/pdf/1804.04235 
321      .. _Decoupled Weight Decay Regularization: 
322          https://arxiv.org/abs/1711.05101 
323      """

327  def _single_tensor_adafactor(
328      params: list[Tensor],
329      grads: list[Tensor],
330      # If grad is 1-dimensional (aka a vector), there is no factorization necessary
331      # so row_var and col_var will be None while variance will be filled.
332      # Contrarily, for a grad with multiple dimensions, we will factor along the last
333      # 2 dimensions, and so row_var and col_var will be filled and variance will be None.
334      row_vars: list[Optional[Tensor]],
335      col_vars: list[Optional[Tensor]],
336      variances: list[Optional[Tensor]],
337      state_steps: list[Tensor],
338      grad_scale: Optional[Tensor],
339      found_inf: Optional[Tensor],
340      *,
341      d: float,
342      lr: Union[Tensor, float],
343      beta2_decay: float,
344      weight_decay: float,
345      eps1: Optional[float],
346      eps2: float,
347      maximize: bool,
348      has_complex: bool,
349  ):
350      assert grad_scale is None and found_inf is None, (
351          "Grad scaling should occur outside of optimizer.step()"
352      )
353  
354      if torch.jit.is_scripting():
355          # this assert is due to JIT being dumb and not realizing that the ops below
356          # have overloads to handle both float and Tensor lrs, so we just assert it's
357          # a float since most people using JIT are using floats
358          assert isinstance(lr, float)
359      else:
360          lr = _to_scalar(lr)
361  
362      for i, param in enumerate(params):
363          grad = grads[i] if not maximize else -grads[i]
364          step_t = state_steps[i]
365          row_var = row_vars[i]
366          col_var = col_vars[i]
367          variance = variances[i]
368          if eps1 is None:
369              eps1 = torch.finfo(param.dtype).eps
370  
371          # update step
372          step_t += 1
373          step_float = step_t.item()
374  
375          one_minus_beta2_t = step_float**beta2_decay
376          rho_t = min(lr, 1 / (step_float**0.5))
377          alpha = max(eps2, param.norm(2).item() / (param.numel() ** 0.5)) * rho_t
378  
379          # Perform stepweight decay
380          if weight_decay != 0:
381              param.mul_(1 - lr * weight_decay)
382  
383          if grad.dim() > 1:
384              assert row_var is not None and col_var is not None, (
385                  "row_var and col_var should be defined when grad is multidimensional"
386              )
387              # same as (g * g).mean(dim=-1) w/o materializing an intermediate size g
388              row_mean = (
389                  torch.norm(grad, dim=-1, keepdim=True).square_().div_(grad.size(-1))
390              )
391              row_var.lerp_(row_mean, one_minus_beta2_t)
392              # same as (g * g).mean(dim=-2) w/o materializing an intermediate size g
393              col_mean = (
394                  torch.norm(grad, dim=-2, keepdim=True).square_().div_(grad.size(-2))
395              )
396              col_var.lerp_(col_mean, one_minus_beta2_t)
397              var_estimate = row_var @ col_var
398              var_estimate.div_(row_var.mean(dim=-2, keepdim=True).clamp_(min=eps1))
399          else:
400              assert variance is not None, (
401                  "variance should be defined when grad is a vector"
402              )
403              grad_squared = grad * grad
404              variance.lerp_(grad_squared, one_minus_beta2_t)
405              # avoid writing into variance during update
406              var_estimate = variance.clone()
407  
408          # square the eps1 as we sqrt after to keep eps1's magnitude
409          update = var_estimate.clamp_(min=eps1 * eps1).rsqrt_()
410          update.mul_(grad)
411          denom = max(1.0, update.norm(2).item() / ((update.numel() ** 0.5) * d))
412          param.add_(update, alpha=-alpha / denom)

415  def _group_tensors_by_device_dtype_and_is_multidim(
416      tensorlists: TensorListList,
417  ) -> dict[
418      tuple[Optional[torch.device], Optional[torch.dtype], bool],
419      list[list[Optional[Tensor]]],
420  ]:
421      """Groups tensors by device, dtype, AND multidimensionality -- whether the tensor 
422      has multiple dims or just one dim (is a vector). This allows the foreach impl of 
423      Adafactor to assume that every group of params will either be factored or not."""
424      grouped_tensors = Optimizer._group_tensors_by_device_and_dtype(tensorlists)
425      ultra_grouped_tensors: dict[
426          tuple[Optional[torch.device], Optional[torch.dtype], bool],
427          list[list[Optional[Tensor]]],
428      ] = {}
429      for (device, dtype), (tensorlists, _) in grouped_tensors.items():
430          matrix_key = (device, dtype, True)
431          vector_key = (device, dtype, False)
432  
433          # assumes grad is the second tensorlist
434          for j, tensor in enumerate(tensorlists[1]):
435              assert tensor is not None, "grad should not be None"
436              if tensor.dim() > 1:
437                  if matrix_key not in ultra_grouped_tensors:
438                      ultra_grouped_tensors[matrix_key] = [[] for _ in tensorlists]
439                  for i in range(len(tensorlists)):
440                      ultra_grouped_tensors[matrix_key][i].append(tensorlists[i][j])
441              else:
442                  if vector_key not in ultra_grouped_tensors:
443                      ultra_grouped_tensors[vector_key] = [[] for _ in tensorlists]
444                  for i in range(len(tensorlists)):
445                      ultra_grouped_tensors[vector_key][i].append(tensorlists[i][j])
446      return ultra_grouped_tensors

449  def _multi_tensor_adafactor(
450      params: list[Tensor],
451      grads: list[Tensor],
452      # If grad is 1-dimensional (aka a vector), there is no factorization necessary
453      # so row_var and col_var will be None while variance will be filled.
454      # Contrarily, for a grad with multiple dimensions, we will factor along the last
455      # 2 dimensions, and so row_var and col_var will be filled and variance will be None.
456      row_vars: list[Optional[Tensor]],
457      col_vars: list[Optional[Tensor]],
458      variances: list[Optional[Tensor]],
459      state_steps: list[Tensor],
460      grad_scale: Optional[Tensor],
461      found_inf: Optional[Tensor],
462      *,
463      d: float,
464      lr: Union[Tensor, float],
465      beta2_decay: float,
466      weight_decay: float,
467      eps1: Optional[float],
468      eps2: float,
469      maximize: bool,
470      has_complex: bool,
471  ):
472      if len(params) == 0:
473          return
474  
475      assert grad_scale is None and found_inf is None, (
476          "Grad scaling should occur outside of optimizer.step()"
477      )
478  
479      lr = _to_scalar(lr)
480  
481      grouped_tensors = _group_tensors_by_device_dtype_and_is_multidim(
482          [params, grads, row_vars, col_vars, variances, state_steps]  # type: ignore[list-item]
483      )
484      for (_, dtype, is_multidim), (
485          (
486              device_params_,
487              device_grads_,
488              device_row_vars_,
489              device_col_vars_,
490              device_variances_,
491              device_state_steps_,
492          )
493      ) in grouped_tensors.items():
494          device_params = cast(list[Tensor], device_params_)
495          device_grads = cast(list[Tensor], device_grads_)
496          device_state_steps = cast(list[Tensor], device_state_steps_)
497          if eps1 is None:
498              assert dtype is not None, (
499                  "dtype is needed to compute eps1 when eps1 is unset"
500              )
501              eps1 = torch.finfo(dtype).eps
502  
503          if TYPE_CHECKING:
504              assert device_state_steps[0] is not None
505  
506          if maximize:
507              device_grads = torch._foreach_neg(device_grads)  # type: ignore[assignment]
508  
509          # Update steps
510          # If steps are on CPU, foreach will fall back to the slow path, which is a for-loop calling t.add(1) over
511          # and over. 1 will then be wrapped into a Tensor over and over again, which is slower than if we just
512          # wrapped it once now. The alpha is required to assure we go to the right overload.
513          if not torch.compiler.is_compiling() and device_state_steps[0].is_cpu:
514              torch._foreach_add_(
515                  device_state_steps, torch.tensor(1.0, device="cpu"), alpha=1.0
516              )
517          else:
518              torch._foreach_add_(device_state_steps, 1.0)
519  
520          one_minus_beta2_ts = []
521          beta2_ts = []
522          rho_ts = []
523          for s in device_state_steps:
524              one_minus_beta2_ts.append(s.item() ** beta2_decay)
525              beta2_ts.append(1 - s.item() ** beta2_decay)
526              rho_ts.append(min(lr, 1 / (s.item() ** 0.5)))
527  
528          alphas = [
529              max(eps2, p.norm(2).item() / (p.numel() ** 0.5)) * r
530              for p, r in zip(device_params, rho_ts)
531          ]
532  
533          # Perform stepweight decay
534          if weight_decay != 0:
535              torch._foreach_mul_(device_params, 1 - lr * weight_decay)
536  
537          if is_multidim:
538              device_row_vars = cast(list[Tensor], device_row_vars_)
539              device_col_vars = cast(list[Tensor], device_col_vars_)
540              assert device_row_vars[0] is not None and device_col_vars[0] is not None, (
541                  "row_var and col_var should be defined when grad is multidimensional"
542              )
543              # same as (g * g).mean(dim=-1) w/o materializing an intermediate size g
544              row_means = [
545                  torch.norm(grad, dim=-1, keepdim=True) for grad in device_grads
546              ]
547              torch._foreach_mul_(row_means, row_means)
548              torch._foreach_div_(row_means, [grad.size(-1) for grad in device_grads])
549              torch._foreach_lerp_(device_row_vars, row_means, one_minus_beta2_ts)
550              del row_means
551  
552              # same as (g * g).mean(dim=-2) w/o materializing an intermediate size g
553              col_means = [
554                  torch.norm(grad, dim=-2, keepdim=True) for grad in device_grads
555              ]
556              torch._foreach_mul_(col_means, col_means)
557              torch._foreach_div_(col_means, [grad.size(-2) for grad in device_grads])
558              torch._foreach_lerp_(device_col_vars, col_means, one_minus_beta2_ts)
559              del col_means
560  
561              var_estimates = [
562                  row_var @ col_var
563                  for row_var, col_var in zip(device_row_vars, device_col_vars)
564              ]
565              row_var_means = [
566                  row_var.mean(dim=-2, keepdim=True) for row_var in device_row_vars
567              ]
568              torch._foreach_clamp_min_(row_var_means, eps1)
569              torch._foreach_div_(var_estimates, row_var_means)
570              del row_var_means
571          else:
572              device_variances = cast(list[Tensor], device_variances_)
573              assert device_variances[0] is not None, (
574                  "variance should be defined when grad is a vector"
575              )
576  
577              grads_squared = torch._foreach_mul(device_grads, device_grads)
578              torch._foreach_lerp_(device_variances, grads_squared, one_minus_beta2_ts)
579              del grads_squared
580  
581              # avoid writing into variance during update
582              var_estimates = [v.clone() for v in device_variances]
583  
584          # square the eps1 as we sqrt after to keep eps1's magnitude
585          torch._foreach_clamp_min_(var_estimates, eps1 * eps1)
586          torch._foreach_rsqrt_(var_estimates)
587          torch._foreach_mul_(var_estimates, device_grads)
588          updates = var_estimates
589  
590          alphas = [
591              -a / (max(1.0, update.norm(2).item() / ((update.numel() ** 0.5) * d)))
592              for a, update in zip(alphas, updates)
593          ]
594          torch._foreach_mul_(updates, alphas)
595          torch._foreach_add_(device_params, updates)
596
