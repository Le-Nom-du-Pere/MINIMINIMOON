26   def _maybe_get_fqn(node: Node, gm: GraphModule) -> Optional[str]:
27       fqn = None
28       if hasattr(gm, "_node_name_to_scope"):
29           # fqn on observers is not present, because they do not
30           # exist when the fqns are created during tracing. If this is
31           # an observer, get the fqn of the node being observed.
32           node_to_use_for_fqn = node
33           if node.op == "call_module":
34               assert isinstance(node.target, str)
35               module = getattr_from_fqn(gm, node.target)
36               if _is_activation_post_process(module):
37                   node_to_use_for_fqn = get_normalized_nth_input(node, gm, 0)
38           fqn = gm._node_name_to_scope[node_to_use_for_fqn.name][0]  # type: ignore[index]
39       return fqn  # type: ignore[return-value]

42   def _insert_logger_after_node(
43       node: Node,
44       gm: GraphModule,
45       logger_cls: Callable,
46       logger_node_name_suffix: str,
47       ref_node_name: str,
48       model_name: str,
49       ref_name: str,
50       ref_node_target_type: str,
51       results_type: str,
52       index_within_arg: int,
53       index_of_arg: int,
54       fqn: Optional[str],
55   ) -> Node:
56       """ 
57       Given a starting graph of 
58    
59       prev_node -> node -> next_node 
60    
61       This function creates a new logger_cls obj and adds it 
62       after node, resulting in 
63    
64       prev_node -> node -> logger_obj -> next_node 
65       """
66       # create new name
67       logger_node_name = get_new_attr_name_with_prefix(
68           node.name + logger_node_name_suffix
69       )(gm)
70       target_type = get_target_type_str(node, gm)
71       # create the logger object
72       logger_obj = logger_cls(
73           ref_node_name,
74           node.name,
75           model_name,
76           ref_name,
77           target_type,
78           ref_node_target_type,
79           results_type,
80           index_within_arg,
81           index_of_arg,
82           fqn,
83       )
84       # attach the logger object to the parent module
85       setattr(gm, logger_node_name, logger_obj)
86       logger_node = node.graph.create_node("call_module", logger_node_name, (node,), {})
87       return logger_node

90   def add_loggers_to_model(
91       gm: GraphModule,
92       node_to_instrument_inputs_to_ref_node_name: dict[Node, tuple[str, str]],
93       node_to_instrument_outputs_to_ref_node_name: dict[Node, tuple[str, str]],
94       logger_cls: Callable,
95       model_name: str,
96   ) -> GraphModule:
97       """ 
98       Takes the graph of gm, adds loggers to the output 
99       of each node in nodes_to_instrument. Returns a GraphModule with the new 
100      graph. 
101      """
102  
103      new_graph = Graph()
104      env: dict[str, Any] = {}
105  
106      def load_arg(a):
107          return map_arg(a, lambda node: env[node.name])
108  
109      for node in gm.graph.nodes:
110          if node.op == "output":
111              new_graph.output(map_arg(get_normalized_nth_input(node, gm, 0), load_arg))
112              continue
113  
114          if (node in node_to_instrument_inputs_to_ref_node_name) or (
115              node in node_to_instrument_outputs_to_ref_node_name
116          ):
117              fqn = _maybe_get_fqn(node, gm)
118  
119              if node in node_to_instrument_inputs_to_ref_node_name:
120                  ref_name, ref_node_type = node_to_instrument_inputs_to_ref_node_name[
121                      node
122                  ]
123                  # Ops such add and mul are special because either
124                  # one or two of the first two arguments can be tensors,
125                  # and if one argument is a tensor it can be first or
126                  # second (x + 1 versus 1 + x).
127                  arg_indices_to_log = get_arg_indices_of_inputs_to_log(node)
128                  for node_arg_idx in arg_indices_to_log:
129                      node_arg = get_normalized_nth_input(node, gm, node_arg_idx)
130                      if type(node_arg) == Node:
131                          # create a single input logger
132                          prev_node = env[node_arg.name]
133                          env[node_arg.name] = _insert_logger_after_node(
134                              prev_node,
135                              gm,
136                              logger_cls,
137                              "_ns_logger_",
138                              node.name,
139                              model_name,
140                              ref_name,
141                              ref_node_type,
142                              NSSingleResultValuesType.NODE_INPUT.value,
143                              index_within_arg=0,
144                              index_of_arg=node_arg_idx,
145                              fqn=fqn,
146                          )
147                      elif (
148                          type(node_arg) == torch.fx.immutable_collections.immutable_list
149                      ):
150                          # create N input loggers, one for each node
151                          for arg_idx, arg in enumerate(node_arg):  # type: ignore[var-annotated, arg-type]
152                              prev_node = env[arg.name]
153                              env[prev_node.name] = _insert_logger_after_node(
154                                  prev_node,
155                                  gm,
156                                  logger_cls,
157                                  "_ns_logger_",
158                                  node.name,
159                                  model_name,
160                                  ref_name,
161                                  ref_node_type,
162                                  NSSingleResultValuesType.NODE_INPUT.value,
163                                  index_within_arg=arg_idx,
164                                  index_of_arg=node_arg_idx,
165                                  fqn=fqn,
166                              )
167                      else:
168                          pass
169  
170              # ensure env is populated with base node
171              # Note: runs for both inputs and outputs
172              env[node.name] = new_graph.node_copy(node, load_arg)
173  
174              if node in node_to_instrument_outputs_to_ref_node_name:
175                  ref_name, ref_node_type = node_to_instrument_outputs_to_ref_node_name[
176                      node
177                  ]
178                  # add the logger after the base node
179                  env[node.name] = _insert_logger_after_node(
180                      env[node.name],
181                      gm,
182                      logger_cls,
183                      "_ns_logger_",
184                      node.name,
185                      model_name,
186                      ref_name,
187                      ref_node_type,
188                      NSSingleResultValuesType.NODE_OUTPUT.value,
189                      index_within_arg=0,
190                      index_of_arg=0,
191                      fqn=fqn,
192                  )
193  
194          else:
195              env[node.name] = new_graph.node_copy(node, load_arg)
196  
197      new_gm = GraphModule(gm, new_graph)
198      return new_gm

201  def _insert_quantize_per_tensor_node(
202      prev_node_c: Node,
203      node_a: Node,
204      gm_b: GraphModule,
205      graph_c: Graph,
206      scale: Union[torch.Tensor, float],
207      zero_point: Union[torch.Tensor, int],
208      dtype_cast_name: str,
209  ) -> Node:
210      # copy scale
211      scale_node_name = get_new_attr_name_with_prefix(node_a.name + "_input_scale_")(gm_b)
212      setattr(gm_b, scale_node_name, scale)
213      scale_node = graph_c.create_node(
214          "get_attr", scale_node_name, (), {}, scale_node_name
215      )
216      # copy zero_point
217      zero_point_node_name = get_new_attr_name_with_prefix(
218          node_a.name + "_input_zero_point_"
219      )(gm_b)
220      setattr(gm_b, zero_point_node_name, zero_point)
221      zero_point_node = graph_c.create_node(
222          "get_attr", zero_point_node_name, (), {}, zero_point_node_name
223      )
224      # create the quantize_per_tensor call
225      return graph_c.create_node(
226          "call_function",
227          torch.quantize_per_tensor,
228          (prev_node_c, scale_node, zero_point_node, torch.quint8),
229          {},
230          dtype_cast_name,
231      )

234  def _insert_dtype_cast_after_node(
235      node_a: Node,
236      node_c: Node,
237      prev_node_c: Union[Node, list[Node]],
238      gm_a: GraphModule,
239      gm_b: GraphModule,
240      graph_c: Graph,
241      node_name_prefix: str,
242      logger_cls: Callable,
243      node_type_to_io_type_map: dict[str, set[NSNodeTargetType]],
244  ) -> Union[Node, list[Node]]:
245      """ 
246      Given a starting graph C (derived from graph B) of 
247   
248      ... -> prev_node_c -> node_c -> ... 
249   
250      And a corresponding related node_a, inserts the correct dtype 
251      cast node after prev_node_c to cast into the dtype expected 
252      by node_a, resulting in: 
253   
254                            dtype_cast 
255                          / 
256      ... -> prev_node_c -> node_c -> ... 
257   
258      For example, if node_c is an int8 op and node_a is an fp32 op, this function 
259      will insert a dequant. 
260      """
261      dtype_cast_op = None
262      dtype_cast_mod_cls = None
263      dtype_cast_method = None
264      dtype_cast_method_dtype = None
265      dtype_cast_scale = None
266      dtype_cast_zero_point = None
267      node_input_type_a, _node_output_type_a = get_node_first_input_and_output_type(
268          node_a, gm_a, logger_cls, node_type_to_io_type_map
269      )
270      node_input_type_c, _node_output_type_c = get_node_first_input_and_output_type(
271          node_c, gm_b, logger_cls, node_type_to_io_type_map
272      )
273  
274      if (
275          (
276              node_input_type_a == NodeInputOrOutputType.FP32
277              and node_input_type_c == NodeInputOrOutputType.INT8
278          )
279          or (
280              node_input_type_a == NodeInputOrOutputType.FP32
281              and node_input_type_c == NodeInputOrOutputType.FP16
282          )
283          or
284          # TODO(future PR): determine the actual dtype of node_c,
285          # the current code only works because dequantize works with
286          # multiple input dtypes.
287          (
288              node_input_type_a == NodeInputOrOutputType.FP32
289              and node_input_type_c == NodeInputOrOutputType.FP32_OR_INT8
290          )
291      ):
292          dtype_cast_op = torch.dequantize
293      elif (
294          node_input_type_a == node_input_type_c
295          and node_input_type_a != NodeInputOrOutputType.UNKNOWN
296      ):
297          dtype_cast_mod_cls = torch.nn.Identity
298      elif (
299          node_input_type_a == NodeInputOrOutputType.INT8
300          and node_input_type_c == NodeInputOrOutputType.FP32
301      ):
302          # int8 shadows fp32, the dtype cast needs to quantize to int8
303          # with the right qparams.
304          node_a_input_qparams = get_node_input_qparams(
305              node_a, gm_a, node_type_to_io_type_map
306          )
307          if node_a_input_qparams is not None:
308              dtype_cast_op = torch.quantize_per_tensor  # type: ignore[assignment]
309              dtype_cast_scale, dtype_cast_zero_point = node_a_input_qparams
310      elif (
311          node_input_type_a == NodeInputOrOutputType.FP16
312          and node_input_type_c == NodeInputOrOutputType.FP32
313      ):
314          dtype_cast_method = "to"
315          dtype_cast_method_dtype = torch.float16
316      else:
317          raise AssertionError(
318              f"dtype cast from {node_input_type_c} {node_c.format_node()} to "
319              + f"{node_input_type_a} {node_a.format_node()} needs to be implemented"
320          )
321  
322      if isinstance(prev_node_c, Node):
323          new_dtype_cast_name = get_new_attr_name_with_prefix(node_name_prefix)(gm_b)
324          if dtype_cast_op:
325              if dtype_cast_scale is not None and dtype_cast_zero_point is not None:
326                  return _insert_quantize_per_tensor_node(
327                      prev_node_c,
328                      node_a,
329                      gm_b,
330                      graph_c,
331                      dtype_cast_scale,
332                      dtype_cast_zero_point,
333                      new_dtype_cast_name,
334                  )
335              else:
336                  return graph_c.create_node(
337                      "call_function",
338                      dtype_cast_op,
339                      (prev_node_c,),
340                      {},
341                      new_dtype_cast_name,
342                  )
343          elif dtype_cast_method:
344              return graph_c.create_node(
345                  "call_method",
346                  dtype_cast_method,
347                  (prev_node_c, dtype_cast_method_dtype),
348                  {},
349                  new_dtype_cast_name,
350              )
351          else:
352              assert dtype_cast_mod_cls
353              dtype_cast_mod = dtype_cast_mod_cls()
354              setattr(gm_b, new_dtype_cast_name, dtype_cast_mod)
355              return graph_c.create_node(
356                  "call_module",
357                  new_dtype_cast_name,
358                  (prev_node_c,),
359                  {},
360                  new_dtype_cast_name,
361              )
362      elif isinstance(prev_node_c, list):
363          results = []
364          for prev_node_c_inner in prev_node_c:
365              new_dtype_cast_name = get_new_attr_name_with_prefix(node_name_prefix)(gm_b)
366              if dtype_cast_op:
367                  # TODO(future PR): add handling for quantize_per_tensor
368                  new_dtype_cast_node = graph_c.create_node(
369                      "call_function",
370                      dtype_cast_op,
371                      (prev_node_c_inner,),
372                      {},
373                      new_dtype_cast_name,
374                  )
375                  results.append(new_dtype_cast_node)
376              else:
377                  assert dtype_cast_mod_cls
378                  dtype_cast_mod = dtype_cast_mod_cls()
379                  setattr(gm_b, new_dtype_cast_name, dtype_cast_mod)
380                  new_dtype_cast_node = graph_c.create_node(
381                      "call_module",
382                      new_dtype_cast_name,
383                      (prev_node_c_inner,),
384                      {},
385                      new_dtype_cast_name,
386                  )
387                  results.append(new_dtype_cast_node)
388          return results
389      else:
390          raise AssertionError(f"type f{type(prev_node_c)} is not handled")

394  def _copy_node_from_a_to_c(
395      node_a: Node,
396      gm_a: GraphModule,
397      gm_b: GraphModule,
398      graph_c: Graph,
399  ) -> Node:
400      """ 
401      Simple copy of node_a to graph_c. 
402      """
403      if node_a.op == "get_attr":
404          node_a_copy_name = get_new_attr_name_with_prefix(node_a.name + "_shadow_copy_")(
405              gm_b
406          )
407          node_a_obj = getattr_from_fqn(gm_a, node_a.target)  # type: ignore[arg-type]
408          if torch.is_tensor(node_a_obj):
409              node_a_obj = node_a_obj.detach()
410          setattr(gm_b, node_a_copy_name, node_a_obj)
411          node_a_copy = graph_c.create_node(
412              node_a.op, node_a_copy_name, (), {}, node_a_copy_name
413          )
414          return node_a_copy
415      elif node_a.op == "call_method":
416          assert node_a.target in (
417              "dequantize",
418              "to",
419          ), f"target {node_a.target} is not implemented"
420          if node_a.target == "dequantize":
421              arg_copy = _copy_node_from_a_to_c(
422                  get_normalized_nth_input(node_a, gm_a, 0), gm_a, gm_b, graph_c
423              )  # type: ignore[arg-type]
424              node_a_copy_name = get_new_attr_name_with_prefix(
425                  node_a.name + "_shadow_copy_"
426              )(gm_b)
427              node_a_copy = graph_c.create_node(
428                  node_a.op, node_a.target, (arg_copy,), {}, node_a_copy_name
429              )
430              return node_a_copy
431          else:  # to
432              arg_copy = _copy_node_from_a_to_c(
433                  get_normalized_nth_input(node_a, gm_a, 0), gm_a, gm_b, graph_c
434              )  # type: ignore[arg-type]
435              node_a_copy_name = get_new_attr_name_with_prefix(
436                  node_a.name + "_shadow_copy_"
437              )(gm_b)
438              node_a_copy = graph_c.create_node(
439                  node_a.op,
440                  node_a.target,
441                  (arg_copy, get_normalized_nth_input(node_a, gm_a, 1)),
442                  {},
443                  node_a_copy_name,
444              )
445              return node_a_copy
446  
447      else:
448          raise AssertionError(
449              f"handling of node {node_a.format_node()} with op {node_a.op} is not implemented"
450          )

453  def _can_insert_copy_of_subgraph_a(
454      subgraph_a: NSSubgraph,
455      gm_a: GraphModule,
456      num_non_param_args_node_a: int,
457  ) -> bool:
458      """ 
459      This function returns `False` if the input subgraph cannot be copied by 
460      `_insert_copy_of_subgraph_a_after_input_node_c`. This usually means 
461      that there is a corner case logic for which copy is not yet implemented. 
462      """
463      # populate the list of nodes we need to check
464      nodes = []
465      cur_node = subgraph_a.end_node
466      while cur_node != subgraph_a.start_node:
467          nodes.append(cur_node)
468          cur_node = get_normalized_nth_input(cur_node, gm_a, 0)  # type: ignore[assignment]
469      nodes.append(cur_node)
470      nodes.reverse()
471  
472      def _can_insert(node_a_arg, gm_a):
473          if isinstance(node_a_arg, Node):
474              arg_a = return_first_non_observer_node(node_a_arg, gm_a)
475              if arg_a.op == "call_method":
476                  return arg_a.target in ("dequantize", "to")
477              elif arg_a.op == "get_attr":
478                  return True
479              else:
480                  return False
481          elif isinstance(node_a_arg, (list, tuple)):
482              for el in node_a_arg:
483                  if not isinstance(el, Node):
484                      return False
485          return True
486  
487      # For each node, check if we handle the copy behavior. This follows the
488      # logic in `_insert_copy_of_subgraph_a_after_input_node_c`.
489      for node_a in nodes:
490          local_num_non_param_args_node_a = (
491              num_non_param_args_node_a if node_a is nodes[0] else 1
492          )
493  
494          norm_args_kwargs = node_a.normalized_arguments(
495              gm_a, normalize_to_only_use_kwargs=True
496          )
497          if norm_args_kwargs is not None:
498              norm_args, norm_kwargs = norm_args_kwargs
499          else:
500              norm_args, norm_kwargs = node_a.args, node_a.kwargs
501  
502          cur_idx = 0
503  
504          while cur_idx < len(norm_args):
505              if cur_idx == 0:
506                  pass
507              elif cur_idx == 1 and local_num_non_param_args_node_a == 2:
508                  pass
509              else:
510                  if not _can_insert(norm_args[cur_idx], gm_a):
511                      return False
512              cur_idx += 1
513  
514          for kwarg_val in norm_kwargs.values():
515              # stitch the inputs from base graph
516              if cur_idx == 0:
517                  pass
518              elif cur_idx == 1 and local_num_non_param_args_node_a == 2:
519                  pass
520              else:
521                  if not _can_insert(kwarg_val, gm_a):
522                      return False
523              cur_idx += 1
524  
525      return True

528  def _insert_copy_of_subgraph_a_after_input_node_c(
529      input_node_c: Union[Node, list[Node]],
530      input_node_c_2: Optional[Union[Node, list[Node]]],
531      subgraph_a: NSSubgraph,
532      gm_a: GraphModule,
533      gm_b: GraphModule,
534      node_name_prefix: str,
535  ) -> Node:
536      """ 
537      TODO(before land): real docblock 
538      """
539      assert isinstance(input_node_c, (Node, list))
540  
541      # create a sequential list of the subgraphs' nodes from start to end,
542      # because we need to add the nodes to graph C in non-reverse order
543      nodes_of_a = [subgraph_a.end_node]
544      cur_node = subgraph_a.end_node
545      while cur_node != subgraph_a.start_node:
546          cur_node = get_normalized_nth_input(cur_node, gm_a, 0)  # type: ignore[assignment]
547          nodes_of_a.insert(0, cur_node)
548  
549      # go through nodes of a in order, and insert them into the graph of c
550      # sequentially
551      cur_node_a = nodes_of_a[0]
552      cur_node_c = _insert_copy_of_node_a_after_input_node_c(
553          input_node_c, input_node_c_2, cur_node_a, gm_a, gm_b, node_name_prefix
554      )
555      for cur_idx_a in range(1, len(nodes_of_a)):
556          cur_node_a = nodes_of_a[cur_idx_a]
557          prev_node_c = cur_node_c  # previous added node is the input to next node
558          cur_node_c = _insert_copy_of_node_a_after_input_node_c(
559              prev_node_c,
560              # TODO(future PR): enable multiple inputs for nodes which are not at start of subgraph
561              None,
562              cur_node_a,
563              gm_a,
564              gm_b,
565              node_name_prefix,
566          )
567      # return the last inserted node
568      return cur_node_c

571  def _insert_copy_of_node_a_after_input_node_c(
572      input_node_c: Union[Node, list[Node]],
573      input_node_c_2: Optional[Union[Node, list[Node]]],
574      node_a: Node,
575      gm_a: GraphModule,
576      gm_b: GraphModule,
577      node_name_prefix: str,
578  ) -> Node:
579      """ 
580      Assume that node_a from graph_a has 
581        args (input, (input2)?, arg1, ...), and 
582        kwargs {kw0: kwarg0, ...} 
583   
584      Note: input2 is optional. If it equals to None, we assume that the op 
585      has a single non-param input.  If it is specified, we assume that the op 
586      has two non-param inputs. 
587   
588      Copies the underlying values of arg1..argn and kwarg0..kwargn into gm_b, 
589      and creates the corresponding nodes in graph_c. Note: observers are ignored, 
590      so if an arg is an observer we navigate up until we find a non-observer parent. 
591   
592      If node_a is a call_module, points the module pointed to by node_a to gm_b. 
593   
594      Creates the copy of node_a in graph_c, with input as the first arg, 
595      and all other args and kwargs pointing to the copies of the objects 
596      in gm_b created above. 
597   
598      An example in pictures: 
599   
600      graph A: 
601      ======== 
602   
603      input -------------> node_a 
604                           / / / 
605      (input_2)?----------/ / / 
606                           / / 
607      weight -> weight_obs  / 
608                           / 
609      bias ---------------- 
610   
611      graph C (derived from B): 
612      ========================= 
613   
614      input_node_c --> node_a_copy 
615                       / / / 
616      (input_node_c_2)? / / 
617                       / / 
618      weight_copy ----/ / 
619                       / 
620      bias_copy ------/ 
621      """
622      if isinstance(input_node_c, Node):
623          graph_c = input_node_c.graph
624      else:
625          assert isinstance(input_node_c, list)
626          graph_c = input_node_c[0].graph
627  
628      norm_args_kwargs = node_a.normalized_arguments(
629          gm_a, normalize_to_only_use_kwargs=True
630      )
631      if norm_args_kwargs is not None:
632          norm_args, norm_kwargs = norm_args_kwargs
633      else:
634          norm_args, norm_kwargs = node_a.args, node_a.kwargs
635  
636      new_args = []
637      new_kwargs = {}
638  
639      def _copy_arg(arg):
640          # copy the other inputs from the other graph
641          if isinstance(arg, Node):
642              arg = return_first_non_observer_node(arg, gm_a)
643              arg = _copy_node_from_a_to_c(arg, gm_a, gm_b, graph_c)
644              return arg
645          elif isinstance(arg, (int, float, torch.dtype)):
646              return arg
647          elif isinstance(kwarg_val, (list, tuple)):
648              for el in kwarg_val:
649                  assert not isinstance(el, Node), (
650                      "handling of Node inside list is not implemented"
651                  )
652              return arg
653          else:
654              raise AssertionError(
655                  f"handling for kwarg of type {type(kwarg_val)} is not implemented"
656              )
657  
658      cur_idx = 0
659  
660      while cur_idx < len(norm_args):
661          if cur_idx == 0:
662              new_arg = input_node_c
663          elif cur_idx == 1 and input_node_c_2 is not None:
664              new_arg = input_node_c_2
665          else:
666              new_arg = _copy_arg(norm_args[cur_idx])
667          new_args.append(new_arg)
668          cur_idx += 1
669  
670      for kwarg_name, kwarg_val in norm_kwargs.items():
671          # stitch the inputs from base graph
672          if cur_idx == 0:
673              new_kwargs[kwarg_name] = input_node_c
674          elif cur_idx == 1 and input_node_c_2 is not None:
675              new_kwargs[kwarg_name] = input_node_c_2
676          else:
677              new_kwargs[kwarg_name] = _copy_arg(kwarg_val)
678          cur_idx += 1
679  
680      new_args = tuple(new_args)  # type: ignore[assignment]
681  
682      node_a_shadows_c_name = get_new_attr_name_with_prefix(node_name_prefix)(gm_b)
683  
684      if node_a.op == "call_module":
685          # if target is a module, we point to the module from gm_b
686          new_mod_copy_name = get_new_attr_name_with_prefix(node_name_prefix)(gm_b)
687          # fetch the corresponding module from gm_a
688          assert isinstance(node_a.target, str)
689          mod_a = getattr_from_fqn(gm_a, node_a.target)
690          setattr(gm_b, new_mod_copy_name, mod_a)
691          node_a_shadows_c = graph_c.create_node(
692              node_a.op,
693              new_mod_copy_name,
694              new_args,  # type: ignore[arg-type]
695              new_kwargs,  # type: ignore[arg-type]
696              node_a_shadows_c_name,
697          )
698          return node_a_shadows_c
699      else:
700          assert node_a.op in ("call_function", "call_method")
701          node_a_shadows_c = graph_c.create_node(
702              node_a.op,
703              node_a.target,
704              new_args,  # type: ignore[arg-type]
705              new_kwargs,  # type: ignore[arg-type]
706              node_a_shadows_c_name,
707          )
708          return node_a_shadows_c
709

711  def create_a_shadows_b(
712      name_a: str,
713      gm_a: GraphModule,
714      name_b: str,
715      gm_b: GraphModule,
716      matched_subgraph_pairs: dict[str, tuple[NSSubgraph, NSSubgraph]],
717      logger_cls: Callable,
718      should_log_inputs: bool,
719      node_type_to_io_type_map: Optional[dict[str, set[NSNodeTargetType]]] = None,
720  ) -> GraphModule:
721      """ 
722      Creates a new GraphModule consisting of the graph of C, with the meaningful 
723      nodes of A shadowing the corresponding nodes of B.  For example, 
724   
725      Graph A: 
726      a0 -> op0_fp32 -> a1 -> op1_fp32 -> a2 
727   
728      Graph B: 
729      b0 -> op0_int8 -> b1 -> op1_int8 -> b2 
730   
731      matched_node_pairs: {'op0': (op0_fp32, op0_int8), 'op1': (op1_fp32, op1_int8)} 
732   
733      Graph C (A shadows B): 
734   
735          / dequant0 -> op0_fp32 -> logger_a_0  / dequant_1 -> op1_fp32 -> logger_a_1 
736         /                                     / 
737      b0 -------------> op0_int8 -> logger_b_0 --------------> op1_int8 -> logger_b_1 
738   
739      In a nutshell, this function does the following for each node pair: 
740      * copies the necessary attributes and modules from gm_a to gm_b, 
741        keeping names unique 
742      * adds a dtype cast op (dequant, quant, etc) 
743      * adds a copy of node_a in gm_b's graph 
744      * adds loggers to the outputs of node_a and node_b 
745      """
746  
747      if node_type_to_io_type_map is None:
748          node_type_to_io_type_map = get_node_type_to_io_type_map()
749  
750      # graph_c is the graph created from copying the nodes of graph_b and inserting
751      # the shadows with the nodes copied from graph_a
752      graph_c = Graph()
753      env_c: dict[str, Any] = {}
754  
755      def load_arg(a):
756          return map_arg(a, lambda node: env_c[node.name])
757  
758      start_node_b_to_matched_subgraph_a_and_name = {}
759      end_node_b_to_matched_subgraph_a_and_name = {}
760      for match_name, match in matched_subgraph_pairs.items():
761          subgraph_a, subgraph_b = match
762          ref_node_type_a = get_target_type_str(subgraph_a.base_op_node, gm_a)
763          ref_node_type_b = get_target_type_str(subgraph_b.base_op_node, gm_b)
764          start_node_b_to_matched_subgraph_a_and_name[subgraph_b.start_node] = (
765              subgraph_a,
766              match_name,
767              ref_node_type_a,
768              ref_node_type_b,
769          )
770          end_node_b_to_matched_subgraph_a_and_name[subgraph_b.end_node] = (
771              subgraph_a,
772              match_name,
773              ref_node_type_a,
774              ref_node_type_b,
775          )
776  
777      for node_b in gm_b.graph.nodes:
778          if node_b.op == "output":
779              graph_c.output(map_arg(node_b.args[0], load_arg))
780              continue
781  
782          # calculate the flags to determine what to do with this node
783          node_b_is_start_node = node_b in start_node_b_to_matched_subgraph_a_and_name
784          node_b_is_end_node = node_b in end_node_b_to_matched_subgraph_a_and_name
785  
786          if node_b_is_start_node or node_b_is_end_node:
787              if node_b_is_start_node:
788                  (
789                      subgraph_a,
790                      ref_name,
791                      ref_node_type_a,
792                      ref_node_type_b,
793                  ) = start_node_b_to_matched_subgraph_a_and_name[node_b]
794              else:
795                  assert node_b_is_end_node
796                  (
797                      subgraph_a,
798                      ref_name,
799                      ref_node_type_a,
800                      ref_node_type_b,
801                  ) = end_node_b_to_matched_subgraph_a_and_name[node_b]
802  
803              all_op_types_support_shadowing = op_type_supports_shadowing(
804                  subgraph_a.start_node
805              ) and op_type_supports_shadowing(node_b)
806              if not all_op_types_support_shadowing:
807                  print(
808                      f"skipping shadow loggers for node_b: {get_target_type_str(node_b, gm_b)}"
809                      + f", start_node_a: {get_target_type_str(subgraph_a.start_node, gm_a)}"
810                      + ", unsupported"
811                  )
812                  env_c[node_b.name] = graph_c.node_copy(node_b, load_arg)
813                  continue
814  
815              # For both start_node and end_node verify that we know how to do
816              # the dtype cast. If we do not, skip.
817              (
818                  node_input_type_a,
819                  node_output_type_a,
820              ) = get_node_first_input_and_output_type(
821                  subgraph_a.start_node, gm_a, logger_cls, node_type_to_io_type_map
822              )
823              (
824                  node_input_type_b,
825                  node_output_type_b,
826              ) = get_node_first_input_and_output_type(
827                  node_b, gm_b, logger_cls, node_type_to_io_type_map
828              )
829              node_io_types_known_a_and_b = (
830                  node_input_type_a != NodeInputOrOutputType.UNKNOWN
831                  and node_output_type_a != NodeInputOrOutputType.UNKNOWN
832                  and node_input_type_b != NodeInputOrOutputType.UNKNOWN
833                  and node_output_type_b != NodeInputOrOutputType.UNKNOWN
834              )
835              if not node_io_types_known_a_and_b:
836                  print(
837                      f"skipping shadow loggers for node_b: {get_target_type_str(node_b, gm_b)}"
838                      + f", start_node_a: {get_target_type_str(subgraph_a.start_node, gm_a)}"
839                      + ", unknown dtype cast"
840                  )
841                  env_c[node_b.name] = graph_c.node_copy(node_b, load_arg)
842                  continue
843  
844              # If we are shadowing from fp32 to int8, we need to insert
845              # quantize_per_tensor call with qparams from the previous node.
846              # Only do this if we are able to infer these qparams from the graph.
847              if (
848                  node_input_type_a == NodeInputOrOutputType.INT8
849                  and node_input_type_b == NodeInputOrOutputType.FP32
850              ):
851                  node_a_input_qparams = get_node_input_qparams(
852                      subgraph_a.start_node, gm_a, node_type_to_io_type_map
853                  )
854                  if not node_a_input_qparams:
855                      print(
856                          f"skipping shadow loggers for node_b: {get_target_type_str(node_b, gm_b)}"
857                          + f", start_node_a: {get_target_type_str(subgraph_a.start_node, gm_a)}"
858                          + ", unknown input qparams"
859                      )
860                      env_c[node_b.name] = graph_c.node_copy(node_b, load_arg)
861                      continue
862  
863              num_non_param_args_node_a = get_number_of_non_param_args(
864                  subgraph_a.start_node, gm_a
865              )
866              if not _can_insert_copy_of_subgraph_a(
867                  subgraph_a, gm_a, num_non_param_args_node_a
868              ):
869                  print(
870                      f"skipping shadow loggers for node_b: {get_target_type_str(node_b, gm_b)}"
871                      + f", start_node_a: {get_target_type_str(subgraph_a.start_node, gm_a)}"
872                      + ", unhandled logic in subgraph copy"
873                  )
874                  env_c[node_b.name] = graph_c.node_copy(node_b, load_arg)
875                  continue
876  
877              fqn_base_a = _maybe_get_fqn(subgraph_a.base_op_node, gm_a)
878              fqn_base_b = _maybe_get_fqn(subgraph_b.base_op_node, gm_b)  # type: ignore[possibly-undefined]
879  
880              if node_b_is_start_node:
881                  # if necessary, log the input of node_c
882                  if should_log_inputs:
883                      prev_node_b = get_normalized_nth_input(node_b, gm_b, 0)
884                      if isinstance(prev_node_b, Node):
885                          prev_node_c = env_c[prev_node_b.name]
886                          env_c[prev_node_c.name] = _insert_logger_after_node(
887                              prev_node_c,
888                              gm_b,
889                              logger_cls,
890                              "_ns_logger_b_inp_",
891                              node_b.name,
892                              name_b,
893                              ref_name,
894                              ref_node_type_b,
895                              NSSingleResultValuesType.NODE_INPUT.value,
896                              index_within_arg=0,
897                              index_of_arg=0,
898                              fqn=fqn_base_b,
899                          )
900                      elif isinstance(prev_node_b, list):
901                          # first, save the prev_node instances, because they
902                          # will be overwritten in the env after the first logger
903                          # is added
904                          prev_node_c_list = [env_c[arg.name] for arg in prev_node_b]
905  
906                          for arg_idx, arg in enumerate(prev_node_b):
907                              prev_node_c = prev_node_c_list[arg_idx]
908                              env_c[prev_node_c.name] = _insert_logger_after_node(
909                                  prev_node_c,
910                                  gm_b,
911                                  logger_cls,
912                                  "_ns_logger_b_inp_",
913                                  node_b.name,
914                                  name_b,
915                                  ref_name,
916                                  ref_node_type_b,
917                                  NSSingleResultValuesType.NODE_INPUT.value,
918                                  index_within_arg=arg_idx,
919                                  index_of_arg=0,
920                                  fqn=fqn_base_b,
921                              )
922                      else:
923                          # logging of inputs which are not lists is not supported yet
924                          raise AssertionError(
925                              f"type {type(prev_node_b)} is not handled yet"
926                          )
927                  # subgraph so far:
928                  #
929                  # (prev_node_c)+ -> (logger_c_input)?
930  
931              # Note: this if statement is always True, spelling it out to clarify code
932              # intent.
933              if node_b_is_start_node or node_b_is_end_node:
934                  # ensure env_c is populated with base node
935                  env_c[node_b.name] = graph_c.node_copy(node_b, load_arg)
936                  node_c = env_c[node_b.name]
937  
938                  # after this point,
939                  #
940                  # node_a is the original node from graph_a, with parent module gm_a
941                  # node_b is the original node from graph_b, with parent module gm_b
942                  # node_c is the copy of node_b in graph_c
943                  #
944                  # subgraph so far:
945                  #
946                  # (prev_node_c)+ -> (logger_c_input)? -> node_start_c
947  
948              if node_b_is_start_node:
949                  # cast dtype from the dtype of node_c's input to the dtype of
950                  # node_a's input (dequant, etc)
951                  # prev_node_c = node_c.args[0]
952                  prev_node_c = get_normalized_nth_input(node_c, gm_b, 0)  # type: ignore[possibly-undefined]
953                  if should_log_inputs:
954                      # skip the input logger when inserting a dtype cast
955                      if isinstance(prev_node_c, Node):
956                          prev_node_c = get_normalized_nth_input(node_c, gm_b, 0)
957                      elif isinstance(prev_node_c, list):
958                          prev_node_c = [
959                              get_normalized_nth_input(arg, gm_b, 0)
960                              for arg in prev_node_c
961                          ]
962                  dtype_cast_node = _insert_dtype_cast_after_node(
963                      subgraph_a.start_node,
964                      node_c,
965                      prev_node_c,
966                      gm_a,
967                      gm_b,
968                      graph_c,
969                      node_b.name + "_dtype_cast_",
970                      logger_cls,
971                      node_type_to_io_type_map,
972                  )
973                  # note: not inserting to env_c because all nodes which use the dtype
974                  #   casts are copied from graph_a
975                  #
976                  # subgraph so far:
977                  #
978                  #           (dtype_cast_node)+
979                  #                  /
980                  # (prev_node_c)+ -> (logger_c_input)? -> node_start_c
981  
982                  # if input logging is enabled, log the input to the subgraph
983                  if should_log_inputs:
984                      # TODO: explain this
985                      ref_node_name = ""
986                      if isinstance(dtype_cast_node, Node):
987                          dtype_cast_node = _insert_logger_after_node(
988                              dtype_cast_node,
989                              gm_b,
990                              logger_cls,
991                              "_ns_logger_a_inp_",
992                              ref_node_name,
993                              name_a,
994                              ref_name,
995                              ref_node_type_a,
996                              NSSingleResultValuesType.NODE_INPUT.value,
997                              index_within_arg=0,
998                              index_of_arg=0,
999                              fqn=fqn_base_a,
1000                         )
1001                         input_logger: Union[Node, list[Node]] = dtype_cast_node
1002                     else:
1003                         assert isinstance(dtype_cast_node, list)
1004                         new_loggers = []
1005                         for dtype_cast_idx, dtype_cast_node_inner in enumerate(
1006                             dtype_cast_node
1007                         ):
1008                             dtype_cast_logger = _insert_logger_after_node(
1009                                 dtype_cast_node_inner,
1010                                 gm_b,
1011                                 logger_cls,
1012                                 "_ns_logger_a_inp_",
1013                                 ref_node_name,
1014                                 name_a,
1015                                 ref_name,
1016                                 ref_node_type_a,
1017                                 NSSingleResultValuesType.NODE_INPUT.value,
1018                                 index_within_arg=dtype_cast_idx,
1019                                 index_of_arg=0,
1020                                 fqn=fqn_base_a,
1021                             )
1022                             new_loggers.append(dtype_cast_logger)
1023                         dtype_cast_node = new_loggers
1024                         input_logger = dtype_cast_node
1025                     # subgraph so far:
1026                     #
1027                     #       (dtype_cast_node)+ -> (logger_a_input)?
1028                     #                  /
1029                     # prev_node_c -> (logger_c_input)? -> node_start_c
1030 
1031                 # hook up the new mod_a copy to be in the graph, receiving the
1032                 # same inputs as mod_b does, with dtype cast to match a
1033                 # Some ops, such as LSTMs, have two non-param inputs. If we have
1034                 # such an op, pass the second param as well. Note: dtype casting
1035                 # for the second param is not implemented yet, it can be added
1036                 # later if there is a use case.
1037                 node_c_second_non_param_arg = None
1038                 num_non_param_args_node_a = get_number_of_non_param_args(
1039                     subgraph_a.start_node, gm_a
1040                 )
1041                 if num_non_param_args_node_a == 2:
1042                     # node_c_second_non_param_arg = node_c.args[1]
1043                     node_c_second_non_param_arg = get_normalized_nth_input(
1044                         node_c, gm_b, 1
1045                     )
1046                 node_a_shadows_c = _insert_copy_of_subgraph_a_after_input_node_c(
1047                     dtype_cast_node,
1048                     node_c_second_non_param_arg,
1049                     subgraph_a,
1050                     gm_a,
1051                     gm_b,
1052                     node_c.name + "_shadow_copy_",
1053                 )
1054                 env_c[node_a_shadows_c.name] = node_a_shadows_c
1055                 # subgraph so far:
1056                 #
1057                 #       dtype_cast_node -> (logger_a_input)? -> subgraph_a_copy(args/kwargs not shown)
1058                 #                  /
1059                 # (prev_node_c)+ -> (logger_c_input)? -> node_start_c
1060 
1061                 if should_log_inputs:
1062                     # When we created the input logger, we left the ref_node_name
1063                     # as an empty string, because the subgraph copy did not exist
1064                     # yet. Now that the subgraph copy exists, we modify this name
1065                     # to its true value.
1066                     # Note: the alternative to this is to create the input logger
1067                     # after creating the subgraph, which is slightly more
1068                     # complicated. This is the lesser of two evils.
1069                     # input_logger = env_c[dtype_cast_node.name]
1070                     # Find the first node in the subgraph
1071                     cur_node = node_a_shadows_c
1072                     while get_normalized_nth_input(cur_node, gm_b, 0) != input_logger:  # type: ignore[possibly-undefined]
1073                         cur_node = get_normalized_nth_input(cur_node, gm_b, 0)  # type: ignore[assignment]
1074                     if isinstance(input_logger, Node):
1075                         input_logger_mod = getattr(gm_b, input_logger.name)
1076                         input_logger_mod.ref_node_name = cur_node.name
1077                     else:
1078                         assert isinstance(input_logger, list)
1079                         for input_logger_inner in input_logger:
1080                             input_logger_mod = getattr(gm_b, input_logger_inner.name)
1081                             input_logger_mod.ref_node_name = cur_node.name
1082 
1083                 # hook up a logger to the mod_a copy
1084                 env_c[node_a_shadows_c.name] = _insert_logger_after_node(
1085                     env_c[node_a_shadows_c.name],
1086                     gm_b,
1087                     logger_cls,
1088                     "_ns_logger_a_",
1089                     node_a_shadows_c.name,
1090                     name_a,
1091                     ref_name,
1092                     ref_node_type_a,
1093                     NSSingleResultValuesType.NODE_OUTPUT.value,
1094                     index_within_arg=0,
1095                     index_of_arg=0,
1096                     fqn=fqn_base_a,
1097                 )
1098                 # subgraph so far:
1099                 #
1100                 #       dtype_cast_node -> (logger_a_input)? -> subgraph_a_copy -> logger_a
1101                 #                  /
1102                 # (prev_node_c)+ -> (logger_c_input)? -> node_start_c
1103 
1104             if node_b_is_end_node:
1105                 # hook up a logger to the mod_b copy
1106                 env_c[node_b.name] = _insert_logger_after_node(
1107                     env_c[node_b.name],
1108                     gm_b,
1109                     logger_cls,
1110                     "_ns_logger_b_",
1111                     node_b.name,
1112                     name_b,
1113                     ref_name,
1114                     ref_node_type_b,
1115                     NSSingleResultValuesType.NODE_OUTPUT.value,
1116                     index_within_arg=0,
1117                     index_of_arg=0,
1118                     fqn=fqn_base_b,
1119                 )
1120                 # subgraph so far:
1121                 #
1122                 #       dtype_cast_node -> (logger_a_input)? -> subgraph_a_copy -> logger_a
1123                 #                  /
1124                 # (prev_node_c+) -> (logger_c_input)? -> node_start_c -> ... -> node_end_c -> logger_c
1125                 #
1126                 # Note: node_start_c may be the same node as node_end_c, or they
1127                 # may have nodes inbetween.
1128 
1129         else:
1130             env_c[node_b.name] = graph_c.node_copy(node_b, load_arg)
1131 
1132     gm_c = GraphModule(gm_b, graph_c)
1133     return gm_c
