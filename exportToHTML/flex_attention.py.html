54   def _get_mod_type(fn: Callable) -> _ModificationType:
55       """Get the type of modification function. 
56       This function inspects the number of positional arguments of the function to determine 
57       the type of modification function. If the function has 5 positional arguments, it is 
58       considered as a score_mod function. If the function has 4 positional arguments, it is 
59       considered as a mask function. 
60       """
61       num_positional_args = sum(
62           1
63           for param in inspect.signature(fn).parameters.values()
64           if param.default == inspect.Parameter.empty
65       )
66       assert num_positional_args == 5 or num_positional_args == 4
67       if num_positional_args == 5:
68           return _ModificationType.SCORE_MOD
69       elif num_positional_args == 4:
70           return _ModificationType.MASK_MOD
71       else:
72           return _ModificationType.UNKNOWN

76   def _vmap_for_bhqkv(
77       fn: Callable,
78       prefix: tuple[Optional[int], ...],
79       suffix: tuple[Optional[int], ...] = (),
80       out_dims: Union[int, list[Optional[int]]] = 0,
81       group_dim: bool = False,
82   ):
83       """Used to vmap both score_mods and mask_mods over 4-dimensional/5-dimension inputs. 
84       Mapping over the [b, hq, q_idx, kv_idx] or [b, hkv, g, q_idx, kv_idx] dimensions. 
85    
86       Args: 
87           fn (callable): The function to vmap. 
88           prefix (tuple): The prefix of the vmap. For score mod functions, 
89                           this should be set to (0,). For mask_mods = () 
90           suffix (tuple): We need to add (0,) if gradOut is being mapped over, 
91                           and (None,) * len(other_buffers). 
92           out_dims (tuple): For forward cases, keep this as the default 0 since 
93                             we are only returning 1 output. For backwards, the joint 
94                             graph returns grads for B, H, Q_idx, KV_idx and other_buffers, 
95                             so we set this to (0, None, None, None, None) + (None,) * len(other_buffers). 
96    
97       Returns: 
98           callable: The vmapped function. 
99       """
100      # We vamp a function 4 times, broadcasting the [b, h, q_idx, kv_idx] dimensions
101      dimensions: list[tuple[None | int, None | int, None | int, None | int]] = []
102      dimensions = [
103          (None, None, None, 0),
104          (None, None, 0, None),
105          (None, 0, None, None),
106      ]
107  
108      if group_dim:
109          dimensions += [
110              (None, 0, None, None),
111          ]
112  
113      dimensions += [
114          (0, None, None, None),
115      ]
116  
117      for dims in dimensions:
118          fn = torch.vmap(fn, in_dims=prefix + dims + suffix, out_dims=out_dims)  # type: ignore[arg-type]
119      return fn

146  def _ordered_to_dense(num_blocks_in_row: Tensor, col_indices: Tensor):
147      num_rows = col_indices.shape[-2]
148      num_cols = col_indices.shape[-1]
149      batch_dims = num_blocks_in_row.shape[:-1]
150      device = num_blocks_in_row.device
151  
152      def create_dense_one(kv_num_blocks, kv_indices):
153          dense_mask = kv_indices.new_zeros(num_rows, num_cols + 1, dtype=torch.int32)
154  
155          row_indices = torch.arange(num_rows, dtype=torch.int, device=device).unsqueeze(
156              -1
157          )
158          col_range = torch.arange(num_cols, dtype=torch.int, device=device)
159          index_mask = col_range < kv_num_blocks.unsqueeze(-1)
160  
161          # We write to one spot "out of bounds"
162          valid_indices = torch.where(index_mask, kv_indices, num_cols)
163  
164          # set the values in 'a' to 1 where the indices are valid
165          dense_mask[row_indices, valid_indices] = dense_mask.new_ones(())
166          return dense_mask[:, :num_cols].contiguous()
167  
168      create_dense_batched = create_dense_one
169      for _ in range(len(batch_dims)):
170          create_dense_batched = torch.vmap(create_dense_batched, in_dims=(0, 0))
171  
172      out = create_dense_batched(num_blocks_in_row, col_indices)
173      return out

176  def _dense_to_ordered(dense_mask) -> tuple[Tensor, Tensor]:
177      dense_mask = dense_mask.to(dtype=torch.int32)
178      num_blocks_in_row = dense_mask.sum(dim=-1)
179      col_indices = torch.argsort(dense_mask, dim=-1, descending=True, stable=True)
180      return (
181          num_blocks_in_row.to(torch.int32, memory_format=torch.contiguous_format),
182          col_indices.to(torch.int32, memory_format=torch.contiguous_format),
183      )

186  def _transpose_ordered(num_blocks_in_row: Tensor, col_indices: Tensor):
187      dense = _ordered_to_dense(num_blocks_in_row, col_indices)
188      return _dense_to_ordered(dense.transpose(-2, -1))

191  def _adjust_num_blocks_and_indices(
192      num_blocks: Tensor,
193      indices: Tensor,
194      new_num_rows: int,
195      new_num_cols: int,
196  ):
197      indices = indices[:, :, :new_num_rows, :new_num_cols]
198      num_blocks = num_blocks[:, :, :new_num_rows]
199      num_blocks = torch.where(num_blocks < new_num_cols, num_blocks, new_num_cols)
200      num_blocks = torch.sum(indices < num_blocks[:, :, :, None], dim=-1).to(torch.int32)
201      return num_blocks, indices

204  class BlockMask:
205      r""" 
206      BlockMask is our format for representing a block-sparse attention mask. 
207      It is somewhat of a cross in-between BCSR and a non-sparse format. 
208   
209      **Basics** 
210   
211      A block-sparse mask means that instead of representing the sparsity of 
212      individual elements in the mask, a KV_BLOCK_SIZE x Q_BLOCK_SIZE block is 
213      considered sparse only if every element within that block is sparse. 
214      This aligns well with hardware, which generally expects to perform 
215      contiguous loads and computation. 
216   
217      This format is primarily optimized for 1. simplicity, and 2. kernel 
218      efficiency. Notably, it is *not* optimized for size, as this mask is always 
219      reduced by a factor of KV_BLOCK_SIZE * Q_BLOCK_SIZE. If the size is a 
220      concern, the tensors can be reduced in size by increasing the block size. 
221   
222      The essentials of our format are: 
223   
224      num_blocks_in_row: Tensor[ROWS]: 
225      Describes the number of blocks present in each row. 
226   
227      col_indices: Tensor[ROWS, MAX_BLOCKS_IN_COL]: 
228      `col_indices[i]` is the sequence of block positions for row i. The values of 
229      this row after `col_indices[i][num_blocks_in_row[i]]` are undefined. 
230   
231      For example, to reconstruct the original tensor from this format: 
232   
233      .. code-block:: python 
234   
235          dense_mask = torch.zeros(ROWS, COLS) 
236          for row in range(ROWS): 
237              for block_idx in range(num_blocks_in_row[row]): 
238                  dense_mask[row, col_indices[row, block_idx]] = 1 
239   
240      Notably, this format makes it easier to implement a reduction along the 
241      *rows* of the mask. 
242   
243      **Details** 
244   
245      The basics of our format require only kv_num_blocks and kv_indices. But, we 
246      have up to 8 tensors on this object. This represents 4 pairs: 
247   
248      1. (kv_num_blocks, kv_indices): Used for the forwards pass of attention, as 
249      we reduce along the KV dimension. 
250   
251      2. [OPTIONAL] (full_kv_num_blocks, full_kv_indices): This is optional and 
252      purely an optimization. As it turns out, applying masking to every block 
253      is quite expensive! If we specifically know which blocks are "full" and 
254      don't require masking at all, then we can skip applying mask_mod to these 
255      blocks. This requires the user to split out a separate mask_mod from the 
256      score_mod. For causal masks, this is about a 15% speedup. 
257   
258      3. [GENERATED] (q_num_blocks, q_indices): Required for the backwards pass, 
259      as computing dKV requires iterating along the mask along the Q dimension. These are autogenerated from 1. 
260   
261      4. [GENERATED] (full_q_num_blocks, full_q_indices): Same as above, but for 
262      the backwards pass. These are autogenerated from 2. 
263      """
264  
265      seq_lengths: tuple[int, int]

317      def from_kv_blocks(
318          cls,
319          kv_num_blocks: Tensor,
320          kv_indices: Tensor,
321          full_kv_num_blocks: Optional[Tensor] = None,
322          full_kv_indices: Optional[Tensor] = None,
323          BLOCK_SIZE: Union[int, tuple[int, int]] = _DEFAULT_SPARSE_BLOCK_SIZE,
324          mask_mod: Optional[_mask_mod_signature] = None,
325          seq_lengths: Optional[tuple[int, int]] = None,
326      ):
327          """ 
328          Creates a BlockMask instance from key-value block information. 
329   
330          Args: 
331              kv_num_blocks (Tensor): Number of kv_blocks in each Q_BLOCK_SIZE row tile. 
332              kv_indices (Tensor): Indices of key-value blocks in each Q_BLOCK_SIZE row tile. 
333              full_kv_num_blocks (Optional[Tensor]): Number of full kv_blocks in each Q_BLOCK_SIZE row tile. 
334              full_kv_indices (Optional[Tensor]): Indices of full key-value blocks in each Q_BLOCK_SIZE row tile. 
335              BLOCK_SIZE (Union[int, tuple[int, int]]): Size of KV_BLOCK_SIZE x Q_BLOCK_SIZE tiles. 
336              mask_mod (Optional[Callable]): Function to modify the mask. 
337   
338          Returns: 
339              BlockMask: Instance with full Q information generated via _transposed_ordered 
340   
341          Raises: 
342              RuntimeError: If kv_indices has < 2 dimensions. 
343              AssertionError: If only one of full_kv_* args is provided. 
344          """
345          if kv_indices.dim() < 2:
346              raise RuntimeError("BlockMask must have at least 2 dimensions")
347  
348          assert (full_kv_num_blocks is None) == (full_kv_indices is None), (
349              "full_kv_num_blocks and full_kv_indices must be both provided or omitted"
350          )
351  
352          # Generate q_num_blocks and q_indices
353          q_num_blocks, q_indices = _transpose_ordered(kv_num_blocks, kv_indices)
354          if full_kv_num_blocks is not None:
355              assert full_kv_indices is not None
356              full_q_num_blocks, full_q_indices = _transpose_ordered(
357                  full_kv_num_blocks, full_kv_indices
358              )
359          else:
360              full_q_num_blocks, full_q_indices = None, None
361  
362          if isinstance(BLOCK_SIZE, int):
363              BLOCK_SIZE = (BLOCK_SIZE, BLOCK_SIZE)
364  
365          mask_mod = mask_mod if mask_mod is not None else noop_mask
366          if seq_lengths is None:
367              q_length = kv_indices.shape[-2] * BLOCK_SIZE[0]
368              kv_length = q_indices.shape[-2] * BLOCK_SIZE[1]
369              seq_lengths = (q_length, kv_length)
370  
371          return cls(
372              seq_lengths=seq_lengths,
373              kv_num_blocks=kv_num_blocks,
374              kv_indices=kv_indices,
375              full_kv_num_blocks=full_kv_num_blocks,
376              full_kv_indices=full_kv_indices,
377              q_num_blocks=q_num_blocks,
378              q_indices=q_indices,
379              full_q_num_blocks=full_q_num_blocks,
380              full_q_indices=full_q_indices,
381              BLOCK_SIZE=BLOCK_SIZE,
382              mask_mod=mask_mod,
383          )

507      def _adjust(self, new_q_len: int, new_kv_len: int):
508          new_num_rows = (new_q_len + self.BLOCK_SIZE[0] - 1) // self.BLOCK_SIZE[0]

536      def numel(self):
537          """Returns the number of elements (not accounting for sparsity) in the mask."""
538          shape = self.shape
539  
540          def _prod(xs):
541              return functools.reduce(operator.mul, xs, 1)
542  
543          return _prod(shape)

545      def sparsity(self) -> float:
546          """Computes the percentage of blocks that are sparse (i.e. not computed)"""
547          total_size = self.numel()
548          computed_blocks = self.kv_num_blocks.sum()
549          if self.full_kv_num_blocks is not None:
550              computed_blocks += self.full_kv_num_blocks.sum()
551  
552          computed_size = computed_blocks.item() * self.BLOCK_SIZE[0] * self.BLOCK_SIZE[1]
553          dense_ratio = computed_size / total_size
554          return 100 * (1 - dense_ratio)

556      def to_dense(self) -> Tensor:
557          """Returns a dense block that is equivalent to the block mask."""
558          partial_dense = _ordered_to_dense(self.kv_num_blocks, self.kv_indices)
559          if self.full_kv_num_blocks is not None:
560              assert self.full_kv_indices is not None
561              return partial_dense | _ordered_to_dense(
562                  self.full_kv_num_blocks, self.full_kv_indices
563              )
564          return partial_dense

587              vis = ", ".join(reversed(descriptors)) + "\n"
588  
589              def summarize_section(section):
590                  percentage = section.float().mean().item()
591                  if percentage == 1:
592                      return "█"
593                  elif percentage == 0:
594                      return " "
595                  else:
596                      return "░"
597  
598              def cdiv(a, b):
599                  return (a + (b - 1)) // b
600  
601              row_step = max(1, cdiv(num_rows, max_rows))
602              col_step = max(1, cdiv(num_cols, max_cols))
603  
604              for r in range(0, num_rows, row_step):
605                  for c in range(0, num_cols, col_step):
606                      cur_mask = dense_mask
607                      for idx in batch_idx:
608                          cur_mask = cur_mask[idx]
609                      char = summarize_section(
610                          cur_mask[r : r + row_step, c : c + col_step]
611                      )
612                      vis += char * 2
613                  vis += "\n"
614              return vis
615  
616          total_vis = []
617          for idx, batch_idx in enumerate(
618              itertools.product(*[range(i) for i in batch_dims])
619          ):
620              if idx == limit:
621                  total_vis.append("...")
622                  total_vis.append("To print out more, set BlockMask.to_string(limit=N)")
623                  total_vis.append(
624                      "You can also index (BlockMask[batch, head]) to choose a specific batch or head"
625                  )
626                  break
627              block_vis = create_block_vis(*batch_idx)
628              total_vis.append(block_vis)
629  
630          return "\n".join(total_vis)

667  def _convert_mask_to_block_mask(
668      mask: Tensor,
669      Q_BLOCK_SIZE=_DEFAULT_SPARSE_BLOCK_SIZE,
670      KV_BLOCK_SIZE=_DEFAULT_SPARSE_BLOCK_SIZE,
671      separate_full_blocks: bool = False,
672  ) -> tuple[Tensor, Optional[Tensor]]:
673      assert mask.dtype == torch.bool
674      mask = _broadcast_to_dim(mask, 4)
675  
676      def padding_needed_for_multiple(x, multiple):
677          return _round_up_to_multiple(x, multiple) - x
678  
679      mask = torch.nn.functional.pad(
680          mask,
681          (
682              0,
683              padding_needed_for_multiple(mask.shape[-1], KV_BLOCK_SIZE),
684              0,
685              padding_needed_for_multiple(mask.shape[-2], Q_BLOCK_SIZE),
686          ),
687      )
688      B, H, Q, KV = mask.shape
689      assert Q % Q_BLOCK_SIZE == 0
690      assert KV % KV_BLOCK_SIZE == 0
691      mask = mask.view(
692          B, H, Q // Q_BLOCK_SIZE, Q_BLOCK_SIZE, KV // KV_BLOCK_SIZE, KV_BLOCK_SIZE
693      )  # [B, H, Q//Q_BLOCK_SIZE, Q_BLOCK_SIZE, KV//KV_BLOCK_SIZE, KV_BLOCK_SIZE]
694      mask = mask.permute(
695          0, 1, 2, 4, 3, 5
696      )  # [B, H, Q//Q_BLOCK_SIZE, KV//KV_BLOCK_SIZE, Q_BLOCK_SIZE, KV_BLOCK_SIZE]
697      mask_block_sum = mask.sum(
698          dim=[-2, -1]
699      )  # [B, H, Q//Q_BLOCK_SIZE, KV//KV_BLOCK_SIZE]
700      if separate_full_blocks:
701          full_block_sum = Q_BLOCK_SIZE * KV_BLOCK_SIZE
702          full_blocks = mask_block_sum == full_block_sum
703          partial_blocks = (mask_block_sum > 0) & (mask_block_sum < full_block_sum)
704          partial_blocks = partial_blocks.to(dtype=torch.int8)
705          full_blocks = full_blocks.to(dtype=torch.int8)
706          return partial_blocks, full_blocks
707      else:
708          partial_blocks = mask_block_sum > 0
709          partial_blocks = partial_blocks.to(dtype=torch.int8)
710          return partial_blocks, None

713  def or_masks(*mask_mods: _mask_mod_signature) -> _mask_mod_signature:
714      """Returns a mask_mod that's the union of provided mask_mods"""
715      if not all(callable(arg) for arg in mask_mods):
716          raise RuntimeError(f"All inputs should be callable mask_mods: {mask_mods}")
717  
718      def or_mask(b, h, q_idx, kv_idx):
719          result = b.new_zeros((), dtype=torch.bool)
720          for mask in mask_mods:
721              result = result | mask(b, h, q_idx, kv_idx)
722          return result
723  
724      return or_mask

727  def and_masks(*mask_mods: _mask_mod_signature) -> _mask_mod_signature:
728      """Returns a mask_mod that's the intersection of provided mask_mods"""
729      if not all(callable(arg) for arg in mask_mods):
730          raise RuntimeError(f"All inputs should be callable mask_mods: {mask_mods}")
731  
732      def and_mask(b, h, q_idx, kv_idx):

734          for mask in mask_mods:
735              result = result & mask(b, h, q_idx, kv_idx)

738      return and_mask

783  def create_mask(
784      mod_fn: Union[_score_mod_signature, _mask_mod_signature],
785      B: Optional[int],
786      H: Optional[int],
787      Q_LEN: int,
788      KV_LEN: int,
789      device: DeviceLikeType = "cuda",
790  ) -> Tensor:
791      r"""This function creates a mask tensor from a mod_fn function. 
792   
793      Args: 
794          mod_fn (Union[_score_mod_signature, _mask_mod_signature]): Function to modify attention scores. 
795          B (int): Batch size. 
796          H (int): Number of query heads. 
797          Q_LEN (int): Sequence length of query. 
798          KV_LEN (int): Sequence length of key/value. 
799          device (str): Device to run the mask creation on. 
800   
801      Returns: 
802          mask (Tensor): A mask tensor with shape (B, H, M, N). 
803      """
804      if B is None:
805          B = 1
806      if H is None:
807          H = 1
808      b = torch.arange(0, B, device=device)
809      h = torch.arange(0, H, device=device)
810      m = torch.arange(0, Q_LEN, device=device)
811      n = torch.arange(0, KV_LEN, device=device)
812      mod_type = _get_mod_type(mod_fn)
813  
814      from torch._dynamo._trace_wrapped_higher_order_op import TransformGetItemToIndex
815  
816      with TransformGetItemToIndex():
817          if mod_type == _ModificationType.SCORE_MOD:
818              score_mod = mod_fn
819              score_mod = _vmap_for_bhqkv(score_mod, prefix=(0,))  # first input is score
820              out = score_mod(torch.zeros(B, H, Q_LEN, KV_LEN, device=device), b, h, m, n)
821              mask = torch.where(torch.isneginf(out), False, True)
822              return mask
823          elif mod_type == _ModificationType.MASK_MOD:
824              mask_mod = mod_fn
825              mask_mod = _vmap_for_bhqkv(mask_mod, prefix=())
826              mask = mask_mod(b, h, m, n)
827              return mask
828          else:
829              raise AssertionError

832  def create_block_mask(
833      mask_mod: _mask_mod_signature,
834      B: Optional[int],
835      H: Optional[int],
836      Q_LEN: int,
837      KV_LEN: int,
838      device: DeviceLikeType = "cuda",
839      BLOCK_SIZE: Union[int, tuple[int, int]] = _DEFAULT_SPARSE_BLOCK_SIZE,
840      _compile=False,
841  ) -> BlockMask:
842      r"""This function creates a block mask tuple from a mask_mod function. 
843   
844      Args: 
845          mask_mod (Callable): mask_mod function. This is a callable that defines the 
846              masking pattern for the attention mechanism. It takes four arguments: 
847              b (batch size), h (number of heads), q_idx (query index), and kv_idx (key/value index). 
848              It should return a boolean tensor indicating which attention connections are allowed (True) 
849              or masked out (False). 
850          B (int): Batch size. 
851          H (int): Number of query heads. 
852          Q_LEN (int): Sequence length of query. 
853          KV_LEN (int): Sequence length of key/value. 
854          device (str): Device to run the mask creation on. 
855          BLOCK_SIZE (int or tuple[int, int]): Block size for the block mask. If a single int is provided it is used for both query and key/value. 
856   
857      Returns: 
858          BlockMask:  A BlockMask object that contains the block mask information. 
859   
860      Example Usage: 
861          .. code-block:: python 
862   
863              def causal_mask(b, h, q_idx, kv_idx): 
864                  return q_idx >= kv_idx 
865   
866   
867              block_mask = create_block_mask(causal_mask, 1, 1, 8192, 8192, device="cuda") 
868              query = torch.randn(1, 1, 8192, 64, device="cuda", dtype=torch.float16) 
869              key = torch.randn(1, 1, 8192, 64, device="cuda", dtype=torch.float16) 
870              value = torch.randn(1, 1, 8192, 64, device="cuda", dtype=torch.float16) 
871              output = flex_attention(query, key, value, block_mask=block_mask) 
872      """
873      mod_type = _get_mod_type(mask_mod)
874      assert mod_type == _ModificationType.MASK_MOD, (
875          f"create-block_mask requires a mask_mod function! Got {mask_mod}"
876      )
877      if B is None:
878          B = 1
879      if H is None:
880          H = 1
881      if isinstance(BLOCK_SIZE, int):
882          Q_BLOCK_SIZE = BLOCK_SIZE
883          KV_BLOCK_SIZE = BLOCK_SIZE
884      else:
885          Q_BLOCK_SIZE, KV_BLOCK_SIZE = BLOCK_SIZE
886  
887      if _compile:
888          warnings.warn(
889              "_compile flag on create_block_mask was originally added to work around a torch.compile limitation. That limitation has since been addressed. So, to compile create_block_mask, we suggest doing torch.compile(create_block_mask). This still works for now, but will be removed in the future.",
890              DeprecationWarning,
891          )
892          return torch.compile(create_block_mask)(
893              mask_mod, B, H, Q_LEN, KV_LEN, device, BLOCK_SIZE
894          )
895  
896      mask_tensor = create_mask(mask_mod, B, H, Q_LEN, KV_LEN, device)
897      partial_block_mask, full_block_mask = _convert_mask_to_block_mask(
898          mask_tensor,
899          Q_BLOCK_SIZE=Q_BLOCK_SIZE,
900          KV_BLOCK_SIZE=KV_BLOCK_SIZE,
901          separate_full_blocks=True,
902      )
903      block_mask = _create_sparse_block_from_block_mask(
904          (partial_block_mask, full_block_mask),
905          mask_mod,
906          (Q_LEN, KV_LEN),
907          Q_BLOCK_SIZE,
908          KV_BLOCK_SIZE,
909      )
910      return block_mask

928  def _nested_mod_func_adapter(
929      orig_mod_func: Union[_score_mod_signature, _mask_mod_signature],
930      q_nt: torch.Tensor,
931      kv_nt: torch.Tensor,
932      is_score_mod: bool,
933  ) -> Union[_score_mod_signature, _mask_mod_signature]:
934      r"""Adapter to convert a score_mod / mask_mod to be NJT-compatible. The given mod func 
935      should be written as if operating over a single sequence at a item. This adapter will 
936      handle conversion from indices operating over a "stacked sequence" of length ``sum(S)`` 
937      for sequence length ``S`` in the NJT to "sequence relative" indices in range ``[0, S)``. 
938   
939      Args: 
940          orig_mod_func (Callable): Function to modify attention scores. It takes four or five 
941              arguments, depending on whether a mask_mod or score_mod func is passed. 
942          q_nt (torch.Tensor): Jagged layout nested tensor (NJT) that defines the sequence length 
943              structure for query. 
944          kv_nt (torch.Tensor): Jagged layout nested tensor (NJT) that defines the sequence length 
945              structure for key / value. 
946          is_score_mod (bool): Indicates whether the mod function is a score_mod. 
947   
948      Returns: 
949          nt_score_mod: An NJT-compatible version of orig_score_mod 
950      """
951  
952      # Used to convert indices within the "stacked" sequence (range [0, sum(*)))
953      # to "sequence local" indices (range [0, S) for each S).
954      def _build_seq_idx(offsets, total_length):
955          range_tensor = torch.arange(
956              total_length, device=offsets.device, dtype=torch.int32
957          )
958  
959          # Use searchsorted to find the index for each position
960          # NB: This assumes offsets[0] to offsets[-1] spans the packed dim of values.
961          # If we ever loosen this restriction, this logic will need to be updated.
962          seq_idx = torch.searchsorted(offsets, range_tensor, right=True) - 1
963          return seq_idx
964  
965      q_offsets = q_nt._offsets  # type: ignore[attr-defined]
966      kv_offsets = kv_nt._offsets  # type: ignore[attr-defined]
967      q_seq_idx = _build_seq_idx(q_offsets, q_nt._values.shape[q_nt._ragged_idx - 1])  # type: ignore[attr-defined]
968      if q_nt is kv_nt:
969          kv_seq_idx = q_seq_idx
970      else:
971          # cross attention case
972          kv_seq_idx = _build_seq_idx(
973              kv_offsets,
974              kv_nt._values.shape[kv_nt._ragged_idx - 1],  # type: ignore[attr-defined]
975          )
976  
977      # Converts q_idx / kv_idx from [0, total_length) -> [0, S), where S refers
978      # to the sequence length for each sequence in the NJT, for use in given
979      # score_mod. This allows the user to write a score_mod as if it were
980      # operating on a single sequence and the "stacked sequence" is split
981      # automatically into individual sequences for them.
982      if is_score_mod:
983  
984          def nt_score_mod(score, b, h, q_idx, kv_idx):
985              b_nested = q_seq_idx[q_idx]
986              q_nested = q_idx - q_offsets[q_seq_idx[q_idx]]
987              kv_nested = kv_idx - kv_offsets[kv_seq_idx[kv_idx]]
988              is_same_sequence = q_seq_idx[q_idx] == kv_seq_idx[kv_idx]
989              return torch.where(
990                  is_same_sequence,
991                  orig_mod_func(score, b_nested, h, q_nested, kv_nested),  # type: ignore[call-arg]
992                  # don't allow inter-sequence attention
993                  float("-inf"),
994              )
995  
996          return nt_score_mod
997      else:
998  
999          def nt_mask_mod(b, h, q_idx, kv_idx):
1000             b_nested = q_seq_idx[q_idx]
1001             q_nested = q_idx - q_offsets[q_seq_idx[q_idx]]
1002             kv_nested = kv_idx - kv_offsets[kv_seq_idx[kv_idx]]
1003             # don't allow inter-sequence attention
1004             is_same_sequence = q_seq_idx[q_idx] == kv_seq_idx[kv_idx]
1005             return orig_mod_func(b_nested, h, q_nested, kv_nested) & is_same_sequence  # type: ignore[call-arg]
1006 
1007         return nt_mask_mod

1012     B: Optional[int],
1013     H: Optional[int],

1016     BLOCK_SIZE: Union[int, tuple[int, int]] = _DEFAULT_SPARSE_BLOCK_SIZE,

1018 ) -> BlockMask:
1019     r"""This function creates a nested tensor compatible block mask tuple from a mask_mod 
1020     function. The returned BlockMask will be on the device specified by the input nested tensor. 
1021  
1022     Args: 
1023         mask_mod (Callable): mask_mod function. This is a callable that defines the

1032             sequence" of length ``sum(S)`` for sequence length ``S`` from the NJT. 
1033         kv_nt (torch.Tensor): Jagged layout nested tensor (NJT) that defines the sequence length 
1034             structure for key / value, allowing for cross attention. The block mask will be 
1035             constructed to operate on a "stacked sequence" of length ``sum(S)`` for sequence

1063  
1064             # shape (B, num_heads, seq_len*, D) where seq_len* varies across the batch 
1065             query = torch.nested.nested_tensor(..., layout=torch.jagged) 
1066             key = torch.nested.nested_tensor(..., layout=torch.jagged) 
1067             value = torch.nested.nested_tensor(..., layout=torch.jagged) 
1068  
1069  
1070             def causal_mask(b, h, q_idx, kv_idx): 
1071                 return q_idx >= kv_idx

1073  
1074             # cross attention case: pass both query and key/value NJTs

1082         kv_nt = q_nt

1085             "create_nested_block_mask(): Expected q_nt and kv_nt to be on the same device"
1086         )
1087     return create_block_mask(
1088         _nested_mod_func_adapter(mask_mod, q_nt, kv_nt, is_score_mod=False),  # type: ignore[arg-type]
1089         B,
1090         H,
1091         q_nt._values.shape[q_nt._ragged_idx - 1],  # type: ignore[attr-defined]
1092         kv_nt._values.shape[kv_nt._ragged_idx - 1],  # type: ignore[attr-defined]

1094         # compile is important so we don't materialize a mask_tensor of
1095         # shape (1, 1, total_seqlen, total_seqlen)
1096         BLOCK_SIZE=BLOCK_SIZE,
1097         _compile=_compile,
1098     )

1101 def _apply_kernel_options(
1102     query: Tensor, key: Tensor, value: Tensor, return_lse: bool, kernel_options
1103 ):
1104     kernel_options = {} if kernel_options is None else dict(kernel_options)
1105

1107     kernel_options.setdefault("ROWS_GUARANTEED_SAFE", False)
1108     kernel_options.setdefault("BLOCKS_ARE_CONTIGUOUS", False)
1109     # This forces all biases grad scatters to be done in the DQ iteration loop of the backwards
1110     kernel_options.setdefault("WRITE_DQ", True)

1113     assert "OUTPUT_LOGSUMEXP" not in kernel_options
1114     kernel_options["OUTPUT_LOGSUMEXP"] = True
1115     if not return_lse:
1116         # We used to check if q,k,v required grads but since captured buffers can require grad

1120         any_inputs_on_cpu_device = (
1121             query.device.type == "cpu"
1122             or key.device.type == "cpu"
1123             or value.device.type == "cpu"
1124         )
1125         if any_inputs_on_cpu_device:
1126             # CPU with torch.compile now supports infernece, and will not return lse
1127             # TODO: support CPU for training and return lse
1128             kernel_options["OUTPUT_LOGSUMEXP"] = False

1175 def _enforce_mem_layouts(
1176     query: Tensor, key: Tensor, value: Tensor
1177 ) -> tuple[torch.Tensor, torch.Tensor, torch.Tensor]:
1178     """ 
1179     Enforce memory layouts for query, key, and value tensors. 
1180  
1181     For non-FP8 dtypes, no action is taken. 
1182  
1183     For FP8 dtypes, we enforce the following memory layouts: 
1184     - Query tensor must be in row-major memory layout, as it will be the left-operand in the FP8 GEMM `q @ k.T`. 
1185     - Key tensor must be in row-major memory layout, as it will be transposed when used as the right-operand 
1186       in the FP8 GEMM `q @ k.T`, meaning it will correctly be in column-major memory layout for the GEMM. 
1187     - Value tensor must be in column-major memory layout, as it will be the right-operand in the FP8 GEMM `softmax_scores @ v`. 
1188  
1189     Returns the query, key, and value tensors with the enforced memory layouts. 
1190     """
1191 
1192     def is_row_major(tensor: Tensor) -> bool:
1193         return tensor.stride()[-1] == 1
1194 
1195     def is_col_major(tensor: Tensor) -> bool:
1196         return tensor.stride()[-2] == 1
1197 
1198     # These memory layout constraint are only for FP8 GEMMs on NVIDIA GPU architectures >= SM89 and < SM100.
1199     # This is because GPU arch < SM89 does not not support FP8 GEMMs, and
1200     # SM100 has support for TN, NT, TT, NN layouts for FP8 GEMMs
1201     # (i.e., left and right operands can be in row or column major layouts)
1202     # so this check is only needed for older architectures.
1203     # See: https://github.com/NVIDIA/cutlass/blob/main/media/docs/cpp/blackwell_functionality.md
1204     fp8_dtypes = (
1205         torch.float8_e4m3fn,
1206         torch.float8_e5m2,
1207     )
1208     gemm_precision = query.dtype
1209 
1210     should_enforce_mem_layout = (
1211         gemm_precision in fp8_dtypes
1212         and torch.version.cuda is not None
1213         and torch.cuda.get_device_capability("cuda") >= (8, 9)
1214         and torch.cuda.get_device_capability("cuda") < (10, 0)
1215     )
1216     if not should_enforce_mem_layout:
1217         return query, key, value
1218 
1219     # Query must be in row-major memory layout as the left-operand in the FP8 GEMM `q @ k.T`
1220     if not is_row_major(query):
1221         query = query.contiguous()
1222 
1223     # Key must be in row-major memory layout as it will be transposed when used as the right-operand
1224     # in the FP8 GEMM `q @ k.T`, meaning it will correctly be in column-major memory layout for the GEMM.
1225     if not is_row_major(key):
1226         key = key.contiguous()
1227 
1228     # Value must be in column-major memory layout as the right-operand in the FP8 GEMM `softmax_scores @ v`
1229     if not is_col_major(value):
1230         value = value.transpose(-2, -1).contiguous().transpose(-2, -1)
1231     return query, key, value

1248     attention score modification function. The attention score modification function will be applied after the attention 
1249     scores have been calculated between the query and key tensors. The attention scores are calculated as follows: 
1250  
1251     The ``score_mod`` function should have the following signature: 
1252  
1253     .. code-block:: python 
1254  
1255         def score_mod( 
1256             score: Tensor, 
1257             batch: Tensor, 
1258             head: Tensor, 
1259             q_idx: Tensor, 
1260             k_idx: Tensor 
1261         ) -> Tensor: 
1262  
1263     Where: 
1264         - ``score``: A scalar tensor representing the attention score, 
1265           with the same data type and device as the query, key, and value tensors. 
1266         - ``batch``, ``head``, ``q_idx``, ``k_idx``: Scalar tensors indicating 
1267           the batch index, query head index, query index, and key/value index, respectively. 
1268           These should have the ``torch.int`` data type and be located on the same device as the score tensor. 
1269  
1270     Args: 
1271         query (Tensor): Query tensor; shape :math:`(B, Hq, L, E)`. For FP8 dtypes, should be in row-major memory layout for optimal performance. 
1272         key (Tensor): Key tensor; shape :math:`(B, Hkv, S, E)`. For FP8 dtypes, should be in row-major memory layout for optimal performance. 
1273         value (Tensor): Value tensor; shape :math:`(B, Hkv, S, Ev)`. For FP8 dtypes, should be in column-major memory layout for optimal performance. 
1274         score_mod (Optional[Callable]): Function to modify attention scores. By default no score_mod is applied. 
1275         block_mask (Optional[BlockMask]): BlockMask object that controls the blocksparsity pattern of the attention. 
1276         scale (Optional[float]): Scaling factor applied prior to softmax. If none, the default value is set to :math:`\frac{1}{\sqrt{E}}`. 
1277         enable_gqa (bool): If set to True, enables Grouped Query Attention (GQA) and broadcasts key/value heads to query heads. 
1278         return_lse (bool): Whether to return the logsumexp of the attention scores. Default is False. 
1279         kernel_options (Optional[Dict[str, Any]]): Options to pass into the Triton kernels. 
1280  
1281     Returns: 
1282         output (Tensor): Attention output; shape :math:`(B, Hq, L, Ev)`. 
1283  
1284     Shape legend: 
1285         - :math:`N: \text{Batch size} ... : \text{Any number of other batch dimensions (optional)}` 
1286         - :math:`S: \text{Source sequence length}` 
1287         - :math:`L: \text{Target sequence length}` 
1288         - :math:`E: \text{Embedding dimension of the query and key}` 
1289         - :math:`Ev: \text{Embedding dimension of the value}` 
1290  
1291     .. warning:: 
1292         `torch.nn.attention.flex_attention` is a prototype feature in PyTorch. 
1293         Please look forward to a more stable implementation in a future version of PyTorch.

1295
