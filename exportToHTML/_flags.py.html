1    """Internal flags for ONNX export."""
2    
3    from __future__ import annotations
4    
5    import functools
6    from typing import Any, Callable, cast, TypeVar
7    
8    
9    _is_onnx_exporting = False
10   
11   TCallable = TypeVar("TCallable", bound=Callable[..., Any])
12   
13   
14   def set_onnx_exporting_flag(func: TCallable) -> TCallable:
15       @functools.wraps(func)
16       def wrapper(*args: Any, **kwargs: Any) -> Any:
17           global _is_onnx_exporting
18           _is_onnx_exporting = True
19           try:
20               return func(*args, **kwargs)
21           finally:
22               # Ensure it resets even if an exception occurs
23               _is_onnx_exporting = False
24   
25       return cast(TCallable, wrapper)
