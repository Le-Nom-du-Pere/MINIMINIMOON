45   @compatibility(is_backward_compatible=False)
46   class Scope:
47       """Scope object that records the module path and the module type 
48       of a module. Scope is used to track the information of the module 
49       that contains a Node in a Graph of GraphModule. For example:: 
50    
51           class Sub(torch.nn.Module): 
52               def forward(self, x): 
53                   # This will be a call_method Node in GraphModule, 
54                   # scope for this would be (module_path="sub", module_type=Sub) 
55                   return x.transpose(1, 2) 
56    
57    
58           class M(torch.nn.Module): 
59               def __init__(self) -> None: 
60                   self.sub = Sub() 
61    
62               def forward(self, x): 
63                   # This will be a call_method Node as well, 
64                   # scope for this would be (module_path="", None) 
65                   x = x.transpose(1, 2) 
66                   x = self.sub(x) 
67                   return x 
68    
69       """
70   
71       def __init__(self, module_path: str, module_type: Any):
72           super().__init__()
73           self.module_path = module_path
74           self.module_type = module_type

77   @compatibility(is_backward_compatible=False)
78   class ScopeContextManager:
79       """A context manager to track the Scope of Node during symbolic tracing. 
80       When entering a forward function of a Module, we'll update the scope information of 
81       the current module, and when we exit, we'll restore the previous scope information. 
82       """
83   
84       def __init__(
85           self,
86           scope: Scope,
87           current_scope: Scope,
88       ):
89           super().__init__()
90           # Keep a copy of prev scope to restore on exit
91           self._prev_scope = copy.copy(scope)
92           # Update scope to current scope
93           scope.module_path = current_scope.module_path
94           scope.module_type = current_scope.module_type
95           # Save a reference so we can restore it
96           self._scope = scope
97   
98       def __enter__(self):
99           return self._scope
100  
101      def __exit__(self, *args):
102          self._scope.module_path = self._prev_scope.module_path
103          self._scope.module_type = self._prev_scope.module_type
104          return

125      graph: Graph

127      # Feature flag for mutable schema checking
128      # Enableby default in 1.12
129      check_mutable_operations: bool = False
130      # Feature flag for assert tracing
131      trace_asserts: bool = False
132      # Feature flag for proxying accesses to buffer values
133      proxy_buffer_attributes: bool = False

136      # ``root`` is an instance of ``nn.Module``
137      traced_func_name: str = "forward"
138  
139      # Maps the containing module's name to the operator name
140      scope: Scope
141  
142      # Records the module call stack
143      module_stack: OrderedDict[str, tuple[str, Any]]

146      node_name_to_scope: dict[str, tuple[str, type]]

148      @compatibility(is_backward_compatible=True)
149      def create_node(
150          self,
151          kind: str,
152          target: Target,
153          args: tuple[Argument, ...],
154          kwargs: dict[str, Argument],
155          name: Optional[str] = None,
156          type_expr: Optional[Any] = None,
157      ) -> Node:
158          """ 
159          Inserts a graph node given target, args, kwargs, and name. 
160   
161          This method can be overridden to do extra checking, validation, or 
162          modification of values used in node creation. For example, one might 
163          want to disallow in-place operations from being recorded. 
164          """
165  
166          if kind == "call_function" and self.check_mutable_operations:
167              check_for_mutable_operation(target, args, kwargs)
168  
169          node = self.graph.create_node(kind, target, args, kwargs, name, type_expr)
170          # TODO node_name_to_scope will be depreciated in favor of
171          # node.meta['nn_module_stack']
172          self.node_name_to_scope[node.name] = (
173              self.scope.module_path,
174              self.scope.module_type,
175          )
176  
177          # Optionally set stack trace on the created Node for debugging purposes
178          if fx_traceback.has_preserved_node_meta():
179              current_meta: dict[str, Any] = fx_traceback.get_current_meta()
180  
181              stack_trace = current_meta.get("stack_trace")
182              if stack_trace:
183                  node.stack_trace = stack_trace
184              # Explicitly set the stack_trace, nn_module_stack and source_fn on the node.meta
185              # If other meta fields are needed, they can be added here
186              for field in _COPY_META_FIELDS:
187                  if field in current_meta:
188                      node.meta[field] = copy.copy(current_meta[field])
189  
190              # Here we decrement to account for the sequence_nr having
191              # just been incremented while tracing this lowered aten op.
192              new_seq_nr = torch.autograd._get_sequence_nr() - 1
193              # The sequence_nr increments every time a new autograd Node
194              # is created. During the FWD pass we store the sequence_nr
195              # corresponding to the last autograd Node created on this fx
196              # node's meta.  A single aten op can create multiple autograd
197              # nodes as is the case with in-place foreach ops. During the
198              # BWD pass we retrieve the sequence_nr stored on the current
199              # executing autograd Node. See NOTE [ Sequence Number ].
200              if current_meta.get("in_grad_fn", 0) > 0:
201                  new_seq_nr = current_meta["grad_fn_seq_nr"][-1]
202              node.meta["seq_nr"] = new_seq_nr
203  
204          elif self.module_stack:
205              node.meta["nn_module_stack"] = copy.copy(self.module_stack)
206  
207          log.debug("create_node %s", node)
208          return node

214      @compatibility(is_backward_compatible=True)
215      def create_proxy(
216          self,
217          kind: str,
218          target: Target,
219          args: tuple[Any, ...],
220          kwargs: dict[str, Any],
221          name: Optional[str] = None,
222          type_expr: Optional[Any] = None,
223          # fix noqa when updating bc tests
224          proxy_factory_fn: Callable[[Node], "Proxy"] = None,  # noqa: RUF013
225      ):
226          """ 
227          Create a Node from the given arguments, then return the Node 
228          wrapped in a Proxy object. 
229   
230          If kind = 'placeholder', then we're creating a Node that 
231          represents the parameter of a function. If we need to encode 
232          a default parameter, we use the ``args`` tuple. ``args`` is 
233          otherwise empty for ``placeholder`` Nodes. 
234          """
235  
236          args_ = self.create_arg(args)
237          kwargs_ = self.create_arg(kwargs)
238          assert isinstance(args_, tuple)
239          assert isinstance(kwargs_, dict)
240  
241          node = self.create_node(kind, target, args_, kwargs_, name, type_expr)
242  
243          if not proxy_factory_fn:
244              proxy = self.proxy(node)
245          else:
246              proxy = proxy_factory_fn(node)
247  
248          if self.record_stack_traces and not proxy.node.stack_trace:
249              from torch.fx.experimental.symbolic_shapes import uninteresting_files
250  
251              user_frame_summary = CapturedTraceback.extract().summary()
252              if user_frame_summary:
253                  first_forward = -1
254                  for i, frame in enumerate(user_frame_summary):
255                      if frame.name == "forward":
256                          user_frame_summary = user_frame_summary[i:]
257                          first_forward = i
258                          break
259  
260                  # Not having a "forward" call in the stacktrace implies the
261                  # stacktrace will probably be irrelevant
262                  if first_forward == -1:
263                      user_frame_summary = []
264  
265                  stack_trace = [
266                      frame
267                      for frame in user_frame_summary
268                      if frame.filename not in uninteresting_files()
269                  ]
270                  stack_trace = traceback.StackSummary.from_list(stack_trace)
271                  proxy.node.stack_trace = "".join(stack_trace.format()).strip()
272  
273          return proxy

275      def _find_user_frame(self):
276          """ 
277          Find the Python stack frame executing the user code during 
278          symbolic tracing. 
279          """
280          # We have to do a little dance here. Basically, walk up the callstack and
281          # record the first frame not in the pytorch source. This is the frame executing
282          # the user code during tracing.
283          frame = inspect.currentframe()
284  
285          pt_files = [
286              "torch/fx/proxy.py",
287              "torch/fx/_symbolic_trace.py",
288              "torch/fx/experimental/proxy_tensor.py",
289              "torch/_ops.py",
290              "torch/_tensor.py",
291              "torch/utils/_python_dispatch.py",
292              "torch/_prims_common/wrappers.py",
293              "torch/_refs/__init__.py",
294              "torch/_refs/nn/functional/__init__.py",
295              "torch/utils/_stats.py",
296          ]
297          while frame:
298              frame = frame.f_back
299              if frame and all(
300                  not frame.f_code.co_filename.endswith(file) for file in pt_files
301              ):
302                  break
303  
304          if not frame:
305              return None
306  
307          return frame

309      @compatibility(is_backward_compatible=True)
310      def create_arg(self, a: Any) -> Argument:
311          """ 
312          A method that lowers the objects seen as arguments during symbolic evaluation 
313          into Argument types that can be stored in IR. 
314   
315          Can be override to support more trace-specific types. 
316          """
317          # IMPORTANT: Are you here because you are trying to proxy a new type into
318          # the graph? Please Please Please contact someone on the PyTorch Compiler team;
319          # the considerations are subtle.
320          #
321          # 1) When you add a new type, all of the downstream consumers and pass writers
322          # need to handle the new type. torch.fx is intended to be easy to write
323          # passes for, so we will push back against new types.
324          # 2) In torch.compile's IR, there are only specific operations that go
325          # into the graph. In particular, Tensor operations should go into the graph,
326          # but non-Tensor operations shouldn't. What that means is that constructors
327          # for new types *SHOULD NOT* become nodes in the FX graph.
328          handler = _create_arg_bypass.get(type(a))
329          if handler is not None:
330              # this is just a performance optimization and can be removed if needed
331              # for common types, we have a fast path to avoid isinstance() overhead
332              # this doesn't remove the checks below since we need to handle subclasses
333              return handler(self, a)
334  
335          if isinstance(a, Proxy):
336              return a.node  # most common arg type goes first
337          elif hasattr(a, "__fx_create_arg__"):
338              return a.__fx_create_arg__(self)
339          # aggregates
340          elif isinstance(a, tuple):
341              if hasattr(a, "_fields"):
342                  # NamedTuple constructors don't seem to like getting a generator
343                  # expression as an argument to their constructor, so build this
344                  # intermediate tuple and unpack it into the NamedTuple constructor
345                  args = [self.create_arg(elem) for elem in a]
346                  return type(a)(*args)  # type: ignore[arg-type]
347              return type(a)([self.create_arg(elem) for elem in a])
348          elif isinstance(a, list):
349              return [self.create_arg(elem) for elem in a]
350          elif isinstance(a, dict):
351              return _create_arg_dict(self, a)
352          elif isinstance(a, slice):
353              return slice(
354                  self.create_arg(a.start),
355                  self.create_arg(a.stop),
356                  self.create_arg(a.step),
357              )
358  
359          elif isinstance(a, range):
360              return range(
361                  self.create_arg(a.start),
362                  self.create_arg(a.stop),
363                  self.create_arg(a.step),
364              )
365  
366          elif isinstance(a, (torch._ops.OpOverload, torch._ops.HigherOrderOperator)):
367              return a
368  
369          elif is_dataclass(a):
370              kwargs = {
371                  field.name: self.create_arg(getattr(a, field.name))
372                  for field in fields(a)
373              }
374              return self.create_node("call_function", a.__class__, (), kwargs)
375  
376          elif isinstance(a, (*base_types, enum.Enum)) or a is None or a is ...:
377              return a
378  
379          raise NotImplementedError(f"argument of type: {type(a)}")

440  @compatibility(is_backward_compatible=True)
441  class Proxy:
442      """ 
443      ``Proxy`` objects are ``Node`` wrappers that flow through the 
444      program during symbolic tracing and record all the operations 
445      (``torch`` function calls, method calls, operators) that they touch 
446      into the growing FX Graph. 
447   
448      If you're doing graph transforms, you can wrap your own ``Proxy`` 
449      method around a raw ``Node`` so that you can use the overloaded 
450      operators to add additional things to a ``Graph``. 
451   
452      ``Proxy`` objects cannot be iterated. In other words, the symbolic 
453      tracer will throw an error if a ``Proxy`` is used in a loop or as 
454      an ``*args``/``**kwargs`` function argument. 
455   
456      There are two main ways around this: 
457      1. Factor out the untraceable logic into a top-level function and 
458      use ``fx.wrap`` on it. 
459      2. If the control flow is static (i.e. the loop trip count is 
460      based on some hyperparameter), the code can be kept in its original 
461      position and refactored into something like:: 
462   
463          for i in range(self.some_hyperparameter): 
464              indexed_item = proxied_value[i] 
465   
466      For a more detailed description into the Proxy internals, check out 
467      the "Proxy" section in `torch/fx/README.md` 
468      """
469  
470      @compatibility(is_backward_compatible=True)
471      def __init__(self, node: Node, tracer: "Optional[TracerBase]" = None):
472          if tracer is None:
473              # This allows you to create a Proxy object around a raw Node
474              tracer = GraphAppendingTracer(node.graph)
475          self.tracer = tracer
476          self.node = node
477  
478      def __repr__(self) -> str:
479          return f"Proxy({self.node.name})"
480  
481      def __getattr__(self, k) -> "Attribute":
482          # note: not added to the graph yet, if this is a method call
483          # we peephole optimize to the method invocation
484          return Attribute(self, k)
485  
486      def __getstate__(self) -> dict:
487          return self.__dict__
488  
489      def __deepcopy__(self, memo) -> dict:
490          # We have to explicitly override this method, because otherwise deepcopy
491          # will go to __getattr__(self, "__deepcopy__") and return a
492          # Attribute(__deepcopy__), and may go into an infinite loop in some cases.
493          import copy
494  
495          new_dict = {}
496          for k, v in self.__dict__.items():
497              try:
498                  new_obj = copy.deepcopy(v, memo)
499              except Exception:
500                  log.warning(
501                      "Shallow copy %s of Proxy because it cannot be deepcopied. "
502                      "Proxy is created for node %s",
503                      k,
504                      self.node.name,
505                  )
506                  new_obj = copy.copy(v)
507              new_dict[k] = new_obj
508          assert "node" in new_dict
509          assert "tracer" in new_dict
510          new_proxy = Proxy(new_dict["node"], new_dict["tracer"])
511          for k, v in new_dict.items():
512              new_proxy.__dict__[k] = v
513          return new_proxy
514  
515      def __setstate__(self, d):
516          # This is called when being unpickled/loaded.
517          self.__dict__ = d
518  
519      def __call__(self, *args, **kwargs) -> "Proxy":
520          return self.tracer.create_proxy(
521              "call_method", "__call__", (self,) + args, kwargs
522          )
523  
524      def __iter__(self) -> Iterator["Proxy"]:
525          frame = inspect.currentframe()
526          assert frame is not None
527          calling_frame = frame.f_back
528          assert calling_frame is not None
529          inst_list = list(dis.get_instructions(calling_frame.f_code))
530          if sys.version_info >= (3, 11):
531              from bisect import bisect_left
532  
533              inst_idx = bisect_left(
534                  inst_list, calling_frame.f_lasti, key=lambda x: x.offset
535              )
536          else:
537              inst_idx = calling_frame.f_lasti // 2
538          inst = inst_list[inst_idx]
539          if inst.opname == "UNPACK_SEQUENCE":
540              return (self[i] for i in range(inst.argval))  # type: ignore[index]
541  
542          return self.tracer.iter(self)
543  
544      def __abs__(self):
545          return self.tracer.create_proxy("call_function", operator.abs, (self,), {})
546  
547      def __bool__(self) -> bool:
548          if self.tracer.trace_asserts:
549              # check if this boolean is used in an assertion, bytecode pattern for assertions
550              # is pretty stable for Python 3.7--3.9
551              frame = inspect.currentframe()
552              assert frame is not None
553              calling_frame = frame.f_back
554              assert calling_frame is not None
555              insts = list(dis.get_instructions(calling_frame.f_code))
556              if sys.version_info >= (3, 11):
557                  from bisect import bisect_left
558  
559                  cur = bisect_left(insts, calling_frame.f_lasti, key=lambda x: x.offset)
560              else:
561                  cur = calling_frame.f_lasti // 2
562              inst = insts[cur]
563  
564              if inst.opname == "POP_JUMP_IF_TRUE":
565                  first = insts[cur + 1]
566                  assert inst.arg is not None
567                  last = insts[inst.arg // 2 - 1]
568                  starts_with_assert = (
569                      first.opname == "LOAD_GLOBAL"
570                      and first.argval == "AssertionError"
571                      or first.opname == "LOAD_ASSERTION_ERROR"
572                  )
573                  if starts_with_assert and last.opname == "RAISE_VARARGS":
574                      self.tracer.create_proxy("call_function", assert_fn, (self,), {})
575                      return True
576  
577          return self.tracer.to_bool(self)
578  
579      @compatibility(is_backward_compatible=True)
580      def keys(self):
581          return self.tracer.keys(self)
582  
583      def __len__(self):
584          raise RuntimeError(
585              "'len' is not supported in symbolic tracing by default. If you want "
586              "this call to be recorded, please call torch.fx.wrap('len') at "
587              "module scope"
588          )

590      @classmethod
591      def __torch_function__(cls, orig_method, types, args=None, kwargs=None):
592          args = args if args else ()
593          kwargs = kwargs if kwargs else {}
594  
595          tracers: dict[Any, None] = {}
596  
597          def find_tracer(a):
598              if isinstance(a, cls):
599                  tracers[a.tracer] = None
600  
601          map_aggregate(args, find_tracer)
602          map_aggregate(kwargs, find_tracer)
603  
604          if len(tracers) > 1:
605              raise RuntimeError(
606                  f"Found multiple different tracers {list(tracers.keys())} while "
607                  f"trying to trace operations {orig_method}"
608              )
609          tracer = next(iter(tracers.keys()))
610  
611          if isinstance(orig_method, torch._C.ScriptMethod):
612              args = (orig_method.owner,) + args
613              return tracer.create_proxy("call_method", orig_method.name, args, kwargs)
614          if torch.overrides.is_tensor_method_or_property(orig_method):
615              return tracer.create_proxy(
616                  "call_method", orig_method.__name__, args, kwargs
617              )
618          else:
619              if isinstance(orig_method, torch._ops.HigherOrderOperator):
620                  # TODO: Define how to symbolically trace HigherOrderOperators
621                  raise RuntimeError("Unable to symbolically trace HigherOrderOperators")
622              return tracer.create_proxy(
623                  "call_function",
624                  orig_method,
625                  args,
626                  kwargs,
627                  name=tracer.graph._target_to_str(orig_method.__name__),
628              )

631  @compatibility(is_backward_compatible=False)
632  class MetaProxy(Proxy):
633      """ 
634      A Proxy subclass that propagates metadata (meta['val']) during graph tracing. 
635      """
636  
637      def __init__(
638          self, node: Node, tracer: "Optional[TracerBase]" = None, fake_mode=None
639      ):
640          super().__init__(node, tracer)
641          self.fake_mode = fake_mode
642  
643      def __repr__(self) -> str:
644          return f"MetaProxy({self.node.name})"
645  
646      @classmethod
647      def __torch_function__(cls, orig_method, types, args=None, kwargs=None):
648          args = args if args else ()
649          kwargs = kwargs if kwargs else {}
650  
651          meta_proxy = None
652          for arg in args:
653              if isinstance(arg, MetaProxy):
654                  meta_proxy = arg
655                  break
656  
657          assert meta_proxy is not None, (
658              "No MetaProxy found in arguments, but one is expected."
659          )
660  
661          proxy = super().__torch_function__(orig_method, types, args, kwargs)
662          with meta_proxy.fake_mode:
663              proxy.node.meta["val"] = orig_method(
664                  *[a.node.meta["val"] if isinstance(a, Proxy) else a for a in args],
665                  **kwargs,
666              )
667          return MetaProxy(proxy.node, proxy.tracer, meta_proxy.fake_mode)

670  @compatibility(is_backward_compatible=True)
671  class Attribute(Proxy):
672      @compatibility(is_backward_compatible=True)
673      def __init__(self, root: Proxy, attr: str):
674          self.root = root
675          self.attr = attr
676          self.tracer = root.tracer
677          self._node: Optional[Node] = None
678  
679      @property
680      def node(self):
681          # the node for attributes is added lazily, since most will just be method calls
682          # which do not rely on the getitem call
683          if self._node is None:
684              self._node = self.tracer.create_proxy(
685                  "call_function", getattr, (self.root, self.attr), {}
686              ).node
687          return self._node
688  
689      def __call__(self, *args, **kwargs):
690          return self.tracer.create_proxy(
691              "call_method", self.attr, (self.root,) + args, kwargs
692          )

735      def _scope(method):
736          def impl(*args, **kwargs):
737              tracer = args[0].tracer
738              target = getattr(operator, method)
739              return tracer.create_proxy("call_function", target, args, kwargs)
740  
741          impl.__name__ = method
742          as_magic = f"__{method.strip('_')}__"
743          setattr(Proxy, as_magic, impl)

748  def _define_reflectable(orig_method_name):
749      method_name = f"__r{orig_method_name.strip('_')}__"
750  
751      def impl(self, rhs):
752          target = getattr(operator, orig_method_name)
753          return self.tracer.create_proxy("call_function", target, (rhs, self), {})
754  
755      impl.__name__ = method_name
756      impl.__qualname__ = method_name
757      setattr(Proxy, method_name, impl)

764  def _no_nodes_error(arg):
765      raise RuntimeError(
766          "Keys for dictionaries used as an argument cannot contain a "
767          f"Node. Got key: {arg}"
768      )

771  def _create_arg_dict(self, a):
772      r = {}
773      for k, v in a.items():
774          if not isinstance(k, str):
775              # Check for invalid dict keys. We do not want a Proxy to appear
776              # anywhere within the key. Since keys can be collection types,
777              # we iterate through the key with map_arg
778              k = self.create_arg(k)
779              map_arg(k, _no_nodes_error)
780          r[k] = self.create_arg(v)
781      return r

785      t: lambda self, a: a
786      for t in [
787          *base_types,
788          type(None),
789          type(...),
790          torch._ops.OpOverload,
791          torch._ops.HigherOrderOperator,
792      ]

794  _create_arg_bypass[Proxy] = lambda self, a: a.node
795  _create_arg_bypass[tuple] = lambda self, a: tuple([self.create_arg(elem) for elem in a])
796  _create_arg_bypass[list] = lambda self, a: [self.create_arg(elem) for elem in a]

798  _create_arg_bypass[immutable_list] = _create_arg_bypass[list]
