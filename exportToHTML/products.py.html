17   class Product(ExprWithIntLimits):
18       r""" 
19       Represents unevaluated products. 
20    
21       Explanation 
22       =========== 
23    
24       ``Product`` represents a finite or infinite product, with the first 
25       argument being the general form of terms in the series, and the second 
26       argument being ``(dummy_variable, start, end)``, with ``dummy_variable`` 
27       taking all integer values from ``start`` through ``end``. In accordance 
28       with long-standing mathematical convention, the end term is included in 
29       the product. 
30    
31       Finite products 
32       =============== 
33    
34       For finite products (and products with symbolic limits assumed to be finite) 
35       we follow the analogue of the summation convention described by Karr [1], 
36       especially definition 3 of section 1.4. The product: 
37    
38       .. math:: 
39    
40           \prod_{m \leq i < n} f(i) 
41    
42       has *the obvious meaning* for `m < n`, namely: 
43    
44       .. math:: 
45    
46           \prod_{m \leq i < n} f(i) = f(m) f(m+1) \cdot \ldots \cdot f(n-2) f(n-1) 
47    
48       with the upper limit value `f(n)` excluded. The product over an empty set is 
49       one if and only if `m = n`: 
50    
51       .. math:: 
52    
53           \prod_{m \leq i < n} f(i) = 1  \quad \mathrm{for} \quad  m = n 
54    
55       Finally, for all other products over empty sets we assume the following 
56       definition: 
57    
58       .. math:: 
59    
60           \prod_{m \leq i < n} f(i) = \frac{1}{\prod_{n \leq i < m} f(i)}  \quad \mathrm{for} \quad  m > n 
61    
62       It is important to note that above we define all products with the upper 
63       limit being exclusive. This is in contrast to the usual mathematical notation, 
64       but does not affect the product convention. Indeed we have: 
65    
66       .. math:: 
67    
68           \prod_{m \leq i < n} f(i) = \prod_{i = m}^{n - 1} f(i) 
69    
70       where the difference in notation is intentional to emphasize the meaning, 
71       with limits typeset on the top being inclusive. 
72    
73       Examples 
74       ======== 
75    
76       >>> from sympy.abc import a, b, i, k, m, n, x 
77       >>> from sympy import Product, oo 
78       >>> Product(k, (k, 1, m)) 
79       Product(k, (k, 1, m)) 
80       >>> Product(k, (k, 1, m)).doit() 
81       factorial(m) 
82       >>> Product(k**2,(k, 1, m)) 
83       Product(k**2, (k, 1, m)) 
84       >>> Product(k**2,(k, 1, m)).doit() 
85       factorial(m)**2 
86    
87       Wallis' product for pi: 
88    
89       >>> W = Product(2*i/(2*i-1) * 2*i/(2*i+1), (i, 1, oo)) 
90       >>> W 
91       Product(4*i**2/((2*i - 1)*(2*i + 1)), (i, 1, oo)) 
92    
93       Direct computation currently fails: 
94    
95       >>> W.doit() 
96       Product(4*i**2/((2*i - 1)*(2*i + 1)), (i, 1, oo)) 
97    
98       But we can approach the infinite product by a limit of finite products: 
99    
100      >>> from sympy import limit 
101      >>> W2 = Product(2*i/(2*i-1)*2*i/(2*i+1), (i, 1, n)) 
102      >>> W2 
103      Product(4*i**2/((2*i - 1)*(2*i + 1)), (i, 1, n)) 
104      >>> W2e = W2.doit() 
105      >>> W2e 
106      4**n*factorial(n)**2/(2**(2*n)*RisingFactorial(1/2, n)*RisingFactorial(3/2, n)) 
107      >>> limit(W2e, n, oo) 
108      pi/2 
109   
110      By the same formula we can compute sin(pi/2): 
111   
112      >>> from sympy import combsimp, pi, gamma, simplify 
113      >>> P = pi * x * Product(1 - x**2/k**2, (k, 1, n)) 
114      >>> P = P.subs(x, pi/2) 
115      >>> P 
116      pi**2*Product(1 - pi**2/(4*k**2), (k, 1, n))/2 
117      >>> Pe = P.doit() 
118      >>> Pe 
119      pi**2*RisingFactorial(1 - pi/2, n)*RisingFactorial(1 + pi/2, n)/(2*factorial(n)**2) 
120      >>> limit(Pe, n, oo).gammasimp() 
121      sin(pi**2/2) 
122      >>> Pe.rewrite(gamma) 
123      (-1)**n*pi**2*gamma(pi/2)*gamma(n + 1 + pi/2)/(2*gamma(1 + pi/2)*gamma(-n + pi/2)*gamma(n + 1)**2) 
124   
125      Products with the lower limit being larger than the upper one: 
126   
127      >>> Product(1/i, (i, 6, 1)).doit() 
128      120 
129      >>> Product(i, (i, 2, 5)).doit() 
130      120 
131   
132      The empty product: 
133   
134      >>> Product(i, (i, n, n-1)).doit() 
135      1 
136   
137      An example showing that the symbolic result of a product is still 
138      valid for seemingly nonsensical values of the limits. Then the Karr 
139      convention allows us to give a perfectly valid interpretation to 
140      those products by interchanging the limits according to the above rules: 
141   
142      >>> P = Product(2, (i, 10, n)).doit() 
143      >>> P 
144      2**(n - 9) 
145      >>> P.subs(n, 5) 
146      1/16 
147      >>> Product(2, (i, 10, 5)).doit() 
148      1/16 
149      >>> 1/Product(2, (i, 6, 9)).doit() 
150      1/16 
151   
152      An explicit example of the Karr summation convention applied to products: 
153   
154      >>> P1 = Product(x, (i, a, b)).doit() 
155      >>> P1 
156      x**(-a + b + 1) 
157      >>> P2 = Product(x, (i, b+1, a-1)).doit() 
158      >>> P2 
159      x**(a - b - 1) 
160      >>> simplify(P1 * P2) 
161      1 
162   
163      And another one: 
164   
165      >>> P1 = Product(i, (i, b, a)).doit() 
166      >>> P1 
167      RisingFactorial(b, a - b + 1) 
168      >>> P2 = Product(i, (i, a+1, b-1)).doit() 
169      >>> P2 
170      RisingFactorial(a + 1, -a + b - 1) 
171      >>> P1 * P2 
172      RisingFactorial(b, a - b + 1)*RisingFactorial(a + 1, -a + b - 1) 
173      >>> combsimp(P1 * P2) 
174      1 
175   
176      See Also 
177      ======== 
178   
179      Sum, summation 
180      product 
181   
182      References 
183      ========== 
184   
185      .. [1] Michael Karr, "Summation in Finite Terms", Journal of the ACM, 
186             Volume 28 Issue 2, April 1981, Pages 305-350

189      .. [3] https://en.wikipedia.org/wiki/Empty_product 
190      """

251      def doit(self, **hints):
252          # first make sure any definite limits have product
253          # variables with matching assumptions
254          reps = {}
255          for xab in self.limits:
256              d = _dummy_with_inherited_properties_concrete(xab)
257              if d:
258                  reps[xab[0]] = d
259          if reps:
260              undo = {v: k for k, v in reps.items()}
261              did = self.xreplace(reps).doit(**hints)
262              if isinstance(did, tuple):  # when separate=True
263                  did = tuple([i.xreplace(undo) for i in did])
264              else:
265                  did = did.xreplace(undo)
266              return did
267  
268          from sympy.simplify.powsimp import powsimp
269          f = self.function
270          for index, limit in enumerate(self.limits):
271              i, a, b = limit
272              dif = b - a
273              if dif.is_integer and dif.is_negative:
274                  a, b = b + 1, a - 1
275                  f = 1 / f
276  
277              g = self._eval_product(f, (i, a, b))
278              if g in (None, S.NaN):
279                  return self.func(powsimp(f), *self.limits[index:])
280              else:
281                  f = g
282  
283          if hints.get('deep', True):
284              return f.doit(**hints)
285          else:
286              return powsimp(f)

291      def _eval_product(self, term, limits):
292  
293          (k, a, n) = limits
294  
295          if k not in term.free_symbols:
296              if (term - 1).is_zero:
297                  return S.One
298              return term**(n - a + 1)
299  
300          if a == n:
301              return term.subs(k, a)
302  
303          from .delta import deltaproduct, _has_simple_delta
304          if term.has(KroneckerDelta) and _has_simple_delta(term, limits[0]):
305              return deltaproduct(term, limits)
306  
307          dif = n - a
308          definite = dif.is_Integer
309          if definite and (dif < 100):
310              return self._eval_product_direct(term, limits)
311  
312          elif term.is_polynomial(k):
313              poly = term.as_poly(k)
314  
315              A = B = Q = S.One
316  
317              all_roots = roots(poly)
318  
319              M = 0
320              for r, m in all_roots.items():
321                  M += m
322                  A *= RisingFactorial(a - r, n - a + 1)**m
323                  Q *= (n - r)**m
324  
325              if M < poly.degree():
326                  arg = quo(poly, Q.as_poly(k))
327                  B = self.func(arg, (k, a, n)).doit()
328  
329              return poly.LC()**(n - a + 1) * A * B
330  
331          elif term.is_Add:
332              factored = factor_terms(term, fraction=True)
333              if factored.is_Mul:
334                  return self._eval_product(factored, (k, a, n))
335  
336          elif term.is_Mul:
337              # Factor in part without the summation variable and part with
338              without_k, with_k = term.as_coeff_mul(k)
339  
340              if len(with_k) >= 2:
341                  # More than one term including k, so still a multiplication
342                  exclude, include = [], []
343                  for t in with_k:
344                      p = self._eval_product(t, (k, a, n))
345  
346                      if p is not None:
347                          exclude.append(p)
348                      else:
349                          include.append(t)
350  
351                  if not exclude:
352                      return None
353                  else:
354                      arg = term._new_rawargs(*include)
355                      A = Mul(*exclude)
356                      B = self.func(arg, (k, a, n)).doit()
357                      return without_k**(n - a + 1)*A * B
358              else:
359                  # Just a single term
360                  p = self._eval_product(with_k[0], (k, a, n))
361                  if p is None:
362                      p = self.func(with_k[0], (k, a, n)).doit()
363                  return without_k**(n - a + 1)*p
364  
365  
366          elif term.is_Pow:
367              if not term.base.has(k):
368                  s = summation(term.exp, (k, a, n))
369  
370                  return term.base**s
371              elif not term.exp.has(k):
372                  p = self._eval_product(term.base, (k, a, n))
373  
374                  if p is not None:
375                      return p**term.exp
376  
377          elif isinstance(term, Product):
378              evaluated = term.doit()
379              f = self._eval_product(evaluated, limits)
380              if f is None:
381                  return self.func(evaluated, limits)
382              else:
383                  return f
384  
385          if definite:
386              return self._eval_product_direct(term, limits)
387

399          (k, a, n) = limits

402      def _eval_derivative(self, x):
403          if isinstance(x, Symbol) and x not in self.free_symbols:
404              return S.Zero
405          f, limits = self.function, list(self.limits)
406          limit = limits.pop(-1)
407          if limits:
408              f = self.func(f, *limits)
409          i, a, b = limit
410          if x in a.free_symbols or x in b.free_symbols:
411              return None
412          h = Dummy()
413          rv = Sum( Product(f, (i, a, h - 1)) * Product(f, (i, h + 1, b)) * Derivative(f, x, evaluate=True).subs(i, h), (h, a, b))
414          return rv

416      def is_convergent(self):
417          r""" 
418          See docs of :obj:`.Sum.is_convergent()` for explanation of convergence 
419          in SymPy. 
420   
421          Explanation 
422          =========== 
423   
424          The infinite product: 
425   
426          .. math:: 
427   
428              \prod_{1 \leq i < \infty} f(i) 
429   
430          is defined by the sequence of partial products: 
431   
432          .. math:: 
433   
434              \prod_{i=1}^{n} f(i) = f(1) f(2) \cdots f(n) 
435   
436          as n increases without bound. The product converges to a non-zero 
437          value if and only if the sum: 
438   
439          .. math:: 
440   
441              \sum_{1 \leq i < \infty} \log{f(n)} 
442   
443          converges. 
444   
445          Examples 
446          ======== 
447   
448          >>> from sympy import Product, Symbol, cos, pi, exp, oo 
449          >>> n = Symbol('n', integer=True) 
450          >>> Product(n/(n + 1), (n, 1, oo)).is_convergent() 
451          False 
452          >>> Product(1/n**2, (n, 1, oo)).is_convergent() 
453          False 
454          >>> Product(cos(pi/n), (n, 1, oo)).is_convergent() 
455          True 
456          >>> Product(exp(-n**2), (n, 1, oo)).is_convergent() 
457          False 
458   
459          References 
460          ========== 
461   
462          .. [1] https://en.wikipedia.org/wiki/Infinite_product 
463          """
464          sequence_term = self.function
465          log_sum = log(sequence_term)
466          lim = self.limits
467          try:
468              is_conv = Sum(log_sum, *lim).is_convergent()
469          except NotImplementedError:
470              if Sum(sequence_term - 1, *lim).is_absolutely_convergent() is S.true:
471                  return S.true
472              raise NotImplementedError("The algorithm to find the product convergence of %s "
473                                          "is not yet implemented" % (sequence_term))
474          return is_conv

476      def reverse_order(expr, *indices):
477          """ 
478          Reverse the order of a limit in a Product. 
479   
480          Explanation 
481          =========== 
482   
483          ``reverse_order(expr, *indices)`` reverses some limits in the expression 
484          ``expr`` which can be either a ``Sum`` or a ``Product``. The selectors in 
485          the argument ``indices`` specify some indices whose limits get reversed. 
486          These selectors are either variable names or numerical indices counted 
487          starting from the inner-most limit tuple. 
488   
489          Examples 
490          ======== 
491   
492          >>> from sympy import gamma, Product, simplify, Sum 
493          >>> from sympy.abc import x, y, a, b, c, d 
494          >>> P = Product(x, (x, a, b)) 
495          >>> Pr = P.reverse_order(x) 
496          >>> Pr 
497          Product(1/x, (x, b + 1, a - 1)) 
498          >>> Pr = Pr.doit() 
499          >>> Pr 
500          1/RisingFactorial(b + 1, a - b - 1) 
501          >>> simplify(Pr.rewrite(gamma)) 
502          Piecewise((gamma(b + 1)/gamma(a), b > -1), ((-1)**(-a + b + 1)*gamma(1 - a)/gamma(-b), True)) 
503          >>> P = P.doit() 
504          >>> P 
505          RisingFactorial(a, -a + b + 1) 
506          >>> simplify(P.rewrite(gamma)) 
507          Piecewise((gamma(b + 1)/gamma(a), a > 0), ((-1)**(-a + b + 1)*gamma(1 - a)/gamma(-b), True)) 
508   
509          While one should prefer variable names when specifying which limits 
510          to reverse, the index counting notation comes in handy in case there 
511          are several symbols with the same name. 
512   
513          >>> S = Sum(x*y, (x, a, b), (y, c, d)) 
514          >>> S 
515          Sum(x*y, (x, a, b), (y, c, d)) 
516          >>> S0 = S.reverse_order(0) 
517          >>> S0 
518          Sum(-x*y, (x, b + 1, a - 1), (y, c, d)) 
519          >>> S1 = S0.reverse_order(1) 
520          >>> S1 
521          Sum(x*y, (x, b + 1, a - 1), (y, d + 1, c - 1)) 
522   
523          Of course we can mix both notations: 
524   
525          >>> Sum(x*y, (x, a, b), (y, 2, 5)).reverse_order(x, 1) 
526          Sum(x*y, (x, b + 1, a - 1), (y, 6, 1)) 
527          >>> Sum(x*y, (x, a, b), (y, 2, 5)).reverse_order(y, x) 
528          Sum(x*y, (x, b + 1, a - 1), (y, 6, 1)) 
529   
530          See Also 
531          ======== 
532   
533          sympy.concrete.expr_with_intlimits.ExprWithIntLimits.index, 
534          reorder_limit, 
535          sympy.concrete.expr_with_intlimits.ExprWithIntLimits.reorder 
536   
537          References 
538          ========== 
539   
540          .. [1] Michael Karr, "Summation in Finite Terms", Journal of the ACM, 
541                 Volume 28 Issue 2, April 1981, Pages 305-350 
542                 https://dl.acm.org/doi/10.1145/322248.322255 
543   
544          """
545          l_indices = list(indices)
546  
547          for i, indx in enumerate(l_indices):
548              if not isinstance(indx, int):
549                  l_indices[i] = expr.index(indx)
550  
551          e = 1
552          limits = []
553          for i, limit in enumerate(expr.limits):
554              l = limit
555              if i in l_indices:
556                  e = -e
557                  l = (limit[0], limit[2] + 1, limit[1] - 1)
558              limits.append(l)
559  
560          return Product(expr.function ** e, *limits)

563  def product(*args, **kwargs):
564      r""" 
565      Compute the product. 
566   
567      Explanation 
568      =========== 
569   
570      The notation for symbols is similar to the notation used in Sum or 
571      Integral. product(f, (i, a, b)) computes the product of f with 
572      respect to i from a to b, i.e., 
573   
574      :: 
575   
576                                       b 
577                                     _____ 
578          product(f(n), (i, a, b)) = |   | f(n) 
579                                     |   | 
580                                     i = a 
581   
582      If it cannot compute the product, it returns an unevaluated Product object. 
583      Repeated products can be computed by introducing additional symbols tuples:: 
584   
585      Examples 
586      ======== 
587   
588      >>> from sympy import product, symbols 
589      >>> i, n, m, k = symbols('i n m k', integer=True) 
590   
591      >>> product(i, (i, 1, k)) 
592      factorial(k) 
593      >>> product(m, (i, 1, k)) 
594      m**k 
595      >>> product(i, (i, 1, k), (k, 1, n)) 
596      Product(factorial(k), (k, 1, n)) 
597   
598      """
599  
600      prod = Product(*args, **kwargs)
601  
602      if isinstance(prod, Product):
603          return prod.doit(deep=False)
604      else:
605          return prod
606
