2    # This module provides a FAST (on GPU) content addressable store for storages
3    # (and tensors on top of them) with VERY WEAK portability guarantees (e.g.,
4    # don't expect CPU/CUDA to address to the same hash, don't expect it to be
5    # portable across devices) that is NOT cryptographically secure.  In return,
6    # we are able to hash 40G of tensor data on GPU in less than a second,
7    # compared to running SHA-1 in CPU which would a minute or so.  The primary
8    # use case is for efficiently snapshotting intermediate tensor data for
9    # offline debugging, but it's been put in this module in case you think of
10   # another use case for it.  The hash function could be replaced with a
11   # straight reimplementation of SHA-1, which would give us much stronger
12   # portability guarantees.
13   #
14   # WARNING: THERE IS NO BC/FC GUARANTEE FOR THIS FORMAT!  If you need to format
15   # shift the result, consider packing it into a single torch.save object
16   # with traditional view sharing.
17   #
18   # Because of the weak portability guarantees, you can only write to the
19   # content store from a single process; we don't provide any capability
20   # of "reopening" a content store to add more things to it.  But we don't
21   # assume that you can keep all of the tensors you want to add to the store
22   # in memory at once, because you probably can't!  Nor do we assume that
23   # you know a priori whether or not two storages can be deduplicated or not.
24   #
25   # Note: only storages are content-addressed; tensors are name addressed
26   #
27   # Note: our padding strategy means that [1, 0] and [1] int16 tensors would
28   # map to the same (padded) storage.  We think this will be immaterial for most
29   # users.

46   def lazy_compile(**compile_kwargs):
47       """Lazily wrap a function with torch.compile on the first call 
48    
49       This avoids eagerly importing dynamo. 
50       """
51   
52       def decorate_fn(fn):
53           @functools.wraps(fn)
54           def compile_hook(*args, **kwargs):
55               compiled_fn = torch.compile(fn, **compile_kwargs)
56               globals()[fn.__name__] = functools.wraps(fn)(compiled_fn)
57               return compiled_fn(*args, **kwargs)
58   
59           return compile_hook
60   
61       return decorate_fn

64   # Use of torch.compile is mandatory for (1) good memory usage
65   # and (2) xor_sum implementation.  This is our first instance of
66   # using PT2 to implement a kernel in PyTorch; if we get AOT capabilities
67   # it would be good to apply it here.
68   @lazy_compile(dynamic=True)
69   def hash_storage_kernel(x):
70       # The randint calls are carefully written to hit things we
71       # have lowerings for in inductor.  Lack of unsigned 32-bit integer
72       # is a pain.
73       a = torch.randint(
74           -(2**31), 2**31, x.shape, device=x.device, dtype=torch.int32
75       ).abs()
76       a = ((a % (2**31 - 1)) + 1).long()
77       b = (
78           torch.randint(-(2**31), 2**31, x.shape, device=x.device, dtype=torch.int32)
79           .abs()
80           .long()
81       )
82       # This is a standard shift-multiply universal hash family
83       # plus xor sum hash, using Philox to generate random numbers.
84       # Our Philox RNG is not deterministic across devices so
85       # don't use this for stable hashing.
86       #
87       # This assumes fixed length so you're also obligated to bucket
88       # by the length of tensor as well
89       return prims.xor_sum((a * x + b).int(), [0])

92   # Returns a hex digest of the data in the storage.  Guaranteed to be
93   # SHA-1 if stable_hash=True, otherwise it will consistent for a single
94   # process run but not necessarily across processes.
95   def hash_storage(storage: torch.UntypedStorage, *, stable_hash: bool = False) -> str:
96       import torch._dynamo
97       from torch._dynamo.utils import is_compile_supported
98   
99       device_type = storage.device.type
100      if stable_hash or not is_compile_supported(device_type):
101          cpu_storage = storage.cpu()
102          # TODO: make storage support buffer protocol so this isn't
103          # necessary
104          buf = (ctypes.c_byte * cpu_storage.nbytes()).from_address(
105              cpu_storage.data_ptr()
106          )
107          sha1 = hashlib.sha1(usedforsecurity=False)
108          sha1.update(buf)
109          return sha1.hexdigest()
110  
111      # TODO: factor this into a random utility
112      if device_type == "cpu":
113          generator = torch._C.default_generator
114      elif device_type == "cuda":
115          generator = torch.cuda.default_generators[storage.device.index]
116      elif device_type == "mps":
117          generator = torch.mps._get_default_mps_generator()
118      elif device_type == "xpu":
119          generator = torch.xpu.default_generators[storage.device.index]
120      else:
121          raise AssertionError(f"unhandled device type {device_type}")
122      state = generator.get_state()
123      try:
124          generator.manual_seed(0)
125          x = torch.empty(0, dtype=torch.uint8, device=storage.device).set_(storage)  # type: ignore[call-overload]
126          # The dtype-casting view cannot be compiled, and so the
127          # padding/reshaping also needs to be done externally even
128          # though it could be profitably fused
129          pad = -x.numel() % 4
130          if pad > 0:
131              x = F.pad(x, (0, pad), "constant", 0)
132          x = x.view(torch.int32)
133          # We run the 32-bit hash five times with differing parameters to
134          # reduce chance of collision
135          ITER = 5
136          cs = [hash_storage_kernel(x).item() for _ in range(ITER)]
137          return struct.pack(">" + "i" * ITER, *cs).hex()
138      finally:
139          generator.set_state(state)

142  class ContentStoreWriter:
143      # Structure:
144      #   storages/
145      #     00/
146      #       0000..00
147      #   tensors/
148      #     name
149      def __init__(self, loc: str, stable_hash: bool = False) -> None:
150          self.loc: str = loc
151          self.seen_storage_hashes: set[str] = set()
152          self.stable_hash = stable_hash
153  
154      # TODO: offer some sort of non-blocking API to speed things up
155      def write_storage(self, storage: torch.UntypedStorage) -> str:
156          h = hash_storage(storage, stable_hash=self.stable_hash)
157          if h in self.seen_storage_hashes:
158              return h
159          # TODO: consider not using torch.save for this; we don't actually
160          # need any metadata for the storage
161          subfolder = os.path.join(self.loc, "storages")
162          os.makedirs(subfolder, exist_ok=True)
163          target = os.path.join(subfolder, h)
164          if os.path.exists(target):
165              return h
166          torch.save(storage, target)
167          self.seen_storage_hashes.add(h)
168          return h
169  
170      def compute_tensor_metadata(self, t: torch.Tensor, h=None):
171          if h is None:
172              h = hash_storage(t.untyped_storage(), stable_hash=self.stable_hash)
173          return (
174              t.dtype,
175              h,
176              t.storage_offset(),
177              tuple(t.shape),
178              t.stride(),
179              torch._utils.get_tensor_metadata(t),
180          )
181  
182      def write_tensor(self, name: str, t: torch.Tensor) -> None:
183          storage = t.untyped_storage()
184          h = self.write_storage(storage)
185          # TODO: Support more advanced snapshotting of requires_grad/grad/etc
186          d, f = os.path.split(name)
187          payload = self.compute_tensor_metadata(t, h=h)
188          subfolder = os.path.join(self.loc, "tensors", d)
189          os.makedirs(subfolder, exist_ok=True)
190          torch.save(payload, os.path.join(subfolder, f))

193  class ContentStoreReader:
194      def __init__(self, loc: str, *, cache=True) -> None:
195          self.loc = loc
196          self.storage_cache: Optional[
197              dict[Optional[torch.device], dict[str, StorageWeakRef]]
198          ] = None
199          if cache:
200              self.storage_cache = defaultdict(dict)
201  
202      def read_storage(self, h: str, *, device=None) -> torch.UntypedStorage:
203          if device is not None:
204              device = torch.device(device)
205          ws = (
206              self.storage_cache[device].get(h)
207              if self.storage_cache is not None
208              else None
209          )
210          s: Optional[torch.UntypedStorage]
211          if ws is not None:
212              s = torch.UntypedStorage._new_with_weak_ptr(ws.cdata)
213              if s is not None:
214                  return s
215          s = torch.load(
216              os.path.join(self.loc, "storages", h),
217              weights_only=True,
218              map_location=device,
219          )._untyped_storage
220          assert s is not None
221          if self.storage_cache is not None:
222              self.storage_cache[device][h] = StorageWeakRef(s)
223          return s
224  
225      def read_tensor_metadata(self, name: str):
226          fn = os.path.join(self.loc, "tensors", name)
227          if not os.path.exists(fn):
228              raise FileNotFoundError(fn)
229          return torch.load(fn, weights_only=True)
230  
231      def read_tensor(self, name: str, *, device=None) -> torch.Tensor:
232          dtype, h, storage_offset, size, stride, metadata = self.read_tensor_metadata(
233              name
234          )
235          storage = self.read_storage(h, device=device)
236          t = torch.tensor([], dtype=dtype, device=storage.device)
237          t.set_(storage, storage_offset, size, stride)
238          torch._utils.set_tensor_metadata(t, metadata)
239          return t
