26   class _PyTreeExtensionContext:
27       """Context manager to register PyTree extension."""
28   
29       _extensions: dict[type, tuple[pytree.FlattenFunc, pytree.UnflattenFunc]]
30   
31       def __init__(self) -> None:
32           self._extensions = {}
33           # Register PyTree extension for HuggingFace model output.
34           self._register_huggingface_model_output_extension()
35   
36       def __enter__(self):
37           for class_type, (flatten_func, unflatten_func) in self._extensions.items():
38               pytree._private_register_pytree_node(
39                   class_type,
40                   flatten_func,
41                   unflatten_func,
42               )
43           return self
44   
45       def __exit__(self, exc_type, exc_val, exc_tb):
46           for class_type in self._extensions:
47               pytree.SUPPORTED_NODES.pop(class_type)
48   
49       def register_pytree_node(
50           self,
51           class_type: type,
52           flatten_func: pytree.FlattenFunc,
53           unflatten_func: pytree.UnflattenFunc,
54       ):
55           """Register PyTree extension for a custom python type. 
56    
57           Args: 
58               class_type: The custom python type. 
59               flatten_func: The flatten function. 
60               unflatten_func: The unflatten function. 
61    
62           Raises: 
63               AssertionError: If the custom python type is already registered. 
64           """
65           if class_type in pytree.SUPPORTED_NODES or class_type in self._extensions:
66               # PyTree node already registered.
67               # E.g., `huggingface/transformer` registers `ModelOutput` as PyTree node after
68               # https://github.com/huggingface/transformers/pull/25358.
69               return
70           self._extensions[class_type] = (flatten_func, unflatten_func)
71   
72       def _register_huggingface_model_output_extension(self):
73           try:
74               from transformers import modeling_outputs  # type: ignore[import]
75           except ImportError:
76               return
77   
78           def model_output_flatten(
79               output: modeling_outputs.ModelOutput,
80           ) -> tuple[list[Any], pytree.Context]:
81               return list(output.values()), (type(output), list(output.keys()))
82   
83           def model_output_unflatten(
84               values: list[Any], context: pytree.Context
85           ) -> modeling_outputs.ModelOutput:
86               output_type, keys = context
87               return output_type(**dict(zip(keys, values)))
88   
89           # All 'ModelOutput' subclasses are defined under module 'modeling_outputs'.
90           named_model_output_classes = inspect.getmembers(
91               modeling_outputs,
92               lambda x: (
93                   inspect.isclass(x)
94                   and issubclass(x, modeling_outputs.ModelOutput)
95                   and x is not modeling_outputs.ModelOutput
96               ),
97           )
98   
99           for _, class_type in named_model_output_classes:
100              self.register_pytree_node(
101                  class_type,
102                  model_output_flatten,
103                  model_output_unflatten,  # type: ignore[arg-type ]
104              )

107  class DynamoFlattenOutputStep(io_adapter.FlattenOutputStep):
108      """Flatten nested collection and custom python types and return a flat list of elements. 
109   
110      Extended from :class:`io_adapter.FlattenOutputStep` to support flattening arbitrary 
111      types via pytree extension. By default this supports many common user defined python 
112      types such as :class:`ModelOutput` from HuggingFace transformers. 
113   
114      The pytree extension can be customized by passing in a ``_PyTreeExtensionContext`` 
115      object. See :meth:`_PyTreeExtensionContext.register_pytree_node`. 
116      """
117  
118      def __init__(self, pytree_extension_context: _PyTreeExtensionContext | None = None):
119          super().__init__()
120          self._pytree_extension_context = (
121              pytree_extension_context or _PyTreeExtensionContext()
122          )
123  
124      def apply(
125          self,
126          model_outputs: Any,
127          model: torch.nn.Module | Callable | torch_export.ExportedProgram | None = None,
128      ) -> Sequence[Any]:
129          """Flatten the model outputs, under the context of pytree extension."""
130          with self._pytree_extension_context:
131              return super().apply(model_outputs, model=model)

134  def _wrap_model_with_output_adapter(
135      model: torch.nn.Module | Callable,
136      output_adapter: DynamoFlattenOutputStep,
137  ) -> Callable:
138      """Wrap model with output adapter. 
139   
140      This is a helper function to enable :func:`dynamo.export` on models that produce 
141      custom user defined types outputs. It wraps the model with an output adapter to 
142      convert the outputs to :func:`dynamo.export` compatible types, i.e. :class:`torch.Tensor`. 
143   
144      The adapting logic is controlled by ``output_adapter``. 
145   
146      Args: 
147          model: PyTorch model or function. 
148          output_adapter: Output adapter to apply to model output. 
149      Returns: 
150          Wrapped model. 
151      """
152      model_func = model.forward if isinstance(model, torch.nn.Module) else model
153  
154      # Preserve original function signature.
155      @functools.wraps(model_func)
156      def wrapped(*args, **kwargs):
157          return output_adapter.apply(model_func(*args, **kwargs), model=model)
158  
159      return wrapped

162  class DynamoExport(_exporter_legacy.FXGraphExtractor):
163      """Generates a FX GraphModule using torch.dynamo.export API 
164      Args: 
165          aten_graph: If True, exports a graph with ATen operators. 
166                      If False, exports a graph with Python operators. 
167      """
168  
169      def __init__(
170          self,
171          aten_graph: bool | None = None,
172      ):
173          super().__init__()
174          self.aten_graph = aten_graph or True
175  
176      def generate_fx(
177          self,
178          options: _exporter_legacy.ResolvedExportOptions,
179          model: torch.nn.Module | Callable,
180          model_args: Sequence[Any],
181          model_kwargs: Mapping[str, Any],
182      ) -> torch.fx.GraphModule:
183          # `dynamo.export` does not recognize custom user defined classes as output type.
184          # Apply wrapper to adapt the outputs back to `dynamo.export` compatible types,
185          # i.e. :class:`torch.Tensor`.
186          dynamo_flatten_output_step = DynamoFlattenOutputStep()
187          wrapped_model = _wrap_model_with_output_adapter(
188              model, dynamo_flatten_output_step
189          )
190          # Record the output adapter step.
191          self.output_adapter.append_step(dynamo_flatten_output_step)
192  
193          # Translate callable to FX graph.
194          #
195          fake_mode = (
196              options.fake_context.fake_mode
197              if options.fake_context
198              else contextlib.nullcontext()
199          )
200          fx_mode = "symbolic" if options.dynamic_shapes else "fake"
201          with fake_mode:  # type: ignore[attr-defined]
202              graph_module, graph_guard = torch._dynamo.export(
203                  wrapped_model,
204                  tracing_mode=fx_mode,
205              )(
206                  *model_args,
207                  **model_kwargs,
208              )
209          del graph_guard  # Unused
210          torch._dynamo.reset()
211  
212          # Export FX graph to ONNX ModelProto.
213          self.input_adapter.append_step(
214              io_adapter.FlattenInputWithTreeSpecValidationInputStep()
215          )
216  
217          updated_model_args = self.input_adapter.apply(
218              *model_args, model=model, **model_kwargs
219          )
220  
221          return self.pre_export_passes(options, model, graph_module, updated_model_args)  # type: ignore[return-value]
222  
223      def pre_export_passes(
224          self,
225          options: _exporter_legacy.ResolvedExportOptions,
226          original_model: torch.nn.Module | Callable,
227          fx_module: torch.fx.GraphModule,
228          fx_module_args: Sequence[Any],
229      ):
230          return _exporter_legacy.common_pre_export_passes(
231              options, original_model, fx_module, fx_module_args
232          )
