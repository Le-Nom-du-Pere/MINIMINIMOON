128  def test_differentiate_finite():
129      x, y, h = symbols('x y h')
130      f = Function('f')
131      with warns_deprecated_sympy():
132          res0 = differentiate_finite(f(x, y) + exp(42), x, y, evaluate=True)
133      xm, xp, ym, yp = [v + sign*S.Half for v, sign in product([x, y], [-1, 1])]
134      ref0 = f(xm, ym) + f(xp, yp) - f(xm, yp) - f(xp, ym)
135      assert (res0 - ref0).simplify() == 0
136  
137      g = Function('g')
138      with warns_deprecated_sympy():
139          res1 = differentiate_finite(f(x)*g(x) + 42, x, evaluate=True)
140      ref1 = (-f(x - S.Half) + f(x + S.Half))*g(x) + \
141             (-g(x - S.Half) + g(x + S.Half))*f(x)
142      assert (res1 - ref1).simplify() == 0
143  
144      res2 = differentiate_finite(f(x) + x**3 + 42, x, points=[x-1, x+1])
145      ref2 = (f(x + 1) + (x + 1)**3 - f(x - 1) - (x - 1)**3)/2
146      assert (res2 - ref2).simplify() == 0
147      raises(TypeError, lambda: differentiate_finite(f(x)*g(x), x,
148                                                     pints=[x-1, x+1]))
149  
150      res3 = differentiate_finite(f(x)*g(x).diff(x), x)
151      ref3 = (-g(x) + g(x + 1))*f(x + S.Half) - (g(x) - g(x - 1))*f(x - S.Half)
152      assert res3 == ref3
153  
154      res4 = differentiate_finite(f(x)*g(x).diff(x).diff(x), x)
155      ref4 = -((g(x - Rational(3, 2)) - 2*g(x - S.Half) + g(x + S.Half))*f(x - S.Half)) \
156             + (g(x - S.Half) - 2*g(x + S.Half) + g(x + Rational(3, 2)))*f(x + S.Half)
157      assert res4 == ref4
158  
159      res5_expr = f(x).diff(x)*g(x).diff(x)
160      res5 = differentiate_finite(res5_expr, points=[x-h, x, x+h])
161      ref5 = (-2*f(x)/h + f(-h + x)/(2*h) + 3*f(h + x)/(2*h))*(-2*g(x)/h + g(-h + x)/(2*h) \
162             + 3*g(h + x)/(2*h))/(2*h) - (2*f(x)/h - 3*f(-h + x)/(2*h) - \
163             f(h + x)/(2*h))*(2*g(x)/h - 3*g(-h + x)/(2*h) - g(h + x)/(2*h))/(2*h)
164      assert res5 == ref5
