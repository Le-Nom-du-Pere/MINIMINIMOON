57   def test_RaisingOp():
58       assert Dagger(ad) == a
59       assert Commutator(ad, a).doit() == Integer(-1)
60       assert Commutator(ad, N).doit() == Integer(-1)*ad
61       assert qapply(ad*k) == (sqrt(k.n + 1)*SHOKet(k.n + 1)).expand()
62       assert qapply(ad*kz) == (sqrt(kz.n + 1)*SHOKet(kz.n + 1)).expand()
63       assert qapply(ad*kf) == (sqrt(kf.n + 1)*SHOKet(kf.n + 1)).expand()
64       assert ad.rewrite('xp').doit() == \
65           (Integer(1)/sqrt(Integer(2)*hbar*m*omega))*(Integer(-1)*I*Px + m*omega*X)
66       assert ad.hilbert_space == ComplexSpace(S.Infinity)
67       for i in range(ndim - 1):
68           assert ad_rep_sympy[i + 1,i] == sqrt(i + 1)
69   
70       if not np:
71           skip("numpy not installed.")
72   
73       ad_rep_numpy = represent(ad, basis=N, ndim=4, format='numpy')
74       for i in range(ndim - 1):
75           assert ad_rep_numpy[i + 1,i] == float(sqrt(i + 1))
76   
77       if not np:
78           skip("numpy not installed.")
79       if not scipy:
80           skip("scipy not installed.")
81   
82       ad_rep_scipy = represent(ad, basis=N, ndim=4, format='scipy.sparse', spmatrix='lil')
83       for i in range(ndim - 1):
84           assert ad_rep_scipy[i + 1,i] == float(sqrt(i + 1))

86       assert ad_rep_numpy.dtype == 'float64'
87       assert ad_rep_scipy.dtype == 'float64'

89   def test_LoweringOp():
90       assert Dagger(a) == ad
91       assert Commutator(a, ad).doit() == Integer(1)
92       assert Commutator(a, N).doit() == a
93       assert qapply(a*k) == (sqrt(k.n)*SHOKet(k.n-Integer(1))).expand()
94       assert qapply(a*kz) == Integer(0)
95       assert qapply(a*kf) == (sqrt(kf.n)*SHOKet(kf.n-Integer(1))).expand()
96       assert a.rewrite('xp').doit() == \
97           (Integer(1)/sqrt(Integer(2)*hbar*m*omega))*(I*Px + m*omega*X)
98       for i in range(ndim - 1):
99           assert a_rep[i,i + 1] == sqrt(i + 1)

101  def test_NumberOp():
102      assert Commutator(N, ad).doit() == ad
103      assert Commutator(N, a).doit() == Integer(-1)*a
104      assert Commutator(N, H).doit() == Integer(0)
105      assert qapply(N*k) == (k.n*k).expand()
106      assert N.rewrite('a').doit() == ad*a
107      assert N.rewrite('xp').doit() == (Integer(1)/(Integer(2)*m*hbar*omega))*(
108          Px**2 + (m*omega*X)**2) - Integer(1)/Integer(2)
109      assert N.rewrite('H').doit() == H/(hbar*omega) - Integer(1)/Integer(2)
110      for i in range(ndim):
111          assert N_rep[i,i] == i
112      assert N_rep == ad_rep_sympy*a_rep

114  def test_Hamiltonian():
115      assert Commutator(H, N).doit() == Integer(0)
116      assert qapply(H*k) == ((hbar*omega*(k.n + Integer(1)/Integer(2)))*k).expand()
117      assert H.rewrite('a').doit() == hbar*omega*(ad*a + Integer(1)/Integer(2))
118      assert H.rewrite('xp').doit() == \
119          (Integer(1)/(Integer(2)*m))*(Px**2 + (m*omega*X)**2)
120      assert H.rewrite('N').doit() == hbar*omega*(N + Integer(1)/Integer(2))
121      for i in range(ndim):
122          assert H_rep[i,i] == hbar*omega*(i + Integer(1)/Integer(2))

133      e1 = Sum(SHOKet(p)*SHOBra(p), (p, 0, 1))

161  def test_sho_coherant_state():
162      alpha = Symbol('alpha', is_complex=True)
163      cstate = exp(-Abs(alpha)**2/S(2))*Sum(((alpha**p)/sqrt(factorial(p)))*SHOKet(p), (p,0,oo))
164      # Projection onto the number eigenstate
165      assert qapply(SHOBra(q)*cstate, sum_doit=True) == exp(-Abs(alpha)**2/S(2))*alpha**q/sqrt(factorial(q))
166      # Ensure that the coherent state is an eigenstate of annihilation operator
167      assert simplify(qapply(SHOBra(q)*a*cstate, sum_doit=True)) == simplify(qapply(SHOBra(q)*alpha*cstate, sum_doit=True))

169  def test_issue_26495():
170      nbar = Symbol('nbar', real=True, nonnegative=True)
171      n = Symbol('n', integer=True)
172      i = Symbol('i', integer=True, nonnegative=True)
