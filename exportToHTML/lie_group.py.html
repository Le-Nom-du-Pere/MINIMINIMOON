54   def _ode_lie_group_try_heuristic(eq, heuristic, func, match, inf):
55   
56       xi = Function("xi")
57       eta = Function("eta")
58       f = func.func
59       x = func.args[0]
60       y = match['y']
61       h = match['h']
62       tempsol = []
63       if not inf:
64           try:
65               inf = infinitesimals(eq, hint=heuristic, func=func, order=1, match=match)
66           except ValueError:
67               return None
68       for infsim in inf:
69           xiinf = (infsim[xi(x, func)]).subs(func, y)
70           etainf = (infsim[eta(x, func)]).subs(func, y)
71           # This condition creates recursion while using pdsolve.
72           # Since the first step while solving a PDE of form
73           # a*(f(x, y).diff(x)) + b*(f(x, y).diff(y)) + c = 0
74           # is to solve the ODE dy/dx = b/a
75           if simplify(etainf/xiinf) == h:
76               continue
77           rpde = f(x, y).diff(x)*xiinf + f(x, y).diff(y)*etainf
78           r = pdsolve(rpde, func=f(x, y)).rhs
79           s = pdsolve(rpde - 1, func=f(x, y)).rhs
80           newcoord = [_lie_group_remove(coord) for coord in [r, s]]
81           r = Dummy("r")
82           s = Dummy("s")
83           C1 = Symbol("C1")
84           rcoord = newcoord[0]
85           scoord = newcoord[-1]
86           try:
87               sol = solve([r - rcoord, s - scoord], x, y, dict=True)
88               if sol == []:
89                   continue
90           except NotImplementedError:
91               continue
92           else:
93               sol = sol[0]
94               xsub = sol[x]
95               ysub = sol[y]
96               num = simplify(scoord.diff(x) + scoord.diff(y)*h)
97               denom = simplify(rcoord.diff(x) + rcoord.diff(y)*h)
98               if num and denom:
99                   diffeq = simplify((num/denom).subs([(x, xsub), (y, ysub)]))
100                  sep = separatevars(diffeq, symbols=[r, s], dict=True)
101                  if sep:
102                      # Trying to separate, r and s coordinates
103                      deq = integrate((1/sep[s]), s) + C1 - integrate(sep['coeff']*sep[r], r)
104                      # Substituting and reverting back to original coordinates
105                      deq = deq.subs([(r, rcoord), (s, scoord)])
106                      try:
107                          sdeq = solve(deq, y)
108                      except NotImplementedError:
109                          tempsol.append(deq)
110                      else:
111                          return [Eq(f(x), sol) for sol in sdeq]
112  
113  
114              elif denom: # (ds/dr) is zero which means s is constant
115                  return [Eq(f(x), solve(scoord - C1, y)[0])]
116  
117              elif num: # (dr/ds) is zero which means r is constant
118                  return [Eq(f(x), solve(rcoord - C1, y)[0])]
119  
120      # If nothing works, return solution as it is, without solving for y
121      if tempsol:
122          return [Eq(sol.subs(y, f(x)), 0) for sol in tempsol]
123      return None

126  def _ode_lie_group( s, func, order, match):
127  
128      heuristics = lie_heuristics
129      inf = {}
130      f = func.func
131      x = func.args[0]
132      df = func.diff(x)
133      xi = Function("xi")
134      eta = Function("eta")
135      xis = match['xi']
136      etas = match['eta']
137      y = match.pop('y', None)
138      if y:
139          h = -simplify(match[match['d']]/match[match['e']])
140      else:
141          y = Dummy("y")
142          h = s.subs(func, y)
143  
144      if xis is not None and etas is not None:
145          inf = [{xi(x, f(x)): S(xis), eta(x, f(x)): S(etas)}]
146  
147          if checkinfsol(Eq(df, s), inf, func=f(x), order=1)[0][0]:
148              heuristics = ["user_defined"] + list(heuristics)
149  
150      match = {'h': h, 'y': y}
151  
152      # This is done so that if any heuristic raises a ValueError
153      # another heuristic can be used.
154      sol = None
155      for heuristic in heuristics:
156          sol = _ode_lie_group_try_heuristic(Eq(df, s), heuristic, func, match, inf)
157          if sol:
158              return sol
159      return sol

162  def infinitesimals(eq, func=None, order=None, hint='default', match=None):
163      r""" 
164      The infinitesimal functions of an ordinary differential equation, `\xi(x,y)` 
165      and `\eta(x,y)`, are the infinitesimals of the Lie group of point transformations 
166      for which the differential equation is invariant. So, the ODE `y'=f(x,y)` 
167      would admit a Lie group `x^*=X(x,y;\varepsilon)=x+\varepsilon\xi(x,y)`, 
168      `y^*=Y(x,y;\varepsilon)=y+\varepsilon\eta(x,y)` such that `(y^*)'=f(x^*, y^*)`. 
169      A change of coordinates, to `r(x,y)` and `s(x,y)`, can be performed so this Lie group 
170      becomes the translation group, `r^*=r` and `s^*=s+\varepsilon`. 
171      They are tangents to the coordinate curves of the new system. 
172   
173      Consider the transformation `(x, y) \to (X, Y)` such that the 
174      differential equation remains invariant. `\xi` and `\eta` are the tangents to 
175      the transformed coordinates `X` and `Y`, at `\varepsilon=0`. 
176   
177      .. math:: \left(\frac{\partial X(x,y;\varepsilon)}{\partial\varepsilon 
178                  }\right)|_{\varepsilon=0} = \xi, 
179                \left(\frac{\partial Y(x,y;\varepsilon)}{\partial\varepsilon 
180                  }\right)|_{\varepsilon=0} = \eta, 
181   
182      The infinitesimals can be found by solving the following PDE: 
183   
184          >>> from sympy import Function, Eq, pprint 
185          >>> from sympy.abc import x, y 
186          >>> xi, eta, h = map(Function, ['xi', 'eta', 'h']) 
187          >>> h = h(x, y)  # dy/dx = h 
188          >>> eta = eta(x, y) 
189          >>> xi = xi(x, y) 
190          >>> genform = Eq(eta.diff(x) + (eta.diff(y) - xi.diff(x))*h 
191          ... - (xi.diff(y))*h**2 - xi*(h.diff(x)) - eta*(h.diff(y)), 0) 
192          >>> pprint(genform) 
193          /d               d           \                     d              2       d                       d             d 
194          |--(eta(x, y)) - --(xi(x, y))|*h(x, y) - eta(x, y)*--(h(x, y)) - h (x, y)*--(xi(x, y)) - xi(x, y)*--(h(x, y)) + --(eta(x, y)) = 0 
195          \dy              dx          /                     dy                     dy                      dx            dx 
196   
197      Solving the above mentioned PDE is not trivial, and can be solved only by 
198      making intelligent assumptions for `\xi` and `\eta` (heuristics). Once an 
199      infinitesimal is found, the attempt to find more heuristics stops. This is done to 
200      optimise the speed of solving the differential equation. If a list of all the 
201      infinitesimals is needed, ``hint`` should be flagged as ``all``, which gives 
202      the complete list of infinitesimals. If the infinitesimals for a particular 
203      heuristic needs to be found, it can be passed as a flag to ``hint``. 
204   
205      Examples 
206      ======== 
207   
208      >>> from sympy import Function 
209      >>> from sympy.solvers.ode.lie_group import infinitesimals 
210      >>> from sympy.abc import x 
211      >>> f = Function('f') 
212      >>> eq = f(x).diff(x) - x**2*f(x) 
213      >>> infinitesimals(eq) 
214      [{eta(x, f(x)): exp(x**3/3), xi(x, f(x)): 0}] 
215   
216      References 
217      ========== 
218   
219      - Solving differential equations by Symmetry Groups, 
220        John Starrett, pp. 1 - pp. 14 
221   
222      """
223  
224      if isinstance(eq, Equality):
225          eq = eq.lhs - eq.rhs
226      if not func:
227          eq, func = _preprocess(eq)
228      variables = func.args
229      if len(variables) != 1:
230          raise ValueError("ODE's have only one independent variable")
231      else:
232          x = variables[0]
233          if not order:
234              order = ode_order(eq, func)
235          if order != 1:
236              raise NotImplementedError("Infinitesimals for only "
237                  "first order ODE's have been implemented")
238          else:
239              df = func.diff(x)
240              # Matching differential equation of the form a*df + b
241              a = Wild('a', exclude = [df])
242              b = Wild('b', exclude = [df])
243              if match:  # Used by lie_group hint
244                  h = match['h']
245                  y = match['y']
246              else:
247                  match = collect(expand(eq), df).match(a*df + b)
248                  if match:
249                      h = -simplify(match[b]/match[a])
250                  else:
251                      try:
252                          sol = solve(eq, df)
253                      except NotImplementedError:
254                          raise NotImplementedError("Infinitesimals for the "
255                              "first order ODE could not be found")
256                      else:
257                          h = sol[0]  # Find infinitesimals for one solution
258                  y = Dummy("y")
259                  h = h.subs(func, y)
260  
261              u = Dummy("u")
262              hx = h.diff(x)
263              hy = h.diff(y)
264              hinv = ((1/h).subs([(x, u), (y, x)])).subs(u, y)  # Inverse ODE
265              match = {'h': h, 'func': func, 'hx': hx, 'hy': hy, 'y': y, 'hinv': hinv}
266              if hint == 'all':
267                  xieta = []
268                  for heuristic in lie_heuristics:
269                      function = globals()['lie_heuristic_' + heuristic]
270                      inflist = function(match, comp=True)
271                      if inflist:
272                          xieta.extend([inf for inf in inflist if inf not in xieta])
273                  if xieta:
274                      return xieta
275                  else:
276                      raise NotImplementedError("Infinitesimals could not be found for "
277                          "the given ODE")
278  
279              elif hint == 'default':
280                  for heuristic in lie_heuristics:
281                      function = globals()['lie_heuristic_' + heuristic]
282                      xieta = function(match, comp=False)
283                      if xieta:
284                          return xieta
285  
286                  raise NotImplementedError("Infinitesimals could not be found for"
287                      " the given ODE")
288  
289              elif hint not in lie_heuristics:
290                  raise ValueError("Heuristic not recognized: " + hint)
291  
292              else:
293                  function = globals()['lie_heuristic_' + hint]
294                  xieta = function(match, comp=True)
295                  if xieta:
296                      return xieta
297                  else:
298                      raise ValueError("Infinitesimals could not be found using the"
299                           " given heuristic")

302  def lie_heuristic_abaco1_simple(match, comp=False):
303      r""" 
304      The first heuristic uses the following four sets of 
305      assumptions on `\xi` and `\eta` 
306   
307      .. math:: \xi = 0, \eta = f(x) 
308   
309      .. math:: \xi = 0, \eta = f(y) 
310   
311      .. math:: \xi = f(x), \eta = 0 
312   
313      .. math:: \xi = f(y), \eta = 0 
314   
315      The success of this heuristic is determined by algebraic factorisation. 
316      For the first assumption `\xi = 0` and `\eta` to be a function of `x`, the PDE 
317   
318      .. math:: \frac{\partial \eta}{\partial x} + (\frac{\partial \eta}{\partial y} 
319                  - \frac{\partial \xi}{\partial x})*h 
320                  - \frac{\partial \xi}{\partial y}*h^{2} 
321                  - \xi*\frac{\partial h}{\partial x} - \eta*\frac{\partial h}{\partial y} = 0 
322   
323      reduces to `f'(x) - f\frac{\partial h}{\partial y} = 0` 
324      If `\frac{\partial h}{\partial y}` is a function of `x`, then this can usually 
325      be integrated easily. A similar idea is applied to the other 3 assumptions as well. 
326   
327   
328      References 
329      ========== 
330   
331      - E.S Cheb-Terrab, L.G.S Duarte and L.A,C.P da Mota, Computer Algebra 
332        Solving of First Order ODEs Using Symmetry Methods, pp. 8 
333   
334   
335      """
336  
337      xieta = []
338      y = match['y']
339      h = match['h']
340      func = match['func']
341      x = func.args[0]
342      hx = match['hx']
343      hy = match['hy']
344      xi = Function('xi')(x, func)
345      eta = Function('eta')(x, func)
346  
347      hysym = hy.free_symbols
348      if y not in hysym:
349          try:
350              fx = exp(integrate(hy, x))
351          except NotImplementedError:
352              pass
353          else:
354              inf = {xi: S.Zero, eta: fx}
355              if not comp:
356                  return [inf]
357              if comp and inf not in xieta:
358                  xieta.append(inf)
359  
360      factor = hy/h
361      facsym = factor.free_symbols
362      if x not in facsym:
363          try:
364              fy = exp(integrate(factor, y))
365          except NotImplementedError:
366              pass
367          else:
368              inf = {xi: S.Zero, eta: fy.subs(y, func)}
369              if not comp:
370                  return [inf]
371              if comp and inf not in xieta:
372                  xieta.append(inf)
373  
374      factor = -hx/h
375      facsym = factor.free_symbols
376      if y not in facsym:
377          try:
378              fx = exp(integrate(factor, x))
379          except NotImplementedError:
380              pass
381          else:
382              inf = {xi: fx, eta: S.Zero}
383              if not comp:
384                  return [inf]
385              if comp and inf not in xieta:
386                  xieta.append(inf)
387  
388      factor = -hx/(h**2)
389      facsym = factor.free_symbols
390      if x not in facsym:
391          try:
392              fy = exp(integrate(factor, y))
393          except NotImplementedError:
394              pass
395          else:
396              inf = {xi: fy.subs(y, func), eta: S.Zero}
397              if not comp:
398                  return [inf]
399              if comp and inf not in xieta:
400                  xieta.append(inf)
401  
402      if xieta:
403          return xieta

405  def lie_heuristic_abaco1_product(match, comp=False):
406      r""" 
407      The second heuristic uses the following two assumptions on `\xi` and `\eta` 
408   
409      .. math:: \eta = 0, \xi = f(x)*g(y) 
410   
411      .. math:: \eta = f(x)*g(y), \xi = 0 
412   
413      The first assumption of this heuristic holds good if 
414      `\frac{1}{h^{2}}\frac{\partial^2}{\partial x \partial y}\log(h)` is 
415      separable in `x` and `y`, then the separated factors containing `x` 
416      is `f(x)`, and `g(y)` is obtained by 
417   
418      .. math:: e^{\int f\frac{\partial}{\partial x}\left(\frac{1}{f*h}\right)\,dy} 
419   
420      provided `f\frac{\partial}{\partial x}\left(\frac{1}{f*h}\right)` is a function 
421      of `y` only. 
422   
423      The second assumption holds good if `\frac{dy}{dx} = h(x, y)` is rewritten as 
424      `\frac{dy}{dx} = \frac{1}{h(y, x)}` and the same properties of the first assumption 
425      satisfies. After obtaining `f(x)` and `g(y)`, the coordinates are again 
426      interchanged, to get `\eta` as `f(x)*g(y)` 
427   
428   
429      References 
430      ========== 
431      - E.S. Cheb-Terrab, A.D. Roche, Symmetries and First Order 
432        ODE Patterns, pp. 7 - pp. 8 
433   
434      """
435  
436      xieta = []
437      y = match['y']
438      h = match['h']
439      hinv = match['hinv']
440      func = match['func']
441      x = func.args[0]
442      xi = Function('xi')(x, func)
443      eta = Function('eta')(x, func)
444  
445  
446      inf = separatevars(((log(h).diff(y)).diff(x))/h**2, dict=True, symbols=[x, y])
447      if inf and inf['coeff']:
448          fx = inf[x]
449          gy = simplify(fx*((1/(fx*h)).diff(x)))
450          gysyms = gy.free_symbols
451          if x not in gysyms:
452              gy = exp(integrate(gy, y))
453              inf = {eta: S.Zero, xi: (fx*gy).subs(y, func)}
454              if not comp:
455                  return [inf]
456              if comp and inf not in xieta:
457                  xieta.append(inf)
458  
459      u1 = Dummy("u1")
460      inf = separatevars(((log(hinv).diff(y)).diff(x))/hinv**2, dict=True, symbols=[x, y])
461      if inf and inf['coeff']:
462          fx = inf[x]
463          gy = simplify(fx*((1/(fx*hinv)).diff(x)))
464          gysyms = gy.free_symbols
465          if x not in gysyms:
466              gy = exp(integrate(gy, y))
467              etaval = fx*gy
468              etaval = (etaval.subs([(x, u1), (y, x)])).subs(u1, y)
469              inf = {eta: etaval.subs(y, func), xi: S.Zero}
470              if not comp:
471                  return [inf]
472              if comp and inf not in xieta:
473                  xieta.append(inf)
474  
475      if xieta:
476          return xieta

478  def lie_heuristic_bivariate(match, comp=False):
479      r""" 
480      The third heuristic assumes the infinitesimals `\xi` and `\eta` 
481      to be bi-variate polynomials in `x` and `y`. The assumption made here 
482      for the logic below is that `h` is a rational function in `x` and `y` 
483      though that may not be necessary for the infinitesimals to be 
484      bivariate polynomials. The coefficients of the infinitesimals 
485      are found out by substituting them in the PDE and grouping similar terms 
486      that are polynomials and since they form a linear system, solve and check 
487      for non trivial solutions. The degree of the assumed bivariates 
488      are increased till a certain maximum value. 
489   
490      References 
491      ========== 
492      - Lie Groups and Differential Equations 
493        pp. 327 - pp. 329 
494   
495      """
496  
497      h = match['h']
498      hx = match['hx']
499      hy = match['hy']
500      func = match['func']
501      x = func.args[0]
502      y = match['y']
503      xi = Function('xi')(x, func)
504      eta = Function('eta')(x, func)
505  
506      if h.is_rational_function():
507          # The maximum degree that the infinitesimals can take is
508          # calculated by this technique.
509          etax, etay, etad, xix, xiy, xid = symbols("etax etay etad xix xiy xid")
510          ipde = etax + (etay - xix)*h - xiy*h**2 - xid*hx - etad*hy
511          num, denom = cancel(ipde).as_numer_denom()
512          deg = Poly(num, x, y).total_degree()
513          deta = Function('deta')(x, y)
514          dxi = Function('dxi')(x, y)
515          ipde = (deta.diff(x) + (deta.diff(y) - dxi.diff(x))*h - (dxi.diff(y))*h**2
516              - dxi*hx - deta*hy)
517          xieq = Symbol("xi0")
518          etaeq = Symbol("eta0")
519  
520          for i in range(deg + 1):
521              if i:
522                  xieq += Add(*[
523                      Symbol("xi_" + str(power) + "_" + str(i - power))*x**power*y**(i - power)
524                      for power in range(i + 1)])
525                  etaeq += Add(*[
526                      Symbol("eta_" + str(power) + "_" + str(i - power))*x**power*y**(i - power)
527                      for power in range(i + 1)])
528              pden, denom = (ipde.subs({dxi: xieq, deta: etaeq}).doit()).as_numer_denom()
529              pden = expand(pden)
530  
531              # If the individual terms are monomials, the coefficients
532              # are grouped
533              if pden.is_polynomial(x, y) and pden.is_Add:
534                  polyy = Poly(pden, x, y).as_dict()
535              if polyy:
536                  symset = xieq.free_symbols.union(etaeq.free_symbols) - {x, y}
537                  soldict = solve(polyy.values(), *symset)
538                  if isinstance(soldict, list):
539                      soldict = soldict[0]
540                  if any(soldict.values()):
541                      xired = xieq.subs(soldict)
542                      etared = etaeq.subs(soldict)
543                      # Scaling is done by substituting one for the parameters
544                      # This can be any number except zero.
545                      dict_ = dict.fromkeys(symset, 1)
546                      inf = {eta: etared.subs(dict_).subs(y, func),
547                          xi: xired.subs(dict_).subs(y, func)}
548                      return [inf]

550  def lie_heuristic_chi(match, comp=False):
551      r""" 
552      The aim of the fourth heuristic is to find the function `\chi(x, y)` 
553      that satisfies the PDE `\frac{d\chi}{dx} + h\frac{d\chi}{dx} 
554      - \frac{\partial h}{\partial y}\chi = 0`. 
555   
556      This assumes `\chi` to be a bivariate polynomial in `x` and `y`. By intuition, 
557      `h` should be a rational function in `x` and `y`. The method used here is 
558      to substitute a general binomial for `\chi` up to a certain maximum degree 
559      is reached. The coefficients of the polynomials, are calculated by by collecting 
560      terms of the same order in `x` and `y`. 
561   
562      After finding `\chi`, the next step is to use `\eta = \xi*h + \chi`, to 
563      determine `\xi` and `\eta`. This can be done by dividing `\chi` by `h` 
564      which would give `-\xi` as the quotient and `\eta` as the remainder. 
565   
566   
567      References 
568      ========== 
569      - E.S Cheb-Terrab, L.G.S Duarte and L.A,C.P da Mota, Computer Algebra 
570        Solving of First Order ODEs Using Symmetry Methods, pp. 8 
571   
572      """
573  
574      h = match['h']
575      hy = match['hy']
576      func = match['func']
577      x = func.args[0]
578      y = match['y']
579      xi = Function('xi')(x, func)
580      eta = Function('eta')(x, func)
581  
582      if h.is_rational_function():
583          schi, schix, schiy = symbols("schi, schix, schiy")
584          cpde = schix + h*schiy - hy*schi
585          num, denom = cancel(cpde).as_numer_denom()
586          deg = Poly(num, x, y).total_degree()
587  
588          chi = Function('chi')(x, y)
589          chix = chi.diff(x)
590          chiy = chi.diff(y)
591          cpde = chix + h*chiy - hy*chi
592          chieq = Symbol("chi")
593          for i in range(1, deg + 1):
594              chieq += Add(*[
595                  Symbol("chi_" + str(power) + "_" + str(i - power))*x**power*y**(i - power)
596                  for power in range(i + 1)])
597              cnum, cden = cancel(cpde.subs({chi : chieq}).doit()).as_numer_denom()
598              cnum = expand(cnum)
599              if cnum.is_polynomial(x, y) and cnum.is_Add:
600                  cpoly = Poly(cnum, x, y).as_dict()
601                  if cpoly:
602                      solsyms = chieq.free_symbols - {x, y}
603                      soldict = solve(cpoly.values(), *solsyms)
604                      if isinstance(soldict, list):
605                          soldict = soldict[0]
606                      if any(soldict.values()):
607                          chieq = chieq.subs(soldict)
608                          dict_ = dict.fromkeys(solsyms, 1)
609                          chieq = chieq.subs(dict_)
610                          # After finding chi, the main aim is to find out
611                          # eta, xi by the equation eta = xi*h + chi
612                          # One method to set xi, would be rearranging it to
613                          # (eta/h) - xi = (chi/h). This would mean dividing
614                          # chi by h would give -xi as the quotient and eta
615                          # as the remainder. Thanks to Sean Vig for suggesting
616                          # this method.
617                          xic, etac = div(chieq, h)
618                          inf = {eta: etac.subs(y, func), xi: -xic.subs(y, func)}
619                          return [inf]

621  def lie_heuristic_function_sum(match, comp=False):
622      r""" 
623      This heuristic uses the following two assumptions on `\xi` and `\eta` 
624   
625      .. math:: \eta = 0, \xi = f(x) + g(y) 
626   
627      .. math:: \eta = f(x) + g(y), \xi = 0 
628   
629      The first assumption of this heuristic holds good if 
630   
631      .. math:: \frac{\partial}{\partial y}[(h\frac{\partial^{2}}{ 
632                  \partial x^{2}}(h^{-1}))^{-1}] 
633   
634      is separable in `x` and `y`, 
635   
636      1. The separated factors containing `y` is `\frac{\partial g}{\partial y}`. 
637         From this `g(y)` can be determined. 
638      2. The separated factors containing `x` is `f''(x)`. 
639      3. `h\frac{\partial^{2}}{\partial x^{2}}(h^{-1})` equals 
640         `\frac{f''(x)}{f(x) + g(y)}`. From this `f(x)` can be determined. 
641   
642      The second assumption holds good if `\frac{dy}{dx} = h(x, y)` is rewritten as 
643      `\frac{dy}{dx} = \frac{1}{h(y, x)}` and the same properties of the first 
644      assumption satisfies. After obtaining `f(x)` and `g(y)`, the coordinates 
645      are again interchanged, to get `\eta` as `f(x) + g(y)`. 
646   
647      For both assumptions, the constant factors are separated among `g(y)` 
648      and `f''(x)`, such that `f''(x)` obtained from 3] is the same as that 
649      obtained from 2]. If not possible, then this heuristic fails. 
650   
651   
652      References 
653      ========== 
654      - E.S. Cheb-Terrab, A.D. Roche, Symmetries and First Order 
655        ODE Patterns, pp. 7 - pp. 8 
656   
657      """
658  
659      xieta = []
660      h = match['h']
661      func = match['func']
662      hinv = match['hinv']
663      x = func.args[0]
664      y = match['y']
665      xi = Function('xi')(x, func)
666      eta = Function('eta')(x, func)
667  
668      for odefac in [h, hinv]:
669          factor = odefac*((1/odefac).diff(x, 2))
670          sep = separatevars((1/factor).diff(y), dict=True, symbols=[x, y])
671          if sep and sep['coeff'] and sep[x].has(x) and sep[y].has(y):
672              k = Dummy("k")
673              try:
674                  gy = k*integrate(sep[y], y)
675              except NotImplementedError:
676                  pass
677              else:
678                  fdd = 1/(k*sep[x]*sep['coeff'])
679                  fx = simplify(fdd/factor - gy)
680                  check = simplify(fx.diff(x, 2) - fdd)
681                  if fx:
682                      if not check:
683                          fx = fx.subs(k, 1)
684                          gy = (gy/k)
685                      else:
686                          sol = solve(check, k)
687                          if sol:
688                              sol = sol[0]
689                              fx = fx.subs(k, sol)
690                              gy = (gy/k)*sol
691                          else:
692                              continue
693                      if odefac == hinv:  # Inverse ODE
694                          fx = fx.subs(x, y)
695                          gy = gy.subs(y, x)
696                      etaval = factor_terms(fx + gy)
697                      if etaval.is_Mul:
698                          etaval = Mul(*[arg for arg in etaval.args if arg.has(x, y)])
699                      if odefac == hinv:  # Inverse ODE
700                          inf = {eta: etaval.subs(y, func), xi : S.Zero}
701                      else:
702                          inf = {xi: etaval.subs(y, func), eta : S.Zero}
703                      if not comp:
704                          return [inf]
705                      else:
706                          xieta.append(inf)
707  
708          if xieta:
709              return xieta

711  def lie_heuristic_abaco2_similar(match, comp=False):
712      r""" 
713      This heuristic uses the following two assumptions on `\xi` and `\eta` 
714   
715      .. math:: \eta = g(x), \xi = f(x) 
716   
717      .. math:: \eta = f(y), \xi = g(y) 
718   
719      For the first assumption, 
720   
721      1. First `\frac{\frac{\partial h}{\partial y}}{\frac{\partial^{2} h}{ 
722         \partial yy}}` is calculated. Let us say this value is A 
723   
724      2. If this is constant, then `h` is matched to the form `A(x) + B(x)e^{ 
725         \frac{y}{C}}` then, `\frac{e^{\int \frac{A(x)}{C} \,dx}}{B(x)}` gives `f(x)` 
726         and `A(x)*f(x)` gives `g(x)` 
727   
728      3. Otherwise `\frac{\frac{\partial A}{\partial X}}{\frac{\partial A}{ 
729         \partial Y}} = \gamma` is calculated. If 
730   
731         a] `\gamma` is a function of `x` alone 
732   
733         b] `\frac{\gamma\frac{\partial h}{\partial y} - \gamma'(x) - \frac{ 
734         \partial h}{\partial x}}{h + \gamma} = G` is a function of `x` alone. 
735         then, `e^{\int G \,dx}` gives `f(x)` and `-\gamma*f(x)` gives `g(x)` 
736   
737      The second assumption holds good if `\frac{dy}{dx} = h(x, y)` is rewritten as 
738      `\frac{dy}{dx} = \frac{1}{h(y, x)}` and the same properties of the first assumption 
739      satisfies. After obtaining `f(x)` and `g(x)`, the coordinates are again 
740      interchanged, to get `\xi` as `f(x^*)` and `\eta` as `g(y^*)` 
741   
742      References 
743      ========== 
744      - E.S. Cheb-Terrab, A.D. Roche, Symmetries and First Order 
745        ODE Patterns, pp. 10 - pp. 12 
746   
747      """
748  
749      h = match['h']
750      hx = match['hx']
751      hy = match['hy']
752      func = match['func']
753      hinv = match['hinv']
754      x = func.args[0]
755      y = match['y']
756      xi = Function('xi')(x, func)
757      eta = Function('eta')(x, func)
758  
759      factor = cancel(h.diff(y)/h.diff(y, 2))
760      factorx = factor.diff(x)
761      factory = factor.diff(y)
762      if not factor.has(x) and not factor.has(y):
763          A = Wild('A', exclude=[y])
764          B = Wild('B', exclude=[y])
765          C = Wild('C', exclude=[x, y])
766          match = h.match(A + B*exp(y/C))
767          try:
768              tau = exp(-integrate(match[A]/match[C]), x)/match[B]
769          except NotImplementedError:
770              pass
771          else:
772              gx = match[A]*tau
773              return [{xi: tau, eta: gx}]
774  
775      else:
776          gamma = cancel(factorx/factory)
777          if not gamma.has(y):
778              tauint = cancel((gamma*hy - gamma.diff(x) - hx)/(h + gamma))
779              if not tauint.has(y):
780                  try:
781                      tau = exp(integrate(tauint, x))
782                  except NotImplementedError:
783                      pass
784                  else:
785                      gx = -tau*gamma
786                      return [{xi: tau, eta: gx}]
787  
788      factor = cancel(hinv.diff(y)/hinv.diff(y, 2))
789      factorx = factor.diff(x)
790      factory = factor.diff(y)
791      if not factor.has(x) and not factor.has(y):
792          A = Wild('A', exclude=[y])
793          B = Wild('B', exclude=[y])
794          C = Wild('C', exclude=[x, y])
795          match = h.match(A + B*exp(y/C))
796          try:
797              tau = exp(-integrate(match[A]/match[C]), x)/match[B]
798          except NotImplementedError:
799              pass
800          else:
801              gx = match[A]*tau
802              return [{eta: tau.subs(x, func), xi: gx.subs(x, func)}]
803  
804      else:
805          gamma = cancel(factorx/factory)
806          if not gamma.has(y):
807              tauint = cancel((gamma*hinv.diff(y) - gamma.diff(x) - hinv.diff(x))/(
808                  hinv + gamma))
809              if not tauint.has(y):
810                  try:
811                      tau = exp(integrate(tauint, x))
812                  except NotImplementedError:
813                      pass
814                  else:
815                      gx = -tau*gamma
816                      return [{eta: tau.subs(x, func), xi: gx.subs(x, func)}]

819  def lie_heuristic_abaco2_unique_unknown(match, comp=False):
820      r""" 
821      This heuristic assumes the presence of unknown functions or known functions 
822      with non-integer powers. 
823   
824      1. A list of all functions and non-integer powers containing x and y 
825      2. Loop over each element `f` in the list, find `\frac{\frac{\partial f}{\partial x}}{ 
826         \frac{\partial f}{\partial x}} = R` 
827   
828         If it is separable in `x` and `y`, let `X` be the factors containing `x`. Then 
829   
830         a] Check if `\xi = X` and `\eta = -\frac{X}{R}` satisfy the PDE. If yes, then return 
831            `\xi` and `\eta` 
832         b] Check if `\xi = \frac{-R}{X}` and `\eta = -\frac{1}{X}` satisfy the PDE. 
833             If yes, then return `\xi` and `\eta` 
834   
835         If not, then check if 
836   
837         a] :math:`\xi = -R,\eta = 1` 
838   
839         b] :math:`\xi = 1, \eta = -\frac{1}{R}` 
840   
841         are solutions. 
842   
843      References 
844      ========== 
845      - E.S. Cheb-Terrab, A.D. Roche, Symmetries and First Order 
846        ODE Patterns, pp. 10 - pp. 12 
847   
848      """
849  
850      h = match['h']
851      hx = match['hx']
852      hy = match['hy']
853      func = match['func']
854      x = func.args[0]
855      y = match['y']
856      xi = Function('xi')(x, func)
857      eta = Function('eta')(x, func)
858  
859      funclist = []
860      for atom in h.atoms(Pow):
861          base, exp = atom.as_base_exp()
862          if base.has(x) and base.has(y):
863              if not exp.is_Integer:
864                  funclist.append(atom)
865  
866      for function in h.atoms(AppliedUndef):
867          syms = function.free_symbols
868          if x in syms and y in syms:
869              funclist.append(function)
870  
871      for f in funclist:
872          frac = cancel(f.diff(y)/f.diff(x))
873          sep = separatevars(frac, dict=True, symbols=[x, y])
874          if sep and sep['coeff']:
875              xitry1 = sep[x]
876              etatry1 = -1/(sep[y]*sep['coeff'])
877              pde1 = etatry1.diff(y)*h - xitry1.diff(x)*h - xitry1*hx - etatry1*hy
878              if not simplify(pde1):
879                  return [{xi: xitry1, eta: etatry1.subs(y, func)}]
880              xitry2 = 1/etatry1
881              etatry2 = 1/xitry1
882              pde2 = etatry2.diff(x) - (xitry2.diff(y))*h**2 - xitry2*hx - etatry2*hy
883              if not simplify(expand(pde2)):
884                  return [{xi: xitry2.subs(y, func), eta: etatry2}]
885  
886          else:
887              etatry = -1/frac
888              pde = etatry.diff(x) + etatry.diff(y)*h - hx - etatry*hy
889              if not simplify(pde):
890                  return [{xi: S.One, eta: etatry.subs(y, func)}]
891              xitry = -frac
892              pde = -xitry.diff(x)*h -xitry.diff(y)*h**2 - xitry*hx -hy
893              if not simplify(expand(pde)):
894                  return [{xi: xitry.subs(y, func), eta: S.One}]

897  def lie_heuristic_abaco2_unique_general(match, comp=False):
898      r""" 
899      This heuristic finds if infinitesimals of the form `\eta = f(x)`, `\xi = g(y)` 
900      without making any assumptions on `h`. 
901   
902      The complete sequence of steps is given in the paper mentioned below. 
903   
904      References 
905      ========== 
906      - E.S. Cheb-Terrab, A.D. Roche, Symmetries and First Order 
907        ODE Patterns, pp. 10 - pp. 12 
908   
909      """
910      hx = match['hx']
911      hy = match['hy']
912      func = match['func']
913      x = func.args[0]
914      y = match['y']
915      xi = Function('xi')(x, func)
916      eta = Function('eta')(x, func)
917  
918      A = hx.diff(y)
919      B = hy.diff(y) + hy**2
920      C = hx.diff(x) - hx**2
921  
922      if not (A and B and C):
923          return
924  
925      Ax = A.diff(x)
926      Ay = A.diff(y)
927      Axy = Ax.diff(y)
928      Axx = Ax.diff(x)
929      Ayy = Ay.diff(y)
930      D = simplify(2*Axy + hx*Ay - Ax*hy + (hx*hy + 2*A)*A)*A - 3*Ax*Ay
931      if not D:
932          E1 = simplify(3*Ax**2 + ((hx**2 + 2*C)*A - 2*Axx)*A)
933          if E1:
934              E2 = simplify((2*Ayy + (2*B - hy**2)*A)*A - 3*Ay**2)
935              if not E2:
936                  E3 = simplify(
937                      E1*((28*Ax + 4*hx*A)*A**3 - E1*(hy*A + Ay)) - E1.diff(x)*8*A**4)
938                  if not E3:
939                      etaval = cancel((4*A**3*(Ax - hx*A) + E1*(hy*A - Ay))/(S(2)*A*E1))
940                      if x not in etaval:
941                          try:
942                              etaval = exp(integrate(etaval, y))
943                          except NotImplementedError:
944                              pass
945                          else:
946                              xival = -4*A**3*etaval/E1
947                              if y not in xival:
948                                  return [{xi: xival, eta: etaval.subs(y, func)}]
949  
950      else:
951          E1 = simplify((2*Ayy + (2*B - hy**2)*A)*A - 3*Ay**2)
952          if E1:
953              E2 = simplify(
954                  4*A**3*D - D**2 + E1*((2*Axx - (hx**2 + 2*C)*A)*A - 3*Ax**2))
955              if not E2:
956                  E3 = simplify(
957                     -(A*D)*E1.diff(y) + ((E1.diff(x) - hy*D)*A + 3*Ay*D +
958                      (A*hx - 3*Ax)*E1)*E1)
959                  if not E3:
960                      etaval = cancel(((A*hx - Ax)*E1 - (Ay + A*hy)*D)/(S(2)*A*D))
961                      if x not in etaval:
962                          try:
963                              etaval = exp(integrate(etaval, y))
964                          except NotImplementedError:
965                              pass
966                          else:
967                              xival = -E1*etaval/D
968                              if y not in xival:
969                                  return [{xi: xival, eta: etaval.subs(y, func)}]

972  def lie_heuristic_linear(match, comp=False):
973      r""" 
974      This heuristic assumes 
975   
976      1. `\xi = ax + by + c` and 
977      2. `\eta = fx + gy + h` 
978   
979      After substituting the following assumptions in the determining PDE, it 
980      reduces to 
981   
982      .. math:: f + (g - a)h - bh^{2} - (ax + by + c)\frac{\partial h}{\partial x} 
983                   - (fx + gy + c)\frac{\partial h}{\partial y} 
984   
985      Solving the reduced PDE obtained, using the method of characteristics, becomes 
986      impractical. The method followed is grouping similar terms and solving the system 
987      of linear equations obtained. The difference between the bivariate heuristic is that 
988      `h` need not be a rational function in this case. 
989   
990      References 
991      ========== 
992      - E.S. Cheb-Terrab, A.D. Roche, Symmetries and First Order 
993        ODE Patterns, pp. 10 - pp. 12 
994   
995      """
996      h = match['h']
997      hx = match['hx']
998      hy = match['hy']
999      func = match['func']
1000     x = func.args[0]
1001     y = match['y']
1002     xi = Function('xi')(x, func)
1003     eta = Function('eta')(x, func)
1004 
1005     coeffdict = {}
1006     symbols = numbered_symbols("c", cls=Dummy)
1007     symlist = [next(symbols) for _ in islice(symbols, 6)]
1008     C0, C1, C2, C3, C4, C5 = symlist
1009     pde = C3 + (C4 - C0)*h - (C0*x + C1*y + C2)*hx - (C3*x + C4*y + C5)*hy - C1*h**2
1010     pde, denom = pde.as_numer_denom()
1011     pde = powsimp(expand(pde))
1012     if pde.is_Add:
1013         terms = pde.args
1014         for term in terms:
1015             if term.is_Mul:
1016                 rem = Mul(*[m for m in term.args if not m.has(x, y)])
1017                 xypart = term/rem
1018                 if xypart not in coeffdict:
1019                     coeffdict[xypart] = rem
1020                 else:
1021                     coeffdict[xypart] += rem
1022             else:
1023                 if term not in coeffdict:
1024                     coeffdict[term] = S.One
1025                 else:
1026                     coeffdict[term] += S.One
1027 
1028     sollist = coeffdict.values()
1029     soldict = solve(sollist, symlist)
1030     if soldict:
1031         if isinstance(soldict, list):
1032             soldict = soldict[0]
1033         subval = soldict.values()
1034         if any(t for t in subval):
1035             onedict = dict(zip(symlist, [1]*6))
1036             xival = C0*x + C1*func + C2
1037             etaval = C3*x + C4*func + C5
1038             xival = xival.subs(soldict)
1039             etaval = etaval.subs(soldict)
1040             xival = xival.subs(onedict)
1041             etaval = etaval.subs(onedict)
1042             return [{xi: xival, eta: etaval}]

1045 def _lie_group_remove(coords):
1046     r""" 
1047     This function is strictly meant for internal use by the Lie group ODE solving 
1048     method. It replaces arbitrary functions returned by pdsolve as follows: 
1049  
1050     1] If coords is an arbitrary function, then its argument is returned. 
1051     2] An arbitrary function in an Add object is replaced by zero. 
1052     3] An arbitrary function in a Mul object is replaced by one. 
1053     4] If there is no arbitrary function coords is returned unchanged. 
1054  
1055     Examples 
1056     ======== 
1057  
1058     >>> from sympy.solvers.ode.lie_group import _lie_group_remove 
1059     >>> from sympy import Function 
1060     >>> from sympy.abc import x, y 
1061     >>> F = Function("F") 
1062     >>> eq = x**2*y 
1063     >>> _lie_group_remove(eq) 
1064     x**2*y 
1065     >>> eq = F(x**2*y) 
1066     >>> _lie_group_remove(eq) 
1067     x**2*y 
1068     >>> eq = x*y**2 + F(x**3) 
1069     >>> _lie_group_remove(eq) 
1070     x*y**2 
1071     >>> eq = (F(x**3) + y)*x**4 
1072     >>> _lie_group_remove(eq) 
1073     x**4*y 
1074  
1075     """
1076     if isinstance(coords, AppliedUndef):
1077         return coords.args[0]
1078     elif coords.is_Add:
1079         subfunc = coords.atoms(AppliedUndef)
1080         if subfunc:
1081             for func in subfunc:
1082                 coords = coords.subs(func, 0)
1083         return coords
1084     elif coords.is_Pow:
1085         base, expr = coords.as_base_exp()
1086         base = _lie_group_remove(base)
1087         expr = _lie_group_remove(expr)
1088         return base**expr
1089     elif coords.is_Mul:
1090         mulargs = []
1091         coordargs = coords.args
1092         for arg in coordargs:
1093             if not isinstance(coords, AppliedUndef):
1094                 mulargs.append(_lie_group_remove(arg))
1095         return Mul(*mulargs)
1096     return coords
1097
