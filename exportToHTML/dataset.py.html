39   class Dataset(Generic[_T_co]):
40       r"""An abstract class representing a :class:`Dataset`. 
41    
42       All datasets that represent a map from keys to data samples should subclass 
43       it. All subclasses should overwrite :meth:`__getitem__`, supporting fetching a 
44       data sample for a given key. Subclasses could also optionally overwrite 
45       :meth:`__len__`, which is expected to return the size of the dataset by many 
46       :class:`~torch.utils.data.Sampler` implementations and the default options 
47       of :class:`~torch.utils.data.DataLoader`. Subclasses could also 
48       optionally implement :meth:`__getitems__`, for speedup batched samples 
49       loading. This method accepts list of indices of samples of batch and returns 
50       list of samples. 
51    
52       .. note:: 
53         :class:`~torch.utils.data.DataLoader` by default constructs an index 
54         sampler that yields integral indices.  To make it work with a map-style 
55         dataset with non-integral indices/keys, a custom sampler must be provided. 
56       """
57   
58       def __getitem__(self, index) -> _T_co:
59           raise NotImplementedError("Subclasses of Dataset should implement __getitem__.")
60   
61       # def __getitems__(self, indices: List) -> List[_T_co]:
62       # Not implemented to prevent false-positives in fetcher check in
63       # torch.utils.data._utils.fetch._MapDatasetFetcher
64   
65       def __add__(self, other: "Dataset[_T_co]") -> "ConcatDataset[_T_co]":
66           return ConcatDataset([self, other])
67   
68       # No `def __len__(self)` default?
69       # See NOTE [ Lack of Default `__len__` in Python Abstract Base Classes ]
70       # in pytorch/torch/utils/data/sampler.py

73   class IterableDataset(Dataset[_T_co], Iterable[_T_co]):
74       r"""An iterable Dataset. 
75    
76       All datasets that represent an iterable of data samples should subclass it. 
77       Such form of datasets is particularly useful when data come from a stream. 
78    
79       All subclasses should overwrite :meth:`__iter__`, which would return an 
80       iterator of samples in this dataset. 
81    
82       When a subclass is used with :class:`~torch.utils.data.DataLoader`, each 
83       item in the dataset will be yielded from the :class:`~torch.utils.data.DataLoader` 
84       iterator. When :attr:`num_workers > 0`, each worker process will have a 
85       different copy of the dataset object, so it is often desired to configure 
86       each copy independently to avoid having duplicate data returned from the 
87       workers. :func:`~torch.utils.data.get_worker_info`, when called in a worker 
88       process, returns information about the worker. It can be used in either the 
89       dataset's :meth:`__iter__` method or the :class:`~torch.utils.data.DataLoader` 's 
90       :attr:`worker_init_fn` option to modify each copy's behavior. 
91    
92       Example 1: splitting workload across all workers in :meth:`__iter__`:: 
93    
94           >>> # xdoctest: +REQUIRES(env:TORCH_DOCTEST_DATALOADER) 
95           >>> # xdoctest: +SKIP("Fails on MacOS12") 
96           >>> class MyIterableDataset(torch.utils.data.IterableDataset): 
97           ...     def __init__(self, start, end): 
98           ...         super(MyIterableDataset).__init__() 
99           ...         assert end > start, "this example code only works with end >= start" 
100          ...         self.start = start 
101          ...         self.end = end 
102          ... 
103          ...     def __iter__(self): 
104          ...         worker_info = torch.utils.data.get_worker_info() 
105          ...         if worker_info is None:  # single-process data loading, return the full iterator 
106          ...             iter_start = self.start 
107          ...             iter_end = self.end 
108          ...         else:  # in a worker process 
109          ...             # split workload 
110          ...             per_worker = int(math.ceil((self.end - self.start) / float(worker_info.num_workers))) 
111          ...             worker_id = worker_info.id 
112          ...             iter_start = self.start + worker_id * per_worker 
113          ...             iter_end = min(iter_start + per_worker, self.end) 
114          ...         return iter(range(iter_start, iter_end)) 
115          ... 
116          >>> # should give same set of data as range(3, 7), i.e., [3, 4, 5, 6]. 
117          >>> ds = MyIterableDataset(start=3, end=7) 
118   
119          >>> # Single-process loading 
120          >>> print(list(torch.utils.data.DataLoader(ds, num_workers=0))) 
121          [tensor([3]), tensor([4]), tensor([5]), tensor([6])] 
122   
123          >>> # xdoctest: +REQUIRES(POSIX) 
124          >>> # Multi-process loading with two worker processes 
125          >>> # Worker 0 fetched [3, 4].  Worker 1 fetched [5, 6]. 
126          >>> # xdoctest: +IGNORE_WANT("non deterministic") 
127          >>> print(list(torch.utils.data.DataLoader(ds, num_workers=2))) 
128          [tensor([3]), tensor([5]), tensor([4]), tensor([6])] 
129   
130          >>> # With even more workers 
131          >>> # xdoctest: +IGNORE_WANT("non deterministic") 
132          >>> print(list(torch.utils.data.DataLoader(ds, num_workers=12))) 
133          [tensor([3]), tensor([5]), tensor([4]), tensor([6])] 
134   
135      Example 2: splitting workload across all workers using :attr:`worker_init_fn`:: 
136   
137          >>> # xdoctest: +REQUIRES(env:TORCH_DOCTEST_DATALOADER) 
138          >>> class MyIterableDataset(torch.utils.data.IterableDataset): 
139          ...     def __init__(self, start, end): 
140          ...         super(MyIterableDataset).__init__() 
141          ...         assert end > start, "this example code only works with end >= start" 
142          ...         self.start = start 
143          ...         self.end = end 
144          ... 
145          ...     def __iter__(self): 
146          ...         return iter(range(self.start, self.end)) 
147          ... 
148          >>> # should give same set of data as range(3, 7), i.e., [3, 4, 5, 6]. 
149          >>> ds = MyIterableDataset(start=3, end=7) 
150   
151          >>> # Single-process loading 
152          >>> print(list(torch.utils.data.DataLoader(ds, num_workers=0))) 
153          [3, 4, 5, 6] 
154          >>> 
155          >>> # Directly doing multi-process loading yields duplicate data 
156          >>> print(list(torch.utils.data.DataLoader(ds, num_workers=2))) 
157          [3, 3, 4, 4, 5, 5, 6, 6] 
158   
159          >>> # Define a `worker_init_fn` that configures each dataset copy differently 
160          >>> def worker_init_fn(worker_id): 
161          ...     worker_info = torch.utils.data.get_worker_info() 
162          ...     dataset = worker_info.dataset  # the dataset copy in this worker process 
163          ...     overall_start = dataset.start 
164          ...     overall_end = dataset.end 
165          ...     # configure the dataset to only process the split workload 
166          ...     per_worker = int(math.ceil((overall_end - overall_start) / float(worker_info.num_workers))) 
167          ...     worker_id = worker_info.id 
168          ...     dataset.start = overall_start + worker_id * per_worker 
169          ...     dataset.end = min(dataset.start + per_worker, overall_end) 
170          ... 
171   
172          >>> # Mult-process loading with the custom `worker_init_fn` 
173          >>> # Worker 0 fetched [3, 4].  Worker 1 fetched [5, 6]. 
174          >>> print(list(torch.utils.data.DataLoader(ds, num_workers=2, worker_init_fn=worker_init_fn))) 
175          [3, 5, 4, 6] 
176   
177          >>> # With even more workers 
178          >>> print(list(torch.utils.data.DataLoader(ds, num_workers=12, worker_init_fn=worker_init_fn))) 
179          [3, 4, 5, 6] 
180      """
181  
182      def __add__(self, other: Dataset[_T_co]):
183          return ChainDataset([self, other])

189  class TensorDataset(Dataset[tuple[Tensor, ...]]):
190      r"""Dataset wrapping tensors. 
191   
192      Each sample will be retrieved by indexing tensors along the first dimension. 
193   
194      Args: 
195          *tensors (Tensor): tensors that have the same size of the first dimension. 
196      """
197  
198      tensors: tuple[Tensor, ...]
199  
200      def __init__(self, *tensors: Tensor) -> None:
201          assert all(
202              tensors[0].size(0) == tensor.size(0) for tensor in tensors
203          ), "Size mismatch between tensors"
204          self.tensors = tensors
205  
206      def __getitem__(self, index):
207          return tuple(tensor[index] for tensor in self.tensors)
208  
209      def __len__(self):
210          return self.tensors[0].size(0)

213  class StackDataset(Dataset[_T_stack]):
214      r"""Dataset as a stacking of multiple datasets. 
215   
216      This class is useful to assemble different parts of complex input data, given as datasets. 
217   
218      Example: 
219          >>> # xdoctest: +SKIP 
220          >>> images = ImageDataset() 
221          >>> texts = TextDataset() 
222          >>> tuple_stack = StackDataset(images, texts) 
223          >>> tuple_stack[0] == (images[0], texts[0]) 
224          >>> dict_stack = StackDataset(image=images, text=texts) 
225          >>> dict_stack[0] == {'image': images[0], 'text': texts[0]} 
226   
227      Args: 
228          *args (Dataset): Datasets for stacking returned as tuple. 
229          **kwargs (Dataset): Datasets for stacking returned as dict. 
230      """
231  
232      datasets: Union[tuple, dict]
233  
234      def __init__(self, *args: Dataset[_T_co], **kwargs: Dataset[_T_co]) -> None:
235          if args:
236              if kwargs:
237                  raise ValueError(
238                      "Supported either ``tuple``- (via ``args``) or"
239                      "``dict``- (via ``kwargs``) like input/output, but both types are given."
240                  )
241              self._length = len(args[0])  # type: ignore[arg-type]
242              if any(self._length != len(dataset) for dataset in args):  # type: ignore[arg-type]
243                  raise ValueError("Size mismatch between datasets")
244              self.datasets = args
245          elif kwargs:
246              tmp = list(kwargs.values())
247              self._length = len(tmp[0])  # type: ignore[arg-type]
248              if any(self._length != len(dataset) for dataset in tmp):  # type: ignore[arg-type]
249                  raise ValueError("Size mismatch between datasets")
250              self.datasets = kwargs
251          else:
252              raise ValueError("At least one dataset should be passed")
253  
254      def __getitem__(self, index):
255          if isinstance(self.datasets, dict):
256              return {k: dataset[index] for k, dataset in self.datasets.items()}
257          return tuple(dataset[index] for dataset in self.datasets)
258  
259      def __getitems__(self, indices: list):
260          # add batched sampling support when parent datasets supports it.
261          if isinstance(self.datasets, dict):
262              dict_batch: list[_T_dict] = [{} for _ in indices]
263              for k, dataset in self.datasets.items():
264                  if callable(getattr(dataset, "__getitems__", None)):
265                      items = dataset.__getitems__(indices)  # type: ignore[attr-defined]
266                      if len(items) != len(indices):
267                          raise ValueError(
268                              "Nested dataset's output size mismatch."
269                              f" Expected {len(indices)}, got {len(items)}"
270                          )
271                      for data, d_sample in zip(items, dict_batch):
272                          d_sample[k] = data
273                  else:
274                      for idx, d_sample in zip(indices, dict_batch):
275                          d_sample[k] = dataset[idx]
276              return dict_batch
277  
278          # tuple data
279          list_batch: list[list] = [[] for _ in indices]
280          for dataset in self.datasets:
281              if callable(getattr(dataset, "__getitems__", None)):
282                  items = dataset.__getitems__(indices)  # type: ignore[attr-defined]
283                  if len(items) != len(indices):
284                      raise ValueError(
285                          "Nested dataset's output size mismatch."
286                          f" Expected {len(indices)}, got {len(items)}"
287                      )
288                  for data, t_sample in zip(items, list_batch):
289                      t_sample.append(data)
290              else:
291                  for idx, t_sample in zip(indices, list_batch):
292                      t_sample.append(dataset[idx])
293          tuple_batch: list[_T_tuple] = [tuple(sample) for sample in list_batch]
294          return tuple_batch
295  
296      def __len__(self):
297          return self._length

300  class ConcatDataset(Dataset[_T_co]):
301      r"""Dataset as a concatenation of multiple datasets. 
302   
303      This class is useful to assemble different existing datasets. 
304   
305      Args: 
306          datasets (sequence): List of datasets to be concatenated 
307      """
308  
309      datasets: list[Dataset[_T_co]]
310      cumulative_sizes: list[int]
311  
312      @staticmethod
313      def cumsum(sequence):
314          r, s = [], 0
315          for e in sequence:
316              l = len(e)
317              r.append(l + s)
318              s += l
319          return r
320  
321      def __init__(self, datasets: Iterable[Dataset]) -> None:
322          super().__init__()
323          self.datasets = list(datasets)
324          assert len(self.datasets) > 0, "datasets should not be an empty iterable"  # type: ignore[arg-type]
325          for d in self.datasets:
326              assert not isinstance(
327                  d, IterableDataset
328              ), "ConcatDataset does not support IterableDataset"
329          self.cumulative_sizes = self.cumsum(self.datasets)
330  
331      def __len__(self):
332          return self.cumulative_sizes[-1]
333  
334      def __getitem__(self, idx):
335          if idx < 0:
336              if -idx > len(self):
337                  raise ValueError(
338                      "absolute value of index should not exceed dataset length"
339                  )
340              idx = len(self) + idx
341          dataset_idx = bisect.bisect_right(self.cumulative_sizes, idx)
342          if dataset_idx == 0:
343              sample_idx = idx
344          else:
345              sample_idx = idx - self.cumulative_sizes[dataset_idx - 1]
346          return self.datasets[dataset_idx][sample_idx]
347  
348      @property
349      @deprecated(
350          "`cummulative_sizes` attribute is renamed to `cumulative_sizes`",
351          category=FutureWarning,
352      )
353      def cummulative_sizes(self):
354          return self.cumulative_sizes

357  class ChainDataset(IterableDataset):
358      r"""Dataset for chaining multiple :class:`IterableDataset` s. 
359   
360      This class is useful to assemble different existing dataset streams. The 
361      chaining operation is done on-the-fly, so concatenating large-scale 
362      datasets with this class will be efficient. 
363   
364      Args: 
365          datasets (iterable of IterableDataset): datasets to be chained together 
366      """
367  
368      def __init__(self, datasets: Iterable[Dataset]) -> None:
369          super().__init__()
370          self.datasets = datasets
371  
372      def __iter__(self):
373          for d in self.datasets:
374              assert isinstance(
375                  d, IterableDataset
376              ), "ChainDataset only supports IterableDataset"
377              yield from d
378  
379      def __len__(self):
380          total = 0
381          for d in self.datasets:
382              assert isinstance(
383                  d, IterableDataset
384              ), "ChainDataset only supports IterableDataset"
385              total += len(d)  # type: ignore[arg-type]
386          return total

389  class Subset(Dataset[_T_co]):
390      r""" 
391      Subset of a dataset at specified indices. 
392   
393      Args: 
394          dataset (Dataset): The whole Dataset 
395          indices (sequence): Indices in the whole set selected for subset 
396      """
397  
398      dataset: Dataset[_T_co]
399      indices: Sequence[int]
400  
401      def __init__(self, dataset: Dataset[_T_co], indices: Sequence[int]) -> None:
402          self.dataset = dataset
403          self.indices = indices
404  
405      def __getitem__(self, idx):
406          if isinstance(idx, list):
407              return self.dataset[[self.indices[i] for i in idx]]
408          return self.dataset[self.indices[idx]]
409  
410      def __getitems__(self, indices: list[int]) -> list[_T_co]:
411          # add batched sampling support when parent dataset supports it.
412          # see torch.utils.data._utils.fetch._MapDatasetFetcher
413          if callable(getattr(self.dataset, "__getitems__", None)):
414              return self.dataset.__getitems__([self.indices[idx] for idx in indices])  # type: ignore[attr-defined]
415          else:
416              return [self.dataset[self.indices[idx]] for idx in indices]
417  
418      def __len__(self):
419          return len(self.indices)

422  def random_split(
423      dataset: Dataset[_T],
424      lengths: Sequence[Union[int, float]],
425      generator: Optional[Generator] = default_generator,
426  ) -> list[Subset[_T]]:
427      r""" 
428      Randomly split a dataset into non-overlapping new datasets of given lengths. 
429   
430      If a list of fractions that sum up to 1 is given, 
431      the lengths will be computed automatically as 
432      floor(frac * len(dataset)) for each fraction provided. 
433   
434      After computing the lengths, if there are any remainders, 1 count will be 
435      distributed in round-robin fashion to the lengths 
436      until there are no remainders left. 
437   
438      Optionally fix the generator for reproducible results, e.g.: 
439   
440      Example: 
441          >>> # xdoctest: +SKIP 
442          >>> generator1 = torch.Generator().manual_seed(42) 
443          >>> generator2 = torch.Generator().manual_seed(42) 
444          >>> random_split(range(10), [3, 7], generator=generator1) 
445          >>> random_split(range(30), [0.3, 0.3, 0.4], generator=generator2) 
446   
447      Args: 
448          dataset (Dataset): Dataset to be split 
449          lengths (sequence): lengths or fractions of splits to be produced 
450          generator (Generator): Generator used for the random permutation. 
451      """
452      if math.isclose(sum(lengths), 1) and sum(lengths) <= 1:
453          subset_lengths: list[int] = []
454          for i, frac in enumerate(lengths):
455              if frac < 0 or frac > 1:
456                  raise ValueError(f"Fraction at index {i} is not between 0 and 1")
457              n_items_in_split = int(
458                  math.floor(len(dataset) * frac)  # type: ignore[arg-type]
459              )
460              subset_lengths.append(n_items_in_split)
461          remainder = len(dataset) - sum(subset_lengths)  # type: ignore[arg-type]
462          # add 1 to all the lengths in round-robin fashion until the remainder is 0
463          for i in range(remainder):
464              idx_to_add_at = i % len(subset_lengths)
465              subset_lengths[idx_to_add_at] += 1
466          lengths = subset_lengths
467          for i, length in enumerate(lengths):
468              if length == 0:
469                  warnings.warn(
470                      f"Length of split at index {i} is 0. "
471                      f"This might result in an empty dataset."
472                  )
473  
474      # Cannot verify that dataset is Sized
475      if sum(lengths) != len(dataset):  # type: ignore[arg-type]
476          raise ValueError(
477              "Sum of input lengths does not equal the length of the input dataset!"
478          )
479  
480      indices = randperm(sum(lengths), generator=generator).tolist()  # type: ignore[arg-type, call-overload]
481      lengths = cast(Sequence[int], lengths)
482      return [
483          Subset(dataset, indices[offset - length : offset])
484          for offset, length in zip(itertools.accumulate(lengths), lengths)
485      ]
