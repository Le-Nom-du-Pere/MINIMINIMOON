8    
9    def asunicode(s):
10       if isinstance(s, bytes):
11           return s.decode('latin1')
12       return str(s)
13   
14   
15   def asbytes(s):
16       if isinstance(s, bytes):
17           return s
18       return str(s).encode('latin1')
