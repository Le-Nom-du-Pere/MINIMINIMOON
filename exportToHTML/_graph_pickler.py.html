47   class GraphPickler(pickle.Pickler):
48       """ 
49       GraphPickler is a Pickler which helps pickling fx graph - in particular 
50       GraphModule. 
51       """
52   
53       def __init__(self, file: io.BytesIO, options: Optional[Options] = None) -> None:
54           super().__init__(file)
55           self.options = options or Options()
56   
57           # This abomination is so we can pass external decoding state to the
58           # unpickler functions. We serialize _unpickle_state as a persistent
59           # external item and when we deserialize it we return the common state
60           # object.
61           self._unpickle_state = _UnpickleStateToken(object())
62   
63           # This is used to describe tensors. It needs to be common across the
64           # pickle so that duplicates and views are properly handled.
65           self._meta_tensor_describer = MetaTensorDescriber(copy_data=False)
66   
67       @override
68       def reducer_override(
69           self, obj: object
70       ) -> tuple[Callable[..., Any], tuple[Any, ...]]:
71           # This function is supposed to return either NotImplemented (meaning to
72           # do the default pickle behavior) or a pair of (unpickle callable, data
73           # to pass to unpickle).
74   
75           # We could instead teach individual classes how to pickle themselves but
76           # that has a few problems:
77           #
78           #   1. If we have some special needs (maybe for this use-case we don't
79           #      want to fully serialize every field) then we're adding private
80           #      details to a public interface.
81           #
82           #   2. If we need to have some common shared data (such as a
83           #      FakeTensorMode) which is passed to each value it's harder to
84           #      support.
85   
86           # These are the types that need special handling. See the individual
87           # *PickleData classes for details on pickling that particular type.
88           if isinstance(obj, FakeTensor):
89               return _TensorPickleData.reduce_helper(self, obj)
90           elif isinstance(obj, torch.fx.GraphModule):
91               return _GraphModulePickleData.reduce_helper(self, obj)
92           elif isinstance(obj, (torch._ops.OperatorBase, torch._ops.OpOverloadPacket)):
93               return _OpPickleData.reduce_helper(self, obj)
94           elif isinstance(obj, ShapeEnv):
95               return _ShapeEnvPickleData.reduce_helper(self, obj)
96           elif isinstance(obj, torch.SymInt):
97               return _SymNodePickleData.reduce_helper(self, obj)
98           elif isinstance(obj, torch._guards.TracingContext):
99               return _TracingContextPickleData.reduce_helper(self, obj)
100          else:
101              # We should never get a raw Node!
102              assert not isinstance(obj, torch.fx.Node)
103              if reduce := _TorchNumpyPickleData.reduce_helper(self, obj):
104                  return reduce
105  
106              # returning `NotImplemented` causes pickle to revert to the default
107              # behavior for this object.
108              return NotImplemented
109  
110      @override
111      def persistent_id(self, obj: object) -> Optional[str]:
112          if obj is self._unpickle_state:
113              return "unpickle_state"
114          else:
115              return None
116  
117      @classmethod
118      def dumps(cls, obj: object, options: Optional[Options] = None) -> bytes:
119          """ 
120          Pickle an object. 
121          """
122          with io.BytesIO() as stream:
123              pickler = cls(stream, options)
124              pickler.dump(obj)
125              return stream.getvalue()
126  
127      @staticmethod
128      def loads(data: bytes, fake_mode: FakeTensorMode) -> object:
129          """ 
130          Unpickle an object. 
131          """
132          state = _UnpickleState(fake_mode)
133          with io.BytesIO(data) as stream:
134              unpickler = _GraphUnpickler(stream, state)
135              return unpickler.load()

162  class _ShapeEnvPickleData:
163      data: dict[str, object]
164  
165      @classmethod
166      def reduce_helper(
167          cls, pickler: GraphPickler, obj: ShapeEnv
168      ) -> tuple[
169          Callable[[Self, _UnpickleState], ShapeEnv], tuple[Self, _UnpickleStateToken]
170      ]:
171          return cls.unpickle, (cls(obj), pickler._unpickle_state)
172  
173      def __init__(self, env: ShapeEnv) -> None:
174          # In theory pickle should recognize that a given ShapeEnv was already
175          # pickled and reuse the resulting _ShapeEnvPickleData (so two objects
176          # pointing at the same ShapeEnv get the same ShapeEnv out).
177          assert not env._translation_validation_enabled
178          self.data = env.__dict__.copy()
179          del self.data["tracked_fakes"]
180          del self.data["fake_tensor_cache"]
181  
182      def unpickle(self, unpickle_state: _UnpickleState) -> ShapeEnv:
183          # Fill in the existing ShapeEnv rather than creating a new one
184          assert unpickle_state.fake_mode
185          assert unpickle_state.fake_mode.shape_env
186  
187          for k, v in self.data.items():
188              setattr(unpickle_state.fake_mode.shape_env, k, v)
189  
190          return unpickle_state.fake_mode.shape_env

224  class _TensorPickleData:
225      metadata: MetaTensorDesc[FakeTensor]
226  
227      @classmethod
228      def reduce_helper(
229          cls, pickler: GraphPickler, obj: FakeTensor
230      ) -> tuple[
231          Callable[[Self, _UnpickleState], FakeTensor], tuple[Self, _UnpickleStateToken]
232      ]:
233          return cls.unpickle, (
234              cls(pickler._meta_tensor_describer, obj),
235              pickler._unpickle_state,
236          )
237  
238      def __init__(self, describer: MetaTensorDescriber, t: Tensor) -> None:
239          # THINGS TO WORRY ABOUT:
240          # 1. Need to make sure that two tensors with the same id end up with the
241          #    same id on the other side of the wire.
242  
243          metadata = describer.describe_tensor(t)
244  
245          # view_func is fine if it's either None or a _FakeTensorViewFunc. A
246          # custom one (which is basically a lambda) can't be serialized.
247          assert not metadata.view_func or isinstance(
248              metadata.view_func, torch._subclasses.meta_utils._FakeTensorViewFunc
249          )
250          self.metadata = dataclasses.replace(metadata, fake_mode=None)
251  
252          # Some debugging/verification
253          for k in MetaTensorDesc._UNSERIALIZABLE:
254              if k in ("fake_mode", "view_func"):
255                  continue
256              assert getattr(self.metadata, k) is None, (
257                  f"not None: {k}: {getattr(self.metadata, k)}"
258              )
259  
260      def unpickle(self, unpickle_state: _UnpickleState) -> FakeTensor:
261          # TODO: make common w/ _output_from_cache_entry() in fake_tensor.py?
262          metadata = dataclasses.replace(
263              self.metadata,
264              fake_mode=unpickle_state.fake_mode,
265          )
266  
267          def with_fake(
268              make_meta_t: Callable[[], torch.Tensor], device: Union[torch.device, str]
269          ) -> FakeTensor:
270              with no_dispatch():
271                  return FakeTensor(
272                      unpickle_state.fake_mode,
273                      make_meta_t(),
274                      device,
275                  )
276  
277          return unpickle_state.meta_converter.meta_tensor(
278              metadata,
279              unpickle_state.fake_mode.shape_env,
280              with_fake,
281              None,
282              None,
283          )

286  class _TorchNumpyPickleData:
287      @classmethod
288      def reduce_helper(
289          cls, pickler: GraphPickler, obj: object
290      ) -> Optional[
291          tuple[
292              Callable[[Self, _UnpickleState], object], tuple[Self, _UnpickleStateToken]
293          ]
294      ]:
295          if data := cls.from_object(obj):
296              return (cls.unpickle, (data, pickler._unpickle_state))
297          else:
298              return None
299  
300      def __init__(self, mod: str, name: str) -> None:
301          self.mod = mod
302          self.name = name
303  
304      def unpickle(self, unpickle_state: _UnpickleState) -> Callable[..., object]:
305          np = getattr(importlib.import_module(self.mod), self.name)
306          return torch._dynamo.variables.misc.get_np_to_tnp_map()[np]
307  
308      @classmethod
309      def from_object(cls, tnp: object) -> Optional[Self]:
310          if not callable(tnp):
311              return None
312  
313          tnp_to_np = torch._dynamo.variables.misc.get_tnp_to_np_map()
314          try:
315              if not (np := tnp_to_np.get(tnp)):
316                  return None
317          except TypeError:
318              return None
319  
320          if not (mod := getattr(np, "__module__", None)):
321              mod = "numpy"
322  
323          if not (name := getattr(np, "__name__", None)):
324              return None
325  
326          assert np == getattr(importlib.import_module(mod), name)
327          return cls(mod, name)

331      @classmethod
332      def reduce_helper(
333          cls, pickler: GraphPickler, obj: torch.fx.GraphModule
334      ) -> tuple[
335          Callable[[Self, _UnpickleState], torch.fx.GraphModule],
336          tuple[Self, _UnpickleStateToken],

343      def __init__(self, gm: torch.fx.GraphModule, options: Options) -> None:
344          # Need to do this to ensure the code is created for later pickling.
345          if isinstance(gm, torch.fx._lazy_graph_module._LazyGraphModule):

347          else:
348              _python_code = gm.recompile()
349          self.gm_dict = gm.__dict__.copy()
350          del self.gm_dict["_graph"]
351          self.graph = _GraphPickleData(gm._graph, options)

354          gm = torch.fx.GraphModule.__new__(torch.fx.GraphModule)
355          gm.__dict__ = self.gm_dict
356          gm._graph = self.graph.unpickle(gm, unpickle_state)
357          return gm

361      def __init__(
362          self,
363          node: torch.fx.Node,
364          mapping: dict[torch.fx.Node, "_NodePickleData"],
365          options: Options,
366      ) -> None:
367          self.args = pytree.tree_map_only(torch.fx.Node, lambda n: mapping[n], node.args)
368          self.kwargs = pytree.tree_map_only(
369              torch.fx.Node, lambda n: mapping[n], node.kwargs
370          )
371          # -- self.graph = node.graph
372          self.name = node.name
373          self.op = node.op
374          self.target = _OpPickleData.pickle(node.target, options)
375          # self.input_nodes = node._input_nodes
376          # self.users = node.users
377          self.type = node.type
378          # self.sort_key = node._sort_key
379          # self.repr_fn = node._repr_fn
380          # self.meta = node.meta
381          self.meta = node.meta
382  
383      def unpickle(
384          self,
385          graph: torch.fx.Graph,
386          mapping: dict["_NodePickleData", torch.fx.Node],
387          unpickle_state: _UnpickleState,
388      ) -> torch.fx.Node:
389          args = pytree.tree_map_only(_NodePickleData, lambda n: mapping[n], self.args)
390          kwargs = pytree.tree_map_only(
391              _NodePickleData, lambda n: mapping[n], self.kwargs
392          )
393          target = self.target.unpickle(unpickle_state)
394          assert callable(target) or isinstance(target, str)
395          node = graph.create_node(self.op, target, args, kwargs, self.name, self.type)
396          node.meta = self.meta
397          return node

400  class _OpPickleData:
401      @classmethod
402      def reduce_helper(
403          cls, pickler: GraphPickler, op: object
404      ) -> tuple[Callable[[_UnpickleState], object], tuple[_UnpickleStateToken]]:
405          result = cls.pickle(op, pickler.options)
406          return (result.unpickle, (pickler._unpickle_state,))
407  
408      @classmethod
409      def pickle(cls, op: object, options: Options) -> "_OpPickleData":
410          if isinstance(op, str):
411              return _OpStrPickleData(op)
412  
413          name = torch.fx.Node._pretty_print_target(op)
414          if isinstance(op, torch._ops.OpOverload):
415              return cls._pickle_op(name, _OpOverloadPickleData, options)
416          elif isinstance(op, torch._ops.OpOverloadPacket):
417              return cls._pickle_op(name, _OpOverloadPacketPickleData, options)
418          elif name.startswith(("builtins.", "math.", "torch.")):
419              root, detail = name.split(".", 1)
420              return _OpBuiltinPickleData(root, detail)
421          elif name.startswith("operator."):
422              _, detail = name.split(".", 1)
423              return _OpOperatorPickleData(detail)
424          else:
425              # TODO: raise a BypassFxGraphCache so we will just bypass this one...
426              raise NotImplementedError(f"TARGET: {type(op)} {op} {name}")
427  
428      @staticmethod
429      def _pickle_op(
430          name: str,
431          datacls: Union[
432              type["_OpOverloadPickleData"], type["_OpOverloadPacketPickleData"]
433          ],
434          options: Options,
435      ) -> "_OpPickleData":
436          if (ops_filter := options.ops_filter) and not ops_filter(name):
437              from torch._inductor.codecache import BypassFxGraphCache
438  
439              raise BypassFxGraphCache(f"Unable to pickle non-standard op: {name}")
440          return datacls(name)
441  
442      @abstractmethod
443      def unpickle(self, unpickle_state: _UnpickleState) -> object:
444          pass
445  
446      @classmethod
447      def _lookup_global_by_name(cls, name: str) -> object:
448          """ 
449          Like `globals()[name]` but supports dotted names. 
450          """
451          if "." in name:
452              mod, rest = name.split(".", 1)
453              root = globals()[mod]
454              return cls._getattr_by_name(root, rest)
455          else:
456              return globals()[name]
457  
458      @staticmethod
459      def _getattr_by_name(root: object, name: str) -> object:
460          """ 
461          Like `getattr(root, name)` but supports dotted names. 
462          """
463          while "." in name:
464              mod, name = name.split(".", 1)
465              root = getattr(root, mod)
466          return getattr(root, name)
