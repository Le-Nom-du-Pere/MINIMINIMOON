36   def get_ema_multi_avg_fn(decay=0.999):
37       """Get the function applying exponential moving average (EMA) across multiple params."""
38   
39       if decay < 0.0 or decay > 1.0:
40           raise ValueError(
41               f"Invalid decay value {decay} provided. Please provide a value in [0,1] range."
42           )
43   
44       @torch.no_grad()
45       def ema_update(ema_param_list: PARAM_LIST, current_param_list: PARAM_LIST, _):
46           # foreach lerp only handles float and complex
47           if torch.is_floating_point(ema_param_list[0]) or torch.is_complex(
48               ema_param_list[0]
49           ):
50               torch._foreach_lerp_(ema_param_list, current_param_list, 1 - decay)
51           else:
52               for p_ema, p_model in zip(ema_param_list, current_param_list):
53                   p_ema.copy_(p_ema * decay + p_model * (1 - decay))
54   
55       return ema_update

58   def get_swa_multi_avg_fn():
59       """Get the function applying stochastic weight average (SWA) across multiple params."""
60   
61       @torch.no_grad()
62       def swa_update(
63           averaged_param_list: PARAM_LIST,
64           current_param_list: PARAM_LIST,
65           num_averaged: Union[Tensor, int],
66       ):
67           # foreach lerp only handles float and complex
68           if torch.is_floating_point(averaged_param_list[0]) or torch.is_complex(
69               averaged_param_list[0]
70           ):
71               torch._foreach_lerp_(
72                   averaged_param_list, current_param_list, 1 / (num_averaged + 1)
73               )
74           else:
75               diffs = torch._foreach_sub(current_param_list, averaged_param_list)
76               if isinstance(num_averaged, Tensor):
77                   torch._foreach_addcdiv_(
78                       averaged_param_list,
79                       diffs,
80                       [num_averaged + 1] * len(averaged_param_list),
81                   )
82               else:
83                   torch._foreach_add_(
84                       averaged_param_list, diffs, alpha=1.0 / (num_averaged + 1)
85                   )
86   
87       return swa_update
88

90   def get_ema_avg_fn(decay=0.999):
91       """Get the function applying exponential moving average (EMA) across a single param."""
92   
93       if decay < 0.0 or decay > 1.0:
94           raise ValueError(
95               f"Invalid decay value {decay} provided. Please provide a value in [0,1] range."
96           )
97   
98       @torch.no_grad()
99       def ema_update(ema_param: Tensor, current_param: Tensor, num_averaged):
100          return decay * ema_param + (1 - decay) * current_param
101  
102      return ema_update

105  def get_swa_avg_fn():
106      """Get the function applying stochastic weight average (SWA) across a single param."""
107  
108      @torch.no_grad()
109      def swa_update(
110          averaged_param: Tensor, current_param: Tensor, num_averaged: Union[Tensor, int]
111      ):
112          return averaged_param + (current_param - averaged_param) / (num_averaged + 1)
113  
114      return swa_update

117  class AveragedModel(Module):
118      r"""Implements averaged model for Stochastic Weight Averaging (SWA) and Exponential Moving Average (EMA). 
119   
120      Stochastic Weight Averaging was proposed in `Averaging Weights Leads to 
121      Wider Optima and Better Generalization`_ by Pavel Izmailov, Dmitrii 
122      Podoprikhin, Timur Garipov, Dmitry Vetrov and Andrew Gordon Wilson 
123      (UAI 2018). 
124   
125      Exponential Moving Average is a variation of `Polyak averaging`_, 
126      but using exponential weights instead of equal weights across iterations. 
127   
128      AveragedModel class creates a copy of the provided module :attr:`model` 
129      on the device :attr:`device` and allows to compute running averages of the 
130      parameters of the :attr:`model`. 
131   
132      Args: 
133          model (torch.nn.Module): model to use with SWA/EMA 
134          device (torch.device, optional): if provided, the averaged model will be 
135              stored on the :attr:`device` 
136          avg_fn (function, optional): the averaging function used to update 
137              parameters; the function must take in the current value of the 
138              :class:`AveragedModel` parameter, the current value of :attr:`model` 
139              parameter, and the number of models already averaged; if None, 
140              an equally weighted average is used (default: None) 
141          multi_avg_fn (function, optional): the averaging function used to update 
142              parameters inplace; the function must take in the current values of the 
143              :class:`AveragedModel` parameters as a list, the current values of :attr:`model` 
144              parameters as a list, and the number of models already averaged; if None, 
145              an equally weighted average is used (default: None) 
146          use_buffers (bool): if ``True``, it will compute running averages for 
147              both the parameters and the buffers of the model. (default: ``False``) 
148   
149      Example: 
150          >>> # xdoctest: +SKIP("undefined variables") 
151          >>> loader, optimizer, model, loss_fn = ... 
152          >>> swa_model = torch.optim.swa_utils.AveragedModel(model) 
153          >>> scheduler = torch.optim.lr_scheduler.CosineAnnealingLR(optimizer, 
154          >>>                                     T_max=300) 
155          >>> swa_start = 160 
156          >>> swa_scheduler = SWALR(optimizer, swa_lr=0.05) 
157          >>> for i in range(300): 
158          >>>      for input, target in loader: 
159          >>>          optimizer.zero_grad() 
160          >>>          loss_fn(model(input), target).backward() 
161          >>>          optimizer.step() 
162          >>>      if i > swa_start: 
163          >>>          swa_model.update_parameters(model) 
164          >>>          swa_scheduler.step() 
165          >>>      else: 
166          >>>          scheduler.step() 
167          >>> 
168          >>> # Update bn statistics for the swa_model at the end 
169          >>> torch.optim.swa_utils.update_bn(loader, swa_model) 
170   
171      You can also use custom averaging functions with the `avg_fn` or `multi_avg_fn` parameters. 
172      If no averaging function is provided, the default is to compute 
173      equally-weighted average of the weights (SWA). 
174   
175      Example: 
176          >>> # xdoctest: +SKIP("undefined variables") 
177          >>> # Compute exponential moving averages of the weights and buffers 
178          >>> ema_model = torch.optim.swa_utils.AveragedModel(model, 
179          >>>             torch.optim.swa_utils.get_ema_multi_avg_fn(0.9), use_buffers=True) 
180   
181      .. note:: 
182          When using SWA/EMA with models containing Batch Normalization you may 
183          need to update the activation statistics for Batch Normalization. 
184          This can be done either by using the :meth:`torch.optim.swa_utils.update_bn` 
185          or by setting :attr:`use_buffers` to `True`. The first approach updates the 
186          statistics in a post-training step by passing data through the model. The 
187          second does it during the parameter update phase by averaging all buffers. 
188          Empirical evidence has shown that updating the statistics in normalization 
189          layers increases accuracy, but you may wish to empirically test which 
190          approach yields the best results in your problem. 
191   
192      .. note:: 
193          :attr:`avg_fn` and `multi_avg_fn` are not saved in the :meth:`state_dict` of the model. 
194   
195      .. note:: 
196          When :meth:`update_parameters` is called for the first time (i.e. 
197          :attr:`n_averaged` is `0`) the parameters of `model` are copied 
198          to the parameters of :class:`AveragedModel`. For every subsequent 
199          call of :meth:`update_parameters` the function `avg_fn` is used 
200          to update the parameters. 
201   
202      .. _Averaging Weights Leads to Wider Optima and Better Generalization: 
203          https://arxiv.org/abs/1803.05407 
204      .. _There Are Many Consistent Explanations of Unlabeled Data: Why You Should 
205          Average: 
206          https://arxiv.org/abs/1806.05594 
207      .. _SWALP: Stochastic Weight Averaging in Low-Precision Training: 
208          https://arxiv.org/abs/1904.11943 
209      .. _Stochastic Weight Averaging in Parallel: Large-Batch Training That 
210          Generalizes Well: 
211          https://arxiv.org/abs/2001.02312 
212      .. _Polyak averaging: 
213          https://paperswithcode.com/method/polyak-averaging 
214      """
215  
216      n_averaged: Tensor
217  
218      def __init__(
219          self,
220          model: Module,
221          device: Optional[Union[int, torch.device]] = None,
222          avg_fn: Optional[Callable[[Tensor, Tensor, Union[Tensor, int]], Tensor]] = None,
223          multi_avg_fn: Optional[
224              Callable[[PARAM_LIST, PARAM_LIST, Union[Tensor, int]], None]
225          ] = None,
226          use_buffers=False,
227      ):  # noqa: D107
228          super().__init__()
229          assert avg_fn is None or multi_avg_fn is None, (
230              "Only one of avg_fn and multi_avg_fn should be provided"
231          )
232          self.module = deepcopy(model)
233          if device is not None:
234              self.module = self.module.to(device)
235          self.register_buffer(
236              "n_averaged", torch.tensor(0, dtype=torch.long, device=device)
237          )
238          self.avg_fn = avg_fn
239          self.multi_avg_fn = multi_avg_fn
240          self.use_buffers = use_buffers
241  
242      def forward(self, *args, **kwargs):
243          """Forward pass."""
244          return self.module(*args, **kwargs)
245  
246      def update_parameters(self, model: Module):
247          """Update model parameters."""
248          self_param = (
249              itertools.chain(self.module.parameters(), self.module.buffers())
250              if self.use_buffers
251              else self.parameters()
252          )
253          model_param = (
254              itertools.chain(model.parameters(), model.buffers())
255              if self.use_buffers
256              else model.parameters()
257          )
258          self_param_detached: list[Optional[Tensor]] = []
259          model_param_detached: list[Optional[Tensor]] = []
260          for p_averaged, p_model in zip(self_param, model_param):
261              p_model_ = p_model.detach().to(p_averaged.device)
262              self_param_detached.append(p_averaged.detach())
263              model_param_detached.append(p_model_)
264              if self.n_averaged == 0:
265                  p_averaged.detach().copy_(p_model_)
266  
267          if self.n_averaged > 0:
268              if self.multi_avg_fn is not None or self.avg_fn is None:
269                  grouped_tensors = _group_tensors_by_device_and_dtype(
270                      [self_param_detached, model_param_detached]
271                  )
272                  for (device, _), (
273                      [self_params, model_params],
274                      _,
275                  ) in grouped_tensors.items():
276                      if self.multi_avg_fn:
277                          self.multi_avg_fn(
278                              self_params,  # type: ignore[arg-type]
279                              model_params,  # type: ignore[arg-type]
280                              self.n_averaged.to(device),
281                          )
282                      elif (
283                          device is not None
284                          and device.type in _get_foreach_kernels_supported_devices()
285                      ):
286                          multi_avg_fn = get_swa_multi_avg_fn()
287                          multi_avg_fn(
288                              self_params, model_params, self.n_averaged.to(device)
289                          )
290                      else:
291                          avg_fn = get_swa_avg_fn()
292                          n_averaged = self.n_averaged.to(device)
293                          for p_averaged, p_model in zip(self_params, model_params):  # type: ignore[assignment]
294                              p_averaged.copy_(avg_fn(p_averaged, p_model, n_averaged))
295              else:
296                  for p_averaged, p_model in zip(  # type: ignore[assignment]
297                      self_param_detached, model_param_detached
298                  ):
299                      n_averaged = self.n_averaged.to(p_averaged.device)
300                      p_averaged.detach().copy_(
301                          self.avg_fn(p_averaged.detach(), p_model, n_averaged)
302                      )
303  
304          if not self.use_buffers:
305              # If not apply running averages to the buffers,
306              # keep the buffers in sync with the source model.
307              for b_swa, b_model in zip(self.module.buffers(), model.buffers()):
308                  b_swa.detach().copy_(b_model.detach().to(b_swa.device))
309          self.n_averaged += 1

312  @torch.no_grad()
313  def update_bn(
314      loader: Iterable[Any],
315      model: Module,
316      device: Optional[Union[int, torch.device]] = None,
317  ):
318      r"""Update BatchNorm running_mean, running_var buffers in the model. 
319   
320      It performs one pass over data in `loader` to estimate the activation 
321      statistics for BatchNorm layers in the model. 
322   
323      Args: 
324          loader (torch.utils.data.DataLoader): dataset loader to compute the 
325              activation statistics on. Each data batch should be either a 
326              tensor, or a list/tuple whose first element is a tensor 
327              containing data. 
328          model (torch.nn.Module): model for which we seek to update BatchNorm 
329              statistics. 
330          device (torch.device, optional): If set, data will be transferred to 
331              :attr:`device` before being passed into :attr:`model`. 
332   
333      Example: 
334          >>> # xdoctest: +SKIP("Undefined variables") 
335          >>> loader, model = ... 
336          >>> torch.optim.swa_utils.update_bn(loader, model) 
337   
338      .. note:: 
339          The `update_bn` utility assumes that each data batch in :attr:`loader` 
340          is either a tensor or a list or tuple of tensors; in the latter case it 
341          is assumed that :meth:`model.forward()` should be called on the first 
342          element of the list or tuple corresponding to the data batch. 
343      """
344      momenta = {}
345      for module in model.modules():
346          if isinstance(module, torch.nn.modules.batchnorm._BatchNorm):
347              module.reset_running_stats()
348              momenta[module] = module.momentum
349  
350      if not momenta:
351          return
352  
353      was_training = model.training
354      model.train()
355      for module in momenta.keys():
356          module.momentum = None
357  
358      for input in loader:
359          if isinstance(input, (list, tuple)):
360              input = input[0]
361          if device is not None:
362              input = input.to(device)
363  
364          model(input)
365  
366      for bn_module in momenta.keys():
367          bn_module.momentum = momenta[bn_module]
368      model.train(was_training)

371  class SWALR(LRScheduler):
372      r"""Anneals the learning rate in each parameter group to a fixed value. 
373   
374      This learning rate scheduler is meant to be used with Stochastic Weight 
375      Averaging (SWA) method (see `torch.optim.swa_utils.AveragedModel`). 
376   
377      Args: 
378          optimizer (torch.optim.Optimizer): wrapped optimizer 
379          swa_lrs (float or list): the learning rate value for all param groups 
380              together or separately for each group. 
381          annealing_epochs (int): number of epochs in the annealing phase 
382              (default: 10) 
383          annealing_strategy (str): "cos" or "linear"; specifies the annealing 
384              strategy: "cos" for cosine annealing, "linear" for linear annealing 
385              (default: "cos") 
386          last_epoch (int): the index of the last epoch (default: -1) 
387   
388      The :class:`SWALR` scheduler can be used together with other 
389      schedulers to switch to a constant learning rate late in the training 
390      as in the example below. 
391   
392      Example: 
393          >>> # xdoctest: +SKIP("Undefined variables") 
394          >>> loader, optimizer, model = ... 
395          >>> lr_lambda = lambda epoch: 0.9 
396          >>> scheduler = torch.optim.lr_scheduler.MultiplicativeLR(optimizer, 
397          >>>        lr_lambda=lr_lambda) 
398          >>> swa_scheduler = torch.optim.swa_utils.SWALR(optimizer, 
399          >>>        anneal_strategy="linear", anneal_epochs=20, swa_lr=0.05) 
400          >>> swa_start = 160 
401          >>> for i in range(300): 
402          >>>      for input, target in loader: 
403          >>>          optimizer.zero_grad() 
404          >>>          loss_fn(model(input), target).backward() 
405          >>>          optimizer.step() 
406          >>>      if i > swa_start: 
407          >>>          swa_scheduler.step() 
408          >>>      else: 
409          >>>          scheduler.step() 
410   
411      .. _Averaging Weights Leads to Wider Optima and Better Generalization: 
412          https://arxiv.org/abs/1803.05407 
413      """
414  
415      def __init__(
416          self,
417          optimizer: Optimizer,
418          swa_lr: float,
419          anneal_epochs=10,
420          anneal_strategy: Literal["cos", "linear"] = "cos",
421          last_epoch=-1,
422      ):  # noqa: D107
423          swa_lrs = _format_param("swa_lr", optimizer, swa_lr)
424          for swa_lr, group in zip(swa_lrs, optimizer.param_groups):
425              group["swa_lr"] = swa_lr
426          if anneal_strategy not in ["cos", "linear"]:
427              raise ValueError(
428                  "anneal_strategy must by one of 'cos' or 'linear', "
429                  f"instead got {anneal_strategy}"
430              )
431          elif anneal_strategy == "cos":
432              self.anneal_func = self._cosine_anneal
433          elif anneal_strategy == "linear":
434              self.anneal_func = self._linear_anneal
435          if not isinstance(anneal_epochs, int) or anneal_epochs < 0:
436              raise ValueError(
437                  f"anneal_epochs must be equal or greater than 0, got {anneal_epochs}"
438              )
439          self.anneal_epochs = anneal_epochs
440          super().__init__(optimizer, last_epoch)
441  
442      @staticmethod
443      def _linear_anneal(t):
444          return t
445  
446      @staticmethod
447      def _cosine_anneal(t):
448          return (1 - math.cos(math.pi * t)) / 2
449  
450      @staticmethod
451      def _get_initial_lr(lr, swa_lr, alpha):
452          if alpha == 1:
453              return swa_lr
454          return (lr - alpha * swa_lr) / (1 - alpha)
455  
456      def get_lr(self):
457          """Get learning rate."""
458          # `_get_lr_called_within_step` is only available `_enable_get_lr_call`,
459          # so we ignore the type error here. See `LRScheduler.step()` for more details.
460          if not self._get_lr_called_within_step:
461              warnings.warn(
462                  "To get the last learning rate computed by the scheduler, "
463                  "please use `get_last_lr()`.",
464                  UserWarning,
465              )
466          # Set in `LRScheduler._initial_step()`
467          step = self._step_count - 1
468          if self.anneal_epochs == 0:
469              step = max(1, step)
470          prev_t = max(0, min(1, (step - 1) / max(1, self.anneal_epochs)))
471          prev_alpha = self.anneal_func(prev_t)
472          prev_lrs = [
473              self._get_initial_lr(group["lr"], group["swa_lr"], prev_alpha)
474              for group in self.optimizer.param_groups
475          ]
476          t = max(0, min(1, step / max(1, self.anneal_epochs)))
477          alpha = self.anneal_func(t)
478          return [
479              group["swa_lr"] * alpha + lr * (1 - alpha)
480              for group, lr in zip(self.optimizer.param_groups, prev_lrs)
481          ]
