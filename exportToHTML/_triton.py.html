118  @functools.cache
119  def has_triton() -> bool:
120      if not has_triton_package():
121          return False
122  
123      from torch._dynamo.device_interface import get_interface_for_device
124  
125      def cuda_extra_check(device_interface: Any) -> bool:
126          return device_interface.Worker.get_device_properties().major >= 7
127  
128      def cpu_extra_check(device_interface: Any) -> bool:
129          import triton.backends
130  
131          return "cpu" in triton.backends.backends
132  
133      def _return_true(device_interface: Any) -> bool:
134          return True
135  
136      triton_supported_devices = {
137          "cuda": cuda_extra_check,
138          "xpu": _return_true,
139          "cpu": cpu_extra_check,
140      }
141  
142      def is_device_compatible_with_triton() -> bool:
143          for device, extra_check in triton_supported_devices.items():
144              device_interface = get_interface_for_device(device)
145              if device_interface.is_available() and extra_check(device_interface):
146                  return True
147          return False
148  
149      return is_device_compatible_with_triton()

164
