48       def _handle_aggregate(ty) -> tuple[dict[str, Any], dict[str, Any], dict[str, Any]]:
49           def dump_type(t, level: int) -> tuple[str, str, str]:
50               if getattr(t, "__name__", None) in cpp_enum_defs:
51                   return t.__name__, "int64_t", t.__name__
52               elif t in _CPP_TYPE_MAP:
53                   return (t.__name__, _CPP_TYPE_MAP[t], _THRIFT_TYPE_MAP[t])
54               elif isinstance(t, str):
55                   assert t in defs
56                   assert t not in cpp_enum_defs
57                   assert "[" not in t
58                   return t, f"ForwardRef<{t}>", t
59               elif isinstance(t, ForwardRef):
60                   return (
61                       t.__forward_arg__,
62                       f"ForwardRef<{t.__forward_arg__}>",
63                       t.__forward_arg__,
64                   )
65               elif o := typing.get_origin(t):
66                   # Lemme know if there's a better way to do this.
67                   if o == list:
68                       yaml_head, cpp_head, thrift_head, thrift_tail = (
69                           "List",
70                           "std::vector",
71                           "list<",
72                           ">",
73                       )
74                   elif o == dict:
75                       yaml_head, cpp_head, thrift_head, thrift_tail = (
76                           "Dict",
77                           "std::unordered_map",
78                           "map<",
79                           ">",
80                       )
81                   elif o == Union:
82                       assert level == 0, "Optional is only supported at the top level."
83                       args = typing.get_args(t)
84                       assert len(args) == 2 and args[1] == type(None)
85                       yaml_type, cpp_type, thrift_type = dump_type(args[0], level + 1)
86                       return (
87                           f"Optional[{yaml_type}]",
88                           f"std::optional<{cpp_type}>",
89                           f"optional {thrift_type}",
90                       )
91                   elif o == Annotated:
92                       return dump_type(t.__origin__, level)
93                   else:
94                       raise AssertionError(f"Type {t} is not supported in export schema.")
95                   yaml_arg_types, cpp_arg_types, thrift_arg_types = zip(
96                       *[dump_type(x, level + 1) for x in typing.get_args(t)]
97                   )
98                   return (
99                       (f"{yaml_head}[{', '.join(yaml_arg_types)}]"),
100                      (f"{cpp_head}<{', '.join(cpp_arg_types)}>"),
101                      f"{thrift_head}{', '.join(thrift_arg_types)}{thrift_tail}",
102                  )
103              elif isinstance(t, type):
104                  return (t.__name__, t.__name__, t.__name__)
105              else:
106                  raise AssertionError(f"Type {t} is not supported in export schema.")

108          def dump_cpp_value(v) -> str:
109              if v is None:
110                  return "std::nullopt"
111              elif v is True:
112                  return "true"
113              elif v is False:
114                  return "false"
115              elif v == {}:
116                  return "{}"
117              elif v == []:
118                  return "{}"
119              elif v == ():
120                  return "{}"
121              elif isinstance(v, str):
122                  return f'"{v}"'
123              else:
124                  raise AssertionError(
125                      f"Default value {v} is not supported yet in export schema."
126                  )

128          def dump_field(f) -> tuple[dict[str, Any], str, Optional[str], str, int]:
129              t, cpp_type, thrift_type = dump_type(f.type, 0)
130              ret = {"type": t}
131              cpp_default: Optional[str] = None
132              assert (
133                  typing.get_origin(f.type) == Annotated
134              ), f"Field {f.name} must be annotated with an integer id."
135              thrift_id = f.type.__metadata__[0]
136              assert (
137                  type(thrift_id) is int
138              ), f"Field {f.name} must be annotated with an integer id."
139  
140              value = dataclasses.MISSING
141              if f.default is not dataclasses.MISSING:
142                  value = f.default
143              elif f.default_factory is not dataclasses.MISSING:
144                  value = f.default_factory()
145  
146              if value is not dataclasses.MISSING:
147                  default = str(value)
148                  ret["default"] = default
149                  cpp_default = dump_cpp_value(value)
150  
151                  if t.startswith("Optional[") and value is not None:
152                      raise AssertionError(
153                          f"Optional field {ty.__name__}.{f.name} must have default value to be None."
154                      )
155  
156              return ret, cpp_type, cpp_default, thrift_type, thrift_id

539  def _diff_schema(dst, src):
540      additions = {key: src[key] for key in src.keys() - dst.keys()}
541      subtractions = {key: dst[key] for key in dst.keys() - src.keys()}
542  
543      common_keys = src.keys() & dst.keys()
544  
545      versions = {"SCHEMA_VERSION", "TREESPEC_VERSION"}
546      common_keys -= versions
547  
548      for key in common_keys:
549          src_kind = src[key]["kind"]
550          src_fields = src[key]["fields"]
551          dst_kind = dst[key]["kind"]
552          dst_fields = dst[key]["fields"]
553          _check(
554              src_kind == dst_kind,
555              f"Type {key} changed kind from {dst_kind} to {src_kind}",
556          )
557          assert isinstance(src_fields, dict) and isinstance(dst_fields, dict)
558          added_fields = {
559              key: src_fields[key] for key in src_fields.keys() - dst_fields.keys()
560          }
561          subtracted_fields = {
562              key: dst_fields[key] for key in dst_fields.keys() - src_fields.keys()
563          }
564          common_fields = src_fields.keys() & dst_fields.keys()
565  
566          for field in common_fields:
567              src_field = src_fields[field]
568              dst_field = dst_fields[field]
569              if src_kind == "struct":
570                  _check(
571                      src_field["type"] == dst_field["type"],
572                      f"Type of the field {key}.{field} changed from {dst_field['type']} to {src_field['type']}",
573                  )
574                  if "default" in src_field and "default" not in dst_field:
575                      added_fields[field] = {}
576                      added_fields[field]["default"] = src_field["default"]
577                  if "default" not in src_field and "default" in dst_field:
578                      subtracted_fields[field] = {}
579                      subtracted_fields[field]["default"] = dst_field["default"]
580              elif src_kind == "enum":
581                  _check(
582                      src_field == dst_field,
583                      f"Value of the enum field {key}.{field} changed from {dst_field} to {src_field}",
584                  )
585              elif src_kind == "union":
586                  _check(
587                      src_field["type"] == dst_field["type"],
588                      f"Type of the field {key}.{field} changed from {dst_field['type']} to {src_field['type']}",
589                  )
590              else:
591                  raise AssertionError(f"Unknown kind {src_kind}: {key}")
592          if len(added_fields) > 0:
593              assert key not in additions
594              additions[key] = {}
595              additions[key]["fields"] = added_fields
596          if len(subtracted_fields) > 0:
597              assert key not in subtractions
598              subtractions[key] = {}
599              subtractions[key]["fields"] = subtracted_fields
600  
601      return additions, subtractions

636          thrift_content = importlib.resources.read_text(
637              __package__, "export_schema.thrift"
638          )
639          match = re.search("checksum<<([A-Fa-f0-9]{64})>>", thrift_content)
640          _check(match is not None, "checksum not found in export_schema.thrift")
641          assert match is not None
642          thrift_checksum_head = match.group(1)
643          thrift_content = thrift_content.splitlines()
644          assert thrift_content[0].startswith("// @" + "generated")
645          assert thrift_content[1].startswith("// checksum<<")
646          thrift_checksum_real = _hash_content("\n".join(thrift_content[2:]))
647  
648          from yaml import load, Loader
649  
650          dst = load(content, Loader=Loader)
651          assert isinstance(dst, dict)
652      else:
653          checksum_head = None
654          thrift_checksum_head = None
655          thrift_checksum_real = None
656          dst = {"SCHEMA_VERSION": None, "TREESPEC_VERSION": None}
657  
658      src, cpp_header, thrift_schema = _staged_schema()
659      additions, subtractions = _diff_schema(dst, src)
660      yaml_path = __package__.replace(".", "/") + "/schema.yaml"
661      thrift_schema_path = __package__.replace(".", "/") + "/export_schema.thrift"
662      torch_prefix = "torch/"
663      assert yaml_path.startswith(torch_prefix)  # sanity check
664      assert thrift_schema_path.startswith(torch_prefix)  # sanity check
665  
666      return _Commit(
667          result=src,
668          checksum_next=_hash_content(repr(src)),
669          yaml_path=yaml_path,
670          additions=additions,
671          subtractions=subtractions,
672          base=dst,
673          checksum_head=checksum_head,
674          cpp_header=cpp_header,
675          cpp_header_path=torch_prefix + "csrc/utils/generated_serialization_types.h",
676          thrift_checksum_head=thrift_checksum_head,
677          thrift_checksum_real=thrift_checksum_real,
678          thrift_checksum_next=_hash_content(thrift_schema),
679          thrift_schema=thrift_schema,
680          thrift_schema_path=thrift_schema_path,
681      )

684  def check(commit: _Commit, force_unsafe: bool = False):
685      next_version = None
686      reason = ""
687      # Step 1: Detect major schema updates.
688      if len(commit.additions) > 0:
689          for k, v in commit.additions.items():
690              if k not in commit.base:
691                  continue
692              kind = commit.result[k]["kind"]
693              fields = v["fields"]
694              for f, d in fields.items():
695                  if kind == "struct" and "default" not in d:
696                      reason += (
697                          f"Field {k}.{f} is added to schema.py without a default value as an incomparible change "
698                          + "which requires major version bump.\n"
699                      )
700                      next_version = [commit.base["SCHEMA_VERSION"][0] + 1, 1]
701  
702      if len(commit.subtractions) > 0:
703          for k, v in commit.subtractions.items():
704              if k not in commit.result:
705                  continue
706              for f in v["fields"]:
707                  reason = f"Field {k}.{f} is removed from schema.py as an incompatible change which requires major version bump.\n"
708              next_version = [commit.base["SCHEMA_VERSION"][0] + 1, 1]
709  
710      if force_unsafe:
711          reason += "--force-unsafe is used."
712          next_version = commit.result["SCHEMA_VERSION"]
713      else:
714          # Step 2: Detect minor schema updates.
715          if next_version is None and len(commit.additions) > 0:
716              for k, v in commit.additions.items():
717                  for f in v["fields"]:
718                      reason += (
719                          f"Field {k}.{f} is added to schema.py as an compatible change "
720                          + "which still requires minor version bump.\n"
721                      )
722              next_version = [
723                  commit.base["SCHEMA_VERSION"][0],
724                  commit.base["SCHEMA_VERSION"][1] + 1,
725              ]
726          if next_version is None and len(commit.subtractions) > 0:
727              for k, v in commit.subtractions.items():
728                  for f in v["fields"]:
729                      reason += (
730                          f"Field {k}.{f} is removed from schema.py as an compatible change "
731                          + "which still requires minor version bump.\n"
732                      )
733              next_version = [
734                  commit.base["SCHEMA_VERSION"][0],
735                  commit.base["SCHEMA_VERSION"][1] + 1,
736              ]
737  
738      return next_version, reason
