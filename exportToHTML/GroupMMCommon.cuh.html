1    #pragma once
2    #include <cutlass/util/packed_stride.hpp>
3    
4    namespace at::cuda::detail {
5    
6    using Strides = std::array<int64_t, 3>;
7    
8    template <
9        typename DtypeA,
10       typename DtypeB,
11       typename DtypeOutput,
12       typename DtypeScale,
13       typename ProblemShape,
14       typename StrideA,
15       typename StrideB,
16       typename StrideOutput>
17   __global__ void prepare_grouped_gemm_data(
18       DtypeA* A,
19       DtypeB* B,
20       DtypeOutput* output,
21       DtypeScale* scale_A,
22       DtypeScale* scale_B,
23       DtypeA** A_ptrs,
24       DtypeB** B_ptrs,
25       DtypeOutput** output_ptrs,
26       DtypeScale** inputA_scale_ptrs,
27       DtypeScale** inputB_scale_ptrs,
28       ProblemShape* problem_sizes,
29       // Strides for cutlass, cute::Stride 
30       StrideA* stride_A,
31       StrideB* stride_B,
32       StrideOutput* stride_output,
33       const int32_t* offs,
34       int32_t M,
35       int32_t N,
36       int32_t K,
37       // Original strides of the input tensors 
38       Strides tensor_StrideA,
39       Strides tensor_StrideB,
40       Strides tensor_StrideOutput,
41       int64_t a_scale_stride,
42       int64_t b_scale_stride,
43       bool a_row_major = true,
44       bool b_row_major = false) {
45     int32_t tid = threadIdx.x;
46     int32_t delta = 0;
47     if (offs != nullptr) {
48       int32_t start = tid == 0 ? 0 : offs[tid - 1];
49       delta = offs[tid] - start;
50       if (K < 0) {
51         if (!a_row_major && b_row_major) {
52           CUDA_KERNEL_ASSERT(delta >=0 && "expected ofsets to be greater or equal 0\n");
53         } else  {
54           // CUTLASS cannot handle delta=0 here. 
55           CUDA_KERNEL_ASSERT(delta >0 && "expected ofsets to be greater than 0\n");
56         }
57       }
58   
59       // TMA transfers require global memory tensor addresses to be 
60       // aligned to 16 bytes. 
61       if (tid < blockDim.x - 1) {
62         // Check this requirement for input tensors, in case group 
63         // addresses are increased along the dynamic dimension. 
64         if ((K < 0 && a_row_major) ||       // 2D/2D: check along K dimension 
65             (M < 0 && !a_row_major)) {      // 3D/2D: check along N dimension 
66           int align = 128 / cutlass::sizeof_bits<DtypeA>::value;
67           CUDA_KERNEL_ASSERT(
68                              delta % align == 0 &&
69                              "expected input tensor dynamic dimension byte size to be non-negative multiple of 16\n");
70         }
71         if ((K < 0 && !b_row_major) ||      // 2D/2D: check along K dimension 
72             (N < 0 && b_row_major)) {       // 3D/2D: check along N dimension 
73           int align = 128 / cutlass::sizeof_bits<DtypeB>::value;
74           CUDA_KERNEL_ASSERT(
75                              delta % align == 0 &&
76                              "expected input tensor dynamic dimension byte size to be non-negative multiple of 16\n");
77         }
78   
79         // Check the same requirement for output tensor (that is always 
80         // contiguous, and in row-major layout). 
81         if (N < 0) {
82           int align = 128 / cutlass::sizeof_bits<DtypeOutput>::value;
83           CUDA_KERNEL_ASSERT(
84                              delta % align == 0 &&
85                              "expected output tensor dynamic dimension byte size to be non-negative multiple of 16\n");
86         }
87       }
88     }
89     int64_t lda, ldb, ldoutput;
90     if (M < 0) {
91       // A and output is 2d 
92       M = delta;
93       lda = a_row_major ? tensor_StrideA[0] : tensor_StrideA[1];
94       ldb = b_row_major ? tensor_StrideB[1] : tensor_StrideB[2];
95       ldoutput = tensor_StrideOutput[0];
96       A_ptrs[tid] = tid == 0 ? A : A + offs[tid - 1] * tensor_StrideA[0];
97       if (scale_A != nullptr) {
98         inputA_scale_ptrs[tid] = tid == 0 ? scale_A : scale_A + offs[tid - 1];
99         inputB_scale_ptrs[tid] = scale_B + tid * b_scale_stride;
100      }
101      output_ptrs[tid] = tid == 0 ? output : output + offs[tid - 1] * ldoutput;
102      B_ptrs[tid] = B + tid * tensor_StrideB[0];
103    } else if (N < 0) {
104      N = delta;
105      lda = a_row_major ? tensor_StrideA[1] : tensor_StrideA[2];
106      ldb = b_row_major ? tensor_StrideB[0] : tensor_StrideB[1]; // B is transposed 
107      ldoutput = tensor_StrideOutput[0];
108      A_ptrs[tid] = A + tid * tensor_StrideA[0];
109      output_ptrs[tid] = tid == 0 ? output : output + offs[tid - 1];
110      B_ptrs[tid] = tid == 0 ? B : B + offs[tid - 1] * tensor_StrideB[1];
111      if (scale_A != nullptr) {
112        inputA_scale_ptrs[tid] = scale_A + tid * a_scale_stride;
113        inputB_scale_ptrs[tid] = tid == 0 ? scale_B : scale_B + offs[tid - 1];
114      }
115    } else if (K < 0) {
116      // A, B is 2d, output is 3d 
117      K = delta;
118      lda = a_row_major ? tensor_StrideA[0] : tensor_StrideA[1];
119      ldb = b_row_major ? tensor_StrideB[0] : tensor_StrideB[1];
120      ldoutput = tensor_StrideOutput[1];
121      A_ptrs[tid] = tid == 0 ? A : A + offs[tid - 1] * tensor_StrideA[1];
122      B_ptrs[tid] = tid == 0 ? B : B + offs[tid - 1] * tensor_StrideB[0];
123      output_ptrs[tid] = output + tid * tensor_StrideOutput[0];
124      if (scale_A != nullptr) {
125        inputA_scale_ptrs[tid] = scale_A + tid * M;
126        inputB_scale_ptrs[tid] = scale_B + tid * N;
127      }
128    } else {
129      // A, B, output are 3D 
130      lda = a_row_major ? tensor_StrideA[1] : tensor_StrideA[2];
131      ldb = b_row_major ? tensor_StrideB[1] : tensor_StrideB[2];
132      ldoutput = tensor_StrideOutput[1];
133      A_ptrs[tid] = A + tid * tensor_StrideA[0];
134      B_ptrs[tid] = B + tid * tensor_StrideB[0];
135      output_ptrs[tid] = output + tid * tensor_StrideOutput[0];
136      if (scale_A != nullptr) {
137        inputA_scale_ptrs[tid] = scale_A + tid * a_scale_stride;
138        inputB_scale_ptrs[tid] = scale_B + tid * b_scale_stride;
139      }
140    }
141    problem_sizes[tid] = ProblemShape(M, N, K);
142  
143    // make_cute_packed_stride only replaces one of the stride elements with 
144    // one the provided values in the shape arguments 
145    // the indices of the src/dst depend on whether A/B are row-major 
146    // so constructing shape argument with two similar lda values 
147    // while it looks non-sensical (and it is a nonsensical shape) 
148    // is fine for these stride construction purposes - the one that will be used 
149    // for replacement is correct, the other one is ignored, and we don't have to 
150    // branch on whether A/B are row-major 
151    stride_A[tid] = cutlass::make_cute_packed_stride(StrideA{}, {lda, lda, 1});
152    stride_B[tid] = cutlass::make_cute_packed_stride(StrideB{}, {ldb, ldb, 1});
153    stride_output[tid] =
154        cutlass::make_cute_packed_stride(StrideOutput{}, {M, ldoutput, 1});
155  }
156  } // namespace at::cuda::detail
