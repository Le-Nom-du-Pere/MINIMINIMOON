83       def __init__(self, settings=None):
84           super().__init__(settings)

86           version = self._settings['tensorflow_version']

91       def _print_Function(self, expr):
92           op = self.mapping.get(type(expr), None)
93           if op is None:
94               return super()._print_Basic(expr)
95           children = [self._print(arg) for arg in expr.args]
96           if len(children) == 1:
97               return "%s(%s)" % (
98                   self._module_format(op),
99                   children[0]
100              )
101          else:
102              return self._expand_fold_binary_op(op, children)

116      def _print_Inverse(self, expr):
117          op = self._module_format('tensorflow.linalg.inv')
118          return "{}({})".format(op, self._print(expr.arg))

120      def _print_Transpose(self, expr):
121          version = self.tensorflow_version
122          if version and version_tuple(version) < version_tuple('1.14'):
123              op = self._module_format('tensorflow.matrix_transpose')
124          else:
125              op = self._module_format('tensorflow.linalg.matrix_transpose')
126          return "{}({})".format(op, self._print(expr.arg))

128      def _print_Derivative(self, expr):
129          variables = expr.variables
130          if any(isinstance(i, Iterable) for i in variables):
131              raise NotImplementedError("derivation by multiple variables is not supported")
132          def unfold(expr, args):
133              if not args:
134                  return self._print(expr)
135              return "%s(%s, %s)[0]" % (
136                      self._module_format("tensorflow.gradients"),
137                      unfold(expr, args[:-1]),
138                      self._print(args[-1]),
139                  )
140          return unfold(expr.expr, variables)
141  
142      def _print_Piecewise(self, expr):
143          version = self.tensorflow_version
144          if version and version_tuple(version) < version_tuple('1.0'):
145              tensorflow_piecewise = "tensorflow.select"
146          else:
147              tensorflow_piecewise = "tensorflow.where"
148  
149          from sympy.functions.elementary.piecewise import Piecewise
150          e, cond = expr.args[0].args
151          if len(expr.args) == 1:
152              return '{}({}, {}, {})'.format(
153                  self._module_format(tensorflow_piecewise),
154                  self._print(cond),
155                  self._print(e),
156                  0)
157  
158          return '{}({}, {}, {})'.format(
159              self._module_format(tensorflow_piecewise),
160              self._print(cond),
161              self._print(e),
162              self._print(Piecewise(*expr.args[1:])))
163  
164      def _print_Pow(self, expr):
165          # XXX May raise error for
166          # int**float or int**complex or float**complex
167          base, exp = expr.args
168          if expr.exp == S.Half:
169              return "{}({})".format(
170                  self._module_format("tensorflow.math.sqrt"), self._print(base))
171          return "{}({}, {})".format(
172              self._module_format("tensorflow.math.pow"),
173              self._print(base), self._print(exp))
174  
175      def _print_MatrixBase(self, expr):
176          tensorflow_f = "tensorflow.Variable" if expr.free_symbols else "tensorflow.constant"
177          data = "["+", ".join(["["+", ".join([self._print(j) for j in i])+"]" for i in expr.tolist()])+"]"
178          return "%s(%s)" % (
179              self._module_format(tensorflow_f),
180              data,
181          )

183      def _print_MatMul(self, expr):
184          from sympy.matrices.expressions import MatrixExpr
185          mat_args = [arg for arg in expr.args if isinstance(arg, MatrixExpr)]
186          args = [arg for arg in expr.args if arg not in mat_args]
187          if args:
188              return "%s*%s" % (
189                  self.parenthesize(Mul.fromiter(args), PRECEDENCE["Mul"]),
190                  self._expand_fold_binary_op(
191                      "tensorflow.linalg.matmul", mat_args)
192              )
193          else:
194              return self._expand_fold_binary_op(
195                  "tensorflow.linalg.matmul", mat_args)
196  
197      def _print_MatPow(self, expr):
198          return self._expand_fold_binary_op(
199              "tensorflow.linalg.matmul", [expr.base]*expr.exp)

201      def _print_CodeBlock(self, expr):
202          # TODO: is this necessary?

204          for subexpr in expr.args:
