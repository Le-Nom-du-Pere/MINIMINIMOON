17   // Different combinations of row, col, and offset can lead to two cases:
18   //
19   // Case 1 - Trapezoid (Triangle as a special case): row + offset <= col
20   //    Example A: offset > 0
21   //      1 1 0 0 0
22   //      1 1 1 0 0
23   //      1 1 1 1 0
24   //    Example B: offset <= 0
25   //      0 0 0
26   //      1 0 0
27   //      1 1 0
28   //    In this case, we calculate the number of elements in the first row and
29   //    last row of the tril respectively, and then compute the tril size.
30   //
31   // Case 2 - Trapezoid + Rectangle: row + offset > col
32   //    Example:
33   //      1 1 0
34   //      1 1 1
35   //      1 1 1
36   //    In this case, we first calculate the size of top trapezoid, and then
37   //    calculate the size of the bottom rectangle.
38   inline int64_t get_tril_size(int64_t row, int64_t col, int64_t offset) {
39     // If either dimension is 0 then the there is no tril
40     if (row == 0 || col == 0) {
41       return 0;
42     }
43     // number of elements in the first row of the tril
44     auto m_first_row = offset > 0 ? std::min<int64_t>(col, 1 + offset)
45                                   : // upper bounded by col
46         row + offset > 0; // either 0 or 1
47     // number of elements in the last row of the tril, bounded by [0, col]
48     auto m_last_row = std::max<int64_t>(0, std::min<int64_t>(col, row + offset));
49     // number of rows, bounded by [0, row]
50     auto n_row_all = std::max<int64_t>(0, std::min<int64_t>(row, row + offset));
51     auto n_row_trapezoid = (m_last_row - m_first_row + 1);
52   
53     // calculate # of elements in the top trapezoid
54     auto tril_size = (m_first_row + m_last_row) * n_row_trapezoid >> 1;
55   
56     // calculate # of elements in the bottom rectangle if there is any
57     auto diff_row = n_row_all - n_row_trapezoid;
58     if (diff_row > 0) {
59       tril_size += diff_row * col;
60     }
61   
62     return tril_size;
63   }

65   inline void check_args(
66       int64_t row,
67       int64_t col,
68       std::optional<Layout> layout_opt) {
69     TORCH_CHECK(row >= 0, "row must be non-negative, got", row);
70     TORCH_CHECK(col >= 0, "col must be non-negative, got", col);
71     if (layout_opt.has_value()) {
72       TORCH_CHECK(
73           *layout_opt == at::kStrided,
74           "only support layout=torch.strided, got",
75           *layout_opt)
76     }
77   }

81   // assumes maximum value in created tensor is n-1 (e.g., torch.randperm(n))
82   inline void check_supported_max_int_with_precision(
83       int64_t n,
84       const Tensor& tensor) {
85     // match defined() to behavior of checks below
86     TORCH_CHECK(
87         at::scalar_tensor(n > 0 ? n - 1 : n, tensor.options()).defined(),
88         "n is too large for result tensor type: '",
89         tensor.toString(),
90         "'");
91   
92     // Ensure sufficient precision for floating point representation.
93     switch (tensor.scalar_type()) {
94       case at::ScalarType::Half:
95         TORCH_CHECK(
96             n <= (int64_t(1) << 11) + 1,
97             "n cannot be greater than 2049 for Half type.");
98         break;
99       case at::ScalarType::Float:
100        TORCH_CHECK(
101            n <= (int64_t(1) << 24) + 1,
102            "n cannot be greater than 2^24+1 for Float type.");
103        break;
104      case at::ScalarType::Double: // Unlikely to happen, but doesn't hurt to
105                                   // check
106        TORCH_CHECK(
107            n <= (int64_t(1) << 53) + 1,
108            "n cannot be greater than 2^53+1 for Double type.");
109        break;
110      default:
111        break;
112    }
113  }

115  // Called by `empty*` functions when deterministic algorithms are enabled to
116  // fill the tensor with NaN if it is floating point or complex type, or fill
117  // with max value if it is integer type
118  inline Tensor& fill_empty_deterministic_(Tensor& tensor) {
119    if (tensor.is_floating_point() || tensor.is_complex()) {
120      AT_DISPATCH_V2(
121          tensor.scalar_type(),
122          "fill_empty_deterministic_",
123          AT_WRAP([&]() {
124            tensor.fill_(std::numeric_limits<scalar_t>::quiet_NaN());
125          }),
126          AT_EXPAND(AT_FLOATING_TYPES),
127          AT_EXPAND(AT_COMPLEX_TYPES),
128          AT_EXPAND(AT_FLOAT8_TYPES),
129          kBFloat16,
130          kHalf,
131          kComplexHalf);
132    } else {
133      AT_DISPATCH_V2(
134          tensor.scalar_type(),
135          "fill_empty_deterministic_",
136          AT_WRAP([&]() { tensor.fill_(std::numeric_limits<scalar_t>::max()); }),
137          kBool,
138          AT_EXPAND(AT_INTEGRAL_TYPES_V2));
139    }
140    return tensor;
141  }

143  // The ZeroTensor allocator ignores whatever allocation is requested and always
144  // gives you nullptr
145  struct ZeroTensorAllocator final : public at::Allocator {
146    ZeroTensorAllocator(at::Device device) : device_(device) {}
147    ~ZeroTensorAllocator() override = default;
148    static void deleter(void* const pointer) {
149      TORCH_INTERNAL_ASSERT(!pointer);
150    }
151    DataPtr allocate(const size_t /*nbytes*/) override {
152      return {nullptr, nullptr, &deleter, device_};
153    }
154    DeleterFnPtr raw_deleter() const override {
155      return deleter;
156    }
157    void copy_data(
158        void* dest [[maybe_unused]],
159        const void* src [[maybe_unused]],
160        std::size_t count [[maybe_unused]]) const final {}
161    at::Device device_;
162  };
