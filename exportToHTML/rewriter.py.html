17   class AST_Rewriter(ast.NodeTransformer):
18       """ 
19       Take a FunctionType object representing a `forward` method, then 
20       perform an AST rewrite to swap out nodes that are not symbolically 
21       traceable with a callsite to the FX alternative. 
22    
23       To support swapping out an AST node, define a new `visit` method on 
24       that node. For more details, see: 
25       https://docs.python.org/3/library/ast.html#ast.NodeTransformer 
26       """
27   
28       # This function checks for new keys added in the globals dict. TorchDynamo
29       # can insert new keys in the global dict and upset the check. Therefore, put
30       # a disable here. This function is an optimization pass and not really
31       # suitable for dynamo tracing anyways.
32       @torch._dynamo.disable
33       def rewrite(self, fn: FunctionType):
34           # Normalize the source lines
35           sourcelines, _ = inspect.getsourcelines(fn)
36           sourcelines = normalize_source_lines(sourcelines)
37           source = "".join(sourcelines)
38           normalized_str = textwrap.dedent(source)
39   
40           # Rewrite the original AST
41           source_ast = ast.parse(normalized_str)
42           dest_ast = ast.fix_missing_locations(self.visit(source_ast))
43   
44           # Pull out the compiled function from the newly-created Module
45           code = compile(dest_ast, "", "exec")
46           globals_dict = copy.copy(fn.__globals__)
47           keys_before = set(globals_dict.keys())
48           exec(code, globals_dict)
49           new_keys = list(set(globals_dict.keys()) - keys_before)
50           assert len(new_keys) == 1
51           fn_compiled = globals_dict[new_keys[0]]
52   
53           # return the compiled function with the original globals
54           def change_func_globals(f, globals):
55               """Based on https://stackoverflow.com/a/13503277/2988730 (@unutbu)"""
56               # __globals__ is a private member of the function class
57               # so we have to copy the function, f, all of its member, except f.__globals__
58               g = FunctionType(
59                   f.__code__,
60                   globals,
61                   name=f.__name__,
62                   argdefs=f.__defaults__,
63                   closure=f.__closure__,
64               )
65               g = functools.update_wrapper(g, f)
66               g.__kwdefaults__ = copy.copy(f.__kwdefaults__)  # type:ignore[attr-defined]
67               return g
68   
69           # Return the correct FunctionType object
70           return change_func_globals(fn_compiled, globals=fn.__globals__)
71   
72       def visit_Assert(self, node):
73           """ 
74           Swap out the Assert node (Python's `assert`) with a callsite to the 
75           symbolically-traceable torch._assert function 
76           """
77           # Create the Call node
78           n = ast.parse("torch._assert()", mode="eval")
79           assert isinstance(n, ast.Expression)
80           call_node = n.body
81           assert isinstance(call_node, ast.Call)
82           msg = node.msg if node.msg else ast.Constant(value="", kind=None)
83           call_node.args = [node.test, msg]
84   
85           # Ensure that the new node conforms to the Python AST grammar
86           expr_wrapper = ast.Expr(value=call_node)
87   
88           # Return the new Call node to signify that we want to use it as
89           # a replacement for the original _assert node
90           return ast.copy_location(expr_wrapper, node)
91   
92       def visit_AnnAssign(self, node):
93           """ 
94           Swap out Python's AnnAssign with an Assign node where the annotation function is called. 
95           Example: 
96                Original: 
97                y: Tensor_Type(1,2,3, Dyn) = f2(x) 
98               Output: 
99                y = annotate(f2(x),Tensor_Type((1,2,3,Dyn))) 
100          """
101          return ast.Assign(
102              targets=[node.target],
103              value=ast.Call(
104                  func=ast.Name(id="annotate", ctx=ast.Load()),
105                  args=[node.value, node.annotation],
106                  keywords=[],
107              ),
108          )

111  class RewritingTracer(Tracer):
112      def trace(
113          self,
114          root: Union[torch.nn.Module, Callable],
115          concrete_args: Optional[dict[str, Any]] = None,
116      ) -> Graph:
117          return super().trace(_rewrite(root), concrete_args)

120  def _rewrite(fn: Union[torch.nn.Module, Callable]) -> Union[torch.nn.Module, Callable]:
121      if isinstance(fn, torch.nn.Module):
122          # Rewrite this module's `forward` as well as the `forward`s of
123          # all of this module's recursive descendents. Return the new,
124          # rewritten module hierarchy.
125          def rewrite_module(m: torch.nn.Module):
126              class RewrittenModule(torch.nn.Module):
127                  def __init__(self, orig):
128                      super().__init__()
129                      for k, v in orig.__dict__.items():
130                          if isinstance(v, torch.nn.Module):
131                              self.__dict__[k] = copy.copy(rewrite_module(v))
132                          else:
133                              self.__dict__[k] = copy.copy(v)
134  
135              RewrittenModule.forward = AST_Rewriter().rewrite(
136                  cast(FunctionType, m.forward)
137              )
138              return RewrittenModule(m)
139  
140          return rewrite_module(fn)
141      else:
142          # Rewrite this single free function
143          return AST_Rewriter().rewrite(cast(FunctionType, fn))
