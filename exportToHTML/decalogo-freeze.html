1    #!/usr/bin/env python3
2    # -*- coding: utf-8 -*-
3    """Genera locks y hashes reproducibles para los decálogos limpios."""
4    
5    from __future__ import annotations
6    
7    import argparse
8    import hashlib
9    import json
10   import sys
11   from pathlib import Path
12   
13   
14   def parse_args() -> argparse.Namespace:
15       parser = argparse.ArgumentParser(description="Congelar bundle de decálogos")
16       parser.add_argument("--bundle", required=True, help="Directorio con los JSON limpios")
17       return parser.parse_args()
18   
19   
20   def main() -> int:
21       args = parse_args()
22       bundle_dir = Path(args.bundle)
23       lock_entries = {}
24       for path in bundle_dir.glob("*.clean.json"):
25           content = path.read_bytes()
26           lock_entries[path.name] = hashlib.sha256(content).hexdigest()
27       crosswalk_path = bundle_dir / "crosswalk.latest.json"
28       if crosswalk_path.exists():
29           lock_entries[crosswalk_path.name] = hashlib.sha256(crosswalk_path.read_bytes()).hexdigest()
30       lock_path = bundle_dir / "decalogo.lock.json"
31       lock_path.write_text(json.dumps(lock_entries, indent=2, sort_keys=True) + "\n", encoding="utf-8")
32       print("Lock file generado en", lock_path)
33       return 0
34   
35   
36   if __name__ == "__main__":
37       sys.exit(main())
