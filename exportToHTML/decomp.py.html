17       from collections.abc import Mapping

22   class Decompose(_pass.Transform):
23       def __init__(
24           self,
25           module: torch.fx.GraphModule,
26           decomposition_table: Mapping[torch._ops.OpOverload, Callable],
27           enable_dynamic_axes: bool,
28           allow_fake_constant: bool | None = False,
29       ):
30           super().__init__(module)
31           self.decomposition_table = decomposition_table
32           self.enable_dynamic_axes = enable_dynamic_axes
33           self.allow_fake_constant = allow_fake_constant
34   
35       def _run(self, *args, **kwargs) -> torch.fx.GraphModule:
36           assert not kwargs, "kwargs is not supported in Decompose."
37   
38           # To preserve stack trace info after `make_fx`.
39           module = _utils.wrap_graph_module_for_node_meta_preservation(self.module)
40   
41           # fake mode use static size to trace the size of tensors. while symbolic
42           # mode generates aten::sym_size to dynamically trace the size of tensors.
43   
44           # e.g. fake mode:
45           #  view: f32[3, 5, 20] = torch.ops.aten.view.default(x, [3, 5, 20])
46   
47           # e.g. symbolic mode:
48           #  sym_size = torch.ops.aten.sym_size(x, 0)
49           #  sym_size_1 = torch.ops.aten.sym_size(x, 1)
50           #  sym_size_2 = torch.ops.aten.sym_size(x, 2)
51           #  sym_size_3 = torch.ops.aten.sym_size(x, 3)
52           #  mul = sym_size_2 * sym_size_3;  sym_size_2 = sym_size_3 = None
53           #  view: f32[3, 5, 20] = torch.ops.aten.view.default(x, [sym_size, sym_size_1, mul])
54   
55           # Mimic `torch._dynamo.export(aten_graph=True)` behavior in invoking `make_fx`.
56           # TODO: May need revisit for user fake mode export + dynamic shape scenario.
57           fake_mode: fake_tensor.FakeTensorMode | None = self.fake_mode
58           maybe_fake_args = self._maybe_fakefy_args(fake_mode, *args)
59           if fake_mode is not None:
60               # Using existing fake mode as context, signal `make_fx` that it does not need
61               # to create a new fake mode by passing tracing_mode as "real".
62               tracing_mode = "real"
63           else:
64               # Existing fake mode not found, signal `make_fx` to create one.
65               fake_mode = contextlib.nullcontext()  # type: ignore[assignment]
66               tracing_mode = "symbolic" if self.enable_dynamic_axes else "fake"
67   
68           # Apply decomposition table to the input graph.
69           assert fake_mode is not None  # for mypy
70           with (
71               fake_tensor.unset_fake_temporarily(),
72               python_dispatch.enable_python_dispatcher(),
73               fake_mode,
74           ):
75               decomposed_module = proxy_tensor.make_fx(
76                   module,
77                   decomposition_table=self.decomposition_table,
78                   tracing_mode=tracing_mode,
79                   _allow_non_fake_inputs=True,
80                   _allow_fake_constant=bool(self.allow_fake_constant),
81               )(*maybe_fake_args)
82   
83           # Rename placeholder targets to match the original module's signature since
84           # We don't want to map forward(x, y, z) to forward(arg0, arg1, arg2).
85           _utils.replace_placeholder_name_and_target(decomposed_module, self.module)
86   
87           return decomposed_module
