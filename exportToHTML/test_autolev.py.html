17   def _test_examples(in_filename, out_filename, test_name=""):
18   
19       in_file_path = os.path.join(FILE_DIR, 'autolev', 'test-examples',
20                                   in_filename)
21       correct_file_path = os.path.join(FILE_DIR, 'autolev', 'test-examples',
22                                        out_filename)
23       with open(in_file_path) as f:
24           generated_code = parse_autolev(f, include_numeric=True)
25   
26       with open(correct_file_path) as f:
27           for idx, line1 in enumerate(f):
28               if line1.startswith("#"):
29                   break
30               try:
31                   line2 = generated_code.split('\n')[idx]
32                   assert line1.rstrip() == line2.rstrip()
33               except Exception:
34                   msg = 'mismatch in ' + test_name + ' in line no: {0}'
35                   raise AssertionError(msg.format(idx+1))

95   def test_output_01():
96       """Autolev example calculates the position, velocity, and acceleration of a 
97       point and expresses in a single reference frame:: 
98    
99             (1) FRAMES C,D,F 
100            (2) VARIABLES FD'',DC'' 
101            (3) CONSTANTS R,L 
102            (4) POINTS O,E 
103            (5) SIMPROT(F,D,1,FD) 
104         -> (6) F_D = [1, 0, 0; 0, COS(FD), -SIN(FD); 0, SIN(FD), COS(FD)] 
105            (7) SIMPROT(D,C,2,DC) 
106         -> (8) D_C = [COS(DC), 0, SIN(DC); 0, 1, 0; -SIN(DC), 0, COS(DC)] 
107            (9) W_C_F> = EXPRESS(W_C_F>, F) 
108         -> (10) W_C_F> = FD'*F1> + COS(FD)*DC'*F2> + SIN(FD)*DC'*F3> 
109            (11) P_O_E>=R*D2>-L*C1> 
110            (12) P_O_E>=EXPRESS(P_O_E>, D) 
111         -> (13) P_O_E> = -L*COS(DC)*D1> + R*D2> + L*SIN(DC)*D3> 
112            (14) V_E_F>=EXPRESS(DT(P_O_E>,F),D) 
113         -> (15) V_E_F> = L*SIN(DC)*DC'*D1> - L*SIN(DC)*FD'*D2> + (R*FD'+L*COS(DC)*DC')*D3> 
114            (16) A_E_F>=EXPRESS(DT(V_E_F>,F),D) 
115         -> (17) A_E_F> = L*(COS(DC)*DC'^2+SIN(DC)*DC'')*D1> + (-R*FD'^2-2*L*COS(DC)*DC'*FD'-L*SIN(DC)*FD'')*D2> + (R*FD''+L*COS(DC)*DC''-L*SIN(DC)*DC'^2-L*SIN(DC)*FD'^2)*D3> 
116   
117      """
118  
119      if not antlr4:
120          skip('Test skipped: antlr4 is not installed.')
121  
122      autolev_input = """\ 
123  FRAMES C,D,F 
124  VARIABLES FD'',DC'' 
125  CONSTANTS R,L 
126  POINTS O,E 
127  SIMPROT(F,D,1,FD) 
128  SIMPROT(D,C,2,DC) 
129  W_C_F>=EXPRESS(W_C_F>,F) 
130  P_O_E>=R*D2>-L*C1> 
131  P_O_E>=EXPRESS(P_O_E>,D) 
132  V_E_F>=EXPRESS(DT(P_O_E>,F),D) 
133  A_E_F>=EXPRESS(DT(V_E_F>,F),D)\ 
134  """
135  
136      sympy_input = parse_autolev(autolev_input)
137  
138      g = {}
139      l = {}
140      exec(sympy_input, g, l)
141  
142      w_c_f = l['frame_c'].ang_vel_in(l['frame_f'])
143      # P_O_E> means "the position of point E wrt to point O"
144      p_o_e = l['point_e'].pos_from(l['point_o'])
145      v_e_f = l['point_e'].vel(l['frame_f'])
146      a_e_f = l['point_e'].acc(l['frame_f'])
147  
148      # NOTE : The Autolev outputs above were manually transformed into
149      # equivalent SymPy physics vector expressions. Would be nice to automate
150      # this transformation.
151      expected_w_c_f = (l['fd'].diff()*l['frame_f'].x +
152                        cos(l['fd'])*l['dc'].diff()*l['frame_f'].y +
153                        sin(l['fd'])*l['dc'].diff()*l['frame_f'].z)
154  
155      assert (w_c_f - expected_w_c_f).simplify() == 0
156  
157      expected_p_o_e = (-l['l']*cos(l['dc'])*l['frame_d'].x +
158                        l['r']*l['frame_d'].y +
159                        l['l']*sin(l['dc'])*l['frame_d'].z)
160  
161      assert (p_o_e - expected_p_o_e).simplify() == 0
162  
163      expected_v_e_f = (l['l']*sin(l['dc'])*l['dc'].diff()*l['frame_d'].x -
164                        l['l']*sin(l['dc'])*l['fd'].diff()*l['frame_d'].y +
165                        (l['r']*l['fd'].diff() +
166                         l['l']*cos(l['dc'])*l['dc'].diff())*l['frame_d'].z)
167      assert (v_e_f - expected_v_e_f).simplify() == 0
168  
169      expected_a_e_f = (l['l']*(cos(l['dc'])*l['dc'].diff()**2 +
170                                sin(l['dc'])*l['dc'].diff().diff())*l['frame_d'].x +
171                        (-l['r']*l['fd'].diff()**2 -
172                         2*l['l']*cos(l['dc'])*l['dc'].diff()*l['fd'].diff() -
173                         l['l']*sin(l['dc'])*l['fd'].diff().diff())*l['frame_d'].y +
174                        (l['r']*l['fd'].diff().diff() +
175                         l['l']*cos(l['dc'])*l['dc'].diff().diff() -
176                         l['l']*sin(l['dc'])*l['dc'].diff()**2 -
177                         l['l']*sin(l['dc'])*l['fd'].diff()**2)*l['frame_d'].z)
178      assert (a_e_f - expected_a_e_f).simplify() == 0
