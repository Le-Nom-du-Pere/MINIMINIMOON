5    inline std::optional<MemoryFormat>
6    check_tensor_options_and_extract_memory_format(
7        const TensorOptions& options,
8        std::optional<MemoryFormat> memory_format) {
9      TORCH_CHECK(
10         options.requires_grad_opt() != true,
11         "Operators taking TensorOptions cannot take a TensorOptions with "
12         "options.requires_grad set as true. This isn't implemented yet.");
13     TORCH_CHECK(
14         !(options.has_memory_format() && memory_format.has_value()),
15         "Cannot set memory_format both in TensorOptions and explicit argument; please delete "
16         "the redundant setter.");
17     if (memory_format.has_value()) {
18       return memory_format;
19     } else {
20       return options.memory_format_opt();
21     }
22   }
