27   # Base Pattern Handler
28   class FuseHandler(ABC):
29       """Base handler class for the fusion patterns"""
30   
31       @abstractmethod
32       def __init__(self, node: Node):
33           pass
34   
35       @abstractmethod
36       def fuse(
37           self,
38           load_arg: Callable,
39           named_modules: dict[str, torch.nn.Module],
40           fused_graph: Graph,
41           root_node: Node,
42           extra_inputs: list[Any],
43           matched_node_pattern: NodePattern,
44           fuse_custom_config: FuseCustomConfig,
45           fuser_method_mapping: dict[Pattern, Union[torch.nn.Sequential, Callable]],
46           is_qat: bool,
47       ) -> Node:
48           pass

51   class DefaultFuseHandler(FuseHandler):
52       def __init__(self, node: Node):
53           super().__init__(node)  # type:ignore[safe-super]
54   
55       def fuse(
56           self,
57           load_arg: Callable,
58           named_modules: dict[str, torch.nn.Module],
59           fused_graph: Graph,
60           root_node: Node,
61           extra_inputs: list[Any],
62           matched_node_pattern: NodePattern,
63           fuse_custom_config: FuseCustomConfig,
64           fuser_method_mapping: dict[Pattern, Union[torch.nn.Sequential, Callable]],
65           is_qat: bool,
66       ) -> Node:
67           assert root_node.op == "call_module", (
68               "Expecting module node to be a call_module Node"
69           )
70           root_module = named_modules[str(root_node.target)]
71   
72           def get_modules(pattern):
73               """Given a node pattern, extract the corresponding modules 
74               e.g. input: (relu_node, (bn_node, conv_node)) 
75                    output: (relu_module, (bn_module, conv_module)) 
76               """
77               if isinstance(pattern, (tuple, list)):
78                   n, *args = pattern
79                   modules: list[torch.nn.Module] = []
80                   modules.append(get_modules(n))
81                   modules.extend(get_modules(a) for a in args)
82                   return tuple(modules)
83               else:
84                   n = pattern
85                   if n.op == "call_module":
86                       return named_modules[n.target]
87                   elif n.op == "call_function" and n.target == torch.nn.functional.relu:
88                       relu = torch.nn.ReLU()
89                       relu.training = root_module.training
90                       return relu
91                   elif n.op == "call_function" or n.op == "call_method":
92                       return n.target
93                   else:
94                       return MatchAllNode
95   
96           # since relu can be used multiple times, we'll need to create a relu module for each match
97           matched_modules = get_modules(matched_node_pattern)
98   
99           def get_matched_types(m):
100              if isinstance(m, tuple):
101                  return tuple(map(get_matched_types, m))
102              if isinstance(m, torch.nn.Module):
103                  return type_before_parametrizations(m)
104              return m
105  
106          matched_module_types = get_matched_types(matched_modules)
107          module_parent_name, module_name = _parent_name(root_node.target)
108          fuser_method = get_fuser_method_new(matched_module_types, fuser_method_mapping)
109          # TODO: change the signature for fuser_method to take matched module patterns
110          # as input
111          fused_module = fuser_method(is_qat, *matched_modules)
112          setattr(named_modules[module_parent_name], module_name, fused_module)
113          extra_args = [load_arg(input) for input in extra_inputs]
114          node = fused_graph.node_copy(root_node, load_arg)
115          args = list(node.args)
116          args.extend(extra_args)
117          node.args = tuple(args)
118          return node

121  def _get_fusion_pattern_to_fuse_handler_cls(
122      backend_config: BackendConfig,
123  ) -> dict[Pattern, Callable]:
124      fusion_pattern_to_fuse_handlers: dict[Pattern, Callable] = {}
125      for pattern, config in backend_config._pattern_complex_format_to_config.items():
126          if config.fuser_method is not None:
127              # TODO: is this logic right?
128              fusion_pattern_to_fuse_handlers[pattern] = DefaultFuseHandler
129      return fusion_pattern_to_fuse_handlers
