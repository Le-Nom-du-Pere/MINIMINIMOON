21   @compatibility(is_backward_compatible=False)
22   def replace_target_nodes_with(
23       fx_module: GraphModule,
24       old_op: str,
25       old_target: Target,
26       new_op: str,
27       new_target: Target,
28   ):
29       """Modifies all nodes in fx_module.graph.nodes which match the specified op code and target, 
30       and updates them to match the new op code and target"""
31       new_graph = Graph()
32       val_map: dict[Node, Node] = {}
33       for node in fx_module.graph.nodes:
34           if node.op == old_op and node.target == old_target:
35               args = map_arg(node.args, lambda n: val_map[n])
36               kwargs = map_arg(node.kwargs, lambda n: val_map[n])
37               assert isinstance(args, tuple)
38               assert isinstance(kwargs, dict)
39               val_map[node] = new_graph.create_node(
40                   new_op, new_target, args, kwargs, node.name
41               )
42           else:
43               val_map[node] = new_graph.node_copy(node, lambda n: val_map[n])
44       fx_module.graph = new_graph

53   @compatibility(is_backward_compatible=False)
54   def get_size_of_all_nodes(
55       fx_module: GraphModule, args: Optional[list[torch.Tensor]] = None
56   ) -> None:
57       """Given a fx graph module, update each node with its total size (weights + bias + output) 
58       and its output_size(output). For a non-module node, the total size is the output size. 
59       return total size"""
60       if args is not None:
61           # Mark shape and dtype for each node (node.shape and node.dtype)
62           ShapeProp(fx_module).propagate(*args)
63       # Calculate the total size of the whole fx graph
64       for node in fx_module.graph.nodes:
65           if node.op == "output":
66               break
67           node.size_bytes = get_size_of_node(fx_module, node)
68       return

71   @compatibility(is_backward_compatible=False)
72   def get_tensor_meta(node: Node) -> Any:
73       tensor_meta = node.meta.get("tensor_meta")
74   
75       if not tensor_meta:
76           raise RuntimeError(
77               f"Node {node} has no tensor metadata associated with it! "
78               f"Check that shape propagation has run."
79           )
80   
81       return tensor_meta

84   @compatibility(is_backward_compatible=False)
85   def get_size_of_node(fx_module: GraphModule, node: Node) -> size_bytes:
86       """Given a node with node.dtype and node.shape, return its total size and its output size. 
87       total_size = weights + bias + output_size 
88       """
89       # Total num of elements
90       total_num_of_elems = 0
91       # For a module, conside all parameters
92       if node.op == "call_module":
93           submodule_dict = dict(fx_module.named_modules())
94           submodule = submodule_dict[node.target]
95           parameters = submodule.named_parameters()
96           # Parameters are named tuples
97           for _name, p in parameters:
98               total_num_of_elems += p.numel()
99       # Don't forget the output size
100      # node.shape is the shape of this node's output
101      tensor_meta = get_tensor_meta(node)
102      output_elem = tensor_meta.shape.numel()
103      total_num_of_elems += output_elem
104      # Assume for now if it's quantized then it's qint8 or quint8
105      if tensor_meta.is_quantized:
106          size_per_elem_bytes = torch._empty_affine_quantized(
107              [], dtype=tensor_meta.dtype
108          ).element_size()
109      else:
110          size_per_elem_bytes = torch.tensor([], dtype=tensor_meta.dtype).element_size()
111      total_size = size_per_elem_bytes * total_num_of_elems
112      output_size = size_per_elem_bytes * output_elem
113      return size_bytes(output_size, total_size)
