15   def _to_domain(m, domain=None):
16       """Convert Matrix to DomainMatrix"""
17       # XXX: deprecated support for RawMatrix:
18       ring = getattr(m, "ring", None)
19       m = m.applyfunc(lambda e: e.as_expr() if isinstance(e, Poly) else e)
20   
21       dM = DomainMatrix.from_Matrix(m)
22   
23       domain = domain or ring
24       if domain is not None:
25           dM = dM.convert_to(domain)
26       return dM

29   def smith_normal_form(m, domain=None):
30       ''' 
31       Return the Smith Normal Form of a matrix `m` over the ring `domain`. 
32       This will only work if the ring is a principal ideal domain. 
33    
34       Examples 
35       ======== 
36    
37       >>> from sympy import Matrix, ZZ 
38       >>> from sympy.matrices.normalforms import smith_normal_form 
39       >>> m = Matrix([[12, 6, 4], [3, 9, 6], [2, 16, 14]]) 
40       >>> print(smith_normal_form(m, domain=ZZ)) 
41       Matrix([[1, 0, 0], [0, 10, 0], [0, 0, 30]]) 
42    
43       '''
44       dM = _to_domain(m, domain)
45       return _snf(dM).to_Matrix()

48   def is_smith_normal_form(m, domain=None):
49       ''' 
50       Checks that the matrix is in Smith Normal Form 
51       '''
52       dM = _to_domain(m, domain)
53       return _is_snf(dM)

56   def smith_normal_decomp(m, domain=None):
57       ''' 
58       Return the Smith Normal Decomposition of a matrix `m` over the ring 
59       `domain`. This will only work if the ring is a principal ideal domain. 
60    
61       Examples 
62       ======== 
63    
64       >>> from sympy import Matrix, ZZ 
65       >>> from sympy.matrices.normalforms import smith_normal_decomp 
66       >>> m = Matrix([[12, 6, 4], [3, 9, 6], [2, 16, 14]]) 
67       >>> a, s, t = smith_normal_decomp(m, domain=ZZ) 
68       >>> assert a == s * m * t 
69       '''
70       dM = _to_domain(m, domain)
71       a, s, t = _snd(dM)
72       return a.to_Matrix(), s.to_Matrix(), t.to_Matrix()

75   def invariant_factors(m, domain=None):
76       ''' 
77       Return the tuple of abelian invariants for a matrix `m` 
78       (as in the Smith-Normal form) 
79    
80       References 
81       ========== 
82    
83       .. [1] https://en.wikipedia.org/wiki/Smith_normal_form#Algorithm 
84       .. [2] https://web.archive.org/web/20200331143852/https://sierra.nmsu.edu/morandi/notes/SmithNormalForm.pdf 
85    
86       '''
87       dM = _to_domain(m, domain)
88       factors = _invf(dM)
89       factors = tuple(dM.domain.to_sympy(f) for f in factors)
90       # XXX: deprecated.
91       if hasattr(m, "ring"):
92           if m.ring.is_PolynomialRing:
93               K = m.ring
94               to_poly = lambda f: Poly(f, K.symbols, domain=K.domain)
95               factors = tuple(to_poly(f) for f in factors)
96       return factors

99   def hermite_normal_form(A, *, D=None, check_rank=False):
100      r""" 
101      Compute the Hermite Normal Form of a Matrix *A* of integers. 
102   
103      Examples 
104      ======== 
105   
106      >>> from sympy import Matrix 
107      >>> from sympy.matrices.normalforms import hermite_normal_form 
108      >>> m = Matrix([[12, 6, 4], [3, 9, 6], [2, 16, 14]]) 
109      >>> print(hermite_normal_form(m)) 
110      Matrix([[10, 0, 2], [0, 15, 3], [0, 0, 2]]) 
111   
112      Parameters 
113      ========== 
114   
115      A : $m \times n$ ``Matrix`` of integers. 
116   
117      D : int, optional 
118          Let $W$ be the HNF of *A*. If known in advance, a positive integer *D* 
119          being any multiple of $\det(W)$ may be provided. In this case, if *A* 
120          also has rank $m$, then we may use an alternative algorithm that works 
121          mod *D* in order to prevent coefficient explosion. 
122   
123      check_rank : boolean, optional (default=False) 
124          The basic assumption is that, if you pass a value for *D*, then 
125          you already believe that *A* has rank $m$, so we do not waste time 
126          checking it for you. If you do want this to be checked (and the 
127          ordinary, non-modulo *D* algorithm to be used if the check fails), then 
128          set *check_rank* to ``True``. 
129   
130      Returns 
131      ======= 
132   
133      ``Matrix`` 
134          The HNF of matrix *A*. 
135   
136      Raises 
137      ====== 
138   
139      DMDomainError 
140          If the domain of the matrix is not :ref:`ZZ`. 
141   
142      DMShapeError 
143          If the mod *D* algorithm is used but the matrix has more rows than 
144          columns. 
145   
146      References 
147      ========== 
148   
149      .. [1] Cohen, H. *A Course in Computational Algebraic Number Theory.* 
150         (See Algorithms 2.4.5 and 2.4.8.) 
151   
152      """
153      # Accept any of Python int, SymPy Integer, and ZZ itself:
154      if D is not None and not ZZ.of_type(D):
155          D = ZZ(int(D))
156      return _hnf(A._rep, D=D, check_rank=check_rank).to_Matrix()
