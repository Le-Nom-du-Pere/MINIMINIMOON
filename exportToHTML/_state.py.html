2    """JIT-related state. 
3     
4    This module stores various pieces of Python-global state relating to the JIT. 
5     
6    This is not intended to be imported directly; please the exposed 
7    functionalities in `torch.jit`. 
8    """
9    
10   import os
11   import weakref
12   from typing import Any
13   
14   import torch
15   
16   
17   class EnabledProxy:
18       """Stores whether the JIT is enabled or not. 
19    
20       This is just a wrapper for a bool, so that we get reference semantics 
21       """
22   
23       def __init__(self) -> None:
24           self.enabled = self.parse_env(
25               "PYTORCH_JIT", True, "> Using PyTorch JIT", "> PyTorch JIT DISABLED"
26           )
27   
28       def parse_env(self, name, default, true_message, false_message):
29           value = os.environ.get(name)
30           if value is None:
31               return default
32           if value.lower() in {"1", "true", "yes"}:
33               return True
34           elif value.lower() in {"0", "false", "no"}:
35               return False
36           if value == "1v":
37               print(true_message)
38               return True
39           elif value == "0v":
40               print(false_message)
41               return False
42           raise ValueError(f"Unknown setting of {name}. Try using 0 or 1.")
43   
44       def __bool__(self):
45           return self.enabled
46   
47   
48   _enabled = EnabledProxy()
49   
50   
51   def disable():
52       _enabled.enabled = False
53   
54   
55   def enable():
56       _enabled.enabled = True
57

59   # The Python CompilationUnit. All functions and modules defined in Python will
60   # live in here. It's defined in Python because doing in cpp creates static
61   # destruction order issues.
62   _python_cu = torch._C.CompilationUnit()
63   
64   
65   # python class => ScriptClass mapping
66   _script_classes: dict[type[Any], type[Any]] = {}
67   _name_to_pyclass: dict[str, type[Any]] = {}
68   
69   
70   def _add_script_class(python_class, script_class):
71       _script_classes[python_class] = script_class
72       _name_to_pyclass[script_class.qualified_name()] = python_class
73   
74   
75   def _get_script_class(python_class):
76       override = getattr(python_class, "_jit_override_qualname", None)
77       if override is not None:
78           python_class = _get_python_class(override)
79       return _script_classes.get(python_class, None)
80   
81   
82   def _get_python_class(qualified_name):
83       return _name_to_pyclass.get(qualified_name, None)
84   
85   
86   def _clear_class_state():
87       _script_classes.clear()
88       _name_to_pyclass.clear()
89

91   # Caching: we currently cache compilation of free functions and overloaded functions.
92   # To cache free functions we hold a weak ref to the function object and
93   # map to the compiled fn's qualified name.
94   # To cache overloaded functions we hold a weak ref to the function obj and
95   # map to all of its overloaded compiled fns.
96   # In the future we could consider caching more types of objects so that
97   # aliasing is preserved across separate compilations of the same object.
98   
99   _jit_caching_layer: weakref.WeakKeyDictionary = weakref.WeakKeyDictionary()
100  _jit_function_overload_caching: weakref.WeakKeyDictionary = weakref.WeakKeyDictionary()
101  
102  
103  def _try_get_jit_cached_overloads(key):
104      qual_names = _jit_function_overload_caching.get(key, None)
105      if qual_names:
106          return [_python_cu.find_function(qual_name) for qual_name in qual_names]
107      else:
108          return None
109  
110  
111  def _set_jit_overload_cache(key, compiled_fns):
112      _jit_function_overload_caching[key] = [fn.qualified_name for fn in compiled_fns]
113  
114  
115  def _try_get_jit_cached_function(key):
116      if getattr(key, "__disable_jit_function_caching__", False) is True:
117          return None
118      qual_name = _jit_caching_layer.get(key, None)
119      if qual_name:
120          return _python_cu.find_function(qual_name)
121      else:
122          return None
123  
124  
125  def _set_jit_function_cache(key, value):
126      # only free functions currently supported
127      assert isinstance(value, torch.jit.ScriptFunction)
128      _jit_caching_layer[key] = value.qualified_name
