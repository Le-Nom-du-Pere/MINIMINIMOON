40   @onnx_impl(aten.group_norm.default, trace_only=True, opset_introduced=21)
41   def aten_group_norm(
42       input: TFloat,
43       num_groups: int,
44       weight: Optional[TFloat] = None,
45       bias: Optional[TFloat] = None,
46       eps: float = 1e-05,
47       cudnn_enabled: bool = True,
48   ) -> TFloat:
49       """group_norm(Tensor input, int num_groups, Tensor? weight=None, Tensor? bias=None, float eps=1e-05, bool cudnn_enabled=True) -> Tensor"""
50   
51       c = op21.Shape(input, start=1, end=2)
52       if weight is None:
53           weight = op21.ConstantOfShape(c, value=ir.tensor(1.0, dtype=input.dtype))
54       if bias is None:
55           bias = op21.ConstantOfShape(c, value=ir.tensor(0.0, dtype=input.dtype))
56       return op21.GroupNormalization(
57           input, weight, bias, epsilon=eps, num_groups=num_groups
58       )

61   @onnx_impl(
62       aten.scaled_dot_product_attention.default, trace_only=True, opset_introduced=23
63   )
64   def aten_scaled_dot_product_attention_23(
65       query: TFloat,
66       key: TFloat,
67       value: TFloat,
68       attn_mask: Optional[TFloat] = None,
69       dropout_p: float = 0.0,
70       is_causal: bool = False,
71       scale: Optional[float] = None,
72       enable_gqa: bool = False,
73   ) -> TFloat:
74       """scaled_dot_product_attention(Tensor query, Tensor key, Tensor value, Tensor? attn_mask=None, float dropout_p=0.0, bool is_causal=False, *, float? scale=None, bool enable_gqa=False) -> Tensor 
75    
76       Reference: 
77           1. https://pytorch.org/docs/stable/generated/torch.nn.functional.scaled_dot_product_attention.html 
78           2. https://onnx.ai/onnx/operators/onnx__Attention.html 
79    
80       Attempts to convert SDPA to Attention onnx op and fallbacks to an onnx graph equivivalent to the following PyTorch code:: 
81           scale_factor = 1 / math.sqrt(Q.size(-1)) if scale is None else scale 
82           attn_mask = ( 
83               torch.ones(L, S, dtype=torch.bool).tril(diagonal=0) 
84               if is_causal 
85               else attn_mask 
86           ) 
87           attn_mask = ( 
88               attn_mask.masked_fill(not attn_mask, -float("inf")) 
89               if attn_mask.dtype == torch.bool 
90               else attn_mask 
91           ) 
92           attn_weight = torch.softmax( 
93               (Q @ K.transpose(-2, -1) *  attn_mask, dim=-1 
94           ) 
95           attn_weight = torch.dropout(attn_weight, dropout_p) 
96           return attn_weight @ V 
97    
98       where Q, K, V are the query, key, and value tensors, respectively. 
99       L is the target sequence length, S is the source sequence length, and E is the embedding size. 
100      """
101      assert (not is_causal) or (is_causal and attn_mask is None), (
102          "is_causal and attn_mask cannot be set at the same time"
103      )
104      assert len(query.shape) == 4 and len(key.shape) == 4 and len(value.shape) == 4, (
105          "only 4D query, key, and value are supported"
106      )
107  
108      # Attention onnx op can only handle non-training scenarios where dropout is disabled.
109      if dropout_p == 0:
110          if enable_gqa:
111              assert (
112                  query.shape[1] > key.shape[1] == value.shape[1]
113                  and query.shape[1] % key.shape[1] == 0
114              ), (
115                  "SDPA (GQA or MQA) requires q_num_heads > kv_num_heads & q_num_heads % kv_num_heads == 0"
116              )
117          else:
118              assert query.shape[1] == key.shape[1] == value.shape[1], (
119                  "SDPA (MHA) requires q_num_heads = kv_num_heads"
120              )
121  
122          # NOTE: num_heads attributes (q_num_heads/kv_num_heads) should not be specified for 4D.
123          # They are not populated with 4D inputs because this information directy comes from input shapes:
124          # `q_num_heads=query.shape[1]` and `kv_num_heads=key.shape[1]`.
125          # This dimension is usually static but it could not be dynamic if also given as an attribute.
126          # num_heads attributes are needed for 3D attention inputs:
127          # (shape: [B, S, N*H]), 4D shape is ([B, N, S, H]).
128  
129          Y, _, _, _ = op23.Attention(
130              query,
131              key,
132              value,
133              attn_mask=attn_mask,
134              scale=scale,
135              is_causal=is_causal,
136          )
137          return Y
138  
139      if scale is None:
140          scale = _attention_scale(query, op23)
141      scale = op23.CastLike(scale, query)
142  
143      if is_causal:
144          attn_mask = _causal_attention_mask(query, key, op23)
145  
146      if attn_mask is None:
147          return _aten_scaled_dot_product_attention_no_mask_onnx(
148              query, key, value, scale, dropout_p, op23
149          )
150  
151      return _aten_scaled_dot_product_attention_float_mask_onnx(
152          query, key, value, attn_mask, scale, dropout_p, op23
153      )

156  def _attention_scale(query: TFloat, op: Opset) -> TFloat:
157      """Calculate the scale factor for the attention result. 
158   
159      Args: 
160          query: Tensor of shape [..., L, E] 
161   
162      Returns: 
163          Scalar scale factor := 1 / math.sqrt(query.size(-1)) 
164      """
165      q_shape = op.Shape(query)
166      q_last_dim = op.Gather(q_shape, op.Constant(value_ints=[-1]))
167      embedding_size = op.CastLike(q_last_dim, query)
168      one = op.Constant(value_float=1.0)
169      cast_one = op.CastLike(one, query)
170      scale = op.Div(cast_one, op.Sqrt(embedding_size))
171      return scale

174  def _causal_attention_mask(query: TFloat, key: TFloat, op: Opset) -> TFloat:
175      """Create a causal mask for the given query and key tensors. 
176   
177      Equivalent to:: 
178          mask = torch.ones(L, S, dtype=torch.bool).tril(diagonal=0) 
179          attn_mask = torch.zeros(L, S, dtype=torch.float) 
180          attn_mask = attn_mask.masked_fill(not mask, -float("inf")) 
181   
182      Args: 
183          query: Tensor of shape [..., L, E] 
184          key: Tensor of shape [..., S, E] 
185   
186      Returns: 
187          Tensor of shape [L, S] 
188      """
189      q_shape = op.Shape(query)
190      k_shape = op.Shape(key)
191  
192      target_length = op.Slice(
193          q_shape, op.Constant(value_ints=[-2]), op.Constant(value_ints=[-1])
194      )
195      source_length = op.Slice(
196          k_shape, op.Constant(value_ints=[-2]), op.Constant(value_ints=[-1])
197      )
198      # attn_mask = torch.ones(L, S) := {
199      size = op.Concat(target_length, source_length, axis=0)
200      attn_mask = op.Expand(op.Constant(value_float=1.0), size)
201      # }
202      attn_mask = op.Trilu(attn_mask, upper=0)
203      # The causal mask has 0s in the lower triangle and -inf in the upper triangle.
204      attn_mask = op.Where(
205          op.Equal(attn_mask, op.Constant(value_float=0.0)),
206          op.Constant(value_float=-float("inf")),
207          op.Constant(value_float=0.0),
208      )
209      attn_mask = op.CastLike(attn_mask, query)
210      return attn_mask

213  def _aten_scaled_dot_product_attention_no_mask_onnx(
214      query: TFloat,
215      key: TFloat,
216      value: TFloat,
217      scale: TFloat,
218      dropout_p: float,
219      op: Opset,
220  ) -> TFloat:
221      # Swap the last two axes of key
222      key_last_dim = op.Shape(key, start=-1)
223      key_second_last_dim = op.Shape(key, start=-2, end=-1)
224      key_first_dims = op.Shape(key, end=-2)
225      # Contract the dimensions that are not the last two so we can transpose
226      # with a static permutation.
227      key_squeezed_shape = op.Concat(
228          op.Constant(value_ints=[-1]), key_second_last_dim, key_last_dim, axis=0
229      )
230      key_squeezed = op.Reshape(key, key_squeezed_shape)
231      key_squeezed_transposed = op.Transpose(key_squeezed, perm=[0, 2, 1])
232      key_transposed_shape = op.Concat(
233          key_first_dims, key_last_dim, key_second_last_dim, axis=0
234      )
235      key_transposed = op.Reshape(key_squeezed_transposed, key_transposed_shape)
236  
237      # https://github.com/pytorch/pytorch/blob/12da0c70378b5be9135c6fda62a9863bce4a4818/aten/src/ATen/native/transformers/attention.cpp#L653
238      # Scale q, k before matmul for stability see https://tinyurl.com/sudb9s96 for math
239      query_scaled = op.Mul(query, op.Sqrt(scale))
240      key_transposed_scaled = op.Mul(
241          key_transposed, op.CastLike(op.Sqrt(scale), key_transposed)
242      )
243      attn_weight = op.Softmax(
244          op.MatMul(query_scaled, key_transposed_scaled),
245          axis=-1,
246      )
247      attn_weight, _ = op.Dropout(attn_weight, dropout_p)
248      return op.MatMul(attn_weight, value)

251  def _aten_scaled_dot_product_attention_float_mask_onnx(
252      query: TFloat,
253      key: TFloat,
254      value: TFloat,
255      attn_mask: TFloat,
256      scale: TFloat,
257      dropout_p: float,
258      op: Opset,
259  ) -> TFloat:
260      # Swap the last two axes of key
261      key_last_dim = op.Shape(key, start=-1)
262      key_second_last_dim = op.Shape(key, start=-2, end=-1)
263      key_first_dims = op.Shape(key, end=-2)
264      # Contract the dimensions that are not the last two so we can transpose
265      # with a static permutation.
266      key_squeezed_shape = op.Concat(
267          op.Constant(value_ints=[-1]), key_second_last_dim, key_last_dim, axis=0
268      )
269      key_squeezed = op.Reshape(key, key_squeezed_shape)
270      key_squeezed_transposed = op.Transpose(key_squeezed, perm=[0, 2, 1])
271      key_transposed_shape = op.Concat(
272          key_first_dims, key_last_dim, key_second_last_dim, axis=0
273      )
274      key_transposed = op.Reshape(key_squeezed_transposed, key_transposed_shape)
275  
276      # https://github.com/pytorch/pytorch/blob/12da0c70378b5be9135c6fda62a9863bce4a4818/aten/src/ATen/native/transformers/attention.cpp#L653
277      # Scale q, k before matmul for stability see https://tinyurl.com/sudb9s96 for math
278      query_scaled = op.Mul(query, op.Sqrt(scale))
279      key_transposed_scaled = op.Mul(key_transposed, op.Sqrt(scale))
280      attn_weight = op.Softmax(
281          op.Add(op.MatMul(query_scaled, key_transposed_scaled), attn_mask),
282          axis=-1,
283      )
284      attn_weight, _ = op.Dropout(attn_weight, dropout_p)
285      return op.MatMul(attn_weight, value)
