61   def run(fn=None):
62       """Don't do any dynamic compiles, just use prior optimizations"""
63       if fn is not None:
64           fn = innermost_fn(fn)
65           assert callable(fn)
66           return RunOnlyContext()(fn)
67       return RunOnlyContext()

70   def disable(fn=None, recursive=True, *, reason=None, wrapping=True):
71       """ 
72       Decorator to disable TorchDynamo 
73    
74       If recursive=True, Dynamo is completely skipped on the decorated function 
75       frame as well as the recursively invoked functions. 
76    
77       If recursive=False, Dynamo skips frames associated with the function code, 
78       but still process recursively invoked frames. 
79    
80       If reason is provided, it will be printed when Dynamo attempts to trace the disabled function. 
81       """
82       if recursive:
83           if fn is not None:
84               fn = innermost_fn(fn)
85               assert callable(fn)
86               return DisableContext(msg=reason, wrapping=wrapping)(fn)
87           return DisableContext(msg=reason, wrapping=wrapping)
88       else:
89   
90           def wrap(fn):
91               fn = innermost_fn(fn)
92               assert callable(fn)
93   
94               nonrecursive_disable_wrapper = get_nonrecursive_disable_wrapper(fn)
95               nonrecursive_disable_wrapper._torchdynamo_disable = True  # type: ignore[attr-defined]
96               nonrecursive_disable_wrapper._torchdynamo_disable_msg = reason  # type: ignore[attr-defined]
97               nonrecursive_disable_wrapper._torchdynamo_orig_callable = fn  # type: ignore[attr-defined]
98               return nonrecursive_disable_wrapper
99   
100          if fn is None:
101              return wrap
102          return wrap(fn)
103

109  def skip(fn=None):
110      """ 
111      Skip frames associated with the function code, but still process recursively 
112      invoked frames 
113      """
114      if fn is None:
115          return skip
116      fn = innermost_fn(fn)
117      assert callable(fn)
118      skip_code(fn.__code__)
119      fn._torchdynamo_disable = True
120      return fn

123  class set_stance(_DecoratorContextManager):
124      """ 
125      Decorator, context manager, function to set the current stance of the compiler. 
126   
127      Stances documented in corresponding function in torch/compiler/__init__.py 
128      """
129  
130      _dynamo_forbidden = True
131  
132      def __init__(
133          self,
134          stance: str = "default",
135          *,
136          skip_guard_eval_unsafe: bool = False,
137          force_backend=None,
138      ) -> None:
139          if force_backend is not None and stance != "default":
140              raise RuntimeError("non-default stance cannot have force_backend set")
141  
142          self.stance = DynamoStance(stance, skip_guard_eval_unsafe, force_backend)
143          self.prev = _set_stance(self.stance)
144  
145      def __call__(self, fn):
146          _set_stance(self.prev)
147          wrapper = super().__call__(fn)
148          # forbid wrapper in graph
149          wrapper._dynamo_forbidden = True  # type: ignore[attr-defined]
150          return wrapper
151  
152      def __enter__(self):
153          _set_stance(self.stance)
154  
155      def __exit__(self, exc_type, exc_val, exc_tb):
156          _set_stance(self.prev)
157  
158      def clone(self):
159          return self.__class__(self.stance.stance, force_backend=self.stance.backend)

167  def allow_in_graph(fn):
168      """ 
169      Tells the compiler frontend (Dynamo) to skip symbolic introspection of the function 
170      and instead directly write it to the graph when encountered. 
171   
172      See :func:`torch.compiler.allow_in_graph`'s docstring for the full documentation 
173   
174      WARNING: this API can be a footgun, please read the documentation carefully. 
175      """
176      if isinstance(fn, (list, tuple)):
177          return [allow_in_graph(x) for x in fn]
178      assert callable(fn), "allow_in_graph expects a callable"
179      if trace_rules.lookup_callable(fn) != variables.TorchInGraphFunctionVariable:
180          fn_id = id(fn)
181          trace_rules._disallowed_callable_ids.remove(fn_id)
182          trace_rules._allowed_callable_ids.add(fn_id)
183  
184          # Avoid id reuse which creates subtle bugs.
185          def deregister():
186              trace_rules._allowed_callable_ids.remove(fn_id)
187  
188          weakref.finalize(fn, deregister)
189      return fn

192  def nonstrict_trace(traceable_fn):
193      # Like `allow_in_graph`, but with the following enhancements/differences:
194      #
195      # 1. Supports user-defined class as inputs, as long as the class has been
196      #    registered with pytree.
197      # 2. Reads to global/captured tensors forces the underlying graph to treat
198      #    those tensors as constant, and we _assume_ they will not be updated. This
199      #    is similar to FX tracing.
200      # 3. In the resulting Dynamo graph, the call to a `nonstrict_trace`-ed function
201      #    will be represented as a call to `torch._higher_order_ops.flat_apply`,
202      #    which takes in the `nonstrict_trace`-ed function and pytree-flattened
203      #    inputs.
204      # 4. Only the returned function is traceable, and the original function will
205      #    not be. Moreover, `nonstrict_trace` can be used inside a `torch.compile`
206      #    region.
207      #
208      # NOTE: like `allow_in_graph`, aliasing information is neither preserved
209      # between inputs themselves, nor between inputs and outputs.
210      assert callable(traceable_fn), "nonstrict_trace expects a callable"
211  
212      @functools.wraps(traceable_fn)
213      def wrapped(*args, **kwargs):
214          return traceable_fn(*args, **kwargs)
215  
216      wrapped_id = id(wrapped)
217  
218      # This line allows us to reuse much of the `allow_in_graph` impl.
219      trace_rules._allowed_callable_ids.add(wrapped_id)
220  
221      # This line allows us to diverge the impl from `allow_in_graph`.
222      trace_rules._nonstrict_trace_callable_ids.add(wrapped_id)
223  
224      # Avoid id reuse which creates subtle bugs.
225      def deregister():
226          trace_rules._allowed_callable_ids.remove(wrapped_id)
227          trace_rules._nonstrict_trace_callable_ids.remove(wrapped_id)
228  
229      weakref.finalize(wrapped, deregister)
230  
231      return wrapped

234  def _disallow_in_graph_helper(throw_if_not_allowed):
235      def inner(fn):
236          if isinstance(fn, (list, tuple)):
237              return [disallow_in_graph(x) for x in fn]
238          assert callable(fn), "disallow_in_graph expects a callable"
239          if (
240              throw_if_not_allowed
241              and trace_rules.lookup_callable(fn)
242              != variables.TorchInGraphFunctionVariable
243              and trace_rules.lookup(fn) != variables.TorchInGraphFunctionVariable
244          ):
245              raise IncorrectUsage(
246                  "disallow_in_graph is expected to be used on an already allowed callable (like torch.* ops). "
247                  "Allowed callables means callables that TorchDynamo puts as-is in the extracted graph."
248              )
249          trace_rules._allowed_callable_ids.remove(id(fn))
250          trace_rules._nonstrict_trace_callable_ids.remove(id(fn))
251          trace_rules._disallowed_callable_ids.add(id(fn))
252          return fn
253  
254      return inner

257  def disallow_in_graph(fn):
258      """ 
259      Customize which functions TorchDynamo will exclude in the generated 
260      graph and force a graph break on. 
261      :: 
262   
263          torch._dynamo.disallow_in_graph(torch.sub) 
264   
265   
266          @torch._dynamo.optimize(...) 
267          def fn(a): 
268              x = torch.add(x, 1) 
269              x = torch.sub(x, 1) 
270              x = torch.add(x, 1) 
271              return x 
272   
273   
274          fn(...) 
275   
276      Will break the graph on `torch.sub`, and give two graphs each with a 
277      single `torch.add()` op. 
278      """
279      return _disallow_in_graph_helper(throw_if_not_allowed=True)(fn)

294      """ 
295      Customize which functions TorchDynamo will assert are not present while tracing.

298      TODO(voz): We now have allow_in_graph, disallow_in_graph, forbid_in_graph - some more robust

308  def substitute_in_graph(
309      original_fn: Callable[_P, _R],
310      *,
311      can_constant_fold_through: bool = False,
312      skip_signature_check: bool = False,
313      # type that is embedded in the Python interpreter
314      is_embedded_type: bool = False,  # internal use only
315  ) -> Callable[[Callable[_P, _R]], Callable[_P, _R]]:
316      """ 
317      Register a polyfill handler for a function, usually a C function from the C extension, to be 
318      used in place of the original function when inlining the original function in the graph. 
319   
320      .. note:: 
321   
322          The polyfill handler is only used when inlining the original function. It is not used when 
323          the original function is called directly. In the eager mode, the decorated function calls 
324          the performant C function rather than the polyfill handler. 
325   
326      The polyfill handler is a function that will be called in place of the original function when 
327      inlining the original function. The polyfill handler should have the same signature and the same 
328      behavior as the original function. 
329   
330      Args: 
331          original_fn (callable): The original function, usually a C function, to register a polyfill 
332              handler for. 
333          can_constant_fold_through (bool, optional): Whether the polyfill handler can be constant 
334              folded through. That is, if the polyfill handler is a pure function and its arguments 
335              are constant, the result of the polyfill handler can be constant folded during the 
336              compilation. Defaults to ``False``. 
337          skip_signature_check (bool, optional): Whether to skip the signature check between the 
338              original function and the polyfill handler. Defaults to ``False``. 
339   
340      Returns: 
341          A decorator that registers the polyfill handler for the original function. 
342   
343      Example:: 
344   
345          >>> # xdoctest: +SKIP("conflict with the tests: duplicate polyfill handlers") 
346          >>> import operator 
347          >>> operator.indexOf([1, 2, 3, 4, 5], 3) 
348          2 
349          >>> torch.compile(operator.indexOf, fullgraph=True)([1, 2, 3, 4, 5], 3) 
350          Traceback (most recent call last): 
351          ... 
352          torch._dynamo.exc.Unsupported: ... 
353   
354          >>> @torch.compiler.substitute_in_graph(operator.indexOf) 
355          ... def indexOf(a, b, /): 
356          ...     for i, item in enumerate(a): 
357          ...         if item is b or item == b: 
358          ...             return i 
359          ...     raise ValueError("sequence.index(x): x not in sequence") 
360          >>> 
361          >>> torch.compile(operator.indexOf, fullgraph=True)([1, 2, 3, 4, 5], 3) 
362          2 
363      """
364      if not is_function(original_fn) and not (
365          is_embedded_type and inspect.isclass(original_fn)
366      ):
367          raise TypeError(
368              f"substitute_in_graph expects a function but got {type(original_fn)!r}"
369          )
370      if is_embedded_type:
371          if not inspect.isclass(original_fn):
372              raise TypeError(
373                  f"substitute_in_graph expects a class but got {type(original_fn)!r}"
374              )
375  
376          from .variables.builder import ITERTOOLS_POLYFILLED_TYPE_IDS, ITERTOOLS_TYPE_IDS
377  
378          if id(original_fn) in ITERTOOLS_TYPE_IDS:
379              ITERTOOLS_POLYFILLED_TYPE_IDS.add(id(original_fn))
380  
381      def wrapper(traceable_fn: Callable[_P, _R]) -> Callable[_P, _R]:
382          if not is_function(traceable_fn):
383              raise TypeError(
384                  f"@substitute_in_graph(...) expects a function but got {type(traceable_fn)!r}"
385              )
386  
387          if not skip_signature_check:
388              try:
389                  original_sig = inspect.signature(original_fn)
390              except ValueError:
391                  pass
392              else:
393                  traceable_sig = inspect.signature(traceable_fn)
394  
395                  def sig_ident(sig):
396                      # Ignore annotations for parameters and return type
397                      return (
398                          tuple(
399                              p.name
400                              for p in sig.parameters.values()
401                              if (
402                                  p.kind
403                                  not in {
404                                      p.KEYWORD_ONLY,
405                                      # the name of *args and **kwargs is not important
406                                      p.VAR_POSITIONAL,
407                                      p.VAR_KEYWORD,
408                                  }
409                              )
410                          ),
411                          {
412                              p.name
413                              for p in sig.parameters.values()
414                              if p.kind == p.KEYWORD_ONLY
415                          },
416                          {
417                              p.name: p.default
418                              for p in sig.parameters.values()
419                              # the name of *args and **kwargs is not important
420                              if p.kind not in {p.VAR_POSITIONAL, p.VAR_KEYWORD}
421                          },
422                      )
423  
424                  wildcard_sig = inspect.signature(lambda *args, **kwargs: None)
425  
426                  if (
427                      sig_ident(original_sig) != sig_ident(traceable_sig)
428                      and sig_ident(original_sig) != sig_ident(wildcard_sig)
429                      and sig_ident(traceable_sig) != sig_ident(wildcard_sig)
430                  ):
431                      raise TypeError(
432                          f"Signature mismatch between {original_fn} and {traceable_fn}: "
433                          f"{original_sig} != {traceable_sig}"
434                      )
435  
436          from torch._dynamo.guards import GuardBuilder
437          from torch._dynamo.trace_rules import (
438              _polyfilled_function_ids,
439              get_torch_obj_rule_map,
440          )
441          from torch._dynamo.variables import PolyfilledFunctionVariable
442          from torch._dynamo.variables.builder import VariableBuilder
443  
444          id_dispatch_map = VariableBuilder._id_dispatch()
445          if id(original_fn) in id_dispatch_map:
446              raise ValueError(
447                  f"Duplicate dispatch rule for {original_fn}: "
448                  "already registered in VariableBuilder's id dispatch map"
449              )
450  
451          if id(original_fn) in _polyfilled_function_ids:
452              raise ValueError(f"Duplicate polyfilled object {original_fn}")
453  
454          rule_map: dict[Any, type[VariableTracker]] = get_torch_obj_rule_map()
455          if original_fn in rule_map:
456              raise ValueError(
457                  f"Duplicate object {original_fn} with different rules: "
458                  f"{PolyfilledFunctionVariable}, {rule_map[original_fn]}"
459              )
460  
461          polyfill_handlers: dict[Callable[..., Any], FunctionType]
462          polyfill_handlers = PolyfilledFunctionVariable._get_polyfill_handlers()
463          if original_fn in polyfill_handlers:
464              raise ValueError(
465                  f"Duplicate polyfill handlers for {original_fn}: "
466                  f"already handled by {polyfill_handlers[original_fn]}"
467              )
468  
469          # Need to wrap the function because we may cannot assign __torch_dynamo_polyfill__ to a
470          # C++ function.
471          @functools.wraps(traceable_fn)
472          def wrapped(*args: _P.args, **kwargs: _P.kwargs) -> _R:
473              return original_fn(*args, **kwargs)
474  
475          def dispatch_fn(self, value: Callable[_P, _R]) -> PolyfilledFunctionVariable:
476              return PolyfilledFunctionVariable(
477                  value,
478                  source=self.source,
479                  **self.install_guards(GuardBuilder.FUNCTION_MATCH),
480              )
481  
482          id_dispatch_map[id(original_fn)] = id_dispatch_map[id(wrapped)] = dispatch_fn
483          _polyfilled_function_ids.add(id(original_fn))
484          _polyfilled_function_ids.add(id(wrapped))
485          rule_map[original_fn] = rule_map[wrapped] = PolyfilledFunctionVariable
486          polyfill_handlers[original_fn] = polyfill_handlers[wrapped] = wrapped  # type: ignore[assignment]
487  
488          wrapped.__torch_dynamo_original__ = original_fn  # type: ignore[attr-defined]
489          wrapped.__torch_dynamo_polyfill__ = traceable_fn  # type: ignore[attr-defined]
490          wrapped.__torch_dynamo_can_constant_fold_through__ = can_constant_fold_through  # type: ignore[attr-defined]
491  
492          return wrapped  # type: ignore[return-value]
493  
494      return wrapper

497  # Helper function to flatten a tensor subclass and apply a function to
498  # all inner tensors that match the outer dim. Used to reduce duplication

500  def _apply_func_to_inner_tensors_of_same_dim(func, t, *args, **kwargs):
501      assert is_traceable_wrapper_subclass(t)
502  
503      attrs, _ctx = t.__tensor_flatten__()
504      assert isinstance(t, torch.Tensor)
505      for attr in attrs:
506          inner = getattr(t, attr)
507          if inner.dim() == t.dim():
508              func(inner, *args, **kwargs)

524  @forbid_in_graph
525  def mark_unbacked(t, index, strict=False, specialize_on=None):
526      """ 
527      Mark a tensor as having an unbacked dim.  This changes the semantics of operations, 
528      we will always report the size does not equal zero/one, we will turn asserts 
529      on this index into runtime asserts, and if you try to get the real value we will 
530      raise an exception.  In other words, we will treat this dimension as if it was 
531      data dependent (we do not know anything about its value.) 
532   
533      For historical reasons, by default if an unbacked dim is specialized, we will 
534      happily specialize it and continue. If you want to error in these cases, pass 
535      strict=True. 
536      """
537      # You could have copied the mark_dynamic behavior but I'm not convinced
538      # it's what you want
539      assert not is_traceable_wrapper_subclass(t), "not implemented yet"
540  
541      if isinstance(index, int):
542          if strict:
543              if not hasattr(t, "_dynamo_strict_unbacked_indices"):
544                  t._dynamo_strict_unbacked_indices = set()
545              t._dynamo_strict_unbacked_indices.add(index)
546              return
547  
548          if not hasattr(t, "_specialized_on"):
549              t._specialize_on = {}
550  
551          if not hasattr(t, "_dynamo_unbacked_indices"):
552              t._dynamo_unbacked_indices = set()
553  
554          # FX tracers don't respect @forbid_in_graph and choke on the following error since it passes in proxies:
555          # TypeError: 'Attribute' object does not support item assignment
556          if isinstance(t._specialize_on, dict):
557              t._specialize_on[index] = specialize_on if specialize_on is not None else []
558  
559          t._dynamo_unbacked_indices.add(index)
560          return
561  
562      assert isinstance(index, (list, tuple))
563      for i in index:
564          mark_unbacked(t, i)

567  @forbid_in_graph
568  def mark_dynamic(t, index, *, min=None, max=None, specialize_on=None):
569      """ 
570      Mark a tensor as having a dynamic dim and set corresponding min and max range for the dim. 
571   
572      [Note - on the state of mark_dynamic] 
573   
574      The behavior of having a dynamic dimension on a tensor is governed by a few factors: 
575   
576      1) torch._dynamo.config dynamic_shapes True or False. 
577          a) dynamic_shapes=True - dynamic_shapes must be True for mark_dynamic to work. 
578          a) dynamic_shapes=False - This config will raise an exception when used in conjunction with 
579          mark_dynamic. We will eventually support this. 
580   
581      2) If the dimension is fully constrained - as in, it does not allow more than a single value 
582      in both eager (torch.compile, torch._dynamo.optimize) mode and export mode (torch._dynamo.export), 
583      we will raise an error 
584   
585      3) If the dimension is partially constrained - allowing at least 2 values but not the full unbounded 
586      range of shapes, in eager we will pass it through, but export will raise an error. 
587   
588      4) Attempts to trace this function will explicitly raise. As such, all calls to mark_dynamic must be made 
589      before torch.compile. 
590   
591      5) If specialize_on is passed in, we will perform a single generic Dynamo trace followed by 
592      multiple specialized compilations in addition to a single generic compilation. NB: For now we only support 
593      per dimension specialization, or in other words we do not generate a cross product of specializations. 
594      At runtime, we will dispatch to a specialized compiled region if the input matches the specialization criteria. 
595   
596      For example: 
597          mark_dynamic(..., specialize_on=[ 
598              lambda x: x == 8, 
599              lambda x: x == 16 
600          ]) 
601   
602      This approach results in one Dynamo trace and two backend compilations. When the input dimension equals 8 or 16 
603      at runtime, execution will be directed to the specialized compiled region. Performance measurements indicate 
604      2-8x speedups depending on the specific specialization and model architecture. 
605      """
606      if is_traceable_wrapper_subclass(t):
607          # default behavior: mirror mark_dynamic() on all inner tensors with same dim as t
608          # TODO: Make this configurable via a supported public API
609          _apply_func_to_inner_tensors_of_same_dim(
610              mark_dynamic, t, index, min=min, max=max
611          )
612  
613      if isinstance(index, int):
614          if not hasattr(t, "_dynamo_dynamic_indices"):
615              t._dynamo_dynamic_indices = set()
616              t._dynamo_dynamic_range = set()
617  
618          if not hasattr(t, "_specialize_on"):
619              t._specialize_on = {}
620  
621          # TODO(voz): Should we bounds check?
622          t._dynamo_dynamic_indices.add(index)
623          t._dynamo_dynamic_range.add(_DimRange(index, min, max))
624  
625          # FX tracers don't respect @forbid_in_graph and choke on the following error since it passes in proxies:
626          # TypeError: 'Attribute' object does not support item assignment
627          if isinstance(t._specialize_on, dict):
628              t._specialize_on[index] = specialize_on if specialize_on is not None else []
629  
630          return
631  
632      assert isinstance(index, (list, tuple))
633      for i in index:
634          mark_dynamic(t, i, min=min, max=max)
635          mark_dynamic(t, i, min=min, max=max, specialize_on=specialize_on)

639  def maybe_mark_dynamic(t, index):
640      """ 
641      Mark a tensor as having a dynamic dim, but don't enforce it (i.e., if this 
642      dimension ends up getting specialized, don't error). 
643      """
644      if is_traceable_wrapper_subclass(t):
645          # default behavior: mirror maybe_mark_dynamic() on all inner tensors with same dim as t
646          # TODO: Make this configurable via a supported public API
647          _apply_func_to_inner_tensors_of_same_dim(maybe_mark_dynamic, t, index)
648  
649      if isinstance(index, int):
650          if not hasattr(t, "_dynamo_weak_dynamic_indices"):
651              t._dynamo_weak_dynamic_indices = set()
652          # TODO(voz): Should we bounds check?
653          t._dynamo_weak_dynamic_indices.add(index)
654          return
655  
656      assert isinstance(index, (list, tuple))
657      for i in index:
658          maybe_mark_dynamic(t, i)

661  def mark_static(t, index=None):
662      """ 
663      Mark a tensor as having a static dim or mark a nn module class as static. 
664   
665      For tensors 
666      =========== 
667      This will prevent us from attempting to compile it dynamically 
668      when dynamic=True; this can improve trace-time performance. 
669   
670      This has lower precedence than mark_dynamic. 
671   
672      Unlike mark_dynamic, this can be done inside a graph, in which case it 
673      induces specialization on the tensor. 
674   
675      For nn.Module classes 
676      ===================== 
677      For static nn.Module classes, TorchDynamo assumes that the module instance 
678      attributes will not be modified after compilation. This will ensure that 
679      TorchDynamo keeps integer attributes CONSTANT and not symints. 
680   
681      From TorchDynamo implementation side, the instances of static-marked 
682      nn.Module class will be converted to UnspecializedBuiltinNNModuleVariable, 
683      which have the same properties. 
684   
685      Note that we still have to guard on the attributes, because different 
686      instances of the nn.Module can have different values of the attributes. The 
687      key point here is that the attributes are static. 
688      """
689      if is_compiling():
690          if index is None:
691              for s in t.size():
692                  comptime.force_static(s)
693          else:
694              comptime.force_static(t.size(index))
695          return
696  
697      if is_traceable_wrapper_subclass(t):
698          # default behavior: mirror mark_static() on all inner tensors with same dim as t
699          # TODO: Make this configurable via a supported public API
700          _apply_func_to_inner_tensors_of_same_dim(mark_static, t, index)
701  
702      if not isinstance(t, torch.Tensor) and issubclass(t, torch.nn.Module):
703          t._dynamo_marked_static = True
704          return t
705  
706      if not isinstance(t, torch.Tensor):
707          raise TypeError(
708              f"mark_static expects a tensor/nn.Module class but received {type(t)}"
709          )
710  
711      if isinstance(index, int):
712          if not hasattr(t, "_dynamo_static_indices"):
713              t._dynamo_static_indices = set()  # type: ignore[attr-defined]
714          # TODO(voz): Should we bounds check?
715          t._dynamo_static_indices.add(index)  # type: ignore[attr-defined]
716      elif index is None:
717          for i in range(t.dim()):
718              mark_static(t, i)
719      else:
720          assert isinstance(index, (list, tuple))
721          for i in index:
722              mark_static(t, i)

728      Marks an input tensor whose data_ptr will not change across multiple calls 
729      to a dynamo-compiled function. This indicates to cudagraphs that an extra allocation

773  # Proxy class for torch._dynamo.config patching - so dynamo can identify context managers/decorators
774  # created by patch_dynamo_config, compared to ones created by a raw torch._dynamo.config.patch.
775  class DynamoConfigPatchProxy:
776      def __init__(self, config_patch):
777          self.config_patch = config_patch
778  
779      @property
780      def changes(self):
781          return self.config_patch.changes
782  
783      # Decorator implementation that simply sets up `self` as a context manager.
784      # Placed in external_utils so that we can trace through it.
785      __call__ = _dynamo_config_patch_proxy_dunder_call
786  
787      def __enter__(self):
788          return self.config_patch.__enter__()
789  
790      def __exit__(self, exc_type, exc_val, exc_tb):
791          return self.config_patch.__exit__(exc_type, exc_val, exc_tb)

794  # Criteria for patchable config:
795  # - Config values must be constants (i.e. int, float, str, bool, None).
796  #     - in particular, NO list, set, dict.
797  # - Traceable config patches are only useful for configs that change dynamo behavior
798  #   from symbolic_convert and below.
799  #     - e.g. patching recompile_limit won't really do anything.
800  # - For patching configs that affect Dynamo behavior above symbolic_convert,
801  #   ensure that Dynamo behaves soundly even if tracing is done with different config.
802  #     - e.g. be careful if patching guard-related configs as configs may have changed
803  #       between guard creation and evaluation.
804  _allowed_config_patches = (
805      "verbose",
806      "verify_correctness",
807      "rewrite_assert_with_torch_assert",
808      "capture_scalar_outputs",
809      "allow_unspec_int_on_nn_module",
810      "skip_torchrec",
811      "dont_skip_tracing",
812  )
813  
814  from . import config
815  
816  
817  for name in _allowed_config_patches:
818      assert hasattr(config, name), "nonexistent config"
819  del config
820

822  def _patch_dynamo_config_check(changes: dict[str, Any]):
823      for k, v in changes.items():
824          if k not in _allowed_config_patches:
825              raise ValueError(
826                  f"patch_dynamo_config does not support patching config {k}"
827              )
828          if not torch._dynamo.utils.is_safe_constant(v):
829              raise ValueError(
830                  f"patch_dynamo_config does not support patching config {k} "
831                  f"with non-safe-constant value {v}"
832              )

835  # TODO: also implement nonrecursive patch_dynamo_config/dont_skip_tracing.
836  # Unlike config.patch, we also need to accept tuple as input in order to
837  # deal with context manager reconstruction.
838  def patch_dynamo_config(
839      arg1: Optional[Union[str, dict[str, Any], tuple[tuple[str, Any], ...]]] = None,
840      arg2: Any = None,
841      **kwargs: Any,
842  ) -> DynamoConfigPatchProxy:
843      """ 
844      A wrapper around torch._dynamo.config.patch that can be traced by Dynamo to 
845      temporarily change config values DURING tracing. 
846   
847      See _allowed_config_patches for the list of allowed config patches. 
848   
849      Arguments are the same as with torch._dynamo.config.patch. 
850   
851      Can be used as a decorator or a context manager. 
852   
853      User code SHOULD NOT MODIFY the return value of this function. 
854   
855      WARNING: changing Dynamo config during tracing can lead to unpredictable tracing behavior! 
856          Proceed only as advised! 
857      """
858      if isinstance(arg1, tuple):
859          arg1 = dict(arg1)
860      config_patch = torch._dynamo.config.patch(arg1, arg2, **kwargs)
861      _patch_dynamo_config_check(config_patch.changes)
862      # check for valid patching using config_patch.changes
863      return DynamoConfigPatchProxy(config_patch)

866  def dont_skip_tracing(fn=None):
867      """ 
868      Context manager/decorator to trace into functions intentionally marked by developers to be skipped 
869      when tracing. 
870   
871      This decorator will also apply to recursively invoked functions. 
872      """
873      ctx = patch_dynamo_config(dont_skip_tracing=True)
874      if fn:
875          return ctx(fn)
876      return ctx
