14   # See also the TYPE_CHECKING block in torch/_dynamo/polyfills/__init__.py
15   POLYFILLED_MODULE_NAMES: tuple[str, ...] = (
16       "builtins",
17       "functools",
18       "itertools",
19       "operator",
20       "os",
21       "pytree",
22       "sys",
23       "fx",

26   POLYFILLED_MODULES: tuple["ModuleType", ...] = tuple(
27       importlib.import_module(f".{submodule}", package=polyfills.__name__)
28       for submodule in POLYFILLED_MODULE_NAMES
29   )
30   
31   
32   # Unregister the builtin functions from _builtin_function_ids to let them to be
33   # dispatched with the appropriate VariableTracker type. Otherwise, they will be
34   # dispatched with BuiltinVariable if present in _builtin_function_ids.
35   for polyfill_module in POLYFILLED_MODULES:
36       for polyfill_name in polyfill_module.__all__:
37           polyfill_handler = getattr(polyfill_module, polyfill_name)
38           original_fn = polyfill_handler.__torch_dynamo_original__
39           trace_rules._builtin_function_ids.remove(id(original_fn))
