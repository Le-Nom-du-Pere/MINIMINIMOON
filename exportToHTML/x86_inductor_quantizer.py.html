60   class _X86InductorQuantizationAnnotation(QuantizationAnnotation):
61       # _is_output_of_quantized_pattern:
62       #  * Node as output node of a fusion pattern.
63       #  * The fusion pattern supports int8 data type.
64       #  * The fusion pattern has inputs annotated to insert observer.
65       #  * The quantization_config is not `None`.
66       _is_output_of_quantized_pattern: bool = False
67   
68   
69   # Operators that:
70   # 1. Operators are optimized to run with int8 when int8 input provided.
71   # 2. Operators do not support int8 input and produce fp32 output.
72   int8_in_int8_out_ops: set = {
73       torch.ops.aten.max_pool2d.default,
74       torch.ops.aten.cat.default,
75       torch.ops.aten.avg_pool2d.default,
76       torch.ops.aten.adaptive_avg_pool2d.default,
77       torch.ops.aten.flatten.using_ints,
78   }
79   
80   # Operators that support the int8 data type for quantization config propagation.
81   # A superset of int8_in_int8_out_ops incorporating additional operators.
82   propagation_quantizable_ops = int8_in_int8_out_ops
83   
84   # Operators support the int8 data type
85   # and recipe is configured by default in X86InductorQuantizer.
86   default_quantizable_ops = propagation_quantizable_ops | {
87       torch.ops.aten.conv1d.default,
88       torch.ops.aten.conv2d.default,
89       torch.ops.aten.linear.default,
90   }
91   
92   # A superset of default_quantizable_ops includes operators support the int8 data type
93   # but not enabled by default recipe of X86InductorQuantizer.
94   quantizable_ops = default_quantizable_ops | {
95       torch.ops.aten.matmul.default,
96   }

101  def _skip_annotate(nodes: list[Node], filter_fn: Optional[FilterFn] = None) -> bool:
102      """Determine whether to skip annotation for a list of nodes."""
103  
104      # 1) Skip annotate if any node is already annotated
105      if _is_any_annotated(nodes):
106          return True
107  
108      # 2) Proceed annotate if a) a filter function is provided
109      # and b) the given nodes list passes the filter function check.
110      if filter_fn and filter_fn(nodes):
111          return False
112  
113      return True

116  def _create_module_name_filter(module_name: str) -> FilterFn:
117      """Create a filter function for a given module name. 
118   
119      The filter function takes a list of nodes (as determined by the annotate function) 
120      and return True if *all* nodes come from the specified module name, False otherwise. 
121   
122      For example: 
123          linear_1: "f32[3, 10]" = torch.ops.aten.linear.default(...) # comes from a module with name `sub.linear1` 
124          relu: "f32[3, 10]" = torch.ops.aten.relu.default(linear_1); # comes from a module with name `sub.relu1` 
125   
126      >> module_name_filter = _create_module_name_filter_inner("sub") 
127      >> print(module_name_filter([relu, linear_1])) 
128      # True  # These two nodes are determined by `_annotate_linear_unary` function and from "sub". 
129      """
130  
131      filter_fn = _get_module_name_filter(module_name)
132  
133      def check_all_nodes_from_module(nodes: list[Node]) -> bool:
134          all_nodes_from_module_name: bool = all(filter_fn(n) for n in nodes)
135          return all_nodes_from_module_name
136  
137      return check_all_nodes_from_module

140  def _create_operator_type_filter(
141      operator_type: Callable,
142  ) -> FilterFn:
143      """Create a filter function for a given operator type. 
144   
145      The filter function takes a list of nodes and returns True if it contains 
146      exactly one node with the specified operator type, False otherwise. 
147   
148      For example: 
149          linear_1: "f32[3, 10]" = torch.ops.aten.linear.default(...) # comes from a module with name `sub.linear1` 
150          relu: "f32[3, 10]" = torch.ops.aten.relu.default(linear_1); # comes from a module with name `sub.relu1` 
151   
152      >> operator_type_filter = _create_operator_type_filter(torch.ops.aten.linear.default) 
153      >> print(operator_type_filter([relu, linear_1])) 
154      # True  # These two nodes are determined by `_annotate_linear_unary` function and the second node is `linear`. 
155      """
156  
157      def operator_type_filter(nodes: list[Node]):
158          num_nodes_with_operator_type = sum(
159              node.target == operator_type for node in nodes
160          )
161          if num_nodes_with_operator_type > 1:
162              raise NotImplementedError(
163                  f"Several nodes within a single pattern are {operator_type}."
164              )
165          return num_nodes_with_operator_type == 1
166  
167      return operator_type_filter

186  def _map_module_function_to_aten_operator_type():
187      module_function_to_aten_operator: dict[Callable, torch._ops.OpOverloadPacket] = {}
188      map_list = (
189          ([torch.nn.Conv2d, F.conv1d], torch.ops.aten.conv1d.default),
190          ([torch.nn.Conv2d, F.conv2d], torch.ops.aten.conv2d.default),
191          ([torch.nn.Linear, F.linear], torch.ops.aten.linear.default),
192          ([torch.nn.MaxPool2d, F.max_pool2d], torch.ops.aten.max_pool2d.default),
193          (
194              [
195                  torch.cat,
196              ],
197              torch.ops.aten.cat.default,
198          ),
199          ([torch.nn.AvgPool2d, F.avg_pool2d], torch.ops.aten.avg_pool2d.default),
200          (
201              [torch.nn.AdaptiveAvgPool2d, F.adaptive_avg_pool2d],
202              torch.ops.aten.adaptive_avg_pool2d.default,
203          ),
204          (
205              [
206                  torch.flatten,
207              ],
208              torch.ops.aten.flatten.using_ints,
209          ),
210          (
211              [
212                  torch.matmul,
213              ],
214              torch.ops.aten.matmul.default,
215          ),
216      )

257  def _is_quantized_op_pt2e(node: torch.fx.Node):
258      """ 
259      Used for pt2e flow to check if the node is a quantized node: 
260      Case1: the node has been annotated as output node of a fusion pattern. 
261      Case2: the node has been annotated as single quantized node. 
262      """
263      if not _is_any_annotated([node]):
264          # The node has not been annotated, directly return False
265          return False

267      assert isinstance(quantization_annotation, _X86InductorQuantizationAnnotation)
268      return quantization_annotation._is_output_of_quantized_pattern

271  @functools.lru_cache
272  def get_default_x86_inductor_quantization_config(
273      is_qat: bool = False,
274      is_dynamic: bool = False,
275      reduce_range: bool = False,
276  ):
277      """ 
278      reduce_range is False by default. Set it to True on earlier CPUs without VNNI to avoid accuracy issue. 
279      """
280      extra_args: dict[str, Any] = {"eps": 2**-12}
281      if is_qat:
282          if is_dynamic:
283              act_observer_or_fake_quant_ctr = FakeQuantize
284              dynamic_quant_observer = MovingAverageMinMaxObserver.with_args(
285                  averaging_constant=1
286              )
287              extra_args["observer"] = dynamic_quant_observer
288          else:
289              act_observer_or_fake_quant_ctr = FusedMovingAvgObsFakeQuantize  # type: ignore[assignment]
290      else:
291          if is_dynamic:
292              act_observer_or_fake_quant_ctr = PlaceholderObserver  # type: ignore[assignment]
293          else:
294              act_observer_or_fake_quant_ctr = HistogramObserver  # type: ignore[assignment]
295  
296      # Copy from x86 default qconfig from torch/ao/quantization/qconfig.py
297      act_quantization_spec = QuantizationSpec(
298          dtype=torch.uint8,
299          quant_min=0,
300          quant_max=127 if reduce_range else 255,
301          qscheme=torch.per_tensor_affine,
302          is_dynamic=is_dynamic,
303          observer_or_fake_quant_ctr=act_observer_or_fake_quant_ctr.with_args(
304              **extra_args
305          ),
306      )
307  
308      weight_observer_or_fake_quant_ctr: _ObserverOrFakeQuantizeConstructor = (
309          FusedMovingAvgObsFakeQuantize if is_qat else PerChannelMinMaxObserver
310      )
311  
312      if is_qat:
313          # Only support per channel quant for now
314          extra_args["observer"] = MovingAveragePerChannelMinMaxObserver  # type: ignore[dict-item]
315      weight_quantization_spec = QuantizationSpec(
316          dtype=torch.int8,
317          quant_min=-128,
318          quant_max=127,
319          qscheme=torch.per_channel_symmetric,
320          ch_axis=0,  # 0 corresponding to weight shape = (oc, ic, kh, kw) of conv
321          is_dynamic=False,
322          observer_or_fake_quant_ctr=weight_observer_or_fake_quant_ctr.with_args(
323              **extra_args
324          ),
325      )
326      bias_quantization_spec = None  # will use placeholder observer by default
327      quantization_config = QuantizationConfig(
328          act_quantization_spec,
329          act_quantization_spec,
330          weight_quantization_spec,
331          bias_quantization_spec,
332          is_qat,
333      )
334      return quantization_config

383  @dataclass
384  class _CurrentQuantizationMode:
385      r"""Configuration defining the current quantization mode for the quantizer. 
386   
387      All possible current quantization modes are listed below: 
388      ---------------------------------------------------------------------------------------------------------- 
389                  |                                       dynamic_state 
390       qat_state  |--------------------------------------------------------------------------------------------- 
391                  |                           None                              |    True       |  False 
392      ---------------------------------------------------------------------------------------------------------- 
393          None    | quantizer does not receive a non-None `quantization_config` | \             | \ 
394          False   | quantizer will not do QAT                                   | dynamic       | static 
395          True    | quantizer will do QAT                                       | QAT + dynamic | QAT + static 
396      """
397  
398      qat_state: Optional[bool]
399      dynamic_state: Optional[bool]
400

402  class X86InductorQuantizer(Quantizer):
403      module_function_to_aten_operator_type = _map_module_function_to_aten_operator_type()
404  
405      def __init__(self) -> None:
406          super().__init__()
407          self.global_config: Optional[QuantizationConfig] = None
408          self.operator_type_qconfig: dict[
409              torch._ops.OpOverloadPacket, Optional[QuantizationConfig]
410          ] = {}
411          self.module_name_qconfig: dict[str, Optional[QuantizationConfig]] = {}
412  
413      def _get_current_quantization_mode(self) -> _CurrentQuantizationMode:
414          """Retrieves the current quantization mode based on all configurations."""
415          qat_state = None
416          dynamic_state = None
417  
418          # As we use `_need_skip_config` to skip all invalid configurations,
419          # we can safely assume that the all existing non-None configurations
420          # have the same quantization mode.
421          for qconfig in (
422              list(self.module_name_qconfig.values())
423              + list(self.operator_type_qconfig.values())
424              + [self.global_config]
425          ):
426              if qconfig is not None:
427                  # Query the `is_qat` state
428                  if qat_state is None:
429                      qat_state = qconfig.is_qat
430                  else:
431                      assert qat_state == qconfig.is_qat, (
432                          f"All non-None quantization configs should have the same `is_qat`,"
433                          f"but got {qat_state} and {qconfig.is_qat}."
434                      )
435                  # Query the `is_dynamic` state
436                  input_activation_spec = qconfig.input_activation
437                  if input_activation_spec is not None:
438                      if dynamic_state is None:
439                          dynamic_state = input_activation_spec.is_dynamic
440                      else:
441                          assert dynamic_state == input_activation_spec.is_dynamic, (
442                              f"All non-None `input_activation_spec` should have the same `is_dynamic`,"
443                              f"but got {dynamic_state} and {input_activation_spec.is_dynamic}."
444                          )
445          return _CurrentQuantizationMode(
446              qat_state=qat_state, dynamic_state=dynamic_state
447          )
448  
449      def _need_skip_config(
450          self, quantization_config: Optional[QuantizationConfig]
451      ) -> bool:
452          """Check if the provided quantization config is valid for X86InductorQuantizer. 
453   
454          Mixed static/dynamic configurations or mixed QAT/non-QAT configurations are not supported. 
455          To avoid such a mix, we compare the incoming configuration with current configuration status. 
456          Refer the `_CurrentQuantizationMode` definition for all possible modes. 
457          """
458          if quantization_config is None:
459              return False
460  
461          need_skip = False
462          current_mode = self._get_current_quantization_mode()
463          if (
464              current_mode.qat_state is not None
465              and current_mode.qat_state != quantization_config.is_qat
466          ):
467              warnings.warn("Mixed QAT and Non-QAT quantization config is not supported.")
468              need_skip = True
469          if current_mode.dynamic_state is not None:
470              input_activation_spec = quantization_config.input_activation
471              if (
472                  input_activation_spec is not None
473                  and current_mode.dynamic_state != input_activation_spec.is_dynamic
474              ):
475                  warnings.warn(
476                      "Mixed dynamic and static quantization config is not supported."
477                  )
478                  need_skip = True
479          return need_skip

481      def set_global(self, quantization_config: QuantizationConfig):
482          if self._need_skip_config(quantization_config):
483              warnings.warn("Skip the global quantization config.")
484              return self
485          self.global_config = quantization_config
486          return self

488      def get_global_quantization_config(self):

558      def _annotate_conv_node_helper(
559          self,
560          conv_node: torch.fx.Node,
561          annotate_output: bool,
562          quantization_config: Optional[QuantizationConfig],
563      ) -> None:
564          """Helper function to annotate the conv node"""
565          if quantization_config is None:
566              _annotate_nodes_not_quantize(conv_node)
567              return
568          input_qspec_map = {}
569          input_node = conv_node.args[0]
570          assert isinstance(input_node, Node)
571          input_qspec_map[input_node] = get_input_act_qspec(quantization_config)
572          weight_node = conv_node.args[1]
573          assert isinstance(weight_node, Node)
574          input_qspec_map[weight_node] = get_weight_qspec(quantization_config)
575          bias_node = None if len(conv_node.args) == 2 else conv_node.args[2]
576          if isinstance(bias_node, Node):
577              input_qspec_map[bias_node] = get_bias_qspec(quantization_config)
578          if annotate_output:
579              conv_node.meta[QUANT_ANNOTATION_KEY] = _X86InductorQuantizationAnnotation(
580                  input_qspec_map=input_qspec_map,
581                  _annotated=True,
582                  _is_output_of_quantized_pattern=True,
583              )
584          else:
585              conv_node.meta[QUANT_ANNOTATION_KEY] = _X86InductorQuantizationAnnotation(
586                  input_qspec_map=input_qspec_map,
587                  _annotated=True,
588              )

590      def _annotate_linear_node_helper(
591          self,
592          linear_node: torch.fx.Node,
593          annotate_output: bool,
594          quantization_config: Optional[QuantizationConfig],
595      ) -> None:
596          """Helper function to annotate the linear node"""
597          if quantization_config is None:
598              _annotate_nodes_not_quantize(linear_node)
599              return
600          input_qspec_map = {}
601          assert linear_node.target in (torch.ops.aten.linear.default,)
602          has_bias = len(linear_node.args) == 3
603          input_index = 0
604          weight_index = 1
605          bias_index = 2
606  
607          input_node = linear_node.args[input_index]
608          assert isinstance(input_node, Node)
609          input_qspec_map[input_node] = get_input_act_qspec(quantization_config)
610  
611          weight_node = linear_node.args[weight_index]
612          assert isinstance(weight_node, Node)
613          input_qspec_map[weight_node] = get_weight_qspec(quantization_config)
614  
615          bias_node = linear_node.args[bias_index] if has_bias else None
616          if isinstance(bias_node, Node):
617              input_qspec_map[bias_node] = get_bias_qspec(quantization_config)
618  
619          if annotate_output:
620              linear_node.meta[QUANT_ANNOTATION_KEY] = _X86InductorQuantizationAnnotation(

636          for partition in partition_list:

672      def annotate(self, model: torch.fx.GraphModule) -> torch.fx.GraphModule:
673          """Annotate the given model with quantization configurations. 
674   
675          Annotation contracts: 
676          1. Annotate each node according to the user's qconfig in the following order: 
677          `module_name_qconfig`, `operator_type_qconfig`, and `global_config`. 
678          2. Avoid re-annotating nodes already annotated in prior stages. For example, 
679          if `linear1` has been annotated by `module_name_qconfig`, it won't be annotated again 
680          during the processing of the 'operator_type_qconfig' or 'global_config'. 
681          3. For config is `None`, the node will be annotated with `_X86InductorQuantizationAnnotation(_annotated=True)`. 
682   
683          For each pair of (module_name_or_operator_type_or_global, qconfig), a filter function is created. 
684          This filter function checks if the node is marked by current stage and not annotated by the previous stage. 
685          """
686          for module_name, quantization_config in self.module_name_qconfig.items():
687              self._annotate_with_config(
688                  model, quantization_config, _create_module_name_filter(module_name)
689              )
690  
691          for operator_type, quantization_config in self.operator_type_qconfig.items():
692              self._annotate_with_config(
693                  model, quantization_config, _create_operator_type_filter(operator_type)
694              )
695  
696          if self.global_config:
697              self._annotate_with_config(
698                  model,
699                  self.global_config,
700                  _global_config_filter,
701              )
702  
703          # Once we've annotated the model with quantization configurations, we also need to annotate
704          # the output of quantizable operations. For example, if we annotated `maxpool2d` to quantize its inputs,
705          # we will quantize its output accordingly. This enables us to fuse the dq-operator-q into a quantized op.
706          # Refer to
707          # https://github.com/intel/intel-extension-for-pytorch/blob/90d19323d96afc53fcc22ba5a7bb3fb07fdd6c1c/intel_extension_for_pytorch/quantization/_recipe.py#L487  # noqa: B950
708  
709          self._annotate_output_for_int8_in_int8_out_pattern_entry(model)
710  
711          return model

713      def _annotate_with_config(
714          self,
715          model: torch.fx.GraphModule,
716          quantization_config: Optional[QuantizationConfig],
717          filter_fn: FilterFn,
718      ) -> None:
719          """Annotate the model with the given quantization configuration. 
720   
721          High-level description of quantization recipe for X86 Inductor Backend: 
722          Step 1: Apply quantization recipe for fusion patterns of conv/linear to enable int8 data type actively. 
723          Step 2: Propagate quantization annotation for patterns besides conv/linear. Go through the pattern in model 
724          from start to the end. If a pattern supports computation with int8 data type and inputs connected to 
725          quantized patterns, annotate its inputs as quantized pattern. 
726          """
727  
728          # Step1: Recipe of fusion patterns like conv/linear.
729          self._annotate_conv2d_fusion_pattern(model, quantization_config, filter_fn)
730          self._annotate_linear_fusion_pattern(model, quantization_config, filter_fn)
731          self._annotate_matmul(model, quantization_config, filter_fn)
732  
733          # Step2: Recipe to propagate annotation for patterns beside conv/linear.
734          # Go through all the nodes from start to end.
735          # Recipe refer to
736          # https://github.com/intel/intel-extension-for-pytorch/blob/90d19323d96afc53fcc22ba5a7bb3fb07fdd6c1c/intel_extension_for_pytorch/quantization/_recipe.py#L538  # noqa: B950
737  
738          self._annotate_propagation_quantizable_pattern_entry(
739              model, quantization_config, filter_fn
740          )

763          for fused_partition in fused_partitions:
764              (
765                  conv_partition,
766                  bn_partition,

769              ) = fused_partition

771              (
772                  conv_node,
773                  bn_output_node,
774                  binary_node,
775                  unary_node,
776              ) = self._get_output_nodes_of_partitions(
777                  [conv_partition, bn_partition, binary_partition, unary_partition]

780                  # Conv BN pattern should only has 1 user.
781                  continue

783                  bn_output_node_idx,
784                  extra_input_node_idx,
785              ) = self._get_input_idx_for_binary_node(bn_output_node, binary_node)
786              if (bn_output_node_idx is None) or (extra_input_node_idx is None):
787                  continue

789                  raise ValueError(f"{bn_output_node} doesn't match input of binary node")
790              extra_input_node = binary_node.args[extra_input_node_idx]
791  
792              if (
793                  conv_node.op != "call_function"
794                  or conv_node.target != torch.ops.aten.conv2d.default
795              ):
796                  continue

799                  [unary_node, binary_node, bn_output_node, conv_node], filter_fn

801                  continue

805              if quantization_config is not None:

808                      quantization_config
809                  )
810                  binary_node.meta[QUANT_ANNOTATION_KEY] = (
811                      _X86InductorQuantizationAnnotation(
812                          input_qspec_map=binary_node_input_qspec_map,
813                          _annotated=True,
814                      )
815                  )
816                  unary_node.meta[QUANT_ANNOTATION_KEY] = (
817                      _X86InductorQuantizationAnnotation(
818                          # TODO<leslie> Remove the annotate of output in QAT when qat util support pattern matcher.
819                          output_qspec=get_output_act_qspec(quantization_config),  # type: ignore[arg-type]
820                          _annotated=True,
821                          _is_output_of_quantized_pattern=True,
822                      )
823                  )
824              else:
825                  _annotate_nodes_not_quantize([binary_node, unary_node])
826              nodes_to_mark_annotated = list(conv_partition.nodes)
827              nodes_to_mark_annotated.extend(list(bn_partition.nodes))
828              nodes_to_mark_annotated.extend(list(binary_partition.nodes))
829              nodes_to_mark_annotated.extend(list(unary_partition.nodes))

1340                         input_node,

1488         self,
1489         gm: torch.fx.GraphModule,
1490         quantization_config: Optional[QuantizationConfig],
1491         filter_fn: Optional[FilterFn] = None,
1492     ) -> None:
1493         # linear + binary_op + (optional) unary op
1494         binary_op_list = [operator.add]
1495         unary_op_list = [torch.nn.ReLU, None]
1496         combinations = itertools.product(binary_op_list, unary_op_list)
1497         for binary_op, unary_op in combinations:
1498             has_unary = unary_op is not None
1499             seq_partition = [torch.nn.Linear, binary_op]
1500             if has_unary:
1501                 seq_partition.append(unary_op)
1502             fused_partitions = find_sequential_partitions(gm, seq_partition)
1503             for fused_partition in fused_partitions:
1504                 unary_partition, unary_node = None, None
1505                 if has_unary:
1506                     (
1507                         linear_partition,
1508                         binary_partition,
1509                         unary_partition,
1510                     ) = fused_partition
1511                     (
1512                         linear_node,
1513                         binary_node,
1514                         unary_node,
1515                     ) = self._get_output_nodes_of_partitions(
1516                         [linear_partition, binary_partition, unary_partition]
1517                     )
1518                 else:
1519                     linear_partition, binary_partition = fused_partition
1520                     linear_node, binary_node = self._get_output_nodes_of_partitions(
1521                         [linear_partition, binary_partition]
1522                     )
1523                 if len(linear_node.users) != 1:
1524                     # Linear Node should only has 1 user node
1525                     continue
1526                 (
1527                     linear_node_idx,
1528                     extra_input_node_idx,
1529                 ) = self._get_input_idx_for_binary_node(linear_node, binary_node)
1530                 if (linear_node_idx is None) or (extra_input_node_idx is None):
1531                     continue
1532                 if linear_node != binary_node.args[linear_node_idx]:
1533                     raise ValueError(
1534                         f"{linear_node} doesn't match input of binary node"
1535                     )

1538                     linear_node.op != "call_function"
1539                     or linear_node.target != torch.ops.aten.linear.default
1540                 ):

1544                     [binary_node, linear_node]
