3    """ 
4    This module provides debugging backends for TorchDynamo to help diagnose and troubleshoot 
5    compilation and execution issues. It includes: 
6     
7    Key Debugging Backends: 
8    - eager: Simple pass-through backend that runs models in eager mode 
9    - eager_noexcept: Similar to eager but with additional exception handling 
10   - eager_debug: Adds schema validation checks for custom operators 
11   - aot_eager: Uses AOT Autograd with nop compiler for debugging 
12   - aot_eager_decomp_partition: Uses TorchInductor decompositions for debugging 
13   - torchscript: Compiles using TorchScript for debugging JIT-related issues 
14    
15   Testing and Development Tools: 
16   - Backends for inducing specific errors (compile/runtime/accuracy) 
17   - ExplainOutput class for detailed graph compilation analysis 
18   - Utilities for cross-referencing and mode management 
19   - Tools for graph detail inspection and break reason analysis 
20    
21   These backends are primarily used for: 
22   1. Debugging graph breaks and compilation failures 
23   2. Testing error handling and recovery mechanisms 
24   3. Analyzing performance bottlenecks 
25   4. Validating operator schemas and decompositions 
26   """

58   def make_eager_backend_with_torch_function_modes(modes):
59       """Used to trace HOPs (cond and while) for eager execution, the metadata 
60       TF mode mutates vars outside of the scope of the HOP, and we can't have graph breaks 
61       in the HOP, so we need to externally run this mode and not trace it."""
62       from contextlib import ExitStack
63   
64       def fn(gm, fake_tensor_inputs, **kwargs):
65           stack = ExitStack()
66           for mode in modes:
67               stack.enter_context(mode)
68   
69           result = gm.forward
70           stack.close()
71           return result
72   
73       return fn

76   @register_backend
77   def eager_noexcept(gm, fake_tensor_inputs, **kwargs):
78       if kwargs:
79           log.warning("eager_noexcept backend ignoring extra kwargs %s", kwargs)
80   
81       # This backend is intended to check that dynamo-generated GraphModules
82       # do not cause errors.
83       def inner(*args):
84           try:
85               return gm(*args)
86           except Exception as e:
87               raise torch._dynamo.exc.TorchDynamoException(
88                   "Unexpected exception when running generated GraphModule"
89               ) from e
90   
91       return inner

94   @register_backend
95   def pre_dispatch_eager(gm, fake_tensor_inputs, **kwargs):
96       if kwargs:
97           log.warning("pre_dispatch_eager backend ignoring extra kwargs %s", kwargs)
98   
99       from torch.fx.experimental.proxy_tensor import make_fx
100  
101      def runnable_gm(*args):
102          return torch.fx.Interpreter(gm).run(*args)
103  
104      pre_dispatch_gm = make_fx(runnable_gm, pre_dispatch=True)(*fake_tensor_inputs)
105      pre_dispatch_gm.print_readable()
106  
107      return pre_dispatch_gm

114  
115      from torch._subclasses.schema_check_mode import SchemaCheckMode

132  # used boxed call to discard inputs when they are no longer needed
133  def boxed_nop(fx_g, example_inputs):
134      def run(args):
135          return torch.fx.Interpreter(fx_g).boxed_run(args)
136  
137      run._boxed_call = True
138      return run

141  def boxed_nop_with_mode(fx_g, example_inputs, *, mode):
142      def run(args):
143          with mode:
144              return torch.fx.Interpreter(fx_g).boxed_run(args)
145  
146      run._boxed_call = True
147      return run

159  def ignore_builtins(op: torch._ops.OpOverload) -> bool:
160      return op.namespace in ("aten", "prims", "prim")
161  
162  
163  def get_nop_func():
164      if not torch._functorch.config.fake_tensor_crossref:
165          return boxed_nop
166      elif torch._functorch.config.fake_tensor_crossref == "all":
167          return fake_crossref_boxed_nop
168      else:
169          assert torch._functorch.config.fake_tensor_crossref == "custom_ops"
170          return functools.partial(fake_crossref_boxed_nop, ignore_op_fn=ignore_builtins)

174  # aot_eager uses AOT Autograd backend with nop compiler. It is helpful in debugging.
175  def aot_eager(

177      fake_tensor_inputs,
178      fw_compiler=None,

200  # Uses TorchInductor AOT Autograd decomps and partitioner to isolate aot vs
201  # inductor problems.
202  # aot_eager_decomp_partition just replaces the inductor compiler with nop to help
203  # isolate inductor vs aot_eager errors
204  def aot_eager_decomp_partition(gm, fake_tensor_inputs, **kwargs):
205      if kwargs:
206          log.warning(
207              "aot_eager_decomp_partition backend ignoring extra kwargs %s", kwargs
208          )
209  
210      from torch._inductor.compiler_bisector import CompilerBisector
211  
212      config_patches = {"unlift_effect_tokens": True}
213      if bisect_changes := CompilerBisector.get_config_change(
214          "aot_eager_decomp_partition"
215      ):
216          config_patches.update(bisect_changes)
217  
218      with functorch_config.patch(config_patches):
219          return aot_autograd(
220              # these are taken from memory_efficient_fusion()
221              fw_compiler=get_nop_func(),
222              bw_compiler=get_nop_func(),
223              # NB: lambda here is to delay import of inductor
224              decompositions=lambda: import_module(
225                  "torch._inductor.compile_fx"
226              ).select_decomp_table(),
227              partition_fn=functools.partial(
228                  min_cut_rematerialization_partition, compiler="inductor"
229              ),
230          )(gm, fake_tensor_inputs)
231  
232  
233  register_backend(
234      name="aot_eager_decomp_partition", compiler_fn=aot_eager_decomp_partition
235  )

261  def aot_eager_decomp_partition_crossref(gm, fake_tensor_inputs, **kwargs):
262      # if the config is set, respect it, otherwise only test custom_ops.
263      # custom_op bad metas always manifest as an error whereas aten will only sometimes.
264      # by default, use the less noisy option

266          "custom_ops"
267          if not functorch_config.fake_tensor_crossref
268          else functorch_config.fake_tensor_crossref
269      )
270      with functorch_config.patch(fake_tensor_crossref=config_val):
271          return aot_eager_decomp_partition(gm, fake_tensor_inputs, **kwargs)

327  @register_backend
328  def non_leaf_compile_error_TESTING_ONLY(gm: torch.fx.GraphModule, example_inputs):
329      # Require at least one non-trivial thing in the graph,
330      # see https://github.com/pytorch/pytorch/issues/102898
331      for node in gm.graph.nodes:
332          if node.op == "call_function":
333              break
334      else:
335          return gm
336      for t in example_inputs:
337          if not t.is_leaf:
338              raise TestingOnlyCompileError
339      return gm

342  @dataclasses.dataclass
343  class ExplainOutput:
344      """ 
345      This is the output of :func:`torch._dynamo.explain()` 
346      There is no reason to create this class directly. 
347      """
348  
349      graphs: list[torch.fx.GraphModule]
350      graph_count: int
351      graph_break_count: int
352      break_reasons: list[
353          Any
354      ]  # Type is GraphCompileReason but doesn't matter for this purpose
355      op_count: int
356      ops_per_graph: Optional[list[torch.fx.Node]] = None
357      out_guards: Optional[list[_guards.Guard]] = None
358      compile_times: Optional[str] = None
359  
360      def __str__(self) -> str:
361          output = f"Graph Count: {self.graph_count}\n"
362          output += f"Graph Break Count: {self.graph_break_count}\n"
363          output += f"Op Count: {self.op_count}\n"
364  
365          output += "Break Reasons:\n"
366          for idx, break_reason in enumerate(self.break_reasons):
367              output += f"  Break Reason {idx + 1}:\n"
368              output += f"    Reason: {break_reason.reason}\n"
369              output += "    User Stack:\n"
370              for frame_summary in break_reason.user_stack:
371                  output += f"      {frame_summary}\n"
372  
373          if self.ops_per_graph is not None:
374              output += "Ops per Graph:\n"
375              for idx, ops in enumerate(self.ops_per_graph):
376                  output += f"  Ops {idx + 1}:\n"
377                  for op in ops:
378                      output += f"    {op}\n"
379  
380          if self.out_guards is not None:
381              output += "Out Guards:\n"
382              for i, guard in enumerate(self.out_guards):
383                  output += f"  Guard {i + 1}:\n"
384                  output += f"    {str(guard)}"
385  
386          if self.compile_times is not None:
387              output += f"Compile Times: {self.compile_times}\n"
388          return output

391  def _explain_graph_detail(
392      gm: torch.fx.GraphModule, graphs, op_count, ops_per_graph, break_reasons
393  ):
394      """ 
395      This function is a utility which processes a torch.fx.GraphModule and 
396      accumulates information about its ops, graph breaks, and other details. It 
397      is intended to be used by the ExplainWithBackend class and 
398      `torch._dynamo.explain()` to provide details from Dynamo's graph capture. 
399   
400      Parameters: 
401          gm (torch.fx.GraphModule): The GraphModule to be processed. 
402          graphs (list): A list that accumulates all the GraphModules processed. 
403          op_count (int): The total count of operations in all GraphModules processed so far. 
404          ops_per_graph (list): A list that accumulates the operations of each GraphModule. 
405          break_reasons (list): A list that accumulates the reasons for breaks in each GraphModule. 
406   
407      Returns: 
408          tuple: A tuple containing the processed GraphModule, the updated lists of graphs, 
409                 operations per graph, and break reasons, and the updated operation count. 
410      """
411      graphs.append(gm)
412      ops = [node.target for node in gm.graph.nodes if node.op == "call_function"]
413      op_count += len(ops)
414      ops_per_graph.append(ops)
415      if gm.compile_subgraph_reason.graph_break:
416          break_reasons.append(gm.compile_subgraph_reason)
417  
418      return gm, graphs, op_count, ops_per_graph, break_reasons

421  class ExplainWithBackend:
422      """ 
423      This class is intended to be used as a backend for `torch.compile`. It is 
424      composable with other backends. When used in this way, it accumulates 
425      information about graph breaks, ops, and other info and provides a string 
426      representation summarizing this information. 
427   
428      Attributes: 
429          backend (str): The name of the backend to use for optimization. 
430          graphs (list): A list of the graphs captured by TorchDynamo. 
431          op_count (int): The total number of operations in all optimized graphs. 
432          break_reasons (list): A list of graph break reasons with stack traces. 
433   
434      Example Usage: 
435          def fn(x): 
436              x = torch.sigmoid(x) 
437              return x 
438   
439          torch._dynamo.reset() 
440          eb = ExplainWithBackend("inductor") 
441          optimized_fn = torch.compile(fn, backend=eb) 
442          result = optimized_fn(torch.randn(5)) 
443          print(eb.output()) 
444      """
445  
446      def __init__(self, backend) -> None:
447          from .registry import lookup_backend
448  
449          self.backend = lookup_backend(backend)
450          self.graphs = []
451          self.op_count = 0
452          self.break_reasons = []
453  
454      def __call__(self, gm: torch.fx.GraphModule, example_inputs):
455          gm, self.graphs, self.op_count, _, self.break_reasons = _explain_graph_detail(
456              gm, self.graphs, self.op_count, [], self.break_reasons
457          )
458          return self.backend(gm, example_inputs)
459  
460      def output(self) -> ExplainOutput:
461          graph_count = len(self.graphs)
462          output = ExplainOutput(
463              self.graphs,
464              graph_count,
465              graph_count - 1,
466              self.break_reasons,
467              self.op_count,
468          )
469  
470          return output
