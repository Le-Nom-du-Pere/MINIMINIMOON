1    #pragma once
2    #include <ATen/core/Tensor.h>
3    #include <c10/util/irange.h>
4    
5    namespace at::native {
6    // input tensors are non-zero dim and non-empty
7    template <typename T1, typename T2, typename Function>
8    
9    void tensor_dim_apply3(
10       const Tensor& self,
11       Tensor& values,
12       Tensor& indices,
13       int64_t dim,
14       Function func) {
15     int ndims = self.dim();
16     int tensor_dim_apply_has_finished = 0;
17     std::vector<int64_t> counter(ndims, 0);
18     const T1* self_data = self.const_data_ptr<T1>();
19     T1* values_data = values.data_ptr<T1>();
20     T2* indices_data = indices.data_ptr<T2>();
21     int64_t self_stride = self.stride(dim);
22     int64_t values_stride = values.stride(dim);
23     int64_t indices_stride = indices.stride(dim);
24     int self_dim_size = self.size(dim);
25   
26     while (!tensor_dim_apply_has_finished) {
27       func(
28           self_data,
29           values_data,
30           indices_data,
31           self_dim_size,
32           self_stride,
33           values_stride,
34           indices_stride);
35       if (ndims == 1) {
36         break;
37       }
38       for (const auto dim_i : c10::irange(ndims)) {
39         if (dim_i == dim) {
40           if (dim_i == (ndims - 1)) {
41             tensor_dim_apply_has_finished = 1;
42             break;
43           }
44           continue;
45         }
46         counter[dim_i]++;
47         self_data += self.stride(dim_i);
48         values_data += values.stride(dim_i);
49         indices_data += indices.stride(dim_i);
50   
51         if (counter[dim_i] == self.size(dim_i)) {
52           if (dim_i == ndims - 1) {
53             tensor_dim_apply_has_finished = 1;
54             break;
55           } else {
56             self_data -= counter[dim_i] * self.stride(dim_i);
57             values_data -= counter[dim_i] * values.stride(dim_i);
58             indices_data -= counter[dim_i] * indices.stride(dim_i);
59             counter[dim_i] = 0;
60           }
61         } else {
62           break;
63         }
64       }
65     }
66   }
67   } // namespace at::native
