9    def _get_foreach_kernels_supported_devices() -> list[str]:
10       r"""Return the device type list that supports foreach kernels."""
11       return ["cuda", "xpu", torch._C._get_privateuse1_backend_name()]

31   # This util function splits tensors into groups by device and dtype, which is useful before sending
32   # tensors off to a foreach implementation, which requires tensors to be on one device and dtype.
33   # If tensorlistlist contains more than one tensorlist, the following assumptions are made BUT NOT verified:
34   #   - tensorlists CAN be None
35   #   - all tensors in the first specified list cannot be None
36   #   - given an index i, all specified tensorlist[i]s match in dtype and device
37   # with_indices (bool, optional): whether to track previous indices as the last list per dictionary entry.
38   #   It comes in handy if there are Nones or literals in the tensorlists that are getting scattered out.
39   #   Whereas mutating a tensor in the resulting split-up tensorlists WILL propagate changes back to the
40   #   original input tensorlists, changing up Nones/literals WILL NOT propagate, and manual propagation
41   #   may be necessary. Check out torch/optim/sgd.py for an example.
42   @no_grad()
43   def _group_tensors_by_device_and_dtype(
44       tensorlistlist: TensorListList,
45       with_indices: bool = False,
46   ) -> dict[tuple[torch.device, torch.dtype], tuple[TensorListList, Indices]]:
47       return torch._C._group_tensors_by_device_and_dtype(tensorlistlist, with_indices)
48   
49   
50   def _device_has_foreach_support(device: torch.device) -> bool:
51       return (
52           device.type in (_get_foreach_kernels_supported_devices() + ["cpu"])
53           and not torch.jit.is_scripting()
54       )
55   
56   
57   def _has_foreach_support(tensors: list[Tensor], device: torch.device) -> bool:
58       return _device_has_foreach_support(device) and all(
59           t is None or type(t) in _foreach_supported_types for t in tensors
60       )
