18   template <typename Iter, typename Scalar>
19   __device__ Iter lower_bound(Iter start, Iter end, Scalar value) {
20     while (start < end) {
21       auto mid = start + ((end - start) >> 1);
22       if (*mid < value) {
23         start = mid + 1;
24       } else {
25         end = mid;
26       }
27     }
28     return end;
29   }
