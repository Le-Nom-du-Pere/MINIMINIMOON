18   class Functionalize(_pass.Transform):
19       """Functionalize a GraphModule. 
20    
21       This pass utilizes ``functionalization`` utility of ``torch._functorch`` to convert 
22       a GraphModule into a functional form. The two main functionalities are (copied from 
23       its documentations): 
24    
25       * ``functionalization`` removes (intermediate) mutations and aliasing from a 
26       function, while preserving the function's semantics. 
27    
28       * ``functionalization`` also removes mutations (and views) that were performed 
29       on function inputs. However to preserve semantics, functionalize will "fix up" the 
30       mutations after the transform has finished running, by detecting if any tensor inputs 
31       "should have" been mutated, and copying the new data back to the inputs if necessary. 
32       For example, consider:: 
33    
34           def fn(a, b): 
35               a.add_(b) 
36               return a 
37    
38         For a call like `fn(x, y)`, the variable `x` outside is also mutated. Hence just 
39         functionalizing is not enough for preserving the original semantics. A "special" 
40         input mutation step needs to be inserted at the end.:: 
41    
42           # After functionalization, without input mutation "fix up". 
43           # This is not semantically the same. The variable outside the function call that 
44           # was passed in as `a` is not mutated. 
45           def fn(a, b): 
46               new_a = a + b 
47               return new_a 
48    
49           # Functionalization with input mutation "fix up" that preserves semantics. 
50           def fn(a, b): 
51               new_a = a + b 
52    
53               # Copying the new data back to the inputs 
54               a.copy_(new_a) 
55    
56               return new_a 
57    
58       For ONNX inference, it is recommended to run ``RemoveInputMutation`` after this pass. 
59       ``RemoveInputMutation`` removes the "fix up" nodes that were added by ``Functionalize``, 
60       which are not needed for ONNX inference. 
61       """
62   
63       def __init__(
64           self,
65           module: torch.fx.GraphModule,
66           enable_dynamic_axes: bool,
67           allow_fake_constant: bool | None = False,
68       ):
69           super().__init__(module)
70           self.enable_dynamic_axes = enable_dynamic_axes
71           self.allow_fake_constant = allow_fake_constant
72   
73       def _functionalize(self, function: Callable) -> Callable:
74           # Working around a dispatcher issue with `torch.func.functionalize` when used
75           # together with `make_fx`.
76           # Ref: https://github.com/pytorch/pytorch/issues/99774#issuecomment-1527949391
77           def wrapped(*inputs):
78               inputs_functional = pytree.tree_map_only(
79                   torch.Tensor, torch._to_functional_tensor, inputs
80               )
81               torch._enable_functionalization(reapply_views=True)
82               try:
83                   out = function(*inputs_functional)
84               finally:
85                   torch._disable_functionalization()
86   
87               flat_inputs_functional = pytree.tree_leaves(inputs_functional)
88               for input_functional in flat_inputs_functional:
89                   if isinstance(input_functional, torch.Tensor):
90                       torch._sync(input_functional)
91               pytree.tree_map(torch._sync, out)
92               out_unwrapped = pytree.tree_map(torch._from_functional_tensor, out)
93               return out_unwrapped
94   
95           return wrapped
96   
97       def _run(self, *args) -> torch.fx.GraphModule:
98           # To preserve stack trace info after `make_fx`.
99           module = _utils.wrap_graph_module_for_node_meta_preservation(self.module)
100  
101          functionalized_callable = self._functionalize(module)
102  
103          # Mimic `torch._dynamo.export(aten_graph=True)` behavior in invoking `make_fx`.
104          # TODO: May need revisit for user fake mode export + dynamic shape scenario.
105          fake_mode: fake_tensor.FakeTensorMode | None = self.fake_mode
106          maybe_fake_args = self._maybe_fakefy_args(fake_mode, *args)
107          if fake_mode is not None:
108              # Using existing fake mode as context, signal `make_fx` that it does not need
109              # to create a new fake mode by passing tracing_mode as "real".
110              tracing_mode = "real"
111          else:
112              # Existing fake mode not found, signal `make_fx` to create one.
113              fake_mode = contextlib.nullcontext()  # type: ignore[assignment]
114              tracing_mode = "symbolic" if self.enable_dynamic_axes else "fake"
115  
116          assert fake_mode is not None  # for mypy
117          with fake_tensor.unset_fake_temporarily(), fake_mode:
118              graph_module = proxy_tensor.make_fx(
119                  functionalized_callable,
120                  decomposition_table={},
121                  tracing_mode=tracing_mode,
122                  _allow_non_fake_inputs=True,
123                  _allow_fake_constant=bool(self.allow_fake_constant),
124              )(*maybe_fake_args)
125  
126          # Rename placeholder targets to match the original module's signature since
127          # We don't want to map forward(x, y, z) to forward(arg0, arg1, arg2).
128          _utils.replace_placeholder_name_and_target(graph_module, self.module)
129  
130          return graph_module

133  class RemoveInputMutation(_pass.Transform):
134      """Remove `aten.copy_.default` nodes that mutate module inputs. 
135   
136      This pass is recommended to be used after ``Functionalization`` pass. 
137      ``Functionalization`` pass adds `aten.copy_.default` nodes to the graph 
138      when it detects mutations to inputs. These nodes are not needed for ONNX export 
139      for inference. They could be useful for training. 
140      """
141  
142      def _run(self, *args) -> torch.fx.GraphModule:
143          for node in reversed(self.module.graph.nodes):
144              if (
145                  node.op == "call_function"
146                  and node.target == torch.ops.aten.copy_.default
147                  and len(node.users) == 0
148                  and isinstance(node.args[0], torch.fx.Node)
149                  and node.args[0].op == "placeholder"
150              ):
151                  self.module.graph.erase_node(node)
152          return self.module
