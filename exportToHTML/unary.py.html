111  def _unary_helper(fn, args, kwargs, inplace):
112      if len(kwargs) != 0:
113          raise ValueError(
114              "MaskedTensor unary ops require that len(kwargs) == 0. "
115              "If you need support for this, please open an issue on Github."
116          )
117      for a in args[1:]:
118          if torch.is_tensor(a):
119              raise TypeError(
120                  "MaskedTensor unary ops do not support additional Tensor arguments"
121              )
122  
123      mask_args, _mask_kwargs = _map_mt_args_kwargs(
124          args, kwargs, lambda x: x._masked_mask
125      )
126      data_args, _data_kwargs = _map_mt_args_kwargs(
127          args, kwargs, lambda x: x._masked_data
128      )
129  
130      if args[0].layout == torch.sparse_coo:
131          data_args[0] = data_args[0].coalesce()
132          s = data_args[0].size()
133          i = data_args[0].indices()
134          data_args[0] = data_args[0].coalesce().values()
135          v = fn(*data_args)
136          result_data = torch.sparse_coo_tensor(i, v, size=s)
137  
138      elif args[0].layout == torch.sparse_csr:
139          crow = data_args[0].crow_indices()
140          col = data_args[0].col_indices()
141          data_args[0] = data_args[0].values()
142          v = fn(*data_args)
143          result_data = torch.sparse_csr_tensor(crow, col, v)
144  
145      else:
146          result_data = fn(*data_args)
147  
148      if inplace:
149          args[0]._set_data_mask(result_data, mask_args[0])
150          return args[0]
151      else:
152          return _wrap_result(result_data, mask_args[0])

155  def _torch_unary(fn_name):
156      fn = getattr(torch.ops.aten, fn_name)
157  
158      def unary_fn(*args, **kwargs):
159          return _unary_helper(fn, args, kwargs, inplace=False)
160  
161      return unary_fn
162  
163  
164  def _torch_inplace_unary(fn_name):
165      fn = getattr(torch.ops.aten, fn_name)
166  
167      def unary_fn(*args, **kwargs):
168          return _unary_helper(fn, args, kwargs, inplace=True)
169  
170      return unary_fn

173  NATIVE_UNARY_MAP = {
174      getattr(torch.ops.aten, name): _torch_unary(name) for name in UNARY_NAMES
175  }
176  NATIVE_INPLACE_UNARY_MAP = {
177      getattr(torch.ops.aten, name): _torch_inplace_unary(name)
178      for name in INPLACE_UNARY_NAMES
179  }
180  
181  NATIVE_UNARY_FNS = list(NATIVE_UNARY_MAP.keys())
182  NATIVE_INPLACE_UNARY_FNS = list(NATIVE_INPLACE_UNARY_MAP.keys())
183  
184  
185  def _is_native_unary(fn):
186      return fn in NATIVE_UNARY_FNS or fn in NATIVE_INPLACE_UNARY_FNS
187  
188  
189  def _apply_native_unary(fn, *args, **kwargs):
190      if fn in NATIVE_UNARY_FNS:
191          return NATIVE_UNARY_MAP[fn](*args, **kwargs)
192      if fn in NATIVE_INPLACE_UNARY_FNS:
193          return NATIVE_INPLACE_UNARY_MAP[fn](*args, **kwargs)
194      return NotImplemented
