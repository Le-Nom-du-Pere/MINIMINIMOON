38   class SparseSemiStructuredTensor(torch.Tensor):
39       """ 
40       This class implements semi-structured sparsity as a Tensor subclass. 
41    
42       Semi-structured sparsity describes a sparsity pattern where n in every 2n elements are sparse, 
43       depending on the datatype. It is also referred to as 2:4 sparsity or fine-grained 
44       structured sparsity. 
45    
46       There are two backends available for semi_structred sparsity, either cuSPARSELt or CUTLASS. 
47       This class is meant to serve as a base class for both implementations. SparseSemiStructuredCUTLASS 
48       and SparseSemiStructuredCUSPARSELT both inherit from this class and define three backend-specific items. 
49       Note that as such, this class cannot be instantiated directly. 
50    
51       -`_DTYPE_SHAPE_CONSTRAINTS` - A dictionary holding backend specific dense/sparse min shape constraints 
52       - `def from_dense()` - backend specific compression routines 
53       - `def _mm()` - backend specific mm op (either torch._cslt_sparse_mm or torch._sparse_semi_structured_(mm|addmm)) 
54       """
55   
56       _DEFAULT_ALG_ID: int = 0
57       _DTYPE_SHAPE_CONSTRAINTS: dict[torch.dtype, _SEMI_STRUCTURED_SPARSE_CONFIG]
58       _FORCE_CUTLASS: bool = False
59       _FUSE_TRANSPOSE: bool = False
60       _PROTOTYPE_WARNING_SHOWN: bool = False
61   
62       BACKEND: str
63       SPARSE_DISPATCH: dict[Callable, Callable]
64   
65       packed: Optional[torch.Tensor]
66       meta: Optional[torch.Tensor]
67       packed_t: Optional[torch.Tensor]
68       meta_t: Optional[torch.Tensor]
69       compressed_swizzled_bitmask: Optional[torch.Tensor]
70       fuse_transpose_cusparselt: bool
71       alg_id_cusparselt: int
72   
73       __slots__ = ["packed", "meta", "packed_t", "meta_t", "compressed_swizzled_bitmask"]
74   
75       @staticmethod
76       def __new__(  # noqa: PYI034
77           cls,
78           shape: torch.Size,
79           packed: Optional[torch.Tensor],
80           meta: Optional[torch.Tensor],
81           packed_t: Optional[torch.Tensor],
82           meta_t: Optional[torch.Tensor],
83           compressed_swizzled_bitmask: Optional[torch.Tensor],
84           fuse_transpose_cusparselt: bool = False,
85           alg_id_cusparselt: int = 0,
86           requires_grad: bool = False,
87       ):
88           """ 
89           Create a new instance of the tensor subclass from the compressed sparse representation. 
90    
91           We have the option to create the subclass with the compressed representations of both X and X', for training. 
92           For inference, we only need a single representation (either X or X'), while the corresponding other set will be None. 
93    
94           Depending on the backend selected, certain fields will be set to None. (CUSPARSELT vs CUTLASS) 
95    
96           Args: 
97               shape: The shape of the original dense tensor 
98               packed: The compressed representation of the original dense tensor 
99               meta: The metadata of the original dense tensor, if it is stored separately 
100              packed_t: The compressed representation of the transposed original dense tensor 
101              meta_t: The metadata of the transposed original dense tensor, if it is stored separately 
102              compressed_swizzled_bitmask: The masks used by the CUTLASS backend to determine which threads should 
103                                           participate in the computation. Used for pointwise ops. 
104              fuse_transpose_cusparselt: When running with cuSPARSELt, we have the option to fuse a transposition 
105                                         with a matmul, which is useful in the case of 2:4 sparse training. 
106              alg_id_cusparselt: The algorithm id to use when using cuSPARSELT, will have effect on performance 
107   
108          Returns: 
109              torch.Tensor: A torch.Tensor wrapper subclass. 
110   
111          Raises: 
112              ValueError: If all of the tensor arguments are None. 
113          """
114          if not cls._PROTOTYPE_WARNING_SHOWN:
115              warnings.warn(
116                  (
117                      "The PyTorch API of SparseSemiStructuredTensor is in prototype stage "
118                      "and will change in the near future. Please open a Github issue "
119                      "for features requests and see our documentation on the torch.sparse "
120                      "module for further information about the project."
121                  ),
122                  UserWarning,
123              )
124              cls._PROTOTYPE_WARNING_SHOWN = True
125  
126              # Because this only runs once, we also load the dispatch table here as well.
127              # We can't define the dispatch table explicitly because of torch.ops import errors, so we do this instead
128              # But this is useful since it allows users to overload the dispatch table for debugging / testing.
129              cls._load_dispatch_table()
130  
131              # we can also register the classes with dynamo when the warning is shown.
132              torch._dynamo.allow_in_graph(cls)
133  
134          if packed is not None:
135              previous_tensor = packed
136          elif packed_t is not None:
137              previous_tensor = packed_t
138          else:
139              raise ValueError("At least one of packed or packed_t must be provided")
140  
141          tensor = torch.Tensor._make_wrapper_subclass(
142              cls,
143              shape,
144              device=previous_tensor.device,
145              dtype=previous_tensor.dtype,
146              layout=previous_tensor.layout,
147              requires_grad=requires_grad,
148          )
149  
150          tensor.packed = packed
151          tensor.meta = meta
152          tensor.packed_t = packed_t
153          tensor.meta_t = meta_t
154          tensor.compressed_swizzled_bitmask = compressed_swizzled_bitmask
155          tensor.fuse_transpose_cusparselt = fuse_transpose_cusparselt
156          tensor.alg_id_cusparselt = alg_id_cusparselt
157          return tensor
158

160          assert hasattr(self, "shape")
161          return f"{self.__class__.__name__}(shape={self.shape})"

235      @classmethod
236      def _validate_device_dim_dtype_shape(cls, original_tensor: torch.Tensor) -> None:
237          """ 
238          Assert that the given tensor is valid for semi-structured sparse compression. 
239          """
240          # check device
241          if not original_tensor.is_cuda:
242              raise RuntimeError(
243                  f"Error original_tensor.device= {original_tensor.device} is not supported! "
244                  "Only CUDA tensors are currently supported."
245              )
246  
247          # check dim
248          if original_tensor.dim() != 2:
249              raise RuntimeError(
250                  f"Error original_tensor.dim = {original_tensor.dim()} is not supported! "
251                  "Only 2d tensors are currently supported."
252              )
253  
254          # check contiguous
255          if not original_tensor.is_contiguous():
256              raise RuntimeError(
257                  "Error original_tensor is not contiguous!"
258                  "Only contiguous tensors are currently supported."
259              )
260  
261          # check dtype
262          if original_tensor.dtype not in cls._DTYPE_SHAPE_CONSTRAINTS:
263              raise RuntimeError(
264                  f"Error original_tensor.dtype {original_tensor.dtype} is not a supported dtype for {cls}!"
265              )
266  
267          # check shape
268          m, n = original_tensor.shape
269          min_rows = cls._DTYPE_SHAPE_CONSTRAINTS[original_tensor.dtype].sparse_min_rows
270          min_cols = cls._DTYPE_SHAPE_CONSTRAINTS[original_tensor.dtype].sparse_min_cols
271          if m < min_rows or m % min_rows or n < min_cols or n % min_cols:
272              # TODO in the future we can add in padding to support sparse dimensions that aren't perfect multiples
273              raise RuntimeError(
274                  f"Error original_tensor.shape {original_tensor.shape} is not supported! "
275                  f"Both dimensions must be larger or equal than and a multiple of ({min_rows}, {min_cols})"
276              )

278      @classmethod
279      def _pad_dense_input(cls, dense_input: torch.Tensor) -> torch.Tensor:
280          """ 
281          Calculates padding for dense tensor and pads tensor if necessary. 
282          If padding is not required, this function returns the original tensor. 
283          """
284          # only 2d matmul
285          assert dense_input.dim() == 2
286  
287          # check shape
288          m, n = dense_input.shape
289          min_rows = cls._DTYPE_SHAPE_CONSTRAINTS[dense_input.dtype].dense_min_rows
290          min_cols = cls._DTYPE_SHAPE_CONSTRAINTS[dense_input.dtype].dense_min_cols
291  
292          # calculate padding
293          to_pad_m = -m % min_rows if m < min_rows or m % min_rows else 0
294          to_pad_n = -n % min_cols if n < min_cols or n % min_rows else 0
295          if to_pad_m or to_pad_n:
296              return torch.nn.functional.pad(dense_input, (0, to_pad_n, 0, to_pad_m))
297          else:
298              return dense_input

300      def to_dense(self):  # type:ignore[override]
301          col = self.shape[-1]
302          return torch.mm(self, torch.eye(col, dtype=self.dtype, device=self.device))

318  def to_sparse_semi_structured(
319      original_tensor: torch.Tensor,
320      transposed: bool = False,
321  ) -> SparseSemiStructuredTensor:
322      """ 
323      This function converts a dense tensor into a sparse semi-structured tensor. 
324      It will return a SparseSemiStructuredTensor, a subclass of torch.Tensor. 
325   
326      This function will check to ensure the dense tensor has the right dtype, size, dims, and device. 
327      We currently only support semi-structured sparse tensors for 2d CUDA tensors. 
328      Additionally, your tensor must be a positive multiple of the minimum sparse block size, given in 
329      `_DTYPE_TO_SHAPE_CONSTRAINTS` for each dtype (float32, float16, bfloat16, int8). 
330   
331      Args: 
332          original_tensor (Tensor): the dense tensor to convert 
333          transposed (bool, optional): deprecated arg to be removed in another release. Do not use. 
334      Returns: 
335          SparseSemiStructuredTensor: A sparse semi-structured tensor created from the given original_tensor 
336      Raises: 
337          None 
338      Example: 
339          >>> # xdoctest: +REQUIRES(env:TORCH_DOCTEST_CUDA) 
340          >>> A = torch.Tensor([0, 0, 1, 1]).tile((128, 32)).half().cuda() 
341          tensor([[0., 0., 1.,  ..., 0., 1., 1.], 
342                  [0., 0., 1.,  ..., 0., 1., 1.], 
343                  [0., 0., 1.,  ..., 0., 1., 1.], 
344                  ..., 
345                  [0., 0., 1.,  ..., 0., 1., 1.], 
346                  [0., 0., 1.,  ..., 0., 1., 1.], 
347                  [0., 0., 1.,  ..., 0., 1., 1.]], device='cuda:0', dtype=torch.float16) 
348          >>> A_sparse = to_sparse_semi_structured(A) 
349          SparseSemiStructuredTensor(shape=torch.Size([128, 128])) 
350          >>> A_sparse.values() 
351          tensor([[1., 1., 1.,  ..., 1., 1., 1.], 
352                  [1., 1., 1.,  ..., 1., 1., 1.], 
353                  [1., 1., 1.,  ..., 1., 1., 1.], 
354                  ..., 
355                  [1., 1., 1.,  ..., 1., 1., 1.], 
356                  [1., 1., 1.,  ..., 1., 1., 1.], 
357                  [1., 1., 1.,  ..., 1., 1., 1.]], device='cuda:0', dtype=torch.float16), 
358          >>> A_sparse.indices() 
359          tensor([[-4370, -4370, -4370,  ..., -4370, -4370, -4370], 
360                  [-4370, -4370, -4370,  ..., -4370, -4370, -4370], 
361                  [-4370, -4370, -4370,  ..., -4370, -4370, -4370], 
362                  ..., 
363                  [-4370, -4370, -4370,  ..., -4370, -4370, -4370], 
364                  [-4370, -4370, -4370,  ..., -4370, -4370, -4370], 
365                  [-4370, -4370, -4370,  ..., -4370, -4370, -4370]], device='cuda:0', dtype=torch.int16)) 
366      """
367      if transposed:
368          warnings.warn(
369              "Setting transpose from `to_sparse_semi_structured` is deprecated "
370              "and will be removed in a future release. "
371              "`SparseSemiStructuredTensor` only support contiguous input tensors.",
372              FutureWarning,
373              stacklevel=2,
374          )
375  
376      # set from _FORCE_CUTLASS flag
377      SPARSE_SUBCLASS = (
378          torch.sparse.SparseSemiStructuredTensorCUTLASS
379          if SparseSemiStructuredTensor._FORCE_CUTLASS
380          else torch.sparse.SparseSemiStructuredTensorCUSPARSELT
381      )
382  
383      return SPARSE_SUBCLASS.from_dense(original_tensor)

386  class SparseSemiStructuredTensorCUTLASS(SparseSemiStructuredTensor):
387      """ 
388      This class implements semi-structured sparsity for the CUTLASS backend. 
389   
390   
391      In this implementation, the specified elements and metadata are stored separately, 
392      in packed and meta respectively. 
393   
394      When _FORCE_CUTLASS is set, or when cuSPARSELt is not available, this subclass calls into _sparse_semi_structured_(mm|addmm) and 
395      sparse_semi_structured_from_dense for conversion to the compressed format. 
396      """
397  
398      BACKEND = "cutlass"
399      _DTYPE_SHAPE_CONSTRAINTS = {
400          torch.int8: _SEMI_STRUCTURED_SPARSE_CONFIG(16, 128, 16, 16),
401          torch.float16: _SEMI_STRUCTURED_SPARSE_CONFIG(32, 64, 8, 8),
402          torch.bfloat16: _SEMI_STRUCTURED_SPARSE_CONFIG(32, 64, 8, 8),
403          torch.float32: _SEMI_STRUCTURED_SPARSE_CONFIG(32, 32, 4, 4),
404      }
405  
406      @classmethod
407      def from_dense(
408          cls, original_tensor: torch.Tensor
409      ) -> "SparseSemiStructuredTensorCUTLASS":
410          cls._validate_device_dim_dtype_shape(original_tensor)
411          (
412              sparse_tensor_cutlass,
413              meta_tensor_cutlass,
414          ) = sparse_semi_structured_from_dense_cutlass(original_tensor)
415          return cls(
416              original_tensor.shape,
417              packed=sparse_tensor_cutlass,
418              meta=meta_tensor_cutlass,
419              packed_t=None,
420              meta_t=None,
421              compressed_swizzled_bitmask=None,
422              requires_grad=original_tensor.requires_grad,
423          )
424  
425      def to_dense(self):  # type: ignore[override]
426          assert self.meta is not None and self.packed is not None
427          return (
428              sparse_semi_structured_to_dense_cutlass(
429                  self.packed,
430                  self.meta,
431              )
432              if self.meta.ndim == 2
433              else super().to_dense()
434          )

436      @classmethod
437      def prune_dense_static_sort(
438          cls, original_tensor: torch.Tensor, algorithm=""
439      ) -> "SparseSemiStructuredTensor":
440          """ 
441          This function takes in a unpruned dense tensor and runs a (branchless) static sort across a 4x4 tile. 
442   
443          It greedily picks the largest values in the tile, upholding the 2:4 sparsity constraint across both rows and columns. 
444          The algorithm used to prune the matrix is implemented in `_sparse_semi_structured_tile`. 
445   
446          Then it creates the packed and meta tensors for the compressed sparse representation of the pruned dense tensor. 
447          It also calculates the packed_t and meta_t tensors for the compressed sparse representation of the transposed 
448          pruned dense tensor. 
449          Since we cannot transpose the compressed representations, we store both for the fw/bw pass respectively. 
450   
451          Finally, this function also computes a compressed swizzled bitmask that encodes the sparsity pattern 
452          This can be used in the backward pass to mask the gradients. 
453   
454          [9 1 7 4]                       [9 0 7 0] 
455          [1 2 3 0]                       [0 2 0 0] 
456          [8 3 5 4] -> prune 4x4 tile  -> [8 0 0 4] -> pack to CUTLASS semi-structured -> packed 
457          [1 2 6 2]                       [0 0 6 2]                                    -> metadata 
458   
459                                                    -> pack to transposed CUTLASS      -> packed_t 
460                                                       semi-structured representation  -> metadata_t 
461   
462                                                    -> compute swizzled bitmask        -> compressed_swizzled_bitmask 
463   
464   
465          The equivalent PyTorch code to create the same five outputs from the dense tensor can be found below: 
466          ``` 
467          from torch.sparse import SparseSemiStructuredTensorCUTLASS 
468          from torch.sparse._semi_structured_conversions import _sparse_semi_structured_tile, _compute_compressed_swizzled_bitmask 
469   
470          pruned = _sparse_semi_structured_tile(dense) 
471          packed_cutlass, meta_cutlass = sparse_semi_structured_from_dense_cutlass(pruned) 
472          packed_t_cutlass, meta_t_cutlass = sparse_semi_structured_from_dense_cutlass(pruned.t().contiguous()) 
473          bitmask = _compute_compressed_swizzled_bitmask(pruned) 
474   
475          SparseSemiStructuredTensorCUTLASS(dense.shape, packed_cutlass, meta_cutlass, packed_t_cutlass, meta_t_cutlass, bitmask) 
476          ``` 
477          """
478          # We can either pack to the CUTLASS or cuSPARSELt representation, depending on the use_cutlass flag.
479          (
480              packed,
481              meta,
482              packed_t,
483              meta_t,
484              compressed_swizzled_bitmask,
485          ) = torch._sparse_semi_structured_tile(
486              original_tensor, algorithm=algorithm, use_cutlass=True
487          )
488  
489          return cls(
490              original_tensor.shape,
491              packed=packed,
492              meta=meta,
493              packed_t=packed_t,
494              meta_t=meta_t,
495              compressed_swizzled_bitmask=compressed_swizzled_bitmask,
496              requires_grad=False,
497          )
498  
499      def _mm(
500          self, B: torch.Tensor, *, bias: Optional[torch.Tensor] = None, **kwargs
501      ) -> torch.Tensor:
502          if isinstance(B, SparseSemiStructuredTensor):
503              raise ValueError(
504                  "`SparseSemiStructuredTensor @ SparseSemiStructuredTensor` is not supported by the hardware"
505              )
506          cls_name = self.__class__.__name__
507          if self.ndim != 2 or B.ndim != 2:
508              raise NotImplementedError(
509                  f"`{cls_name}` matmul: Broadcasting is not implemented"
510              )
511          if self.packed is None or self.meta is None:
512              raise NotImplementedError(
513                  f"`{cls_name}` matmul: operation is not supported"
514              )
515          else:
516              if bias is None:
517                  res = torch._sparse_semi_structured_mm(self.packed, self.meta, B)
518              else:
519                  res = torch._sparse_semi_structured_addmm(
520                      bias, self.packed, self.meta, B
521                  )
522              return res[: self.shape[0]]

525  class SparseSemiStructuredTensorCUSPARSELT(SparseSemiStructuredTensor):
526      """ 
527      The cuSPARSELt backend expects the specified elements and the metadata to be stored in a single tensor: 
528      packed = [ specified elements of original tensor | metadata ] 
529      For an original tensor of size (m, k) we expect the first m * k // 2 elements to be the kept elements 
530      The rest of the tensor is metadata. Since there is only one tensor, we only use the packed and packed_t 
531      attributes respectively. 
532   
533      cuSPARSELt also supports transposition fusion, which is necessary for performant 2:4 sparse training, as well 
534      as specifying alg_id, a config that affects the performance of the matmul depending on matmul sizes. 
535      """
536  
537      BACKEND = "cusparselt"
538      _DTYPE_SHAPE_CONSTRAINTS = {
539          torch.float8_e4m3fn: _SEMI_STRUCTURED_SPARSE_CONFIG(32, 32, 16, 16),
540          torch.int8: _SEMI_STRUCTURED_SPARSE_CONFIG(32, 32, 16, 16),
541          torch.float16: _SEMI_STRUCTURED_SPARSE_CONFIG(16, 16, 8, 8),
542          torch.bfloat16: _SEMI_STRUCTURED_SPARSE_CONFIG(16, 16, 8, 8),
543      }
544  
545      @classmethod
546      def from_dense(
547          cls, original_tensor: torch.Tensor
548      ) -> "SparseSemiStructuredTensorCUSPARSELT":
549          cls._validate_device_dim_dtype_shape(original_tensor)
550          return cls(
551              shape=original_tensor.shape,
552              packed=torch._cslt_compress(original_tensor),
553              meta=None,
554              packed_t=None,
555              meta_t=None,
556              compressed_swizzled_bitmask=None,
557              fuse_transpose_cusparselt=SparseSemiStructuredTensor._FUSE_TRANSPOSE,
558              alg_id_cusparselt=SparseSemiStructuredTensor._DEFAULT_ALG_ID,
559              requires_grad=original_tensor.requires_grad,
560          )
561  
562      @classmethod
563      def prune_dense_static_sort(
564          cls, original_tensor: torch.Tensor, algorithm=""
565      ) -> "SparseSemiStructuredTensor":
566          """ 
567          This function does the same thing as described in SparseSemiStructuredCUTLASS, but uses the cuSPASRELt metadata 
568          layout and sparse matmul. 
569   
570          The only functional difference is that cuSPARSELt stores `metadata` and `packed` together into a single tensor. 
571   
572          [9 1 7 4]                       [9 0 7 0] 
573          [1 2 3 0]                       [0 2 0 0] 
574          [8 3 5 4] -> prune 4x4 tile  -> [8 0 0 4] -> pack to cuSPARSELT semi-structured -> packed 
575          [1 2 6 2]                       [0 0 6 2] 
576   
577                                                    -> pack to transposed cuSPARSELt      -> packed_t 
578                                                       semi-structured representation 
579   
580                                                    -> compute swizzled bitmask           -> compressed_swizzled_bitmask 
581   
582   
583          The equivalent PyTorch code to create the same three outputs from the dense tensor can be found below: 
584          ``` 
585          from torch.sparse import SparseSemiStructuredTensorCUSPARSELT 
586          from torch.sparse._semi_structured_conversions import _sparse_semi_structured_tile, _compute_compressed_swizzled_bitmask 
587   
588          pruned = _sparse_semi_structured_tile(dense) 
589          packed_cusparselt = torch._cslt_compress(pruned) 
590          packed_t_cusparselt = torch._cslt_compress(pruned.t().contiguous()) 
591          bitmask = _compute_compressed_swizzled_bitmask(pruned) 
592   
593          SparseSemiStructuredTensorCUSPARSELT(dense.shape, packed_cutlass, None, packed_t_cutlass, None, bitmask) 
594          ``` 
595          """
596          (
597              packed,
598              meta,
599              packed_t,
600              meta_t,
601              compressed_swizzled_bitmask,
602          ) = torch._sparse_semi_structured_tile(
603              original_tensor, algorithm=algorithm, use_cutlass=False
604          )
605  
606          return cls(
607              original_tensor.shape,
608              packed=packed,
609              meta=meta,
610              packed_t=packed_t,
611              meta_t=meta_t,
612              compressed_swizzled_bitmask=compressed_swizzled_bitmask,
613              requires_grad=False,
614          )
615  
616      def _mm(
617          self, B: torch.Tensor, *, bias: Optional[torch.Tensor] = None, **kwargs
618      ) -> torch.Tensor:
619          if isinstance(B, SparseSemiStructuredTensor):
620              raise ValueError(
621                  "`SparseSemiStructuredTensor @ SparseSemiStructuredTensor` is not supported by the hardware"
622              )
623          if self.ndim != 2 or B.ndim != 2:
624              raise NotImplementedError(
625                  f"`{self.__class__.__name__}` matmul: Broadcasting is not implemented"
626              )
627          if B.dtype != self.dtype:
628              raise NotImplementedError(
629                  f"`{self.__class__.__name__}` matmul: trying to do `A={tuple(self.shape)} @ B={tuple(B.shape)}`, "
630                  f"with A.dtype={self.dtype} and B.dtype={B.dtype}. "
631                  "This operation is only supported when A and B have the same data type."
632              )
633          if bias is not None and bias.dtype != self.dtype:
634              raise NotImplementedError(
635                  f"`{self.__class__.__name__}` matmul: trying to do `A={tuple(self.shape)} @ B={tuple(B.shape)} + C`, "
636                  f"with A.dtype=B.dtype={self.dtype} and C.dtype={B.dtype}. "
637                  "This operation is only supported when A, B and C have the same data type."
638              )
639          # Force fp8 mm to error to be consistent with torch
640          if self.dtype == torch.float8_e4m3fn:
641              raise NotImplementedError(
642                  f"`{self.__class__.__name__}` matmul: trying to do `A={tuple(self.shape)} @ B={tuple(B.shape)}`, "
643                  f"with A.dtype=B.dtype={self.dtype}. "
644                  "mm is not supported for float8_e4m3fn, please use `torch._scaled_mm` instead."
645              )
646          if self.packed is None:
647              raise NotImplementedError(
648                  f"`{self.__class__.__name__}` matmul: operation is not supported"
649              )
650          else:
651              res = torch._cslt_sparse_mm(
652                  self.packed,
653                  B,
654                  bias=bias,
655                  transpose_result=self.fuse_transpose_cusparselt,
656                  alg_id=self.alg_id_cusparselt,
657              )
658              return res.t() if self.fuse_transpose_cusparselt else res
