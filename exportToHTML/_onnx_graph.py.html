17   def parse(graph):
18       nodes = []
19       import itertools
20   
21       nodes_proto = list(itertools.chain(graph.input, graph.output))
22   
23       for node in nodes_proto:
24           print(node.name)
25           shapeproto = TensorShapeProto(
26               dim=[
27                   TensorShapeProto.Dim(size=d.dim_value)
28                   for d in node.type.tensor_type.shape.dim
29               ]
30           )
31           nodes.append(
32               NodeDef(
33                   name=node.name.encode(encoding="utf_8"),
34                   op="Variable",
35                   input=[],
36                   attr={
37                       "dtype": AttrValue(type=node.type.tensor_type.elem_type),
38                       "shape": AttrValue(shape=shapeproto),
39                   },
40               )
41           )
42   
43       for node in graph.node:
44           _attr = [" = ".join([str(f[1]) for f in s.ListFields()]) for s in node.attribute]
45           attr = ", ".join(_attr).encode(encoding="utf_8")
46           print(node.output[0])
47           nodes.append(
48               NodeDef(
49                   name=node.output[0].encode(encoding="utf_8"),
50                   op=node.op_type,
51                   input=node.input,
52                   attr={"parameters": AttrValue(s=attr)},
53               )
54           )
55   
56       # two pass token replacement, appends opname to object id
57       mapping = {}
58       for node in nodes:
59           mapping[node.name] = node.op + "_" + node.name
60   
61       return GraphDef(node=nodes, versions=VersionDef(producer=22))
