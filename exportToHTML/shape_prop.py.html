19   @compatibility(is_backward_compatible=True)
20   class TensorMetadata(NamedTuple):
21       # TensorMetadata is a structure containing pertinent information
22       # about a tensor within a PyTorch program.
23   
24       # General Tensor metadata
25       shape: torch.Size
26       dtype: torch.dtype
27       requires_grad: bool
28       stride: tuple[int, ...]
29       memory_format: Optional[torch.memory_format]
30   
31       # Quantization metadata
32       is_quantized: bool
33       qparams: dict[str, Any]
34   
35   
36   # When include_contiguity is True, we will set contiguity when its always true for the tensor.
37   # Some tensors can represent both contiguous and non-contiguous tensors. e.g: (u0, u1) with (u2, u3).
38   # In such situation contiguity is not set. We could also make it a tri-state i.e: (definitely_contiguous,
39   # contiguous, and unknown).
40   def _extract_tensor_metadata(
41       result: torch.Tensor, include_contiguity=True
42   ) -> TensorMetadata:
43       """ 
44       Extract a TensorMetadata NamedTuple describing `result`. 
45       """
46       shape = result.shape
47       dtype = result.dtype
48       requires_grad = result.requires_grad
49       stride = result.stride() if not is_sparse_any(result) else ()
50   
51       memory_format = None
52   
53       if include_contiguity and not is_sparse_any(result):
54           memory_formats = {
55               torch.contiguous_format,
56               torch.channels_last,
57               torch.channels_last_3d,
58           }
59           for query_format in memory_formats:
60               if definitely_contiguous_for_memory_format(
61                   result, memory_format=query_format
62               ):
63                   memory_format = query_format
64                   break
65   
66       is_quantized = result.is_quantized
67       qparams: dict[str, Any] = {}
68       if is_quantized:
69           qscheme = result.qscheme()
70           qparams["qscheme"] = qscheme
71           if qscheme in {torch.per_tensor_affine, torch.per_tensor_symmetric}:
72               qparams["scale"] = result.q_scale()  # type: ignore[assignment]
73               qparams["zero_point"] = result.q_zero_point()  # type: ignore[assignment]
74           elif qscheme in {
75               torch.per_channel_affine,
76               torch.per_channel_affine_float_qparams,
77               torch.per_channel_symmetric,
78           }:
79               # In this branch, scale and zero_point are expected to be tensors,
80               # we store the values as immutable_list in TensorMetadata for
81               # easier serialization downstream
82               qparams["scale"] = result.q_per_channel_scales().tolist()  # type: ignore[assignment]
83               qparams["zero_point"] = result.q_per_channel_zero_points().tolist()  # type: ignore[assignment]
84               qparams["axis"] = result.q_per_channel_axis()  # type: ignore[assignment]
85   
86       return TensorMetadata(
87           shape, dtype, requires_grad, stride, memory_format, is_quantized, qparams
88       )

91   @compatibility(is_backward_compatible=True)
92   class ShapeProp(torch.fx.Interpreter):
93       """ 
94       Execute an FX graph Node-by-Node and 
95       record the shape and type of the result 
96       into the corresponding node. 
97    
98       Example: 
99            In this example, we record the shape 
100           and data type of a module given 
101           an example input ``torch.randn(50, D_in)``. 
102           We print the name, shape and dtype of each node. 
103   
104          class TwoLayerNet(torch.nn.Module): 
105              def __init__(self, D_in, H, D_out): 
106                  super().__init__() 
107                  self.linear1 = torch.nn.Linear(D_in, H) 
108                  self.linear2 = torch.nn.Linear(H, D_out) 
109              def forward(self, x): 
110                  h_relu = self.linear1(x).clamp(min=0) 
111                  y_pred = self.linear2(h_relu) 
112                  return y_pred 
113          N, D_in, H, D_out = 64, 1000, 100, 10 
114          x = torch.randn(N, D_in) 
115          y = torch.randn(N, D_out) 
116          model = TwoLayerNet(D_in, H, D_out) 
117          gm = torch.fx.symbolic_trace(model) 
118          sample_input = torch.randn(50, D_in) 
119          ShapeProp(gm).propagate(sample_input) 
120   
121          for node in gm.graph.nodes: 
122              print(node.name, node.meta['tensor_meta'].dtype, 
123                  node.meta['tensor_meta'].shape) 
124   
125          The output of this code is: 
126   
127          x torch.float32 torch.Size([50, 1000]) 
128          linear1 torch.float32 torch.Size([50, 100]) 
129          clamp_1 torch.float32 torch.Size([50, 100]) 
130          linear2 torch.float32 torch.Size([50, 10]) 
131          output torch.float32 torch.Size([50, 10]) 
132   
133      Args: 
134           module (GraphModule): The module to be executed 
135           fake_mode (FakeTensorMode): A fake mode for copying the gm 
136   
137      """
138  
139      def __init__(self, gm, fake_mode=None):
140          super().__init__(gm)
141          if fake_mode is None:
142              fake_mode = detect_fake_mode()
143          if fake_mode is not None:
144              from torch._dynamo.utils import deepcopy_to_fake_tensor
145  
146              # Note:
147              # We need fake execution cause the inputs are fake, however, we cannot fakify the module
148              # - because we need to write to the tensor_meta of the real module. So we fakify to
149              # produce a result (L131 below), to extract tensor meta, and then keep going.
150              #
151              # If we were to fakify, we would write to the wrong node, and then downstream fusion
152              # would be missing the tensor_meta.
153              #
154              # See torch/_inductor/overrides.py for where this is called upstream of fusion.
155              self.fake_module = deepcopy_to_fake_tensor(self.module, fake_mode)
156              self.fake_mode = fake_mode
157          else:
158              self.fake_module = None
159              self.fake_mode = None
160  
161          self.real_module = self.module
162  
163      def run_node(self, n: Node) -> Any:
164          from torch.fx.experimental.symbolic_shapes import (
165              compute_unbacked_bindings,
166              rebind_unbacked,
167          )
168  
169          try:
170              if self.fake_module is not None:
171                  # Hacky swap. Alternatively, we could do this with overriding
172                  # call_module and get_attr.
173                  self.module = self.fake_module
174              try:
175                  if self.fake_mode is not None:
176                      with self.fake_mode, enable_python_dispatcher():
177                          result = super().run_node(n)
178                          rebind_unbacked(self.fake_mode.shape_env, n, result)
179                  else:
180                      result = super().run_node(n)
181              finally:
182                  self.module = self.real_module
183          except Exception as e:
184              traceback.print_exc()
185              raise RuntimeError(
186                  f"ShapeProp error for: node={n.format_node()} with meta={n.meta}"
187              ) from e
188  
189          found_tensor = False
190  
191          def extract_tensor_meta(obj):
192              if isinstance(obj, torch.Tensor):
193                  nonlocal found_tensor
194                  found_tensor = True
195                  return _extract_tensor_metadata(obj)
196              else:
197                  return obj
198  
199          meta = map_aggregate(result, extract_tensor_meta)
200          if found_tensor:
201              n.meta["tensor_meta"] = meta
202  
203          if self.fake_mode:
204              if (shape_env := self.fake_mode.shape_env) and (
205                  symbol_to_path := compute_unbacked_bindings(shape_env, result)
206              ):
207                  n.meta["unbacked_bindings"] = symbol_to_path
208  
209          n.meta["type"] = type(result)
210          return result
211  
212      def propagate(self, *args):
213          """ 
214          Run `module` via interpretation and return the result and 
215          record the shape and type of each node. 
216   
217          Args: 
218              *args (Tensor): the sample input. 
219   
220          Returns: 
221              Any: The value returned from executing the Module 
222          """
223          if self.fake_mode is not None:
224              fake_args = [
225                  self.fake_mode.from_tensor(t) if isinstance(t, torch.Tensor) else t
226                  for t in args
227              ]
228          else:
229              fake_args = args
230          return super().run(*fake_args)
