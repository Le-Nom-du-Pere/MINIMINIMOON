41   @dataclass
42   class _Config(Generic[T]):
43       """Represents a config with richer behaviour than just a default value. 
44       :: 
45           i.e. 
46           foo = Config(justknob="//foo:bar", default=False) 
47           install_config_module(...) 
48    
49       This configs must be installed with install_config_module to be used 
50    
51       Precedence Order: 
52           alias: If set, the directly use the value of the alias. 
53           env_name_force: If set, this environment variable has precedence over 
54               everything after this. 
55               If multiple env variables are given, the precendence order is from 
56               left to right. 
57           user_override: If a user sets a value (i.e. foo.bar=True), that 
58               has precedence over everything after this. 
59           env_name_default: If set, this environment variable will override everything 
60               after this. 
61               If multiple env variables are given, the precendence order is from 
62               left to right. 
63           justknob: If this pytorch installation supports justknobs, that will 
64               override defaults, but will not override the user_override precendence. 
65           default: This value is the lowest precendance, and will be used if nothing is 
66               set. 
67    
68       Environment Variables: 
69           These are interpreted to be either "0" or "1" to represent true and false. 
70    
71       Arguments: 
72           justknob: the name of the feature / JK. In OSS this is unused. 
73           default: is the value to default this knob to in OSS. 
74           alias: The alias config to read instead. 
75           env_name_force: The environment variable, or list of, to read that is a FORCE 
76               environment variable. I.e. it overrides everything except for alias. 
77           env_name_default: The environment variable, or list of, to read that changes the 
78               default behaviour. I.e. user overrides take preference. 
79       """
80   
81       default: Union[T, object]
82       justknob: Optional[str] = None
83       env_name_default: Optional[list[str]] = None
84       env_name_force: Optional[list[str]] = None
85       alias: Optional[str] = None
86   
87       def __init__(
88           self,
89           default: Union[T, object] = _UNSET_SENTINEL,
90           justknob: Optional[str] = None,
91           env_name_default: Optional[Union[str, list[str]]] = None,
92           env_name_force: Optional[Union[str, list[str]]] = None,
93           value_type: Optional[type] = None,
94           alias: Optional[str] = None,
95       ):
96           # python 3.9 does not support kw_only on the dataclass :(.
97           self.default = default
98           self.justknob = justknob
99           self.env_name_default = _Config.string_or_list_of_string_to_list(
100              env_name_default
101          )
102          self.env_name_force = _Config.string_or_list_of_string_to_list(env_name_force)
103          self.value_type = value_type
104          self.alias = alias
105          if self.alias is not None:
106              assert (
107                  default is _UNSET_SENTINEL
108                  and justknob is None
109                  and env_name_default is None
110                  and env_name_force is None
111              ), "if alias is set, none of {default, justknob and env var} can be set"
112  
113      @staticmethod
114      def string_or_list_of_string_to_list(
115          val: Optional[Union[str, list[str]]]
116      ) -> Optional[list[str]]:
117          if val is None:
118              return None
119          if isinstance(val, str):
120              return [val]
121          assert isinstance(val, list)
122          return val

156  def _read_env_variable(name: str) -> Optional[Union[bool, str]]:
157      value = os.environ.get(name)
158      if value == "1":
159          return True
160      if value == "0":
161          return False
162      return value

165  def install_config_module(module: ModuleType) -> None:
166      """ 
167      Converts a module-level config into a `ConfigModule()`. 
168   
169      See _config_typing.pyi for instructions on how to get the converted module to typecheck. 
170      """
171  
172      class ConfigModuleInstance(ConfigModule):
173          # __annotations__ is written to by Sphinx autodoc
174          _bypass_keys = set({"_is_dirty", "_hash_digest", "__annotations__"})
175  
176      def visit(
177          source: Union[ModuleType, type],
178          dest: Union[ModuleType, SubConfigProxy],
179          prefix: str,
180      ) -> None:
181          """Walk the module structure and move everything to module._config"""
182          if sys.version_info[:2] < (3, 10):
183              type_hints = getattr(source, "__annotations__", {})
184          else:
185              type_hints = inspect.get_annotations(source)
186          for key, value in list(source.__dict__.items()):
187              if (
188                  key.startswith("__")
189                  or isinstance(value, (ModuleType, FunctionType))
190                  or (hasattr(value, "__module__") and value.__module__ == "typing")
191                  # Handle from torch.utils._config_module import Config
192                  or (isinstance(value, type) and issubclass(value, _Config))
193              ):
194                  continue
195  
196              name = f"{prefix}{key}"
197              annotated_type = type_hints.get(key, None)
198              if isinstance(value, CONFIG_TYPES):
199                  config[name] = _ConfigEntry(
200                      _Config(default=value, value_type=annotated_type)
201                  )
202                  if dest is module:
203                      delattr(module, key)
204              elif isinstance(value, _Config):
205                  if annotated_type is not None and value.value_type is None:
206                      value.value_type = annotated_type
207  
208                  config[name] = _ConfigEntry(value)
209  
210                  if dest is module:
211                      delattr(module, key)
212              elif isinstance(value, type):
213                  assert value.__module__ == module.__name__
214                  # a subconfig with `class Blah:` syntax
215                  proxy = SubConfigProxy(module, f"{name}.")
216                  visit(value, proxy, f"{name}.")
217                  if dest is module:
218                      setattr(dest, key, proxy)
219                  else:
220                      dest.__dict__[key] = proxy
221              else:
222                  raise AssertionError(f"Unhandled config {key}={value} ({type(value)})")
223  
224      config: dict[str, _ConfigEntry] = {}
225  
226      compile_ignored_keys = get_assignments_with_compile_ignored_comments(module)
227  
228      visit(module, module, "")
229      module._config = config  # type: ignore[attr-defined]
230      module._compile_ignored_keys = compile_ignored_keys  # type: ignore[attr-defined]
231      module.__class__ = ConfigModuleInstance
232      module._is_dirty = True  # type: ignore[attr-defined]
233      module._hash_digest = None  # type: ignore[attr-defined]
234

236  COMPILE_IGNORED_MARKER = "@compile_ignored"
237  
238  
239  # Gets all the keys (i.e. assignments) with a @compile_ignored comment
240  def get_assignments_with_compile_ignored_comments(module: ModuleType) -> set[str]:
241      source_code = inspect.getsource(module)
242      assignments = set()
243  
244      # Tokenize the source code to retrieve comments
245      tokens = tokenize.tokenize(io.BytesIO(source_code.encode("utf-8")).readline)
246      current_comment = "", -1
247      prev_name = ""
248  
249      for token in tokens:
250          if token.type == tokenize.COMMENT:
251              prev_name = ""
252              maybe_current = token.string.strip()
253              if COMPILE_IGNORED_MARKER in maybe_current:
254                  assert current_comment == (
255                      "",
256                      -1,
257                  ), f"unconsumed {COMPILE_IGNORED_MARKER}"
258                  current_comment = maybe_current, token.start[0]
259          elif token.type == tokenize.NAME:
260              # Only accept the first name token, to handle if you have
261              # something like foo: Bar = ...
262              if not prev_name:
263                  prev_name = token.string
264          elif token.type == tokenize.OP and token.string == "=":
265              # Check if the current assignment follows a comment
266              # with COMPILE_IGNORED_MARKER
267              if (
268                  COMPILE_IGNORED_MARKER in current_comment[0]
269                  and current_comment[1] == token.start[0] - 1
270              ):
271                  assignments.add(prev_name)
272                  current_comment = "", -1  # reset
273              prev_name = ""
274      assert current_comment == ("", -1), f"unconsumed {COMPILE_IGNORED_MARKER}"
275      return assignments

278  @dataclass
279  class _ConfigEntry:
280      # The default value specified in the configuration
281      default: Any
282      # The type of the configuration value
283      value_type: type
284      # The value specified by the user when they overrode the configuration
285      # _UNSET_SENTINEL indicates the value is not set.
286      user_override: Any = _UNSET_SENTINEL
287      # The justknob to check for this config
288      justknob: Optional[str] = None
289      # environment variables are read at install time
290      env_value_force: Any = _UNSET_SENTINEL
291      env_value_default: Any = _UNSET_SENTINEL
292      # Used to work arounds bad assumptions in unittest.mock.patch
293      # The code to blame is
294      # https://github.com/python/cpython/blob/94a7a4e22fb8f567090514785c69e65298acca42/Lib/unittest/mock.py#L1637
295      # Essentially, mock.patch requires, that if __dict__ isn't accessible
296      # (which it isn't), that after delattr is called on the object, the
297      # object must throw when hasattr is called. Otherwise, it doesn't call
298      # setattr again.
299      # Technically we'll have an intermediate state of hiding the config while
300      # mock.patch is unpatching itself, but it calls setattr after the delete
301      # call so the final state is correct. It's just very unintuitive.
302      # upstream bug - python/cpython#126886
303      hide: bool = False
304      alias: Optional[str] = None
305  
306      def __init__(self, config: _Config):
307          self.default = config.default
308          self.value_type = (
309              config.value_type if config.value_type is not None else type(self.default)
310          )
311          self.justknob = config.justknob
312          self.alias = config.alias
313          if config.env_name_default is not None:
314              for val in config.env_name_default:
315                  if (env_value := _read_env_variable(val)) is not None:
316                      self.env_value_default = env_value
317                      break
318          if config.env_name_force is not None:
319              for val in config.env_name_force:
320                  if (env_value := _read_env_variable(val)) is not None:
321                      self.env_value_force = env_value
322                      break
323  
324          # Ensure justknobs and envvars are allowlisted types
325          if self.justknob is not None and self.default is not None:
326              assert isinstance(
327                  self.default, bool
328              ), f"justknobs only support booleans, {self.default} is not a boolean"
329          if self.value_type is not None and (
330              config.env_name_default is not None or config.env_name_force is not None
331          ):
332              assert self.value_type in (
333                  bool,
334                  str,
335                  Optional[bool],
336                  Optional[str],
337              ), f"envvar configs only support (optional) booleans or strings, {self.value_type} is neither"

340  class ConfigModule(ModuleType):
341      # NOTE: This should be kept in sync with _config_typing.pyi.
342  
343      # The actual configuration settings.  E.g., torch._dynamo.config.debug
344      # would live as "debug" in the key, and torch._inductor.config.triton.cudagraphs
345      # maps as "triton.cudagraphs". See discussion on the class for meaning of various sub items
346      _config: dict[str, _ConfigEntry]
347      _bypass_keys: set[str]
348      _compile_ignored_keys: set[str]
349      _is_dirty: bool
350      _hash_digest: Optional[bytes]
351  
352      def __init__(self) -> None:
353          raise NotImplementedError(
354              f"use {__name__}.install_config_module(sys.modules[__name__])"
355          )
356  
357      def __setattr__(self, name: str, value: object) -> None:
358          if name in self._bypass_keys:
359              super().__setattr__(name, value)
360          elif name not in self._config:
361              raise AttributeError(f"{self.__name__}.{name} does not exist")
362          elif self._config[name].alias is not None:
363              self._set_alias_val(self._config[name], value)
364          else:
365              self._config[name].user_override = value
366              self._is_dirty = True
367              self._config[name].hide = False
368  
369      def __getattr__(self, name: str) -> Any:
370          try:
371              config = self._config[name]
372  
373              if config.hide:
374                  raise AttributeError(f"{self.__name__}.{name} does not exist")
375  
376              alias_val = self._get_alias_val(config)
377              if alias_val is not _UNSET_SENTINEL:
378                  return alias_val
379  
380              if config.env_value_force is not _UNSET_SENTINEL:
381                  return config.env_value_force
382  
383              if config.user_override is not _UNSET_SENTINEL:
384                  return config.user_override
385  
386              if config.env_value_default is not _UNSET_SENTINEL:
387                  return config.env_value_default
388  
389              if config.justknob is not None:
390                  # JK only supports bools and ints
391                  return justknobs_check(name=config.justknob, default=config.default)
392  
393              # Note that reference types can still be modified, so we
394              # copy them to user_overrides in case the user overrides
395              # them
396              if isinstance(config.default, (list, set, dict)):
397                  config.user_override = copy.deepcopy(config.default)
398                  return config.user_override
399              return config.default
400  
401          except KeyError as e:
402              # make hasattr() work properly
403              raise AttributeError(f"{self.__name__}.{name} does not exist") from e
404  
405      def __delattr__(self, name: str) -> None:
406          self._is_dirty = True
407          # must support delete because unittest.mock.patch deletes
408          # then recreate things
409          self._config[name].user_override = _UNSET_SENTINEL
410          self._config[name].hide = True
411  
412      def _get_alias_module_and_name(
413          self, entry: _ConfigEntry
414      ) -> Optional[tuple[ModuleType, str]]:
415          alias = entry.alias
416          if alias is None:
417              return None
418          module_name, constant_name = alias.rsplit(".", 1)
419          try:
420              module = importlib.import_module(module_name)
421          except ImportError as e:
422              raise AttributeError("config alias {alias} does not exist") from e
423          return module, constant_name
424  
425      def _get_alias_val(self, entry: _ConfigEntry) -> Any:
426          data = self._get_alias_module_and_name(entry)
427          if data is None:
428              return _UNSET_SENTINEL
429          module, constant_name = data
430          constant_value = getattr(module, constant_name)
431          return constant_value
432  
433      def _set_alias_val(self, entry: _ConfigEntry, val: Any) -> None:
434          data = self._get_alias_module_and_name(entry)
435          assert data is not None
436          module, constant_name = data
437          setattr(module, constant_name, val)
438  
439      def _is_default(self, name: str) -> bool:
440          """ 
441          Returns true if the config is at its default value. 
442          configs overridden by the env are not considered default. 
443          """
444          config_val = self._config[name]
445          # The config is not overridden by the user, and the env_value_default
446          # is different from the default value (meaning user has set the env to
447          # change the default value).
448          not_set_env_default = (
449              config_val.env_value_default is _UNSET_SENTINEL
450              or config_val.env_value_default == config_val.default
451          )
452          not_set_env_force = (
453              config_val.env_value_force is _UNSET_SENTINEL
454              or config_val.env_value_force == config_val.default
455          )
456  
457          unset = config_val.user_override is _UNSET_SENTINEL
458          # Handle reference types specially to avoid spammy warnings
459          if isinstance(config_val.default, (list, set, dict)):
460              unset = unset or config_val.user_override == config_val.default
461          return unset and not_set_env_default and not_set_env_force
462  
463      def _get_dict(
464          self,
465          ignored_keys: Optional[list[str]] = None,
466          ignored_prefixes: Optional[list[str]] = None,
467          skip_default: bool = False,
468      ) -> dict[str, Any]:
469          """Export a dictionary of current configuration keys and values. 
470   
471          This function is design to provide a single point which handles 
472          accessing config options and exporting them into a dictionary. 
473          This is used by a number of different user facing export methods 
474          which all have slightly different semantics re: how and what to 
475          skip. 
476          If a config is aliased, it skips this config. 
477   
478          Arguments: 
479              ignored_keys are keys that should not be exported. 
480              ignored_prefixes are prefixes that if a key matches should 
481                  not be exported 
482              skip_default does two things. One if a key has not been modified 
483                  it skips it. 
484          """
485          config: dict[str, Any] = {}
486          for key in self._config:
487              if ignored_keys and key in ignored_keys:
488                  continue
489              if ignored_prefixes:
490                  if any(key.startswith(prefix) for prefix in ignored_prefixes):
491                      continue
492              if skip_default and self._is_default(key):
493                  continue
494              if self._config[key].alias is not None:
495                  continue
496              config[key] = copy.deepcopy(getattr(self, key))
497  
498          return config
499  
500      def get_type(self, config_name: str) -> type:
501          return self._config[config_name].value_type
502  
503      def save_config(self) -> bytes:
504          """Convert config to a pickled blob"""
505          ignored_keys = getattr(self, "_save_config_ignore", [])
506          return pickle.dumps(
507              self._get_dict(ignored_keys=ignored_keys),
508              protocol=2,
509          )
510  
511      def save_config_portable(
512          self, *, ignore_private_configs: bool = True
513      ) -> dict[str, Any]:
514          """Convert config to portable format"""
515          prefixes = []
516          if ignore_private_configs:
517              prefixes.append("_")
518          prefixes.extend(getattr(self, "_cache_config_ignore_prefix", []))
519          return self._get_dict(ignored_prefixes=prefixes)
520  
521      def codegen_config(self) -> str:
522          """Convert config to Python statements that replicate current config. 
523          This does NOT include config settings that are at default values. 
524          """
525  
526          # additional imports required
527          imports = set()
528  
529          def get_module_name(func: Callable, add_dot: bool) -> str:
530              module_name = func.__module__
531              if module_name == "builtins":
532                  module_name = ""
533              if add_dot and module_name != "":
534                  module_name += "."
535              return module_name
536  
537          def add_import(func: Callable) -> None:
538              module_name = get_module_name(func, False)
539              if module_name:
540                  imports.add(module_name)
541  
542          def list_of_callables_to_string(v: Union[list, set]) -> list[str]:
543              return [f"{get_module_name(item, True)}{item.__name__}" for item in v]
544  
545          def importable_callable(v: Any) -> bool:
546              # functools.partial has no attributes below but is a callable
547              return callable(v) and hasattr(v, "__module__") and hasattr(v, "__name__")
548  
549          def get_config_line(mod, k, v) -> str:  # type: ignore[no-untyped-def]
550              """ 
551              Return a string version of the config line. 
552              Handle v when v is a callable, or a list/dict of callables. Add import statements for callables if necessary. 
553              We assume that the value of a single config won't be a mix of callables and non-callables. 
554   
555              Example output: 
556                  import logging 
557                  import _warnings 
558                  torch._dynamo.config.reorderable_logging_functions = { _warnings.warn, logging.warn, print } 
559              """
560              if importable_callable(v):
561                  add_import(v)
562                  return f"{mod}.{k} = {get_module_name(v, True)}{v.__name__}"
563              elif isinstance(v, (list, set)) and all(
564                  importable_callable(item) for item in v
565              ):
566                  for item in v:
567                      add_import(item)
568                  v_list = list_of_callables_to_string(v)
569                  if isinstance(v, list):
570                      return f"{mod}.{k} = {v_list}"
571                  else:
572                      return f"{mod}.{k} = {{ {', '.join(v_list)} }}"
573              else:
574                  return f"{mod}.{k} = {v!r}"
575  
576          lines = []
577          mod = self.__name__
578          for k, v in self._get_dict(
579              ignored_keys=getattr(self, "_save_config_ignore", []), skip_default=True
580          ).items():
581              lines.append(get_config_line(mod, k, v))
582          for import_name in imports:
583              lines.insert(0, f"import {import_name}")
584          return "\n".join(lines)
585  
586      def get_hash(self) -> bytes:
587          """Hashes the configs that are not compile_ignored"""
588          if self._is_dirty or self._hash_digest is None:
589              dict_to_hash = self._get_dict(ignored_keys=list(self._compile_ignored_keys))
590              string_to_hash = repr(sorted(dict_to_hash.items()))
591              self._hash_digest = hashlib.md5(
592                  string_to_hash.encode("utf-8"), usedforsecurity=False
593              ).digest()
594              self._is_dirty = False
595          return self._hash_digest
596  
597      @deprecated(

599          " use `config.get_config_copy()` instead if you just want a copy of the config, or "
600          "config.load_config if you need mutable access",
601          category=FutureWarning,
602      )
603      def to_dict(self) -> dict[str, Any]:
604          return self.get_config_copy()

610          category=FutureWarning,

612      def shallow_copy_dict(self) -> dict[str, Any]:

615      def load_config(self, maybe_pickled_config: Union[bytes, dict[str, Any]]) -> None:
616          """Restore from a prior call to save_config() or shallow_copy_dict()"""
617          if not isinstance(maybe_pickled_config, dict):
618              config = pickle.loads(maybe_pickled_config)
619          else:
620              config = maybe_pickled_config
621          for k, v in config.items():
622              if k in self._config:
623                  setattr(self, k, v)
624              else:
625                  from torch._dynamo.utils import warn_once
626  
627                  warn_once(f"key {k} with value {v} is not understood by this config")
628  
629      def get_config_copy(self) -> dict[str, Any]:
630          return self._get_dict()
631  
632      def patch(
633          self,
634          arg1: Optional[Union[str, dict[str, Any]]] = None,
635          arg2: Any = None,
636          **kwargs: dict[str, Any],
637      ) -> "ContextDecorator":
638          """ 
639          Decorator and/or context manager to make temporary changes to a config. 
640   
641          As a decorator: 
642   
643              @config.patch("name", val) 
644              @config.patch(name1=val1, name2=val2) 
645              @config.patch({"name1": val1, "name2", val2}) 
646              def foo(...): 
647                  ... 
648   
649          As a context manager: 
650   
651              with config.patch("name", val): 
652                  ... 
653          """
654          changes: dict[str, Any]
655          if arg1 is not None:
656              if arg2 is not None:
657                  assert isinstance(arg1, str)
658                  # patch("key", True) syntax
659                  changes = {arg1: arg2}
660              else:
661                  assert isinstance(arg1, dict)
662                  # patch({"key": True}) syntax
663                  changes = arg1
664              assert not kwargs
665          else:
666              # patch(key=True) syntax
667              changes = kwargs
668              assert arg2 is None
669          assert isinstance(changes, dict), f"expected `dict` got {type(changes)}"
670          prior: dict[str, Any] = {}
671          config = self
672  
673          class ConfigPatch(ContextDecorator):
674              def __init__(self) -> None:
675                  self.changes = changes
676  
677              def __enter__(self) -> None:
678                  assert not prior
679                  for key in self.changes.keys():
680                      # KeyError on invalid entry
681                      prior[key] = config.__getattr__(key)
682                  for k, v in self.changes.items():
683                      config.__setattr__(k, v)
684  
685              def __exit__(self, exc_type, exc_val, exc_tb):  # type: ignore[no-untyped-def]
686                  for k, v in prior.items():
687                      config.__setattr__(k, v)
688                  prior.clear()
689  
690          return ConfigPatch()
691  
692      def _make_closure_patcher(self, **changes: dict[str, Any]) -> Any:
693          """ 
694          A lower-overhead version of patch() for things on the critical path. 
695   
696          Usage: 
697   
698              # do this off the critical path 
699              change_fn = config.make_closure_patcher(foo=True) 
700   
701              ... 
702   
703              revert = change_fn() 
704              try: 
705                ... 
706              finally: 
707                  revert() 
708   
709          """
710          config = self._config
711  
712          def change() -> Callable[[], None]:
713              prior = {k: config[k].user_override for k in changes}
714              for k, v in changes.items():
715                  self._config[k].user_override = v
716  
717              def revert() -> None:
718                  for k, v in prior.items():
719                      self._config[k].user_override = v
720  
721              return revert
722  
723          return change

726  class ContextDecorator(contextlib.ContextDecorator):
727      """ 
728      Same as contextlib.ContextDecorator, but with support for 
729      `unittest.TestCase` 
730      """
731  
732      def __enter__(self) -> None:
733          raise NotImplementedError("NYI")
734  
735      def __exit__(self, exc_type, exc_val, exc_tb) -> NoReturn:  # type: ignore[no-untyped-def]
736          raise NotImplementedError("NYI")
737  
738      def __call__(self, func: Callable[[Any], Any]) -> Any:
739          if isinstance(func, type) and issubclass(func, unittest.TestCase):
740  
741              class _TestCase(func):  # type: ignore[valid-type, misc]
742                  @classmethod
743                  def setUpClass(cls) -> None:
744                      self.__enter__()
745                      try:
746                          super().setUpClass()
747                      except Exception:
748                          self.__exit__(None, None, None)
749                          raise
750  
751                  @classmethod
752                  def tearDownClass(cls) -> None:
753                      try:
754                          super().tearDownClass()
755                      finally:
756                          self.__exit__(None, None, None)
757  
758              _TestCase.__name__ = func.__name__
759              _TestCase.__qualname__ = func.__qualname__
760              _TestCase.__module__ = func.__module__
761  
762              return _TestCase
763  
764          return super().__call__(func)
