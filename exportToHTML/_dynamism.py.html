19   def normalize_source_name(name: str) -> str:
20       # Match attribute access like .x and replace with ['x']
21       return re.sub(r"\.([a-zA-Z_][a-zA-Z0-9_]*)", r"['\1']", name)

24   def module_to_nested_dict(module: torch.nn.Module) -> dict[str, Any]:
25       """Recursively converts an nn.Module into a nested dictionary with explicit 'parameters' and 'modules' keys."""
26       self_dict: dict[str, Any] = {}
27   
28       self_dict["_parameters"] = {}
29       self_dict["_modules"] = {}
30   
31       for attr_name in dir(module):
32           try:
33               if not attr_name.startswith("_") and not callable(
34                   getattr(module, attr_name)
35               ):
36                   attr_value = getattr(module, attr_name)
37                   if (
38                       not isinstance(attr_value, torch.nn.Module)
39                       and isinstance(attr_value, (int, float, torch.Tensor))
40                       and type(attr_value) is not bool
41                   ):
42                       self_dict[attr_name] = attr_value
43           except NotImplementedError:
44               # Skip attributes that raise NotImplementedError since they won't
45               # contain any dynamism anyways.
46               continue
47   
48       for name, param in module.named_parameters(recurse=False):
49           self_dict["_parameters"][name] = param
50       for name, buffer in module.named_buffers(recurse=False):
51           self_dict["_parameters"][name] = buffer
52   
53       for name, submodule in module.named_children():
54           self_dict["_modules"][name] = module_to_nested_dict(submodule)
55   
56       return self_dict

59   def track_dynamism_across_examples(
60       example_inputs: list[Any],
61   ) -> dict[Any, Any]:
62       """ 
63       This function analyzes a list of example inputs to determine the dynamism of their shapes. 
64       It tracks whether the dimensions of tensors or non-tensor values change across 
65       different examples. The function returns a dictionary where each key represents 
66       a path to a value in the input examples, and the corresponding value is a tuple 
67       indicating which dimensions are dynamic (i.e., change across examples). This 
68       helps in understanding how the structure of data varies across different instances. 
69       """
70       tracking: dict[KeyPath, tuple[list[set[Any]], bool]] = {}
71   
72       for ex in example_inputs:
73           if "self" in ex and isinstance(ex["self"], torch.nn.Module):
74               ex["self"] = module_to_nested_dict(ex["self"])
75           leaves_with_paths, _ = tree_flatten_with_path(ex)
76           for key_path, value in leaves_with_paths:
77               if not isinstance(value, (int, float, torch.Tensor)):
78                   continue
79               if isinstance(value, torch.Tensor):
80                   shape: tuple[int | float, ...] = tuple(value.shape)
81                   is_tensor = True
82               else:
83                   shape = (value,)
84                   is_tensor = False
85               if key_path not in tracking:
86                   tracking[key_path] = ([set() for _ in range(len(shape))], is_tensor)
87               else:
88                   dim_sets, flag = tracking[key_path]
89                   if flag != is_tensor:
90                       pass
91                   while len(dim_sets) < len(shape):
92                       dim_sets.append(set())
93               for i, dim in enumerate(shape):
94                   tracking[key_path][0][i].add(dim)
95   
96       output: dict[Any, Any] = {}
97       for key_path, (dim_sets, _is_tensor) in tracking.items():
98           final_dyn = tuple(len(s) > 1 for s in dim_sets)
99           key_str = "L" + "".join(f"{str(k)}" for k in key_path)
100          key = key_path[0].key  # type: ignore[attr-defined]
101          if key not in output:
102              output[key] = {}
103          output[key][key_str] = final_dyn
104      return output

107  def clone_and_convert_to_meta(example_input: Any) -> Any:
108      """ 
109      This function takes a list of example inputs and for each tensor, clones it and converts it to device=meta. 
110      For non-tensor values, it keeps the reference. It uses pytree to handle nested structures recursively. 
111      """
112  
113      def transform_fn(value: Any) -> Any:
114          if isinstance(value, torch.Tensor):
115              return value.clone().to(device="meta")
116          return value
117  
118      return tree_map(transform_fn, example_input)
