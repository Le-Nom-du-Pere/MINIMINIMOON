3    """Common utilities for testing Dynamo's minifier functionality. 
4     
5    This module provides the base infrastructure for running minification tests in Dynamo. 
6    It includes: 
7    - MinifierTestResult: A dataclass for storing and processing minifier test results 
8    - MinifierTestBase: A base test class with utilities for: 
9      - Running tests in isolated environments 
10     - Managing temporary directories and configurations 
11     - Executing minifier launcher scripts 
12     - Running and validating reproduction scripts 
13     - Supporting both compile-time and runtime error testing 
14    
15   The minifier helps reduce failing Dynamo compilations to minimal reproductions. 
16   """

38   @dataclasses.dataclass
39   class MinifierTestResult:
40       minifier_code: str
41       repro_code: str

43       def _get_module(self, t):
44           match = re.search(r"class Repro\(torch\.nn\.Module\):\s+([ ].*\n| *\n)+", t)
45           assert match is not None, "failed to find module"
46           r = match.group(0)
47           r = re.sub(r"\s+$", "\n", r, flags=re.MULTILINE)
48           r = re.sub(r"\n{3,}", "\n\n", r)
49           return r.strip()
50   
51       def get_exported_program_path(self):
52           # Extract the exported program file path from AOTI minifier's repro.py
53           # Regular expression pattern to match the file path
54           pattern = r'torch\.export\.load\(\s*["\'](.*?)["\']\s*\)'
55           # Search for the pattern in the text
56           match = re.search(pattern, self.repro_code)
57           # Extract and print the file path if a match is found
58           if match:
59               file_path = match.group(1)
60               return file_path
61           return None
62   
63       def minifier_module(self):
64           return self._get_module(self.minifier_code)
65   
66       def repro_module(self):
67           return self._get_module(self.repro_code)

70   class MinifierTestBase(torch._dynamo.test_case.TestCase):
71       DEBUG_DIR = tempfile.mkdtemp()
72   
73       @classmethod
74       def setUpClass(cls):
75           super().setUpClass()
76           if not os.path.exists(cls.DEBUG_DIR):
77               cls.DEBUG_DIR = tempfile.mkdtemp()
78           cls._exit_stack.enter_context(  # type: ignore[attr-defined]
79               torch._dynamo.config.patch(debug_dir_root=cls.DEBUG_DIR)
80           )
81           # These configurations make new process startup slower.  Disable them
82           # for the minification tests to speed them up.
83           cls._exit_stack.enter_context(  # type: ignore[attr-defined]
84               torch._inductor.config.patch(
85                   {
86                       # https://github.com/pytorch/pytorch/issues/100376
87                       "pattern_matcher": False,
88                       # multiprocess compilation takes a long time to warmup
89                       "compile_threads": 1,
90                       # https://github.com/pytorch/pytorch/issues/100378
91                       "cpp.vec_isa_ok": False,
92                   }
93               )
94           )
95   
96       @classmethod
97       def tearDownClass(cls):
98           if os.getenv("PYTORCH_KEEP_TMPDIR", "0") != "1":
99               shutil.rmtree(cls.DEBUG_DIR)
100          else:
101              print(f"test_minifier_common tmpdir kept at: {cls.DEBUG_DIR}")
102          cls._exit_stack.close()  # type: ignore[attr-defined]

104      def _gen_codegen_fn_patch_code(self, device, bug_type):
105          assert bug_type in ("compile_error", "runtime_error", "accuracy")
106          return f"""\ 
107  {torch._dynamo.config.codegen_config()}
108  {torch._inductor.config.codegen_config()}
109  torch._inductor.config.{"cpp" if device == "cpu" else "triton"}.inject_relu_bug_TESTING_ONLY = {bug_type!r}
110  """

112      def _maybe_subprocess_run(self, args, *, isolate, cwd=None):
113          if not isolate:
114              assert len(args) >= 2, args
115              assert args[0] == "python3", args
116              if args[1] == "-c":
117                  assert len(args) == 3, args
118                  code = args[2]
119                  args = ["-c"]
120              else:
121                  assert len(args) >= 2, args
122                  with open(args[1]) as f:
123                      code = f.read()
124                  args = args[1:]
125  
126              # WARNING: This is not a perfect simulation of running
127              # the program out of tree.  We only interpose on things we KNOW we
128              # need to handle for tests.  If you need more stuff, you will
129              # need to augment this appropriately.
130  
131              # NB: Can't use save_config because that will omit some fields,
132              # but we must save and reset ALL fields
133              dynamo_config = torch._dynamo.config.get_config_copy()
134              inductor_config = torch._inductor.config.get_config_copy()
135              try:
136                  stderr = io.StringIO()
137                  log_handler = logging.StreamHandler(stderr)
138                  log = logging.getLogger("torch._dynamo")
139                  log.addHandler(log_handler)
140                  try:
141                      prev_cwd = _as_posix_path(os.getcwd())
142                      if cwd is not None:
143                          cwd = _as_posix_path(cwd)
144                          os.chdir(cwd)
145                      with patch("sys.argv", args), report_compile_source_on_error():
146                          exec(code, {"__name__": "__main__", "__compile_source__": code})
147                      rc = 0
148                  except Exception:
149                      rc = 1
150                      traceback.print_exc(file=stderr)
151                  finally:
152                      log.removeHandler(log_handler)
153                      if cwd is not None:
154                          os.chdir(prev_cwd)  # type: ignore[possibly-undefined]
155                      # Make sure we don't leave buggy compiled frames lying
156                      # around
157                      torch._dynamo.reset()
158              finally:
159                  torch._dynamo.config.load_config(dynamo_config)
160                  torch._inductor.config.load_config(inductor_config)
161  
162              # TODO: return a more appropriate data structure here
163              return subprocess.CompletedProcess(
164                  args,
165                  rc,
166                  b"",
167                  stderr.getvalue().encode("utf-8"),
168              )
169          else:
170              if cwd is not None:
171                  cwd = _as_posix_path(cwd)
172              return subprocess.run(args, capture_output=True, cwd=cwd, check=False)

174      # Run `code` in a separate python process.
175      # Returns the completed process state and the directory containing the
176      # minifier launcher script, if `code` outputted it.
177      def _run_test_code(self, code, *, isolate):
178          proc = self._maybe_subprocess_run(
179              ["python3", "-c", code], isolate=isolate, cwd=self.DEBUG_DIR
180          )
181  
182          print("test stdout:", proc.stdout.decode("utf-8"))
183          print("test stderr:", proc.stderr.decode("utf-8"))
184          repro_dir_match = re.search(
185              r"(\S+)minifier_launcher.py", proc.stderr.decode("utf-8")
186          )
187          if repro_dir_match is not None:
188              return proc, repro_dir_match.group(1)
189          return proc, None
190  
191      # Runs the minifier launcher script in `repro_dir`
192      def _run_minifier_launcher(
193          self, repro_dir, isolate, *, minifier_args=(), repro_after=None
194      ):
195          self.assertIsNotNone(repro_dir)
196          launch_file = _as_posix_path(os.path.join(repro_dir, "minifier_launcher.py"))
197          with open(launch_file) as f:
198              launch_code = f.read()
199          self.assertTrue(os.path.exists(launch_file))
200  
201          args = ["python3", launch_file, "minify", *minifier_args]
202          if not isolate and repro_after != "aot_inductor":
203              # AOTI minifier doesn't have --no-isolate flag.
204              # Everything in AOTI minifier is in no-isolate mode.
205              args.append("--no-isolate")
206          launch_proc = self._maybe_subprocess_run(args, isolate=isolate, cwd=repro_dir)
207          print("minifier stdout:", launch_proc.stdout.decode("utf-8"))
208          stderr = launch_proc.stderr.decode("utf-8")
209          print("minifier stderr:", stderr)
210          self.assertNotIn("Input graph did not fail the tester", stderr)
211  
212          return launch_proc, launch_code
213  
214      # Runs the repro script in `repro_dir`
215      def _run_repro(self, repro_dir, *, isolate=True):
216          self.assertIsNotNone(repro_dir)
217          repro_file = _as_posix_path(os.path.join(repro_dir, "repro.py"))
218          with open(repro_file) as f:
219              repro_code = f.read()
220          self.assertTrue(os.path.exists(repro_file))
221  
222          repro_proc = self._maybe_subprocess_run(
223              ["python3", repro_file], isolate=isolate, cwd=repro_dir
224          )
225          print("repro stdout:", repro_proc.stdout.decode("utf-8"))
226          print("repro stderr:", repro_proc.stderr.decode("utf-8"))
227          return repro_proc, repro_code

229      # Template for testing code.
230      # `run_code` is the code to run for the test case.
231      # `patch_code` is the code to be patched in every generated file; usually
232      # just use this to turn on bugs via the config
233      def _gen_test_code(self, run_code, repro_after, repro_level):
234          repro_after_line = ""
235          if repro_after == "aot_inductor":
236              repro_after_line = (
237                  "torch._inductor.config.aot_inductor.dump_aoti_minifier = True"
238              )
239          elif repro_after:
240              repro_after_line = f"""\ 
241  torch._dynamo.config.repro_after = "{repro_after}"
242          """
243          return f"""\ 
244  import torch
245  import torch._dynamo
246  import torch._inductor
247  {_as_posix_path(torch._dynamo.config.codegen_config())}
248  {_as_posix_path(torch._inductor.config.codegen_config())}
249  {repro_after_line}
250  torch._dynamo.config.repro_level = {repro_level}
251  torch._inductor.config.aot_inductor.repro_level = {repro_level}
252  torch._dynamo.config.debug_dir_root = "{_as_posix_path(self.DEBUG_DIR)}"
253  {run_code}
254  """

256      # Runs a full minifier test.
257      # Minifier tests generally consist of 3 stages:
258      # 1. Run the problematic code
259      # 2. Run the generated minifier launcher script
260      # 3. Run the generated repro script
261      #
262      # If possible, you should run the test with isolate=False; use
263      # isolate=True only if the bug you're testing would otherwise
264      # crash the process
265      def _run_full_test(
266          self, run_code, repro_after, expected_error, *, isolate, minifier_args=()
267      ) -> Optional[MinifierTestResult]:
268          if isolate:
269              repro_level = 3
270          elif expected_error is None or expected_error == "AccuracyError":
271              repro_level = 4
272          else:
273              repro_level = 2
274          test_code = self._gen_test_code(run_code, repro_after, repro_level)
275          print("running test", file=sys.stderr)
276          test_proc, repro_dir = self._run_test_code(test_code, isolate=isolate)
277          if expected_error is None:
278              # Just check that there was no error
279              self.assertEqual(test_proc.returncode, 0)
280              self.assertIsNone(repro_dir)
281              return None
282          # NB: Intentionally do not test return code; we only care about
283          # actually generating the repro, we don't have to crash
284          self.assertIn(expected_error, test_proc.stderr.decode("utf-8"))
285          self.assertIsNotNone(repro_dir)
286          print("running minifier", file=sys.stderr)
287          _minifier_proc, minifier_code = self._run_minifier_launcher(
288              repro_dir,
289              isolate=isolate,
290              minifier_args=minifier_args,
291              repro_after=repro_after,
292          )
293          print("running repro", file=sys.stderr)
294          repro_proc, repro_code = self._run_repro(repro_dir, isolate=isolate)
295          self.assertIn(expected_error, repro_proc.stderr.decode("utf-8"))
296          self.assertNotEqual(repro_proc.returncode, 0)
297          return MinifierTestResult(minifier_code=minifier_code, repro_code=repro_code)
