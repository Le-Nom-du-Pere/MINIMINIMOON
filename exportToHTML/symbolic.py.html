20   def _call_symbolic_op(
21       op_type: str,
22       domain: str,
23       args: Sequence[ir.Value | None],
24       kwargs: dict[str, int | float | str | bool | list[int] | list[float] | list[str]],
25       dtypes: Sequence[int],
26       version: int | None,
27       metadata_props: dict[str, str] | None,
28   ) -> Sequence[ir.Value]:
29       """Call an operator with the given arguments and keyword arguments. 
30    
31       Arguments are always inputs, while keyword arguments are attributes. 
32       """
33       # This is a wrapper around the IR node creation that hooks into the _builder.OpRecorder
34       # tracer so that all nodes created are recorded the same way as if we were to use
35       # onnxscript ops directly.
36   
37       assert _core.current_tracer is not None
38       tracer = _core.current_tracer
39   
40       inputs = list(args)
41   
42       # If final inputs are None, strip them from the node inputs
43       for input in reversed(inputs):
44           if input is not None:
45               break
46           inputs.pop()
47   
48       # Construct and filter out None attributes
49       attributes = [
50           attr
51           for attr in ir_convenience.convert_attributes(kwargs)  # type: ignore[arg-type]
52           if attr.value is not None  # type: ignore[union-attr]
53       ]
54       tracer.nodes.append(
55           node := ir.Node(
56               domain,
57               op_type,
58               inputs=inputs,
59               attributes=attributes,
60               num_outputs=len(dtypes),
61               version=version,
62               metadata_props=metadata_props,
63           )
64       )
65       # Set the dtypes for the outputs. We set them here because the graph builder
66       # Uses PyTorch types which are sometimes inaccurate when they are ONNX only
67       # types like float4e2m1.
68       for value, dtype in zip(node.outputs, dtypes):
69           value.dtype = ir.DataType(dtype)
70           # The shape is set by the graph builder. We don't need to set it here.
71       return node.outputs

74   @onnx_impl(torch.ops.onnx_symbolic._symbolic.default, no_compile=True)
75   def onnx_symbolic_symbolic(
76       inputs: Sequence[ir.Value | None],
77       op_type: str,
78       onnx_dtype: int,
79       *,
80       shape: Sequence[int | ir.Value],
81       attr_keys: Sequence[str],
82       attr_types: Sequence[str],
83       attr_pos: Sequence[tuple[int, int]],
84       attr_ints: Sequence[int],
85       attr_floats: Sequence[float],
86       attr_strs: Sequence[str],
87       metadata_props_keys: Sequence[str] = (),
88       metadata_props_values: Sequence[str] = (),
89       domain: str = "",
90       version: int | None = None,
91   ) -> ir.Value:
92       del shape  # Unused. The shapes are set by the graph builder
93       encoded = _symbolic_impl.EncodedAttrs(
94           attr_keys=list(attr_keys),
95           attr_types=list(attr_types),
96           attr_pos=list(attr_pos),
97           attr_ints=list(attr_ints),
98           attr_floats=list(attr_floats),
99           attr_strs=list(attr_strs),
100      )
101      attrs = encoded.to_dict()
102      return _call_symbolic_op(
103          op_type,
104          domain,
105          inputs,
106          attrs,
107          dtypes=[onnx_dtype],
108          version=version,
109          metadata_props=dict(zip(metadata_props_keys, metadata_props_values)),
110      )[0]
