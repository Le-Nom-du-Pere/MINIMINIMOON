15   class EntityType(Enum):
16       PERSON = "PERSON"
17       ORGANIZATION = "ORG"
18       GOVERNMENT = "GOVERNMENT"
19       POSITION = "POSITION"
20   
21   
22   @dataclass
23   class ResponsibilityEntity:
24       text: str
25       entity_type: EntityType
26       confidence: float
27       start_pos: int
28       end_pos: int
29       role: Optional[str] = None
30       is_government: bool = False
31   
32   
33   class ResponsibilityDetector:
34       """Detects responsibility entities in document segments using spaCy NER and lexical rules."""
35   
36       def __init__(self, model_name: str = "es_core_news_sm"):
37           """ 
38           Initialize the responsibility detector. 
39    
40           Args: 
41               model_name: spaCy model name (Spanish model recommended) 
42           """
43           self.nlp = self._load_spacy_pipeline(model_name)
44   
45           # Government institution patterns (high priority)
46           self.government_patterns = [
47               r"alcald[ií]a(?:\s+(?:de|del|municipal))?",
48               r"secretar[ií]a(?:\s+(?:de|del))?",
49               r"programa(?:\s+(?:de|del))?",
50               r"ministerio(?:\s+(?:de|del))?",
51               r"gobernaci[oó]n(?:\s+(?:de|del))?",
52               r"municipalidad",
53               r"ayuntamiento",
54               r"concejal[ií]a",
55               r"departamento\s+administrativo",
56               r"instituto\s+(?:nacional|municipal|departamental)",
57               r"direcci[oó]n\s+(?:nacional|municipal|departamental)",
58               r"superintendencia",
59               r"procuradur[ií]a",
60               r"defensor[ií]a\s+del\s+pueblo",
61               r"contralor[ií]a",
62           ]
63   
64           # Official position patterns
65           self.position_patterns = [
66               r"alcalde(?:sa)?",
67               r"secretari[oa](?:\s+(?:de|del))?",
68               r"ministr[oa](?:\s+(?:de|del))?",
69               r"director(?:a)?(?:\s+(?:general|ejecutiv[oa]|t[eé]cnic[oa]))?",
70               r"coordinador(?:a)?(?:\s+(?:de|del))?",
71               r"jefe(?:\s+(?:de|del))?",
72               r"responsable(?:\s+(?:de|del))?",
73               r"encargad[oa](?:\s+(?:de|del))?",
74               r"gerente(?:\s+(?:de|del))?",
75               r"presidente(?:a)?(?:\s+(?:de|del))?",
76               r"gobernador(?:a)?",
77               r"concejal(?:a)?",
78               r"comisionad[oa]",
79               r"subdirector(?:a)?",
80               r"viceministr[oa]",
81           ]
82   
83           # Generic institutional terms
84           self.institutional_patterns = [
85               r"entidad(?:\s+(?:p[úu]blica|gubernamental))?",
86               r"organizaci[oó]n(?:\s+(?:p[úu]blica|gubernamental))?",
87               r"instituci[oó]n(?:\s+(?:p[úu]blica|gubernamental))?",
88               r"dependencia(?:\s+(?:p[úu]blica|gubernamental))?",
89               r"oficina(?:\s+(?:p[úu]blica|gubernamental))?",
90               r"unidad(?:\s+(?:administrativa|t[eé]cnica))?",
91           ]
92   
93           # Contextual responsibility cues (increase confidence if in proximity)
94           self.context_cues = re.compile(
95               r"\b(responsable(?:s)?\s+de|a\s+cargo\s+de|competencia\s+de|deber(?:es)?\s+de|obligaci[oó]n\s+de|encargad[oa]\s+de|lidera(?:r|do)|coordina(?:r|ci[oó]n)|supervisa(?:r|ci[oó]n))\b",
96               re.IGNORECASE,
97           )
98   
99           # Compile patterns
100          self.compiled_gov_patterns = [
101              re.compile(p, re.IGNORECASE) for p in self.government_patterns
102          ]
103          self.compiled_pos_patterns = [
104              re.compile(p, re.IGNORECASE) for p in self.position_patterns
105          ]
106          self.compiled_inst_patterns = [
107              re.compile(p, re.IGNORECASE) for p in self.institutional_patterns
108          ]
109  
110          # Pre-compile normalization for government keyword check
111          self.gov_keyword_re = re.compile(
112              r"\b(alcald[ií]a|secretar[ií]a|programa|ministerio|gobernaci[oó]n|municipalidad|ayuntamiento|concejal[ií]a|instituto|direcci[oó]n|departamento|nacional|municipal|departamental|superintendencia|procuradur[ií]a|defensor[ií]a\s+del\s+pueblo|contralor[ií]a)\b",
113              re.IGNORECASE,
114          )

116      # ---------- Public API ----------
117  
118      def detect_entities(self, text: str) -> List[ResponsibilityEntity]:
119          """ 
120          Detect responsibility entities in text using NER and lexical rules. 
121   
122          Args: 
123              text: Input text segment 
124   
125          Returns: 
126              List of ResponsibilityEntity objects with confidence scores 
127          """
128          safe_text = text if isinstance(text, str) else ""
129          doc = self.nlp(safe_text)
130  
131          entities: List[ResponsibilityEntity] = []
132          entities.extend(self._extract_ner_entities(doc))
133          entities.extend(self._extract_pattern_entities(safe_text))
134  
135          entities = self._merge_overlapping_entities(
136              self._dedupe_entities(entities))
137          entities = self._calculate_final_scores(safe_text, entities)
138  
139          return sorted(entities, key=lambda x: x.confidence, reverse=True)
140  
141      def calculate_responsibility_score(self, text: str) -> Dict:
142          """ 
143          Calculate overall responsibility score for a document segment. 
144   
145          Args: 
146              text: Document segment text 
147   
148          Returns: 
149              Dictionary with responsibility assessment 
150          """
151          entities = self.detect_entities(text)
152  
153          gov_score = sum(e.confidence for e in entities if e.is_government)
154          person_score = sum(
155              e.confidence for e in entities if e.entity_type == EntityType.PERSON
156          )
157          position_score = sum(
158              e.confidence for e in entities if e.entity_type == EntityType.POSITION
159          )
160          org_score = sum(
161              e.confidence
162              for e in entities
163              if e.entity_type == EntityType.ORGANIZATION and not e.is_government
164          )
165  
166          # Weighted linear combination; cap by theoretical max (2.0 per entity as en el diseño original)
167          total_score = (
168              gov_score * 2.0
169              + position_score * 1.5
170              + person_score * 1.0
171              + org_score * 0.8
172          )
173          max_possible = len(entities) * 2.0 if entities else 1.0
174          factibility_score = min(1.0, total_score / max_possible)
175  
176          return {
177              "factibility_score": factibility_score,
178              "entities": entities,
179              "breakdown": {
180                  "government_score": gov_score,
181                  "position_score": position_score,
182                  "person_score": person_score,
183                  "organization_score": org_score,
184                  "total_entities": len(entities),
185              },
186              "has_government_entities": any(e.is_government for e in entities),
187              "has_official_positions": any(
188                  e.entity_type == EntityType.POSITION for e in entities
189              ),
190          }

194      @staticmethod
195      def _load_spacy_pipeline(preferred: str):
196          """ 
197          Load spaCy pipeline with robust fallbacks while preserving the same constructor signature. 
198          """
199          tried = []
200          candidates = [
201              preferred,
202              "es_core_news_md",
203              "es_core_news_lg",
204              "xx_ent_wiki_sm",
205              "xx_sent_ud_sm",
206          ]
207          for name in candidates:
208              try:
209                  nlp = spacy.load(name)
210                  # Disable unnecessary components if present to speed up (keeps NER)
211                  for pipe in [
212                      "lemmatizer",
213                      "morphologizer",
214                      "attribute_ruler",
215                      "tagger",
216                      "parser",
217                      "senter",
218                  ]:
219                      if pipe in nlp.pipe_names and pipe != "ner":
220                          try:
221                              nlp.disable_pipe(pipe)
222                          except Exception:
223                              pass
224                  return nlp
225              except Exception as e:
226                  tried.append(f"{name}: {type(e).__name__}")
227                  continue
228          # Ultimate fallback: blank Spanish with an empty NER (will rely on rules)
229          nlp = spacy.blank("es")
230          return nlp

232      def _extract_ner_entities(self, doc) -> List[ResponsibilityEntity]:
233          """Extract PERSON and ORG entities from spaCy NER."""
234          entities: List[ResponsibilityEntity] = []
235          if not hasattr(doc, "ents"):
236              return entities
237  
238          for ent in doc.ents:
239              label = ent.label_
240              if label in ("PERSON", "PER"):
241                  entities.append(
242                      ResponsibilityEntity(
243                          text=ent.text,
244                          entity_type=EntityType.PERSON,
245                          confidence=0.72,  # slightly tuned base
246                          start_pos=ent.start_char,
247                          end_pos=ent.end_char,
248                      )
249                  )
250              elif label == "ORG":
251                  text = ent.text
252                  is_gov = self._is_government_entity(text)
253                  entities.append(
254                      ResponsibilityEntity(
255                          text=text,
256                          entity_type=(
257                              EntityType.GOVERNMENT if is_gov else EntityType.ORGANIZATION
258                          ),
259                          confidence=0.82 if is_gov else 0.62,
260                          start_pos=ent.start_char,
261                          end_pos=ent.end_char,
262                          is_government=is_gov,
263                      )
264                  )
265          return entities

267      def _extract_pattern_entities(self, text: str) -> List[ResponsibilityEntity]:
268          """Extract entities using lexical patterns."""
269          entities: List[ResponsibilityEntity] = []
270  
271          # Government institutions (highest priority)
272          for pattern in self.compiled_gov_patterns:
273              for m in pattern.finditer(text):
274                  entities.append(
275                      ResponsibilityEntity(
276                          text=m.group(),
277                          entity_type=EntityType.GOVERNMENT,
278                          confidence=0.90,
279                          start_pos=m.start(),
280                          end_pos=m.end(),
281                          is_government=True,
282                      )
283                  )
284  
285          # Official positions
286          for pattern in self.compiled_pos_patterns:
287              for m in pattern.finditer(text):
288                  entities.append(
289                      ResponsibilityEntity(
290                          text=m.group(),
291                          entity_type=EntityType.POSITION,
292                          confidence=0.80,
293                          start_pos=m.start(),
294                          end_pos=m.end(),
295                          role=m.group(),
296                          is_government=True,  # cargos públicos implican poder/deber institucional
297                      )
298                  )
299  
300          # Generic institutional terms (lower priority)
301          for pattern in self.compiled_inst_patterns:
302              for m in pattern.finditer(text):
303                  entities.append(
304                      ResponsibilityEntity(
305                          text=m.group(),
306                          entity_type=EntityType.ORGANIZATION,
307                          confidence=0.50,
308                          start_pos=m.start(),
309                          end_pos=m.end(),
310                          is_government=False,
311                      )
312                  )
313  
314          return entities

316      @staticmethod
317      def _strip_accents(s: str) -> str:
318          return "".join(
319              c
320              for c in unicodedata.normalize("NFD", s)
321              if unicodedata.category(c) != "Mn"
322          )
323  
324      def _is_government_entity(self, text: str) -> bool:
325          """Check if an entity is likely a government organization."""
326          t = self._strip_accents(text.lower())
327          return bool(self.gov_keyword_re.search(t))

329      @staticmethod
330      def _span_iou(a: Tuple[int, int], b: Tuple[int, int]) -> float:
331          """Intersection-over-Union for character spans."""
332          a0, a1 = a
333          b0, b1 = b
334          inter = max(0, min(a1, b1) - max(a0, b0))
335          union = max(a1, b1) - min(a0, b0)
336          return (inter / union) if union > 0 else 0.0
337  
338      @staticmethod
339      def _normalize_text_for_key(s: str) -> str:
340          s = s.strip()
341          s = unicodedata.normalize("NFKC", s)
342          s = s.lower()
343          s = "".join(
344              c
345              for c in unicodedata.normalize("NFD", s)
346              if unicodedata.category(c) != "Mn"
347          )
348          return re.sub(r"\s+", " ", s)

350      def _dedupe_entities(
351          self, entities: List[ResponsibilityEntity]
352      ) -> List[ResponsibilityEntity]:
353          """Deduplicate by normalized text and dominant span; keep highest-confidence instance."""
354          bucket: Dict[str, ResponsibilityEntity] = {}
355          for e in entities:
356              key = f"{e.entity_type.value}:{self._normalize_text_for_key(e.text)}"
357              if key not in bucket or e.confidence > bucket[key].confidence:
358                  bucket[key] = e
359          return list(bucket.values())
360  
361      def _merge_overlapping_entities(
362          self, entities: List[ResponsibilityEntity]
363      ) -> List[ResponsibilityEntity]:
364          """Merge overlapping entities, keeping the most responsible-leaning candidate with IoU-aware logic."""
365          if not entities:
366              return entities
367  
368          entities = sorted(entities, key=lambda x: (x.start_pos, -x.end_pos))
369          merged: List[ResponsibilityEntity] = []
370          current = entities[0]
371  
372          def priority_score(e: ResponsibilityEntity) -> Tuple[int, float]:
373              # Prefer GOVERNMENT > POSITION > ORGANIZATION > PERSON, then confidence
374              order = {
375                  EntityType.GOVERNMENT: 3,
376                  EntityType.POSITION: 2,
377                  EntityType.ORGANIZATION: 1,
378                  EntityType.PERSON: 0,
379              }
380              return (order.get(e.entity_type, 0), e.confidence)
381  
382          for nxt in entities[1:]:
383              iou = self._span_iou(
384                  (current.start_pos, current.end_pos), (nxt.start_pos, nxt.end_pos)
385              )
386              if iou > 0.0:
387                  # Choose by priority; if tie, higher confidence
388                  cur_p, nxt_p = priority_score(current), priority_score(nxt)
389                  if nxt_p > cur_p:
390                      current = nxt
391                  elif nxt_p == cur_p and nxt.confidence > current.confidence:
392                      current = nxt
393                  else:
394                      # keep current; ignore nxt
395                      pass
396              else:
397                  merged.append(current)
398                  current = nxt
399  
400          merged.append(current)
401          return merged

403      def _calculate_final_scores(
404          self, text: str, entities: List[ResponsibilityEntity]
405      ) -> List[ResponsibilityEntity]:
406          """Calculate final confidence scores considering entity types, patterns, form, and contextual cues."""
407          # Precompute context cue windows to cheaply boost nearby entities
408          cue_spans = [m.span() for m in self.context_cues.finditer(text)]
409  
410          def near_cue(e: ResponsibilityEntity, window: int = 40) -> bool:
411              for c0, c1 in cue_spans:
412                  if abs(e.start_pos - c1) <= window or abs(e.end_pos - c0) <= window:
413                      return True
414              return False
415  
416          for e in entities:
417              base = e.confidence
418  
419              # Boost government entities
420              if e.is_government:
421                  base *= 1.20
422  
423              # Boost official positions
424              if e.entity_type == EntityType.POSITION:
425                  base *= 1.12
426  
427              # Slight penalty for generic non-gov organizations
428              if e.entity_type == EntityType.ORGANIZATION and not e.is_government:
429                  base *= 0.90
430  
431              # Form-based calibration: title-case and token length (reduce false short hits)
432              token_len = max(1, len(e.text.strip().split()))
433              if token_len >= 3:
434                  base *= 1.05
435              elif token_len == 1 and e.entity_type in (
436                  EntityType.ORGANIZATION,
437                  EntityType.GOVERNMENT,
438              ):
439                  base *= 0.92
440  
441              # Title/Proper-casing heuristic (e.g., "Ministerio de Educación")
442              if re.match(r"^[A-ZÁÉÍÓÚÑ][^\n]+", e.text.strip()):
443                  base *= 1.03
444  
445              # Contextual cue proximity
446              if near_cue(e):
447                  base *= 1.08
448  
449              # Cap to [0.1, 1.0] to keep scores sane and comparable
450              e.confidence = max(0.10, min(1.00, base))
451  
452          return entities
