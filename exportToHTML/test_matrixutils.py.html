36   def test_matrix_tensor_product():
37       if not np:
38           skip("numpy not installed.")
39   
40       l1 = zeros(4)
41       for i in range(16):
42           l1[i] = 2**i
43       l2 = zeros(4)
44       for i in range(16):
45           l2[i] = i
46       l3 = zeros(2)
47       for i in range(4):
48           l3[i] = i
49       vec = Matrix([1, 2, 3])
50   
51       #test for Matrix known 4x4 matrices
52       numpyl1 = np.array(l1.tolist())
53       numpyl2 = np.array(l2.tolist())
54       numpy_product = np.kron(numpyl1, numpyl2)
55       args = [l1, l2]
56       sympy_product = matrix_tensor_product(*args)
57       assert numpy_product.tolist() == sympy_product.tolist()
58       numpy_product = np.kron(numpyl2, numpyl1)
59       args = [l2, l1]
60       sympy_product = matrix_tensor_product(*args)
61       assert numpy_product.tolist() == sympy_product.tolist()
62   
63       #test for other known matrix of different dimensions
64       numpyl2 = np.array(l3.tolist())
65       numpy_product = np.kron(numpyl1, numpyl2)
66       args = [l1, l3]
67       sympy_product = matrix_tensor_product(*args)
68       assert numpy_product.tolist() == sympy_product.tolist()
69       numpy_product = np.kron(numpyl2, numpyl1)
70       args = [l3, l1]
71       sympy_product = matrix_tensor_product(*args)
72       assert numpy_product.tolist() == sympy_product.tolist()
73   
74       #test for non square matrix
75       numpyl2 = np.array(vec.tolist())
76       numpy_product = np.kron(numpyl1, numpyl2)
77       args = [l1, vec]
78       sympy_product = matrix_tensor_product(*args)
79       assert numpy_product.tolist() == sympy_product.tolist()
80       numpy_product = np.kron(numpyl2, numpyl1)
81       args = [vec, l1]
82       sympy_product = matrix_tensor_product(*args)
83       assert numpy_product.tolist() == sympy_product.tolist()
84   
85       #test for random matrix with random values that are floats
86       random_matrix1 = np.random.rand(randint(1, 5), randint(1, 5))
87       random_matrix2 = np.random.rand(randint(1, 5), randint(1, 5))
88       numpy_product = np.kron(random_matrix1, random_matrix2)
89       args = [Matrix(random_matrix1.tolist()), Matrix(random_matrix2.tolist())]
90       sympy_product = matrix_tensor_product(*args)
91       assert not (sympy_product - Matrix(numpy_product.tolist())).tolist() > \
92           (ones(sympy_product.rows, sympy_product.cols)*epsilon).tolist()
93   
94       #test for three matrix kronecker
95       sympy_product = matrix_tensor_product(l1, vec, l2)
96   
97       numpy_product = np.kron(l1, np.kron(vec, l2))
98       assert numpy_product.tolist() == sympy_product.tolist()
