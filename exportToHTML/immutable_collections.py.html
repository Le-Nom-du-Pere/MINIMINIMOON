36   def _no_mutation(self: Any, *args: Any, **kwargs: Any) -> NoReturn:
37       raise TypeError(
38           f"{type(self).__name__!r} object does not support mutation. {_help_mutation}",
39       )

42   @compatibility(is_backward_compatible=True)
43   class immutable_list(list[_T]):
44       """An immutable version of :class:`list`."""
45   
46       __delitem__ = _no_mutation
47       __iadd__ = _no_mutation
48       __imul__ = _no_mutation
49       __setitem__ = _no_mutation
50       append = _no_mutation
51       clear = _no_mutation
52       extend = _no_mutation
53       insert = _no_mutation
54       pop = _no_mutation
55       remove = _no_mutation
56       reverse = _no_mutation
57       sort = _no_mutation
58   
59       def __hash__(self) -> int:  # type: ignore[override]
60           return hash(tuple(self))
61   
62       def __reduce__(self) -> tuple[type[Self], tuple[tuple[_T, ...]]]:
63           return (type(self), (tuple(self),))

66   @compatibility(is_backward_compatible=True)
67   class immutable_dict(dict[_KT, _VT]):
68       """An immutable version of :class:`dict`."""
69   
70       __delitem__ = _no_mutation
71       __ior__ = _no_mutation
72       __setitem__ = _no_mutation
73       clear = _no_mutation
74       pop = _no_mutation
75       popitem = _no_mutation
76       setdefault = _no_mutation
77       update = _no_mutation  # type: ignore[assignment]
78   
79       def __hash__(self) -> int:  # type: ignore[override]
80           return hash(frozenset(self.items()))
81   
82       def __reduce__(self) -> tuple[type[Self], tuple[tuple[tuple[_KT, _VT], ...]]]:
83           return (type(self), (tuple(self.items()),))
