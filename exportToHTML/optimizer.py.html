3    """ 
4    This module implements variable tracking for PyTorch optimizers during Dynamo tracing. 
5     
6    The OptimizerVariable class provides specialized handling for optimizer instances by: 
7    - Optimizing the tracing of expensive optimizer initialization 
8    - Managing optimizer state and parameter group tracking 
9    - Handling tensor sources and guards for optimizer state tensors 
10   - Supporting CUDA graph execution through static tensor address management 
11   - Providing special handling for parameter gradients and optimizer state tensors 
12    
13   Key features include: 
14   - Efficient initialization tracing via _init_group optimization 
15   - Automatic marking of optimizer state tensors as static for CUDA graphs 
16   - Proper source tracking for parameter groups, gradients, and state tensors 
17   - Guard installation for optimizer state structure 
18   - Support for both CPU and GPU tensor handling 
19   - Cleanup of static tensor references via finalizers 
20    
21   The module integrates with Dynamo's broader tracing system while providing 
22   optimizer-specific optimizations and safety guarantees. 
23   """

66   def _is_static_for_cudagraphs(x):
67       from torch._inductor.cudagraph_trees import get_manager
68   
69       if x.is_cuda:
70           manager = get_manager(x.device.index, False)
71           is_static_address = torch._dynamo.utils.get_static_address_type(x) is not None
72           if manager:
73               return (
74                   is_static_address
75                   or manager.current_node._is_cuda_graph_recorded_tensor(x)
76               )
77           else:
78               return is_static_address
79       else:
80           # Don't print a warning for non-cuda tensors
81           return True

85       _nonvar_fields = {

92       def __init__(
93           self,
94           value,
95           grad_to_source=None,
96           static_tensor_names=None,
97           tensor_to_source=None,
98           **kwargs,
99       ) -> None:
100          super().__init__(value, **kwargs)
101          self.grad_to_source = grad_to_source or {}
102          self.tensor_to_source = tensor_to_source or {}
103          self.static_tensor_names = static_tensor_names or set()

105      def call_method(
106          self,
107          tx,
108          name,
109          args: "list[VariableTracker]",
110          kwargs: "dict[str, VariableTracker]",
111      ) -> "VariableTracker":
112          """This is an optimization to avoid tracing the very slow initialization of the optimizer"""
113          if name == "_init_group":
114              try:
115                  self.graph_break_if_pending_mutation(tx)
116                  self.move_step_if_cpu()
117                  py_args, py_kwargs = self.get_python_args(*args, **kwargs)
118                  ret_val = self.value._init_group(*py_args, **py_kwargs)
119                  self.map_sources_and_install_guards(tx)
120                  self.update_list_args(tx, args, kwargs, py_args, py_kwargs)
121                  # stash a weak_ptr to optimizer to invalidate code
122                  # if the optimizer object dies
123                  mangled_name = f"__optimizer_{id(self.value)}"
124                  tx.store_global_weakref_by_id(mangled_name, self.value)
125                  self.create_finalizer(tx)
126  
127                  # This is currently safe only because the only actual `ret_val`s returned
128                  # by the `_init_group` of existing optimizers are properties that are invariant
129                  # to the input tensors (e.g. dtype, layout). Changing these would trigger a
130                  # recompilation and hence never result in the wrong specialization of `ret_val`.
131                  return ConstantVariable.create(ret_val)
132              except (ArgMappingException, GuardInstallException) as _:
133                  # trace normally if we can't map args or install guards correctly
134                  pass
135  
136          return super().call_method(tx, name, args, kwargs)

138      def var_getattr(self, tx: "InstructionTranslator", name):
139          # Note: this allows us to intercept the call in call_method
140          # in the typical case, we return a UserMethodVariable
141          # which will directly inline
142          if name in ("_init_group", "step"):
143              return GetAttrVariable(self, name, source=AttrSource(self.source, name))
144  
145          if name == "param_groups":
146              from ..decorators import mark_static_address
147  
148              for group in self.value.param_groups:
149                  for p in group["params"]:
150                      mark_static_address(p)
151  
152              self._set_capturable(tx)
153  
154          return super().var_getattr(tx, name)

156      def graph_break_if_pending_mutation(self, tx):
157          # If there are pending mutations on a parameter (due to using closure)
158          # then we need to graph break to allow the python version of the parameter
159          # to update, so that running _init_group will initialize the states with
160          # the correct values
161          for g in self.value.param_groups:
162              for p in g["params"]:
163                  side_effects = tx.output.side_effects
164                  variable = side_effects.id_to_variable.get(id(p), None)
165                  if variable and side_effects.has_pending_mutation(variable):
166                      from ..exc import Unsupported
167  
168                      raise Unsupported("Pending mutation on parameter")

170      def _set_capturable(self, tx):
171          from . import LazyVariableTracker
172  
173          # We only set capturable if params are on cuda
174          # and the state is not initialized
175          def safe_to_set_capturable(group):
176              all_uninitialized = True
177              all_gpu = True
178  
179              for p in group.get("params", []):
180                  all_gpu &= p.is_cuda or p.is_xpu
181                  all_uninitialized &= p not in self.value.state
182  
183              return "capturable" in group and all_uninitialized and all_gpu
184  
185          # track indices to not set so we don't need to
186          # in the variable tracker realize the whole state
187          # we handle guarding the state specially
188          for group in self.value.param_groups:
189              if safe_to_set_capturable(group):
190                  group["capturable"] = True
191  
192          source = self.source and AttrSource(self.source, "param_groups")
193          param_groups_vt = LazyVariableTracker.realize_all(
194              VariableTracker.build(tx, self.value.param_groups, source)
195          )
196          for param_group_vt in param_groups_vt.items:
197              key = ConstDictVariable._HashableTracker(
198                  ConstantVariable.create("capturable")
199              )
200              param_group_vt.items[key] = ConstantVariable.create(True)
201  
202      def get_python_args(self, *args, **kwargs):
203          """Get python values equivalent to the variable tracker args"""
204  
205          def map_arg(arg):
206              if isinstance(arg, ConstantVariable):
207                  return arg.as_python_constant()
208              elif isinstance(arg, ListVariable) and not arg.items:
209                  return []
210              elif (
211                  isinstance(arg, ConstDictVariable)
212                  and isinstance(arg.source, GetItemSource)
213                  and isinstance(arg.source.base, AttrSource)
214                  and arg.source.base.member == "param_groups"
215              ):
216                  return self.value.param_groups[arg.source.index]
217  
218              raise ArgMappingException
219  
220          new_args = [map_arg(arg) for arg in args]
221          new_kwargs = {k: map_arg(v) for k, v in kwargs.items()}
222  
223          return new_args, new_kwargs

225      # If users load an old state dictionary,
226      # it's possible that step could be on the cpu
227      # if this is the case, move it to the GPU
228      # corresponding to the parameter
229      # in most cases this is a no-op because the state is empty
230      def move_step_if_cpu(self):
231          for p, state in self.value.state.items():
232              if "step" in state and state["step"].is_cpu:
233                  state["step"] = state["step"].to(p.device)

235      def map_sources_and_install_guards(self, tx):
236          from ..decorators import mark_static_address
237          from .lazy import LazyVariableTracker
238  
239          self.grad_to_source = {}
240          self.tensor_to_source = {}
241  
242          # Tracing the _init_group is expensive. But we still have to insert the
243          # necessary guards for _init_group. So, we manually handle insertion of
244          # guards. We also want to mark all the tensors inside the state dict to
245          # be static address.
246  
247          # Mark all the tensors in the state dict to be static address. This has
248          # to be done first because the variable builder relies on the static
249          # address annotation.
250          def mark_static(x):
251              mark_static_address(x)
252  
253          tree_map_only(torch.Tensor, mark_static, self.value.state)
254  
255          # Recursively realize the variable trackers for optim.state and
256          # optim.param_groups, which recursively install the necessary guards.
257          params_groups_source = self.source and AttrSource(self.source, "param_groups")
258          param_groups_vt = LazyVariableTracker.realize_all(
259              VariableTracker.build(tx, self.value.param_groups, params_groups_source)
260          )
261  
262          state_source = self.source and AttrSource(self.source, "state")
263  
264          state_vt = VariableTracker.build(tx, self.value.state, state_source)
265  
266          # We need to realize the top level state dict to populate
267          # the guard locals
268          state_vt.realize()
269          tx.output.guard_on_key_order.add(state_source)
270  
271          # Populate self.grad_to_source and self.tensor_to_source so that we can
272          # manually update_list_args
273          for group, group_vt in zip(self.value.param_groups, param_groups_vt.items):
274              # we assume here that all params within a param group
275              # are initialized similarly
276              if len(group["params"]) > 0:
277                  for param in group["params"]:
278                      if param.grad is not None:
279                          key_index = None
280                          for i, k in enumerate(self.value.state.keys()):
281                              if k is param:
282                                  key_index = i
283                                  break
284                          if key_index:
285                              LazyVariableTracker.realize_all(
286                                  VariableTracker.build(
287                                      tx,
288                                      self.value.state[param],
289                                      DictGetItemSource(
290                                          state_source,
291                                          ConstDictKeySource(state_source, key_index),
292                                      ),
293                                  )
294                              )
295                              break
296  
297              params_vt = group_vt.getitem_const(tx, ConstantVariable.create("params"))
298              all_static = True
299              non_static_grads = []
300              for p_ind, (p, p_vt) in enumerate(
301                  zip(group["params"], params_vt.unpack_var_sequence(tx))
302              ):
303                  param_source = p_vt.source
304                  self.tensor_to_source[p] = param_source
305                  grad_source = GradSource(
306                      param_source,
307                      "grad",
308                  )
309  
310                  if p.grad is not None:
311                      self.grad_to_source[p.grad] = grad_source
312                      if not _is_static_for_cudagraphs(p.grad):
313                          all_static = False
314                          non_static_grads.append(grad_source)
315                  else:
316                      install_guard(grad_source.make_guard(GuardBuilder.CONSTANT_MATCH))
317  
318              # Note: to avoid spam logs only warn if perf hint artifact is enabled
319              # (NB: artifacts are only enabled at the debug or warning level)
320              if not all_static and perf_hint_log.isEnabledFor(logging.DEBUG):
321                  non_static_grads = [src.name() for src in non_static_grads]
322                  perf_hint_log.warning(
323                      (
324                          "Grad tensors %s will be copied during cudagraphs execution."
325                          "If using cudagraphs and the grad tensor addresses will be the same across runs,"
326                          " use torch._dynamo.decorators.mark_static_address to elide this copy.",
327                      ),
328                      non_static_grads,
329                  )
330  
331          # We have to again iterate over the state dict to collect the
332          # tensor_to_source dict. This is used for the finalizer.
333          for idx, (p, value) in enumerate(self.value.state.items()):
334              p_state_source = DictGetItemSource(
335                  state_source, ConstDictKeySource(state_source, idx)
336              )
337              tx.output.guard_on_key_order.add(p_state_source)
338              for inner_idx, (k, v) in enumerate(value.items()):
339                  if (
340                      isinstance(v, torch.Tensor)
341                      and v not in self.grad_to_source
342                      and v not in self.tensor_to_source
343                  ):
344                      self.tensor_to_source[v] = DictGetItemSource(
345                          p_state_source, ConstDictKeySource(p_state_source, inner_idx)
346                      )

348      def wrap_tensor(self, tx: "InstructionTranslator", tensor_value):
349          """Wrap state tensor in a TensorVariable"""
350          from ..decorators import mark_static_address
351  
352          # If we have a source for a tensor already use it,
353          # if we have not seen a tensor before, stash and use a
354          # global weak ref source, since it must be an optimizer tensor
355          # that we have missed
356  
357          if tensor_value in self.tensor_to_source:
358              # mark these tensors as static for cudagraphs
359              mark_static_address(tensor_value)
360              source = self.tensor_to_source[tensor_value]
361              self.static_tensor_names.add(tx.output.module_key_name(source.name()))
362          elif tensor_value in self.grad_to_source:
363              source = self.grad_to_source[tensor_value]
364          else:
365              # mark these tensors as static for cudagraphs
366              mark_static_address(tensor_value)
367  
368              global_name = tx.store_global_weakref_by_id(GLOBAL_KEY_PREFIX, tensor_value)
369              source = GlobalWeakRefSource(global_name)
370              self.static_tensor_names.add(tx.output.module_key_name(source.name()))
371  
372          return VariableTracker.build(tx, tensor_value, source)

374      def update_list_args(
375          self, tx: "InstructionTranslator", args, kwargs, py_args, py_kwargs
376      ):
377          """Update the args and kwargs to the traced optimizer call"""
378          for arg, py_arg in zip(args, py_args):
379              if isinstance(arg, ListVariable):
380                  assert isinstance(py_arg, list), (
381                      "py_arg should be a list in optimizer variable"
382                  )
383                  for i, val in enumerate(py_arg):
384                      tx.output.side_effects.mutation(arg)
385                      if isinstance(val, torch.Tensor):
386                          arg.items.append(self.wrap_tensor(tx, val))
387                      else:
388                          source = arg.source and GetItemSource(arg.source, i)
389                          arg.items.append(VariableTracker.build(tx, val, source))

391      def create_finalizer(self, tx):
392          names_to_delete = self.static_tensor_names
393          value = self.value
394          tc = tx.output.tracing_context
395  
396          def init_finalizer(gm):
397              def clear_static_tensor_refs():
398                  for name in names_to_delete:
399                      gm._buffers.pop(name, None)
400                      gm._parameters.pop(name, None)
401                      if tc.params_flat:
402                          tc.params_flat.clear()
403                      if tc.params_flat_unwrap_subclasses:
404                          tc.params_flat_unwrap_subclasses.clear()
405  
406              weakref.finalize(value, clear_static_tensor_refs)
407  
408          tx.output.add_graph_finalizer(init_finalizer)
