16   /** 
17    * Note [CUDA Graph-safe RNG states] 
18    * ~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~~ 
19    * 
20    * Strategy: 
21    * ~~~~~~~~~ 
22    * (It helps to look at 
23    * cuda/detail/PhiloxCudaStateRaw.cuh and 
24    * cuda/detail/UnpackRaw.cuh 
25    * while you read this.) 
26    * 
27    * A CUDA graph containing multiple RNG ops behaves like a 
28    * single giant kernel from the perspective of ops external 
29    * to the graph.  During graph capture, logic in CUDAGeneratorImpl 
30    * records the total of all offset increments that occur in the 
31    * graphed region, and records the final total as the offset for 
32    * the entire graph. 
33    * 
34    * When the graph reruns, the logic that reruns it 
35    * increments this device's CUDA generator's offset 
36    * by that total. 
37    * 
38    * Meanwhile, within the graph, at capture time, instead of 
39    * populating PhiloxCudaStates with the uint64_t offset pulled 
40    * directly from the global state, PhiloxCudaState uses a pointer 
41    * to a one-element stream-local int64_t device tensor 
42    * holding an initial offset value, and a uint64_t holding an 
43    * intra-graph offset. (The intra-graph offset starts from zero 
44    * when capture begins.)  In each consumer kernel, 
45    * at::cuda::philox::unpack computes the offset to use for this kernel 
46    * as intra-graph offset + *initial offset. 
47    * 
48    * When the graph reruns, the logic that reruns it first 
49    * fill_s the initial offset tensor with this device's 
50    * CUDA generator's current offset. 
51    * 
52    * The control flow above ensures graphed execution is bitwise 
53    * identical to eager execution as long as RNG ops are enqueued 
54    * from a single thread, even if RNG ops and graphs containing 
55    * RNG ops are enqueued and run simultaneously on multiple streams. 
56    * 
57    * Usage: 
58    * ~~~~~~ 
59    * PhiloxCudaState in this file, and unpack() in 
60    * cuda/CUDAGraphsUtils.cuh allow non-divergent use of 
61    * CUDAGeneratorImpl whether graph capture is underway or not. 
62    * 
63    * Each PhiloxCudaState instance should be used for one and only one 
64    * consumer kernel. 
65    * 
66    * Example (see e.g. native/cuda/Dropout.cu): 
67    * 
68    * #include <ATen/cuda/CUDAGeneratorImpl.h> 
69    * #include <ATen/cuda/CUDAGraphsUtils.cuh> 
70    * 
71    * __global__ void kernel(..., PhiloxCudaState philox_args) { 
72    *   auto seeds = at::cuda::philox::unpack(philox_args); 
73    *   IndexType idx = blockIdx.x * blockDim.x + threadIdx.x; 
74    *   curandStatePhilox4_32_10_t state; 
75    *   curand_init(std::get<0>(seeds), // seed 
76    *               idx,                // per-thread subsequence 
77    *               std::get<1>(seeds), // offset in subsequence 
78    *               &state); 
79    *   ... 
80    * } 
81    * 
82    * host_caller(...) { 
83    *   PhiloxCudaState rng_engine_inputs; 
84    *   { 
85    *     // See Note [Acquire lock when using random generators] 
86    *     std::lock_guard<std::mutex> lock(gen->mutex_); 
87    * 
88    *     // gen could be HostState or DevState here! No divergent code needed! 
89    *     rng_engine_inputs = gen->philox_cuda_state(offset_increment); 
90    *   } 
91    *   kernel<<<...>>>(..., rng_engine_inputs); 
92    * } 
93    * 
94    */
95   
96   struct CUDAGeneratorState : public c10::intrusive_ptr_target {
97     uint64_t seed_;
98     uint64_t philox_offset_per_thread_;
99     uint32_t offset_intragraph_;
100    bool capturing_{};
101    std::unordered_set<cuda::CUDAGraph*> registered_graphs_;
102    at::TensorBase seed_extragraph_{};
103    at::TensorBase offset_extragraph_{};
104  
105    CUDAGeneratorState(
106        uint64_t seed = default_rng_seed_val,
107        uint64_t philox_offset_per_thread = 0,
108        uint32_t offset_intragraph = 0)
109        : seed_(seed),
110          philox_offset_per_thread_(philox_offset_per_thread),
111          offset_intragraph_(offset_intragraph) {}
112  
113    void increase(uint64_t increment);
114  
115    void register_graph(cuda::CUDAGraph* graph);
116    void unregister_graph(cuda::CUDAGraph* graph);
117  
118    void capture_prologue();
119    // capture_epilogue returns the wholegraph_increment
120    uint64_t capture_epilogue();
121    void replay_prologue(uint64_t wholegraph_increment);
122    c10::intrusive_ptr<CUDAGeneratorState> clone();
123  };
