46   class CUDAGraph(torch._C._CUDAGraph):
47       r"""Wrapper around a CUDA graph. 
48    
49       Arguments: 
50           keep_graph (bool, optional): If ``keep_graph=False``, the 
51               cudaGraphExec_t will be instantiated on GPU at the end of 
52               ``capture_end`` and the underlying cudaGraph_t will be 
53               destroyed. Users who want to query or otherwise modify the 
54               underlying cudaGraph_t before instantiatiation can set 
55               ``keep_graph=True`` and access it via ``raw_cuda_graph`` after 
56               ``capture_end``. Note that the cudaGraphExec_t will not be 
57               instantiated at the end of ``capture_end`` in this 
58               case. Instead, it wil be instantiated via an explicit called 
59               to ``instantiate`` or automatically on the first call to 
60               ``replay`` if ``instantiate`` was not already called. Calling 
61               ``instantiate`` manually before ``replay`` is recommended to 
62               prevent increased latency on the first call to ``replay``. It 
63               is allowed to modify the raw cudaGraph_t after first calling 
64               ``instantiate``, but the user must call ``instantiate`` again 
65               manually to make sure the instantiated graph has these 
66               changes. Pytorch has no means of tracking these changes. 
67    
68       .. warning:: 
69           This API is in beta and may change in future releases. 
70    
71       """
72   
73       def __new__(cls, keep_graph=False):
74           return super().__new__(cls, keep_graph)
75   
76       def capture_begin(self, pool=None, capture_error_mode="global"):
77           r"""Begin capturing CUDA work on the current stream. 
78    
79           Typically, you shouldn't call ``capture_begin`` yourself. 
80           Use :class:`~torch.cuda.graph` or :func:`~torch.cuda.make_graphed_callables`, 
81           which call ``capture_begin`` internally. 
82    
83           Arguments: 
84               pool (optional): Token (returned by :func:`~torch.cuda.graph_pool_handle` or 
85                   :meth:`other_Graph_instance.pool()<torch.cuda.CUDAGraph.pool>`) that hints this graph may share memory 
86                   with the indicated pool.  See :ref:`Graph memory management<graph-memory-management>`. 
87               capture_error_mode (str, optional): specifies the cudaStreamCaptureMode for the graph capture stream. 
88                   Can be "global", "thread_local" or "relaxed". During cuda graph capture, some actions, such as cudaMalloc, 
89                   may be unsafe. "global" will error on actions in other threads, "thread_local" will only error for 
90                   actions in the current thread, and "relaxed" will not error on these actions. Do NOT change this setting 
91                   unless you're familiar with `cudaStreamCaptureMode <https://docs.nvidia.com/cuda/cuda-runtime-api/group__CUDART__STREAM.html#group__CUDART__STREAM_1g9d0535d93a214cbf126835257b16ba85>`_ 
92           """  # noqa: B950
93           super().capture_begin(pool=pool, capture_error_mode=capture_error_mode)
94   
95       def capture_end(self):
96           r"""End CUDA graph capture on the current stream. 
97    
98           After ``capture_end``, ``replay`` may be called on this instance. 
99    
100          Typically, you shouldn't call ``capture_end`` yourself. 
101          Use :class:`~torch.cuda.graph` or :func:`~torch.cuda.make_graphed_callables`, 
102          which call ``capture_end`` internally. 
103          """
104          super().capture_end()
105  
106      def instantiate(self):
107          r"""Instantiate the CUDA graph. Will be called by 
108          ``capture_end`` if ``keep_graph=False``, or by ``replay`` if 
109          ``keep_graph=True`` and ``instantiate`` has not already been 
110          explicitly called. Does not destroy the cudaGraph_t returned 
111          by ``raw_cuda_graph``. 
112          """
113          super().instantiate()
114  
115      def replay(self):
116          r"""Replay the CUDA work captured by this graph."""
117          super().replay()
118  
119      def reset(self):
120          r"""Delete the graph currently held by this instance."""
121          super().reset()

126          This id can optionally be passed to another graph's ``capture_begin``, 
127          which hints the other graph may share the same memory pool.

153  class graph:
154      r"""Context-manager that captures CUDA work into a :class:`torch.cuda.CUDAGraph` object for later replay. 
155   
156      See :ref:`CUDA Graphs <cuda-graph-semantics>` for a general introduction, 
157      detailed use, and constraints. 
158   
159      Arguments: 
160          cuda_graph (torch.cuda.CUDAGraph): Graph object used for capture. 
161          pool (optional): Opaque token (returned by a call to :func:`~torch.cuda.graph_pool_handle()` or 
162              :meth:`other_Graph_instance.pool()<torch.cuda.CUDAGraph.pool>`) hinting this graph's capture 
163              may share memory from the specified pool. See :ref:`Graph memory management<graph-memory-management>`. 
164          stream (torch.cuda.Stream, optional): If supplied, will be set as the current stream in the context. 
165              If not supplied, ``graph`` sets its own internal side stream as the current stream in the context. 
166          capture_error_mode (str, optional): specifies the cudaStreamCaptureMode for the graph capture stream. 
167              Can be "global", "thread_local" or "relaxed". During cuda graph capture, some actions, such as cudaMalloc, 
168              may be unsafe. "global" will error on actions in other threads, "thread_local" will only error for 
169              actions in the current thread, and "relaxed" will not error on actions. Do NOT change this setting 
170              unless you're familiar with `cudaStreamCaptureMode <https://docs.nvidia.com/cuda/cuda-runtime-api/group__CUDART__STREAM.html#group__CUDART__STREAM_1g9d0535d93a214cbf126835257b16ba85>`_ 
171   
172      .. note:: 
173          For effective memory sharing, if you pass a ``pool`` used by a previous capture and the previous capture 
174          used an explicit ``stream`` argument, you should pass the same ``stream`` argument to this capture. 
175   
176      .. warning:: 
177          This API is in beta and may change in future releases. 
178   
179      .. _cudaStreamCaptureMode: 
180          https://docs.nvidia.com/cuda/cuda-runtime-api/group__CUDART__STREAM.html#group__CUDART__STREAM_1g9d0535d93a214cbf126835257b16ba85 
181      """  # noqa: B950
182  
183      default_capture_stream: typing.Optional["torch.cuda.Stream"] = None
184  
185      def __init__(
186          self,
187          cuda_graph,
188          pool=None,
189          stream=None,
190          capture_error_mode: str = "global",
191      ):
192          # Lazy-init of default_capture_stream helps avoid circular-import errors.
193          # Not thread safe, but graphs already have the general (explicitly documented)
194          # restriction that only one capture may be underway at a time in the process.
195          if self.__class__.default_capture_stream is None:
196              self.__class__.default_capture_stream = torch.cuda.Stream()
197  
198          self.pool = () if pool is None else (pool,)
199          self.capture_stream = (
200              stream if stream is not None else self.__class__.default_capture_stream
201          )
202          assert self.capture_stream is not None
203          self.stream_ctx = torch.cuda.stream(self.capture_stream)
204          self.cuda_graph = cuda_graph
205          self.capture_error_mode = capture_error_mode
206  
207      def __enter__(self):
208          # Free as much memory as we can for the graph
209          torch.cuda.synchronize()
210          gc.collect()
211          torch.cuda.empty_cache()
212  
213          # Stackoverflow seems comfortable with this pattern
214          # https://stackoverflow.com/questions/26635684/calling-enter-and-exit-manually#39172487
215          self.stream_ctx.__enter__()
216  
217          self.cuda_graph.capture_begin(
218              *self.pool, capture_error_mode=self.capture_error_mode
219          )
220  
221      def __exit__(self, exc_type, exc_value, traceback):
222          self.cuda_graph.capture_end()
223          self.stream_ctx.__exit__(exc_type, exc_value, traceback)
224          # returning None should propagate exceptions from either capture_end or stream_ctx.__exit__()

227  def make_graphed_callables(
228      callables, sample_args, num_warmup_iters=3, allow_unused_input=False, pool=None
229  ):
230      r"""Accept callables (functions or :class:`nn.Module<torch.nn.Module>`\ s) and returns graphed versions. 
231   
232      Each graphed callable's forward pass runs its source callable's 
233      forward CUDA work as a CUDA graph inside a single autograd node. 
234   
235      The graphed callable's forward pass also appends 
236      a backward node to the autograd graph. During backward, this node runs the 
237      callable's backward work as a CUDA graph. 
238   
239      Therefore, each graphed callable should be a drop-in replacement for its source callable 
240      in an autograd-enabled training loop. 
241   
242      See :ref:`Partial-network capture<partial-network-capture>` for detailed use and constraints. 
243   
244      If you pass a tuple of several callables, their captures will use the same memory pool. 
245      See :ref:`Graph memory management<graph-memory-management>` for when this is appropriate. 
246   
247      Arguments: 
248          callables (torch.nn.Module or Python function, or tuple of these): Callable or callables to graph. 
249              See :ref:`Graph memory management<graph-memory-management>` for when passing a tuple of callables 
250              is appropriate.  If you pass a tuple of callables, their order in the tuple must be the same order 
251              they'll run in the live workload. 
252          sample_args (tuple of Tensors, or tuple of tuples of Tensors): Samples args for each callable. 
253              If a single callable was passed, ``sample_args`` must be a single tuple of argument Tensors. 
254              If a tuple of callables was passed, ``sample_args`` must be tuple of tuples of argument Tensors. 
255          num_warmup_iters (int): The number of warmup iterations. Currently, ``DataDistributedParallel`` needs 
256              11 iterations for warm up. Default: ``3``. 
257          allow_unused_input (bool): If False, specifying inputs that were not used when computing outputs 
258              (and therefore their grad is always zero) is an error. Defaults to False. 
259          pool (optional): Token (returned by :func:`~torch.cuda.graph_pool_handle` or 
260              :meth:`other_Graph_instance.pool()<torch.cuda.CUDAGraph.pool>`) that hints this graph may share memory 
261              with the indicated pool.  See :ref:`Graph memory management<graph-memory-management>`. 
262      .. note:: 
263          The ``requires_grad`` state of each Tensor in ``sample_args`` must match the state 
264          that's expected for the corresponding real input in the training loop. 
265   
266      .. warning:: 
267          This API is in beta and may change in future releases. 
268   
269      .. warning:: 
270          ``sample_args`` for each callable must contain only Tensors. Other types are not allowed. 
271   
272      .. warning:: 
273          Returned callables do not support higher order differentiation (e.g., double backward). 
274   
275      .. warning:: 
276          In any :class:`~torch.nn.Module` passed to :func:`~make_graphed_callables`, only parameters 
277          may be trainable. Buffers must have ``requires_grad=False``. 
278   
279      .. warning:: 
280          After you pass a :class:`torch.nn.Module` through :func:`~make_graphed_callables`, 
281          you may not add or remove any of that Module's parameters or buffers. 
282   
283      .. warning:: 
284          :class:`torch.nn.Module`\s passed to :func:`~torch.cuda.make_graphed_callables` must not have module hooks 
285          registered on them at the time they are passed. However, registering hooks on modules *after* passing them 
286          through :func:`~torch.cuda.make_graphed_callables` is allowed. 
287   
288      .. warning:: 
289          When running a graphed callable, you must pass its arguments in the same order and format 
290          they appeared in that callable's ``sample_args``. 
291   
292      .. warning:: 
293          The automatic mixed precision is supported in :func:`~torch.cuda.make_graphed_callables` only with disabled 
294          caching. The context manager `torch.cuda.amp.autocast()` must have `cache_enabled=False`. 
295      """
296      if torch.is_autocast_enabled() and torch.is_autocast_cache_enabled():
297          raise RuntimeError(
298              "make_graphed_callables does not support the autocast caching. Please set `cache_enabled=False`."
299          )
300  
301      just_one_callable = False
302  
303      if not isinstance(callables, tuple):
304          just_one_callable = True
305          callables = (callables,)
306          sample_args = (sample_args,)
307  
308      flatten_sample_args = []
309  
310      for c, args in zip(callables, sample_args):
311          if isinstance(c, torch.nn.Module):
312              assert (
313                  len(c._backward_hooks) == 0
314                  and len(c._forward_hooks) == 0
315                  and len(c._forward_pre_hooks) == 0
316              ), (
317                  "Modules must not have hooks registered at the time they are passed. However, registering hooks "
318                  + "on modules after passing them through make_graphed_callables is allowed."
319              )
320              assert all(b.requires_grad is False for b in c.buffers()), (
321                  "In any :class:`~torch.nn.Module` passed to "
322                  + ":func:`~make_graphed_callables`, only parameters may be trainable. All buffers must have "
323                  + "``requires_grad=False``."
324              )
325          flatten_arg = torch.utils._pytree.arg_tree_leaves(*args)
326          flatten_sample_args.append(tuple(flatten_arg))
327          assert all(isinstance(arg, torch.Tensor) for arg in flatten_arg), (
328              "In the beta API, sample_args "
329              + "for each callable must contain only Tensors. Other types are not allowed."
330          )
331  
332      # If a callable is an nn.Module, its graph's full input surface is the args the user explicitly
333      # passes to forward (ie, its sample_args) AND the module's parameter attributes.
334      per_callable_len_user_args = [len(args) for args in flatten_sample_args]
335      per_callable_module_params = [
336          tuple(c.parameters()) if isinstance(c, torch.nn.Module) else ()
337          for c in callables
338      ]
339      per_callable_static_input_surfaces = [
340          flatten_sample_args[i] + per_callable_module_params[i]
341          for i in range(len(callables))
342      ]
343  
344      fwd_graphs = [torch.cuda.CUDAGraph() for _ in range(len(callables))]
345      bwd_graphs = [torch.cuda.CUDAGraph() for _ in range(len(callables))]
346  
347      mempool = graph_pool_handle() if pool is None else pool
348  
349      # Warmup
350      # Hopefully prevents cudnn benchmarking and other lazy-initialization cuda work
351      # from ending up in any captures.
352      torch.cuda.synchronize()
353      with torch.cuda.stream(torch.cuda.Stream()):
354          for func, args, static_input_surface in zip(
355              callables, sample_args, per_callable_static_input_surfaces
356          ):
357              grad_inputs, outputs, outputs_grad = None, None, None
358              for _ in range(num_warmup_iters):
359                  outputs = torch.utils._pytree.tree_leaves(func(*args))
360                  outputs_grad = tuple(o for o in outputs if o.requires_grad)
361                  if len(outputs_grad) > 0:
362                      grad_inputs = torch.autograd.grad(
363                          outputs=outputs_grad,
364                          inputs=tuple(
365                              i for i in static_input_surface if i.requires_grad
366                          ),
367                          grad_outputs=tuple(
368                              torch.empty_like(o) for o in outputs if o.requires_grad
369                          ),
370                          only_inputs=True,
371                          allow_unused=allow_unused_input,
372                      )
373              for v in [outputs, outputs_grad, grad_inputs]:
374                  del v
375  
376      torch.cuda.synchronize()
377  
378      # All captures here share a mempool. To avoid replays corrupting each other's memory,
379      # the safest approach is to capture all passes in the same order they'll run:
380      # fwd 1, fwd 2, ... fwd N, then bwd N, bwd N-1, ... bwd 1.
381  
382      # Capture forward graphs
383      per_callable_static_outputs = []
384      per_callable_output_unflatten_spec = []
385      for func, args, fwd_graph in zip(callables, sample_args, fwd_graphs):
386          with torch.cuda.graph(fwd_graph, pool=mempool):
387              outputs = func(*args)
388  
389          flatten_outputs, spec = torch.utils._pytree.tree_flatten(outputs)
390          per_callable_static_outputs.append(tuple(flatten_outputs))
391          per_callable_output_unflatten_spec.append(spec)
392  
393      # Capture backward graphs in reverse order
394      per_callable_static_grad_outputs = []
395      per_callable_static_grad_inputs = []
396      for static_input_surface, static_outputs, bwd_graph in zip(
397          reversed(per_callable_static_input_surfaces),
398          reversed(per_callable_static_outputs),
399          reversed(bwd_graphs),
400      ):
401          # For now, assumes all static_outputs require grad
402          # assert all(o.requires_grad for o in static_outputs), "Outputs of graphed callables must require grad."
403          static_grad_outputs = tuple(
404              torch.empty_like(o) if o.requires_grad else None for o in static_outputs
405          )
406  
407          outputs_grad = tuple(o for o in static_outputs if o.requires_grad)
408          grad_inputs = None
409          if len(outputs_grad) > 0:
410              with torch.cuda.graph(bwd_graph, pool=mempool):
411                  grad_inputs = torch.autograd.grad(
412                      outputs=outputs_grad,
413                      inputs=tuple(i for i in static_input_surface if i.requires_grad),
414                      grad_outputs=tuple(o for o in static_grad_outputs if o is not None),
415                      only_inputs=True,
416                      allow_unused=allow_unused_input,
417                  )
418  
419          # Constructs a tuple suitable for returning from Graphed.backward:
420          # Pads out the actually-needed grads with Nones in gradient slots for inputs that don't require grad.
421          # I couldn't think of a slick one-liner for this pattern.
422          static_grad_inputs = []
423          grad_idx = 0
424          for arg in static_input_surface:
425              if arg.requires_grad and grad_inputs is not None:
426                  static_grad_inputs.append(grad_inputs[grad_idx])
427                  grad_idx += 1
428              else:
429                  static_grad_inputs.append(None)  # type: ignore[arg-type]
430          static_grad_inputs = tuple(static_grad_inputs)  # type: ignore[assignment]
431  
432          per_callable_static_grad_outputs.append(static_grad_outputs)
433          per_callable_static_grad_inputs.append(static_grad_inputs)
434  
435      # Reverses the most recent two lists
436      per_callable_static_grad_outputs.reverse()
437      per_callable_static_grad_inputs.reverse()
438      # Now for every per_callable list, per_callable_*[i] holds the stuff for the ith callable.
439  
440      def make_graphed_autograd_function(
441          fwd_graph,
442          bwd_graph,
443          module_params,
444          len_user_args,
445          output_unflatten_spec,
446          static_input_surface,
447          static_outputs,
448          static_grad_outputs,
449          static_grad_inputs,
450      ):
451          class Graphed(torch.autograd.Function):
452              @staticmethod
453              def forward(ctx, *inputs):
454                  # At this stage, only the user args may (potentially) be new tensors.
455                  for i in range(len_user_args):
456                      if static_input_surface[i].data_ptr() != inputs[i].data_ptr():
457                          static_input_surface[i].copy_(inputs[i])
458                  fwd_graph.replay()
459                  assert isinstance(static_outputs, tuple)
460                  return tuple(o.detach() for o in static_outputs)
461  
462              @staticmethod
463              @torch.autograd.function.once_differentiable
464              def backward(ctx, *grads):
465                  assert len(grads) == len(static_grad_outputs)
466                  for g, grad in zip(static_grad_outputs, grads):
467                      if g is not None:
468                          # don't copy if autograd gods have been kind and the
469                          # incoming grad is already in the right place
470                          if g.data_ptr() != grad.data_ptr():
471                              g.copy_(grad)
472                  bwd_graph.replay()
473  
474                  # Input args that didn't require grad expect a None gradient.
475                  assert isinstance(static_grad_inputs, tuple)
476                  return tuple(
477                      b.detach() if b is not None else b for b in static_grad_inputs
478                  )
479  
480          def functionalized(*user_args):
481              # Runs the autograd function with inputs == all inputs to the graph that might require grad
482              # (explicit user args + module parameters)
483              # Assumes module params didn't change since capture.
484              flatten_user_args = torch.utils._pytree.arg_tree_leaves(*user_args)
485              out = Graphed.apply(*(tuple(flatten_user_args) + module_params))
486              return torch.utils._pytree.tree_unflatten(out, output_unflatten_spec)
487  
488          return functionalized
489  
490      # Put together the final graphed callables
491      ret = []
492      for i, func in enumerate(callables):
493          graphed = make_graphed_autograd_function(
494              fwd_graphs[i],
495              bwd_graphs[i],
496              per_callable_module_params[i],
497              per_callable_len_user_args[i],
498              per_callable_output_unflatten_spec[i],
499              per_callable_static_input_surfaces[i],
500              per_callable_static_outputs[i],
501              per_callable_static_grad_outputs[i],
502              per_callable_static_grad_inputs[i],
503          )
504  
505          if isinstance(func, torch.nn.Module):
506  
507              def make_graphed_forward(func, graph_training_state, graphed, orig_fwd):
508                  def new_fwd(*user_args):
509                      # If the module's training-or-eval state matches what we graphed,
510                      # run the graph, otherwise run the original forward method
511                      if func.training == graph_training_state:
512                          return graphed(*user_args)
513                      else:
514                          return orig_fwd(*user_args)
515  
516                  return new_fwd
517  
518              func.forward = make_graphed_forward(func, func.training, graphed, func.forward)  # type: ignore[assignment]
519              ret.append(func)
520          else:
521              ret.append(graphed)
522  
523      if just_one_callable:
524          return ret[0]
525  
526      return tuple(ret)
527
