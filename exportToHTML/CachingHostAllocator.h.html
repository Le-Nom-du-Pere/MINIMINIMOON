10   //
11   // A caching allocator for CUDA host allocations (pinned memory).
12   //
13   // This provides a drop-in replacement for THCudaHostAllocator, which re-uses
14   // freed pinned (page-locked) memory allocations. This avoids device
15   // synchronizations due to cudaFreeHost calls.
16   //
17   // To ensure correct behavior, THCCachingHostAllocator_recordEvent must be
18   // called anytime a pointer from this allocator is used in a cudaMemcpyAsync
19   // call between host and device, and passed the corresponding context from the
20   // allocation. This is currently invoked by at::native::copy_kernel_cuda.
21   //
22   C10_DEPRECATED_MESSAGE(
23     "at::cuda::getCachingHostAllocator() is deprecated. Please use at::getHostAllocator(at::kCUDA) instead.")
24   inline TORCH_CUDA_CPP_API at::HostAllocator* getCachingHostAllocator() {
25     return at::getHostAllocator(at::kCUDA);
26   }
27   
28   // Records an event in the specified stream. The allocation corresponding to the
29   // input `ptr`/`ctx` will not be re-used until the event has occurred.
30   C10_DEPRECATED_MESSAGE(
31     "at::cuda::CachingHostAllocator_recordEvent(...) is deprecated. Please use at::getHostAllocator(at::kCUDA)->record_event(...) instead.")
32   inline TORCH_CUDA_CPP_API bool CachingHostAllocator_recordEvent(
33       void* ptr,
34       void* ctx,
35       c10::cuda::CUDAStream stream) {
36     return getHostAllocator(at::kCUDA)->record_event(ptr, ctx, stream.unwrap());
37   }
38   
39   // Releases cached pinned memory allocations via cudaHostFree
40   C10_DEPRECATED_MESSAGE(
41     "at::cuda::CachingHostAllocator_emptyCache() is deprecated. Please use at::getHostAllocator(at::kCUDA)->empty_cache() instead.")
42   inline TORCH_CUDA_CPP_API void CachingHostAllocator_emptyCache() {
43     getHostAllocator(at::kCUDA)->empty_cache();
44   }
45   
46   C10_DEPRECATED_MESSAGE(

48   inline TORCH_CUDA_CPP_API at::DataPtr HostAlloc(size_t size) {
49     return getHostAllocator(at::kCUDA)->allocate(size);
50   }
