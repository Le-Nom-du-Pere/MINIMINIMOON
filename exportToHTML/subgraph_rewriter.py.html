44   def _replace_attributes(gm: GraphModule, replacement: torch.nn.Module) -> None:
45       gm.delete_all_unused_submodules()
46   
47       if isinstance(replacement, GraphModule):
48           replacement.graph.lint()
49   
50       def try_get_attr(gm: torch.nn.Module, target: str) -> Optional[Any]:
51           module_path, _, attr_name = target.rpartition(".")
52           try:
53               mod: torch.nn.Module = gm.get_submodule(module_path)
54           except AttributeError:
55               return None
56           attr = getattr(mod, attr_name, None)
57           return attr
58   
59       for node in gm.graph.nodes:
60           if node.op == "call_module" or node.op == "get_attr":
61               gm_attr = try_get_attr(gm, node.target)
62               replacement_attr = try_get_attr(replacement, node.target)
63   
64               # CASE 1: This target already exists as an attribute in our
65               # result GraphModule. Whether or not it exists in
66               # `replacement`, the existing submodule takes precedence.
67               if gm_attr is not None:
68                   continue
69   
70               # CASE 2: The target exists as an attribute in `replacement`
71               # only, so we need to copy it over.
72               elif replacement_attr is not None:
73                   new_attr = copy.deepcopy(replacement_attr)
74                   if isinstance(replacement_attr, torch.nn.Module):
75                       gm.add_submodule(node.target, new_attr)
76                   else:
77                       setattr(gm, node.target, new_attr)
78   
79               # CASE 3: The target doesn't exist as an attribute in `gm`
80               # or `replacement`
81               else:
82                   raise RuntimeError(
83                       'Attempted to create a "',
84                       node.op,
85                       '" node during subgraph rewriting '
86                       f"with target {node.target}, but "
87                       "the referenced attribute does not "
88                       "exist in the replacement GraphModule",
89                   )
90   
91       gm.graph.lint()

94   @compatibility(is_backward_compatible=True)
95   def replace_pattern(
96       gm: GraphModule,
97       pattern: Union[Callable, GraphModule],
98       replacement: Union[Callable, GraphModule],
99   ) -> list[Match]:
100      """ 
101      Matches all possible non-overlapping sets of operators and their 
102      data dependencies (``pattern``) in the Graph of a GraphModule 
103      (``gm``), then replaces each of these matched subgraphs with another 
104      subgraph (``replacement``). 
105   
106      Args: 
107          ``gm``: The GraphModule that wraps the Graph to operate on 
108          ``pattern``: The subgraph to match in ``gm`` for replacement 
109          ``replacement``: The subgraph to replace ``pattern`` with 
110   
111      Returns: 
112          List[Match]: A list of ``Match`` objects representing the places 
113          in the original graph that ``pattern`` was matched to. The list 
114          is empty if there are no matches. ``Match`` is defined as: 
115   
116          .. code-block:: python 
117   
118              class Match(NamedTuple): 
119                  # Node from which the match was found 
120                  anchor: Node 
121                  # Maps nodes in the pattern subgraph to nodes in the larger graph 
122                  nodes_map: Dict[Node, Node] 
123   
124      Examples: 
125   
126      .. code-block:: python 
127   
128          import torch 
129          from torch.fx import symbolic_trace, subgraph_rewriter 
130   
131   
132          class M(torch.nn.Module): 
133              def __init__(self) -> None: 
134                  super().__init__() 
135   
136              def forward(self, x, w1, w2): 
137                  m1 = torch.cat([w1, w2]).sum() 
138                  m2 = torch.cat([w1, w2]).sum() 
139                  return x + torch.max(m1) + torch.max(m2) 
140   
141   
142          def pattern(w1, w2): 
143              return torch.cat([w1, w2]) 
144   
145   
146          def replacement(w1, w2): 
147              return torch.stack([w1, w2]) 
148   
149   
150          traced_module = symbolic_trace(M()) 
151   
152          subgraph_rewriter.replace_pattern(traced_module, pattern, replacement) 
153   
154      The above code will first match ``pattern`` in the ``forward`` 
155      method of ``traced_module``. Pattern-matching is done based on 
156      use-def relationships, not node names. For example, if you had 
157      ``p = torch.cat([a, b])`` in ``pattern``, you could match 
158      ``m = torch.cat([a, b])`` in the original ``forward`` function, 
159      despite the variable names being different (``p`` vs ``m``). 
160   
161      The ``return`` statement in ``pattern`` is matched based on its 
162      value only; it may or may not match to the ``return`` statement in 
163      the larger graph. In other words, the pattern doesn't have to extend 
164      to the end of the larger graph. 
165   
166      When the pattern is matched, it will be removed from the larger 
167      function and replaced by ``replacement``. If there are multiple 
168      matches for ``pattern`` in the larger function, each non-overlapping 
169      match will be replaced. In the case of a match overlap, the first 
170      found match in the set of overlapping matches will be replaced. 
171      ("First" here being defined as the first in a topological ordering 
172      of the Nodes' use-def relationships. In most cases, the first Node 
173      is the parameter that appears directly after ``self``, while the 
174      last Node is whatever the function returns.) 
175   
176      One important thing to note is that the parameters of the 
177      ``pattern`` Callable must be used in the Callable itself, 
178      and the parameters of the ``replacement`` Callable must match 
179      the pattern. The first rule is why, in the above code block, the 
180      ``forward`` function has parameters ``x, w1, w2``, but the 
181      ``pattern`` function only has parameters ``w1, w2``. ``pattern`` 
182      doesn't use ``x``, so it shouldn't specify ``x`` as a parameter. 
183      As an example of the second rule, consider replacing 
184   
185      .. code-block:: python 
186   
187          def pattern(x, y): 
188              return torch.neg(x) + torch.relu(y) 
189   
190      with 
191   
192      .. code-block:: python 
193   
194          def replacement(x, y): 
195              return torch.relu(x) 
196   
197      In this case, ``replacement`` needs the same number of parameters 
198      as ``pattern`` (both ``x`` and ``y``), even though the parameter 
199      ``y`` isn't used in ``replacement``. 
200   
201      After calling ``subgraph_rewriter.replace_pattern``, the generated 
202      Python code looks like this: 
203   
204      .. code-block:: python 
205   
206          def forward(self, x, w1, w2): 
207              stack_1 = torch.stack([w1, w2]) 
208              sum_1 = stack_1.sum() 
209              stack_2 = torch.stack([w1, w2]) 
210              sum_2 = stack_2.sum() 
211              max_1 = torch.max(sum_1) 
212              add_1 = x + max_1 
213              max_2 = torch.max(sum_2) 
214              add_2 = add_1 + max_2 
215              return add_2 
216      """
217      match_and_replacements = _replace_pattern(gm, pattern, replacement)
218      return [
219          Match(anchor=m.anchor, nodes_map=m.nodes_map) for m in match_and_replacements
220      ]

223  # Experimental API, not backward compatible
224  @compatibility(is_backward_compatible=False)
225  def replace_pattern_with_filters(
226      gm: GraphModule,
227      pattern: Union[Callable, Graph, GraphModule],
228      replacement: Union[Callable, Graph, GraphModule, None] = None,
229      match_filters: Optional[
230          list[Callable[["InternalMatch", Graph, Graph], bool]]
231      ] = None,
232      ignore_literals: bool = False,
233      # Placed at the end to avoid breaking backward compatibility
234      replacement_callback: Optional[
235          Callable[["InternalMatch", Graph, Graph], Graph]
236      ] = None,
237  ) -> list[ReplacedPatterns]:
238      """ 
239      See replace_pattern for documentation. This function is an overload with an additional match_filter argument. 
240   
241      Args: 
242          ``match_filters``: A list of functions that take in 
243              (match: InternalMatch, original_graph: Graph, pattern_graph: Graph) and return a boolean indicating 
244              whether the match satisfies the condition. 
245              See matcher_utils.py for definition of InternalMatch. 
246          ``replacement_callback``: A function that takes in a match and returns a 
247              Graph to be used as the replacement. This allows you to construct a 
248              replacement graph based on the match. 
249      """
250  
251      return _replace_pattern(
252          gm, pattern, replacement, match_filters, ignore_literals, replacement_callback
253      )

256  def _replace_pattern(
257      gm: GraphModule,
258      pattern: Union[Callable, Graph, GraphModule],
259      replacement: Union[Callable, Graph, GraphModule, None] = None,
260      match_filters: Optional[
261          list[Callable[["InternalMatch", Graph, Graph], bool]]
262      ] = None,
263      ignore_literals: bool = False,
264      # Placed at the end to avoid breaking backward compatibility
265      replacement_callback: Optional[
266          Callable[["InternalMatch", Graph, Graph], Graph]
267      ] = None,
268  ) -> list[ReplacedPatterns]:
269      from torch.fx.passes.utils.matcher_utils import InternalMatch, SubgraphMatcher
270  
271      if match_filters is None:
272          match_filters = []
273  
274      # Get the graphs for `gm`, `pattern`, `replacement`
275      original_graph: Graph = gm.graph
276  
277      if isinstance(pattern, GraphModule):
278          pattern_graph = pattern.graph
279      elif isinstance(pattern, Graph):
280          pattern_graph = pattern
281      else:
282          pattern_graph = symbolic_trace(pattern).graph
283  
284      matcher = SubgraphMatcher(
285          pattern_graph,
286          match_output=False,
287          match_placeholder=False,
288          remove_overlapping_matches=True,
289          ignore_literals=ignore_literals,
290      )
291      _matches: list[InternalMatch] = matcher.match(original_graph)
292  
293      # Filter out matches that don't match the filter
294      _matches = [
295          m
296          for m in _matches
297          if all(
298              match_filter(m, original_graph, pattern_graph)
299              for match_filter in match_filters
300          )
301      ]
302  
303      if isinstance(replacement, GraphModule):
304          common_replacement_graph = replacement.graph
305      elif isinstance(replacement, Graph):
306          common_replacement_graph = replacement
307      elif callable(replacement):
308          common_replacement_graph = symbolic_trace(replacement).graph
309      else:
310          assert replacement_callback is not None, (
311              "Must provide either a replacement GraphModule or a replacement callback"
312          )
313          common_replacement_graph = None
314  
315      # As we progressively replace nodes, we'll need to keep track of how the match results should change
316      match_changed_node: dict[Node, Node] = {}
317  
318      match_and_replacements = []
319      for match in _matches:
320          if replacement_callback is not None:
321              replacement_graph = replacement_callback(
322                  match, original_graph, pattern_graph
323              )
324          else:
325              assert common_replacement_graph is not None, (
326                  "Must provide either a replacement GraphModule or a replacement callback"
327              )
328              replacement_graph = common_replacement_graph
329          replacement_placeholders = [
330              n for n in replacement_graph.nodes if n.op == "placeholder"
331          ]
332  
333          # Build connecting between replacement graph's input and original graph input producer node
334  
335          # Initialize `val_map` with mappings from placeholder nodes in
336          # `replacement` to their corresponding node in `original_graph`
337          assert len(match.placeholder_nodes) == len(replacement_placeholders)
338          val_map: dict[Node, Node] = {}
339          for rn, gn in zip(replacement_placeholders, match.placeholder_nodes):
340              if isinstance(gn, Node):
341                  val_map[rn] = match_changed_node.get(gn, gn)
342                  if gn != val_map[rn]:
343                      # Update match.placeholder_nodes and match.nodes_map with the node that replaced gn
344                      gn_ind = match.placeholder_nodes.index(gn)
345                      match.placeholder_nodes[gn_ind] = match_changed_node[gn]
346                      map_key = list(match.nodes_map.keys())[
347                          list(match.nodes_map.values()).index(gn)
348                      ]
349                      match.nodes_map[map_key] = match_changed_node[gn]
350              else:
351                  val_map[rn] = gn
352  
353          # Copy the replacement graph over
354          user_nodes: set[Node] = set()
355          for n in match.returning_nodes:
356              user_nodes.update(n.users)
357  
358          first_user_node = None
359          if len(user_nodes) == 0:
360              first_user_node = None
361          elif len(user_nodes) == 1:
362              first_user_node = next(iter(user_nodes))
363          else:
364              # If there are multiple user nodes, we need to find the first user node
365              # in the current execution order of the `original_graph`
366              for n in original_graph.nodes:
367                  if n in user_nodes:
368                      first_user_node = n
369                      break
370  
371          first_next_node = None
372          if first_user_node is None:
373              # no users, so we insert the replacement graph before the first next
374              # node of returning nodes
375              next_node = None
376              for n in reversed(original_graph.nodes):
377                  if n in match.returning_nodes:
378                      first_next_node = next_node
379                      break
380                  else:
381                      next_node = n
382          insert_point = (
383              first_user_node if first_user_node is not None else first_next_node
384          )
385          assert insert_point is not None, "The insert point can't be None"
386          with original_graph.inserting_before(insert_point):
387              copied_returning_nodes = original_graph.graph_copy(
388                  replacement_graph, val_map
389              )
390  
391          if isinstance(copied_returning_nodes, Node):
392              copied_returning_nodes = (copied_returning_nodes,)
393  
394          # Get a list of nodes that have been replaced into the graph
395          replacement_nodes: list[Node] = [
396              v for v in val_map.values() if v not in match.placeholder_nodes
397          ]
398  
399          # Hook the output Node of the replacement subgraph in to the
400          # original Graph at the correct location
401          assert len(match.returning_nodes) == len(copied_returning_nodes)  # type: ignore[arg-type]
402          for gn, copied_node in zip(match.returning_nodes, copied_returning_nodes):  # type: ignore[arg-type]
403              gn.replace_all_uses_with(copied_node)
404              match_changed_node[gn] = copied_node
405          # Remove the original nodes
406          for node in reversed(pattern_graph.nodes):
407              if node.op != "placeholder" and node.op != "output":
408                  gn = match.nodes_map[node]
409                  gm.graph.erase_node(gn)
410  
411          match_and_replacements.append(
412              ReplacedPatterns(
413                  anchor=match.anchors[0],
414                  nodes_map=match.nodes_map,
415                  replacements=replacement_nodes,
416              )
417          )
418  
419      # Update the passed-in GraphModule to reflect the new state of
420      # `original_graph`
421      gm.recompile()
422  
423      # If `replacement` was an nn.Module, we'll need to make sure that
424      # all the submodules have been copied over correctly
425      if isinstance(replacement, torch.nn.Module):
426          _replace_attributes(gm, replacement)
427  
428      return match_and_replacements
