2    """ 
3    This file contains canonical definitions for our symbol naming conventions, 
4    across torch.fx.experimental.symbolic_shapes and torch._inductor.  The 
5    intention is: 
6     
7    1. To make it easily greppable where all the sites we use a prefix are 
8    2. Make it possible to easily tell if we can introduce a new prefix without 
9       introducing a conflict 
10    
11   You can occasionally test if prefixes have been hardcoded by renaming prefixes 
12   in this file and seeing what breaks. 
13   """
14   
15   from collections.abc import Iterable
16   from enum import auto, Enum
17   from typing import Union
18   
19   import sympy
20   
21   
22   class SymT(Enum):
23       SIZE = auto()
24       FLOAT = auto()
25       UNBACKED_INT = auto()
26       UNBACKED_FLOAT = auto()
27       # Inductor: The intermediates in inner_fn tmp0, one generated per ops call.
28       # If one of these shows up in an indexing expression, that means an
29       # indirect load is happening.
30       TMP = auto()
31       # Inductor: Placeholder variable that is later replaced with TMP
32       INDIRECT = auto()
33       # Inductor: Some size expressions are replaced with a precomputed size ps0
34       # which is computed host side, and then directly reused in the kernel, so
35       # we don't repeatedly recompute it on device.
36       PRECOMPUTED_SIZE = auto()
37       # Inductor: An indexing variable i0 in loops IR which ranges over non-reduced
38       # dim in the loop
39       INDEX = auto()
40       # Inductor: A reduction indexing (r0, r1) variables in loops IR which ranges over
41       # reduced dim(s) in the loop
42       R0_INDEX = auto()
43       R1_INDEX = auto()
44       # Inductor: In templated kernels torch._inductor.kernel, we have a hook to
45       # store the final output and append epilogue fusions.  To do this, we must
46       # know what the indexes the outputs range over.  NB: These will also
47       # advertise as INDEX, this is... probably OK?
48       TEMPLATE_INDEX = auto()
49       # Inductor: iteration domain for blockIdx.x/blockIdx.y
50       XBLOCK = auto()
51       YBLOCK = auto()
52       ZBLOCK = auto()
53       # Inductor: this is used solely for dynamic_reshape_indexer
54       VIEW = auto()
55       # Alternate (non-modular) indexing used in halide kernels
56       HALIDE = auto()
57

59   # Invariant: there must not be a prefix which is a prefix of another string,
60   # as this introduces ambiguity
61   prefix_str = {
62       SymT.SIZE: "s",  # integer
63       SymT.UNBACKED_INT: "u",  # integer
64       # Prefix z here is chosen to avoid false aliasing in symbol_is_type test
65       # DO NOT add a "z" type.  You also need to avoid conflicts on these
66       # prefixes but this is somewhat easier to manage
67       SymT.FLOAT: "zf",
68       SymT.UNBACKED_FLOAT: "zuf",
69       SymT.TMP: "tmp",
70       SymT.PRECOMPUTED_SIZE: "ps",
71       SymT.INDEX: "i",
72       SymT.R0_INDEX: "r0_",
73       SymT.R1_INDEX: "r1_",
74       SymT.TEMPLATE_INDEX: "idx",
75       SymT.XBLOCK: "x",
76       SymT.YBLOCK: "y",
77       SymT.ZBLOCK: "z",
78       SymT.INDIRECT: "indirect",  # false aliasing?
79       SymT.VIEW: "view",
80       SymT.HALIDE: "h",
81   }

84   def make_symbol(prefix: SymT, idx: int, **kwargs) -> sympy.Symbol:
85       # TODO: maybe put the assumptions here directly
86       return sympy.Symbol(f"{prefix_str[prefix]}{idx}", **kwargs)

89   # This type is a little wider than it should be, because free_symbols says
90   # that it contains Basic, rather than Symbol
91   def symbol_is_type(sym: sympy.Basic, prefix: Union[SymT, Iterable[SymT]]) -> bool:
92       assert isinstance(sym, sympy.Symbol)
93       name_str = sym.name.lower()  # Match capitalized names like XBLOCK, RBLOCK
94       if isinstance(prefix, SymT):
95           return name_str.startswith(prefix_str[prefix])
96       else:
97           return name_str.startswith(tuple(prefix_str[p] for p in prefix))
98   
99   
100  def free_symbol_is_type(e: sympy.Expr, prefix: Union[SymT, Iterable[SymT]]) -> bool:
101      return any(symbol_is_type(v, prefix) for v in e.free_symbols)
