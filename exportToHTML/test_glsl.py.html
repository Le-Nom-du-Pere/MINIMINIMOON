299      n, m, o, p = symbols('n m o p', integer=True)
300      a = IndexedBase('a')
301      b = IndexedBase('b')
302      c = IndexedBase('c')
303      y = IndexedBase('y')
304      i = Idx('i', m)
305      j = Idx('j', n)
306      k = Idx('k', o)

308      s0 = (
309          'for (int i=0; i<m; i++){\n'
310          '   y[i] = 0.0;\n'

314          'for (int i=0; i<m; i++){\n'
315          '   for (int j=0; j<n; j++){\n'
316          '      for (int k=0; k<o; k++){\n'

318          '      }\n'
319          '   }\n'
320          '}\n'
321      )
322      s2 = (
323          'for (int i=0; i<m; i++){\n'
324          '   for (int k=0; k<o; k++){\n'
325          '      y[i] = a[%s]*b[k] + y[i];\n' % (i*o + k) +\
326          '   }\n'
327          '}\n'
328      )
329      s3 = (
330          'for (int i=0; i<m; i++){\n'
331          '   for (int j=0; j<n; j++){\n'
