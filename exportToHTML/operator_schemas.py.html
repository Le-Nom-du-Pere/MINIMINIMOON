87   def _torchscript_type_to_python_type(ts_type: "torch._C.JitType") -> Any:
88       """ 
89       Convert a TorchScript type to a Python type (including subtypes) via 
90       eval'ing the annotation_str. _type_eval_globals sets up expressions 
91       like "List" and "Future" to map to actual types (typing.List and jit.Future) 
92       """
93       return eval(ts_type.annotation_str, _type_eval_globals)

96   def _torchscript_schema_to_signature_impl(
97       ts_schema: torch._C.FunctionSchema,
98   ) -> inspect.Signature:
99       from inspect import Parameter
100  
101      parameters: list[Parameter] = []
102      for arg in ts_schema.arguments:
103          arg_type = _torchscript_type_to_python_type(arg.type)
104          default = arg.default_value if arg.has_default_value() else Parameter.empty
105          # TODO: Figure out if this is safe. It seems like when generating the type signatures for
106          # PythonArgParser, we emit signatures with `input` instead of `self` as the first tensor
107          # argument name. Downstream, if someone converts that positional argument to a keyword
108          # argument, the name mismatch will break things, so here we're going to normalize the
109          # name to "input"
110          name = arg.name if arg.name != "self" else "input"
111          kind = (
112              Parameter.KEYWORD_ONLY
113              if arg.kwarg_only
114              else Parameter.POSITIONAL_OR_KEYWORD
115          )
116          # "from" is a keyword therefore it must be a POSITIONAL_ONLY argument
117          if name == "from":
118              assert kind == Parameter.POSITIONAL_OR_KEYWORD
119              # ParameterKind type is internal implementation detail to inspec package
120              # which makes it hard to do type annotation
121              kind = Parameter.POSITIONAL_ONLY  # type: ignore[assignment]
122              # This renders all previous arguments to positional only
123              for idx, p in enumerate(parameters):
124                  assert p.kind == Parameter.POSITIONAL_OR_KEYWORD
125                  parameters[idx] = Parameter(
126                      name=p.name,
127                      kind=Parameter.POSITIONAL_ONLY,
128                      default=p.default,
129                      annotation=p.annotation,
130                  )
131          parameters.append(
132              Parameter(name=name, kind=kind, default=default, annotation=arg_type)
133          )
134      return_types = [
135          _torchscript_type_to_python_type(ret.type) for ret in ts_schema.returns
136      ]
137      if len(return_types) == 0:
138          return_type = None
139      elif len(return_types) == 1:
140          return_type = return_types[0]
141      else:
142          return_type = tuple(return_types)
143  
144      return inspect.Signature(parameters, return_annotation=return_type)

150  def _torchscript_schema_to_signature(
151      ts_schema: torch._C.FunctionSchema,
152  ) -> inspect.Signature:
153      # Cached as it's called in the hot path of FakeTensor dispatch
154      cache_key = ts_schema.name, ts_schema.overload_name
155      cache_val = _SCHEMA_TO_SIGNATURE_CACHE.get(cache_key)
156      if cache_val is not None:
157          return cache_val
158  
159      res = _torchscript_schema_to_signature_impl(ts_schema)
160      _SCHEMA_TO_SIGNATURE_CACHE[cache_key] = res
161      return res

164  @compatibility(is_backward_compatible=False)
165  def check_for_mutable_operation(
166      target: Callable, args: tuple["Argument", ...], kwargs: dict[str, "Argument"]
167  ):
168      signatures, schemas = get_signature_for_torch_op(target, return_schemas=True)
169  
170      if signatures and schemas:
171          matched_schemas = []
172  
173          # Iterate through all of the schema until we find one that matches
174          # If one matches, populate `new_args_and_kwargs` with the new args/kwargs
175          # values. If none matches, `new_args_and_kwargs` will be None
176          for candidate_signature, schema in zip(signatures, schemas):
177              try:
178                  candidate_signature.bind(*args, **kwargs)
179                  matched_schemas.append((candidate_signature, schema))
180              except TypeError:
181                  continue
182  
183          def throw_if_mutable(schema):
184              if schema.is_mutable:
185                  raise RuntimeError(
186                      f"Tried to trace mutable operation {schema}. FX only supports functional "
187                      f"code, so operations that mutate operands in-place (e.g. via `out` arguments) "
188                      f"are not supported"
189                  )
190  
191          if len(matched_schemas) == 0:
192              # Did not match any schema. Cannot check for mutation
193              pass
194          elif len(matched_schemas) == 1:
195              # Matched exactly one schema, unambiguous
196              _, schema_to_check = matched_schemas[0]
197              throw_if_mutable(schema_to_check)
198          else:
199              # Ambiguous schema match. Since mutability checking is best effort,
200              # do nothing.
201              pass

204  @compatibility(is_backward_compatible=False)
205  def get_signature_for_torch_op(op: Callable, return_schemas: bool = False):
206      """ 
207      Given an operator on the `torch` namespace, return a list of `inspect.Signature` 
208      objects corresponding to the overloads of that op.. May return `None` if a signature 
209      could not be retrieved. 
210   
211      Args: 
212          op (Callable): An operator on the `torch` namespace to look up a signature for 
213   
214      Returns: 
215          Optional[List[inspect.Signature]]: A list of signatures for the overloads of this 
216              operator, or None if the operator signatures could not be retrieved. If 
217              return_schemas=True, returns a tuple containing the optional Python signatures 
218              and the optional TorchScript Function signature 
219      """
220      if isinstance(op, OpOverload):
221          schemas = [op._schema]
222      elif isinstance(op, OpOverloadPacket):
223          schemas = [getattr(op, overload)._schema for overload in op.overloads()]
224      else:
225          override = _manual_overrides.get(op)
226          if override:
227              return (override, None) if return_schemas else None
228  
229          aten_fn = torch.jit._builtins._find_builtin(op)
230  
231          if aten_fn is None:
232              return (None, None) if return_schemas else None
233          schemas = torch._C._jit_get_schemas_for_operator(aten_fn)
234  
235      signatures = [_torchscript_schema_to_signature(schema) for schema in schemas]
236      return (signatures, schemas) if return_schemas else signatures

239  @compatibility(is_backward_compatible=False)
240  def create_type_hint(x):
241      """ 
242      Produces a type hint for the given argument. 
243   
244      The :func:`create_type_hint` looks for a type hint compatible with the input argument `x`. 
245   
246      If `x` is a `list` or `tuple`, it looks for an object in the list whose type is a superclass 
247      of the rest, and uses that as `base_type` for the `List` or `Tuple` to be returned. 
248      If no such object is found, it defaults to `List[Any]`. 
249   
250      If `x` is neither a `list` nor a `tuple`, it returns `x`. 
251      """
252      try:
253          if isinstance(x, (list, tuple)):
254              # todo(chilli): Figure out the right way for mypy to handle this
255              if isinstance(x, list):
256  
257                  def ret_type(x):
258                      return list[x]  # type: ignore[valid-type]
259  
260              else:
261  
262                  def ret_type(x):
263                      return tuple[x, ...]  # type: ignore[valid-type]
264  
265              if len(x) == 0:
266                  return ret_type(Any)
267              base_type = x[0]
268              for t in x:
269                  if issubclass(t, base_type):
270                      continue
271                  elif issubclass(base_type, t):
272                      base_type = t
273                  else:
274                      return ret_type(Any)
275              return ret_type(base_type)
276      except Exception:
277          # We tried to create a type hint for list but failed.
278          warnings.warn(
279              f"We were not able to successfully create type hint from the type {x}"
280          )
281      return x

284  @compatibility(is_backward_compatible=False)
285  def type_matches(signature_type: Any, argument_type: Any):
286      sig_origin_type = getattr(signature_type, "__origin__", signature_type)
287  
288      if signature_type is argument_type:
289          return True
290  
291      # Union types in signature. Given type needs to match one of the
292      # contained types in the Union
293      if sig_origin_type is typing.Union and signature_type != argument_type:
294          sig_contained = signature_type.__args__
295          return any(type_matches(c, argument_type) for c in sig_contained)
296  
297      if getattr(signature_type, "__origin__", None) is list:
298          sig_el_type = signature_type.__args__[0]
299  
300          # int can be promoted to list[int]
301          if argument_type is int and sig_el_type is int:
302              return True
303  
304          if not inspect.isclass(sig_el_type):
305              warnings.warn(
306                  f"Does not support nested parametric types, got {signature_type}. Please file a bug."
307              )
308              return False
309          if getattr(argument_type, "__origin__", None) is list:
310              return issubclass(argument_type.__args__[0], sig_el_type)
311  
312          def is_homogeneous_tuple(t):
313              if getattr(t, "__origin__", None) is not tuple:
314                  return False
315              contained = t.__args__
316              if t.__args__ == ((),):  # Tuple[()].__args__ == ((),) for some reason
317                  return True
318              return all((c is Ellipsis) or issubclass(c, sig_el_type) for c in contained)
319  
320          # Tuple[T] is accepted for List[T] parameters
321          return is_homogeneous_tuple(argument_type)
322  
323      # Dtype is an int in schemas
324      if signature_type is int and argument_type is torch.dtype:
325          return True
326  
327      if signature_type is numbers.Number and argument_type in {int, float}:
328          return True
329      if inspect.isclass(argument_type) and inspect.isclass(signature_type):
330          return issubclass(argument_type, signature_type)
331  
332      return False

335  @compatibility(is_backward_compatible=False)
336  def normalize_function(
337      target: Callable,
338      args: tuple[Any, ...],
339      kwargs: Optional[dict[str, Any]] = None,
340      arg_types: Optional[tuple[Any]] = None,
341      kwarg_types: Optional[dict[str, Any]] = None,
342      normalize_to_only_use_kwargs: bool = False,
343  ) -> Optional[ArgsKwargsPair]:
344      """ 
345      Returns normalized arguments to PyTorch functions. This means that 
346      `args/kwargs` will be matched up to the functional's 
347      signature and return exclusively kwargs in positional order if 
348      `normalize_to_only_use_kwargs` is True. 
349      Also populates default values. Does not support positional-only 
350      parameters or varargs parameters (*args, **kwargs). Does not support modules. 
351   
352      May require `arg_types` and `kwarg_types` in order to disambiguate overloads. 
353   
354      Args: 
355          target (Callable): Function that we are normalizing 
356          args (Tuple[Any]): Tuple of args to the function 
357          kwargs (Optional[Dict[str, Any]]): Dict of kwargs to the function 
358          arg_types (Optional[Tuple[Any]]): Tuple of arg types for the args 
359          kwarg_types (Optional[Dict[str, Any]]): Dict of arg types for the kwargs 
360          normalize_to_only_use_kwargs (bool): Whether to normalize to only use kwargs. 
361   
362      Returns: 
363   
364          Returns normalized_args_and_kwargs, or `None` if not successful. 
365      """
366      if kwargs is None:
367          kwargs = {}
368      new_args_and_kwargs = None
369      if (
370          not isinstance(target, types.BuiltinFunctionType)
371          and not (isinstance(target, (OpOverloadPacket, OpOverload)))
372          and hasattr(target, "_op")
373      ):
374          # ExecuTorch's EdgeOpOverload are a wrapper around PyTorch's OpOverload,
375          # so we can unwrap it here to get its schema
376          # Can't import EdgeOpOverload directly because of a circular dependency,
377          # so checking for "_op" existing is the next best thing.
378          target = target._op
379  
380      # Repeat the condition after checking for the inner _op field.
381      if not isinstance(target, types.BuiltinFunctionType) and not (
382          isinstance(target, (OpOverloadPacket, OpOverload))
383      ):
384          target_for_analysis = target
385          if target in boolean_dispatched:
386              # HACK: `boolean_dispatch` as used in `torch.nn.functional` makes it so that we have
387              # a 2-way dispatch based on a boolean value. Here we check that the `true` and `false`
388              # branches of the dispatch have exactly the same signature. If they do, use the `true`
389              # branch signature for analysis. Otherwise, leave this un-normalized
390              assert not isinstance(target, str)
391              dispatched = boolean_dispatched[target]
392              if_true, if_false = dispatched["if_true"], dispatched["if_false"]
393              if (
394                  inspect.signature(if_true).parameters
395                  != inspect.signature(if_false).parameters
396              ):
397                  return None
398              target_for_analysis = if_true
399  
400          assert callable(target_for_analysis)
401          sig = inspect.signature(inspect.unwrap(target_for_analysis))
402          new_args_and_kwargs = _args_kwargs_to_normalized_args_kwargs(
403              sig, args, kwargs, normalize_to_only_use_kwargs
404          )
405      else:
406          assert callable(target)
407          torch_op_schemas = get_signature_for_torch_op(target)
408          matched_schemas = []
409          if torch_op_schemas:
410              # Iterate through all of the schema until we find one that matches
411              # If one matches, populate `new_args_and_kwargs` with the new args/kwargs
412              # values. If none matches, `new_args_and_kwargs` will be None
413              for candidate_signature in torch_op_schemas:
414                  try:
415                      candidate_signature.bind(*args, **kwargs)
416                      matched_schemas.append(candidate_signature)
417                  except TypeError:
418                      continue
419  
420              if len(matched_schemas) == 0:
421                  # Did not match any schema. Cannot normalize
422                  pass
423              elif len(matched_schemas) == 1:
424                  # Matched exactly one schema, unambiguous
425                  new_args_and_kwargs = _args_kwargs_to_normalized_args_kwargs(
426                      matched_schemas[0], args, kwargs, normalize_to_only_use_kwargs
427                  )
428              else:
429                  if arg_types is not None or kwarg_types is not None:
430                      arg_types = arg_types if arg_types else cast(tuple[Any], ())
431                      kwarg_types = kwarg_types if kwarg_types else {}
432                      for candidate_signature in torch_op_schemas:
433                          sig_matches = True
434                          try:
435                              bound_types = candidate_signature.bind(
436                                  *arg_types, **kwarg_types
437                              )
438                              for arg_name, arg_type in bound_types.arguments.items():
439                                  param = candidate_signature.parameters[arg_name]
440                                  sig_matches = sig_matches and type_matches(
441                                      param.annotation, arg_type
442                                  )
443                          except TypeError:
444                              sig_matches = False
445                          if sig_matches:
446                              new_args_and_kwargs = (
447                                  _args_kwargs_to_normalized_args_kwargs(
448                                      candidate_signature,
449                                      args,
450                                      kwargs,
451                                      normalize_to_only_use_kwargs,
452                                  )
453                              )
454                              break
455                  else:
456                      # Matched more than one schema. In this situation, the caller must provide the types of
457                      # the arguments of the overload they expect.
458                      schema_printouts = "\n".join(
459                          str(schema) for schema in matched_schemas
460                      )
461                      raise RuntimeError(
462                          f"Tried to normalize arguments to {torch.typename(target)} but "
463                          f"the schema match was ambiguous! Please provide argument types to "
464                          f"the normalize_arguments() call. Available schemas:\n{schema_printouts}"
465                      )
466  
467      return new_args_and_kwargs

470  @compatibility(is_backward_compatible=False)
471  def normalize_module(
472      root: torch.nn.Module,
473      target: str,
474      args: tuple[Any],
475      kwargs: Optional[dict[str, Any]] = None,
476      normalize_to_only_use_kwargs: bool = False,
477  ) -> Optional[ArgsKwargsPair]:
478      """ 
479      Returns normalized arguments to PyTorch modules. This means that 
480      `args/kwargs` will be matched up to the functional's 
481      signature and return exclusively kwargs in positional order if 
482      `normalize_to_only_use_kwargs` is True. 
483      Also populates default values. Does not support positional-only 
484      parameters or varargs parameters (*args, **kwargs). 
485   
486      Args: 
487          root (nn.Module): root module upon which we query modules 
488          target (Callable): Function that we are normalizing 
489          args (Tuple[Any]): Tuple of args to the function 
490          kwargs (Optional[Dict[str, Any]]): Dict of kwargs to the function 
491          normalize_to_only_use_kwargs (bool): Whether to normalize to only use kwargs. 
492   
493      Returns: 
494   
495          Returns normalized_args_and_kwargs, or `None` if not successful. 
496      """
497      try:
498          submod = root.get_submodule(target)
499      except AttributeError as e:
500          raise RuntimeError(
501              f"Tried to normalize node with target {target} but root did not "
502              f"have that target!"
503          ) from e
504      if hasattr(submod.__class__, "__name__"):
505          classname = submod.__class__.__name__
506          if getattr(torch.nn, classname, None) == submod.__class__:
507              sig = inspect.signature(inspect.unwrap(submod.forward))
508              if kwargs is None:
509                  kwargs = {}
510              new_args_and_kwargs = _args_kwargs_to_normalized_args_kwargs(
511                  sig, args, kwargs, normalize_to_only_use_kwargs
512              )
513              return new_args_and_kwargs
514      return None

517  def _args_kwargs_to_normalized_args_kwargs(
518      sig: inspect.Signature,
519      args: tuple[Any, ...],
520      kwargs: dict[str, Any],
521      normalize_to_only_use_kwargs: bool,
522  ) -> Optional[ArgsKwargsPair]:
523      """ 
524      Given a call target, args, and kwargs, return the arguments normalized into 
525      an ArgsKwargsPair, or None if the type signature is not supported by 
526      this normalization. 
527   
528      Args: 
529   
530          sig (inspect.Signature): Signature object for the target 
531          args (Tuple): Arguments that appear at the callsite for `target` 
532          kwargs (Dict): Keyword arguments that appear at the callsite for `target` 
533          normalize_to_only_use_kwargs (bool): Whether to normalize to only use kwargs. 
534   
535      Returns: 
536   
537          Optional[ArgsKwargsPair]: Normalized args and kwargs for `target`, or `None` if 
538              this target is not supported. 
539      """
540  
541      # Don't currently support positional-only
542      # or varargs (*args, **kwargs) signatures
543      supported_parameter_types = {
544          inspect.Parameter.POSITIONAL_OR_KEYWORD,
545          inspect.Parameter.KEYWORD_ONLY,
546      }
547      if any(p.kind not in supported_parameter_types for p in sig.parameters.values()):
548          # Add an exception for one signature, which is common for random/uniform, i.e.:
549          # Tensor(a!) self, float from=0, float to=1, *, Generator? generator=None
550          # `from` is Python keyword and as such functions with that signature should have
551          # positional-only args, but at the same time they could be dispatched as kwargs
552          if list(sig.parameters.keys()) != ["input", "from", "to", "generator"]:
553              return None
554  
555      bound_args = sig.bind(*args, **kwargs)
556      bound_args.apply_defaults()
557  
558      new_kwargs: dict[str, Any] = {}
559      new_args: list[Any] = []
560      for i, param in enumerate(sig.parameters):
561          if not normalize_to_only_use_kwargs and i < len(args):
562              new_args.append(bound_args.arguments[param])
563          else:
564              new_kwargs[param] = bound_args.arguments[param]
565  
566      return ArgsKwargsPair(tuple(new_args), new_kwargs)
