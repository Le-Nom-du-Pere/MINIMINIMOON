663    static Vectorized<c10::complex<float>> loadu(
664        const void* ptr,
665        int64_t count = size()) {
666      if (count == size())
667        return _mm512_loadu_ps(reinterpret_cast<const float*>(ptr));
668  
669      __at_align__ float tmp_values[2 * size()];
670      // Ensure uninitialized memory does not change the output value See
671      // https://github.com/pytorch/pytorch/issues/32502 for more details. We do
672      // not initialize arrays to zero using "={0}" because gcc would compile it
673      // to two instructions while a loop would be compiled to one instruction.
674      for (const auto i : c10::irange(2 * size())) {
675        tmp_values[i] = 0.0;
676      }
677      std::memcpy(
678          tmp_values,
679          reinterpret_cast<const float*>(ptr),
680          count * sizeof(c10::complex<float>));
681      return _mm512_load_ps(tmp_values);
682    }
683    void store(void* ptr, int count = size()) const {
684      if (count == size()) {
685        _mm512_storeu_ps(reinterpret_cast<float*>(ptr), values);
686      } else if (count > 0) {
687        float tmp_values[2 * size()];
688        _mm512_storeu_ps(reinterpret_cast<float*>(tmp_values), values);
689        std::memcpy(ptr, tmp_values, count * sizeof(c10::complex<float>));
690      }
691    }

695      __m512i idx1 = _mm512_set_epi32(
696          30, 14, 28, 12, 26, 10, 24, 8, 22, 6, 20, 4, 18, 2, 16, 0);
