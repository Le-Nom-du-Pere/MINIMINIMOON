179  // C10_LIKELY/C10_UNLIKELY
180  //
181  // These macros provide parentheses, so you can use these macros as:
182  //
183  //    if C10_LIKELY(some_expr) {
184  //      ...
185  //    }
186  //
187  // NB: static_cast to boolean is mandatory in C++, because __builtin_expect
188  // takes a long argument, which means you may trigger the wrong conversion
189  // without it.
190  //
191  #if defined(__GNUC__) || defined(__ICL) || defined(__clang__)
192  #define C10_LIKELY(expr) (__builtin_expect(static_cast<bool>(expr), 1))
193  #define C10_UNLIKELY(expr) (__builtin_expect(static_cast<bool>(expr), 0))
194  #else
195  #define C10_LIKELY(expr) (expr)
196  #define C10_UNLIKELY(expr) (expr)
197  #endif

253  // constants from
254  // (https://docs.nvidia.com/cuda/cuda-c-programming-guide/index.html#features-and-technical-specifications)
255  // The maximum number of threads per multiprocessor is 1024 for Turing
256  // architecture (7.5), 1536 for Geforce Ampere (8.6)/Jetson Orin (8.7), and
257  // 2048 for all other architectures. You'll get warnings if you exceed these
258  // constants. Hence, the following macros adjust the input values from the user
259  // to resolve potential warnings.
260  #if __CUDA_ARCH__ == 750
261  constexpr uint32_t CUDA_MAX_THREADS_PER_SM = 1024;
262  #elif __CUDA_ARCH__ == 860 || __CUDA_ARCH__ == 870 || __CUDA_ARCH__ == 890
263  constexpr uint32_t CUDA_MAX_THREADS_PER_SM = 1536;
264  #else
265  constexpr uint32_t CUDA_MAX_THREADS_PER_SM = 2048;
266  #endif
267  // CUDA_MAX_THREADS_PER_BLOCK is same for all architectures currently
268  constexpr uint32_t CUDA_MAX_THREADS_PER_BLOCK = 1024;
269  // CUDA_THREADS_PER_BLOCK_FALLBACK is the "canonical fallback" choice of block
270  // size. 256 is a good number for this fallback and should give good occupancy
271  // and versatility across all architectures.
272  constexpr uint32_t CUDA_THREADS_PER_BLOCK_FALLBACK = 256;
273  // NOTE: if you are thinking of constexpr-ify the inputs to launch bounds, it
274  //       turns out that although __launch_bounds__ can take constexpr, it
275  //       can't take a constexpr that has anything to do with templates.
276  //       Currently we use launch_bounds that depend on template arguments in
277  //       Loops.cuh, Reduce.cuh and LossCTC.cuh. Hence, C10_MAX_THREADS_PER_BLOCK
278  //       and C10_MIN_BLOCKS_PER_SM are kept as macros.
279  // Suppose you were planning to write __launch_bounds__(a, b), based on your
280  // performance tuning on a modern GPU. Instead, you should write
281  // __launch_bounds__(C10_MAX_THREADS_PER_BLOCK(a), C10_MIN_BLOCKS_PER_SM(a, b)),
282  // which will also properly respect limits on old architectures.
283  #define C10_MAX_THREADS_PER_BLOCK(val)           \
284    (((val) <= CUDA_MAX_THREADS_PER_BLOCK) ? (val) \
285                                           : CUDA_THREADS_PER_BLOCK_FALLBACK)
286  #define C10_MIN_BLOCKS_PER_SM(threads_per_block, blocks_per_sm)        \
287    ((((threads_per_block) * (blocks_per_sm) <= CUDA_MAX_THREADS_PER_SM) \
288          ? (blocks_per_sm)                                              \
289          : ((CUDA_MAX_THREADS_PER_SM + (threads_per_block) - 1) /       \
290             (threads_per_block))))
291  // C10_LAUNCH_BOUNDS is analogous to __launch_bounds__
292  #define C10_LAUNCH_BOUNDS_0 \
293    __launch_bounds__(        \
294        256, 4) // default launch bounds that should give good occupancy and
295                // versatility across all architectures.
296  #define C10_LAUNCH_BOUNDS_1(max_threads_per_block) \
297    __launch_bounds__((C10_MAX_THREADS_PER_BLOCK((max_threads_per_block))))
298  #define C10_LAUNCH_BOUNDS_2(max_threads_per_block, min_blocks_per_sm) \
299    __launch_bounds__(                                                  \
300        (C10_MAX_THREADS_PER_BLOCK((max_threads_per_block))),           \
301        (C10_MIN_BLOCKS_PER_SM((max_threads_per_block), (min_blocks_per_sm))))

406  // ROCm disables kernel assert by default for performance considerations.
407  // Though ROCm supports __assert_fail, it uses kernel printf which has
408  // a non-negligible performance impact even if the assert condition is
409  // never triggered. We choose to use abort() instead which will still
410  // terminate the application but without a more useful error message.
411  #if !defined(C10_USE_ROCM_KERNEL_ASSERT) and defined(USE_ROCM)
412  #define CUDA_KERNEL_ASSERT(cond) \
413    if C10_UNLIKELY (!(cond)) {    \
414      abort();                     \
415    }
416  #define CUDA_KERNEL_ASSERT_MSG(cond, msg) \
417    if C10_UNLIKELY (!(cond)) {             \
418      abort();                              \
419    }
420  #define SYCL_KERNEL_ASSERT(cond) \
421    if C10_UNLIKELY (!(cond)) {    \
422      abort();                     \
423    }
424  #else
425  #define CUDA_KERNEL_ASSERT(cond)                                         \
426    if (C10_UNLIKELY(!(cond))) {                                           \
427      __assert_fail(                                                       \
428          #cond, __FILE__, static_cast<unsigned int>(__LINE__), __func__); \
429    }

431    if (C10_UNLIKELY(!(cond))) {                                         \
