107  /// Runs the RHS packing micro-kernel.
108  ///
109  /// The int4 values are stored in a N x K matrix, where N is number of rows and K is the number of columns.
110  /// Two int4 values are stored in one byte. The lower order part of the byte (low) holds
111  /// the first nibble (K-index + 0). The higher order of the byte holds the second nibble (K-index + 1).
112  ///
113  /// @param[in]  num_groups   The number of groups. It must be 1.
114  /// @param[in]  n            The number of rows in the RHS matrix (not packed).
115  /// @param[in]  k            The number of columns in the RHS matrix (not packed).
116  /// @param[in]  nr           The number of columns written by the matmul micro-kernel. It must be a multiple of 4.
117  /// @param[in]  kr           The number of columns loaded in the single inner most loop of the matmul micro-kernel.
118  /// @param[in]  sr           The number of kr splits. It can be 1 (no splits) up to kr.
119  ///                          However, kr must be multiple of sr.
120  /// @param[in]  bl           The block length, which defines the number of
121  ///                          K values stored in a single block. It must be a multiple of 32.
122  /// @param[in]  rhs          The RHS matrix containing the 4-bit values.
123  ///                          Size in bytes is expected to be greater than or equal to n * k * (sizeof(uint8_t) / 2).
124  /// @param[in]  rhs_stride   The number of bytes per row in bytes of the RHS matrix
125  /// @param[in]  bias         The biases.
126  /// @param[in]  scale        The per-block quantization scales.
127  ///                          The scale data type must be provided with the params object.
128  ///                          Supported scale data types are FP32, FP16 and BF16.
129  /// @param[in]  scale_stride The number of bytes per row in bytes of the scale matrix
130  /// @param[out] rhs_packed   The packed RHS matrix.
131  /// @param[in]  extra_bytes  Extra bytes to append to the end of each row of the packed RHS matrix.
132  /// @param[in]  params       Parameters for the micro-kernel.
133  void kai_run_rhs_pack_nxk_qsi4c32p_qsu4c32s1s0(
134      size_t num_groups,                                                   //
135      size_t n,                                                            //
136      size_t k,                                                            //
137      size_t nr,                                                           //
138      size_t kr,                                                           //
139      size_t sr,                                                           //
140      size_t bl,                                                           //
141      const uint8_t* rhs,                                                  //
142      size_t rhs_stride,                                                   //
143      const float* bias,                                                   //
144      const void* scale,                                                   //
145      size_t scale_stride,                                                 //
146      void* rhs_packed,                                                    //
147      size_t extra_bytes,                                                  //
148      const struct kai_rhs_pack_nxk_qsi4c32p_qsu4c32s1s0_params* params);  //
