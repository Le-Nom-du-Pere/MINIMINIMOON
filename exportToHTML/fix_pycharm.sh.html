fix_pycharm.sh 
1    #!/bin/bash
2    set -e
3    
4    VENV_PY="$HOME/.venvs/policy311/bin/python"
5    echo "Configurando DataSpell en: $(pwd)"
6    echo "Para usar: $VENV_PY"
7    
8    # Backup
9    zip -qr ".idea_backup_$(date +%s).zip" ".idea" 2>/dev/null || true
10   
11   # Limpiar configuraciones tóxicas
12   if [ -f ".idea/workspace.xml" ]; then
13     sed -i '' 's/ADD_CONTENT_ROOTS" value="true"/ADD_CONTENT_ROOTS" value="false"/g' .idea/workspace.xml
14     sed -i '' 's/ADD_SOURCE_ROOTS" value="true"/ADD_SOURCE_ROOTS" value="false"/g' .idea/workspace.xml
15     echo "✓ workspace.xml limpio"
16   fi
17   
18   # Limpiar variables de entorno tóxicas en todas las configuraciones
19   find .idea -name "*.xml" -exec sed -i '' '/PYTHONHOME/d; /PYTHONPATH/d' {} \; 2>/dev/null || true
20   echo "✓ Variables PYTHONHOME/PYTHONPATH eliminadas"
21   
22   echo "Listo. Abre DataSpell y configura el intérprete a: $VENV_PY"
23
