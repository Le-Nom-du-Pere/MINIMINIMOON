6    def configure_logging(log_level: Optional[str] = None) -> None:
7        """ 
8        Configure logging with LOG_LEVEL environment variable support. 
9         
10       Args: 
11           log_level: Override log level, if None uses LOG_LEVEL env var 
12       """
13       # Get log level from parameter or environment variable
14       level_string = log_level or os.getenv('LOG_LEVEL', 'INFO')
15       
16       # Validate and convert log level string to logging constant
17       level = get_log_level(level_string)
18       
19       # Configure logging
20       logging.basicConfig(
21           level=level,
22           format='%(asctime)s - %(name)s - %(levelname)s - %(message)s',
23           datefmt='%Y-%m-%d %H:%M:%S'
24       )
25       
26       # Log configuration info at DEBUG level
27       logger = logging.getLogger(__name__)
28       logger.debug(f"Logging configured with level: {logging.getLevelName(level)} ({level})")
29       
30       # Optionally warn about invalid values
31       if level_string.upper() != logging.getLevelName(level):
32           logger.warning(f"Invalid LOG_LEVEL value '{level_string}' - falling back to INFO")

35   def get_log_level(level_string: str) -> int:
36       """ 
37       Convert log level string to logging constant with validation. 
38        
39       Args: 
40           level_string: Log level as string (e.g., 'DEBUG', 'INFO', 'WARNING') 
41            
42       Returns: 
43           Logging level constant (int) 
44       """
45       # Normalize to uppercase
46       level_string = level_string.upper()
47       
48       # Valid log levels
49       valid_levels = {
50           'DEBUG': logging.DEBUG,
51           'INFO': logging.INFO,
52           'WARNING': logging.WARNING,
53           'WARN': logging.WARNING,  # Common alias
54           'ERROR': logging.ERROR,
55           'CRITICAL': logging.CRITICAL,
56           'FATAL': logging.CRITICAL  # Common alias
57       }
58       
59       return valid_levels.get(level_string, logging.INFO)
