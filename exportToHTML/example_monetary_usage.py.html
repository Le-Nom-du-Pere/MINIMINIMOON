example_monetary_usage.py 
1    #!/usr/bin/env python3
2    """ 
3    Example usage of the MonetaryDetector for Spanish text processing. 
4     
5    This module demonstrates how to use the MonetaryDetector to find and normalize 
6    monetary amounts, percentages, and numeric values with scales in Spanish text. 
7    """
8    
9    from monetary_detector import create_monetary_detector, MonetaryType
10   
11   def main():
12       """Demonstrate monetary detection capabilities."""
13       # Create detector instance
14       detector = create_monetary_detector()
15       
16       print("=== Spanish Monetary Expression Detection Demo ===\n")
17       
18       # Example Spanish text with various monetary expressions
19       sample_texts = [
20           "El presupuesto total es de $2.5 millones COP para el próximo año.",
21           "La empresa reportó ingresos de €1.234.567,89 con un crecimiento del 15%.",
22           "El proyecto requiere una inversión de USD 500K aproximadamente.",
23           "Los costos operativos aumentaron a 1,8M respecto a los 1.200 miles del año anterior.",
24           "La meta de ventas es alcanzar ¥2.500.000 con un margen del 12,5%.",
25           "El fondo tiene assets por £750K y espera un retorno del 8,3%.",
26       ]
27       
28       # Process each text sample
29       for i, text in enumerate(sample_texts, 1):
30           print(f"Sample {i}: {text}")
31           results = detector.detect_monetary_expressions(text)
32           
33           if results:
34               print("  Detected expressions:")
35               for j, match in enumerate(results, 1):
36                   type_name = match.type.value.upper()
37                   currency_info = f" ({match.currency})" if match.currency else ""
38                   print(f"    [{j}] '{match.text}' -> {match.value:,.2f}{currency_info} [{type_name}]")
39           else:
40               print("  No monetary expressions detected")
41           print()
42       
43       print("=== Convention Examples ===\n")
44       
45       # Demonstrate handling of ambiguous abbreviations
46       abbreviation_examples = [
47           ("$1M", "M = million"),
48           ("€2.5MM", "MM = million (same as M)"),
49           ("£500K", "K = thousand"),
50           ("¥3B", "B = billion"),
51       ]
52       
53       print("Abbreviation conventions:")
54       for expr, explanation in abbreviation_examples:
55           result = detector.normalize_monetary_expression(expr)
56           print(f"  {expr:8} -> {result:>12,.0f} ({explanation})")
57       print()
58       
59       # Demonstrate decimal separator handling
60       decimal_examples = [
61           ("$1.234.567,89", "Spanish format: period=thousands, comma=decimal"),
62           ("$1,234,567.89", "English format: comma=thousands, period=decimal"),
63           ("€12.345,60", "Mixed format: automatic detection"),
64       ]
65       
66       print("Decimal separator conventions:")
67       for expr, explanation in decimal_examples:
68           result = detector.normalize_monetary_expression(expr)
69           print(f"  {expr:15} -> {result:>12,.2f} ({explanation})")
70       print()
71       
72       # Demonstrate percentage conversion
73       percentage_examples = ["50%", "12,5%", "0,75%", "150%"]
74       
75       print("Percentage conversion (to decimal):")
76       for expr in percentage_examples:
77           result = detector.normalize_monetary_expression(expr)
78           print(f"  {expr:6} -> {result:.4f}")
79       print()
80       
81       print("=== Advanced Features ===\n")
82       
83       # Complex text analysis
84       complex_text = """ 
85       El informe financiero muestra que los ingresos alcanzaron $15.5 millones USD, 
86       un incremento del 23% comparado con los €12M del período anterior. Los gastos 
87       operativos fueron de 8,2M COP, mientras que la inversión en I+D representó 
88       el 4,5% del presupuesto total. Se proyecta un crecimiento del 18,7% para el 
89       siguiente trimestre. 
90       """
91       
92       print("Complex text analysis:")
93       print(complex_text.strip())
94       print("\nExtracted values:")
95       
96       results = detector.detect_monetary_expressions(complex_text)
97       for i, match in enumerate(results, 1):
98           type_name = match.type.value.upper()
99           currency_info = f" {match.currency}" if match.currency else ""
100          position_info = f" at position {match.start_pos}-{match.end_pos}"
101          print(f"  [{i}] {match.value:>12,.2f}{currency_info:>5} [{type_name}] - '{match.text}'{position_info}")
102      
103      print(f"\nTotal expressions found: {len(results)}")
104      
105      # Summary statistics
106      currencies = [r for r in results if r.type == MonetaryType.CURRENCY and r.currency]
107      percentages = [r for r in results if r.type == MonetaryType.PERCENTAGE]
108      numerics = [r for r in results if r.type == MonetaryType.NUMERIC]
109      
110      print(f"  - Currency amounts: {len(currencies)}")
111      print(f"  - Percentages: {len(percentages)}")
112      print(f"  - Numeric with scales: {len(numerics)}")
113      
114      if currencies:
115          unique_currencies = set(r.currency for r in currencies)
116          print(f"  - Unique currencies: {', '.join(sorted(unique_currencies))}")
117  
118  
119  if __name__ == "__main__":
120      main()
